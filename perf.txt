package org.pronzato.fabric.api.control.discovery.scope;

import java.util.LinkedHashMap;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Locale;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import org.pronzato.fabric.api.control.discovery.CanonicalIds;
import org.pronzato.fabric.api.control.discovery.definitions.Definition;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEntitlement;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEnvelope;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionPlatform;
import org.pronzato.fabric.api.control.entitlements.simple.PlatformModelMembership;
import org.pronzato.fabric.api.util.Strings;

public final class DiscoveryScopeResolver {
  private static final String PLATFORM_GROUP = "platform";

  public DiscoveryScope resolve(
      String primaryGroup,
      String subject,
      String delegatedSubject,
      List<Definition> definitions) {
    List<Definition> records = definitions == null ? List.of() : definitions;
    DiscoveryScope base = DiscoveryScope.primaryOnly(primaryGroup, subject, delegatedSubject);

    String effectiveSubject = firstNonBlank(normalize(delegatedSubject), normalize(subject));
    if (effectiveSubject == null) {
      return base;
    }

    Set<String> subjectGroups = resolveSubjectGroups(effectiveSubject, records);
    LinkedHashSet<String> visibleGroups = new LinkedHashSet<>(base.visibleGroups());
    LinkedHashSet<String> visibleAssetIds = new LinkedHashSet<>(base.visibleAssetIds());
    LinkedHashMap<String, LinkedHashSet<String>> reasons = mutableReasons(base.visibilityReasons());

    if (isPlatformAdmin(subjectGroups)) {
      for (String group : groupsFromDefinitions(records)) {
        addGroupVisibility(visibleGroups, reasons, group, "platform-admin");
      }
    }

    for (Definition definition : records) {
      if (!(definition instanceof DefinitionEntitlement entitlement)) {
        continue;
      }
      if (!enabled(entitlement.envelope() == null ? null : entitlement.envelope().enabled())) {
        continue;
      }
      if (!isEntitlementAllowed(entitlement, effectiveSubject, subjectGroups)) {
        continue;
      }
      String source = entitlementSource(entitlement.envelope());
      for (String target : entitlementTargets(entitlement)) {
        String normalizedTarget = normalize(target);
        if (normalizedTarget == null) {
          continue;
        }
        visibleAssetIds.add(normalizedTarget);
        String group = groupFromResourceId(normalizedTarget);
        if (group != null) {
          addGroupVisibility(visibleGroups, reasons, group, source);
        }
      }
    }

    return base.withGroupVisibility(visibleGroups, visibleAssetIds, immutableReasons(reasons));
  }

  private static Set<String> resolveSubjectGroups(String effectiveSubject, List<Definition> definitions) {
    DefinitionPlatform platform = findPlatformModel(definitions);
    if (platform == null || platform.root() == null) {
      return Set.of();
    }
    Map<String, Set<String>> membership = PlatformModelMembership.buildMembershipIndex(platform);
    if (membership.isEmpty()) {
      return Set.of();
    }

    LinkedHashSet<String> groups = new LinkedHashSet<>();
    Set<String> candidates = PlatformModelMembership.membershipSubjectCandidates(effectiveSubject);
    for (String candidate : candidates) {
      Set<String> candidateGroups = membership.get(candidate);
      if (candidateGroups != null) {
        for (String group : candidateGroups) {
          String normalized = normalizeGroup(group);
          if (normalized != null) {
            groups.add(normalized);
          }
        }
      }
    }

    String directGroup = normalizeGroup(effectiveSubject);
    if (directGroup != null) {
      groups.add(directGroup);
    }
    return groups.isEmpty() ? Set.of() : Set.copyOf(groups);
  }

  private static DefinitionPlatform findPlatformModel(List<Definition> definitions) {
    for (Definition definition : definitions) {
      if (!(definition instanceof DefinitionPlatform platform)) {
        continue;
      }
      DefinitionEnvelope envelope = platform.envelope();
      if (envelope == null) {
        continue;
      }
      if (PLATFORM_GROUP.equalsIgnoreCase(normalize(envelope.group()))
          && "platform".equalsIgnoreCase(normalize(envelope.name()))) {
        return platform;
      }
    }
    return null;
  }

  private static Set<String> groupsFromDefinitions(List<Definition> definitions) {
    LinkedHashSet<String> groups = new LinkedHashSet<>();
    for (Definition definition : definitions) {
      DefinitionEnvelope envelope = definition == null ? null : definition.envelope();
      String group = envelope == null ? null : normalize(envelope.group());
      if (group != null) {
        groups.add(group);
      }
    }
    return groups.isEmpty() ? Set.of() : Set.copyOf(groups);
  }

  private static boolean isEntitlementAllowed(
      DefinitionEntitlement entitlement,
      String effectiveSubject,
      Set<String> subjectGroups) {
    boolean allow = false;
    boolean deny = false;

    if (entitlement.policies() != null) {
      for (DefinitionEntitlement.PolicyDocument policy : entitlement.policies()) {
        if (policy == null || !enabled(policy.enabled()) || policy.bindings() == null) {
          continue;
        }
        for (DefinitionEntitlement.PolicyBinding binding : policy.bindings()) {
          if (binding == null || !enabled(binding.enabled())) {
            continue;
          }
          if (!subjectMatches(binding.subject(), effectiveSubject, subjectGroups)) {
            continue;
          }
          String effect = normalize(binding.effect());
          if (effect == null) {
            continue;
          }
          if ("DENY".equalsIgnoreCase(effect)) {
            deny = true;
          } else if ("ALLOW".equalsIgnoreCase(effect)) {
            allow = true;
          }
        }
      }
    }

    if (entitlement.relationships() != null) {
      for (DefinitionEntitlement.RelationshipTuple relationship : entitlement.relationships()) {
        if (relationship == null) {
          continue;
        }
        if (!subjectMatches(relationship.subject(), effectiveSubject, subjectGroups)) {
          continue;
        }
        String relation = normalize(relationship.relation());
        if (relation == null) {
          continue;
        }
        String token = relation.toLowerCase(Locale.ROOT);
        if (token.contains("deny")) {
          deny = true;
          continue;
        }
        if (token.contains("allow")
            || token.contains("read")
            || token.contains("invoke")
            || token.contains("write")
            || token.contains("start")) {
          allow = true;
        }
      }
    }

    if (deny) {
      return false;
    }
    return allow;
  }

  private static List<String> entitlementTargets(DefinitionEntitlement entitlement) {
    LinkedHashSet<String> targets = new LinkedHashSet<>();
    addTarget(targets, entitlement.scopeId());

    if (entitlement.policies() != null) {
      for (DefinitionEntitlement.PolicyDocument policy : entitlement.policies()) {
        if (policy == null || !enabled(policy.enabled())) {
          continue;
        }
        addTarget(targets, policy.resourceId());
        if (policy.bindings() == null) {
          continue;
        }
        for (DefinitionEntitlement.PolicyBinding binding : policy.bindings()) {
          if (binding == null || !enabled(binding.enabled()) || binding.obligations() == null) {
            continue;
          }
          List<String> datasetScope = binding.obligations().datasetScope();
          if (datasetScope == null) {
            continue;
          }
          for (String resourceId : datasetScope) {
            addTarget(targets, resourceId);
          }
        }
      }
    }

    if (entitlement.relationships() != null) {
      for (DefinitionEntitlement.RelationshipTuple tuple : entitlement.relationships()) {
        if (tuple == null) {
          continue;
        }
        addTarget(targets, tuple.resourceId());
        addTarget(targets, tuple.resourceSubject());
      }
    }

    return targets.isEmpty() ? List.of() : List.copyOf(targets);
  }

  private static void addTarget(LinkedHashSet<String> targets, String value) {
    String normalized = normalize(value);
    if (normalized != null) {
      targets.add(normalized);
    }
  }

  private static String groupFromResourceId(String resourceId) {
    String normalized = normalize(resourceId);
    if (normalized == null) {
      return null;
    }
    try {
      CanonicalIds.ParsedId parsed = CanonicalIds.parse(normalized);
      String group = normalize(parsed.group());
      if (group != null) {
        return group;
      }
    } catch (RuntimeException ignored) {
      // fall through to lenient parse for non-canonical config scope ids.
    }

    int separator = normalized.indexOf(':');
    if (separator <= 0 || separator >= normalized.length() - 1) {
      return null;
    }
    String payload = normalized.substring(separator + 1);
    String[] parts = payload.split("/");
    if (parts.length == 0) {
      return null;
    }
    String type = normalized.substring(0, separator).toLowerCase(Locale.ROOT);
    if ("group".equals(type)) {
      return normalizeGroup(normalized);
    }
    if ("external".equals(type)
        || "storage".equals(type)
        || "user".equals(type)
        || "role".equals(type)) {
      return null;
    }
    return normalize(parts[0]);
  }

  private static void addGroupVisibility(
      Set<String> visibleGroups,
      Map<String, LinkedHashSet<String>> reasons,
      String group,
      String sourceReason) {
    String normalizedGroup = normalize(group);
    if (normalizedGroup == null) {
      return;
    }
    visibleGroups.add(normalizedGroup);
    String reason = normalize(sourceReason);
    if (reason == null) {
      return;
    }
    reasons.computeIfAbsent(normalizedGroup, ignored -> new LinkedHashSet<>()).add(reason);
  }

  private static boolean subjectMatches(String configuredSubject, String subject, Set<String> groups) {
    String normalizedConfigured = normalize(configuredSubject);
    if (normalizedConfigured == null) {
      return false;
    }
    if (normalizedConfigured.startsWith("group:")) {
      String configuredGroup = normalizeGroup(normalizedConfigured);
      if (configuredGroup == null || groups == null || groups.isEmpty()) {
        return false;
      }
      for (String group : groups) {
        if (Objects.equals(configuredGroup, normalizeGroup(group))) {
          return true;
        }
      }
      return false;
    }
    return Objects.equals(normalizedConfigured, normalize(subject));
  }

  private static String entitlementSource(DefinitionEnvelope envelope) {
    String group = envelope == null ? null : normalize(envelope.group());
    String name = envelope == null ? null : normalize(envelope.name());
    if (group == null || name == null) {
      return "entitlement";
    }
    return "entitlement:" + group + "/" + name;
  }

  private static boolean enabled(Boolean value) {
    return value == null || value.booleanValue();
  }

  private static boolean isPlatformAdmin(Set<String> groups) {
    if (groups == null || groups.isEmpty()) {
      return false;
    }
    for (String group : groups) {
      String normalized = normalizeGroup(group);
      if (normalized != null && PLATFORM_GROUP.equalsIgnoreCase(normalized)) {
        return true;
      }
    }
    return false;
  }

  private static LinkedHashMap<String, LinkedHashSet<String>> mutableReasons(
      Map<String, List<String>> source) {
    LinkedHashMap<String, LinkedHashSet<String>> reasons = new LinkedHashMap<>();
    if (source == null || source.isEmpty()) {
      return reasons;
    }
    for (Map.Entry<String, List<String>> entry : source.entrySet()) {
      String group = normalize(entry.getKey());
      if (group == null) {
        continue;
      }
      LinkedHashSet<String> values = new LinkedHashSet<>();
      List<String> reasonValues = entry.getValue();
      if (reasonValues != null) {
        for (String reason : reasonValues) {
          String normalizedReason = normalize(reason);
          if (normalizedReason != null) {
            values.add(normalizedReason);
          }
        }
      }
      reasons.put(group, values);
    }
    return reasons;
  }

  private static Map<String, List<String>> immutableReasons(Map<String, LinkedHashSet<String>> source) {
    if (source == null || source.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, List<String>> reasons = new LinkedHashMap<>();
    for (Map.Entry<String, LinkedHashSet<String>> entry : source.entrySet()) {
      if (entry.getValue() == null || entry.getValue().isEmpty()) {
        reasons.put(entry.getKey(), List.of());
      } else {
        reasons.put(entry.getKey(), List.copyOf(entry.getValue()));
      }
    }
    return Map.copyOf(reasons);
  }

  private static String firstNonBlank(String first, String second) {
    if (first != null) {
      return first;
    }
    return second;
  }

  private static String normalizeGroup(String value) {
    String normalized = normalize(value);
    if (normalized == null) {
      return null;
    }
    if (normalized.startsWith("group:") && normalized.length() > "group:".length()) {
      return normalize(normalized.substring("group:".length()));
    }
    return normalized;
  }

  private static String normalize(String value) {
    return Strings.trimToNull(value);
  }
}
