package org.pronzato.fabric.api.control.security.identity;

import java.util.Locale;
import java.util.Map;
import java.util.Objects;
import java.util.Optional;
import java.util.function.Supplier;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryCache;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionApp;
import org.pronzato.fabric.api.control.security.UserContext;
import org.pronzato.fabric.api.control.security.UserPrincipal;
import org.pronzato.fabric.api.control.security.headers.FabricRequestHeaders;
import org.pronzato.fabric.api.util.Strings;

public final class RequestIdentityResolver {
  private static final String USER_PREFIX = "user:";
  private static final String APP_PREFIX = "app:";

  private RequestIdentityResolver() {}

  public static RequestIdentity resolve(
      FabricRequestHeaders headers, DiscoveryCache DiscoveryCache, String runtimeEnvironment) {
    Objects.requireNonNull(headers, "headers");

    String rawObo = Strings.trimToNull(headers.oboUser());
    CallerAppIdentity requestCallerApp = normalizeCallerApp(headers.callerApp());
    String authenticatedSubject =
        normalizeAuthenticatedPrincipal(headers.authenticatedPrincipal(), requestCallerApp, rawObo);

    if (authenticatedSubject == null) {
      throw failure(
          Code.UNAUTHENTICATED,
          OboDenyRule.OBO_PRESENT_BUT_NOT_AUTHENTICATED,
          requestCallerApp,
          rawObo);
    }

    String normalizedObo = null;
    String effectiveUserSubject = authenticatedSubject;
    CallerAppIdentity resolvedCallerApp = mergeCallerAppForContext(requestCallerApp, authenticatedSubject);
    if (rawObo != null) {
      if (!requestCallerApp.isPresent()) {
        throw failure(
            Code.PERMISSION_DENIED,
            OboDenyRule.OBO_CALLER_APP_HEADERS_REQUIRED,
            requestCallerApp,
            rawObo);
      }
      resolvedCallerApp = requestCallerApp;
      DefinitionApp callerDefinition = requireCallerAppDefinition(DiscoveryCache, resolvedCallerApp, rawObo);
      Config config = Config.from(callerDefinition, runtimeEnvironment, headers.requestMetadata().env());

      if (!config.delegatedUserEnabled()) {
        throw failure(
            Code.PERMISSION_DENIED, OboDenyRule.OBO_DISABLED_FOR_APP, resolvedCallerApp, rawObo);
      }

      normalizedObo =
          normalizeOboUser(
              rawObo,
              false,
              true,
              resolvedCallerApp,
              rawObo);
      effectiveUserSubject = normalizedObo;
    }

    return new RequestIdentity(
        effectiveUserSubject,
        callerAppSubject(resolvedCallerApp),
        headers.requestMetadata().requestId(),
        headers.requestMetadata().traceId(),
        normalizedObo);
  }

  public static <T> T callAsRequestIdentity(RequestIdentity identity, Supplier<T> action) {
    Objects.requireNonNull(identity, "identity");
    Objects.requireNonNull(action, "action");
    UserPrincipal previousUser = UserContext.get().orElse(null);
    CallerAppContext.ContextValue previousCallerApp = CallerAppContext.get().orElse(null);
    UserContext.set(new UserPrincipal(identity.effectiveUserSubject(), Map.of()));
    CallerAppContext.set(identity);
    try {
      return action.get();
    } finally {
      if (previousUser == null) {
        UserContext.clear();
      } else {
        UserContext.set(previousUser);
      }
      if (previousCallerApp == null) {
        CallerAppContext.clear();
      } else {
        CallerAppContext.set(previousCallerApp);
      }
    }
  }

  public static void runAsRequestIdentity(RequestIdentity identity, Runnable action) {
    callAsRequestIdentity(
        identity,
        () -> {
          action.run();
          return null;
        });
  }

  private static String normalizeAuthenticatedPrincipal(
      String principal, CallerAppIdentity callerApp, String rawObo) {
    String candidate = Strings.trimToNull(principal);
    if (candidate == null) {
      return null;
    }
    if (startsWithIgnoreCase(candidate, USER_PREFIX)) {
      String userId = normalizeUserId(candidate.substring(USER_PREFIX.length()));
      if (userId == null) {
        throw failure(
            Code.PERMISSION_DENIED, OboDenyRule.OBO_REQUIRES_APP_SUBJECT, callerApp, rawObo);
      }
      return USER_PREFIX + userId;
    }
    if (startsWithIgnoreCase(candidate, APP_PREFIX)) {
      String appId = normalizeAppId(candidate.substring(APP_PREFIX.length()));
      if (appId == null) {
        throw failure(
            Code.PERMISSION_DENIED, OboDenyRule.OBO_REQUIRES_APP_SUBJECT, callerApp, rawObo);
      }
      return APP_PREFIX + appId;
    }

    String withoutRealm = stripRealm(candidate);
    String asApp = normalizeAppId(withoutRealm);
    if (asApp != null) {
      return APP_PREFIX + asApp;
    }
    String userId = normalizeUserId(candidate);
    if (userId == null) {
      throw failure(
          Code.PERMISSION_DENIED, OboDenyRule.OBO_REQUIRES_APP_SUBJECT, callerApp, rawObo);
    }
    return USER_PREFIX + userId;
  }

  private static String normalizeOboUser(
      String rawObo,
      boolean requireCanonicalUser,
      boolean allowRawUsernameInDev,
      CallerAppIdentity callerApp,
      String originalObo) {
    String candidate = Strings.trimToNull(rawObo);
    if (candidate == null) {
      throw failure(
          Code.PERMISSION_DENIED,
          OboDenyRule.OBO_CANONICAL_REQUIRED,
          callerApp,
          originalObo);
    }

    if (startsWithIgnoreCase(candidate, USER_PREFIX)) {
      String userId = normalizeUserId(candidate.substring(USER_PREFIX.length()));
      if (userId == null) {
        throw failure(
            Code.PERMISSION_DENIED,
            OboDenyRule.OBO_CANONICAL_REQUIRED,
            callerApp,
            originalObo);
      }
      return USER_PREFIX + userId;
    }

    if (requireCanonicalUser) {
      throw failure(
          Code.PERMISSION_DENIED,
          OboDenyRule.OBO_CANONICAL_REQUIRED,
          callerApp,
          originalObo);
    }
    if (!allowRawUsernameInDev) {
      throw failure(
          Code.PERMISSION_DENIED,
          OboDenyRule.OBO_RAW_USERNAME_DISALLOWED,
          callerApp,
          originalObo);
    }
    String userId = normalizeUserId(candidate);
    if (userId == null) {
      throw failure(
          Code.PERMISSION_DENIED,
          OboDenyRule.OBO_CANONICAL_REQUIRED,
          callerApp,
          originalObo);
    }
    return USER_PREFIX + userId;
  }

  private static CallerAppIdentity normalizeCallerApp(CallerAppIdentity callerApp) {
    if (callerApp == null) {
      return new CallerAppIdentity(null, null, null);
    }
    return new CallerAppIdentity(callerApp.group(), callerApp.name(), callerApp.instance());
  }

  private static CallerAppIdentity mergeCallerAppForContext(
      CallerAppIdentity requestCallerApp, String authenticatedSubject) {
    if (requestCallerApp != null && requestCallerApp.isPresent()) {
      return requestCallerApp;
    }
    CallerAppIdentity authenticatedCaller = callerAppFromAppSubject(authenticatedSubject);
    if (authenticatedCaller != null && authenticatedCaller.isPresent()) {
      return authenticatedCaller;
    }
    return requestCallerApp == null ? new CallerAppIdentity(null, null, null) : requestCallerApp;
  }

  private static CallerAppIdentity callerAppFromAppSubject(String subject) {
    String normalized = Strings.trimToNull(subject);
    if (normalized == null || !startsWithIgnoreCase(normalized, APP_PREFIX)) {
      return null;
    }
    String appId = normalizeAppId(normalized.substring(APP_PREFIX.length()));
    if (appId == null) {
      return null;
    }
    int delimiter = appId.indexOf('/');
    if (delimiter <= 0 || delimiter >= appId.length() - 1) {
      return null;
    }
    return new CallerAppIdentity(appId.substring(0, delimiter), appId.substring(delimiter + 1), null);
  }

  private static String callerAppSubject(CallerAppIdentity callerApp) {
    if (callerApp == null || !callerApp.isPresent()) {
      return null;
    }
    return callerApp.asSubject();
  }

  private static DefinitionApp requireCallerAppDefinition(
      DiscoveryCache DiscoveryCache, CallerAppIdentity callerApp, String rawObo) {
    if (callerApp == null || !callerApp.isPresent()) {
      throw failure(
          Code.PERMISSION_DENIED,
          OboDenyRule.OBO_CALLER_APP_HEADERS_REQUIRED,
          callerApp,
          rawObo);
    }
    if (DiscoveryCache == null) {
      throw failure(
          Code.PERMISSION_DENIED, OboDenyRule.OBO_APP_DEFINITION_MISSING, callerApp, rawObo);
    }
    Optional<DefinitionApp> definition =
        DiscoveryCache.get(DefinitionApp.class, callerApp.group(), callerApp.name());
    if (definition.isEmpty()) {
      throw failure(
          Code.PERMISSION_DENIED, OboDenyRule.OBO_APP_DEFINITION_MISSING, callerApp, rawObo);
    }
    return definition.get();
  }

  private static String normalizeUserId(String value) {
    String candidate = Strings.trimToNull(value);
    if (candidate == null) {
      return null;
    }
    int slash = candidate.lastIndexOf('\\');
    if (slash >= 0 && slash < candidate.length() - 1) {
      candidate = candidate.substring(slash + 1);
    }
    candidate = stripRealm(candidate);
    candidate = Strings.trimToNull(candidate);
    if (candidate == null) {
      return null;
    }
    if (candidate.contains("/") || candidate.contains(":")) {
      return null;
    }
    for (int i = 0; i < candidate.length(); i++) {
      if (Character.isWhitespace(candidate.charAt(i))) {
        return null;
      }
    }
    return candidate;
  }

  private static String stripRealm(String value) {
    String candidate = Strings.trimToNull(value);
    if (candidate == null) {
      return null;
    }
    int realm = candidate.indexOf('@');
    if (realm > 0) {
      return candidate.substring(0, realm);
    }
    return candidate;
  }

  private static String normalizeAppId(String value) {
    String candidate = Strings.trimToNull(value);
    if (candidate == null) {
      return null;
    }
    if (candidate.startsWith("/") || candidate.endsWith("/")) {
      return null;
    }
    int firstSlash = candidate.indexOf('/');
    if (firstSlash <= 0 || firstSlash >= candidate.length() - 1) {
      return null;
    }
    int secondSlash = candidate.indexOf('/', firstSlash + 1);
    if (secondSlash >= 0) {
      return null;
    }
    String group = candidate.substring(0, firstSlash).trim();
    String app = candidate.substring(firstSlash + 1).trim();
    if (group.isEmpty() || app.isEmpty()) {
      return null;
    }
    if (!isSimpleToken(group) || !isSimpleToken(app)) {
      return null;
    }
    return group + "/" + app;
  }

  private static boolean isSimpleToken(String token) {
    if (token == null || token.isBlank()) {
      return false;
    }
    for (int i = 0; i < token.length(); i++) {
      char ch = token.charAt(i);
      boolean ok =
          (ch >= 'a' && ch <= 'z')
              || (ch >= 'A' && ch <= 'Z')
              || (ch >= '0' && ch <= '9')
              || ch == '-'
              || ch == '_'
              || ch == '.';
      if (!ok) {
        return false;
      }
    }
    return true;
  }

  private static boolean startsWithIgnoreCase(String value, String prefix) {
    if (value == null || prefix == null || value.length() < prefix.length()) {
      return false;
    }
    return value.substring(0, prefix.length()).toLowerCase(Locale.ROOT)
        .equals(prefix.toLowerCase(Locale.ROOT));
  }

  private static ResolutionException failure(
      Code code, OboDenyRule rule, CallerAppIdentity callerApp, String oboUser) {
    return new ResolutionException(
        code,
        "OBO_DENIED rule="
            + Objects.requireNonNull(rule, "rule").name()
            + " caller="
            + callerAppToken(callerApp)
            + " obo="
            + oboToken(oboUser));
  }

  private static String callerAppToken(CallerAppIdentity callerApp) {
    if (callerApp == null || !callerApp.isPresent()) {
      return "<none>";
    }
    return callerApp.asSubject();
  }

  private static String oboToken(String oboUser) {
    String normalized = Strings.trimToNull(oboUser);
    return normalized == null ? "<none>" : normalized;
  }

  private enum OboDenyRule {
    OBO_PRESENT_BUT_NOT_AUTHENTICATED,
    OBO_CALLER_APP_HEADERS_REQUIRED,
    OBO_REQUIRES_APP_SUBJECT,
    OBO_APP_DEFINITION_MISSING,
    OBO_DISABLED_FOR_APP,
    OBO_CANONICAL_REQUIRED,
    OBO_RAW_USERNAME_DISALLOWED,
    OBO_UNTRUSTED_HEADERS
  }

  public enum Code {
    UNAUTHENTICATED,
    PERMISSION_DENIED
  }

  public static final class ResolutionException extends RuntimeException {
    private final Code code;

    public ResolutionException(Code code, String message) {
      super(message);
      this.code = Objects.requireNonNull(code, "code");
    }

    public Code code() {
      return code;
    }
  }

  public record Config(
      boolean delegatedUserEnabled,
      boolean delegatedUserRequireCanonical,
      boolean delegatedUserAllowRawUsernameInDev,
      boolean delegatedUserTrustUserHeaders,
      boolean devEnvironment) {
    public static Config from(
        DefinitionApp appDefinition, String runtimeEnvironment, String requestEnvironment) {
      String effectiveEnvironment = Strings.firstNonBlank(runtimeEnvironment, requestEnvironment);
      boolean dev = isDevEnvironment(effectiveEnvironment);
      boolean enabled =
          appDefinition != null && Boolean.TRUE.equals(appDefinition.delegatedUserEnabled());
      boolean requireCanonical =
          appDefinition == null || appDefinition.delegatedUserRequireCanonical() == null
              ? true
              : appDefinition.delegatedUserRequireCanonical();
      boolean allowRawDev =
          appDefinition != null
              && Boolean.TRUE.equals(appDefinition.delegatedUserAllowRawUsernameInDev());
      boolean trustHeaders =
          appDefinition != null && Boolean.TRUE.equals(appDefinition.delegatedUserTrustUserHeaders());
      return new Config(enabled, requireCanonical, allowRawDev, trustHeaders, dev);
    }

    private static boolean isDevEnvironment(String value) {
      String normalized = Strings.trimToNull(value);
      if (normalized == null) {
        return false;
      }
      return normalized.equalsIgnoreCase("dev")
          || normalized.equalsIgnoreCase("local")
          || normalized.equalsIgnoreCase("localhost");
    }
  }
}
