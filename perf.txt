package org.pronzato.fabric.api.control.status.service;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.SerializationFeature;
import java.io.Closeable;
import java.util.Map;
import java.util.Objects;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.ScheduledFuture;
import java.util.concurrent.TimeUnit;
import org.pronzato.fabric.api.control.ControlConnection;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryTopics;
import org.pronzato.fabric.api.control.discovery.codec.DiscoveryCodec;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEnvelope;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionHbService;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionStatusService;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionType;
import org.pronzato.fabric.api.services.common.ServiceAdvertise;
import org.pronzato.fabric.api.util.Strings;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public final class ServiceStatusPublisher implements Closeable {
  private static final Logger LOGGER = LoggerFactory.getLogger(ServiceStatusPublisher.class);
  private static final boolean STATUS_DEBUG = false;
  private static final long DEFAULT_HEARTBEAT_INTERVAL_SECONDS = 10L;
  private static final long DEFAULT_STATUS_INTERVAL_SECONDS = 30L;
  private static final int DEFAULT_TTL_SECONDS = 30;
  private static final String STATUS_ONLINE = "ONLINE";
  private static final String STATUS_OFFLINE = "OFFLINE";

  private final ObjectMapper om =
      DiscoveryCodec.mapper().copy().configure(SerializationFeature.ORDER_MAP_ENTRIES_BY_KEYS, true);
  private final ControlConnection control;
  private final ScheduledExecutorService ses;
  private final boolean ownsScheduler;
  private final long heartbeatIntervalSec;
  private final long statusIntervalSec;
  private final int ttlSeconds;
  private final String group;
  private final String region;
  private final String serviceName;
  private final String serviceInstanceId;
  private final String serviceHost;
  private final int servicePort;
  private final boolean tlsRequired;
  private final String routeHost;
  private final Integer routePort;
  private final String routeAuthority;
  private final long serviceStartEpochMs;
  private final String serviceAppName;
  private final String serviceAppInstanceId;
  private final String serviceAppGroup;
  private final Map<String, String> attrs;
  private volatile String sessionId;
  private volatile String sessionKey;
  private volatile boolean open = true;
  private volatile ScheduledFuture<?> heartbeatTask;
  private volatile ScheduledFuture<?> statusTask;

  public ServiceStatusPublisher(
      ControlConnection control,
      String group,
      String service,
      String instanceId,
      String host,
      int port,
      String region,
      long heartbeatSeconds,
      long statusSeconds,
      int ttlSeconds,
      boolean tlsRequired,
      String routeHost,
      Integer routePort,
      String routeAuthority,
      String appName,
      String appInstance,
      String appGroup,
      Map<String, String> attrs,
      ScheduledExecutorService scheduler) {
    this.control = Objects.requireNonNull(control, "control");
    this.group = Objects.requireNonNull(group, "group");
    this.serviceName = Objects.requireNonNull(service, "service");
    this.serviceInstanceId = Objects.requireNonNull(instanceId, "instanceId");
    this.serviceHost = host;
    this.servicePort = port;
    this.heartbeatIntervalSec =
        heartbeatSeconds <= 0 ? DEFAULT_HEARTBEAT_INTERVAL_SECONDS : heartbeatSeconds;
    this.statusIntervalSec = statusSeconds <= 0 ? DEFAULT_STATUS_INTERVAL_SECONDS : statusSeconds;
    this.ttlSeconds = ttlSeconds <= 0 ? DEFAULT_TTL_SECONDS : ttlSeconds;
    this.tlsRequired = tlsRequired;
    this.routeHost = routeHost;
    this.routePort = routePort;
    this.routeAuthority = routeAuthority;
    this.serviceStartEpochMs = System.currentTimeMillis();
    Map<String, String> attrCopy = Strings.isEmpty(attrs) ? Map.of() : Map.copyOf(attrs);
    this.attrs = new ConcurrentHashMap<>(attrCopy);
    String resolvedRegion = region;
    if (Strings.isBlank(resolvedRegion)) {
      resolvedRegion = attrCopy.get("region");
    }
    this.region = Strings.isBlank(resolvedRegion) ? "AMRS" : resolvedRegion;
    this.serviceAppName = Strings.firstNonBlank(appName, serviceName);
    this.serviceAppInstanceId = Strings.firstNonBlank(appInstance, serviceInstanceId);
    this.serviceAppGroup = Strings.firstNonBlank(appGroup, group);
    if (scheduler == null) {
      this.ses = Executors.newSingleThreadScheduledExecutor();
      this.ownsScheduler = true;
    } else {
      this.ses = scheduler;
      this.ownsScheduler = false;
    }
  }

  public void start() {
    publishServiceStartOrThrow();
    heartbeatTask =
        ses.scheduleAtFixedRate(
            this::publishHeartbeatSafe, heartbeatIntervalSec, heartbeatIntervalSec, TimeUnit.SECONDS);
    statusTask =
        ses.scheduleAtFixedRate(
            this::publishStatusSafe, statusIntervalSec, statusIntervalSec, TimeUnit.SECONDS);
  }

  public ServiceStatusPublisher withSessionLineage(String sessionId, String sessionKey) {
    this.sessionId = Strings.trimToNull(sessionId);
    this.sessionKey = Strings.trimToNull(sessionKey);
    return this;
  }

  @Override
  public void close() {
    if (!open) {
      return;
    }
    open = false;
    publishStatusSafe(STATUS_OFFLINE);
    publishHeartbeatSafe();
    ScheduledFuture<?> task = heartbeatTask;
    heartbeatTask = null;
    if (task != null) {
      task.cancel(true);
    }
    ScheduledFuture<?> status = statusTask;
    statusTask = null;
    if (status != null) {
      status.cancel(true);
    }
    if (ownsScheduler) {
      ses.shutdownNow();
    }
  }

  private void publishHeartbeatSafe() {
    try {
      publishServiceHeartbeat(STATUS_ONLINE);
    } catch (Exception e) {
      LOGGER.error("[Status] service heartbeat publish failed", e);
    }
  }

  private void publishStatusSafe() {
    publishStatusSafe(STATUS_ONLINE);
  }

  private void publishStatusSafe(String status) {
    try {
      publishServiceStatus(status);
    } catch (Exception e) {
      LOGGER.error("[Status] service status publish failed for status={}", status, e);
    }
  }

  private void publishServiceStartOrThrow() {
    validateServiceHeartbeatContract();
    try {
      publishServiceStatus(STATUS_ONLINE);
      publishServiceHeartbeat(STATUS_ONLINE);
    } catch (Exception ex) {
      throw new IllegalStateException(
          "Failed to publish initial service heartbeat/status: serviceId="
              + valueOrUnknown(group)
              + "/"
              + valueOrUnknown(serviceName)
              + ", instanceId="
              + valueOrUnknown(serviceInstanceId),
          ex);
    }
  }

  private void validateServiceHeartbeatContract() {
    ServiceAdvertise.requireRoutableHost(serviceHost, "advertiseHost");
    if (servicePort <= 0) {
      throw new IllegalStateException(
          "Failed to publish service heartbeat: serviceId="
              + valueOrUnknown(group)
              + "/"
              + valueOrUnknown(serviceName)
              + ", instanceId="
              + valueOrUnknown(serviceInstanceId)
              + ", advertisePort is missing/invalid. Configure bind.port or advertise.port.");
    }
    if (tlsRequired) {
      String tlsTermination = Strings.trimToNull(attrs.get("tlsTermination"));
      if (tlsTermination == null) {
        throw new IllegalStateException(
            "Failed to publish service heartbeat: serviceId="
                + valueOrUnknown(group)
                + "/"
                + valueOrUnknown(serviceName)
                + ", instanceId="
                + valueOrUnknown(serviceInstanceId)
                + ", tlsEnabled=true but tlsTermination is missing.");
      }
    }
  }

  private void publishServiceStatus(String status) throws Exception {
    DefinitionStatusService definition = buildServiceStatus(status);
    if (definition == null) {
      return;
    }
    publishStatusDefinition(definition);
  }

  private void publishServiceHeartbeat(String status) throws Exception {
    DefinitionHbService heartbeat = buildServiceHeartbeat();
    if (heartbeat == null) {
      return;
    }
    String payload = om.writeValueAsString(heartbeat);
    if (STATUS_DEBUG) {
      LOGGER.debug("[Status] publish topic={} status={} payload={}", DiscoveryTopics.HEARTBEAT, status, payload);
    }
    int ttl = ttlSeconds > 0 ? ttlSeconds : 0;
    control.amps().publish(DiscoveryTopics.HEARTBEAT, payload, ttl);
  }

  private void publishStatusDefinition(
      org.pronzato.fabric.api.control.discovery.definitions.Definition definition) throws Exception {
    if (definition == null) {
      return;
    }
    String payload = om.writeValueAsString(definition);
    if (STATUS_DEBUG) {
      LOGGER.debug("[Status] publish topic={} payload={}", DiscoveryTopics.STATUS, payload);
    }
    control.amps().publish(DiscoveryTopics.STATUS, payload);
  }

  private DefinitionStatusService buildServiceStatus(String status) {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedName = Strings.trimToNull(serviceName);
    String normalizedInstance = Strings.trimToNull(serviceInstanceId);
    if (normalizedGroup == null || normalizedName == null || normalizedInstance == null) {
      return null;
    }
    DefinitionEnvelope envelope = statusEnvelope(DefinitionType.STATUS_SERVICE, normalizedName);
    if (envelope == null) {
      return null;
    }
    long nowMs = System.currentTimeMillis();
    Integer port = servicePort > 0 ? servicePort : null;
    String transport = resolveTransport();
    String uri = resolveUri(port);
    Map<String, String> tagsSnapshot = attrs.isEmpty() ? Map.of() : Map.copyOf(attrs);
    String advertiseHost = Strings.trimToNull(serviceHost);
    Integer advertisePort = port;
    String bindHost = Strings.trimToNull(attrs.get("bindHost"));
    Integer bindPort = parseInteger(attrs.get("bindPort"));
    Boolean tlsEnabled = tlsRequired;
    String tlsTermination =
        firstNonBlank(Strings.trimToNull(attrs.get("tlsTermination")), tlsRequired ? "PASSTHROUGH" : null);
    Boolean kerberosEnabled = parseBoolean(attrs.get("kerberosEnabled"));
    if (kerberosEnabled == null) {
      kerberosEnabled = false;
    }
    String kerberosSpn = Strings.trimToNull(attrs.get("kerberosSpn"));
    String derivedSpn = Strings.trimToNull(attrs.get("derivedSpn"));
    Long startEpochMs = serviceStartEpochMs > 0L ? serviceStartEpochMs : null;
    DefinitionStatusService.State state = null;
    if (status != null) {
      try {
        state = DefinitionStatusService.State.valueOf(status);
      } catch (IllegalArgumentException ignore) {
        state = DefinitionStatusService.State.UNKNOWN;
      }
    }
    if (state == null) {
      state = DefinitionStatusService.State.UNKNOWN;
    }
    return new DefinitionStatusService(
        envelope,
        normalizedGroup,
        DefinitionType.STATUS_SERVICE,
        normalizedName,
        normalizedInstance,
        state,
        nowMs,
        startEpochMs,
        null,
        transport,
        uri,
        advertiseHost,
        advertisePort,
        bindHost,
        bindPort,
        tlsEnabled,
        tlsTermination,
        kerberosEnabled,
        kerberosSpn,
        derivedSpn,
        Strings.trimToNull(serviceHost),
        port,
        tlsRequired,
        Strings.trimToNull(routeHost),
        routePort,
        Strings.trimToNull(routeAuthority),
        Strings.trimToNull(region),
        Strings.trimToNull(serviceAppName),
        Strings.trimToNull(serviceAppInstanceId),
        Strings.trimToNull(serviceAppGroup),
        tagsSnapshot,
        sessionId,
        sessionKey);
  }

  private DefinitionHbService buildServiceHeartbeat() {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedName = Strings.trimToNull(serviceName);
    String normalizedInstance = Strings.trimToNull(serviceInstanceId);
    if (normalizedGroup == null || normalizedName == null || normalizedInstance == null) {
      return null;
    }
    DefinitionEnvelope envelope = statusEnvelope(DefinitionType.HEARTBEAT_SERVICE, normalizedName);
    if (envelope == null) {
      return null;
    }
    Integer port = servicePort > 0 ? servicePort : null;
    String transport = resolveTransport();
    String uri = resolveUri(port);
    Map<String, String> tagsSnapshot = attrs.isEmpty() ? Map.of() : Map.copyOf(attrs);
    String advertiseHost = Strings.trimToNull(serviceHost);
    Integer advertisePort = port;
    String bindHost = Strings.trimToNull(attrs.get("bindHost"));
    Integer bindPort = parseInteger(attrs.get("bindPort"));
    Boolean tlsEnabled = tlsRequired;
    String tlsTermination =
        firstNonBlank(Strings.trimToNull(attrs.get("tlsTermination")), tlsRequired ? "PASSTHROUGH" : null);
    Boolean kerberosEnabled = parseBoolean(attrs.get("kerberosEnabled"));
    if (kerberosEnabled == null) {
      kerberosEnabled = false;
    }
    String kerberosSpn = Strings.trimToNull(attrs.get("kerberosSpn"));
    String derivedSpn = Strings.trimToNull(attrs.get("derivedSpn"));
    long nowMs = System.currentTimeMillis();
    return new DefinitionHbService(
        envelope,
        normalizedGroup,
        DefinitionType.HEARTBEAT_SERVICE,
        normalizedName,
        normalizedInstance,
        transport,
        uri,
        advertiseHost,
        advertisePort,
        bindHost,
        bindPort,
        tlsEnabled,
        tlsTermination,
        kerberosEnabled,
        kerberosSpn,
        derivedSpn,
        Strings.trimToNull(serviceHost),
        port,
        tlsRequired,
        Strings.trimToNull(routeHost),
        routePort,
        Strings.trimToNull(routeAuthority),
        Strings.trimToNull(region),
        nowMs,
        tagsSnapshot);
  }

  private DefinitionEnvelope statusEnvelope(DefinitionType type, String name) {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedName = Strings.trimToNull(name);
    if (normalizedGroup == null || normalizedName == null || type == null) {
      return null;
    }
    return new DefinitionEnvelope(
        normalizedGroup,
        type,
        normalizedName,
        null,
        null,
        null,
        null,
        null,
        null,
        null);
  }

  private String resolveTransport() {
    if (Strings.isEmpty(attrs)) {
      return null;
    }
    return firstNonBlank(attrs.get("transport"), attrs.get("protocol"), attrs.get("serviceType"));
  }

  private String resolveUri(Integer port) {
    String authority = firstNonBlank(routeAuthority, routeHost, serviceHost);
    if (authority == null) {
      return null;
    }
    if (authority.contains(":")) {
      return authority;
    }
    int resolvedPort = port == null ? 0 : port;
    if (resolvedPort <= 0 && routePort != null) {
      resolvedPort = routePort;
    }
    return resolvedPort > 0 ? authority + ":" + resolvedPort : authority;
  }

  private static String firstNonBlank(String... values) {
    if (values == null) {
      return null;
    }
    for (String value : values) {
      if (Strings.isNotBlank(value)) {
        return value.trim();
      }
    }
    return null;
  }

  private static Integer parseInteger(String value) {
    if (Strings.isBlank(value)) {
      return null;
    }
    try {
      return Integer.parseInt(value.trim());
    } catch (NumberFormatException ignore) {
      return null;
    }
  }

  private static Boolean parseBoolean(String value) {
    if (Strings.isBlank(value)) {
      return null;
    }
    String normalized = value.trim();
    if ("true".equalsIgnoreCase(normalized) || "yes".equalsIgnoreCase(normalized)) {
      return true;
    }
    if ("false".equalsIgnoreCase(normalized) || "no".equalsIgnoreCase(normalized)) {
      return false;
    }
    return null;
  }

  private static String valueOrUnknown(String value) {
    return Strings.isBlank(value) ? "<unknown>" : value;
  }
}
