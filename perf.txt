public final class OdbcSessionSupport {
  private static final String PLATFORM_GROUP = "platform";
  private static final String ODBC_QUERY_FUNCTION = "odbc_query(?, ?)";
  private static final String ODBC_QUERY_SQL = "select * from odbc_query(?, ?)";
  private static final String ODBC_CONNECTION_PROBE_SQL = "select 1 as fabric_odbc_probe";

  private OdbcSessionSupport() {}

  public static OdbcHandle installDatasetView(
      DuckdbSession session,
      DiscoveryCache cache,
      DatasetId datasetId,
      DefinitionDataset.OdbcBinding binding) {
    Objects.requireNonNull(datasetId, "datasetId");
    Objects.requireNonNull(binding, "binding");
    String canonicalDatasetId = datasetId.toString();
    OdbcConnectionSpec connection =
        resolveConnection(
            cache,
            datasetId.group(),
            binding.connectionName(),
            binding.databaseType(),
            canonicalDatasetId);
    String remoteSql = datasetSql(binding, canonicalDatasetId);
    try {
      installDatasetViewUsingParameterizedQuery(session, datasetId, connection, remoteSql);
      return OdbcHandle.noOp(session, connection.connectionName(), connection.databaseType());
    } catch (RuntimeException ex) {
      if (!shouldFallbackToOdbcScan(ex)) {
        throw queryFailed(canonicalDatasetId, connection, ex);
      }
      try {
        installDatasetViewUsingOdbcScan(session, datasetId, connection, remoteSql);
        return OdbcHandle.noOp(session, connection.connectionName(), connection.databaseType());
      } catch (RuntimeException fallbackEx) {
        throw queryFailed(canonicalDatasetId, connection, fallbackEx);
      }
    }
  }

  public static OdbcQueryResult openQueryResult(
      DuckdbSession session, OdbcConnectionSpec spec, String remoteSql) {
    Objects.requireNonNull(session, "session");
    Objects.requireNonNull(spec, "spec");
    String resolvedRemoteSql = Strings.requireNonBlank(remoteSql, "remoteSql");
    try {
      DuckdbExtensions.requireOdbc(session);
    } catch (RuntimeException ex) {
      throw ex;
    }
    String queryConnectionString = queryConnectionString(spec);
    PreparedStatement preparedStatement = null;
    try {
      preparedStatement = session.prepareStatement(ODBC_QUERY_SQL);
      preparedStatement.setString(1, queryConnectionString);
      preparedStatement.setString(2, resolvedRemoteSql);
      ResultSet resultSet = preparedStatement.executeQuery();
      return new OdbcQueryResult(
          preparedStatement, resultSet, OdbcQueryPath.ODBC_QUERY_PARAMETERIZED);
    } catch (SQLException | RuntimeException ex) {
      closeStatementQuietly(preparedStatement);
      if (!shouldFallbackToOdbcScan(ex)) {
        throw new IllegalStateException(compactReason(ex), ex);
      }
      Statement fallbackStatement = null;
      try {
        fallbackStatement = session.createStatement();
        ResultSet fallbackResultSet =
            fallbackStatement.executeQuery(
                "select * from " + odbcScanExpression(queryConnectionString, resolvedRemoteSql));
        return new OdbcQueryResult(
            fallbackStatement, fallbackResultSet, OdbcQueryPath.ODBC_SCAN_LITERAL);
      } catch (SQLException | RuntimeException fallbackEx) {
        closeStatementQuietly(fallbackStatement);
        throw new IllegalStateException(compactReason(fallbackEx), fallbackEx);
      }
    }
  }

  public static OdbcConnectionHealth probeConnection(DuckdbSession session, OdbcConnectionSpec spec) {
    Objects.requireNonNull(session, "session");
    Objects.requireNonNull(spec, "spec");
    long startNanos = System.nanoTime();
    List<String> warnings = new ArrayList<>();
    DuckdbExtensions.OdbcExtensionLoadResult extensionResult = null;
    try {
      extensionResult = DuckdbExtensions.loadOdbc(session);
      OdbcQueryPath queryPath;
      String remoteProductName;
      String remoteVersion;
      try (OdbcQueryResult queryResult = openQueryResult(session, spec, ODBC_CONNECTION_PROBE_SQL)) {
        queryPath = queryResult.path();
        ResultSet resultSet = queryResult.resultSet();
        if (!resultSet.next()) {
          warnings.add("probe_query_returned_no_rows");
        }
        remoteProductName = resolveRemoteProductName(resultSet);
        remoteVersion = resolveRemoteVersion(resultSet);
        if (remoteProductName == null) {
          warnings.add("remote_product_name_unavailable");
        }
        if (remoteVersion == null) {
          warnings.add("remote_version_unavailable");
        }
      }
      boolean fallbackInUse =
          queryPath == OdbcQueryPath.ODBC_SCAN_LITERAL
              || (extensionResult != null && extensionResult.fallbackInUse());
      return new OdbcConnectionHealth(
          true,
          spec.connectionName(),
          spec.databaseType(),
          extensionResult == null ? null : extensionResult.extensionNameUsed(),
          extensionResult != null && extensionResult.extensionLoaded(),
          true,
          queryPath,
          "ODBC_CONNECTION_OK",
          remoteProductName,
          remoteVersion,
          elapsedMillis(startNanos),
          List.copyOf(warnings),
          fallbackInUse);
    } catch (SQLException | RuntimeException ex) {
      String message = probeFailureMessage(spec, ex);
      return new OdbcConnectionHealth(
          false,
          spec.connectionName(),
          spec.databaseType(),
          extensionResult == null ? null : extensionResult.extensionNameUsed(),
          extensionResult != null && extensionResult.extensionLoaded(),
          !isExtensionUnavailable(message),
          null,
          message,
          null,
          null,
          elapsedMillis(startNanos),
          List.copyOf(warnings),
          extensionResult != null && extensionResult.fallbackInUse());
    }
  }

  public static OdbcSchemaInspection inspectSchema(
      DuckdbSession session, OdbcConnectionSpec spec, String remoteSql) {
    Objects.requireNonNull(session, "session");
    Objects.requireNonNull(spec, "spec");
    String resolvedRemoteSql = Strings.requireNonBlank(remoteSql, "remoteSql");
    long startNanos = System.nanoTime();
    List<OdbcColumnMetadata> columns = new ArrayList<>();
    List<String> warnings = new ArrayList<>();
    try (OdbcQueryResult result =
        openQueryResult(
            session,
            spec,
            "select * from (" + resolvedRemoteSql + ") as fabric_odbc_schema_preview limit 0")) {
      ResultSetMetaData metadata = result.resultSet().getMetaData();
      int count = metadata.getColumnCount();
      for (int index = 1; index <= count; index++) {
        String name = Strings.firstNonBlank(metadata.getColumnLabel(index), metadata.getColumnName(index));
        int nullableCode = metadata.isNullable(index);
        String nullability =
            nullableCode == ResultSetMetaData.columnNoNulls
                ? "NOT_NULL"
                : nullableCode == ResultSetMetaData.columnNullable ? "NULLABLE" : "UNKNOWN";
        columns.add(
            new OdbcColumnMetadata(
                name,
                metadata.getColumnType(index),
                metadata.getColumnTypeName(index),
                metadata.getPrecision(index),
                metadata.getScale(index),
                nullability));
      }
      return new OdbcSchemaInspection(
          true,
          spec.connectionName(),
          spec.databaseType(),
          result.path(),
          "ODBC_SCHEMA_OK columns=" + columns.size(),
          List.copyOf(columns),
          elapsedMillis(startNanos),
          List.copyOf(warnings));
    } catch (SQLException ex) {
      return new OdbcSchemaInspection(
          false,
          spec.connectionName(),
          spec.databaseType(),
          null,
          compactReason(ex),
          List.of(),
          elapsedMillis(startNanos),
          List.copyOf(warnings));
    } catch (RuntimeException ex) {
      return new OdbcSchemaInspection(
          false,
          spec.connectionName(),
          spec.databaseType(),
          null,
          compactReason(ex),
          List.of(),
          elapsedMillis(startNanos),
          List.copyOf(warnings));
    }
  }

  public static OdbcRuntimeStatus runtimeStatus(boolean configured, OdbcConnectionHealth health) {
    if (!configured) {
      return new OdbcRuntimeStatus(
          false,
          false,
          false,
          false,
          ODBC_QUERY_FUNCTION,
          false,
          null,
          "NOT_CONFIGURED",
          0L,
          List.of());
    }
    if (health == null) {
      return new OdbcRuntimeStatus(
          true,
          false,
          false,
          false,
          ODBC_QUERY_FUNCTION,
          false,
          null,
          "NOT_TESTED",
          0L,
          List.of());
    }
    return new OdbcRuntimeStatus(
        true,
        health.extensionAvailable(),
        health.extensionLoaded(),
        health.ok(),
        ODBC_QUERY_FUNCTION,
        health.fallbackInUse(),
        health.extensionNameUsed(),
        Strings.firstNonBlank(Strings.trimToNull(health.message()), health.ok() ? "OK" : "FAILED"),
        health.roundTripMillis(),
        health.warnings());
  }

  private static void installDatasetViewUsingParameterizedQuery(
      DuckdbSession session, DatasetId datasetId, OdbcConnectionSpec spec, String remoteSql) {
    Objects.requireNonNull(session, "session");
    Objects.requireNonNull(datasetId, "datasetId");
    Objects.requireNonNull(spec, "spec");
    String normalizedRemoteSql = Strings.requireNonBlank(remoteSql, "remoteSql");
    String queryConnectionString = queryConnectionString(spec);
    try {
      DuckdbExtensions.requireOdbc(session);
    } catch (RuntimeException ex) {
      throw ex;
    }
    String sql =
        "create or replace view "
            + quoteIdentifier(viewName(datasetId))
            + " as select * from odbc_query(?, ?)";
    try (PreparedStatement statement = session.prepareStatement(sql)) {
      statement.setString(1, queryConnectionString);
      statement.setString(2, normalizedRemoteSql);
      statement.execute();
    } catch (SQLException ex) {
      throw new IllegalStateException(ex.getMessage(), ex);
    }
  }

  private static void installDatasetViewUsingOdbcScan(
      DuckdbSession session, DatasetId datasetId, OdbcConnectionSpec spec, String remoteSql) {
    Objects.requireNonNull(session, "session");
    Objects.requireNonNull(datasetId, "datasetId");
    Objects.requireNonNull(spec, "spec");
    String normalizedRemoteSql = Strings.requireNonBlank(remoteSql, "remoteSql");
    String queryConnectionString = queryConnectionString(spec);
    session.exec(
        "create or replace view "
            + quoteIdentifier(viewName(datasetId))
            + " as select * from "
            + odbcScanExpression(queryConnectionString, normalizedRemoteSql));
  }

  private static boolean shouldFallbackToOdbcScan(Throwable cause) {
    String reason =
        Strings.firstNonBlank(
            Strings.trimToNull(cause == null ? null : cause.getMessage()),
            cause == null ? null : cause.getClass().getSimpleName());
    if (reason == null) {
      return false;
    }
    String normalized = reason.toLowerCase();
    if (!normalized.contains("odbc_query")) {
      return false;
    }
    return normalized.contains("does not exist")
        || normalized.contains("not found")
        || normalized.contains("catalog error")
        || normalized.contains("binder error")
        || normalized.contains("no function matches")
        || normalized.contains("unknown table function")
        || (normalized.contains("parameter") && normalized.contains("not"));
  }

  private static String odbcScanExpression(String queryConnectionString, String remoteSql) {
    return "odbc_scan('"
        + escapeSqlLiteral(Strings.requireNonBlank(queryConnectionString, "queryConnectionString"))
        + "', '"
        + escapeSqlLiteral(Strings.requireNonBlank(remoteSql, "remoteSql"))
        + "')";
  }

  private static String queryConnectionString(OdbcConnectionSpec spec) {
    Objects.requireNonNull(spec, "spec");
    String base = Strings.requireNonBlank(spec.connectionString(), "connectionString");
    StringBuilder builder = new StringBuilder(base);
    if (!base.endsWith(";")) {
      builder.append(';');
    }
    appendCredentialOption(builder, "UID", spec.username(), "UID", "USER ID");
    appendCredentialOption(builder, "PWD", spec.password(), "PWD", "PASSWORD");
    return builder.toString();
  }

  private static void appendCredentialOption(
      StringBuilder builder, String optionName, String value, String... aliases) {
    String normalizedValue = Strings.trimToNull(value);
    if (normalizedValue == null) {
      return;
    }
    if (hasOption(builder.toString(), aliases)) {
      return;
    }
    builder.append(optionName).append('=').append(normalizedValue).append(';');
  }

  private static boolean hasOption(String connectionString, String... aliases) {
    String normalized = Strings.trimToNull(connectionString);
    if (normalized == null || aliases == null || aliases.length == 0) {
      return false;
    }
    for (String token : normalized.split(";")) {
      int equalsIndex = token.indexOf('=');
      if (equalsIndex <= 0) {
        continue;
      }
      String key = Strings.trimToNull(token.substring(0, equalsIndex));
      if (key == null) {
        continue;
      }
      for (String alias : aliases) {
        if (alias != null && key.equalsIgnoreCase(alias)) {
          return true;
        }
      }
    }
    return false;
  }

  private static void closeStatementQuietly(Statement statement) {
    if (statement == null) {
      return;
    }
    try {
      statement.close();
    } catch (SQLException | RuntimeException ignore) {
      // best effort cleanup.
    }
  }

  private static void closeResultSetQuietly(ResultSet resultSet) {
    if (resultSet == null) {
      return;
    }
    try {
      resultSet.close();
    } catch (SQLException | RuntimeException ignore) {
      // best effort cleanup.
    }
  }

  private static String compactReason(Throwable cause) {
    String message =
        Strings.firstNonBlank(
            Strings.trimToNull(cause == null ? null : cause.getMessage()),
            cause == null ? null : cause.getClass().getSimpleName());
    if (message == null) {
      return "unknown";
    }
    return message.replace('\n', ' ').replace('\r', ' ');
  }

  private static String probeFailureMessage(OdbcConnectionSpec spec, Throwable cause) {
    String reason = compactReason(cause);
    if (reason.startsWith("ODBC_EXTENSION_UNAVAILABLE") || reason.startsWith("ODBC_EXTENSION_LOAD_FAILED")) {
      return reason;
    }
    return connectionFailed(
            spec == null ? null : spec.connectionName(),
            spec == null ? null : spec.databaseType(),
            reason,
            cause)
        .getMessage();
  }

  private static boolean isExtensionUnavailable(String message) {
    String normalized = Strings.firstNonBlank(Strings.trimToNull(message), "").toUpperCase();
    return normalized.startsWith("ODBC_EXTENSION_UNAVAILABLE")
        || normalized.startsWith("ODBC_EXTENSION_LOAD_FAILED");
  }

  private static String resolveRemoteProductName(ResultSet resultSet) {
    if (resultSet == null) {
      return null;
    }
    try {
      String value = Strings.trimToNull(resultSet.getStatement().getConnection().getMetaData().getDatabaseProductName());
      if (value == null || value.equalsIgnoreCase("duckdb")) {
        return null;
      }
      return value;
    } catch (SQLException | RuntimeException ignore) {
      return null;
    }
  }

  private static String resolveRemoteVersion(ResultSet resultSet) {
    if (resultSet == null) {
      return null;
    }
    try {
      String value = Strings.trimToNull(resultSet.getStatement().getConnection().getMetaData().getDatabaseProductVersion());
      if (value == null) {
        return null;
      }
      if (value.toLowerCase().contains("duckdb")) {
        return null;
      }
      return value;
    } catch (SQLException | RuntimeException ignore) {
      return null;
    }
  }

  private static long elapsedMillis(long startNanos) {
    return Math.max(0L, (System.nanoTime() - startNanos) / 1_000_000L);
  }

  public static String ingestSourceSql(
      DefinitionIngestSource.OdbcSource source, String ingestName, String connectionName) {
    Objects.requireNonNull(source, "source");
    String remoteSql = Strings.trimToNull(source.querySql());
    if (remoteSql != null) {
      return remoteSql;
    }
    String tableName = Strings.trimToNull(source.tableName());
    if (tableName != null) {
      return "select * from " + tableName;
    }
    String resolvedConnectionName =
        Strings.firstNonBlank(Strings.trimToNull(connectionName), Strings.trimToNull(source.connectionName()));
    throw ingestSourceFailed(
        ingestName,
        resolvedConnectionName,
        source.databaseType(),
        "missing_table_or_query",
        null);
  }

  public static String datasetSql(DefinitionDataset.OdbcBinding binding, String datasetId) {
    Objects.requireNonNull(binding, "binding");
    return datasetRemoteSql(binding, datasetId);
  }

  public static OdbcConnectionSpec resolveConnection(
      DiscoveryCache cache,
      String preferredGroup,
      String connectionName,
      DefinitionConnection.Odbc.DatabaseType requestedDatabaseType,
      String datasetOrIngestId) {
    Objects.requireNonNull(cache, "cache");
    String resolvedConnectionName = Strings.requireNonBlank(connectionName, "connectionName");
    String normalizedGroup = Strings.trimToNull(preferredGroup);
    if (normalizedGroup == null) {
      normalizedGroup = PLATFORM_GROUP;
    }
    Optional<DefinitionConnection> resolved =
        cache.get(DefinitionConnection.class, normalizedGroup, resolvedConnectionName);
    if (resolved.isEmpty() && !PLATFORM_GROUP.equals(normalizedGroup)) {
      resolved = cache.get(DefinitionConnection.class, PLATFORM_GROUP, resolvedConnectionName);
    }
    if (resolved.isEmpty()) {
      throw connectionFailed(
          resolvedConnectionName,
          requestedDatabaseType,
          "connection_not_found id=" + Strings.firstNonBlank(datasetOrIngestId, "<unknown>"),
          null);
    }
    DefinitionConnection connection = resolved.get();
    if (connection.connectionType() != DefinitionConnection.ConnectionType.ODBC
        || connection.odbc() == null) {
      throw connectionFailed(
          resolvedConnectionName,
          requestedDatabaseType,
          "connection_not_odbc",
          null);
    }
    DefinitionConnection.Odbc odbc = connection.odbc();
    DefinitionConnection.Odbc.DatabaseType databaseType =
        requestedDatabaseType == null
            ? (odbc.databaseType() == null ? DefinitionConnection.Odbc.DatabaseType.OTHER : odbc.databaseType())
            : requestedDatabaseType;
    String connectionString = buildConnectionString(odbc);
    String username = Strings.trimToNull(odbc.username());
    String password = Strings.firstNonBlank(Strings.trimToNull(odbc.password()), Strings.trimToNull(odbc.secretRef()));
    return new OdbcConnectionSpec(resolvedConnectionName, databaseType, connectionString, username, password);
  }

  public static OdbcHandle openHandle(DuckdbSession session, OdbcConnectionSpec spec) {
    Objects.requireNonNull(session, "session");
    Objects.requireNonNull(spec, "spec");
    try {
      DuckdbExtensions.requireOdbc(session);
    } catch (RuntimeException ex) {
      throw ex;
    }
    String sql = connectSql(spec);
    try (Statement statement = session.createStatement();
        ResultSet resultSet = statement.executeQuery(sql)) {
      if (!resultSet.next()) {
        throw connectionFailed(spec.connectionName(), spec.databaseType(), "empty_handle_result", null);
      }
      long handle = resultSet.getLong(1);
      if (resultSet.wasNull()) {
        throw connectionFailed(spec.connectionName(), spec.databaseType(), "null_handle_result", null);
      }
      return new OdbcHandle(session, spec.connectionName(), spec.databaseType(), handle, true);
    } catch (SQLException ex) {
      throw connectionFailed(spec.connectionName(), spec.databaseType(), ex.getMessage(), ex);
    } catch (RuntimeException ex) {
      throw connectionFailed(spec.connectionName(), spec.databaseType(), ex.getMessage(), ex);
    }
  }

  public static String queryExpression(long handle, String remoteSql) {
    return "odbc_query(" + handle + ", '" + escapeSqlLiteral(Strings.requireNonBlank(remoteSql, "remoteSql")) + "')";
  }

  public static String viewName(DatasetId id) {
    Objects.requireNonNull(id, "id");
    return sanitizeIdentifier(id.group())
        + "__"
        + sanitizeIdentifier(id.namespace())
        + "__"
        + sanitizeIdentifier(id.name());
  }

  public static RuntimeException queryFailed(
      String datasetId, OdbcConnectionSpec spec, Throwable cause) {
    return queryFailed(
        datasetId,
        spec == null ? null : spec.connectionName(),
        spec == null ? null : spec.databaseType(),
        compactReason(cause),
        cause);
  }

  public static RuntimeException queryFailed(
      String datasetId,
      String connectionName,
      DefinitionConnection.Odbc.DatabaseType databaseType,
      String reason,
      Throwable cause) {
    return new IllegalStateException(
        "ODBC_QUERY_FAILED datasetId="
            + Strings.firstNonBlank(Strings.trimToNull(datasetId), "<unknown>")
            + " connectionName="
            + Strings.firstNonBlank(Strings.trimToNull(connectionName), "<unknown>")
            + " databaseType="
            + databaseTypeName(databaseType)
            + " reason="
            + Strings.firstNonBlank(Strings.trimToNull(reason), "unknown"),
        cause);
  }

  public static RuntimeException connectionFailed(
      String connectionName,
      DefinitionConnection.Odbc.DatabaseType databaseType,
      String reason,
      Throwable cause) {
    return new IllegalStateException(
        "ODBC_CONNECTION_FAILED connectionName="
            + Strings.firstNonBlank(Strings.trimToNull(connectionName), "<unknown>")
            + " databaseType="
            + databaseTypeName(databaseType)
            + " reason="
            + Strings.firstNonBlank(Strings.trimToNull(reason), "unknown"),
        cause);
  }

  public static RuntimeException ingestSourceFailed(

            + Strings.firstNonBlank(Strings.trimToNull(connectionName), "<unknown>")
            + " databaseType="
            + databaseTypeName(databaseType)
            + " reason="
            + Strings.firstNonBlank(Strings.trimToNull(reason), "unknown"),
        cause);
  }

  public static void closeQuietly(OdbcHandle handle) {
    if (handle == null) {
      return;
    }
    try {
      handle.close();
    } catch (RuntimeException ignore) {
      // keep query path deterministic and do not mask upstream failures.
    }
  }

  private static String datasetRemoteSql(DefinitionDataset.OdbcBinding binding, String datasetId) {
    String querySql = Strings.trimToNull(binding.querySql());
    if (querySql != null) {
      return querySql;
    }
    String tableName = Strings.trimToNull(binding.tableName());
    if (tableName == null) {
      throw queryFailed(
          datasetId,
          binding.connectionName(),
          binding.databaseType(),
          "missing_table_or_query",
          null);
    }
    String catalogName = Strings.trimToNull(binding.catalogName());
    String schemaName = Strings.trimToNull(binding.schemaName());
    StringBuilder sql = new StringBuilder("select * from ");
    appendQualifiedTable(sql, catalogName, schemaName, tableName);
    return sql.toString();
  }

  private static void appendQualifiedTable(
      StringBuilder sql, String catalogName, String schemaName, String tableName) {
    if (catalogName != null) {
      sql.append(quoteRemoteIdentifier(catalogName)).append('.');
    }
    if (schemaName != null) {
      sql.append(quoteRemoteIdentifier(schemaName)).append('.');
    }
    sql.append(quoteRemoteIdentifier(Strings.requireNonBlank(tableName, "tableName")));
  }

  private static String buildConnectionString(DefinitionConnection.Odbc odbc) {
    String explicit = Strings.trimToNull(odbc.connectionString());
    StringBuilder builder = new StringBuilder();
    if (explicit != null) {
      builder.append(explicit);
      if (!explicit.endsWith(";")) {
        builder.append(';');
      }
    } else {
      String dsn = Strings.trimToNull(odbc.dsn());
      if (dsn != null) {
        appendOption(builder, "DSN", dsn);
      }
      String driver = Strings.trimToNull(odbc.driverName());
      if (driver != null) {
        appendOption(builder, "Driver", "{" + driver + "}");
      }
    }
    if (builder.length() == 0) {
      throw connectionFailed(
          odbc.connectionName(),
          odbc.databaseType(),
          "missing_connection_string_or_dsn",
          null);
    }
    if (odbc.sslEnabled() != null) {
      appendOption(builder, "SSL", Boolean.TRUE.equals(odbc.sslEnabled()) ? "1" : "0");
    }
    if (odbc.connectTimeoutSeconds() != null) {
      appendOption(builder, "ConnectTimeout", String.valueOf(odbc.connectTimeoutSeconds()));
    }
    if (odbc.queryTimeoutSeconds() != null) {
      appendOption(builder, "QueryTimeout", String.valueOf(odbc.queryTimeoutSeconds()));
    }
    if (odbc.fetchSize() != null) {
      appendOption(builder, "FetchSize", String.valueOf(odbc.fetchSize()));
    }
    appendOptions(builder, odbc.properties());
    appendOptions(builder, odbc.options());
    return builder.toString();
  }

  private static void appendOptions(StringBuilder builder, Map<String, String> values) {
    if (values == null || values.isEmpty()) {
      return;
    }
    TreeMap<String, String> ordered = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
    ordered.putAll(values);
    for (Map.Entry<String, String> entry : ordered.entrySet()) {
      appendOption(builder, entry.getKey(), entry.getValue());
    }
  }

  private static void appendOption(StringBuilder builder, String key, String value) {
    String normalizedKey = Strings.trimToNull(key);
    String normalizedValue = Strings.trimToNull(value);
    if (normalizedKey == null || normalizedValue == null) {
      return;
    }
    builder.append(normalizedKey).append('=').append(normalizedValue).append(';');
  }

  private static String connectSql(OdbcConnectionSpec spec) {
    String connectionString = escapeSqlLiteral(spec.connectionString());
    String username = Strings.trimToNull(spec.username());
    String password = Strings.trimToNull(spec.password());
    if (username == null && password == null) {
      return "select odbc_connect('" + connectionString + "')";
    }
    return "select odbc_connect('"
        + connectionString
        + "', '"
        + escapeSqlLiteral(Strings.firstNonBlank(username, ""))
        + "', '"
        + escapeSqlLiteral(Strings.firstNonBlank(password, ""))
        + "')";
  }

  private static String quoteIdentifier(String identifier) {
    return "\"" + Strings.requireNonBlank(identifier, "identifier").replace("\"", "\"\"") + "\"";
  }

  private static String quoteRemoteIdentifier(String identifier) {
    return "\"" + Strings.requireNonBlank(identifier, "identifier").replace("\"", "\"\"") + "\"";
  }

  private static String sanitizeIdentifier(String value) {
    String token = Strings.requireNonBlank(value, "identifier");
    StringBuilder normalized = new StringBuilder(token.length());
    for (int index = 0; index < token.length(); index++) {
      char ch = token.charAt(index);
      if (Character.isLetterOrDigit(ch) || ch == '_') {
        normalized.append(ch);
      } else {
        normalized.append('_');
      }
    }
    String resolved = Strings.trimToNull(normalized.toString());
    return resolved == null ? "dataset" : resolved;
  }

  private static String databaseTypeName(DefinitionConnection.Odbc.DatabaseType databaseType) {
    return databaseType == null ? DefinitionConnection.Odbc.DatabaseType.OTHER.name() : databaseType.name();
  }

  private static String escapeSqlLiteral(String value) {
    return value.replace("'", "''");
  }

  public static final class OdbcQueryResult implements AutoCloseable {
    private final Statement statement;
    private final ResultSet resultSet;
    private final OdbcQueryPath path;
    private boolean closed;

    private OdbcQueryResult(Statement statement, ResultSet resultSet, OdbcQueryPath path) {
      this.statement = Objects.requireNonNull(statement, "statement");
      this.resultSet = Objects.requireNonNull(resultSet, "resultSet");
      this.path = Objects.requireNonNull(path, "path");
    }

    public ResultSet resultSet() {
      return resultSet;
    }

    public OdbcQueryPath path() {
      return path;
    }

    @Override
    public void close() {
      if (closed) {
        return;
      }
      closed = true;
      closeResultSetQuietly(resultSet);
      closeStatementQuietly(statement);
    }
  }

  public static final class OdbcHandle implements AutoCloseable {
    private final DuckdbSession session;
    private final String connectionName;
    private final DefinitionConnection.Odbc.DatabaseType databaseType;
    private final long handle;
    private final boolean closeOnClose;
    private boolean closed;

    private OdbcHandle(
        DuckdbSession session,
        String connectionName,
        DefinitionConnection.Odbc.DatabaseType databaseType,
        long handle,
        boolean closeOnClose) {
      this.session = Objects.requireNonNull(session, "session");
      this.connectionName = connectionName;
      this.databaseType = databaseType;
      this.handle = handle;
      this.closeOnClose = closeOnClose;
    }

    private static OdbcHandle noOp(
        DuckdbSession session,
        String connectionName,
        DefinitionConnection.Odbc.DatabaseType databaseType) {
      return new OdbcHandle(session, connectionName, databaseType, -1L, false);
    }

    public long handle() {
      return handle;
    }

    public String connectionName() {
      return connectionName;
    }

    public DefinitionConnection.Odbc.DatabaseType databaseType() {
      return databaseType;
    }

    @Override
    public void close() {
      if (closed) {
        return;
      }
      closed = true;
      if (!closeOnClose || handle < 0L) {
        return;
      }
      try (Statement statement = session.createStatement()) {
        statement.execute("select odbc_close(" + handle + ")");
      } catch (SQLException | RuntimeException ignore) {
        // best-effort close.
      }
    }
  }
