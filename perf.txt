  public AccessManagerImpl(
      SessionImpl session,
      Supplier<EntitlementsManagerImpl> entitlementsSupplier,
      Supplier<DiscoveryCache> discoveryCacheSupplier,
      Supplier<ServiceStatusCache> serviceStatusCacheSupplier) {
    this.session = Objects.requireNonNull(session, "session");
    this.entitlementsSupplier = Objects.requireNonNull(entitlementsSupplier, "entitlementsSupplier");
    this.discoveryCacheSupplier = Objects.requireNonNull(discoveryCacheSupplier, "discoveryCacheSupplier");
    this.serviceStatusCacheSupplier =
        Objects.requireNonNull(serviceStatusCacheSupplier, "serviceStatusCacheSupplier");
  }

  @Override
  public void close() {
    serviceTableCache.clear();
    serviceTableFailureCooldownUntilMs.clear();
    ServiceInstanceResolver resolver = dynamicServiceResolver;
    dynamicServiceResolver = null;
    if (ownsDynamicServiceResolver) {
      CloseSupport.closeQuietly(resolver);
    }
    ownsDynamicServiceResolver = false;
  }

    return datasets(null);

  @Override
  public List<DatasetAccess> datasets(String subject) {
    String resolvedSubject = resolveSubject(subject);
    List<DatasetDescriptor> datasetDescriptors = listDatasetDescriptors();
    if (datasetDescriptors.isEmpty()) {
      return List.of();
    }
    Map<String, DatasetServiceRoute> routes = resolveDatasetRoutes(resolvedSubject, datasetDescriptors);
    DataEntitlements dataEntitlements = requireDataEntitlements();
    List<DatasetAccess> rows = new ArrayList<>(datasetDescriptors.size());
    for (DatasetDescriptor descriptor : datasetDescriptors) {
      EnumSet<DatasetAction> allowed = EnumSet.noneOf(DatasetAction.class);
      DataEntitlements.Identity identity = new DataEntitlements.Identity(resolvedSubject, null);
      if (isAllowed(dataEntitlements.authorizeRead(identity, descriptor.datasetId()))) {
        allowed.add(DatasetAction.READ);
      }
      if (isAllowed(dataEntitlements.authorizeWrite(identity, descriptor.datasetId()))) {
        allowed.add(DatasetAction.WRITE);
      }
      rows.add(
          new DatasetAccess(
              descriptor.datasetId(),
              descriptor.group(),
              descriptor.namespace(),
              descriptor.name(),
              allowed.isEmpty() ? Set.of() : Set.copyOf(allowed),
              allowed.contains(DatasetAction.READ) ? routes.get(descriptor.datasetId()) : null));
    }
    return List.copyOf(rows);
  }

  @Override
  public List<String> queryableDatasets() {
    return queryableDatasets(null);
  }

  @Override
  public List<String> queryableDatasets(String subject) {
    List<String> allowed = new ArrayList<>();
    for (DatasetAccess access : datasets(subject)) {
      if (access != null && access.allowedActions().contains(DatasetAction.READ)) {
        allowed.add(access.datasetId());
      }
    }
    return List.copyOf(allowed);
  }

  @Override
  public boolean canQueryDataset(String datasetId) {
    return canQueryDataset(null, datasetId);
  }

  @Override
  public boolean canQueryDataset(String subject, String datasetId) {
    String canonicalDataset = DataEntitlements.requireDatasetResourceId(datasetId);
    DataEntitlements dataEntitlements = requireDataEntitlements();
    DataEntitlements.Decision decision =
        dataEntitlements.authorizeRead(
            new DataEntitlements.Identity(resolveSubject(subject), null), canonicalDataset);
    return isAllowed(decision);
  }

  @Override
  public Optional<DatasetServiceRoute> datasetQueryRoute(String datasetId) {
    return datasetQueryRoute(null, datasetId);
  }

  @Override
  public Optional<DatasetServiceRoute> datasetQueryRoute(String subject, String datasetId) {
    String canonicalDataset = DataEntitlements.requireDatasetResourceId(datasetId);
    String resolvedSubject = resolveSubject(subject);
    if (!canQueryDataset(resolvedSubject, canonicalDataset)) {
      return Optional.empty();
    }
    List<DatasetDescriptor> descriptors = listDatasetDescriptors();
    for (DatasetDescriptor descriptor : descriptors) {
      if (canonicalDataset.equals(descriptor.datasetId())) {
        return Optional.ofNullable(
            resolveDatasetRoutes(resolvedSubject, List.of(descriptor)).get(canonicalDataset));
      }
    }
    return Optional.empty();
  }

  @Override
  public List<ServiceAccess> services() {
    return services(null);
  }

  @Override
  public List<ServiceAccess> services(String subject) {
    String resolvedSubject = resolveSubject(subject);
    List<ServiceAccess> rows = new ArrayList<>();
    for (ServiceDescriptor descriptor : listServiceDescriptors()) {
      Map<ServiceEntitlements.Transport, Set<ServiceAction>> byTransport =
          new EnumMap<>(ServiceEntitlements.Transport.class);
      for (ServiceEntitlements.Transport transport : transportsForService(descriptor.serviceType())) {
        EnumSet<ServiceAction> allowed = EnumSet.noneOf(ServiceAction.class);
        for (ServiceAction action : ServiceAction.values()) {
          if (isServiceActionAllowed(resolvedSubject, descriptor.serviceId(), transport, action)) {
            allowed.add(action);
          }
        }
        if (!allowed.isEmpty()) {
          byTransport.put(transport, Set.copyOf(allowed));
        }
      }
      rows.add(
          new ServiceAccess(
              descriptor.serviceId(),
              descriptor.group(),
              descriptor.name(),
              descriptor.serviceType() == null ? null : descriptor.serviceType().name(),
              descriptor.online(),
              byTransport.isEmpty() ? Map.of() : Map.copyOf(byTransport)));
    }
    return List.copyOf(rows);
  }

  @Override
  public List<String> callableServices() {
    return callableServices(null);
  }

  @Override
  public List<String> callableServices(String subject) {
    List<String> allowed = new ArrayList<>();
    for (ServiceAccess access : services(subject)) {
      if (access == null || access.allowedActionsByTransport().isEmpty()) {
        continue;
      }
      boolean callable = false;
      for (Set<ServiceAction> actions : access.allowedActionsByTransport().values()) {
        if (actions != null && actions.contains(ServiceAction.INVOKE)) {
          callable = true;
          break;
        }
      }
      if (callable) {
        allowed.add(access.serviceId());
      }
    }
    return List.copyOf(allowed);
  }

  @Override
  public boolean canCallService(String serviceId) {
    return canCallService(null, serviceId);
  }

  @Override
  public boolean canCallService(String subject, String serviceId) {
    String canonicalService = ServiceEntitlements.requireServiceResourceId(serviceId);
    String resolvedSubject = resolveSubject(subject);
    Set<ServiceEntitlements.Transport> transports = transportsForService(resolveServiceType(canonicalService));
    if (transports.isEmpty()) {
      transports = Set.of(ServiceEntitlements.Transport.values());
    }
    for (ServiceEntitlements.Transport transport : transports) {
      if (isServiceActionAllowed(resolvedSubject, canonicalService, transport, ServiceAction.INVOKE)) {
        return true;
      }
    }
    return false;
  }

  @Override
  public boolean canCallService(
      String subject,
      String serviceId,
      ServiceEntitlements.Transport transport,
      ServiceAction action) {
    if (transport == null || action == null) {
      throw new IllegalArgumentException("transport and action are required");
    }
    String canonicalService = ServiceEntitlements.requireServiceResourceId(serviceId);
    return isServiceActionAllowed(resolveSubject(subject), canonicalService, transport, action);
  }

  @Override
  public boolean isServiceOnline(String serviceId) {
    return onlineServiceKeys().contains(serviceKeyFromResourceId(serviceId));
  }

  private Map<String, DatasetServiceRoute> resolveDatasetRoutes(
      String subject, List<DatasetDescriptor> datasets) {
    if (datasets == null || datasets.isEmpty()) {
      return Map.of();
    }
    Map<String, DatasetServiceRoute> selections = new LinkedHashMap<>();
    Map<String, List<DatasetDescriptor>> byNamespaceAndName = new HashMap<>();
    Map<String, List<DatasetDescriptor>> byGroupNamespaceAndName = new HashMap<>();
    for (DatasetDescriptor dataset : datasets) {
      if (dataset == null || dataset.datasetId() == null) {
        continue;
      }
      if (dataset.directQuery()) {
        selections.put(
            dataset.datasetId(),
            new DatasetServiceRoute(
                dataset.datasetId(),
                DIRECT_QUERY_SERVICE_ID,
                DIRECT_QUERY_SERVICE_DISPLAY,
                DIRECT_QUERY_SERVICE_DISPLAY,
                1,
                true));
        continue;
      }
      String key = datasetKey(dataset.namespace(), dataset.name());
      if (key == null) {
        continue;
      }
      byNamespaceAndName.computeIfAbsent(key, ignored -> new ArrayList<>()).add(dataset);
      String grouped = groupedDatasetKey(dataset.group(), dataset.namespace(), dataset.name());
      if (grouped != null) {
        byGroupNamespaceAndName.computeIfAbsent(grouped, ignored -> new ArrayList<>()).add(dataset);
      }
    }
    if (byNamespaceAndName.isEmpty()) {
      return selections;
    }

    Map<String, Map<String, DatasetServiceCandidate>> candidatesByDatasetId = new HashMap<>();
    for (ServiceReference serviceRef : listAvailableArrowServices(subject)) {
      Set<String> supportedKeys = fetchSupportedDatasetKeys(subject, serviceRef);
      if (supportedKeys.isEmpty()) {
        continue;
      }
      for (String key : supportedKeys) {
        List<DatasetDescriptor> matches =
            byGroupNamespaceAndName.get(groupedDatasetKey(serviceRef.group(), key));
        if (matches == null || matches.isEmpty()) {
          matches = byNamespaceAndName.get(key);
        }
        if (matches == null || matches.isEmpty()) {
          continue;
        }
        for (DatasetDescriptor match : matches) {
          Map<String, DatasetServiceCandidate> perDataset =
              candidatesByDatasetId.computeIfAbsent(match.datasetId(), ignored -> new LinkedHashMap<>());
          perDataset.putIfAbsent(
              serviceRef.serviceId(),
              new DatasetServiceCandidate(
                  serviceRef.serviceId(),
                  serviceRef.serviceDisplay(),
                  serviceRef.serviceType(),
                  Strings.trimToNull(match.group()) != null
                      && Strings.trimToNull(match.group()).equalsIgnoreCase(serviceRef.group()),
                  serviceRef.online()));
        }
      }
    }
    for (Map.Entry<String, Map<String, DatasetServiceCandidate>> entry :
        candidatesByDatasetId.entrySet()) {
      String datasetId = entry.getKey();
      List<DatasetServiceCandidate> candidates = List.copyOf(entry.getValue().values());
      if (candidates.isEmpty()) {
        continue;
      }
      DatasetServiceCandidate selected =
          candidates.stream().min(this::compareServiceCandidates).orElse(null);
      if (selected == null) {
        continue;
      }
      int serviceCount = candidates.size();
      String display =
          serviceCount > 1
              ? selected.serviceDisplay() + " (+" + (serviceCount - 1) + ")"
              : selected.serviceDisplay();
      selections.put(
          datasetId,
          new DatasetServiceRoute(
              datasetId,
              selected.serviceId(),
              display,
              selected.serviceType() == null ? null : selected.serviceType().name(),
              serviceCount,
              selected.online()));
    }
    return selections;
  }

  private List<ServiceReference> listAvailableArrowServices(String subject) {
    Set<String> onlineKeys = onlineServiceKeys();
    if (onlineKeys.isEmpty()) {
      return List.of();
    }
    List<ServiceReference> resolved = new ArrayList<>();
    for (ServiceDescriptor descriptor : listServiceDescriptors()) {
      if (descriptor == null || descriptor.serviceType() == null) {
        continue;
      }
      if (descriptor.serviceType() != DefinitionService.ServiceType.ARROW_FLIGHT_SQL
          && descriptor.serviceType() != DefinitionService.ServiceType.ARROW_FLIGHT) {
        continue;
      }
      if (!onlineKeys.contains(serviceKey(descriptor.group(), descriptor.name()))) {
        continue;
      }
      ServiceEntitlements.Transport requiredTransport =
          descriptor.serviceType() == DefinitionService.ServiceType.ARROW_FLIGHT_SQL
              ? ServiceEntitlements.Transport.ARROW_FLIGHT_SQL
              : ServiceEntitlements.Transport.ARROW_FLIGHT;
      if (!isServiceActionAllowed(subject, descriptor.serviceId(), requiredTransport, ServiceAction.INVOKE)) {
        continue;
      }
      resolved.add(
          new ServiceReference(
              descriptor.group(),
              descriptor.name(),
              descriptor.serviceId(),
              descriptor.serviceType(),
              descriptor.group() + "/" + descriptor.name(),
              descriptor.online()));
    }
    resolved.sort(
        Comparator.comparing(
                (ServiceReference value) -> Strings.firstNonBlank(value.group(), ""),
                String.CASE_INSENSITIVE_ORDER)
            .thenComparing(
                value -> Strings.firstNonBlank(value.serviceName(), ""),
                String.CASE_INSENSITIVE_ORDER));
    return List.copyOf(resolved);
  }

  private Set<String> fetchSupportedDatasetKeys(String subject, ServiceReference serviceRef) {
    if (serviceRef == null
        || serviceRef.serviceType() != DefinitionService.ServiceType.ARROW_FLIGHT_SQL) {
      return Set.of();
    }
    String serviceKey = serviceKey(serviceRef.group(), serviceRef.serviceName());
    if (serviceKey == null) {
      return Set.of();
    }
    String cacheKey = tableCacheKey(serviceKey, subject);
    long nowMs = System.currentTimeMillis();
    CachedServiceTables cached = cachedServiceTables(cacheKey, nowMs);
    if (cached != null) {
      return cached.datasetKeys();
    }
    if (isServiceTableLookupCoolingDown(cacheKey, nowMs)) {
      return Set.of();
    }
    try (RemoteTabularManager tabular = newRemoteTabularManager(serviceRef)) {
      List<TabularTableMetadata> tables = runAsSubject(resolveSubject(subject), tabular::listTables);
      Set<String> supported = new HashSet<>();
      if (tables != null) {
        for (TabularTableMetadata table : tables) {
          if (table == null) {
            continue;
          }
          String key = datasetKey(table.schemaName(), table.tableName());
          if (key != null) {
            supported.add(key);
          }
        }
      }
      cacheServiceTables(cacheKey, supported, nowMs);
      clearServiceTableFailureCooldown(cacheKey, nowMs);
      return supported;
    } catch (Exception ex) {
      markServiceTableFailure(cacheKey, nowMs);
      CachedServiceTables fallback = cachedServiceTables(cacheKey, nowMs);
      LOGGER.debug(
          "[Access][ServiceTables][LookupFailed] service={} subject={} message={} fallbackDatasetCount={}",
          serviceRef.serviceId(),
          resolveSubject(subject),
          Strings.firstNonBlank(ex.getMessage(), ex.getClass().getSimpleName()),
          fallback == null ? 0 : fallback.datasetKeys().size());
      return fallback == null ? Set.of() : fallback.datasetKeys();
    }
  }

  private RemoteTabularManager newRemoteTabularManager(ServiceReference serviceRef) {
    Objects.requireNonNull(serviceRef, "serviceRef");
    DiscoveryCache cache = requireDiscoveryCache();
    SessionContext context = session.context();
    String sessionGroup = Strings.firstNonBlank(context == null ? null : context.getGroup(), "platform");
    String sessionApp = Strings.firstNonBlank(context == null ? null : context.getAppName(), "app");
    String sessionInstance =
        Strings.firstNonBlank(context == null ? null : context.getInstance(), "instance");
    String sessionEnv = Strings.trimToNull(context == null ? null : context.getEnv());
    String sessionRegion = Strings.trimToNull(context == null ? null : context.getRegion());
    String sessionSite = Strings.trimToNull(context == null ? null : context.getSite());
    DataOptions.ServiceOptions serviceOptions = serviceOptionsFor(serviceRef, sessionGroup);
    ServiceInstanceResolver resolver = resolveDynamicServiceResolver();
    return new RemoteTabularManager(
        session,
        cache,
        sessionGroup,
        sessionApp,
        sessionInstance,
        sessionEnv,
        sessionRegion,
        sessionSite,
        serviceOptions,
        session.kerberosPrincipal(),
        session.kerberosKeytab(),
        resolver);
  }

  private DataOptions.ServiceOptions serviceOptionsFor(
      ServiceReference serviceRef, String defaultGroup) {
    String group = Strings.firstNonBlank(serviceRef.group(), defaultGroup);
    ServiceRunOptions.Builder optionsBuilder = ServiceRunOptions.builder();
    if (group != null) {
      optionsBuilder.withGroup(group);
    }
    Path trustCert = session.trustCert();
    if (trustCert != null) {
      optionsBuilder.withTlsTrustCert(trustCert).withTlsClientTrustCert(trustCert);
    }
    return DataOptions.ServiceOptions.builder()
        .withEnabled(true)
        .withQueryService(serviceRef.serviceName())
        .withServiceOptions(optionsBuilder.build())
        .build();
  }

  private ServiceInstanceResolver resolveDynamicServiceResolver() {
    ServiceInstanceResolver existing = dynamicServiceResolver;
    if (existing != null) {
      return existing;
    }
    synchronized (this) {
      existing = dynamicServiceResolver;
      if (existing != null) {
        return existing;
      }
      ServiceInstanceResolver fromSession = session.serviceInstanceResolverIfInitialized();
      if (fromSession != null) {
        dynamicServiceResolver = fromSession;
        ownsDynamicServiceResolver = false;
        return fromSession;
      }
      ControlConnection control = session.controlPlaneConnection();
      if (control == null) {
        throw new IllegalStateException("Session control plane is not initialized");
      }
      ServiceInstanceResolver created = new ServiceInstanceResolver(control, session.statusScheduler());
      dynamicServiceResolver = created;
      ownsDynamicServiceResolver = true;
      return created;
    }
  }

  private boolean isServiceActionAllowed(
      String subject,
      String serviceResourceId,
      ServiceEntitlements.Transport transport,
      ServiceAction action) {
    try {
      ServiceEntitlements entitlements = requireServiceEntitlements();
      ServiceEntitlements.Identity identity = new ServiceEntitlements.Identity(subject, null);
      ServiceEntitlements.Request request =
          new ServiceEntitlements.Request(
              transport,
              toCoreAction(action),
              ServiceEntitlements.requireServiceResourceId(serviceResourceId),
              null,
              null,
              null,
              Map.of(),
              Map.of());
      ServiceEntitlements.Decision decision = entitlements.authorize(identity, request);
      return decision != null && decision.allowed();
    } catch (Exception ex) {
      return false;
    }
  }

  private static ServiceEntitlements.Action toCoreAction(ServiceAction action) {
    return switch (action) {
      case START -> ServiceEntitlements.Action.START;
      case INVOKE -> ServiceEntitlements.Action.INVOKE;
      case READ -> ServiceEntitlements.Action.READ;
      case WRITE -> ServiceEntitlements.Action.WRITE;
    };
  }

  private DefinitionService.ServiceType resolveServiceType(String serviceResourceId) {
    String canonical = ServiceEntitlements.requireServiceResourceId(serviceResourceId);
    CanonicalIds.ParsedId parsed = CanonicalIds.parse(canonical);
    if (!parsed.isService()) {
      return null;
    }
    DiscoveryCache cache = discoveryCacheSupplier.get();
    if (cache == null) {
      return null;
    }
    DefinitionService definition =
        cache.get(DefinitionService.class, parsed.group(), parsed.name()).orElse(null);
    return definition == null ? null : definition.serviceType();
  }

  private Set<ServiceEntitlements.Transport> transportsForService(
      DefinitionService.ServiceType serviceType) {
    if (serviceType == null) {
      return Set.of();
    }
    return switch (serviceType) {
      case GRPC -> Set.of(ServiceEntitlements.Transport.GRPC);
      case AMPS -> Set.of(ServiceEntitlements.Transport.AMPS);
      case ARROW_FLIGHT -> Set.of(ServiceEntitlements.Transport.ARROW_FLIGHT);
      case ARROW_FLIGHT_INGEST -> Set.of(ServiceEntitlements.Transport.ARROW_FLIGHT_INGEST);
      case ARROW_FLIGHT_SQL -> Set.of(ServiceEntitlements.Transport.ARROW_FLIGHT_SQL);
      case UNKNOWN -> Set.of();
    };
  }

  private List<DatasetDescriptor> listDatasetDescriptors() {
    DiscoveryCache cache = discoveryCacheSupplier.get();
    if (cache == null) {
      return List.of();
    }
    List<DatasetDescriptor> rows = new ArrayList<>();
    for (DefinitionDataset definition : cache.list(DefinitionDataset.class)) {
      if (definition == null || definition.envelope() == null) {
        continue;
      }
      if (Boolean.FALSE.equals(definition.envelope().enabled())) {
        continue;
      }
      String group = Strings.trimToNull(definition.envelope().group());
      String namespace =
          Strings.firstNonBlank(
              Strings.trimToNull(definition.namespace()), namespaceFromEnvelopeName(definition.envelope()));
      String name = datasetName(definition);
      if (group == null || namespace == null || name == null) {
        continue;
      }
      String datasetId;
      try {
        datasetId = DataEntitlements.canonicalDatasetResourceId(group, namespace, name);
      } catch (Exception ex) {
        continue;
      }
      rows.add(
          new DatasetDescriptor(
              datasetId, group, namespace, name, prefersDirectQuery(definition)));
    }
    rows.sort(
        Comparator.comparing(DatasetDescriptor::group, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(DatasetDescriptor::namespace, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(DatasetDescriptor::name, String.CASE_INSENSITIVE_ORDER));
    return List.copyOf(rows);
  }

  private List<ServiceDescriptor> listServiceDescriptors() {
    DiscoveryCache cache = discoveryCacheSupplier.get();
    if (cache == null) {
      return List.of();
    }
    Set<String> online = onlineServiceKeys();
    List<ServiceDescriptor> rows = new ArrayList<>();
    for (DefinitionService service : cache.list(DefinitionService.class)) {
      if (service == null || service.envelope() == null) {
        continue;
      }
      if (Boolean.FALSE.equals(service.envelope().enabled())) {
        continue;
      }
      String group = Strings.trimToNull(service.envelope().group());
      String name = Strings.trimToNull(service.envelope().name());
      if (group == null || name == null) {
        continue;
      }
      String canonical = Strings.firstNonBlank(service.resourceId(), "service:" + group + "/" + name);
      String serviceId;
      try {
        serviceId = ServiceEntitlements.requireServiceResourceId(canonical);
      } catch (Exception ex) {
        continue;
      }
      rows.add(
          new ServiceDescriptor(
              serviceId,
              group,
              name,
              service.serviceType(),
              online.contains(serviceKey(group, name))));
    }
    rows.sort(
        Comparator.comparing(ServiceDescriptor::group, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(ServiceDescriptor::name, String.CASE_INSENSITIVE_ORDER));
    return List.copyOf(rows);
  }

  private Set<String> onlineServiceKeys() {
    ServiceStatusCache cache = serviceStatusCacheSupplier.get();
    if (cache == null) {
      return Set.of();
    }
    Set<String> keys = new HashSet<>();
    for (ServiceStatusRow row : cache.snapshot()) {
      if (row == null) {
        continue;
      }
      if (!"ONLINE".equalsIgnoreCase(Strings.trimToNull(row.state()))) {
        continue;
      }
      String key = serviceKey(row.group(), row.name());
      if (key != null) {
        keys.add(key);
      }
    }
    return keys;
  }

  private String resolveSubject(String requestedSubject) {
    String explicit = Strings.trimToNull(requestedSubject);
    if (explicit != null) {
      return explicit;
    }
    String current =
        UserContext.get().map(UserPrincipal::username).map(Strings::trimToNull).orElse(null);
    if (current != null) {
      return current;
    }
    SessionContext context = session.context();
    String group = Strings.trimToNull(context == null ? null : context.getGroup());
    String appName = Strings.trimToNull(context == null ? null : context.getAppName());
    if (group != null && appName != null) {
      try {
        return CanonicalIds.appPrincipal(group, appName);
      } catch (Exception ignored) {
      }
    }
    return "app:unknown/unknown";
  }

  private DiscoveryCache requireDiscoveryCache() {

    if (cache == null) {
      throw new IllegalStateException("DiscoveryCache is not initialized");
    }

  }

  private int compareServiceCandidates(DatasetServiceCandidate left, DatasetServiceCandidate right) {
    int leftGroup = left.sameGroup() ? 0 : 1;
    int rightGroup = right.sameGroup() ? 0 : 1;
    if (leftGroup != rightGroup) {
      return Integer.compare(leftGroup, rightGroup);
    }
    int leftType = serviceTypeRank(left.serviceType());
    int rightType = serviceTypeRank(right.serviceType());
    if (leftType != rightType) {
      return Integer.compare(leftType, rightType);
    }
    return Strings.firstNonBlank(left.serviceDisplay(), "")
        .compareToIgnoreCase(Strings.firstNonBlank(right.serviceDisplay(), ""));
  }

  private static int serviceTypeRank(DefinitionService.ServiceType serviceType) {
    if (serviceType == null) {
      return 99;
    }
    return switch (serviceType) {
      case ARROW_FLIGHT_SQL -> 0;
      case ARROW_FLIGHT -> 1;
      default -> 99;
    };
  }

  private CachedServiceTables cachedServiceTables(String cacheKey, long nowMs) {
    if (cacheKey == null) {
      return null;
    }
    CachedServiceTables cached = serviceTableCache.get(cacheKey);
    if (cached == null) {
      return null;
    }
    if (nowMs - cached.loadedAtMs() > SERVICE_TABLE_CACHE_TTL_MILLIS) {
      serviceTableCache.remove(cacheKey, cached);
      return null;
    }
    return cached;
  }

  private boolean isServiceTableLookupCoolingDown(String cacheKey, long nowMs) {
    if (cacheKey == null) {
      return false;
    }
    Long cooldownUntil = serviceTableFailureCooldownUntilMs.get(cacheKey);
    if (cooldownUntil == null) {
      return false;
    }
    if (cooldownUntil <= nowMs) {
      serviceTableFailureCooldownUntilMs.remove(cacheKey, cooldownUntil);
      return false;
    }
    return true;
  }

  private void cacheServiceTables(String cacheKey, Set<String> datasetKeys, long nowMs) {
    if (cacheKey == null || datasetKeys == null) {
      return;
    }
    Set<String> normalized =
        new LinkedHashSet<>(datasetKeys.stream().sorted(String.CASE_INSENSITIVE_ORDER).toList());
    serviceTableCache.put(cacheKey, new CachedServiceTables(normalized, nowMs));
  }

  private void markServiceTableFailure(String cacheKey, long nowMs) {
    if (cacheKey == null) {
      return;
    }
    serviceTableFailureCooldownUntilMs.put(
        cacheKey, nowMs + SERVICE_TABLE_FAILURE_COOLDOWN_MILLIS);
  }

  private void clearServiceTableFailureCooldown(String cacheKey, long nowMs) {
    if (cacheKey == null) {
      return;
    }
    Long cooldownUntil = serviceTableFailureCooldownUntilMs.get(cacheKey);
    if (cooldownUntil != null && cooldownUntil <= nowMs) {
      serviceTableFailureCooldownUntilMs.remove(cacheKey, cooldownUntil);
      return;
    }
    serviceTableFailureCooldownUntilMs.remove(cacheKey);
  }

  private static String tableCacheKey(String serviceKey, String subject) {
    String normalizedService = Strings.trimToNull(serviceKey);
    if (normalizedService == null) {
      return null;
    }
    String normalizedSubject = Strings.trimToNull(subject);
    if (normalizedSubject == null) {
      return normalizedService;
    }
    return normalizedService + "|" + normalizedSubject.toLowerCase(Locale.ROOT);
  }

  private static <T> T runAsSubject(String subject, Supplier<T> callback) {
    Objects.requireNonNull(callback, "callback");
    String normalizedSubject = Strings.trimToNull(subject);
    if (normalizedSubject == null) {
      return callback.get();
    }
    UserPrincipal previous = UserContext.get().orElse(null);
    UserContext.set(new UserPrincipal(normalizedSubject, Map.of()));
    try {
      return callback.get();
    } finally {
      if (previous == null) {
        UserContext.clear();
      } else {
        UserContext.set(previous);
      }
    }
  }

  private static boolean isAllowed(DataEntitlements.Decision decision) {
    return decision != null && decision.allowed();
  }

  private static String datasetName(DefinitionDataset definition) {
    if (definition == null) {
      return null;
    }
    DefinitionEnvelope envelope = definition.envelope();
    String name = Strings.trimToNull(envelope == null ? null : envelope.name());
    if (name == null) {
      return null;
    }
    int slash = name.indexOf('/');
    if (slash < 0 || slash == name.length() - 1) {
      return name;
    }
    return name.substring(slash + 1);
  }

  private static String namespaceFromEnvelopeName(DefinitionEnvelope envelope) {
    String name = Strings.trimToNull(envelope == null ? null : envelope.name());
    if (name == null) {
      return null;
    }
    int slash = name.indexOf('/');
    if (slash <= 0) {
      return null;
    }
    return name.substring(0, slash);
  }

  private static String datasetKey(String namespace, String name) {
    String normalizedNamespace = Strings.trimToNull(namespace);
    String normalizedName = Strings.trimToNull(name);
    if (normalizedNamespace == null || normalizedName == null) {
      return null;
    }
    return normalizedNamespace.toLowerCase(Locale.ROOT) + "/" + normalizedName.toLowerCase(Locale.ROOT);
  }

  private static String groupedDatasetKey(String group, String namespace, String name) {
    String normalizedGroup = Strings.trimToNull(group);
    String key = datasetKey(namespace, name);
    if (normalizedGroup == null || key == null) {
      return null;
    }
    return normalizedGroup.toLowerCase(Locale.ROOT) + "|" + key;
  }

  private static String groupedDatasetKey(String group, String datasetKey) {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedKey = Strings.trimToNull(datasetKey);
    if (normalizedGroup == null || normalizedKey == null) {
      return null;
    }
    return normalizedGroup.toLowerCase(Locale.ROOT) + "|" + normalizedKey;
  }

  private static String serviceKey(String group, String name) {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedName = Strings.trimToNull(name);
    if (normalizedGroup == null || normalizedName == null) {
      return null;
    }
    return normalizedGroup.toLowerCase(Locale.ROOT) + "/" + normalizedName.toLowerCase(Locale.ROOT);
  }

  private static String serviceKeyFromResourceId(String serviceResourceId) {
    String normalized = Strings.trimToNull(serviceResourceId);
    if (normalized == null || isDirectServiceId(normalized)) {
      return null;
    }
    String token =
        normalized.regionMatches(true, 0, "service:", 0, "service:".length())
            ? normalized.substring("service:".length())
            : normalized;
    int slash = token.indexOf('/');
    if (slash <= 0 || slash >= token.length() - 1) {
      return null;
    }
    String group = Strings.trimToNull(token.substring(0, slash));
    String name = Strings.trimToNull(token.substring(slash + 1));
    return serviceKey(group, name);
  }

  private static boolean prefersDirectQuery(DefinitionDataset definition) {
    return definition != null && definition.binding() != null && definition.binding().parquet() != null;
  }

  private static boolean isDirectServiceId(String serviceId) {
    String normalized = Strings.trimToNull(serviceId);
    return normalized != null
        && (DIRECT_QUERY_SERVICE_ID.equalsIgnoreCase(normalized)
            || DIRECT_QUERY_SERVICE_DISPLAY.equalsIgnoreCase(normalized));
  }
