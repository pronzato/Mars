  static ServiceManagerImpl.RunningGrpcServer startGrpcServer(
      ServiceManagerImpl manager,
      DefinitionService def,
      BindableService svc,
      ServiceRunOptions options)
      throws Exception {
    Objects.requireNonNull(def, "def");
    Objects.requireNonNull(svc, "svc");
    String serviceName = ServiceManagerImpl.resolveServiceName(def);
    String serviceId = ServiceManagerImpl.serviceId(def);
    DefinitionService.Grpc grpc = manager.requireGrpc(def, serviceName);
    manager.requireGrpcRoutingMode(def, "server");
    options = options == null ? ServiceRunOptions.defaults() : options;

    int bindPort = manager.resolveBindPort(grpc, options);
    String bindHost = manager.resolveBindHost(grpc, options);
    String advertiseHost = manager.resolveAdvertiseHost(grpc, options);
    int advertisePort = manager.resolveAdvertisePort(grpc, options, bindPort);
    boolean tlsEnabled = ServiceManagerImpl.tlsEnabled(grpc);
    boolean kerberosEnabled = ServiceManagerImpl.kerberosEnabled(grpc);
    String instanceId = resolveInstanceId(options, advertiseHost, advertisePort);

    GrpcServiceRuntime.Config cfg = new GrpcServiceRuntime.Config();
    cfg.bindHost = bindHost;
    cfg.bindPort = bindPort;
    cfg.advertiseHost = advertiseHost;
    cfg.advertisePort = advertisePort;
    cfg.portFailback = Boolean.TRUE.equals(grpc.portFailback());
    cfg.tlsEnabled = tlsEnabled;
    cfg.certChainPem =
        ServiceManagerImpl.toFile(
            ServiceManagerImpl.firstNonNull(
                options.tlsCertChain(), manager.session().serviceTlsCertChain()));
    cfg.privateKeyPem =
        ServiceManagerImpl.toFile(
            ServiceManagerImpl.firstNonNull(
                options.tlsPrivateKey(), manager.session().serviceTlsPrivateKey()));
    cfg.kerberosEnabled = kerberosEnabled;
    cfg.kerberosSpn = grpc.kerberos() == null ? null : grpc.kerberos().spn();
    cfg.kerberosPrincipal = manager.sessionKerberosPrincipal();
    cfg.kerberosKeytab = manager.sessionKerberosKeytab();
    cfg.kerberosRealm = KerberosSpn.extractRealm(cfg.kerberosPrincipal);

    authorizeStart(
        manager,
        serviceId,
        resolveServerStartSubject(manager, grpc),
        ServiceEntitlements.Transport.GRPC);
    GrpcServiceRuntime runtime =
        GrpcServiceRuntime.start(cfg, svc, manager.buildServerInterceptors(def, serviceId, grpc, options));
    ServiceStatusPublisher publisher =
        createGrpcStatusPublisher(
            manager,
            serviceName,
            instanceId,
            advertiseHost,
            advertisePort,
            bindHost,
            bindPort,
            grpc,
            tlsEnabled,
            options);
    try {
      publisher.start();
    } catch (Exception ex) {
      runtime.close();
      throw ex;
    }
    manager
        .logger()
        .info(
            "[ServiceManager][gRPC][server-start] serviceId={} bind={}:{} advertise={}:{} instanceId={} tlsEnabled={} kerberosEnabled={} identityMode={}",
            serviceId,
            bindHost,
            runtime.port(),
            advertiseHost,
            advertisePort,
            instanceId,
            tlsEnabled,
            kerberosEnabled,
            ServiceManagerImpl.resolveGrpcIdentityMode(grpc));
    return new ServiceManagerImpl.RunningGrpcServer(serviceId, instanceId, runtime, publisher);
  }

  static ManagedChannel createGrpcChannel(
      ServiceManagerImpl manager, DefinitionService def, ServiceRunOptions options) throws Exception {
    Objects.requireNonNull(def, "def");
    String serviceName = ServiceManagerImpl.resolveServiceName(def);
    String serviceId = ServiceManagerImpl.serviceId(def);
    String group = manager.resolveGroup(def);
    DefinitionService.Grpc grpc = manager.requireGrpc(def, serviceName);
    ServiceRoutingMode routingMode = manager.requireGrpcRoutingMode(def, "client");
    options = options == null ? ServiceRunOptions.defaults() : options;

    ServiceInstanceResolver instanceResolver = manager.session().serviceInstanceResolverIfInitialized();
    GrpcClientRuntime.RoutingConfig cfg = new GrpcClientRuntime.RoutingConfig();
    cfg.serviceId = serviceId;
    if (instanceResolver != null) {
      cfg.instanceResolver = instanceResolver;
      cfg.serviceGroup = ServiceManagerImpl.requireValue(group, "group");
      cfg.serviceName = ServiceManagerImpl.requireValue(serviceName, "serviceName");
      cfg.closeInstanceResolverOnCleanup = false;
    } else {
      cfg.statusCache = createRoutingStatusCache(manager, group, serviceName, serviceId);
    }
    cfg.tlsEnabled = ServiceManagerImpl.tlsEnabled(grpc);
    cfg.trustCertPem =
        ServiceManagerImpl.toFile(
            ServiceManagerImpl.firstNonNull(
                options.tlsTrustCert(), options.clientTrustCert(), manager.session().trustCert()));
    cfg.kerberosEnabled = ServiceManagerImpl.kerberosEnabled(grpc);
    cfg.kerberosSpn = grpc.kerberos() == null ? null : grpc.kerberos().spn();
    cfg.kerberosPrincipal = manager.sessionKerberosPrincipal();
    cfg.kerberosKeytab = manager.sessionKerberosKeytab();
    cfg.kerberosRealm = KerberosSpn.extractRealm(cfg.kerberosPrincipal);
    cfg.kerberosRequireTls = true;
    cfg.logRoutingSelections = manager.session().grpcRoutingLogSelections();
    cfg.routingMode = routingMode;
    cfg.defaultEffectiveSubject = manager.sessionKerberosPrincipal();
    cfg.callerAppSubject = manager.sessionAppSubject();
    cfg.routingKey = options.routingKey();

    ManagedChannel channel =
        GrpcClientRuntime.createRoutingChannel(
            cfg, manager.buildClientInterceptors(def, serviceId, options));
    manager
        .logger()
        .info(
            "[ServiceManager][gRPC][client-routing] serviceId={} group={} service={} mode={} tlsEnabled={} kerberosEnabled={} routingKeyConfigured={}",
            serviceId,
            group,
            serviceName,
            routingMode,
            cfg.tlsEnabled,
            cfg.kerberosEnabled,
            Strings.isNotBlank(cfg.routingKey));
    return channel;
  }

  static ServiceHbCache createRoutingStatusCache(
      ServiceManagerImpl manager, String group, String serviceName, String serviceId) throws Exception {
    ControlConnection control = manager.session().controlPlaneConnection();
    if (control == null) {
      throw new IllegalStateException(
          "Session control-plane client is not initialized for service: " + serviceId);
    }
    ServiceHbCache cache =
        new ServiceHbCache(
            control,
            ServiceManagerImpl.requireValue(group, "group"),
            ServiceManagerImpl.requireValue(serviceName, "serviceName"),
            new ServiceHbCache.Config(),
            manager.session().statusScheduler());
    boolean success = false;
    try {
      cache.start();
      success = true;
      return cache;
    } finally {
      if (!success) {
        cache.close();
      }
    }
  }

  static FlightSqlServiceProducer toFlightSqlProducer(
      ServiceManagerImpl manager,
      Object implementation,
      String serviceResourceId,
      String advertiseHost,
      java.util.function.IntSupplier portSupplier,
      boolean tlsEnabled,
      ServiceRunOptions options) {
    List<String> datasetAllowlist =
        normalizeDatasetAllowlist(options == null ? null : options.datasetIds());
    Set<DatasetId> allowedDatasets = parseDatasetAllowlist(datasetAllowlist);
    if (implementation == null) {
      TabularManager tabularManager =
          resolveFlightSqlTabularManager(manager, serviceResourceId, datasetAllowlist);
      return new FlightSqlServiceProducer(
          tabularManager,
          manager.entitlements(),
          serviceResourceId,
          manager.session().discoveryCache(),
          resolveRuntimeEnvironment(manager),
          advertiseHost,
          portSupplier,
          tlsEnabled,
          manager.checkpointStore(),
          allowedDatasets);
    }
    if (implementation instanceof FlightSqlServiceProducer producer) {
      return producer;
    }
    throw new IllegalArgumentException(
        "Service implementation must be "
            + FlightSqlServiceProducer.class.getName()
            + " for Arrow Flight SQL services: "
            + implementation.getClass().getName());
  }

  static TabularManager resolveFlightSqlTabularManager(
      ServiceManagerImpl manager, String serviceResourceId, List<String> datasetAllowlist) {
    TabularManager directTabular = manager.session().data().direct().tabular();
    if (datasetAllowlist == null || datasetAllowlist.isEmpty()) {
      return directTabular;
    }
    if (!(directTabular instanceof TabularManagerImpl localTabularManager)) {
      manager
          .logger()
          .warn(
              "[ServiceManager][FlightSQL][dataset-scope-ignored] serviceId={} reason=direct-tabular-not-local datasetIds={}",
              serviceResourceId,
              datasetAllowlist);
      return directTabular;
    }
    String group = manager.resolveGroupName();
    if (group == null) {
      throw new IllegalStateException(
          "Session group is required when ServiceRunOptions.datasetIds is configured");
    }
    DuckdbCatalogImpl scopedCatalog =
        new DuckdbCatalogImpl(manager.session().discoveryCache(), localTabularManager.rowManager(), group);
    TabularManagerImpl scopedTabularManager =
        new TabularManagerImpl(localTabularManager.rowManager(), scopedCatalog);
    scopedTabularManager.installDatasets(datasetAllowlist);
    manager
        .logger()
        .info(
            "[ServiceManager][FlightSQL][dataset-scope] serviceId={} datasetCount={} datasetIds={}",
            serviceResourceId,
            datasetAllowlist.size(),
            datasetAllowlist);
    return scopedTabularManager;
  }

  static List<String> normalizeDatasetAllowlist(List<String> datasetIds) {
    if (datasetIds == null || datasetIds.isEmpty()) {
      return List.of();
    }
    LinkedHashSet<String> normalized = new LinkedHashSet<>();
    for (String value : datasetIds) {
      String trimmed = Strings.trimToNull(value);
      if (trimmed != null) {
        normalized.add(trimmed);
      }
    }
    return normalized.isEmpty() ? List.of() : List.copyOf(normalized);
  }

  static Set<DatasetId> parseDatasetAllowlist(List<String> datasetIds) {
    if (datasetIds == null || datasetIds.isEmpty()) {
      return Set.of();
    }
    LinkedHashSet<DatasetId> parsed = new LinkedHashSet<>();
    for (String value : datasetIds) {
      parsed.add(DatasetId.parse(Strings.requireNonBlank(value, "datasetIds entry")));
    }
    return parsed.isEmpty() ? Set.of() : Set.copyOf(parsed);
  }

  static FlightIngestProducer toFlightIngestProducer(
      ServiceManagerImpl manager, Object implementation, String serviceResourceId) {
    if (implementation == null) {
      return new FlightIngestProducer(
          manager.session().data().direct().ingest(),
          manager.entitlements(),
          serviceResourceId,
          manager.session().discoveryCache(),
          resolveRuntimeEnvironment(manager));
    }
    if (implementation instanceof FlightIngestProducer producer) {
      return producer;
    }
    throw new IllegalArgumentException(
        "Service implementation must be "
            + FlightIngestProducer.class.getName()
            + " for Arrow Flight ingest services: "
            + implementation.getClass().getName());
  }

  static void authorizeStart(ServiceManagerImpl manager, String serviceId, String subject) {
    authorizeStart(manager, serviceId, subject, ServiceEntitlements.Transport.GRPC);
  }

  static void authorizeStart(
      ServiceManagerImpl manager,
      String serviceId,
      String subject,
      ServiceEntitlements.Transport transport) {
    ServiceEntitlements.Decision decision =
        manager
            .entitlements()
            .authorize(
                new ServiceEntitlements.Identity(subject, null),
                new ServiceEntitlements.Request(
                    transport,
                    ServiceEntitlements.Action.START,
                    serviceId,
                    ServiceManagerImpl.startOperationName(transport),
                    null,
                    null,
                    Map.of(),
                    Map.of()));
    if (decision == null || !decision.allowed()) {
      throw new IllegalArgumentException(ServiceManagerImpl.decisionReason(decision));
    }
  }

  static String resolveServerStartSubject(ServiceManagerImpl manager, DefinitionService.Grpc grpc) {
    String appSubject = manager.sessionAppSubject();
    if (appSubject != null) {
      return appSubject;
    }
    String kerberosPrincipal = manager.sessionKerberosPrincipal();
    String serviceSpn = grpc == null || grpc.kerberos() == null ? null : grpc.kerberos().spn();
    return ServiceManagerImpl.firstNonBlank(kerberosPrincipal, serviceSpn);
  }

  static FlightProducer maybeInstrumentArrowServerProducer(
      ServiceManagerImpl manager,
      DefinitionService service,
      FlightProducer producer,
      String transport,
      boolean transportEnabled) {
    if (!transportEnabled) {
      return producer;
    }
    OtelManager otelManager = sessionOtelManager(manager);
    if (!otelManager.enabled()) {
      return producer;
    }
    return FabricArrowFlightOtel.instrumentServer(
        otelManager,
        producer,
        manager.resolveGroup(service),
        manager.currentAppName(),
        ServiceManagerImpl.resolveServiceName(service),
        transport);
  }

  static OtelManager sessionOtelManager(ServiceManagerImpl manager) {
    OtelManager otelManager = manager.session().telemetry();
    return otelManager == null ? OtelManagerNoop.instance() : otelManager;
  }

  static String resolveRuntimeEnvironment(ServiceManagerImpl manager) {
    return manager.session().context() == null
        ? null
        : Strings.trimToNull(manager.session().context().getEnv());
  }

  static ServiceStatusPublisher createGrpcStatusPublisher(
      ServiceManagerImpl manager,
      String serviceName,
      String instanceId,
      String advertiseHost,
      int advertisePort,
      String bindHost,
      int bindPort,
      DefinitionService.Grpc grpc,
      boolean tlsEnabled,
      ServiceRunOptions options) {
    ControlConnection control = manager.session().controlPlaneConnection();
    if (control == null) {
      throw new IllegalStateException("Session control-plane client is not initialized");
    }
    String group = ServiceManagerImpl.requireValue(manager.resolveGroupName(), "group");
    String region =
        manager.session().context() == null
            ? null
            : Strings.trimToNull(manager.session().context().getRegion());
    Map<String, String> attrs = new java.util.HashMap<>();
    attrs.put("bindHost", bindHost);
    attrs.put("bindPort", Integer.toString(bindPort));
    attrs.put("tlsEnabled", Boolean.toString(tlsEnabled));
    attrs.put("kerberosEnabled", Boolean.toString(ServiceManagerImpl.kerberosEnabled(grpc)));
    addDataResources(attrs, options);
    if (grpc.tls() != null) {
      String tlsTermination = Strings.trimToNull(grpc.tls().termination());
      if (tlsTermination != null) {
        attrs.put("tlsTermination", tlsTermination);
      }
    }
    if (grpc.kerberos() != null) {
      String kerberosSpn = Strings.trimToNull(grpc.kerberos().spn());
      if (kerberosSpn != null) {
        attrs.put("kerberosSpn", kerberosSpn);
      }
    }
    return new ServiceStatusPublisher(
        control,
        group,
        ServiceManagerImpl.requireValue(serviceName, "serviceName"),
        ServiceManagerImpl.requireValue(instanceId, "instanceId"),
        advertiseHost,
        advertisePort,
        region,
        manager.resolvedServiceHeartbeatSeconds(),
        manager.resolvedServiceStatusSeconds(),
        manager.resolvedServiceHeartbeatTtlSeconds(),
        tlsEnabled,
        null,
        null,
        null,
        manager.session().context() == null
            ? null
            : manager.session().context().getAppName(),
        manager.session().context() == null
            ? null
            : manager.session().context().getInstance(),
        group,
        attrs,
        manager.session().statusScheduler());
  }

        attrs.put("tlsTermination", tlsTermination);
      }
    }
    if (arrow.kerberos() != null) {
      String kerberosSpn = Strings.trimToNull(arrow.kerberos().spn());
      if (kerberosSpn != null) {
        attrs.put("kerberosSpn", kerberosSpn);
      }
    }
    return new ServiceStatusPublisher(
        control,
        group,
        ServiceManagerImpl.requireValue(serviceName, "serviceName"),
        ServiceManagerImpl.requireValue(instanceId, "instanceId"),
        advertiseHost,
        advertisePort,
        region,
        manager.resolvedServiceHeartbeatSeconds(),
        manager.resolvedServiceStatusSeconds(),
        manager.resolvedServiceHeartbeatTtlSeconds(),
        tlsEnabled,
        null,
        null,
        null,
        manager.session().context() == null
            ? null
            : manager.session().context().getAppName(),
        manager.session().context() == null
            ? null
            : manager.session().context().getInstance(),
        group,
        attrs,
        manager.session().statusScheduler());
  }

  static String resolveInstanceId(
      ServiceRunOptions options, String advertiseHost, int advertisePort) {
    String optionInstance = options == null ? null : Strings.trimToNull(options.instanceName());
    if (optionInstance != null) {
      return optionInstance;
    }
    String host = Strings.trimToNull(advertiseHost);
    if (host == null) {
      host = "host";
    }
    return "svc-" + host.replace('.', '-') + "-" + advertisePort;
  }

  private static void addDataResources(Map<String, String> attrs, ServiceRunOptions options) {
    if (attrs == null || options == null || Strings.isEmpty(options.datasetIds())) {
      return;
    }
    LinkedHashSet<String> normalized = new LinkedHashSet<>();
    for (String value : options.datasetIds()) {
      String datasetId = Strings.trimToNull(value);
      if (datasetId != null) {
        normalized.add(datasetId);
      }
    }
    if (normalized.isEmpty()) {
      return;
    }
    attrs.put("datasetIds", String.join(",", normalized));
    attrs.put("datasetCount", Integer.toString(normalized.size()));
  }
