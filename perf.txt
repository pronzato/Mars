    append(manager, ingestName, payload, DuckDbActiveMutation.UPSERT, true);
  }

    append(manager, ingestName, payload, mutation, true);
  }

  static void append(
      IngestManagerImpl manager,
      String ingestName,
      IngestPayload payload,
      DuckDbActiveMutation mutation,
      boolean enforceWriteGate) {
    String normalizedName = Strings.trimToNull(ingestName);
    if (normalizedName == null) {
      throw new IllegalArgumentException("ingest name is required");
    }
    Objects.requireNonNull(payload, "payload");
    DuckDbActiveMutation resolvedMutation =
        mutation == null ? DuckDbActiveMutation.UPSERT : mutation;

    DefinitionIngest ingest = IngestManagerImplDataset.resolveIngestDefinition(manager, normalizedName);
    Optional<DefinitionIngestSource> sourceDefinition =
        manager.DiscoveryCache.get(DefinitionIngestSource.class, manager.group, normalizedName);
    if (sourceDefinition.isEmpty()) {
      throw new IllegalStateException(
          "IngestSource not found for ingest: " + manager.group + "/" + normalizedName);
    }

    DefinitionIngestSource ingestSource = sourceDefinition.get();
    DefinitionEnvelope sourceEnvelope = ingestSource.envelope();
    if (sourceEnvelope != null && Boolean.FALSE.equals(sourceEnvelope.enabled())) {
      throw new IllegalStateException("IngestSource is disabled: " + manager.group + "/" + normalizedName);
    }

    String actualFormat =
        Strings.requireNonBlank(ingestSource.format(), "ingestSource.format")
            .toUpperCase(Locale.ROOT);
    String expectedFormat;
    FormatDecoder decoder;
    byte[] payloadUtf8Bytes;

    if (payload instanceof IngestPayloadJson jsonPayload) {
      expectedFormat = IngestManagerImpl.FORMAT_JSON;
      decoder = new FormatDecoderJson(ingestSource.json());
      payloadUtf8Bytes = jsonPayload.json().getBytes(StandardCharsets.UTF_8);
    } else if (payload instanceof IngestPayloadNvFix nvfixPayload) {
      expectedFormat = IngestManagerImpl.FORMAT_NVFIX;
      decoder = new FormatDecoderNvFix(ingestSource.nvfix());
      payloadUtf8Bytes = nvfixPayload.nvfix().getBytes(StandardCharsets.UTF_8);
    } else {
      throw new IllegalArgumentException(
          "Unsupported payload type: " + payload.getClass().getName());
    }

    if (!expectedFormat.equals(actualFormat)) {
      throw new IllegalStateException(
          "IngestSource format mismatch: ingest="
              + manager.group
              + "/"
              + normalizedName
              + " expected="
              + expectedFormat
              + " actual="
              + actualFormat);
    }

    String datasetId = Strings.trimToNull(ingest.datasetId());
    if (datasetId == null) {
      throw new IllegalArgumentException(
          "Ingest datasetId is required: " + manager.group + "/" + normalizedName);
    }
    DatasetId parsedDatasetId = DatasetId.parse(datasetId);
    if (enforceWriteGate) {
      enforceWrite(manager, parsedDatasetId.toString(), IngestManagerImpl.OPERATION_APPEND, null);
    }

    String outputSchemaId = Strings.trimToNull(ingest.outputSchemaId());
    if (outputSchemaId == null) {
      throw new IllegalArgumentException(
          "Ingest outputSchemaId is required: " + manager.group + "/" + normalizedName);
    }
    DefinitionIngest.SchemaMode schemaMode = ingest.effectiveSchemaMode();
    String ingestId = manager.group + "/" + normalizedName;
    DefinitionSchema expectedSchemaDefinition =
        IngestSchemaResolver.resolveExpectedSchema(
            manager.DiscoveryCache, manager.group, outputSchemaId, ingestId);

    byte[] ipcBytes =
        decoder.decodeToArrowIpc(
            new FormatDecodeRequest(
                ingestId,
                outputSchemaId,
                expectedSchemaDefinition,
                payloadUtf8Bytes,
                schemaMode,
                FormatDecodeRequest.DEFAULT_ROW_LIMIT_PER_BATCH));
    appendIpc(manager, normalizedName, ipcBytes, resolvedMutation, false);
  }

  static void appendIpc(IngestManagerImpl manager, String ingestName, byte[] arrowIpcStreamBytes) {
    appendIpc(manager, ingestName, arrowIpcStreamBytes, DuckDbActiveMutation.UPSERT, true);
  }

  static void appendIpc(
      IngestManagerImpl manager,
      String ingestName,
      byte[] arrowIpcStreamBytes,
      DuckDbActiveMutation mutation) {
    appendIpc(manager, ingestName, arrowIpcStreamBytes, mutation, true);
  }

  static void appendIpc(
      IngestManagerImpl manager,
      String ingestName,
      byte[] arrowIpcStreamBytes,
      DuckDbActiveMutation mutation,
      boolean enforceWriteGate) {
    String normalizedName = Strings.trimToNull(ingestName);
    if (normalizedName == null) {
      throw new IllegalArgumentException("ingest name is required");
    }
    DuckDbActiveMutation resolvedMutation =
        mutation == null ? DuckDbActiveMutation.UPSERT : mutation;

    DefinitionIngest ingest = IngestManagerImplDataset.resolveIngestDefinition(manager, normalizedName);
    DefinitionEnvelope envelope = ingest.envelope();
    if (envelope != null && Boolean.FALSE.equals(envelope.enabled())) {
      throw new IllegalStateException("Ingest is disabled: " + manager.group + "/" + normalizedName);
    }

    String inputType = ingest.input() == null ? null : Strings.trimToNull(ingest.input().type());
    if (!IngestManagerImpl.INPUT_TYPE_API_IPC.equals(inputType)) {
      throw new IllegalArgumentException(
          "Unsupported ingest input.type for "
              + manager.group
              + "/"
              + normalizedName
              + ": "
              + inputType
              + " (expected "
              + IngestManagerImpl.INPUT_TYPE_API_IPC
              + ")");
    }

    String datasetId = Strings.trimToNull(ingest.datasetId());
    if (datasetId == null) {
      throw new IllegalArgumentException(
          "Ingest datasetId is required: " + manager.group + "/" + normalizedName);
    }
    DatasetId parsedDatasetId = DatasetId.parse(datasetId);
    if (enforceWriteGate) {
      enforceWrite(manager, parsedDatasetId.toString(), IngestManagerImpl.OPERATION_INGEST, null);
    }

    String outputSchemaId = Strings.trimToNull(ingest.outputSchemaId());
    if (outputSchemaId == null) {
      throw new IllegalArgumentException(
          "Ingest outputSchemaId is required: " + manager.group + "/" + normalizedName);
    }

    String ingestId = manager.group + "/" + normalizedName;
    DefinitionIngest.SchemaMode schemaMode = ingest.effectiveSchemaMode();
    DefinitionSchema expectedSchemaDefinition =
        IngestSchemaResolver.resolveExpectedSchema(
            manager.DiscoveryCache, manager.group, outputSchemaId, ingestId);
    IngestManagerImplSchema.ParsedExpectedSchema expectedSchema =
        IngestManagerImplSchema.parseExpectedSchema(expectedSchemaDefinition, ingestId, outputSchemaId);
    Schema actualSchema =
        IngestManagerImplSchema.readArrowSchema(
            arrowIpcStreamBytes, ingestId, manager.schemaIpcAllocator);
    IngestManagerImplSchema.SchemaComparison comparison =
        IngestManagerImplSchema.compareSchemas(expectedSchema, actualSchema, schemaMode);
    if (!comparison.matches()) {
      throw new IllegalStateException(
          IngestManagerImplSchema.formatSchemaMismatch(
              ingestId,
              schemaMode,
              outputSchemaId,
              actualSchema,
              comparison.missingFields(),
              comparison.extraFields(),
              comparison.typeMismatches(),
              comparison.nullabilityMismatches(),
              comparison.appendOnlyViolations(),
              comparison.nonNullableExtraFields(),
              comparison.expectedFieldCount(),
              comparison.actualFieldCount()));
    }

    DefinitionDataset datasetDefinition =
        IngestManagerImplDataset.resolveDatasetDefinition(manager, datasetId);
    appendToDatasetWithRolling(
        manager,
        normalizedName,
        parsedDatasetId,
        datasetDefinition,
        arrowIpcStreamBytes,
        resolvedMutation);
  }

  static void flushPendingParquetBuffer(IngestManagerImpl manager, String ingestName, String reason) {
    if (Strings.isBlank(ingestName)) {
      return;
    }
    IngestManagerImpl.ParquetRollBuffer buffer = manager.parquetRollBuffers.get(ingestName);
    if (buffer == null) {
      return;
    }
    synchronized (buffer.monitor) {
      flushBufferLocked(manager, ingestName, buffer, reason);
      if (buffer.pendingIpcBatches.isEmpty()) {
        manager.parquetRollBuffers.remove(ingestName, buffer);
      }
    }
  }

  static void flushAllPendingParquetBuffers(IngestManagerImpl manager, String reason) {
    for (String ingestName : new ArrayList<>(manager.parquetRollBuffers.keySet())) {
      flushPendingParquetBuffer(manager, ingestName, reason);
    }
  }

  static void flushIdleParquetBuffersSafe(IngestManagerImpl manager) {
    long now = System.currentTimeMillis();
    for (Map.Entry<String, IngestManagerImpl.ParquetRollBuffer> entry :
        manager.parquetRollBuffers.entrySet()) {
      String ingestName = entry.getKey();
      IngestManagerImpl.ParquetRollBuffer buffer = entry.getValue();
      try {
        synchronized (buffer.monitor) {
          if (buffer.pendingIpcBatches.isEmpty()) {
            continue;
          }
          long maxIdleMillis = buffer.policy.maxIdleMillis();
          if (maxIdleMillis <= 0L) {
            continue;
          }
          long idleFor = now - buffer.lastAppendMillis;
          if (idleFor < maxIdleMillis) {
            continue;
          }
          flushBufferLocked(
              manager,
              ingestName,
              buffer,
              "maxIdleMillis=" + maxIdleMillis + " elapsed (idleFor=" + idleFor + "ms)");
        }
      } catch (RuntimeException ex) {
        IngestManagerImpl.LOGGER.error(
            "[IngestManager] idle parquet roll failed ingest={}: {}",
            ingestName,
            ex.getMessage(),
            ex);
      }
    }
  }

  static void ensureRollScanTask(IngestManagerImpl manager) {
    if (manager.rollScheduler == null) {
      return;
    }
    ScheduledFuture<?> existing = manager.rollScanTask.get();
    if (existing != null && !existing.isCancelled()) {
      return;
    }
    ScheduledFuture<?> created;
    try {
      created =
          manager.rollScheduler.scheduleWithFixedDelay(
              () -> flushIdleParquetBuffersSafe(manager),
              IngestManagerImpl.ROLL_SCAN_INTERVAL_MILLIS,
              IngestManagerImpl.ROLL_SCAN_INTERVAL_MILLIS,
              TimeUnit.MILLISECONDS);
    } catch (RuntimeException ex) {
      IngestManagerImpl.LOGGER.warn(
          "[IngestManager] unable to start parquet idle roll scan: {}", ex.getMessage());
      return;
    }
    if (!manager.rollScanTask.compareAndSet(existing, created)) {
      created.cancel(false);
    }
  }

  static void cancelRollScanTask(IngestManagerImpl manager) {
    ScheduledFuture<?> task = manager.rollScanTask.getAndSet(null);
    if (task != null) {
      task.cancel(false);
    }
  }

  static IngestManagerImpl.ParquetRollPolicy resolveParquetRollPolicy(
      IngestManagerImpl manager, String ingestName, DefinitionDataset datasetDefinition) {
    if (datasetDefinition == null || datasetDefinition.binding() == null) {
      return IngestManagerImpl.ParquetRollPolicy.disabled();
    }
    DefinitionDataset.ParquetBinding parquet = datasetDefinition.binding().parquet();
    if (parquet == null) {
      return IngestManagerImpl.ParquetRollPolicy.disabled();
    }
    IngestManagerImpl.ParquetRollPolicy runtimePolicy =
        Strings.isBlank(ingestName) ? null : manager.parquetRollPolicies.get(ingestName);
    if (runtimePolicy != null) {
      return runtimePolicy;
    }
    return new IngestManagerImpl.ParquetRollPolicy(
        IngestManagerImpl.DEFAULT_PARQUET_MAX_ROWS_PER_FILE,
        IngestManagerImpl.DEFAULT_PARQUET_MAX_IDLE_MILLIS);
  }

  static void requireWrite(IngestManagerImpl manager, String datasetId, String operation) {
    enforceWrite(manager, datasetId, operation, null);
  }

  static void enforceWrite(
      IngestManagerImpl manager, String datasetId, String operation, List<String> columnsOrNull) {
    if (!(manager.rowManager instanceof RowManagerImpl localDataManager)) {
      return;
    }
    DataWriteAuthorization.requireWrite(
        localDataManager.dataEntitlementsForSqlPlanning(),
        localDataManager.effectiveIdentityForSqlPlanning(),
        datasetId,
        operation,
        columnsOrNull);
  }

  static int payloadBytes(IngestPayload payload) {
    if (payload instanceof IngestPayloadJson json) {
      return json.json().getBytes(StandardCharsets.UTF_8).length;
    }
    if (payload instanceof IngestPayloadNvFix nvfix) {
      return nvfix.nvfix().getBytes(StandardCharsets.UTF_8).length;
    }
    return 0;
  }

  private static void appendToDataset(
      IngestManagerImpl manager,
      DatasetId datasetId,
      byte[] arrowIpcStreamBytes,
      DuckDbActiveMutation mutation,
      String parquetFilenamePrefix) {
    DuckDbActiveMutation resolvedMutation =
        mutation == null ? DuckDbActiveMutation.UPSERT : mutation;
    if (manager.rowManager instanceof RowManagerImpl local) {
      local.appendIpc(datasetId, arrowIpcStreamBytes, resolvedMutation, parquetFilenamePrefix);
      return;
    }
    if (resolvedMutation != DuckDbActiveMutation.UPSERT) {
      throw new IllegalStateException(
          "DuckDB ACTIVE mutation mode " + resolvedMutation + " requires DIRECT runtime");
    }
    if (manager.rowManager instanceof RemoteRowManager remote) {
      remote.appendIpc(datasetId, arrowIpcStreamBytes);
      return;
    }
    throw new IllegalStateException(
        "Arrow IPC append requires a direct or remote row manager implementation");
  }

  private static void appendToDatasetWithRolling(
      IngestManagerImpl manager,
      String ingestName,
      DatasetId datasetId,
      DefinitionDataset datasetDefinition,
      byte[] arrowIpcStreamBytes,
      DuckDbActiveMutation mutation) {
    if (datasetDefinition != null
        && datasetDefinition.binding() != null
        && datasetDefinition.binding().odbc() != null
        && (datasetDefinition.type() == null || datasetDefinition.type() == DatasetType.EXTERNAL)) {
      flushPendingParquetBuffer(manager, ingestName, "odbc-target");
      OdbcRuntimeGuard.failIngestRuntime(ingestName, datasetDefinition.binding().odbc().databaseType());
    }
    if (datasetDefinition != null
        && datasetDefinition.binding() != null
        && datasetDefinition.binding().external() != null
        && (datasetDefinition.type() == null || datasetDefinition.type() == DatasetType.EXTERNAL)) {
      flushPendingParquetBuffer(manager, ingestName, "external-target");
      appendToExternalTarget(
          manager,
          ingestName,
          datasetId,
          datasetDefinition,
          datasetDefinition.binding().external(),
          arrowIpcStreamBytes,
          mutation);
      return;
    }

    String parquetFilenamePrefix = resolveParquetFilenamePrefix(manager, ingestName);
    DuckDbActiveMutation resolvedMutation =
        mutation == null ? DuckDbActiveMutation.UPSERT : mutation;
    if (resolvedMutation != DuckDbActiveMutation.UPSERT) {
      flushPendingParquetBuffer(manager, ingestName, "mutation=" + resolvedMutation.name());
      appendToDataset(manager, datasetId, arrowIpcStreamBytes, resolvedMutation, parquetFilenamePrefix);
      return;
    }

    IngestManagerImpl.ParquetRollPolicy policy =
        resolveParquetRollPolicy(manager, ingestName, datasetDefinition);
    if (manager.rollScheduler == null && policy.maxIdleMillis() > 0L) {
      policy = new IngestManagerImpl.ParquetRollPolicy(policy.maxRowsPerFile(), 0L);
    }
    if (!policy.enabled()) {
      flushPendingParquetBuffer(manager, ingestName, "rolling-disabled");
      manager.parquetRollBuffers.remove(ingestName);
      appendToDataset(
          manager,
          datasetId,
          arrowIpcStreamBytes,
          DuckDbActiveMutation.UPSERT,
          parquetFilenamePrefix);
      return;
    }

    IngestManagerImpl.ArrowIpcBatchMeta batchMeta =
        inspectArrowIpcBatch(manager, arrowIpcStreamBytes, manager.group + "/" + ingestName);
    if (batchMeta.rowCount() <= 0L) {
      return;
    }
    if (policy.maxIdleMillis() > 0L) {
      ensureRollScanTask(manager);
    }

    IngestManagerImpl.ParquetRollPolicy effectivePolicy = policy;
    IngestManagerImpl.ParquetRollBuffer buffer =
        manager.parquetRollBuffers.computeIfAbsent(
            ingestName,
            key -> new IngestManagerImpl.ParquetRollBuffer(datasetId, effectivePolicy, parquetFilenamePrefix));
    synchronized (buffer.monitor) {
      if (!datasetId.equals(buffer.datasetId)
          || !buffer.policy.equals(policy)
          || !Objects.equals(buffer.parquetFilenamePrefix, parquetFilenamePrefix)) {
        flushBufferLocked(manager, ingestName, buffer, "policy-change");
        buffer.datasetId = datasetId;
        buffer.policy = policy;
        buffer.parquetFilenamePrefix = parquetFilenamePrefix;
        buffer.schema = null;
      }
      if (buffer.schema != null && batchMeta.schema() != null && !buffer.schema.equals(batchMeta.schema())) {
        flushBufferLocked(manager, ingestName, buffer, "schema-change");
      }
      buffer.pendingIpcBatches.add(arrowIpcStreamBytes);
      buffer.pendingRows += batchMeta.rowCount();
      buffer.lastAppendMillis = System.currentTimeMillis();
      if (buffer.schema == null && batchMeta.schema() != null) {
        buffer.schema = batchMeta.schema();
      }
      if (policy.maxRowsPerFile() > 0 && buffer.pendingRows >= policy.maxRowsPerFile()) {
        flushBufferLocked(
            manager,
            ingestName,
            buffer,
            "maxRowsPerFile="
                + policy.maxRowsPerFile()
                + " reached (rows="
                + buffer.pendingRows
                + ")");
      }
    }
  }

  private static void appendToExternalTarget(
      IngestManagerImpl manager,
      String ingestName,
      DatasetId datasetId,
      DefinitionDataset datasetDefinition,
      DefinitionDataset.ExternalBinding externalBinding,
      byte[] arrowIpcStreamBytes,
      DuckDbActiveMutation mutation) {
    if (manager.session == null) {
      throw new IllegalStateException(
          "External ingest target requires a session-backed ingest manager");
    }
    String providerId =
        Strings.requireNonBlank(
            externalBinding == null ? null : externalBinding.externalType(),
            "dataset.binding.external.externalType");
    if (OdbcRuntimeGuard.isOdbcType(providerId)) {
      OdbcRuntimeGuard.failIngestRuntime(
          ingestName,
          Strings.trimToNull(
              externalBinding.options() == null ? null : externalBinding.options().get("databaseType")));
    }
    IngestTargetProvider provider = manager.plugins.findIngestTargetProvider(providerId);
    if (provider == null) {
      throw new IllegalStateException(
          "No ingest target provider registered for externalType '" + providerId + "'");
    }
    List<Map<String, Object>> rows =
        ArrowIpcRowDecoder.decodeToRowMaps(arrowIpcStreamBytes, manager.schemaIpcAllocator);
    try {
      provider.append(
          new IngestTargetAppendContext(
              manager.session,
              ingestName,
              datasetId,
              datasetDefinition,
              externalBinding,
              arrowIpcStreamBytes,
              rows,
              mutation == null ? DuckDbActiveMutation.UPSERT.name() : mutation.name()));
    } catch (RuntimeException ex) {
      throw ex;
    } catch (Exception ex) {
      throw new IllegalStateException(
          "External ingest target append failed for provider '" + provider.providerId() + "'",
          ex);
    }
  }

  private static String resolveParquetFilenamePrefix(IngestManagerImpl manager, String ingestName) {
    String override = Strings.trimToNull(manager.parquetFilenamePrefixOverride.get());
    if (override != null) {
      return override;
    }
    String normalizedIngest = Strings.trimToNull(ingestName);
    if (normalizedIngest == null) {
      return null;
    }
    return Strings.trimToNull(manager.parquetFilenamePrefixes.get(normalizedIngest));
  }

  private static void flushBufferLocked(
      IngestManagerImpl manager,
      String ingestName,
      IngestManagerImpl.ParquetRollBuffer buffer,
      String reason) {
    if (buffer.pendingIpcBatches.isEmpty()) {
      return;
    }
    byte[] merged =
        mergeArrowIpcBatches(
            manager,
            buffer.pendingIpcBatches,
            buffer.schema,
            manager.group + "/" + Strings.firstNonBlank(ingestName, "<unknown>"));
    if (merged.length > 0) {
      appendToDataset(
          manager,
          buffer.datasetId,
          merged,
          DuckDbActiveMutation.UPSERT,
          buffer.parquetFilenamePrefix);
    }
    IngestManagerImpl.LOGGER.info(
        "[IngestManager] rolled parquet ingest={} reason={} rows={} batches={}",
        ingestName,
        reason,
        buffer.pendingRows,
        buffer.pendingIpcBatches.size());
    buffer.pendingIpcBatches.clear();
    buffer.pendingRows = 0L;
    buffer.lastAppendMillis = 0L;
    buffer.schema = null;
  }

  private static IngestManagerImpl.ArrowIpcBatchMeta inspectArrowIpcBatch(
      IngestManagerImpl manager, byte[] arrowIpcStreamBytes, String ingestId) {
    try (ArrowStreamReader reader =
        new ArrowStreamReader(new ByteArrayInputStream(arrowIpcStreamBytes), manager.schemaIpcAllocator)) {
      VectorSchemaRoot root = reader.getVectorSchemaRoot();
      Schema schema = root == null ? null : root.getSchema();
      long rowCount = 0L;
      while (reader.loadNextBatch()) {
        rowCount += root.getRowCount();
      }
      return new IngestManagerImpl.ArrowIpcBatchMeta(rowCount, schema);
    } catch (IOException | RuntimeException ex) {
      throw new IllegalStateException(
          "Failed to inspect Arrow IPC rows for ingest=" + ingestId + ": " + ex.getMessage(), ex);
    }
  }

  private static byte[] mergeArrowIpcBatches(
      IngestManagerImpl manager, List<byte[]> batches, Schema expectedSchema, String ingestId) {
    if (batches == null || batches.isEmpty()) {
      return new byte[0];
    }
    if (batches.size() == 1) {
      return batches.get(0);
    }
    try (ByteArrayOutputStream output = new ByteArrayOutputStream()) {
      Schema resolvedSchema = expectedSchema;
      VectorSchemaRoot outputRoot = null;
      ArrowStreamWriter writer = null;
      try {
        for (byte[] batchBytes : batches) {
          try (ArrowStreamReader reader =
              new ArrowStreamReader(new ByteArrayInputStream(batchBytes), manager.rollIpcAllocator)) {
            VectorSchemaRoot inputRoot = reader.getVectorSchemaRoot();
            Schema inputSchema = inputRoot == null ? null : inputRoot.getSchema();
            if (resolvedSchema == null) {
              resolvedSchema = inputSchema;
            } else if (inputSchema != null && !resolvedSchema.equals(inputSchema)) {
              throw new IllegalStateException(
                  "Arrow IPC schema mismatch while rolling ingest="
                      + ingestId
                      + " expected="
                      + resolvedSchema
                      + " actual="
                      + inputSchema);
            }
            if (resolvedSchema == null) {
              continue;
            }
            if (outputRoot == null) {
              outputRoot = VectorSchemaRoot.create(resolvedSchema, manager.rollIpcAllocator);
              writer = new ArrowStreamWriter(outputRoot, null, output);
              writer.start();
            }
            VectorLoader loader = new VectorLoader(outputRoot);
            while (reader.loadNextBatch()) {
              try (ArrowRecordBatch recordBatch = new VectorUnloader(inputRoot).getRecordBatch()) {
                loader.load(recordBatch);
                writer.writeBatch();
              }
            }
          }
        }
        if (writer == null) {
          return new byte[0];
        }
        writer.end();
        return output.toByteArray();
      } finally {
        if (writer != null) {
          writer.close();
        }
        if (outputRoot != null) {
          outputRoot.close();
        }
      }
    } catch (IOException | RuntimeException ex) {
      throw new IllegalStateException(
          "Failed to merge Arrow IPC batches for ingest=" + ingestId + ": " + ex.getMessage(), ex);
    }
  }
