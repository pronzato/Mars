package org.pronzato.fabric.api.data.catalog;

import java.util.ArrayList;
import java.util.Comparator;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Locale;
import java.util.Objects;
import java.util.Set;
import org.pronzato.fabric.api.data.dataset.DatasetId;
import org.pronzato.fabric.api.util.Strings;

public final class DatasetNameResolver {
  private static final Comparator<DatasetId> DATASET_ID_ORDER =
      Comparator.comparing(DatasetId::toString, String.CASE_INSENSITIVE_ORDER)
          .thenComparing(DatasetId::toString);
  private static final Set<String> TABLE_CLAUSE_KEYWORDS =
      Set.of("from", "join", "update", "into");
  private static final Set<String> STOP_KEYWORDS =
      Set.of(
          "where",
          "group",
          "order",
          "having",
          "qualify",
          "window",
          "union",
          "except",
          "intersect",
          "limit",
          "offset",
          "fetch",
          "on",
          "using",
          "set",
          "values",
          "returning");

  private final String sessionGroup;
  private final String defaultNamespace;
  private final CatalogSnapshot snapshot;

  public DatasetNameResolver(String sessionGroup, String defaultNamespace, CatalogSnapshot snapshot) {
    this.sessionGroup = Strings.requireNonBlank(sessionGroup, "sessionGroup");
    this.defaultNamespace = Strings.requireNonBlank(defaultNamespace, "defaultNamespace");
    this.snapshot = Objects.requireNonNull(snapshot, "snapshot");
  }

  public DatasetId resolveTableToken(String token) {
    String normalizedToken = normalizeToken(token);
    if (normalizedToken.indexOf('.') >= 0) {
      DatasetId qualified = snapshot.qualifiedNameMap().get(normalizedToken.toLowerCase(Locale.ROOT));
      if (qualified == null) {
        throw unknownTableToken(token);
      }
      return qualified;
    }

    List<DatasetId> candidates = snapshot.unqualifiedNameMap().get(normalizedToken.toLowerCase(Locale.ROOT));
    if (candidates == null || candidates.isEmpty()) {
      throw unknownTableToken(token);
    }
    if (candidates.size() == 1) {
      return candidates.get(0);
    }
    DatasetId preferred = preferDefaultNamespaceCandidate(candidates);
    if (preferred != null) {
      return preferred;
    }
    List<String> ids = new ArrayList<>(candidates.size());
    List<DatasetId> sorted = new ArrayList<>(candidates);
    sorted.sort(DATASET_ID_ORDER);
    for (DatasetId candidate : sorted) {
      ids.add(candidate.toString());
    }
    throw new IllegalArgumentException(
        "Ambiguous table token '"
            + token
            + "' resolves to multiple datasets: ["
            + String.join(",", ids)
            + "]. Use namespace.foo or fully qualify group.namespace.foo");
  }

  public List<DatasetId> resolveAllTableTokensInSql(String sql) {
    return resolveAllTableTokensInSql(sql, false);
  }

  public List<DatasetId> resolveAllTableTokensInSql(String sql, boolean strict) {
    List<DatasetToken> tokens = scanTableTokensInSql(sql, strict);
    LinkedHashSet<DatasetId> resolved = new LinkedHashSet<>();
    for (DatasetToken token : tokens) {
      resolved.add(resolveTableToken(token.normalizedToken()));
    }
    return List.copyOf(resolved);
  }

  private DatasetId preferDefaultNamespaceCandidate(List<DatasetId> candidates) {
    if (candidates == null || candidates.isEmpty()) {
      return null;
    }
    for (DatasetId candidate : candidates) {
      if (candidate == null) {
        continue;
      }
      if (!sessionGroup.equals(candidate.group())) {
        continue;
      }
      if (defaultNamespace.equals(candidate.namespace())) {
        return candidate;
      }
    }
    return null;
  }

  List<DatasetToken> scanTableTokensInSql(String sql) {
    return scanTableTokensInSql(sql, false);
  }

  List<DatasetToken> scanTableTokensInSql(String sql, boolean strict) {
    String query = Strings.requireNonBlank(sql, "sql");
    List<SqlToken> tokens = tokenize(query);
    Set<String> cteNames = collectLeadingCteNames(tokens);
    List<DatasetToken> resolved = new ArrayList<>();
    for (int index = 0; index < tokens.size(); index++) {
      SqlToken token = tokens.get(index);
      if (!token.word()) {
        continue;
      }
      String keyword = token.normalizedLower();
      if (!TABLE_CLAUSE_KEYWORDS.contains(keyword)) {
        continue;
      }
      switch (keyword) {
        case "from" -> index = scanFromClause(query, tokens, cteNames, strict, index + 1, resolved) - 1;
        case "join", "update", "into" -> {
          int next = scanSingleTableReference(query, tokens, cteNames, keyword, strict, index + 1, resolved);
          index = Math.max(index, next - 1);
        }
        default -> {
          // no-op
        }
      }
    }
    return List.copyOf(resolved);
  }

  private int scanFromClause(
      String sql,
      List<SqlToken> tokens,
      Set<String> cteNames,
      boolean strict,
      int start,
      List<DatasetToken> resolved) {
    int index = start;
    while (index < tokens.size()) {
      SqlToken token = tokens.get(index);
      if (isStopKeyword(token)) {
        return index;
      }
      if (token.symbol("(")) {
        if (strict) {
          throw unsupportedSql("from", token.start(), "derived table reference", null);
        }
        return index;
      }
      int next = addTableReference(sql, tokens, cteNames, "from", strict, index, resolved);
      if (next < 0) {
        return index + 1;
      }
      index = next;
      if (index < tokens.size() && tokens.get(index).symbol(",")) {
        index++;
        continue;
      }
      return index;
    }
    return index;
  }

  private int scanSingleTableReference(
      String sql,
      List<SqlToken> tokens,
      Set<String> cteNames,
      String keyword,
      boolean strict,
      int start,
      List<DatasetToken> resolved) {
    return Math.max(start, addTableReference(sql, tokens, cteNames, keyword, strict, start, resolved));
  }

  private int addTableReference(
      String sql,
      List<SqlToken> tokens,
      Set<String> cteNames,
      String keyword,
      boolean strict,
      int start,
      List<DatasetToken> resolved) {
    if (start >= tokens.size()) {
      if (strict) {
        throw unsupportedSql(keyword, sql.length(), "unable to parse table reference", null);
      }
      return -1;
    }
    SqlToken token = tokens.get(start);
    if (token.symbol("(")) {
      if (strict) {
        throw unsupportedSql(keyword, token.start(), "derived table reference", null);
      }
      return -1;
    }
    ParsedIdentifier parsed = parseIdentifier(sql, tokens, start, strict);
    if (parsed == null) {
      if (strict) {
        throw unsupportedSql(keyword, token.start(), "unable to parse table reference", null);
      }
      return -1;
    }
    if (parsed.nextIndex() < tokens.size() && tokens.get(parsed.nextIndex()).symbol("(")) {
      if (strict) {
        throw unsupportedSql(null, parsed.startOffset(), "table function reference", parsed.rawToken());
      }
      return -1;
    }
    AliasReference alias = parseAlias(sql, tokens, parsed.nextIndex(), strict);
    int referenceEnd = alias == null ? parsed.endOffset() : alias.endOffset();
    int nextIndex = alias == null ? parsed.nextIndex() : alias.nextIndex();
    String normalizedLower = parsed.normalizedToken().toLowerCase(Locale.ROOT);
    if (cteNames.contains(normalizedLower)) {
      return nextIndex;
    }
    resolved.add(
        new DatasetToken(
            parsed.rawToken(),
            parsed.normalizedToken(),
            parsed.startOffset(),
            referenceEnd,
            alias == null ? null : alias.aliasToken(),
            alias != null && alias.usesAsKeyword()));
    return nextIndex;
  }

  private static boolean isStopKeyword(SqlToken token) {
    return token != null && token.word() && STOP_KEYWORDS.contains(token.normalizedLower());
  }

  private static AliasReference parseAlias(String sql, List<SqlToken> tokens, int start) {
    return parseAlias(sql, tokens, start, false);
  }

  private static AliasReference parseAlias(String sql, List<SqlToken> tokens, int start, boolean strict) {
    if (start >= tokens.size()) {
      return null;
    }
    SqlToken token = tokens.get(start);
    if (token.word() && "as".equals(token.normalizedLower())) {
      int aliasIndex = start + 1;
      if (aliasIndex >= tokens.size()) {
        return null;
      }
      SqlToken aliasToken = tokens.get(aliasIndex);
      if (!isAliasToken(aliasToken)) {
        return null;
      }
      return new AliasReference(
          sql.substring(aliasToken.start(), aliasToken.end()), true, aliasToken.end(), aliasIndex + 1);
    }
    if (!isAliasToken(token)) {
      return null;
    }
    return new AliasReference(sql.substring(token.start(), token.end()), false, token.end(), start + 1);
  }

  private static boolean isAliasToken(SqlToken token) {
    if (token == null || (!token.word() && !token.quotedIdentifier())) {
      return false;
    }
    String lower = token.normalizedLower();
    if (isReservedWord(lower) || "join".equals(lower)) {
      return false;
    }
    return true;
  }

  private static Set<String> collectLeadingCteNames(List<SqlToken> tokens) {
    if (tokens.isEmpty()) {
      return Set.of();
    }
    int index = 0;
    SqlToken first = tokens.get(index);
    if (!first.word() || !"with".equals(first.normalizedLower())) {
      return Set.of();
    }
    index++;
    if (index < tokens.size()
        && tokens.get(index).word()
        && "recursive".equals(tokens.get(index).normalizedLower())) {
      index++;
    }

    LinkedHashSet<String> names = new LinkedHashSet<>();
    while (index < tokens.size()) {
      SqlToken cteName = tokens.get(index);
      if (!cteName.word() && !cteName.quotedIdentifier()) {
        break;
      }
      if (isReservedWord(cteName.normalizedLower())) {
        break;
      }
      names.add(cteName.identifierValue().toLowerCase(Locale.ROOT));
      index++;

      if (index < tokens.size() && tokens.get(index).symbol("(")) {
        int next = skipBalancedParentheses(tokens, index);
        if (next < 0) {
          break;
        }
        index = next;
      }
      if (index >= tokens.size() || !tokens.get(index).word() || !"as".equals(tokens.get(index).normalizedLower())) {
        break;
      }
      index++;
      if (index >= tokens.size() || !tokens.get(index).symbol("(")) {
        break;
      }
      int next = skipBalancedParentheses(tokens, index);
      if (next < 0) {
        break;
      }
      index = next;

      if (index < tokens.size() && tokens.get(index).symbol(",")) {
        index++;
        continue;
      }
      break;
    }
    return names.isEmpty() ? Set.of() : Set.copyOf(names);
  }

  private static int skipBalancedParentheses(List<SqlToken> tokens, int openIndex) {
    if (openIndex >= tokens.size() || !tokens.get(openIndex).symbol("(")) {
      return -1;
    }
    int depth = 0;
    for (int index = openIndex; index < tokens.size(); index++) {
      SqlToken token = tokens.get(index);
      if (token.symbol("(")) {
        depth++;
        continue;
      }
      if (token.symbol(")")) {
        depth--;
        if (depth == 0) {
          return index + 1;
        }
      }
    }
    return -1;
  }

  private static ParsedIdentifier parseIdentifier(String sql, List<SqlToken> tokens, int start) {
    return parseIdentifier(sql, tokens, start, false);
  }

  private static ParsedIdentifier parseIdentifier(
      String sql, List<SqlToken> tokens, int start, boolean strict) {
    if (start >= tokens.size()) {
      return null;
    }
    SqlToken first = tokens.get(start);
    if (!first.word() && !first.quotedIdentifier()) {
      return null;
    }
    if (isReservedWord(first.normalizedLower())) {
      return null;
    }
    String firstPart = first.identifierValue();
    if (firstPart == null) {
      return null;
    }
    int end = first.end();
    int next = start + 1;
    String normalized = firstPart;
    if (next + 1 < tokens.size() && tokens.get(next).symbol(".")) {
      SqlToken second = tokens.get(next + 1);
      if (!second.word() && !second.quotedIdentifier()) {
        return null;
      }
      if (isReservedWord(second.normalizedLower())) {
        return null;
      }
      String secondPart = second.identifierValue();
      if (secondPart == null) {
        return null;
      }
      normalized = firstPart + "." + secondPart;
      end = second.end();
      next = next + 2;
      if (next + 1 < tokens.size() && tokens.get(next).symbol(".")) {
        SqlToken third = tokens.get(next + 1);
        if (!third.word() && !third.quotedIdentifier()) {
          return null;
        }
        if (isReservedWord(third.normalizedLower())) {
          return null;
        }
        String thirdPart = third.identifierValue();
        if (thirdPart == null) {
          return null;
        }
        normalized = normalized + "." + thirdPart;
        end = third.end();
        next = next + 2;
        if (next + 1 < tokens.size() && tokens.get(next).symbol(".")) {
          if (strict) {
            int rawEnd = tokens.get(next).end();
            if (next + 1 < tokens.size()) {
              SqlToken extra = tokens.get(next + 1);
              if (extra.word() || extra.quotedIdentifier()) {
                rawEnd = extra.end();
              }
            }
            String rawToken = sql.substring(first.start(), rawEnd);
            throw unsupportedSql(null, first.start(), "identifier with more than 3 parts", rawToken);
          }
          return null;
        }
      }
    }
    String raw = sql.substring(first.start(), end);
    return new ParsedIdentifier(raw, normalized, first.start(), end, next);
  }

  private IllegalArgumentException unknownTableToken(String token) {
    return new IllegalArgumentException(
        "Unknown table token '"
            + token
            + "' (searched group="
            + sessionGroup
            + " defaultNamespace="
            + defaultNamespace
            + ")");
  }

  private static String normalizeToken(String token) {
    String raw = Strings.requireNonBlank(token, "token");
    ParsedIdentifier parsed = parseIdentifier(raw, tokenize(raw), 0);
    if (parsed == null) {
      return raw;
    }
    return parsed.normalizedToken();
  }

  private static boolean isReservedWord(String tokenLower) {
    if (tokenLower == null) {
      return true;
    }
    if (TABLE_CLAUSE_KEYWORDS.contains(tokenLower) || STOP_KEYWORDS.contains(tokenLower)) {
      return true;
    }
    return switch (tokenLower) {
      case "select",
          "with",
          "left",
          "right",
          "full",
          "inner",
          "outer",
          "cross",
          "lateral",
          "as",
          "and",
          "or",
          "not" -> true;
      default -> false;
    };
  }

  private static IllegalArgumentException unsupportedSql(
      String keyword, int offset, String reason, String tokenOrNull) {
    StringBuilder message =
        new StringBuilder("Unsupported SQL for entitled queries: ").append(reason);
    if (tokenOrNull != null) {
      message.append(" '").append(tokenOrNull).append("'");
    }
    if (keyword != null) {
      message.append(" after ").append(keyword.toUpperCase(Locale.ROOT));
    }
    message.append(" at offset=").append(Math.max(offset, 0));
    return new IllegalArgumentException(message.toString());
  }

  private static List<SqlToken> tokenize(String sql) {
    List<SqlToken> tokens = new ArrayList<>();
    int index = 0;
    while (index < sql.length()) {
      char ch = sql.charAt(index);
      if (Character.isWhitespace(ch)) {
        index++;
        continue;
      }
      if (ch == '-' && index + 1 < sql.length() && sql.charAt(index + 1) == '-') {
        index += 2;
        while (index < sql.length() && sql.charAt(index) != '\n') {
          index++;
        }
        continue;
      }
      if (ch == '/' && index + 1 < sql.length() && sql.charAt(index + 1) == '*') {
        index += 2;
        while (index + 1 < sql.length()) {
          if (sql.charAt(index) == '*' && sql.charAt(index + 1) == '/') {
            index += 2;
            break;
          }
          index++;
        }
        continue;
      }
      if (ch == '\'') {
        index++;
        while (index < sql.length()) {
          if (sql.charAt(index) == '\'') {
            if (index + 1 < sql.length() && sql.charAt(index + 1) == '\'') {
              index += 2;
              continue;
            }
            index++;
            break;
          }
          index++;
        }
        continue;
      }
      if (ch == '"' || ch == '`' || ch == '[') {
        int start = index;
        ParsedQuotedIdentifier quoted = parseQuotedIdentifier(sql, index);
        if (quoted == null) {
          index++;
          continue;
        }
        index = quoted.end();
        tokens.add(
            new SqlToken(
                sql.substring(start, quoted.end()),
                quoted.identifier(),
                quoted.identifier().toLowerCase(Locale.ROOT),
                start,
                quoted.end(),
                false,
                true));
        continue;
      }
      if (isIdentifierStart(ch)) {
        int start = index;
        index++;
        while (index < sql.length() && isIdentifierPart(sql.charAt(index))) {
          index++;
        }
        String text = sql.substring(start, index);
        tokens.add(new SqlToken(text, text, text.toLowerCase(Locale.ROOT), start, index, true, false));
        continue;
      }
      tokens.add(
          new SqlToken(
              String.valueOf(ch),
              String.valueOf(ch),
              String.valueOf(ch).toLowerCase(Locale.ROOT),
              index,
              index + 1,
              false,
              false));
      index++;
    }
    return List.copyOf(tokens);
  }

  private static ParsedQuotedIdentifier parseQuotedIdentifier(String sql, int start) {
    if (start >= sql.length()) {
      return null;
    }
    char open = sql.charAt(start);
    char close = open == '[' ? ']' : open;
    int index = start + 1;
    StringBuilder value = new StringBuilder();
    while (index < sql.length()) {
      char current = sql.charAt(index);
      if (current == close) {
        if (open != '['
            && index + 1 < sql.length()
            && sql.charAt(index + 1) == close
            && close != '`') {
          value.append(close);
          index += 2;
          continue;
        }
        return new ParsedQuotedIdentifier(value.toString(), index + 1);
      }
      value.append(current);
      index++;
    }
    return null;
  }

  private static boolean isIdentifierStart(char ch) {
    return Character.isLetter(ch) || ch == '_' || ch == '$';
  }

  private static boolean isIdentifierPart(char ch) {
    return Character.isLetterOrDigit(ch) || ch == '_' || ch == '$';
  }

  private record ParsedQuotedIdentifier(String identifier, int end) {}

  private record ParsedIdentifier(
      String rawToken,
      String normalizedToken,
      int startOffset,
      int endOffset,
      int nextIndex) {}

  private record AliasReference(
      String aliasToken,
      boolean usesAsKeyword,
      int endOffset,
      int nextIndex) {}

  private record SqlToken(
      String text,
      String normalized,
      String normalizedLower,
      int start,
      int end,
      boolean word,
      boolean quotedIdentifier) {
    private boolean symbol(String candidate) {
      return !word && !quotedIdentifier && Objects.equals(text, candidate);
    }

    private String identifierValue() {
      if (word) {
        return normalized;
      }
      if (quotedIdentifier) {
        return normalized;
      }
      return null;
    }
  }
}
