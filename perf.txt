  private IngestManagerImplDataset() {}

  static void register(IngestManagerImpl manager, IngestModel model) {
    IngestModel normalized = Objects.requireNonNull(model, "model");
    if (normalized.target() == null) {
      throw new IllegalArgumentException("ingest target is required");
    }
    String ingestName = Strings.requireNonBlank(normalized.name(), "model.name");
    IngestManagerImplAppend.flushPendingParquetBuffer(manager, ingestName, "register-update");
    applyRuntimeParquetRollPolicyFromModel(manager, ingestName, normalized.target());
    String namespace = Strings.requireNonBlank(normalized.namespace(), "model.namespace");
    SchemaModel schema = Objects.requireNonNull(normalized.schema(), "model.schema");
    validateTargetSchemaReferences(ingestName, normalized.target(), schema);
    IngestManagerImpl.ActiveTableDescriptor activeDescriptor = null;
    if (normalized.target() instanceof IngestTargetDuckDbActiveModel duckDbActiveTarget) {
      activeDescriptor = buildActiveTableDescriptor(manager, ingestName, schema, duckDbActiveTarget);
    }
    String schemaName = manager.modelMapper.deterministicSchemaName(manager.group, namespace, ingestName);
    SchemaModel schemaToRegister =
        schema.toBuilder()
            .withName(schemaName)
            .withType(SchemaType.TABULAR)
            .withCompatibility(
                schema.getCompatibility() == null
                    ? SchemaCompatibility.BACKWARDS
                    : schema.getCompatibility())
            .build();
    manager.schemaManager.register(schemaName, SchemaType.TABULAR, schemaToRegister);
    DefinitionIngest ingestDefinition =
        manager.modelMapper.toDefinition(normalized, manager.group, IngestManagerImpl.DEFAULT_ACTOR, schemaName);
    DefinitionIngestSource sourceDefinition =
        manager.modelMapper.toDefinition(
            Objects.requireNonNull(normalized.source(), "model.source"),
            manager.group,
            IngestManagerImpl.DEFAULT_ACTOR);
    String parquetUriOverride = inferParquetUri(manager, normalized, normalized.target());
    DefinitionDataset datasetDefinition =
        manager.modelMapper.toDefinitionDataset(
            normalized, manager.group, IngestManagerImpl.DEFAULT_ACTOR, parquetUriOverride);
    try (org.pronzato.fabric.api.control.discovery.cache.DiscoveryCache.ConfigCacheTx tx =
        manager.DiscoveryCache.beginTx("ingest:" + manager.group + "/" + ingestName)) {
      if (datasetDefinition != null) {
        manager.DiscoveryCache.upsertTx(tx, datasetDefinition);
        NamespaceEntitlementDefaults.ensureDefaultNamespaceAccess(
            manager.DiscoveryCache, tx, datasetDefinition, IngestManagerImpl.DEFAULT_ACTOR);
      }
      manager.DiscoveryCache.upsertTx(tx, ingestDefinition);
      manager.DiscoveryCache.upsertTx(tx, sourceDefinition);
      manager.DiscoveryCache.commitTx(tx);
    }
    if (activeDescriptor != null) {
      manager.activeTableDescriptors.put(ingestName, activeDescriptor);
    } else {
      manager.activeTableDescriptors.remove(ingestName);
    }
  }

  static void deleteDatasetDefinition(IngestManagerImpl manager, String datasetId) {
    DatasetId parsed = DatasetId.parse(Strings.requireNonBlank(datasetId, "datasetId"));
    org.pronzato.fabric.api.control.discovery.lifecycle.DeleteResult result =
        manager.deleteManager.delete(
            parsed.toString(),
            new org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOptions(
                false, false, false, false, false, false, true, false, List.of()));
    if (result.outcome() == org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOutcome.BLOCKED) {
      throw new IllegalStateException(
          "Delete blocked for dataset definition " + parsed + ": " + result.blockers());
    }
    if (result.outcome() == org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOutcome.PARTIAL) {
      throw new IllegalStateException(
          "Delete partially completed for dataset definition " + parsed + ": " + result.warnings());
    }
  }

  static void applyRuntimeParquetRollPolicyFromModel(
      IngestManagerImpl manager, String ingestName, IngestTargetModel target) {
    if (!(target instanceof IngestTargetParquetModel parquetTarget)) {
      manager.parquetRollPolicies.remove(ingestName);
      return;
    }
    Integer maxRowsPerFile = parquetTarget.maxRowsPerFile();
    Long maxIdleMillis = parquetTarget.maxIdleMillis();
    if (maxRowsPerFile == null && maxIdleMillis == null) {
      return;
    }
    IngestManagerImpl.ParquetRollPolicy policy =
        new IngestManagerImpl.ParquetRollPolicy(
            IngestManagerImpl.resolveMaxRowsPerFile(maxRowsPerFile),
            IngestManagerImpl.resolveMaxIdleMillis(maxIdleMillis));
    manager.parquetRollPolicies.put(ingestName, policy);
  }

  static String inferParquetUri(IngestManagerImpl manager, IngestModel model, IngestTargetModel target) {
    if (!(target instanceof IngestTargetParquetModel parquet)) {
      return null;
    }
    String parquetPath = Strings.trimToNull(parquet.parquetPath());
    String connectionName = Strings.trimToNull(parquet.connectionName());
    if (connectionName == null) {
      return Strings.requireNonBlank(parquetPath, "target.parquet.parquetPath");
    }
    if (parquetPath != null) {
      throw new IllegalArgumentException(
          "target.parquet.parquetPath is not supported when target.parquet.connectionName is set; "
              + "managed S3 path is derived as /fabric/data/{group}/{datasetName}");
    }
    DatasetId datasetId =
        manager.modelMapper.resolveDatasetId(
            manager.group,
            Strings.requireNonBlank(model == null ? null : model.namespace(), "model.namespace"),
            Objects.requireNonNull(model, "model"));
    parquetPath = datasetId.name();
    IngestManagerImpl.S3PathRoot root = resolveS3PathRoot(manager, connectionName);
    String normalizedClientPath = normalizeS3ClientPath(parquetPath);
    String resolvedClientPath = stripManagedS3Prefix(manager, normalizedClientPath);
    if (Strings.trimToNull(resolvedClientPath) == null) {
      throw new IllegalArgumentException(
          "target.parquet.parquetPath must include a dataset path after managed prefix "
              + managedS3Prefix(manager));
    }
    String prefix = joinPath(managedS3Prefix(manager), resolvedClientPath);
    return "s3://" + root.bucket() + (prefix == null ? "" : "/" + prefix);
  }

  static IngestManagerImpl.S3PathRoot resolveS3PathRoot(IngestManagerImpl manager, String connectionName) {
    String normalizedName = Strings.requireNonBlank(connectionName, "target.parquet.connectionName");
    DefinitionConnection connection = resolveConnection(manager, normalizedName);
    if (connection.connectionType() != DefinitionConnection.ConnectionType.S3) {
      throw new IllegalArgumentException(
          "Connection '" + normalizedName + "' is not S3 (type=" + connection.connectionType() + ")");
    }
    DefinitionConnection.S3 s3 = connection.s3();
    String bucket = Strings.trimToNull(s3 == null ? null : s3.bucket());
    String basePrefix = Strings.trimToNull(s3 == null ? null : s3.basePrefix());
    if (bucket != null) {
      return new IngestManagerImpl.S3PathRoot(bucket, basePrefix);
    }
    String physicalRef =
        connection.client() == null ? null : Strings.trimToNull(connection.client().physicalRef());
    if (physicalRef != null) {
      IngestManagerImpl.ScopedConnectionRef ref = parseScopedConnectionRef(physicalRef);
      DefinitionConnection physicalConnection = resolveConnection(manager, ref.group(), ref.name());
      if (physicalConnection.connectionType() != DefinitionConnection.ConnectionType.S3
          || physicalConnection.s3() == null) {
        throw new IllegalArgumentException(
            "Connection '" + normalizedName + "' physicalRef is not an S3 connection: " + physicalRef);
      }
      String fallbackBucket = Strings.trimToNull(physicalConnection.s3().bucket());
      String fallbackPrefix = Strings.trimToNull(physicalConnection.s3().basePrefix());
      return new IngestManagerImpl.S3PathRoot(
          Strings.requireNonBlank(fallbackBucket, "s3.bucket"),
          Strings.firstNonBlank(basePrefix, fallbackPrefix));
    }
    throw new IllegalArgumentException(
        "Connection '" + normalizedName + "' is missing S3 bucket information");
  }

  static DefinitionConnection resolveConnection(IngestManagerImpl manager, String connectionName) {
    Optional<DefinitionConnection> inGroup =
        manager.DiscoveryCache.get(DefinitionConnection.class, manager.group, connectionName);
    if (inGroup.isPresent()) {
      return inGroup.get();
    }
    return manager.DiscoveryCache
        .get(DefinitionConnection.class, IngestManagerImpl.PLATFORM_GROUP, connectionName)
        .orElseThrow(
            () ->
                new IllegalArgumentException(
                    "Connection '"
                        + manager.group
                        + "/"
                        + connectionName
                        + "' or '"
                        + IngestManagerImpl.PLATFORM_GROUP
                        + "/"
                        + connectionName
                        + "' was not found"));
  }

  static DefinitionConnection resolveConnection(
      IngestManagerImpl manager, String groupName, String connectionName) {
    return manager.DiscoveryCache
        .get(DefinitionConnection.class, groupName, connectionName)
        .orElseThrow(
            () ->
                new IllegalArgumentException(
                    "Connection '" + groupName + "/" + connectionName + "' was not found"));
  }

  static IngestManagerImpl.ScopedConnectionRef parseScopedConnectionRef(String scopedRef) {
    String normalized = Strings.requireNonBlank(scopedRef, "connection client.physicalRef");
    int slash = normalized.indexOf('/');
    if (slash < 0) {
      return new IngestManagerImpl.ScopedConnectionRef(IngestManagerImpl.PLATFORM_GROUP, normalized);
    }
    if (slash == 0 || slash == normalized.length() - 1) {
      throw new IllegalArgumentException("Invalid connection reference: " + normalized);
    }
    return new IngestManagerImpl.ScopedConnectionRef(
        normalized.substring(0, slash), normalized.substring(slash + 1));
  }

  static String normalizeS3ClientPath(String parquetPath) {
    String normalized = Strings.requireNonBlank(parquetPath, "target.parquet.parquetPath");
    if (normalized.regionMatches(true, 0, "s3://", 0, 5)) {
      throw new IllegalArgumentException("target.parquet.parquetPath must be relative, not an s3:// URI");
    }
    String candidate = normalized.replace('\\', '/');
    while (candidate.startsWith("/")) {
      candidate = candidate.substring(1);
    }
    String[] segments = candidate.split("/");
    List<String> sanitized = new ArrayList<>(segments.length);
    for (String segment : segments) {
      String trimmed = Strings.trimToNull(segment);
      if (trimmed == null) {
        continue;
      }
      if (".".equals(trimmed) || "..".equals(trimmed)) {
        throw new IllegalArgumentException(
            "target.parquet.parquetPath contains unsupported segment: " + trimmed);
      }
      if (trimmed.indexOf(':') >= 0) {
        throw new IllegalArgumentException("target.parquet.parquetPath must not contain ':' segments");
      }
      sanitized.add(trimmed);
    }
    if (sanitized.isEmpty()) {
      throw new IllegalArgumentException(
          "target.parquet.parquetPath must include at least one path segment");
    }
    return String.join("/", sanitized);
  }

  static String stripManagedS3Prefix(IngestManagerImpl manager, String parquetPath) {
    String normalized = normalizePathToken(parquetPath);
    if (normalized == null) {
      return null;
    }
    String managed = managedS3Prefix(manager);
    if (startsWithPathPrefixIgnoreCase(normalized, managed)) {
      return trimLeadingPathSeparator(normalized.substring(managed.length()));
    }
    String groupOnly = normalizedGroupToken(manager);
    if (startsWithPathPrefixIgnoreCase(normalized, groupOnly)) {
      return trimLeadingPathSeparator(normalized.substring(groupOnly.length()));
    }
    return normalized;
  }

  static String toClientVisibleS3Path(
      IngestManagerImpl manager, String parquetPath, String connectionName) {
    String connection = Strings.trimToNull(connectionName);
    String value = Strings.trimToNull(parquetPath);
    if (connection == null || value == null) {
      return parquetPath;
    }
    String normalized = value;
    if (normalized.regionMatches(true, 0, "s3://", 0, 5)) {
      String withoutScheme = normalized.substring(5);
      int slash = withoutScheme.indexOf('/');
      if (slash >= 0 && slash + 1 < withoutScheme.length()) {
        normalized = withoutScheme.substring(slash + 1);
      } else {
        return parquetPath;
      }
    } else if (normalized.contains("://")) {
      return parquetPath;
    }
    String trimmed = stripManagedS3Prefix(manager, normalized);
    return Strings.trimToNull(trimmed) == null ? parquetPath : trimmed;
  }

  static String managedS3Prefix(IngestManagerImpl manager) {
    return joinPath(IngestManagerImpl.S3_MANAGED_ROOT, normalizedGroupToken(manager));
  }

  static String normalizedGroupToken(IngestManagerImpl manager) {
    return Strings.requireNonBlank(manager.group, "session group").toLowerCase(Locale.ROOT);
  }

  static boolean startsWithPathPrefixIgnoreCase(String path, String prefix) {
    String normalizedPath = normalizePathToken(path);
    String normalizedPrefix = normalizePathToken(prefix);
    if (normalizedPath == null || normalizedPrefix == null) {
      return false;
    }
    if (normalizedPath.equalsIgnoreCase(normalizedPrefix)) {
      return true;
    }
    return normalizedPath.length() > normalizedPrefix.length()
        && normalizedPath.regionMatches(true, 0, normalizedPrefix, 0, normalizedPrefix.length())
        && normalizedPath.charAt(normalizedPrefix.length()) == '/';
  }

  static String normalizePathToken(String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      return null;
    }
    String candidate = normalized.replace('\\', '/');
    while (candidate.startsWith("/")) {
      candidate = candidate.substring(1);
    }
    while (candidate.endsWith("/")) {
      candidate = candidate.substring(0, candidate.length() - 1);
    }
    return Strings.trimToNull(candidate);
  }

  static String trimLeadingPathSeparator(String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      return null;
    }
    String out = normalized;
    while (out.startsWith("/")) {
      out = out.substring(1);
    }
    return Strings.trimToNull(out);
  }

  static String joinPath(String... parts) {
    List<String> normalized = new ArrayList<>();
    for (String part : parts) {
      String trimmed = Strings.trimToNull(part);
      if (trimmed == null) {
        continue;
      }
      String candidate = trimmed.replace('\\', '/');
      while (candidate.startsWith("/")) {
        candidate = candidate.substring(1);
      }
      while (candidate.endsWith("/")) {
        candidate = candidate.substring(0, candidate.length() - 1);
      }
      String normalizedPart = Strings.trimToNull(candidate);
      if (normalizedPart != null) {
        normalized.add(normalizedPart);
      }
    }
    return normalized.isEmpty() ? null : String.join("/", normalized);
  }

  static DefinitionIngest resolveIngestDefinition(IngestManagerImpl manager, String ingestName) {
    Optional<DefinitionIngest> definition =
        manager.DiscoveryCache.get(DefinitionIngest.class, manager.group, ingestName);
    if (definition.isEmpty()) {
      throw new IllegalArgumentException("Unknown ingest: " + manager.group + "/" + ingestName);
    }
    return definition.get();
  }

  static SchemaModel resolveSchemaModel(IngestManagerImpl manager, DefinitionIngest ingest) {
    Objects.requireNonNull(ingest, "ingest");
    String schemaName = Strings.requireNonBlank(ingest.outputSchemaId(), "ingest.outputSchemaId");
    String ingestName =
        Strings.requireNonBlank(
            ingest.envelope() == null ? null : ingest.envelope().name(), "ingest.envelope.name");
    return manager.schemaManager
        .get(manager.group, schemaName)
        .orElseThrow(
            () ->
                new IllegalStateException(
                    "Schema not found for ingest="
                        + manager.group
                        + "/"
                        + ingestName
                        + " schema="
                        + schemaName));
  }

  static DefinitionDataset resolveDatasetDefinition(
      IngestManagerImpl manager, String datasetCanonicalId) {
    String datasetIdText = Strings.trimToNull(datasetCanonicalId);
    if (datasetIdText == null) {
      return null;
    }
    DatasetId datasetId;
    try {
      datasetId = DatasetId.parse(datasetIdText);
    } catch (RuntimeException ex) {
      return null;
    }
    try {
      return new DatasetResolver(manager.DiscoveryCache).resolveDefinition(datasetId);
    } catch (RuntimeException ex) {
      return null;
    }
  }

  static void prepareActiveDataset(
      IngestManagerImpl manager, DefinitionIngest ingestDefinition, DatasetId datasetId) {
    if (!(manager.rowManager instanceof RowManagerImpl localDataManager)) {
      return;
    }
    DefinitionDataset dataset = resolveDatasetDefinition(manager, ingestDefinition.datasetId());
    if (dataset == null || dataset.binding() == null || dataset.binding().duckdb() == null) {
      return;
    }
    String targetType = Strings.trimToNull(dataset.binding().duckdb().targetType());
    if (!"DUCKDB_ACTIVE".equalsIgnoreCase(Strings.firstNonBlank(targetType, ""))) {
      return;
    }
    localDataManager.prepareDuckDbActive(datasetId);
  }

  static IngestManagerImpl.ActiveTableDescriptor buildActiveTableDescriptor(
      IngestManagerImpl manager,
      String ingestName,
      SchemaModel schema,
      IngestTargetDuckDbActiveModel target) {
    String tableName = Strings.requireNonBlank(target.tableName(), "target.duckdbActive.tableName");
    List<String> primaryKey = target.primaryKey();
    if (primaryKey == null || primaryKey.isEmpty()) {
      throw new IllegalArgumentException("target.duckdbActive.primaryKey is required");
    }
    List<String> schemaFields =
        schema.getFields() == null
            ? List.of()
            : schema.getFields().stream()
                .map(SchemaModel.Field::getName)
                .filter(Objects::nonNull)
                .toList();
    for (String key : primaryKey) {
      if (!schemaFields.contains(key)) {
        throw new IllegalArgumentException(
            "target.duckdbActive.primaryKey contains unknown field '"
                + key
                + "' for ingest "
                + ingestName);
      }
    }
    return new IngestManagerImpl.ActiveTableDescriptor(
        ingestName,
        tableName,
        List.copyOf(primaryKey),
        target.memory(),
        target.databasePath(),
        target.parquetHistoryPath(),
        target.persistOnStop(),
        target.createIfMissing(),
        schemaFields);
  }

  static void validateTargetSchemaReferences(
      String ingestName, IngestTargetModel target, SchemaModel schema) {
    if (!(target instanceof IngestTargetParquetModel parquetTarget)) {
      return;
    }
    String partitionValueField = Strings.trimToNull(parquetTarget.partitionValueField());
    if (partitionValueField == null) {
      return;
    }
    Set<String> schemaFields = new HashSet<>();
    if (schema.getFields() != null) {
      for (SchemaModel.Field field : schema.getFields()) {
        if (field == null) {
          continue;
        }
        String fieldName = Strings.trimToNull(field.getName());
        if (fieldName == null) {
          continue;
        }
        schemaFields.add(fieldName.toLowerCase(Locale.ROOT));
      }
    }
    if (!schemaFields.contains(partitionValueField.toLowerCase(Locale.ROOT))) {
      throw new IllegalArgumentException(
          "target.parquet.partitionValueField contains unknown field '"
              + partitionValueField
              + "' for ingest "
              + ingestName);
    }
  }
}
