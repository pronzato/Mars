public record AgentLimits(
    int maxSteps,
    int maxToolCalls,
    int maxAssetsInspected,
    int maxRelatedDepth,
    int maxDatasetQueries) {

  public static final int DEFAULT_MAX_STEPS = 8;
  public static final int DEFAULT_MAX_TOOL_CALLS = 8;
  public static final int DEFAULT_MAX_ASSETS_INSPECTED = 5;
  public static final int DEFAULT_MAX_RELATED_DEPTH = 1;
  public static final int DEFAULT_MAX_DATASET_QUERIES = 1;

  public static final int HARD_MAX_STEPS = 20;
  public static final int HARD_MAX_TOOL_CALLS = 20;
  public static final int HARD_MAX_ASSETS_INSPECTED = 20;
  public static final int HARD_MAX_RELATED_DEPTH = 3;
  public static final int HARD_MAX_DATASET_QUERIES = 3;

  public AgentLimits {
    maxSteps = clamp(maxSteps, DEFAULT_MAX_STEPS, HARD_MAX_STEPS);
    maxToolCalls = clamp(maxToolCalls, DEFAULT_MAX_TOOL_CALLS, HARD_MAX_TOOL_CALLS);
    maxAssetsInspected =
        clamp(maxAssetsInspected, DEFAULT_MAX_ASSETS_INSPECTED, HARD_MAX_ASSETS_INSPECTED);
    maxRelatedDepth = clamp(maxRelatedDepth, DEFAULT_MAX_RELATED_DEPTH, HARD_MAX_RELATED_DEPTH);
    maxDatasetQueries =
        clamp(maxDatasetQueries, DEFAULT_MAX_DATASET_QUERIES, HARD_MAX_DATASET_QUERIES);
  }

  public static AgentLimits defaults() {
    return new AgentLimits(
        DEFAULT_MAX_STEPS,
        DEFAULT_MAX_TOOL_CALLS,
        DEFAULT_MAX_ASSETS_INSPECTED,
        DEFAULT_MAX_RELATED_DEPTH,
        DEFAULT_MAX_DATASET_QUERIES);
  }

  public AgentLimits merge(AgentLimits overrides) {
    if (overrides == null) {
      return this;
    }
    return new AgentLimits(
        clamp(overrides.maxSteps(), maxSteps, HARD_MAX_STEPS),
        clamp(overrides.maxToolCalls(), maxToolCalls, HARD_MAX_TOOL_CALLS),
        clamp(overrides.maxAssetsInspected(), maxAssetsInspected, HARD_MAX_ASSETS_INSPECTED),
        clamp(overrides.maxRelatedDepth(), maxRelatedDepth, HARD_MAX_RELATED_DEPTH),
        clamp(overrides.maxDatasetQueries(), maxDatasetQueries, HARD_MAX_DATASET_QUERIES));
  }

  private static int clamp(int value, int fallback, int max) {
    int candidate = value <= 0 ? fallback : value;
    return Math.max(1, Math.min(max, candidate));
  }
}

    if (value == null) {

    if (normalized.isEmpty()) {

    } catch (IllegalArgumentException ex) {
      return AUTO;
    }
  }

  private static AgentResponse buildResponse(

      String subject,

      AgentLimits limits,
      AgentReasonerResult result) {

    long durationMs = Math.max(0L, System.currentTimeMillis() - startedMs);
    AgentReasonerResult resolved =
        result == null
            ? new AgentReasonerResult(

                "Agent execution failed.",

            lane,
            selectedMode,
            scenarioId,
            resolved.interpretedIntent(),
            limits,
            resolved.candidateAssetIds(),
            resolved.selectedAssetIds(),
            resolved.deniedAssetIds(),
            resolved.filteredAssetIds(),
            resolved.answerAssembly(),
            startedAt,
            finishedAt,
            durationMs,
            resolved.steps().size(),
            resolved.toolCalls().size(),
            resolved.steps(),
            resolved.toolCalls());
    AgentAnswer answer =
        new AgentAnswer(
            resolved.answerText(), resolved.usedAssets(), resolved.datasetAssets(), resolved.evidence());
    List<String> usedAssets =
        resolved.usedAssets() == null ? List.of() : List.copyOf(new ArrayList<>(resolved.usedAssets()));
    List<String> warnings =
        resolved.warnings() == null ? List.of() : List.copyOf(new ArrayList<>(resolved.warnings()));
    List<String> errors =

    AgentStatus status = resolved.status() == null ? AgentStatus.FAILED : resolved.status();
    LOGGER.info(
        "[AgentRuntime] requestId={} subject={} lane={} mode={} scenario={} status={} steps={} toolCalls={} warnings={} errors={} durationMs={}",
        requestId,

    return new AgentResponse(requestId, status, answer, usedAssets, warnings, errors, trace);
  }

  private AgentReasoner selectReasoner(AgentReasonerMode mode) {
    if (mode == AgentReasonerMode.SCRIPTED) {
      return scriptedReasoner;
    }
    if (mode == AgentReasonerMode.MODEL) {
      return modelReasoner;
    }
    return deterministicReasoner;
  }

  private static AgentReasonerMode selectMode(AgentReasonerMode requestedMode, String scenarioId) {
    AgentReasonerMode mode = requestedMode == null ? AgentReasonerMode.AUTO : requestedMode;
    if (mode == AgentReasonerMode.AUTO) {
      return scenarioId == null ? AgentReasonerMode.DETERMINISTIC : AgentReasonerMode.SCRIPTED;
    }
    return mode;
  }

  private static boolean containsForbiddenIntent(String prompt) {
    String normalized = normalize(prompt);
    if (normalized == null) {
      return false;
    }
    String lower = normalized.toLowerCase(Locale.ROOT);
    for (String token : FORBIDDEN_INTENT_TOKENS) {
      if (containsToken(lower, token)) {
        return true;
      }
    }
    return false;
  }

  private static boolean containsToken(String text, String token) {
    int index = text.indexOf(token);
    while (index >= 0) {
      boolean leftBoundary = index == 0 || !Character.isLetterOrDigit(text.charAt(index - 1));
      int rightIndex = index + token.length();
      boolean rightBoundary =
          rightIndex >= text.length() || !Character.isLetterOrDigit(text.charAt(rightIndex));
      if (leftBoundary && rightBoundary) {
        return true;
      }
      index = text.indexOf(token, index + 1);
    }
    return false;
  }

  private static String normalize(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }

  static boolean canRunStep(int currentStepNumber, AgentLimits limits, List<String> warnings) {
    if (currentStepNumber < limits.maxSteps()) {
      return true;
    }
    warnings.add("Agent max step limit reached: " + limits.maxSteps());
    return false;
  }

  static boolean canRunTool(List<AgentToolCall> toolCalls, AgentLimits limits, List<String> warnings) {
    if (toolCalls.size() < limits.maxToolCalls()) {
      return true;
    }
    warnings.add("Agent max tool call limit reached: " + limits.maxToolCalls());
    return false;
  }

  static AgentStatus deriveStatus(
      List<AgentToolCall> toolCalls,
      List<String> warnings,
      List<String> errors,
      Map<String, CatalogDetails> detailsByAsset,
      QueryDatasetResponse datasetResponse) {
    if (!errors.isEmpty() && detailsByAsset.isEmpty() && !querySucceeded(datasetResponse)) {
      boolean onlyDeniedOrMissing = true;
      for (AgentToolCall toolCall : toolCalls) {
        if (toolCall == null) {
          continue;
        }
        ToolStatus status = toolCall.status();
        if (status != ToolStatus.DENIED && status != ToolStatus.NOT_FOUND) {
          onlyDeniedOrMissing = false;
          break;
        }
      }
      return onlyDeniedOrMissing ? AgentStatus.DENIED : AgentStatus.FAILED;
    }
    if (!errors.isEmpty() || !warnings.isEmpty()) {
      return AgentStatus.PARTIAL;
    }
    return AgentStatus.COMPLETED;
  }

  static void recordToolIssue(
      ToolStatus status, String message, List<String> warnings, List<String> errors) {
    String resolved = Strings.firstNonBlank(normalize(message), "Tool call did not succeed.");
    if (status == ToolStatus.ERROR || status == ToolStatus.INVALID_REQUEST) {
      errors.add(resolved);
      return;
    }
    warnings.add(resolved);
  }

  static boolean querySucceeded(QueryDatasetResponse response) {
    return response != null && response.status() == ToolStatus.OK;
  }

  static boolean isDataset(CatalogSummary summary, String assetId) {
    if (summary != null && summary.type() == DefinitionType.DATASET) {
      return true;
    }
    return assetId != null && assetId.startsWith("dataset:");
  }

  static String firstDatasetId(Set<String> datasetAssets) {
    if (datasetAssets == null || datasetAssets.isEmpty()) {
      return null;
    }
    return datasetAssets.iterator().next();
  }

  static String firstAssetId(Map<String, CatalogDetails> detailsByAsset, Set<String> usedAssets) {
    if (!detailsByAsset.isEmpty()) {
      return detailsByAsset.keySet().iterator().next();
    }
    if (usedAssets == null || usedAssets.isEmpty()) {
      return null;
    }
    return usedAssets.iterator().next();
  }

  static List<String> collectRelatedAssetIds(Map<?, List<CatalogRelationshipView>> relationshipsByType) {
    if (relationshipsByType == null || relationshipsByType.isEmpty()) {
      return List.of();
    }
    LinkedHashSet<String> assetIds = new LinkedHashSet<>();
    for (List<CatalogRelationshipView> values : relationshipsByType.values()) {
      if (values == null || values.isEmpty()) {
        continue;
      }
      for (CatalogRelationshipView relationship : values) {
        if (relationship == null) {
          continue;
        }
        String from = normalize(relationship.fromAssetId());
        String to = normalize(relationship.toAssetId());
        if (from != null) {
          assetIds.add(from);
        }
        if (to != null) {
          assetIds.add(to);
        }
      }
    }
    return List.copyOf(assetIds);
  }

  static String normalize(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }

      if (!candidate.isEmpty() && lowerPrompt.contains(candidate)) {
        return true;
      }
    }
    return false;
  }

                "traverse_related_assets",
                AgentStepStatus.COMPLETED,
                "Collected " + relatedResponse.relationshipCount() + " relationships.",
                "get_related_assets",
                toolRequestId,
                relatedIds));
      } else {
        if (relatedResponse.status() == ToolStatus.DENIED) {
          deniedAssets.add(relatedRoot);
        }
        AgentReasonerSupport.recordToolIssue(
            relatedResponse.status(), relatedResponse.message(), warnings, errors);
        steps.add(
            new AgentPlanStep(
                stepNumber,
                "traverse_related_assets",
                relatedResponse.status() == ToolStatus.ERROR
                    ? AgentStepStatus.FAILED
                    : AgentStepStatus.SKIPPED,
                Strings.firstNonBlank(relatedResponse.message(), "Relationship traversal failed"),
                "get_related_assets",
                toolRequestId,
                List.of(relatedRoot)));
      }
    }

    String queryDatasetId =
        Strings.firstNonBlank(
            AgentReasonerSupport.normalize(resolved.datasetId()),
            AgentReasonerSupport.firstDatasetId(datasetAssets));
    if (queryDatasetId != null
        && limits.maxDatasetQueries() > 0
        && AgentReasonerSupport.canRunStep(stepNumber, limits, warnings)
        && AgentReasonerSupport.canRunTool(toolCalls, limits, warnings)) {
      stepNumber++;
      String toolRequestId = resolved.requestId() + "-tool-" + (toolCalls.size() + 1);
      long startedTool = System.currentTimeMillis();
      datasetResponse =
          toolGatewayService.queryDataset(
              new QueryDatasetRequest(
                  queryDatasetId,
                  AgentReasonerSupport.normalize(resolved.datasetQuery()),
                  null,
                  null,
                  toolRequestId,
                  resolved.subject(),
                  resolved.delegatedSubject(),
                  resolved.lane()));
      long duration = System.currentTimeMillis() - startedTool;
      toolCalls.add(
          new AgentToolCall(
              toolCalls.size() + 1,
              "query_dataset",
              toolRequestId,
              queryDatasetId,
              queryDatasetId,
              datasetResponse.status(),
              datasetResponse.message(),
              duration));
      if (datasetResponse.status() == ToolStatus.OK) {
        datasetAssets.add(queryDatasetId);
        evidence.add(
            "query_dataset rows="
                + datasetResponse.rowCount()
                + " truncated="
                + datasetResponse.truncated());
        steps.add(
            new AgentPlanStep(
                stepNumber,
                "query_dataset",
                AgentStepStatus.COMPLETED,
                "Queried dataset rows=" + datasetResponse.rowCount(),
                "query_dataset",
                toolRequestId,
                List.of(queryDatasetId)));
      } else {
        if (datasetResponse.status() == ToolStatus.DENIED) {
          deniedAssets.add(queryDatasetId);
        }
        AgentReasonerSupport.recordToolIssue(
            datasetResponse.status(), datasetResponse.message(), warnings, errors);
        steps.add(
            new AgentPlanStep(
                stepNumber,
                "query_dataset",
                datasetResponse.status() == ToolStatus.ERROR
                    ? AgentStepStatus.FAILED
                    : AgentStepStatus.SKIPPED,
                Strings.firstNonBlank(datasetResponse.message(), "Dataset query failed"),
                "query_dataset",
                toolRequestId,
                List.of(queryDatasetId)));
      }
    }

    answerAssembly.add("resolved_assets=" + detailsByAsset.size());
    answerAssembly.add("related_traversal=" + (relatedResponse != null));
    answerAssembly.add("dataset_query_attempted=" + (datasetResponse != null));
    String answerText =
        buildAnswerText(
            resolved.prompt(),
            detailsByAsset,
            relatedResponse,
            datasetResponse,
            warnings,
            errors,
            limits,
            toolCalls.size(),
            interpretedIntent);
    AgentStatus status =
        AgentReasonerSupport.deriveStatus(toolCalls, warnings, errors, detailsByAsset, datasetResponse);
    return new AgentReasonerResult(
        status,
        interpretedIntent,
        answerText,
        new ArrayList<>(usedAssets),
        new ArrayList<>(datasetAssets),
        evidence,
        warnings,
        errors,
        new ArrayList<>(candidateAssets),
        new ArrayList<>(selectedAssets),
        new ArrayList<>(deniedAssets),
        new ArrayList<>(filteredAssets),
        answerAssembly,
        steps,
        toolCalls);
  }

  private static String buildAnswerText(
      String prompt,
      Map<String, CatalogDetails> detailsByAsset,
      GetRelatedAssetsResponse relatedResponse,
      QueryDatasetResponse datasetResponse,
      List<String> warnings,
      List<String> errors,
      AgentLimits limits,
      int toolCalls,
      String interpretedIntent) {
    StringBuilder builder = new StringBuilder();
    builder.append("Question: ").append(prompt).append('\n');
    builder.append("Intent: ").append(interpretedIntent).append('\n');
    builder
        .append("Execution: toolCalls=")
        .append(toolCalls)
        .append(", maxSteps=")
        .append(limits.maxSteps())
        .append(", maxToolCalls=")
        .append(limits.maxToolCalls())
        .append('\n');

    if (detailsByAsset.isEmpty()) {
      builder.append("No authorized assets were resolved.\n");
    } else {
      builder.append("Relevant assets:\n");
      for (Map.Entry<String, CatalogDetails> entry : detailsByAsset.entrySet()) {
        CatalogSummary summary = entry.getValue() == null ? null : entry.getValue().summary();
        String title = summary == null ? null : AgentReasonerSupport.normalize(summary.title());
        String type = summary == null || summary.type() == null ? "UNKNOWN" : summary.type().name();
        builder.append("- ").append(entry.getKey()).append(" (").append(type).append(')');
        if (title != null) {
          builder.append(" - ").append(title);
        }
        builder.append('\n');
      }
    }

    if (relatedResponse != null && relatedResponse.status() == ToolStatus.OK) {
      builder
          .append("Relationship traversal returned ")
          .append(relatedResponse.relationshipCount())
          .append(" relationships at depth ")
          .append(relatedResponse.depth())
          .append(".\n");
    }

    if (datasetResponse != null) {
      builder.append("Dataset query status: ").append(datasetResponse.status().name());
      if (datasetResponse.status() == ToolStatus.OK) {
        builder
            .append(" (rows=")
            .append(datasetResponse.rowCount())
            .append(", truncated=")
            .append(datasetResponse.truncated())
            .append(')');
      } else if (AgentReasonerSupport.normalize(datasetResponse.message()) != null) {
        builder.append(" (").append(datasetResponse.message()).append(')');
      }
      builder.append(".\n");
    }

    if (!warnings.isEmpty()) {
      builder.append("Warnings: ").append(warnings.size()).append(".\n");
    }
    if (!errors.isEmpty()) {
      builder.append("Errors: ").append(errors.size()).append(".\n");
    }
    return builder.toString().trim();
  }

  private static String interpretIntent(String prompt, String datasetQuery) {
    String normalizedPrompt = AgentReasonerSupport.normalize(prompt);
    if (normalizedPrompt == null) {
      return "unknown";
    }
    String lower = normalizedPrompt.toLowerCase();
    if (lower.contains("why")) {
      return "explain_reasoning";
    }
    if (lower.contains("entitlement") || lower.contains("access") || lower.contains("visible")) {
      return "visibility_analysis";
    }
    if (lower.contains("dataset") || AgentReasonerSupport.normalize(datasetQuery) != null) {
      return "dataset_discovery_or_query";
    }
    if (lower.contains("related") || lower.contains("depend")) {
      return "relationship_traversal";
    }
    return "semantic_asset_discovery";
  }

  private static AgentReasonerRequest emptyRequest() {
    return new AgentReasonerRequest(
        "agent-unknown",
        null,
        null,
        null,
        null,
        null,
        null,
        null,
        AgentLimits.defaults(),
        AgentReasonerMode.DETERMINISTIC,
        null);
  }
}

  private static void runScenarioFlow(Execution execution) {
    String scenarioId = execution.scenario.id();
    if (AgentScenarioRegistry.RELATED_ASSETS.equals(scenarioId)) {
      runRelatedAssetsScenario(execution);
      return;
    }
    if (AgentScenarioRegistry.EXPLAIN_RELATIONSHIPS.equals(scenarioId)) {
      runExplainRelationshipsScenario(execution);
      return;
    }
    if (AgentScenarioRegistry.DATASET_RECOMMENDATION.equals(scenarioId)) {
      runDatasetRecommendationScenario(execution);
      return;
    }
    if (AgentScenarioRegistry.ENTITLEMENT_VISIBILITY.equals(scenarioId)) {
      runEntitlementVisibilityScenario(execution);
      return;
    }
    runToolChoiceScenario(execution);
  }

  private static void runRelatedAssetsScenario(Execution execution) {
    execution.answerAssembly.add("scripted_step=resolve_seed_or_search");
    execution.resolveSeedAsset();
    if (execution.usedAssets.isEmpty()) {
      execution.searchAssets("search_assets", execution.request.limits().maxAssetsInspected());
    }
    execution.traverseRelated(execution.firstKnownAsset(), execution.request.limits().maxRelatedDepth());
  }

  private static void runExplainRelationshipsScenario(Execution execution) {
    execution.answerAssembly.add("scripted_step=collect_assets_to_explain");
    execution.resolveSeedAsset();
    if (execution.usedAssets.isEmpty()) {
      execution.searchAssets("search_assets", Math.min(3, execution.request.limits().maxAssetsInspected()));
    }
    execution.traverseRelated(execution.firstKnownAsset(), Math.min(2, execution.request.limits().maxRelatedDepth()));
  }

  private static void runDatasetRecommendationScenario(Execution execution) {
    execution.answerAssembly.add("scripted_step=discover_dataset_candidates");
    if (execution.request.datasetId() != null) {
      execution.getAsset(execution.request.datasetId(), "resolve_dataset_seed");
    }
    execution.searchAssets("search_assets", execution.request.limits().maxAssetsInspected());
    execution.traverseRelated(execution.firstKnownAsset(), 1);
  }

  private static void runEntitlementVisibilityScenario(Execution execution) {
    execution.answerAssembly.add("scripted_step=demonstrate_visibility");
    execution.resolveSeedAsset();
    execution.searchAssets("search_assets", Math.min(4, execution.request.limits().maxAssetsInspected()));
    if (!execution.selectedAssets.isEmpty()) {
      int inspected = 0;
      for (String assetId : execution.selectedAssets) {
        execution.getAsset(assetId, "inspect_visibility_asset");
        inspected++;
        if (inspected >= 2) {
          break;
        }
      }
    }
  }

  private static void runToolChoiceScenario(Execution execution) {
    execution.answerAssembly.add("scripted_step=explain_tool_choices");
    execution.searchAssets("search_assets", Math.min(3, execution.request.limits().maxAssetsInspected()));
    execution.traverseRelated(execution.firstKnownAsset(), 1);
  }

    if (execution.request.datasetId() != null || execution.request.datasetQuery() != null) {
      return true;

    return AgentScenarioRegistry.DATASET_RECOMMENDATION.equals(execution.scenario.id());

  private final class Execution {

    private final AgentScenario scenario;
    private final List<AgentPlanStep> steps = new ArrayList<>();
    private final List<AgentToolCall> toolCalls = new ArrayList<>();
    private final List<String> warnings = new ArrayList<>();
    private final List<String> errors = new ArrayList<>();
    private final List<String> evidence = new ArrayList<>();
    private final List<String> answerAssembly = new ArrayList<>();
    private final LinkedHashSet<String> usedAssets = new LinkedHashSet<>();
    private final LinkedHashSet<String> datasetAssets = new LinkedHashSet<>();
    private final LinkedHashSet<String> candidateAssets = new LinkedHashSet<>();
    private final LinkedHashSet<String> selectedAssets = new LinkedHashSet<>();
    private final LinkedHashSet<String> deniedAssets = new LinkedHashSet<>();
    private final LinkedHashSet<String> filteredAssets = new LinkedHashSet<>();
    private final Map<String, CatalogDetails> detailsByAsset = new LinkedHashMap<>();
    private int stepNumber = 0;
    private GetRelatedAssetsResponse relatedResponse;
    private QueryDatasetResponse datasetResponse;

    private void resolveSeedAsset() {
      if (request.seedAssetId() == null) {
        return;
      }
      getAsset(request.seedAssetId(), "resolve_seed_asset");
    }

    private void getAsset(String assetId, String stepName) {
      if (assetId == null) {
        return;
      }
      if (!canRun()) {
        return;
      }
      stepNumber++;
      String toolRequestId = toolRequestId();
      long startedTool = System.currentTimeMillis();
      GetAssetResponse response =
          toolGatewayService.getAsset(
              new GetAssetRequest(
                  assetId,
                  toolRequestId,
                  request.subject(),
                  request.delegatedSubject(),
                  request.lane()));
      long duration = System.currentTimeMillis() - startedTool;
      toolCalls.add(
          new AgentToolCall(
              toolCalls.size() + 1,
              "get_asset",
              toolRequestId,
              assetId,
              null,
              response.status(),
              response.message(),
              duration));
      if (response.status() == ToolStatus.OK && response.asset() != null) {
        CatalogDetails details = response.asset();
        String canonicalId =
            AgentReasonerSupport.normalize(
                details.summary() == null ? response.assetId() : details.summary().assetId());
        String resolvedId = canonicalId == null ? assetId : canonicalId;
        candidateAssets.add(resolvedId);
        selectedAssets.add(resolvedId);
        usedAssets.add(resolvedId);
        detailsByAsset.put(resolvedId, details);
        if (AgentReasonerSupport.isDataset(details.summary(), resolvedId)) {
          datasetAssets.add(resolvedId);
        }
        steps.add(
            new AgentPlanStep(
                stepNumber,
                stepName,
                AgentStepStatus.COMPLETED,
                "Loaded asset details for scenario.",
                "get_asset",
                toolRequestId,
                List.of(resolvedId)));
        return;
      }

      if (response.status() == ToolStatus.DENIED) {
        deniedAssets.add(assetId);
      } else if (response.status() == ToolStatus.NOT_FOUND) {
        filteredAssets.add(assetId);
      }
      AgentReasonerSupport.recordToolIssue(response.status(), response.message(), warnings, errors);
      steps.add(
          new AgentPlanStep(
              stepNumber,
              stepName,
              response.status() == ToolStatus.ERROR ? AgentStepStatus.FAILED : AgentStepStatus.SKIPPED,
              Strings.firstNonBlank(response.message(), "Asset could not be resolved"),
              "get_asset",
              toolRequestId,
              List.of(assetId)));
    }

    private void searchAssets(String stepName, int preferredLimit) {
      if (!canRun()) {
        return;
      }
      stepNumber++;
      int limit = Math.max(1, Math.min(preferredLimit, request.limits().maxAssetsInspected()));
      String toolRequestId = toolRequestId();
      long startedTool = System.currentTimeMillis();
      SearchAssetsResponse response =
          toolGatewayService.searchAssets(
              new SearchAssetsRequest(
                  request.prompt(),
                  null,
                  null,
                  List.of(),
                  null,
                  null,
                  null,
                  limit,
                  toolRequestId,
                  request.subject(),
                  request.delegatedSubject(),
                  request.lane(),
                  request.seedAssetId()));
      long duration = System.currentTimeMillis() - startedTool;
      toolCalls.add(
          new AgentToolCall(
              toolCalls.size() + 1,
              "search_assets",
              toolRequestId,
              null,
              null,
              response.status(),
              response.message(),
              duration));
      if (response.status() == ToolStatus.OK) {
        List<String> selected = new ArrayList<>();
        for (CatalogSummary summary : response.results()) {
          if (summary == null) {
            continue;
          }
          String assetId = AgentReasonerSupport.normalize(summary.assetId());
          if (assetId == null) {
            continue;
          }
          candidateAssets.add(assetId);
          selectedAssets.add(assetId);
          usedAssets.add(assetId);
          selected.add(assetId);
          if (AgentReasonerSupport.isDataset(summary, assetId)) {
            datasetAssets.add(assetId);
          }
          if (selected.size() >= request.limits().maxAssetsInspected()) {
            break;
          }
        }
        if (response.totalCount() > response.results().size()) {
          filteredAssets.add(
              "search_assets:omitted=" + (response.totalCount() - response.results().size()));
        }
        AgentStepStatus status = selected.isEmpty() ? AgentStepStatus.SKIPPED : AgentStepStatus.COMPLETED;
        String summary =
            selected.isEmpty()
                ? "No authorized assets matched scripted search."
                : "Scripted search selected " + selected.size() + " assets.";
        steps.add(
            new AgentPlanStep(
                stepNumber, stepName, status, summary, "search_assets", toolRequestId, selected));
        return;
      }

      AgentReasonerSupport.recordToolIssue(response.status(), response.message(), warnings, errors);
      steps.add(
          new AgentPlanStep(
              stepNumber,
              stepName,
              response.status() == ToolStatus.ERROR ? AgentStepStatus.FAILED : AgentStepStatus.SKIPPED,
              Strings.firstNonBlank(response.message(), "search_assets failed"),
              "search_assets",
              toolRequestId,
              List.of()));
    }

    private void inspectDiscoveredAssets() {
      int inspected = 0;
      for (String assetId : new ArrayList<>(selectedAssets)) {
        if (detailsByAsset.containsKey(assetId)) {
          continue;
        }
        if (inspected >= request.limits().maxAssetsInspected()) {
          break;
        }
        getAsset(assetId, "inspect_asset");
        inspected++;
      }
    }

    private void traverseRelated(String rootAssetId, int depth) {
      if (rootAssetId == null || depth <= 0) {
        return;
      }
      if (!canRun()) {
        return;
      }
      stepNumber++;
      String toolRequestId = toolRequestId();
      int boundedDepth = Math.max(1, Math.min(depth, request.limits().maxRelatedDepth()));
      long startedTool = System.currentTimeMillis();
      relatedResponse =
          toolGatewayService.getRelatedAssets(
              new GetRelatedAssetsRequest(
                  rootAssetId,
                  null,
                  boundedDepth,
                  toolRequestId,
                  request.subject(),
                  request.delegatedSubject(),
                  request.lane()));
      long duration = System.currentTimeMillis() - startedTool;
      toolCalls.add(
          new AgentToolCall(
              toolCalls.size() + 1,
              "get_related_assets",
              toolRequestId,
              rootAssetId,
              null,
              relatedResponse.status(),
              relatedResponse.message(),
              duration));
      if (relatedResponse.status() == ToolStatus.OK) {
        List<String> relatedIds =
            AgentReasonerSupport.collectRelatedAssetIds(relatedResponse.relationshipsByType());
        for (String relatedId : relatedIds) {
          candidateAssets.add(relatedId);
          usedAssets.add(relatedId);
          if (relatedId.startsWith("dataset:")) {
            datasetAssets.add(relatedId);
          }
        }
        steps.add(
            new AgentPlanStep(
                stepNumber,
                "traverse_related_assets",
                AgentStepStatus.COMPLETED,
                "Scripted traversal returned " + relatedResponse.relationshipCount() + " relationships.",
                "get_related_assets",
                toolRequestId,
                relatedIds));
        return;
      }

      if (relatedResponse.status() == ToolStatus.DENIED) {
        deniedAssets.add(rootAssetId);
      }
      AgentReasonerSupport.recordToolIssue(
          relatedResponse.status(), relatedResponse.message(), warnings, errors);
      steps.add(
          new AgentPlanStep(
              stepNumber,
              "traverse_related_assets",
              relatedResponse.status() == ToolStatus.ERROR
                  ? AgentStepStatus.FAILED
                  : AgentStepStatus.SKIPPED,
              Strings.firstNonBlank(relatedResponse.message(), "Relationship traversal failed"),
              "get_related_assets",
              toolRequestId,
              List.of(rootAssetId)));
    }

    private void queryDataset(String datasetId, String query, String stepName) {
      if (datasetId == null || request.limits().maxDatasetQueries() <= 0) {
        return;
      }
      if (!canRun()) {
        return;
      }
      stepNumber++;
      String toolRequestId = toolRequestId();
      long startedTool = System.currentTimeMillis();
      datasetResponse =
          toolGatewayService.queryDataset(
              new QueryDatasetRequest(
                  datasetId,
                  AgentReasonerSupport.normalize(query),
                  null,
                  null,
                  toolRequestId,
                  request.subject(),
                  request.delegatedSubject(),
                  request.lane()));
      long duration = System.currentTimeMillis() - startedTool;
      toolCalls.add(
          new AgentToolCall(
              toolCalls.size() + 1,
              "query_dataset",
              toolRequestId,
              datasetId,
              datasetId,
              datasetResponse.status(),
              datasetResponse.message(),
              duration));
      if (datasetResponse.status() == ToolStatus.OK) {
        datasetAssets.add(datasetId);
        evidence.add(
            "scripted_query_dataset rows="
                + datasetResponse.rowCount()
                + " truncated="
                + datasetResponse.truncated());
        steps.add(
            new AgentPlanStep(
                stepNumber,
                stepName,
                AgentStepStatus.COMPLETED,
                "Queried dataset rows=" + datasetResponse.rowCount(),
                "query_dataset",
                toolRequestId,
                List.of(datasetId)));
        return;
      }

      if (datasetResponse.status() == ToolStatus.DENIED) {
        deniedAssets.add(datasetId);
      }
      AgentReasonerSupport.recordToolIssue(
          datasetResponse.status(), datasetResponse.message(), warnings, errors);
      steps.add(
          new AgentPlanStep(
              stepNumber,
              stepName,
              datasetResponse.status() == ToolStatus.ERROR
                  ? AgentStepStatus.FAILED
                  : AgentStepStatus.SKIPPED,
              Strings.firstNonBlank(datasetResponse.message(), "Dataset query failed"),
              "query_dataset",
              toolRequestId,
              List.of(datasetId)));
    }

    private String firstDatasetId() {
      return AgentReasonerSupport.firstDatasetId(datasetAssets);
    }

    private String firstKnownAsset() {
      return AgentReasonerSupport.firstAssetId(detailsByAsset, usedAssets);
    }

    private String toolRequestId() {
      return request.requestId() + "-tool-" + (toolCalls.size() + 1);
    }

    private boolean canRun() {
      return AgentReasonerSupport.canRunStep(stepNumber, request.limits(), warnings)
          && AgentReasonerSupport.canRunTool(toolCalls, request.limits(), warnings);
    }
  }
}
