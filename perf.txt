  private static final String ODBC = "ODBC";

  private final FabricPlugins plugins;

  public IngestModelMapper() {
    this(FabricPlugins.system());
  }

  public IngestModelMapper(FabricPlugins plugins) {
    this.plugins = Objects.requireNonNull(plugins, "plugins");
  }

  public DefinitionIngest toDefinition(
      IngestModel model, String sessionGroup, String actor, String outputSchemaName) {
    Objects.requireNonNull(model, "model");
    String group = DefinitionModelMapperSupport.requireNonBlank(sessionGroup, "sessionGroup");
    String ingestName = DefinitionModelMapperSupport.requireNonBlank(model.name(), "model.name");
    String namespace =
        DefinitionModelMapperSupport.requireNonBlank(model.namespace(), "model.namespace");
    String resolvedSchemaName =
        DefinitionModelMapperSupport.requireNonBlank(outputSchemaName, "outputSchemaName");
    DatasetId datasetId = resolveDatasetId(group, namespace, model);
    DefinitionEnvelope envelope =
        DefinitionModelMapperSupport.envelopeNow(
            group,
            DefinitionType.INGEST,
            ingestName,
            Strings.firstNonBlank(
                model.uuid(),
                DefinitionModelMapperSupport.buildDefinitionId("ingest", group, ingestName)),
            model.enabled(),
            model.lifecycleStatus(),
            actor);
    return new DefinitionIngest(
        envelope,
        datasetId.toString(),
        resolvedSchemaName,
        new DefinitionIngest.Input(INPUT_TYPE_API_IPC),
        toDefinitionSchemaMode(model.schemaMode()),
        null);
  }

  public DefinitionIngestSource toDefinition(
      IngestSourceModel model, String sessionGroup, String actor) {
    Objects.requireNonNull(model, "model");
    String group = DefinitionModelMapperSupport.requireNonBlank(sessionGroup, "sessionGroup");
    String ingestName =
        DefinitionModelMapperSupport.requireNonBlank(model.ingestName(), "source.ingestName");
    String namespace =
        DefinitionModelMapperSupport.requireNonBlank(model.namespace(), "source.namespace");
    DefinitionEnvelope envelope =
        DefinitionModelMapperSupport.envelopeNow(
            group,
            DefinitionType.INGEST_SOURCE,
            ingestName,
            Strings.firstNonBlank(
                model.uuid(),
                DefinitionModelMapperSupport.buildDefinitionId(
                    "ingest-source", group, ingestName)),
            model.enabled() == null || model.enabled(),
            model.lifecycleStatus(),
            actor);
    String format = toFormatToken(model.format());

    if (model instanceof IngestSourceApiModel api) {
      DefinitionIngestSource.JsonConfig json =
          "JSON".equals(format)
              ? new DefinitionIngestSource.JsonConfig("OBJECT_PER_MESSAGE", api.jsonRootField())
              : null;
      DefinitionIngestSource.NvFixConfig nvfix =
          "NVFIX".equals(format)
              ? new DefinitionIngestSource.NvFixConfig(
                  normalizedNvFixPairDelimiter(api.nvFixPairDelimiter()),
                  normalizedNvFixKvDelimiter(api.nvFixKvDelimiter()),
                  normalizedNvFixTrim(api.nvFixTrimKeys()),
                  normalizedNvFixTrim(api.nvFixTrimValues()))
              : null;
      return new DefinitionIngestSource(
          envelope, namespace, ingestName, format, json, nvfix, null, null, null, null);
    }

    if (model instanceof IngestSourceAmpsModel amps) {
      DefinitionIngestSource.JsonConfig json =
          "JSON".equals(format)
              ? new DefinitionIngestSource.JsonConfig("OBJECT_PER_MESSAGE", amps.jsonRootField())
              : null;
      DefinitionIngestSource.NvFixConfig nvfix =
          "NVFIX".equals(format)
              ? new DefinitionIngestSource.NvFixConfig(
                  normalizedNvFixPairDelimiter(amps.nvFixPairDelimiter()),
                  normalizedNvFixKvDelimiter(amps.nvFixKvDelimiter()),
                  normalizedNvFixTrim(amps.nvFixTrimKeys()),
                  normalizedNvFixTrim(amps.nvFixTrimValues()))
              : null;
      DefinitionIngestSource.Mode mode =
          amps.mode() == IngestSourceAmpsModel.Mode.BATCH
              ? DefinitionIngestSource.Mode.BATCH
              : DefinitionIngestSource.Mode.STREAM;
      DefinitionIngestSource.OofHandling definitionOofHandling = null;
      if (amps.oofHandling() == IngestSourceAmpsModel.OofHandling.MARK_AS_DELETED) {
        definitionOofHandling = DefinitionIngestSource.OofHandling.MARK_AS_DELETED;
      } else if (hasSubscribeOption(amps.options(), "oof")) {
        definitionOofHandling = DefinitionIngestSource.OofHandling.DELETE_EVENT;
      }
      DefinitionIngestSource.AmpsSource ampsSource =
          new DefinitionIngestSource.AmpsSource(
              Strings.requireNonBlank(amps.connectionName(), "source.amps.connectionName"),
              Strings.requireNonBlank(amps.topic(), "source.amps.topic"),
              amps.sow(),
              amps.sowOnly(),
              amps.options(),
              amps.filter(),
              mode,
              definitionOofHandling,
              amps.batchMaxMessages(),
              amps.batchMaxBytes(),
              amps.batchFlushMillis());
      return new DefinitionIngestSource(
          envelope, namespace, ingestName, format, json, nvfix, ampsSource, null, null, null);
    }
    if (model instanceof IngestSourceExternalModel externalModel) {
      if (ODBC.equalsIgnoreCase(Strings.trimToNull(externalModel.providerId()))) {
        return new DefinitionIngestSource(
            envelope,
            namespace,
            ingestName,
            format,
            null,
            null,
            null,
            null,
            toDefinitionOdbcSource(externalModel),
            null);
      }
      return new DefinitionIngestSource(
          envelope,
          namespace,
          ingestName,
          format,
          null,
          null,
          null,
          new DefinitionIngestSource.ExternalSource(
              externalModel.providerId(), externalModel.reference(), externalModel.options()),
          null,
          null);
    }
    @SuppressWarnings("unchecked")
    IngestSourceProvider provider =
        plugins.findIngestSourceProviderForModel((Class<? extends IngestSourceModel>) model.getClass());
    if (provider != null) {
      try {
        DefinitionIngestSource.ExternalSource external = provider.toExternalSource(model);
        if (external == null) {
          throw new IllegalStateException(
              "Ingest source provider '" + provider.providerId() + "' returned null external mapping");
        }
        String externalType = Strings.firstNonBlank(external.sourceType(), provider.providerId());
        return new DefinitionIngestSource(
            envelope,
            namespace,
            ingestName,
            format,
            null,
            null,
            null,
            new DefinitionIngestSource.ExternalSource(
                externalType,
                external.reference(),
                external.options()),
            null,
            null);
      } catch (RuntimeException ex) {
        throw ex;
      } catch (Exception ex) {
        throw new IllegalStateException(
            "Failed to map source model via provider '" + provider.providerId() + "'",
            ex);
      }
    }
    throw new IllegalArgumentException(
        "Unsupported source model: " + model.getClass().getName());
  }

  public DefinitionDataset toDefinitionDataset(IngestModel model, String sessionGroup, String actor) {
    return toDefinitionDataset(model, sessionGroup, actor, null);
  }

  public DefinitionDataset toDefinitionDataset(
      IngestModel model, String sessionGroup, String actor, String parquetUriOverride) {
    Objects.requireNonNull(model, "model");
    String group = DefinitionModelMapperSupport.requireNonBlank(sessionGroup, "sessionGroup");
    String ingestName = DefinitionModelMapperSupport.requireNonBlank(model.name(), "model.name");
    String namespace =
        DefinitionModelMapperSupport.requireNonBlank(model.namespace(), "model.namespace");
    DatasetId datasetId = resolveDatasetId(group, namespace, model);
    IngestTargetModel target = model.target();
    if (target == null) {
      return null;
    }
    DefinitionDataset.Binding binding;
    DatasetType type;
    if (target instanceof IngestTargetParquetModel parquet) {
      type = DatasetType.PARQUET;
      String parquetUri = Strings.firstNonBlank(parquetUriOverride, parquet.parquetPath());
      binding =
          new DefinitionDataset.Binding(
              null,
              new DefinitionDataset.ParquetBinding(
                  parquetUri,
                  parquet.glob(),
                  parquet.hivePartitioning(),
                  null,
                  parquet.partitionKey(),
                  parquet.partitionValueField(),
                  parquet.connectionName()),
              null);
    } else if (target instanceof IngestTargetDuckDbActiveModel duckdbActive) {
      type = DatasetType.DUCKDB;
      String relationName = Strings.firstNonBlank(duckdbActive.tableName(), datasetId.name());
      binding =
          new DefinitionDataset.Binding(
              new DefinitionDataset.DuckdbBinding(
                  relationName,
                  IngestTargetType.DUCKDB_ACTIVE.name(),
                  duckdbActive.primaryKey(),
                  duckdbActive.memory(),
                  duckdbActive.databasePath(),
                  duckdbActive.parquetHistoryPath(),
                  duckdbActive.persistOnStop(),
                  duckdbActive.createIfMissing()),
              null,
              null);
    } else if (target instanceof IngestTargetDuckDbModel duckdb) {
      type = DatasetType.DUCKDB;
      String relationName = Strings.firstNonBlank(duckdb.table(), datasetId.name());
      binding =
          new DefinitionDataset.Binding(
              new DefinitionDataset.DuckdbBinding(
                  relationName, IngestTargetType.DUCKDB.name(), List.of(), null, null, null, null, null),
              null,
              null);
    } else if (target instanceof IngestTargetExternalModel external) {
      type = DatasetType.EXTERNAL;
      if (ODBC.equalsIgnoreCase(Strings.trimToNull(external.providerId()))) {
        binding =
            new DefinitionDataset.Binding(
                null,
                null,
                null,
                toDefinitionOdbcBinding(external));
      } else {
        binding =
            new DefinitionDataset.Binding(
                null,
                null,
                new DefinitionDataset.ExternalBinding(
                    external.providerId(),
                    external.reference(),
                    external.options()),
                null);
      }
    } else {
      @SuppressWarnings("unchecked")
      IngestTargetProvider provider =
          plugins.findIngestTargetProviderForModel((Class<? extends IngestTargetModel>) target.getClass());
      if (provider == null) {
        throw new IllegalArgumentException("Unsupported target model: " + target.getClass().getName());
      }
      DefinitionDataset.ExternalBinding externalBinding;
      try {
        externalBinding = provider.toExternalBinding(target);
      } catch (RuntimeException ex) {
        throw ex;
      } catch (Exception ex) {
        throw new IllegalStateException(
            "Failed to map target model via provider '" + provider.providerId() + "'",
            ex);
      }
      if (externalBinding == null) {
        throw new IllegalStateException(
            "Ingest target provider '" + provider.providerId() + "' returned null external binding");
      }
      type = DatasetType.EXTERNAL;
      binding =
          new DefinitionDataset.Binding(
              null,
              null,
              new DefinitionDataset.ExternalBinding(
                  Strings.firstNonBlank(externalBinding.externalType(), provider.providerId()),
                  externalBinding.reference(),
                  externalBinding.options()),
              null);
    }
    DefinitionEnvelope envelope =
        DefinitionModelMapperSupport.envelopeNow(
            group,
            DefinitionType.DATASET,
            DefinitionModelMapperSupport.buildScopedName(
                datasetId.namespace(), datasetId.name(), "dataset.namespace", "dataset.name"),
            DefinitionModelMapperSupport.buildDefinitionId(
                "dataset",
                group,
                DefinitionModelMapperSupport.buildScopedName(
                    datasetId.namespace(), datasetId.name(), "dataset.namespace", "dataset.name")),
            model.enabled(),
            model.lifecycleStatus(),
            actor);
    return new DefinitionDataset(envelope, type, datasetId.namespace(), binding);
  }

  public IngestModel fromDefinitions(
      DefinitionIngest ingest,
      DefinitionIngestSource sourceOrNull,
      SchemaModel schemaOrNull) {
    Objects.requireNonNull(ingest, "ingest");
    DefinitionEnvelope envelope = ingest.envelope();
    String ingestName =
        Strings.requireNonBlank(envelope == null ? null : envelope.name(), "ingest.envelope.name");
    String namespace =
        Strings.requireNonBlank(
            Strings.firstNonBlank(
                sourceOrNull == null ? null : sourceOrNull.namespace(),
                parseNamespace(ingest.datasetId())),
            "ingest.namespace");
    IngestModel.Builder builder =
        IngestModel.builder()
            .withName(ingestName)
            .withNamespace(namespace)
            .withResolvedUuid(envelope == null ? null : envelope.uuid())
            .withEnabled(envelope == null || !Boolean.FALSE.equals(envelope.enabled()))
            .withResolvedLifecycleStatus(envelope == null ? null : envelope.lifecycleStatus())
            .withSchemaMode(fromDefinitionSchemaMode(ingest.schemaMode()))
            .withSchema(Objects.requireNonNull(schemaOrNull, "ingest schema is required"))
            .withResolvedDatasetId(ingest.datasetId());

    if (sourceOrNull != null) {
      builder.withSource(fromDefinitionSource(sourceOrNull));
    } else {
      builder.withSourceApi();
    }
    return builder.build();
  }

  public IngestSourceModel fromDefinitionSource(DefinitionIngestSource source) {
    if (source == null) {
      return null;
    }
    DefinitionEnvelope envelope = source.envelope();
    String ingestName = Strings.trimToNull(envelope == null ? null : envelope.name());
    String namespace = Strings.requireNonBlank(source.namespace(), "source.namespace");
    IngestFormat format = IngestFormat.fromToken(Strings.requireNonBlank(source.format(), "source.format"));
    if (source.external() != null) {
      DefinitionIngestSource.ExternalSource external = source.external();
      String sourceType = Strings.requireNonBlank(external.sourceType(), "source.external.sourceType");
      IngestSourceProvider provider = plugins.findIngestSourceProvider(sourceType);
      if (provider != null) {
        try {
          return provider.fromExternalSource(
              new IngestSourceProviderContext(
                  source,
                  Strings.requireNonBlank(ingestName, "source.envelope.name"),
                  namespace,
                  envelope == null ? null : envelope.uuid(),
                  envelope == null ? null : envelope.enabled(),
                  envelope == null ? null : envelope.lifecycleStatus(),
                  format,
                  external));
        } catch (RuntimeException ex) {
          throw ex;
        } catch (Exception ex) {
          throw new IllegalStateException(
              "Failed to rebuild source model via provider '" + provider.providerId() + "'",
              ex);
        }
      }
      return IngestSourceExternalModel.builder()
          .withResolvedIngestName(ingestName)
          .withNamespace(namespace)
          .withUuid(envelope == null ? null : envelope.uuid())
          .withEnabled(envelope == null ? null : envelope.enabled())
          .withLifecycleStatus(envelope == null ? null : envelope.lifecycleStatus())
          .withFormat(format)
          .withProviderId(sourceType)
          .withReference(external.reference())
          .withOptions(external.options())
          .build();
    }
    if (source.odbc() != null) {
      DefinitionIngestSource.OdbcSource odbc = source.odbc();
      return IngestSourceExternalModel.builder()
          .withResolvedIngestName(ingestName)
          .withNamespace(namespace)
          .withUuid(envelope == null ? null : envelope.uuid())
          .withEnabled(envelope == null ? null : envelope.enabled())
          .withLifecycleStatus(envelope == null ? null : envelope.lifecycleStatus())
          .withFormat(format)
          .withProviderId(ODBC)
          .withReference(odbc.connectionName())
          .withOptions(toExternalOptions(odbc))
          .build();
    }
    if (source.amps() == null) {
      return IngestSourceApiModel.builder()
          .withResolvedIngestName(ingestName)
          .withNamespace(namespace)
          .withUuid(envelope == null ? null : envelope.uuid())
          .withEnabled(envelope == null ? null : envelope.enabled())
          .withLifecycleStatus(envelope == null ? null : envelope.lifecycleStatus())
          .withFormat(format)
          .withJsonRootField(source.json() == null ? null : source.json().rootField())
          .withNvFixPairDelimiter(source.nvfix() == null ? null : source.nvfix().pairDelimiter())
          .withNvFixKvDelimiter(source.nvfix() == null ? null : source.nvfix().kvDelimiter())
          .withNvFixTrimKeys(source.nvfix() == null ? null : source.nvfix().trimKeys())
          .withNvFixTrimValues(source.nvfix() == null ? null : source.nvfix().trimValues())
          .build();
    }
    DefinitionIngestSource.AmpsSource amps = source.amps();
    IngestSourceAmpsModel.Builder builder =
        IngestSourceAmpsModel.builder()
        .withResolvedIngestName(ingestName)
        .withNamespace(namespace)
        .withUuid(envelope == null ? null : envelope.uuid())
        .withEnabled(envelope == null ? null : envelope.enabled())
        .withLifecycleStatus(envelope == null ? null : envelope.lifecycleStatus())
        .withFormat(format)
        .withConnectionName(amps.connectionName())
        .withTopic(amps.topic())
        .withSowSubscribe(amps.sow())
        .withSow(amps.sowOnly())
        .withOptions(amps.options())
        .withFilter(amps.filter())
        .withMode(
            amps.mode() == DefinitionIngestSource.Mode.BATCH
                ? IngestSourceAmpsModel.Mode.BATCH
                : IngestSourceAmpsModel.Mode.STREAM)
        .withBatchMaxMessages(amps.batchMaxMessages())
        .withBatchMaxBytes(amps.batchMaxBytes())
        .withBatchFlushMillis(amps.batchFlushMillis())
        .withJsonRootField(source.json() == null ? null : source.json().rootField())
        .withNvFixPairDelimiter(source.nvfix() == null ? null : source.nvfix().pairDelimiter())
        .withNvFixKvDelimiter(source.nvfix() == null ? null : source.nvfix().kvDelimiter())
        .withNvFixTrimKeys(source.nvfix() == null ? null : source.nvfix().trimKeys())
        .withNvFixTrimValues(source.nvfix() == null ? null : source.nvfix().trimValues());
    if (amps.oofHandling() != null) {
      builder.withOofHandling(
          amps.oofHandling() == DefinitionIngestSource.OofHandling.MARK_AS_DELETED
              ? IngestSourceAmpsModel.OofHandling.MARK_AS_DELETED
              : IngestSourceAmpsModel.OofHandling.DELETE_EVENT);
    }
    return builder.build();
  }

  public IngestTargetModel fromDefinitionDataset(DefinitionDataset dataset) {
    if (dataset == null || dataset.binding() == null) {
      return null;
    }
    if (dataset.type() == DatasetType.PARQUET && dataset.binding().parquet() != null) {
      DefinitionDataset.ParquetBinding parquet = dataset.binding().parquet();
      return IngestTargetParquetModel.builder()
          .withParquetPath(parquet.uri())
          .withGlob(parquet.glob())
          .withHivePartitioning(parquet.hivePartitioning())
          .withPartitionKey(parquet.partitionKey())
          .withPartitionValueField(parquet.partitionValueField())
          .withConnectionName(parquet.connectionName())
          .build();
    }
    if (dataset.type() == DatasetType.DUCKDB && dataset.binding().duckdb() != null) {
      DefinitionDataset.DuckdbBinding duckdb = dataset.binding().duckdb();
      String targetTypeToken = Strings.trimToNull(duckdb.targetType());
      if (IngestTargetType.DUCKDB_ACTIVE.name().equalsIgnoreCase(Strings.firstNonBlank(targetTypeToken, ""))) {
        return IngestTargetDuckDbActiveModel.builder()
            .withTableName(duckdb.relationName())
            .withPrimaryKey(duckdb.primaryKey())
            .withMemory(duckdb.memory())
            .withDatabasePath(duckdb.databasePath())
            .withParquetHistoryPath(duckdb.parquetHistoryPath())
            .withPersistOnStop(duckdb.persistOnStop())
            .withCreateIfMissing(duckdb.createIfMissing())
            .build();
      }
      return IngestTargetDuckDbModel.builder()
          .withType(IngestTargetDuckDbModel.Type.MEMORY)
          .withTable(duckdb.relationName())
          .build();
    }
    if (dataset.binding().external() != null
        && (dataset.type() == DatasetType.EXTERNAL || dataset.type() == null)) {
      DefinitionDataset.ExternalBinding external = dataset.binding().external();
      String targetType = Strings.requireNonBlank(external.externalType(), "dataset.binding.external.externalType");
      IngestTargetProvider provider = plugins.findIngestTargetProvider(targetType);
      if (provider != null) {
        try {
          return provider.fromExternalBinding(
              new IngestTargetProviderContext(dataset, toDatasetId(dataset), external));
        } catch (RuntimeException ex) {
          throw ex;
        } catch (Exception ex) {
          throw new IllegalStateException(
              "Failed to rebuild target model via provider '" + provider.providerId() + "'",
              ex);
        }
      }
      return IngestTargetExternalModel.builder()
          .withProviderId(targetType)
          .withReference(external.reference())
          .withOptions(external.options())
          .build();
    }
    if (dataset.binding().odbc() != null
        && (dataset.type() == DatasetType.EXTERNAL || dataset.type() == null)) {
      DefinitionDataset.OdbcBinding odbc = dataset.binding().odbc();
      return IngestTargetExternalModel.builder()
          .withProviderId(ODBC)
          .withReference(odbc.connectionName())
          .withOptions(toExternalOptions(odbc))
          .build();
    }
    return null;
  }

  public String deterministicSchemaName(String sessionGroup, String namespace, String ingestName) {
    DefinitionModelMapperSupport.requireNonBlank(sessionGroup, "sessionGroup");
    String normalizedNamespace =
        DefinitionModelMapperSupport.requireNonBlank(namespace, "namespace");
    String normalizedIngestName =
        DefinitionModelMapperSupport.requireNonBlank(ingestName, "ingestName");
    return "ingest:" + normalizedNamespace + "/" + normalizedIngestName;
  }

  public DatasetId resolveDatasetId(String group, String namespace, IngestModel model) {
    String datasetCanonical = Strings.trimToNull(model.datasetId());
    if (datasetCanonical != null) {
      return DatasetId.parse(datasetCanonical);
    }
    String datasetName = defaultDatasetName(model);
    return DatasetId.of(group, namespace, datasetName);
  }

  private static String defaultDatasetName(IngestModel model) {
    IngestTargetModel target = model.target();
    if (target instanceof IngestTargetDuckDbModel duckdb) {
      return Strings.firstNonBlank(duckdb.table(), model.name());
    }
    if (target instanceof IngestTargetDuckDbActiveModel duckdbActive) {
      return Strings.firstNonBlank(duckdbActive.tableName(), model.name());
    }
    return model.name();
  }

  private static String normalizedNvFixPairDelimiter(String configured) {
    String value = Strings.trimToNull(configured);
    return value == null ? String.valueOf(AmpsConnection.SOH) : value;
  }

  private static String normalizedNvFixKvDelimiter(String configured) {
    String value = Strings.trimToNull(configured);
    if (value == null) {
      return "=";
    }
    if (value.isEmpty()) {
      throw new IllegalArgumentException("nvFixKvDelimiter must not be empty");
    }
    return value;
  }

  private static Boolean normalizedNvFixTrim(Boolean configured) {
    return configured == null || configured;
  }

  private static IngestSchemaMode fromDefinitionSchemaMode(DefinitionIngest.SchemaMode mode) {
    if (mode == null) {
      return IngestSchemaMode.STRICT;
    }
    return switch (mode) {
      case STRICT -> IngestSchemaMode.STRICT;
      case ADD_COLUMNS -> IngestSchemaMode.ADD_COLUMNS;
      case RELAXED -> IngestSchemaMode.RELAXED;
      case UNKNOWN ->
          throw new IllegalArgumentException("Unsupported ingest schemaMode UNKNOWN");
    };
  }

  private static DefinitionIngest.SchemaMode toDefinitionSchemaMode(IngestSchemaMode mode) {
    IngestSchemaMode normalized = mode == null ? IngestSchemaMode.STRICT : mode;
    return switch (normalized) {
      case STRICT -> DefinitionIngest.SchemaMode.STRICT;
      case ADD_COLUMNS -> DefinitionIngest.SchemaMode.ADD_COLUMNS;
      case RELAXED -> DefinitionIngest.SchemaMode.RELAXED;
    };
  }

  private static String toFormatToken(IngestFormat format) {
    IngestFormat normalized = format == null ? IngestFormat.JSON : format;
    return normalized.name().toUpperCase(Locale.ROOT);
  }

  private static String parseNamespace(String datasetIdCanonical) {
    String datasetId = Strings.trimToNull(datasetIdCanonical);
    if (datasetId == null) {
      return null;
    }
    try {
      return DatasetId.parse(datasetId).namespace();
    } catch (RuntimeException ex) {
      return null;
    }
  }

  private static DatasetId toDatasetId(DefinitionDataset dataset) {
    DefinitionEnvelope envelope = dataset == null ? null : dataset.envelope();
    String group =
        DefinitionModelMapperSupport.requireNonBlank(
            envelope == null ? null : envelope.group(), "dataset.envelope.group");
    String namespace =
        DefinitionModelMapperSupport.requireNonBlank(
            dataset == null ? null : dataset.namespace(), "dataset.namespace");
    String envelopeName = Strings.trimToNull(envelope == null ? null : envelope.name());
    String datasetName = envelopeName;
    if (datasetName != null && datasetName.contains("/")) {
      int delimiter = datasetName.indexOf('/');
      datasetName = datasetName.substring(delimiter + 1);
    }
    datasetName =
        DefinitionModelMapperSupport.requireNonBlank(datasetName, "dataset.envelope.name");
    return DatasetId.of(group, namespace, datasetName);
  }

  private static boolean hasSubscribeOption(String options, String optionToken) {
    String normalizedToken = Strings.trimToNull(optionToken);
    String normalizedOptions = Strings.trimToNull(options);
    if (normalizedToken == null || normalizedOptions == null) {
      return false;
    }
    for (String token : normalizedOptions.split(";")) {
      String normalized = Strings.trimToNull(token);
      if (normalized != null && normalizedToken.equalsIgnoreCase(normalized)) {
        return true;
      }
    }
    return false;
  }

  private static DefinitionIngestSource.OdbcSource toDefinitionOdbcSource(
      IngestSourceExternalModel externalModel) {
    String connectionName =
        Strings.firstNonBlank(
            Strings.trimToNull(externalModel.reference()),
            optionValue(externalModel.options(), "connectionName"));
    DefinitionConnection.Odbc.DatabaseType databaseType =
        parseDatabaseType(optionValue(externalModel.options(), "databaseType"));
    String tableName = optionValue(externalModel.options(), "tableName");
    String querySql = optionValue(externalModel.options(), "querySql");
    return new DefinitionIngestSource.OdbcSource(
        connectionName,
        databaseType,
        tableName,
        querySql,
        copyOptions(externalModel.options()));
  }

  private static DefinitionDataset.OdbcBinding toDefinitionOdbcBinding(
      IngestTargetExternalModel externalModel) {
    if (!ODBC.equalsIgnoreCase(Strings.trimToNull(externalModel.providerId()))) {
      return null;
    }
    String connectionName =
        Strings.firstNonBlank(
            Strings.trimToNull(externalModel.reference()),
            optionValue(externalModel.options(), "connectionName"));
    return new DefinitionDataset.OdbcBinding(
        connectionName,
        parseDatabaseType(optionValue(externalModel.options(), "databaseType")),
        optionValue(externalModel.options(), "catalogName"),
        optionValue(externalModel.options(), "schemaName"),
        optionValue(externalModel.options(), "tableName"),
        optionValue(externalModel.options(), "querySql"),
        optionValue(externalModel.options(), "primaryKey"),
        copyOptions(externalModel.options()));
  }

  private static Map<String, String> toExternalOptions(DefinitionIngestSource.OdbcSource odbc) {
    java.util.LinkedHashMap<String, String> options = new java.util.LinkedHashMap<>();
    putOption(options, "connectionName", odbc.connectionName());
    putOption(options, "databaseType", odbc.databaseType() == null ? null : odbc.databaseType().name());
    putOption(options, "tableName", odbc.tableName());
    putOption(options, "querySql", odbc.querySql());
    if (odbc.options() != null) {
      options.putAll(odbc.options());
    }
    return options.isEmpty() ? Map.of() : Map.copyOf(options);
  }

  private static Map<String, String> toExternalOptions(DefinitionDataset.OdbcBinding odbc) {
    java.util.LinkedHashMap<String, String> options = new java.util.LinkedHashMap<>();
    putOption(options, "connectionName", odbc.connectionName());
    putOption(options, "databaseType", odbc.databaseType() == null ? null : odbc.databaseType().name());
    putOption(options, "catalogName", odbc.catalogName());
    putOption(options, "schemaName", odbc.schemaName());
    putOption(options, "tableName", odbc.tableName());
    putOption(options, "querySql", odbc.querySql());
    putOption(options, "primaryKey", odbc.primaryKey());
    if (odbc.options() != null) {
      options.putAll(odbc.options());
    }
    return options.isEmpty() ? Map.of() : Map.copyOf(options);
  }

  private static DefinitionConnection.Odbc.DatabaseType parseDatabaseType(String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      return null;
    }
    try {
      return DefinitionConnection.Odbc.DatabaseType.valueOf(normalized.toUpperCase(Locale.ROOT));
    } catch (IllegalArgumentException ex) {
      return DefinitionConnection.Odbc.DatabaseType.OTHER;
    }
  }

  private static String optionValue(Map<String, String> options, String key) {
    if (options == null || options.isEmpty()) {
      return null;
    }
    return Strings.trimToNull(options.get(key));
  }

  private static Map<String, String> copyOptions(Map<String, String> options) {
    if (options == null || options.isEmpty()) {
      return Map.of();
    }
    return Map.copyOf(options);
  }

  private static void putOption(Map<String, String> options, String key, String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized != null) {
      options.put(key, normalized);
    }
  }
