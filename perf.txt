<?xml version="1.0" encoding="UTF-8"?>
<configuration scan="true" scanPeriod="30 seconds">
  <property name="FABRIC_LOG_LEVEL" value="${fabric.log.level:-INFO}" />

  <appender name="STDOUT" class="ch.qos.logback.core.ConsoleAppender">
    <encoder>
      <pattern>%d{HH:mm:ss.SSS} %-5level [%thread] %logger{48} - %msg%n</pattern>
    </encoder>
  </appender>

  <logger name="org.pronzato.fabric" level="${FABRIC_LOG_LEVEL}" additivity="false">
    <appender-ref ref="STDOUT" />
  </logger>
  <logger name="org.apache.arrow" level="INFO" additivity="false">
    <appender-ref ref="STDOUT" />
  </logger>
  <logger name="io.grpc" level="INFO" additivity="false">
    <appender-ref ref="STDOUT" />
  </logger>
  <logger name="io.netty" level="INFO" additivity="false">
    <appender-ref ref="STDOUT" />
  </logger>

  <root level="WARN">
    <appender-ref ref="STDOUT" />
  </root>
</configuration>
