    }

  public IngestAmpsPump start() throws Exception {
    if (!started.compareAndSet(false, true)) {
      return this;
    }
    try {
      AmpsConnection amps =
          connectionInstanceName == null
              ? session.connections().amps(connectionName)
              : session
                  .connections()
                  .amps(
                      connectionName,
                      ConnectionOptions.builder().withPinnedInstance(connectionInstanceName).build());
      BatchAccumulator accumulator = null;
      if (batchSpec.mode() == AmpsConnection.BatchMode.BATCH) {
        accumulator =
            new BatchAccumulator(
                batchSpec,
                SystemClock.INSTANCE,
                (payloads, messages, bytes, elapsedMs) -> {
                  List<PendingBatchMessage> batchMessages = drainPendingBatchMessages(payloads.size());
                  for (int index = 0; index < payloads.size(); index++) {
                    PendingBatchMessage batchMessage = batchMessages.get(index);
                    append(batchMessage.message(), payloads.get(index), batchMessage.ampsCommand());
                  }
                  flushPendingCheckpoint();
                  LOGGER.info("[IngestAmpsPump] batch flush ingestName={} messageCount={} totalBytes={}", ingestName, messages, bytes);
                });
        batchAccumulator = accumulator;
        batchScheduler = createBatchScheduler();
        long intervalMs = Math.max(1L, batchSpec.batchFlushMillis());
        batchScheduler.scheduleAtFixedRate(
            accumulator::flushIfDue, intervalMs, intervalMs, TimeUnit.MILLISECONDS);
      }
      BatchAccumulator resolvedAccumulator = accumulator;
      subscription =
          amps.subscribe(
              subscriptionSpec,
              (AmpsConnection.SubscriptionListener)
                  message -> onMessage(message, resolvedAccumulator));
      LOGGER.info("[IngestAmpsPump] start ingestName={} ingestSourceName={} connectionName={} connectionInstance={} topic={} sowSubscribe={} sowOnly={} mode={} batchMaxMessages={} batchMaxBytes={} batchFlushMillis={} logIncoming={} logOutgoing={} transformerEnabled={}", ingestName, ingestSourceName, connectionName, connectionInstanceName, subscriptionSpec.topic(), subscriptionSpec.sow(), subscriptionSpec.sowOnly(), batchSpec.mode(), batchSpec.batchMaxMessages(), batchSpec.batchMaxBytes(), batchSpec.batchFlushMillis(), logIncomingMessages, logOutgoingMessages, payloadTransformer != null || payloadTransformerWithCommand != null);
      return this;
    } catch (Exception ex) {
      started.set(false);
      closeScheduler();
      batchAccumulator = null;
      subscription = null;
      throw ex;
    }
  }

  public String lastCheckpoint() {
    String pending = Strings.trimToNull(pendingBookmark);
    return pending == null ? lastCheckpoint : pending;
  }

  public String lastCheckpointFlushUtc() {
    return lastCheckpointFlushUtc;
  }

  @Override
  public void close() {
    if (!closed.compareAndSet(false, true)) {
      return;
    }
    closeScheduler();
    BatchAccumulator accumulator = batchAccumulator;
    if (accumulator != null) {
      try {
        accumulator.flush();
      } catch (Exception ex) {
        LOGGER.info("[IngestAmpsPump] error ingestName={} topic={} exceptionClass={} message={}", ingestName, subscriptionSpec.topic(), ex.getClass().getName(), ex.getMessage());
      }
    }
    flushPendingCheckpoint();
    AmpsConnection.Subscription active = subscription;
    subscription = null;
    if (active != null) {
      active.close();
    }
    synchronized (pendingMessagesLock) {
      pendingBatchMessages.clear();
    }
  }

  private void onMessage(AmpsConnection.AmpsMessage message, BatchAccumulator accumulator) {
    try {
      Message raw = message.raw();
      if (raw != null && raw.getCommand() == Message.Command.GroupEnd) {
        notifySowComplete();
        if (subscriptionSpec.sowOnly()) {
          close();
        }
        return;
      }
      if (!isMutationCommand(raw)) {
        return;
      }
      String line = toSingleLinePayload(message);
      if (line == null) {
        return;
      }
      if (logIncomingMessages) {
        logIncomingMessage(raw, message, line);
      }
      int command = messageCommand(raw);
      String transformed = applyPayloadTransformer(line, command);
      if (transformed == null) {
        persistBookmarkIfPresent(message);
        if (accumulator == null || !hasPendingBatchMessages()) {
          flushPendingCheckpoint();
        }
        return;
      }
      int effectiveCommand = effectiveCommand(command);
      requireWrite(commandOperation(effectiveCommand));
      if (accumulator != null) {
        synchronized (pendingMessagesLock) {
          pendingBatchMessages.add(new PendingBatchMessage(message, effectiveCommand));
        }
        try {
          accumulator.onMessage(transformed);
        } catch (RuntimeException ex) {
          rollbackPendingBatchMessage(message);
          throw ex;
        }
        return;
      }
      append(message, transformed, effectiveCommand);
    } catch (RuntimeException ex) {
      notifyError(ex);
      LOGGER.info("[IngestAmpsPump] error ingestName={} topic={} exceptionClass={} message={}", ingestName, subscriptionSpec.topic(), ex.getClass().getName(), ex.getMessage());
      throw ex;
    }
  }

  private String toSingleLinePayload(AmpsConnection.AmpsMessage message) {
    Objects.requireNonNull(message, "message");
    String data = message.data();
    if (data == null) {
      return null;
    }
    return AmpsConnection.ensureTrailingNewline(data);
  }

  private String applyPayloadTransformer(String payloadText, int ampsCommand) {
    if (payloadText == null) {
      return null;
    }
    if (payloadTransformerWithCommand != null) {
      String transformed =
          Strings.trimToNull(payloadTransformerWithCommand.apply(payloadText, ampsCommand));
      if (transformed == null) {
        return null;
      }
      return AmpsConnection.ensureTrailingNewline(transformed);
    }
    if (payloadTransformer == null) {
      return payloadText;
    }
    String transformed = Strings.trimToNull(payloadTransformer.apply(payloadText));
    if (transformed == null) {
      return null;
    }
    return AmpsConnection.ensureTrailingNewline(transformed);
  }

  private void requireWrite(String operation) {
    if (writeAuthorizer == null) {
      return;
    }
    writeAuthorizer.requireWrite(
        Strings.requireNonBlank(datasetId, "datasetId"),
        Strings.requireNonBlank(operation, "operation"));
  }

  private void notifyError(Throwable error) {
    try {
      errorListener.onError(error);
    } catch (Exception ignore) {
    }
  }

  private void notifySowComplete() {
    if (sowCompleteListener == null) {
      return;
    }
    try {
      sowCompleteListener.run();
    } catch (Exception ignore) {
    }
  }

  private void append(AmpsConnection.AmpsMessage message, String payloadText, int ampsCommand) {
    Message raw = message == null ? null : message.raw();
    int command = ampsCommand;
    if (logOutgoingMessages) {
      logOutgoingMessage(message, payloadText, command);
    }
    if (ampsMessageHandler != null) {
      ampsMessageHandler.onMessage(
          ingestName,
          payloadText,
          payloadFormat,
          command,
          oofHandling,
          parquetFilenamePrefix);
      persistBookmarkIfPresent(message);
      return;
    }
    if (isDeleteCommand(command)) {
      return;
    }
    IngestPayload payload = toIngestPayload(payloadText);
    session.data().direct().ingest().append(ingestName, payload);
    persistBookmarkIfPresent(message);
  }

  private IngestPayload toIngestPayload(String payloadText) {
    return payloadFormat == PayloadFormat.JSON
        ? new IngestPayloadJson(payloadText)
        : new IngestPayloadNvFix(payloadText);
  }

  private void persistBookmarkIfPresent(AmpsConnection.AmpsMessage message) {
    if (message == null || !isBookmarkCandidate(message)) {
      return;
    }
    String bookmark = Strings.trimToNull(message.bookmark());
    if (bookmark == null) {
      return;
    }
    pendingBookmark = bookmark;
    if (batchSpec.mode() == AmpsConnection.BatchMode.STREAM) {
      flushPendingCheckpoint();
    }
  }

  private synchronized void flushPendingCheckpoint() {
    String bookmark = Strings.trimToNull(pendingBookmark);
    if (bookmark == null || checkpointStore == null || checkpointKey == null) {
      return;
    }
    checkpointStore.save(
        CheckpointRecord.of(
            checkpointKey,
            Map.of(
                "bookmark", bookmark,
                "ingestName", ingestName,
                "connectionName", connectionName,
                "topic", subscriptionSpec.topic())));
    lastCheckpoint = bookmark;
    lastCheckpointFlushUtc = Instant.now().toString();
    pendingBookmark = null;
  }

  private List<PendingBatchMessage> drainPendingBatchMessages(int expected) {
    synchronized (pendingMessagesLock) {
      if (pendingBatchMessages.size() < expected) {
        throw new IllegalStateException(
            "Ingest AMPS batch state mismatch: expected="
                + expected
                + " buffered="
                + pendingBatchMessages.size());
      }
      List<PendingBatchMessage> drained = new ArrayList<>(expected);
      for (int index = 0; index < expected; index++) {
        drained.add(pendingBatchMessages.remove(0));
      }
      return drained;
    }
  }

  private void rollbackPendingBatchMessage(AmpsConnection.AmpsMessage message) {
    synchronized (pendingMessagesLock) {
      int last = pendingBatchMessages.size() - 1;
      if (last >= 0 && pendingBatchMessages.get(last).message() == message) {
        pendingBatchMessages.remove(last);
      }
    }
  }

  private boolean hasPendingBatchMessages() {
    synchronized (pendingMessagesLock) {
      return !pendingBatchMessages.isEmpty();
    }
  }

  private static boolean isMutationCommand(Message message) {
    if (message == null) {
      return true;
    }
    int command = message.getCommand();
    return command == Message.Command.Publish
        || command == Message.Command.SOW
        || command == Message.Command.OOF
        || command == Message.Command.SOWDelete;
  }

  private static boolean isBookmarkCandidate(AmpsConnection.AmpsMessage message) {
    Message raw = message == null ? null : message.raw();
    if (raw == null) {
      return false;
    }
    int command = raw.getCommand();
    if (command != Message.Command.Publish
        && command != Message.Command.SOW
        && command != Message.Command.OOF
        && command != Message.Command.SOWDelete) {
      return false;
    }
    if (Strings.isBlank(message.topic())) {
      return false;
    }
    return Strings.isNotBlank(message.data());
  }

  private static String commandOperation(int command) {
    if (isDeleteCommand(command)) {
      return "DELETE";
    }
    return "INGEST";
  }

  private int effectiveCommand(int command) {
    if (persistOofAsRows && isDeleteCommand(command)) {
      return Message.Command.Publish;
    }
    return command;
  }

  private void logIncomingMessage(Message raw, AmpsConnection.AmpsMessage message, String payloadText) {
    LOGGER.info("[IngestAmpsPump][incoming] ingestName={} connectionName={} connectionInstance={} topic={} command={} bytes={} bookmark={} payload={}", ingestName, connectionName, connectionInstanceName, subscriptionSpec.topic(), commandLabel(raw), payloadText == null ? 0 : payloadText.length(), message == null ? null : Strings.trimToNull(message.bookmark()), payloadPreview(payloadText));
  }

  private void logOutgoingMessage(
      AmpsConnection.AmpsMessage message, String payloadText, int command) {
    LOGGER.info("[IngestAmpsPump][outgoing] ingestName={} datasetId={} payloadFormat={} command={} oofHandling={} bookmark={} payload={}", ingestName, datasetId, payloadFormat, commandLabel(command), oofHandling, message == null ? null : Strings.trimToNull(message.bookmark()), payloadPreview(payloadText));
  }

  private static String commandLabel(Message message) {
    return commandLabel(message == null ? Message.Command.Publish : message.getCommand());
  }

  private static String commandLabel(int command) {
    return switch (command) {
      case Message.Command.Publish -> "PUBLISH";
      case Message.Command.SOW -> "SOW";
      case Message.Command.OOF -> "OOF";
      case Message.Command.SOWDelete -> "SOW_DELETE";
      case Message.Command.GroupEnd -> "GROUP_END";
      default -> Integer.toString(command);
    };
  }

  private static int messageCommand(Message message) {
    return message == null ? Message.Command.Publish : message.getCommand();
  }

  private static boolean isDeleteCommand(int command) {
    return command == Message.Command.OOF || command == Message.Command.SOWDelete;
  }

  private static String payloadPreview(String payloadText) {
    String normalized = Strings.trimToNull(payloadText);
    if (normalized == null) {
      return null;
    }
    String singleLine = normalized.replace('\n', ' ').replace('\r', ' ').trim();
    if (singleLine.length() <= MAX_VERBOSE_PAYLOAD_CHARS) {
      return singleLine;
    }
    return singleLine.substring(0, MAX_VERBOSE_PAYLOAD_CHARS) + "...";
  }

  private ScheduledExecutorService createBatchScheduler() {
    ThreadFactory threadFactory =
        runnable -> {
          Thread thread = new Thread(runnable, "ingest-amps-pump-" + ingestName);
          thread.setDaemon(true);
          return thread;
        };
    return Executors.newSingleThreadScheduledExecutor(threadFactory);
  }

  private void closeScheduler() {
    ScheduledExecutorService scheduler = batchScheduler;
    batchScheduler = null;
    if (scheduler == null) {
      return;
    }
    scheduler.shutdownNow();
  }

    public IngestAmpsPump build() {
      String resolvedIngestName = Strings.requireNonBlank(ingestName, "ingestName");
      String resolvedIngestSourceName =
          Strings.firstNonBlank(ingestSourceName, resolvedIngestName);
      String resolvedDatasetId = Strings.trimToNull(datasetId);
      if (writeAuthorizer != null) {
        resolvedDatasetId = Strings.requireNonBlank(resolvedDatasetId, "datasetId");
      }
      String resolvedConnectionName = Strings.requireNonBlank(connectionName, "connectionName");
      String resolvedTopic = Strings.requireNonBlank(topic, "topic");
      PayloadFormat resolvedPayloadFormat =
          Objects.requireNonNull(payloadFormat, "payloadFormat");
      String bookmarkGroup =
          session.context() == null
              ? null
              : Strings.trimToNull(session.context().getGroup());
      String bookmarkStart = null;
      CheckpointStore resolvedCheckpointStore = null;
      CheckpointKey checkpointKey = null;
      if (bookmarkResume && bookmarkGroup != null) {
        String checkpointConnectionName =
            connectionInstanceName == null
                ? resolvedConnectionName
                : resolvedConnectionName + "@" + connectionInstanceName;
        String subscriptionKey =
            AmpsConnection.BookmarkStoreLocal.subscriptionKey(
                resolvedIngestName, checkpointConnectionName, resolvedTopic);
        checkpointKey =
            new CheckpointKey(
                bookmarkGroup,
                CheckpointType.INGEST,
                resolvedIngestName,
                checkpointConnectionName + ":" + subscriptionKey);
        resolvedCheckpointStore =
            checkpointStore == null
                ? CheckpointStores.failFast(
                    "Ingest AMPS checkpoint store is not configured. Configure session with withCheckpointPath(...)")
                : checkpointStore;
        bookmarkStart =
            resolvedCheckpointStore
                .load(checkpointKey)
                .map(record -> Strings.trimToNull(record.payload().get("bookmark")))
                .orElse(null);
        if (bookmarkStart == null) {
          bookmarkStart = Client.Bookmarks.NOW;
        }
      }
      boolean isSowSubscription = Boolean.TRUE.equals(sow) || Boolean.TRUE.equals(sowOnly);
      String resolvedOptions =
          (oofHandlingConfigured && isSowSubscription)
              ? ensureSubscribeOption(options, "oof")
              : options;
      AmpsConnection.SubscriptionSpec subscriptionSpec =
          new AmpsConnection.SubscriptionSpec(
              Boolean.TRUE.equals(sow),
              Boolean.TRUE.equals(sowOnly),
              resolvedTopic,
              resolvedOptions,
              filter,
              bookmarkStart,
              null);
      AmpsConnection.BatchSpec batchSpec =
          new AmpsConnection.BatchSpec(mode, batchMaxMessages, batchMaxBytes, batchFlushMillis);
      return new IngestAmpsPump(
          session,
          resolvedIngestName,
          resolvedIngestSourceName,
          resolvedDatasetId,
          resolvedConnectionName,
          connectionInstanceName,
          subscriptionSpec,
          batchSpec,
          resolvedPayloadFormat,
          oofHandling,
          parquetFilenamePrefix,
          logIncomingMessages,
          logOutgoingMessages,
          payloadTransformer,
          payloadTransformerWithCommand,
          persistOofAsRows,
          writeAuthorizer,
          ampsMessageHandler,
          errorListener,
          sowCompleteListener,
          resolvedCheckpointStore,
          checkpointKey);
    }

    private static String ensureSubscribeOption(String options, String optionToken) {
      String normalizedToken = Strings.requireNonBlank(optionToken, "optionToken");
      String normalizedOptions = Strings.trimToNull(options);
      if (normalizedOptions == null) {
        return normalizedToken;
      }
      for (String token : normalizedOptions.split(";")) {
        String normalized = Strings.trimToNull(token);
        if (normalized != null && normalizedToken.equalsIgnoreCase(normalized)) {
          return normalizedOptions;
        }
      }
      return normalizedOptions + ";" + normalizedToken;
    }
  }

  @FunctionalInterface
  interface Clock {
    long nowMillis();
  }

  enum SystemClock implements Clock {
    INSTANCE;

    @Override
    public long nowMillis() {
      return System.currentTimeMillis();
    }
  }

  @FunctionalInterface
  interface FlushHandler {
    void flush(List<String> payloads, int messageCount, int byteCount, long elapsedMs);
  }

  record PendingBatchMessage(AmpsConnection.AmpsMessage message, int ampsCommand) {}

  @FunctionalInterface
  public interface ErrorListener {
    ErrorListener NOOP = ignored -> {};

    void onError(Throwable error);
  }

  static final class BatchAccumulator {
    private final AmpsConnection.BatchSpec batchSpec;
    private final Clock clock;
    private final FlushHandler flushHandler;
    private final List<String> payloads = new ArrayList<>();
    private int messageCount;
    private int byteCount;
    private long lastFlushMs;

    BatchAccumulator(AmpsConnection.BatchSpec batchSpec, Clock clock, FlushHandler flushHandler) {
      this.batchSpec = Objects.requireNonNull(batchSpec, "batchSpec");
      this.clock = Objects.requireNonNull(clock, "clock");
      this.flushHandler = Objects.requireNonNull(flushHandler, "flushHandler");
      this.lastFlushMs = clock.nowMillis();
    }

    synchronized void onMessage(String line) {
      String payloadLine = Objects.requireNonNull(line, "line");
      if (payloadLine.isEmpty()) {
        throw new IllegalArgumentException("line is required");
      }
      int lineBytes = payloadLine.getBytes(StandardCharsets.UTF_8).length;
      if (batchSpec.mode() == AmpsConnection.BatchMode.STREAM) {
        long now = clock.nowMillis();
        long elapsedMs = Math.max(0L, now - lastFlushMs);
        flushHandler.flush(List.of(payloadLine), 1, lineBytes, elapsedMs);
        lastFlushMs = now;
        return;
      }
      if (lineBytes > batchSpec.batchMaxBytes()) {
        throw new IllegalStateException(
            "single message bytes="
                + lineBytes
                + " exceeds batchMaxBytes="
                + batchSpec.batchMaxBytes()
                + " mode="
                + batchSpec.mode());
      }
      flushIfDue();
      payloads.add(payloadLine);
      messageCount++;
      byteCount += lineBytes;
      if (messageCount >= batchSpec.batchMaxMessages() || byteCount >= batchSpec.batchMaxBytes()) {
        flush();
      }
    }

    synchronized void flushIfDue() {
      if (batchSpec.mode() != AmpsConnection.BatchMode.BATCH || messageCount <= 0) {
        return;
      }
      long now = clock.nowMillis();
      if (now - lastFlushMs >= batchSpec.batchFlushMillis()) {
        flushInternal(now);
      }
    }

    synchronized void flush() {
      if (messageCount <= 0) {
        return;
      }
      flushInternal(clock.nowMillis());
    }

    private void flushInternal(long nowMs) {
      List<String> batchPayloads = List.copyOf(payloads);
      int messages = messageCount;
      int bytes = byteCount;
      long elapsedMs = Math.max(0L, nowMs - lastFlushMs);
      payloads.clear();
      messageCount = 0;
      byteCount = 0;
      lastFlushMs = nowMs;
      flushHandler.flush(batchPayloads, messages, bytes, elapsedMs);
    }
  }
