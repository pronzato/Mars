package org.pronzato.fabric.api.control.status.service;

import com.crankuptheamps.client.Message;
import java.io.Closeable;
import java.time.Duration;
import java.util.ArrayDeque;
import java.util.Comparator;
import java.util.Deque;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.Optional;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ConcurrentMap;
import java.util.concurrent.CopyOnWriteArrayList;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.ScheduledFuture;
import java.util.concurrent.TimeUnit;
import java.util.function.Consumer;
import java.util.regex.Pattern;
import org.pronzato.fabric.api.control.ControlConnection;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryTopics;
import org.pronzato.fabric.api.control.discovery.codec.DiscoveryCodec;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionHbService;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionType;
import org.pronzato.fabric.api.util.Strings;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public final class ServiceHbCache implements Closeable {
  private static final Logger LOGGER = LoggerFactory.getLogger(ServiceHbCache.class);
  private static final String FABRIC_HB_TOPIC = DiscoveryTopics.HEARTBEAT;

  public static final class Target {
    public final String instanceId;
    public final String host;
    public final int port;
    public final boolean tls;
    public final String authorityOverride;

    public Target(String instanceId, String host, int port, boolean tls, String authorityOverride) {
      this.instanceId = Strings.emptyToNull(instanceId);
      this.host = host;
      this.port = port;
      this.tls = tls;
      this.authorityOverride = Strings.emptyToNull(authorityOverride);
    }

    @Override
    public String toString() {
      String id = instanceId == null ? "" : instanceId + " ";
      String base = id + host + ":" + port + (tls ? " (TLS)" : "");
      return authorityOverride == null ? base : base + " auth=" + authorityOverride;
    }
  }

  public enum AlertType {
    FIRST_SEEN,
    GONE,
    FLAP_SUSPECT,
    DEADMAN_TIMEOUT
  }

  public static final class Config {
    private Duration flapWindow = Duration.ofSeconds(30);
    private int flapThreshold = 4;
    private Duration deadmanTimeout = Duration.ofSeconds(20);
    private String regionFilter = null;
    private boolean notifyOnTimestampOnly = false;

    public Config withFlapWindow(Duration w) {
      this.flapWindow = w;
      return this;
    }

    public Config withFlapThreshold(int t) {
      this.flapThreshold = t;
      return this;
    }

    public Config withDeadmanTimeout(Duration t) {
      this.deadmanTimeout = t;
      return this;
    }

    public Config withRegionFilter(String regionOrNull) {
      this.regionFilter = Strings.isBlank(regionOrNull) ? null : regionOrNull;
      return this;
    }

    public Config withNotifyOnTimestampOnly(boolean v) {
      this.notifyOnTimestampOnly = v;
      return this;
    }

    Duration flapWindow() {
      return flapWindow;
    }

    int flapThreshold() {
      return flapThreshold;
    }

    Duration deadmanTimeout() {
      return deadmanTimeout;
    }

    String regionFilter() {
      return regionFilter;
    }

    boolean notifyOnTimestampOnly() {
      return notifyOnTimestampOnly;
    }
  }

  private final String group;
  private final String service;
  private final Config cfg;
  private final Pattern servicePattern;
  private final ControlConnection control;
  private final com.fasterxml.jackson.databind.ObjectMapper om = DiscoveryCodec.mapper();
  private final ConcurrentMap<String, ServiceStatusInstance> byInstance = new ConcurrentHashMap<>();
  private final ConcurrentMap<String, Long> lastSeen = new ConcurrentHashMap<>();
  private final ConcurrentMap<String, String> lastStatus = new ConcurrentHashMap<>();
  private final ConcurrentMap<String, Deque<Long>> flipTimes = new ConcurrentHashMap<>();
  private final List<Consumer<List<ServiceStatusInstance>>> snapshotListeners = new CopyOnWriteArrayList<>();
  private final List<Consumer<ServiceStatusAlert>> alertListeners = new CopyOnWriteArrayList<>();
  private final RoundRobin rr = new RoundRobin();
  private final ScheduledExecutorService ses;
  private final boolean ownsScheduler;
  private volatile boolean running = false;
  private volatile ScheduledFuture<?> deadmanTask;
  private volatile ControlConnection.Subscription subscription;

  public ServiceHbCache(
      ControlConnection control,
      String group,
      String service,
      Config cfg,
      ScheduledExecutorService scheduler) {
    this.control = Objects.requireNonNull(control, "control");
    this.group = Objects.requireNonNull(group, "group");
    this.service = Strings.isBlank(service) ? ".*" : service;
    this.cfg = cfg == null ? new Config() : cfg;
    this.servicePattern = compileServicePattern(this.service);
    if (scheduler == null) {
      this.ses = Executors.newScheduledThreadPool(1);
      this.ownsScheduler = true;
    } else {
      this.ses = scheduler;
      this.ownsScheduler = false;
    }
  }

  public void start() throws Exception {
    if (running) {
      return;
    }
    running = true;
    subscribe();
    deadmanTask = ses.scheduleAtFixedRate(this::deadmanScanSafe, 5, 5, TimeUnit.SECONDS);
  }

  private void subscribe() throws Exception {
    String filter = filterExpression();
    subscription =
        control
            .sowAndSubscribe(
                FABRIC_HB_TOPIC,
                "oof",
                filter.isBlank() ? null : filter,
                msg -> {
                  try {
                    int command = msg.raw().getCommand();
                    switch (command) {
                      case Message.Command.SOW:
                      case Message.Command.Publish:
                        onUpsert(msg.data());
                        break;
                      case Message.Command.SOWDelete:
                        onDelete(msg.data());
                        break;
                      case Message.Command.OOF:
                        notifySnapshot();
                        break;
                      default:
                        break;
                    }
                  } catch (Exception e) {
                    LOGGER.error("[ServiceHbCache] Unexpected failure processing status update", e);
                  }
                });
  }

  @Override
  public void close() {
    running = false;
    ControlConnection.Subscription currentSubscription = subscription;
    subscription = null;
    if (currentSubscription != null) {
      try {
        currentSubscription.close();
      } catch (Exception ignore) {
      }
    }
    ScheduledFuture<?> task = deadmanTask;
    deadmanTask = null;
    if (task != null) {
      task.cancel(true);
    }
    if (ownsScheduler) {
      ses.shutdownNow();
    }
  }

  public AutoCloseable onSnapshot(Consumer<List<ServiceStatusInstance>> listener) {
    Consumer<List<ServiceStatusInstance>> callback = Objects.requireNonNull(listener, "listener");
    snapshotListeners.add(callback);
    return () -> snapshotListeners.remove(callback);
  }

  public void onAlert(Consumer<ServiceStatusAlert> listener) {
    alertListeners.add(Objects.requireNonNull(listener, "listener"));
  }

  public List<ServiceStatusInstance> listUp() {
    pruneStaleInstances(System.currentTimeMillis(), false);
    String rf = cfg.regionFilter();
    return byInstance.values().stream()
        .filter(i -> "ONLINE".equalsIgnoreCase(i.status))
        .filter(i -> rf == null || rf.equalsIgnoreCase(i.region))
        .sorted(Comparator.comparing(i -> i.instanceId))
        .toList();
  }

  public Optional<ServiceStatusInstance> roundRobin() {
    List<ServiceStatusInstance> list = listUp();
    return list.isEmpty() ? Optional.empty() : Optional.of(list.get(rr.next(list.size())));
  }

  public Optional<Target> chooseTargetRoundRobin() {
    return roundRobin().map(ServiceStatusInstance::asTarget);
  }

  private void onUpsert(String json) throws Exception {
    DefinitionHbService heartbeat = om.readValue(json, DefinitionHbService.class);
    if (!matchesScope(heartbeat)) {
      return;
    }
    String instanceId = Strings.trimToNull(heartbeat.instance());
    if (instanceId == null) {
      return;
    }
    String host = Strings.trimToNull(heartbeat.advertiseHost());
    int port = heartbeat.advertisePort() == null ? 0 : heartbeat.advertisePort();
    String region = Strings.trimToNull(heartbeat.region());
    String status = "ONLINE";
    long ts =
        heartbeat.heartbeatEpochMs() == null
            ? System.currentTimeMillis()
            : heartbeat.heartbeatEpochMs();
    boolean tlsRequired = Boolean.TRUE.equals(heartbeat.tlsEnabled());
    String routeHost = Strings.trimToNull(heartbeat.routeHost());
    Integer routePort = heartbeat.routePort();
    String routeAuthority = Strings.trimToNull(heartbeat.routeAuthority());
    Map<String, String> attrs =
        heartbeat.tags() == null || heartbeat.tags().isEmpty() ? Map.of() : Map.copyOf(heartbeat.tags());
    ServiceStatusInstance newInstance =
        new ServiceStatusInstance(
            instanceId,
            host,
            port,
            region == null ? "" : region,
            ts,
            status,
            tlsRequired,
            routeHost,
            routePort,
            routeAuthority,
            attrs);
    ServiceStatusInstance old = byInstance.put(instanceId, newInstance);
    if (old == null) {
      emitAlert(new ServiceStatusAlert(AlertType.FIRST_SEEN, instanceId, host + ":" + port + " " + status));
    }
    lastSeen.put(instanceId, ts);
    handleFlip(instanceId, old == null ? null : old.status, status, ts);
    boolean diff;
    if (old == null) {
      diff = true;
    } else if (cfg.notifyOnTimestampOnly()) {
      diff = !old.equals(newInstance);
    } else {
      ServiceStatusInstance newNoTs =
          new ServiceStatusInstance(
              newInstance.instanceId,
              newInstance.host,
              newInstance.port,
              newInstance.region,
              old.tsEpochMs,
              newInstance.status,
              newInstance.tlsRequired,
              newInstance.routeHost,
              newInstance.routePort,
              newInstance.routeAuthority,
              newInstance.attrs);
      diff = !old.equals(newNoTs);
    }
    if (diff) {
      notifySnapshot();
    }
  }

  private void onDelete(String json) throws Exception {
    DefinitionHbService heartbeat = om.readValue(json, DefinitionHbService.class);
    String instanceId = Strings.trimToNull(heartbeat.instance());
    if (instanceId == null) {
      return;
    }
    if (!matchesScope(heartbeat) && !byInstance.containsKey(instanceId)) {
      return;
    }
    ServiceStatusInstance removed = byInstance.remove(instanceId);
    lastSeen.remove(instanceId);
    lastStatus.remove(instanceId);
    flipTimes.remove(instanceId);
    if (removed != null) {
      emitAlert(new ServiceStatusAlert(AlertType.GONE, instanceId, "removed"));
      notifySnapshot();
    }
  }

  private void handleFlip(String instanceId, String prev, String cur, long whenMs) {
    if (prev == null || Objects.equals(prev, cur)) {
      lastStatus.put(instanceId, cur);
      return;
    }
    lastStatus.put(instanceId, cur);
    Deque<Long> q = flipTimes.computeIfAbsent(instanceId, ignored -> new ArrayDeque<>());
    q.addLast(whenMs);
    long cutoff = whenMs - cfg.flapWindow().toMillis();
    while (!q.isEmpty() && q.peekFirst() < cutoff) {
      q.removeFirst();
    }
    if (q.size() >= cfg.flapThreshold()) {
      emitAlert(
          new ServiceStatusAlert(
              AlertType.FLAP_SUSPECT,
              instanceId,
              "flips=" + q.size() + " window=" + cfg.flapWindow().toSeconds() + "s"));
    }
  }

  private void deadmanScanSafe() {
    try {
      deadmanScan();
    } catch (Throwable t) {
      LOGGER.error("[ServiceHbCache] Deadman scan failed", t);
    }
  }

  private void deadmanScan() {
    pruneStaleInstances(System.currentTimeMillis(), true);
  }

  private void pruneStaleInstances(long nowEpochMs, boolean emitTimeoutAlerts) {
    long limitMs = cfg.deadmanTimeout().toMillis();
    if (limitMs <= 0L) {
      return;
    }
    boolean changed = false;
    for (var entry : lastSeen.entrySet()) {
      String instanceId = entry.getKey();
      long lastSeenEpochMs = entry.getValue() == null ? 0L : entry.getValue();
      long ageMs = nowEpochMs - lastSeenEpochMs;
      if (ageMs <= limitMs) {
        continue;
      }
      if (emitTimeoutAlerts) {
        emitAlert(
            new ServiceStatusAlert(
                AlertType.DEADMAN_TIMEOUT, instanceId, "ageMs=" + ageMs + " > " + limitMs));
      }
      ServiceStatusInstance removed = byInstance.remove(instanceId);
      lastSeen.remove(instanceId);
      lastStatus.remove(instanceId);
      flipTimes.remove(instanceId);
      if (removed != null) {
        if (emitTimeoutAlerts && LOGGER.isDebugEnabled()) {
          LOGGER.debug(
              "[ServiceHbCache][stale-evict] instance={} endpoint={}:{} ageMs={} limitMs={}",
              instanceId,
              removed.host,
              removed.port,
              ageMs,
              limitMs);
        }
        changed = true;
      }
    }
    if (changed) {
      notifySnapshot();
    }
  }

  private boolean matchesScope(DefinitionHbService heartbeat) {
    if (heartbeat == null) {
      return false;
    }
    if (heartbeat.type() != DefinitionType.HEARTBEAT_SERVICE) {
      return false;
    }
    String messageGroup = Strings.trimToNull(heartbeat.group());
    if (messageGroup == null || !messageGroup.equals(group)) {
      return false;
    }
    String svc = Strings.trimToNull(heartbeat.name());
    if (Strings.isBlank(svc)) {
      return matchAllServices();
    }
    return matchesService(svc);
  }

  private String filterExpression() {
    return "/group = '"
        + escapeFilterValue(group)
        + "' and /type = '"
        + DefinitionType.HEARTBEAT_SERVICE.name()
        + "'";
  }

  private static String escapeFilterValue(String value) {
    return value == null ? "" : value.replace("'", "''");
  }

  private boolean matchesService(String value) {
    String candidate = value == null ? "" : value;
    return servicePattern.matcher(candidate).matches();
  }

  private boolean matchAllServices() {
    return ".*".equals(servicePattern.pattern());
  }

  private static Pattern compileServicePattern(String raw) {
    if (Strings.isBlank(raw) || raw.equals("*") || raw.equals(".*")) {
      return Pattern.compile(".*");
    }
    if (raw.contains("*") || raw.contains("?")) {
      return Pattern.compile(globToRegex(raw));
    }
    return Pattern.compile(Pattern.quote(raw));
  }

  private static String globToRegex(String glob) {
    StringBuilder regex = new StringBuilder();
    for (int i = 0; i < glob.length(); i++) {
      char c = glob.charAt(i);
      switch (c) {
        case '*':
          regex.append(".*");
          break;
        case '?':
          regex.append('.');
          break;
        case '.':
          regex.append("\\.");
          break;
        case '\\':
          regex.append("\\\\");
          break;
        default:
          if ("+()[]{}^$|".indexOf(c) >= 0) {
            regex.append('\\');
          }
          regex.append(c);
      }
    }
    return regex.toString();
  }

  private void notifySnapshot() {
    if (snapshotListeners.isEmpty()) {
      return;
    }
    List<ServiceStatusInstance> snapshot = List.copyOf(listUp());
    for (Consumer<List<ServiceStatusInstance>> listener : snapshotListeners) {
      safe(() -> listener.accept(snapshot));
    }
  }

  private void emitAlert(ServiceStatusAlert alert) {
    for (Consumer<ServiceStatusAlert> listener : alertListeners) {
      safe(() -> listener.accept(alert));
    }
  }

  private static void safe(Runnable runnable) {
    try {
      runnable.run();
    } catch (Throwable ignore) {
    }
  }

  private static final class RoundRobin {
    private int index = -1;

    synchronized int next(int size) {
      index = (index + 1) % size;
      return index;
    }
  }
}
