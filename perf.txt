  private DiscoveryCache(
      ControlConnection control,
      String sessionGroup,
      boolean adminMode,
      boolean verbose,
      boolean offlineMode) {
    if (!offlineMode) {
      this.control = Objects.requireNonNull(control, "control");
    } else {
      this.control = control;
    }
    this.sessionGroup = normalize(sessionGroup);
    this.adminMode = adminMode;
    this.verbose = verbose;
    this.offlineMode = offlineMode;
    this.discoveryScope = defaultScope();
  }

  public static DiscoveryCache recoveredSnapshotCache(
      String sessionGroup, boolean adminMode, boolean verbose, List<Definition> definitions) {
    DiscoveryCache cache = new DiscoveryCache(null, sessionGroup, adminMode, verbose, true);
    cache.replaceWithRecoveredSnapshot(definitions);
    return cache;
  }

  public void start() throws Exception {
    if (offlineMode) {
      started.set(true);
      recovered.set(true);
      recoveryLatch.countDown();
      return;
    }
    if (!started.compareAndSet(false, true)) {
      awaitRecovery(Duration.ofSeconds(30));
      return;
    }
    openSubscription(Duration.ofSeconds(30), true);
  }

  public boolean recovered() {
    return recovered.get();
  }

  public boolean offlineMode() {
    return offlineMode;
  }

  public int replaceWithRecoveredSnapshot(List<Definition> snapshotDefinitions) {
    List<Definition> source = snapshotDefinitions == null ? List.of() : snapshotDefinitions;
    synchronized (subscriptionLock) {
      closeSubscriptionQuietly();
      definitions.clear();
      int loaded = 0;
      for (Definition definition : source) {
        if (definition == null) {
          throw new IllegalArgumentException("Snapshot contains a null definition");
        }
        if (!shouldAcceptDefinition(definition)) {
          continue;
        }
        Key key = requireDefinitionKey(definition);
        cacheUpsert(key, definition);
        loaded++;
      }
      started.set(true);
      recovered.set(true);
      recoveryLatch.countDown();
      return loaded;
    }
  }

  public int size() {
    return definitions.size();
  }

  public DiscoveryScope discoveryScope() {
    return discoveryScope;
  }

  public DiscoveryScope updateDiscoveryScope(DiscoveryScope nextScope) {
    DiscoveryScope resolvedScope = nextScope == null ? defaultScope() : nextScope;
    DiscoveryScope previous = discoveryScope;
    if (Objects.equals(previous, resolvedScope)) {
      return previous;
    }
    discoveryScope = resolvedScope;
    evictOutOfScopeDefinitions(resolvedScope);
    if (started.get() && !offlineMode) {
      openSubscription(Duration.ofSeconds(10), false);
    }
    return previous;
  }

  public List<Definition> allDefinitions() {
    return new ArrayList<>(definitions.values());
  }

  public <T extends Definition> List<T> list(Class<T> clazz) {
    return list(clazz, null);
  }

  public <T extends Definition> List<T> list(Class<T> clazz, String group) {
    if (clazz == null) {
      return List.of();
    }
    DefinitionType type = toType(clazz);
    if (type == DefinitionType.UNKNOWN) {
      return List.of();
    }
    String normalizedGroup = normalize(group);
    List<T> result = new ArrayList<>();
    for (Definition definition : definitions.values()) {
      if (definition == null || definitionType(definition) != type) {
        continue;
      }
      if (normalizedGroup != null && !normalizedGroup.equals(normalize(definitionGroup(definition)))) {
        continue;
      }
      if (clazz.isInstance(definition)) {
        result.add(clazz.cast(definition));
      }
    }
    return result;
  }

  public Definition get(DefinitionType type, String group, String name) {
    return definitions.get(new Key(type, normalize(group), normalize(name)));
  }

  public <T extends Definition> Optional<T> get(Class<T> clazz, String group, String name) {
    return getOrFetch(clazz, group, name);
  }

  <T extends Definition> Optional<T> getOrFetch(Class<T> clazz, String group, String name) {
    if (clazz == null) {
      return Optional.empty();
    }
    DefinitionType type = toType(clazz);
    if (type == DefinitionType.UNKNOWN) {
      return Optional.empty();
    }
    String normalizedGroup = normalize(group);
    String normalizedName = normalize(name);
    if (normalizedGroup == null || normalizedName == null) {
      return Optional.empty();
    }
    Definition definition = get(type, normalizedGroup, normalizedName);
    if (definition == null) {
      Optional<Definition> fetched = fetchOnMiss(type, normalizedGroup, normalizedName);
      if (fetched.isEmpty()) {
        return Optional.empty();
      }
      definition = fetched.get();
    }
    if (!clazz.isInstance(definition)) {
      return Optional.empty();
    }
    return Optional.of(clazz.cast(definition));
  }

  public Optional<DefinitionSchema> getSchema(SchemaReference ref) {
    if (ref == null) {
      return Optional.empty();
    }
    String group = normalize(ref.group());
    String name = normalize(ref.name());
    if (group == null || name == null) {
      return Optional.empty();
    }
    Definition definition = get(DefinitionType.SCHEMA, group, name);
    if (definition instanceof DefinitionSchema schema) {
      return Optional.of(schema);
    }
    return Optional.empty();
  }

  public List<DefinitionSchema> listSchemas(String group) {
    List<DefinitionSchema> schemas = new ArrayList<>();
    for (DefinitionSchema schema : list(DefinitionSchema.class, group)) {
      if (schema == null || schema.schemaType() == null) {
        continue;
      }
      schemas.add(schema);
    }
    schemas.sort(SCHEMA_NAME_ORDER);
    return List.copyOf(schemas);
  }

  public List<DefinitionSchema> listSchemas(String group, DefinitionSchema.SchemaType type) {
    DefinitionSchema.SchemaType normalizedType = Objects.requireNonNull(type, "schemaType");
    List<DefinitionSchema> filtered = new ArrayList<>();
    for (DefinitionSchema schema : listSchemas(group)) {
      if (schema.schemaType() == normalizedType) {
        filtered.add(schema);
      }
    }
    return List.copyOf(filtered);
  }

  public void addListener(DiscoveryListener listener) {
    if (listener == null) {
      return;
    }
    listeners.addIfAbsent(listener);
  }

  public void removeListener(DiscoveryListener listener) {
    if (listener == null) {
      return;
    }
    listeners.remove(listener);
  }

  public AutoCloseable registerListener(
      DefinitionType type, BiConsumer<ConfigKey, Definition> listener) {
    Objects.requireNonNull(type, "type");
    Objects.requireNonNull(listener, "listener");
    if (type == DefinitionType.UNKNOWN) {
      throw new IllegalArgumentException("type is required");
    }
    typedUpsertListeners
        .computeIfAbsent(type, ignored -> new CopyOnWriteArrayList<>())
        .add(listener);
    return () -> unregisterListener(type, listener);
  }

  public ConfigCacheTx beginTx(String context) {
    return new ConfigCacheTx(this, Strings.requireNonBlank(context, "context"));
  }

  public void upsertTx(ConfigCacheTx tx, Definition definition) {
    Objects.requireNonNull(definition, "definition");
    ConfigCacheTx activeTx = requireTx(tx);
    Key key = requireDefinitionKey(definition);
    activeTx.stageUpsert(key, definitions.get(key), definition);
  }

  public void removeTx(
      ConfigCacheTx tx, Class<? extends Definition> type, String group, String name) {
    ConfigCacheTx activeTx = requireTx(tx);
    Key key = requireKey(type, group, name);
    activeTx.stageDelete(key, definitions.get(key));
  }

  public void commitTx(ConfigCacheTx tx) {
    ConfigCacheTx activeTx = requireTx(tx);
    activeTx.requireCommittable();
    List<ConfigCacheTx.WriteOp> ordered = activeTx.orderedWrites(TX_KEY_ORDER);
    if (ordered.isEmpty()) {
      activeTx.markCommitted();
      return;
    }
    RuntimeException originalFailure = null;
    for (ConfigCacheTx.WriteOp op : ordered) {
      try {
        applyWriteOp(op);
        activeTx.markApplied(op.key());
      } catch (RuntimeException ex) {
        originalFailure = txCommitFailure(activeTx.context(), op.key(), ex);
        break;
      }
    }
    if (originalFailure == null) {
      activeTx.markCommitted();
      return;
    }
    try {
      rollbackApplied(activeTx);
    } catch (RuntimeException rollbackFailure) {
      originalFailure.addSuppressed(rollbackFailure);
    }
    throw originalFailure;
  }

  public void rollbackTx(ConfigCacheTx tx) {
    ConfigCacheTx activeTx = requireTx(tx);
    if (activeTx.isCompleted()) {
      return;
    }
    rollbackApplied(activeTx);
  }

  public void upsert(Definition definition) {
    Objects.requireNonNull(definition, "definition");
    DefinitionType type = definitionType(definition);
    if (type == null || type == DefinitionType.UNKNOWN) {
      type = toType(definition.getClass());
    }
    if (type == null || type == DefinitionType.UNKNOWN) {
      throw new IllegalArgumentException("Unsupported definition type: " + definition.getClass().getName());
    }
    String group = normalize(definitionGroup(definition));
    String name = normalize(definitionName(definition));
    if (group == null || name == null) {
      throw new IllegalArgumentException("definition envelope requires group/name");
    }
    String payload = serializeDefinition(definition);
    try {
      ControlConnection activeControl = requireControlConnection("upsert");
      activeControl.amps().publish(DiscoveryTopics.DISCOVERY, payload);
      if (shouldAcceptDefinition(definition)) {
        Key key = new Key(type, group, name);
        cacheUpsert(key, definition);
      }
    } catch (Exception ex) {
      throw new IllegalStateException(
          "Failed to publish definition " + type + " " + group + "/" + name, ex);
    }
  }

  public void delete(DefinitionType type, String group, String name) {
    Objects.requireNonNull(type, "type");
    if (type == DefinitionType.UNKNOWN) {
      throw new IllegalArgumentException("type is required");
    }
    String normalizedGroup = normalize(group);
    String normalizedName = normalize(name);
    if (normalizedGroup == null || normalizedName == null) {
      throw new IllegalArgumentException("group/name are required");
    }
    Command command = new Command("sow_delete");
    command.setTopic(DiscoveryTopics.DISCOVERY);
    command.setFilter(
        "(/envelope/group='"
            + escape(normalizedGroup)
            + "' AND /envelope/type='"
            + escape(type.name())
            + "' AND /envelope/name='"
            + escape(normalizedName)
            + "')");
    ControlConnection activeControl = requireControlConnection("delete");
    try (MessageStream ignored = activeControl.amps().client().execute(command)) {
      Key key = new Key(type, normalizedGroup, normalizedName);
      definitions.remove(key);
      logDefinitionChange("delete", key);
      notifyDelete(
          new DefinitionEnvelope(
              normalizedGroup,
              type,
              normalizedName,
              null,
              null,
              null,
              null,
              null,
              null,
              null));
    } catch (Exception ex) {
      throw new IllegalStateException(
          "Failed to delete definition " + type + " " + normalizedGroup + "/" + normalizedName, ex);
    }
  }

  public void stop() {
    synchronized (subscriptionLock) {
      closeSubscriptionQuietly();
    }
  }

  @Override
  public void close() {
    stop();
  }

  private DiscoveryScope defaultScope() {
    return DiscoveryScope.primaryOnly(sessionGroup, null, null);
  }

  private void openSubscription(Duration timeout, boolean failOnTimeout) {
    synchronized (subscriptionLock) {
      ControlConnection activeControl = requireControlConnection("subscription");
      recovered.set(false);
      recoveryLatch = new CountDownLatch(1);
      String filter = subscriptionFilter();
      boolean replaced = false;
      if (subscription != null) {
        try {
          subscription.replaceFilter(filter);
          replaced = true;
          if (verbose) {
            LOGGER.info("[DiscoveryCache] context=scope-refresh mode=replace filter={}", safe(filter));
          }
        } catch (Exception ex) {
          if (verbose) {
            LOGGER.warn(
                "[DiscoveryCache] context=scope-refresh mode=replace-failed reason={}",
                Strings.firstNonBlank(ex.getMessage(), ex.getClass().getSimpleName()));
          }
          closeSubscriptionQuietly();
        }
      }
      if (!replaced) {
        closeSubscriptionQuietly();
        Command command = new Command(Message.Command.SOWAndSubscribe);
        command.setTopic(DiscoveryTopics.DISCOVERY);
        command.setOptions("oof");
        if (filter != null) {
          command.setFilter(filter);
        }
        try {
          subscription =
              activeControl.sowAndSubscribe(
                  command.getTopic(),
                  command.getOptions(),
                  command.getFilter(),
                  message -> handleMessage(message.raw()));
        } catch (Exception ex) {
          throw new IllegalStateException(
              "Failed to subscribe discovery cache topic with scope filter", ex);
        }
      }
    }
    if (failOnTimeout) {
      awaitRecovery(timeout);
      return;
    }
    try {
      awaitRecovery(timeout);
    } catch (RuntimeException ex) {
      if (verbose) {
        LOGGER.warn("[DiscoveryCache] scope subscription refresh timed out: {}", Strings.firstNonBlank(ex.getMessage(), ex.getClass().getSimpleName()));
      }
    }
  }

  private void closeSubscriptionQuietly() {
    ControlConnection.Subscription id = subscription;
    if (id == null) {
      return;
    }
    try {
      id.close();
    } catch (Exception ignore) {
    }
    subscription = null;
  }

  private void awaitRecovery(Duration timeout) {
    long millis = timeout == null ? 30000L : Math.max(1L, timeout.toMillis());
    try {
      boolean done = recoveryLatch.await(millis, TimeUnit.MILLISECONDS);
      if (!done) {
        throw new IllegalStateException(
            "DiscoveryCache initial recovery did not complete (GroupEnd not observed)");
      }
    } catch (InterruptedException ex) {
      Thread.currentThread().interrupt();
      throw new IllegalStateException("Interrupted while waiting for DiscoveryCache recovery", ex);
    }
  }

  private void handleMessage(Message message) {
    if (message == null) {
      return;
    }
    int command = message.getCommand();
    if (command == Message.Command.GroupEnd) {
      recovered.set(true);
      recoveryLatch.countDown();
      if (verbose) {
        LOGGER.info("[DiscoveryCache] context=recovered definitions={}", definitions.size());
      }
      return;
    }
    if (command != Message.Command.SOW
        && command != Message.Command.Publish
        && command != Message.Command.OOF) {
      return;
    }
    String payload = message.getData();
    if (Strings.isBlank(payload)) {
      return;
    }
    final Definition definition;
    try {
      definition = DiscoveryCodec.decode(payload);
    } catch (DefinitionDecodeException ex) {
      logDecodeFailureWithDedupe(ex);
      return;
    } catch (RuntimeException ex) {
      return;
    }
    if (definition == null || !shouldAcceptDefinition(definition)) {
      return;
    }
    DefinitionType type = definitionType(definition);
    if (type == null || type == DefinitionType.UNKNOWN) {
      type = toType(definition.getClass());
    }
    if (type == null || type == DefinitionType.UNKNOWN) {
      return;
    }
    Key key = new Key(type, normalize(definitionGroup(definition)), normalize(definitionName(definition)));
    if (key.group() == null || key.name() == null) {
      return;
    }
    if (command == Message.Command.OOF) {
      definitions.remove(key);
      logDefinitionChange("remove", key);
      notifyDelete(definition.envelope());
      return;
    }
    cacheUpsert(key, definition);
  }

  private void logDecodeFailureWithDedupe(DefinitionDecodeException ex) {
    if (ex == null) {
      return;
    }
    DefinitionEnvelope env = ex.getEnvelope();
    String key =
        safe(env == null ? null : env.group())
            + "|"
            + safe(env == null ? null : env.type())
            + "|"
            + safe(env == null ? null : env.name())
            + "|"
            + Integer.toHexString(Objects.hashCode(ex.getMessage()));
    long now = System.currentTimeMillis();
    Long last = DECODE_ERROR_LAST_LOG.get(key);
    if (last != null && now - last <= DECODE_ERROR_WINDOW_MS) {
      return;
    }
    DECODE_ERROR_LAST_LOG.put(key, now);
    String payloadPretty = pretty(ex.getPayload());
    LOGGER.error("[DiscoveryCache][DecodeFailed] group={} type={} name={} uuid={} payload=\n{}", safe(env == null ? null : env.group()), safe(env == null ? null : env.type()), safe(env == null ? null : env.name()), safe(env == null ? null : env.uuid()), payloadPretty, ex);
  }

  private boolean shouldAcceptDefinition(Definition definition) {
    if (definition == null) {
      return false;
    }
    DefinitionType type = definitionType(definition);
    if (type == null || type == DefinitionType.UNKNOWN) {
      type = toType(definition.getClass());
    }
    if (!isCacheableType(type)) {
      return false;
    }
    if (adminMode) {
      return true;
    }
    if (type == DefinitionType.ENTITLEMENT) {
      return true;
    }
    return isGroupVisible(normalize(definitionGroup(definition)), discoveryScope);
  }

  private String subscriptionFilter() {
    if (adminMode) {
      return null;
    }
    DiscoveryScope scope = discoveryScope == null ? defaultScope() : discoveryScope;
    List<String> groups = new ArrayList<>(scope.visibleGroups());
    if (groups.stream().noneMatch(value -> PLATFORM_GROUP.equalsIgnoreCase(value))) {
      groups.add(PLATFORM_GROUP);
    }
    StringBuilder groupFilter = new StringBuilder();
    for (String group : groups) {
      String normalized = normalize(group);
      if (normalized == null) {
        continue;
      }
      if (groupFilter.length() > 0) {
        groupFilter.append(" OR ");
      }
      groupFilter.append("(/envelope/group='").append(escape(normalized)).append("')");
    }
    if (groupFilter.length() == 0) {
      return "(/envelope/type='" + ENTITLEMENT_TYPE + "')";
    }
    return "(/envelope/type='" + ENTITLEMENT_TYPE + "' OR " + groupFilter + ")";
  }

  private void evictOutOfScopeDefinitions(DiscoveryScope scope) {
    if (adminMode || scope == null) {
      return;
    }
    List<Key> evictions = new ArrayList<>();
    for (Map.Entry<Key, Definition> entry : definitions.entrySet()) {
      Definition definition = entry.getValue();
      if (definition == null) {
        continue;
      }
      if (shouldRetainDefinition(definition, scope)) {
        continue;
      }
      evictions.add(entry.getKey());
    }
    for (Key key : evictions) {
      removeCached(key);
    }
  }

  private boolean shouldRetainDefinition(Definition definition, DiscoveryScope scope) {
    DefinitionType type = definitionType(definition);
    if (type == null || type == DefinitionType.UNKNOWN) {
      type = toType(definition.getClass());
    }
    if (type == DefinitionType.ENTITLEMENT) {
      return true;
    }
    return isGroupVisible(normalize(definitionGroup(definition)), scope);
  }

  private boolean isGroupVisible(String group, DiscoveryScope scope) {
    if (group == null) {
      return false;
    }
    DiscoveryScope effectiveScope = scope == null ? defaultScope() : scope;
    return effectiveScope.includesGroup(group);
  }

  private static String pretty(JsonNode node) {
    if (node == null) {
      return "";
    }
    try {
      return DiscoveryCodec.mapper().writerWithDefaultPrettyPrinter().writeValueAsString(node);
    } catch (Exception ex) {
      return node.toString();
    }
  }

  private static String serializeDefinition(Definition definition) {
    try {
      return DiscoveryCodec.mapper().writerWithDefaultPrettyPrinter().writeValueAsString(definition);
    } catch (Exception ex) {
      throw new IllegalArgumentException("Unable to serialize definition", ex);
    }
  }

  private static String escape(String value) {
    return value == null ? "" : value.replace("'", "''");
  }

    return DiscoveryCacheType.schemaName(schema);
  }

    return DiscoveryCacheType.schemaTypeName(schema);
  }

  private static String txTypeToken(Key key) {
    return key == null || key.type() == null ? "" : key.type().name();
  }

  private static String txGroupToken(Key key) {
    return key == null ? "" : Strings.firstNonBlank(key.group(), "");
  }

  private static String txNameToken(Key key) {
    return key == null ? "" : Strings.firstNonBlank(key.name(), "");
  }

  private void logDefinitionChange(String context, Key key) {
    if (!verbose) {
      return;
    }
    LOGGER.info("[DiscoveryCache] context={} definitions={} key={}", context, definitions.size(), formatKey(key));
  }

  private static String formatKey(Key key) {
    if (key == null) {
      return "unknown";
    }
    return key.type() + ":" + key.group() + "/" + key.name();
  }

  private void notifyUpsert(Definition definition) {
    if (definition == null || listeners.isEmpty()) {
      return;
    }
    for (DiscoveryListener listener : listeners) {
      try {
        listener.onUpsert(definition);
      } catch (Exception ex) {
        LOGGER.warn("[DiscoveryCache] listener upsert failed: {}", ex.toString());
      }
    }
  }

  private void notifyDelete(DefinitionEnvelope envelope) {
    if (envelope == null || listeners.isEmpty()) {
      return;
    }
    for (DiscoveryListener listener : listeners) {
      try {
        listener.onDelete(envelope);
      } catch (Exception ex) {
        LOGGER.warn("[DiscoveryCache] listener delete failed: {}", ex.toString());
      }
    }
  }

  private void unregisterListener(
      DefinitionType type, BiConsumer<ConfigKey, Definition> listener) {
    CopyOnWriteArrayList<BiConsumer<ConfigKey, Definition>> registered = typedUpsertListeners.get(type);
    if (registered == null) {
      return;
    }
    registered.remove(listener);
    if (registered.isEmpty()) {
      typedUpsertListeners.remove(type, registered);
    }
  }

  private void notifyTypedUpsert(Key key, Definition definition) {
    if (key == null || definition == null) {
      return;
    }
    List<BiConsumer<ConfigKey, Definition>> listenersByType = typedUpsertListeners.get(key.type());
    if (listenersByType == null || listenersByType.isEmpty()) {
      return;
    }
    ConfigKey configKey = new ConfigKey(key.type(), key.group(), key.name());
    for (BiConsumer<ConfigKey, Definition> listener : listenersByType) {
      try {
        listener.accept(configKey, definition);
      } catch (Exception ex) {
        LOGGER.warn("[DiscoveryCache] typed listener upsert failed type={} key={} message={}", key.type(), formatKey(key), ex.toString());
      }
    }
  }

  private Optional<Definition> fetchOnMiss(DefinitionType type, String group, String name) {
    Key key = new Key(type, group, name);
    Definition cached = definitions.get(key);
    if (cached != null) {
      return Optional.of(cached);
    }

    CompletableFuture<Optional<Definition>> mine = new CompletableFuture<>();
    CompletableFuture<Optional<Definition>> inFlight = inFlightFetches.putIfAbsent(key, mine);
    if (inFlight != null) {
      try {
        return inFlight.join();
      } catch (RuntimeException ex) {
        return Optional.empty();
      }
    }

    try {
      Optional<Definition> fetched = fetchSingle(type, group, name);
      mine.complete(fetched);
      return fetched;
    } catch (Exception ex) {
      mine.complete(Optional.empty());
      LOGGER.warn("[DiscoveryCache] context=fetch-miss key={} message={}", formatKey(key), ex.toString());
      return Optional.empty();
    } finally {
      inFlightFetches.remove(key, mine);
    }
  }

  private Optional<Definition> fetchSingle(DefinitionType type, String group, String name) throws Exception {
    Key requestedKey = new Key(type, group, name);
    Definition cached = definitions.get(requestedKey);
    if (cached != null) {
      return Optional.of(cached);
    }

    Command command = new Command(Message.Command.SOW);
    command.setTopic(DiscoveryTopics.DISCOVERY);
    command.setFilter(
        "(/envelope/group='"
            + escape(group)
            + "' AND /envelope/type='"
            + escape(type.name())
            + "' AND /envelope/name='"
            + escape(name)
            + "')");

    ControlConnection activeControl = requireControlConnection("fetch-single");
    try (MessageStream stream = activeControl.amps().client().execute(command)) {
      for (Message message : stream) {
        if (message == null) {
          continue;
        }
        int commandType = message.getCommand();
        if (commandType != Message.Command.SOW && commandType != Message.Command.Publish) {
          continue;
        }
        String payload = message.getData();
        if (Strings.isBlank(payload)) {
          continue;
        }
        Definition definition;
        try {
          definition = DiscoveryCodec.decode(payload);
        } catch (DefinitionDecodeException ex) {
          logDecodeFailureWithDedupe(ex);
          continue;
        } catch (RuntimeException ex) {
          continue;
        }
        if (definition == null || !shouldAcceptDefinition(definition)) {
          continue;
        }
        DefinitionType foundType = definitionType(definition);
        if (foundType == null || foundType == DefinitionType.UNKNOWN) {
          foundType = toType(definition.getClass());
        }
        if (foundType != type) {
          continue;
        }
        String foundGroup = normalize(definitionGroup(definition));
        String foundName = normalize(definitionName(definition));
        if (!group.equals(foundGroup) || !name.equals(foundName)) {
          continue;
        }
        Key foundKey = new Key(foundType, foundGroup, foundName);
        cacheUpsert(foundKey, definition);
        return Optional.of(definition);
      }
    }
    return Optional.ofNullable(definitions.get(requestedKey));
  }

  private void cacheUpsert(Key key, Definition definition) {
    definitions.put(key, definition);
    logDefinitionChange("upsert", key);
    notifyUpsert(definition);
    notifyTypedUpsert(key, definition);
  }

  private void removeCached(Key key) {
    Definition removed = definitions.remove(key);
    if (removed == null) {
      return;
    }
    logDefinitionChange("remove", key);
    DefinitionEnvelope envelope = definitionEnvelope(removed);
    if (envelope != null) {
      notifyDelete(envelope);
      return;
    }
    notifyDelete(
        new DefinitionEnvelope(
            key.group(),
            key.type(),
            key.name(),
            null,
            null,
            null,
            null,
            null,
            null,
            null));
  }

  private ConfigCacheTx requireTx(ConfigCacheTx tx) {
    ConfigCacheTx activeTx = Objects.requireNonNull(tx, "tx");
    activeTx.requireOwner(this);
    return activeTx;
  }

  private void applyWriteOp(ConfigCacheTx.WriteOp op) {
    if (op.delete()) {
      delete(op.key().type(), op.key().group(), op.key().name());
      return;
    }
    upsert(op.newValue());
  }

  private void rollbackApplied(ConfigCacheTx tx) {
    RuntimeException failure = null;
    List<ConfigCacheTx.WriteOp> ordered = tx.orderedWrites(TX_KEY_ORDER);
    for (ConfigCacheTx.WriteOp op : ordered) {
      if (!tx.wasApplied(op.key())) {
        continue;
      }
      try {
        Definition snapshot = op.oldValueSnapshot();
        if (snapshot == null) {
          removeCached(op.key());
        } else {
          cacheUpsert(op.key(), snapshot);
        }
      } catch (RuntimeException ex) {
        if (failure == null) {
          failure =
              new IllegalStateException(
                  "DiscoveryCache tx rollback failed context="
                      + tx.context()
                      + " key="
                      + formatKey(op.key()),
                  ex);
        } else {
          failure.addSuppressed(ex);
        }
      }
    }
    tx.markRolledBack();
    if (failure != null) {
      throw failure;
    }
  }

  private static RuntimeException txCommitFailure(
      String context, Key key, RuntimeException original) {
    String message =
        "DiscoveryCache tx commit failed context="
            + context
            + " key="
            + formatKey(key)
            + " reason="
            + Strings.firstNonBlank(
                Strings.trimToNull(original.getMessage()),
                original.getClass().getSimpleName());
    if (original instanceof IllegalStateException
        && message.equals(Strings.trimToNull(original.getMessage()))) {
      return original;
    }
    return new IllegalStateException(message, original);
  }

  private static Key requireKey(
      Class<? extends Definition> clazz, String group, String name) {
    Class<? extends Definition> normalizedClass = Objects.requireNonNull(clazz, "type");
    DefinitionType type = toType(normalizedClass);
    if (type == DefinitionType.UNKNOWN) {
      throw new IllegalArgumentException(
          "Unsupported definition class: " + normalizedClass.getName());
    }
    String normalizedGroup = normalize(group);
    String normalizedName = normalize(name);
    if (normalizedGroup == null || normalizedName == null) {
      throw new IllegalArgumentException("group/name are required");
    }
    return new Key(type, normalizedGroup, normalizedName);
  }

  private Key requireDefinitionKey(Definition definition) {
    Objects.requireNonNull(definition, "definition");
    DefinitionType type = definitionType(definition);
    if (type == null || type == DefinitionType.UNKNOWN) {
      type = toType(definition.getClass());
    }
    if (type == null || type == DefinitionType.UNKNOWN) {
      throw new IllegalArgumentException(
          "Unsupported definition type: " + definition.getClass().getName());
    }
    String group = normalize(definitionGroup(definition));
    String name = normalize(definitionName(definition));
    if (group == null || name == null) {
      throw new IllegalArgumentException("definition envelope requires group/name");
    }
    return new Key(type, group, name);
  }

  private static Map<DefinitionType, Class<? extends Definition>> cacheableDefinitions() {
    return DiscoveryCacheType.cacheableDefinitions();
  }

  private ControlConnection requireControlConnection(String operation) {
    ControlConnection active = control;
    if (active != null) {
      return active;
    }
    throw new IllegalStateException(
        "DiscoveryCache control connection is unavailable for "
            + Strings.firstNonBlank(operation, "operation")
            + " (recovered snapshot mode)");
  }

  private static Map<Class<? extends Definition>, DefinitionType> classToType() {
    return DiscoveryCacheType.classToType(CACHEABLE_DEFINITIONS);
  }

  public static final class ConfigCacheTx implements AutoCloseable {
    private final DiscoveryCache owner;
    private final String context;
    private final LinkedHashMap<Key, WriteOp> writesByKey = new LinkedHashMap<>();
    private final Set<Key> appliedKeys = new HashSet<>();
    private boolean committed;
    private boolean rolledBack;

    private ConfigCacheTx(DiscoveryCache owner, String context) {
      this.owner = Objects.requireNonNull(owner, "owner");
      this.context = Strings.requireNonBlank(context, "context");
    }

    @Override
    public void close() {
      if (isCompleted()) {
        return;
      }
      owner.rollbackTx(this);
    }

    private void requireOwner(DiscoveryCache expected) {
      if (owner != expected) {
        throw new IllegalArgumentException("DiscoveryCache tx owner mismatch context=" + context);
      }
    }

    private void requireCommittable() {
      if (committed) {
        throw new IllegalStateException("DiscoveryCache tx is already committed context=" + context);
      }
      if (rolledBack) {
        throw new IllegalStateException("DiscoveryCache tx is already rolled back context=" + context);
      }
    }

    private boolean isCompleted() {
      return committed || rolledBack;
    }

    private String context() {
      return context;
    }

    private void markCommitted() {
      committed = true;
      rolledBack = false;
      writesByKey.clear();
      appliedKeys.clear();
    }

    private void markRolledBack() {
      rolledBack = true;
      committed = false;
      writesByKey.clear();
      appliedKeys.clear();
    }

    private void markApplied(Key key) {
      appliedKeys.add(key);
    }

    private boolean wasApplied(Key key) {
      return appliedKeys.contains(key);
    }

    private void stageUpsert(Key key, Definition oldSnapshot, Definition newValue) {
      requireCommittable();
      Objects.requireNonNull(key, "key");
      Objects.requireNonNull(newValue, "definition");
      WriteOp existing = writesByKey.get(key);
      Definition snapshot = existing == null ? oldSnapshot : existing.oldValueSnapshot();
      writesByKey.put(key, new WriteOp(key, snapshot, newValue, false));
    }

    private void stageDelete(Key key, Definition oldSnapshot) {
      requireCommittable();
      Objects.requireNonNull(key, "key");
      WriteOp existing = writesByKey.get(key);
      Definition snapshot = existing == null ? oldSnapshot : existing.oldValueSnapshot();
      writesByKey.put(key, new WriteOp(key, snapshot, null, true));
    }

    private List<WriteOp> orderedWrites(Comparator<Key> comparator) {
      List<WriteOp> ordered = new ArrayList<>(writesByKey.values());
      ordered.sort((left, right) -> comparator.compare(left.key(), right.key()));
      return ordered;
    }

    public static final class WriteOp {
      private final Key key;
      private final Definition oldValueSnapshot;
      private final Definition newValue;
      private final boolean delete;

      private WriteOp(Key key, Definition oldValueSnapshot, Definition newValue, boolean delete) {
        this.key = key;
        this.oldValueSnapshot = oldValueSnapshot;
        this.newValue = newValue;
        this.delete = delete;
      }

      private Key key() {
        return key;
      }

      private Definition oldValueSnapshot() {
        return oldValueSnapshot;
      }

      private Definition newValue() {
        return newValue;
      }

      private boolean delete() {
        return delete;
      }
    }
  }

  private record Key(DefinitionType type, String group, String name) {}

  public record ConfigKey(DefinitionType type, String group, String name) {}
}
