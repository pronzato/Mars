public final class AmpsConnection implements AutoCloseable {
  public static final char SOH = '\u0001';
  public static final int DEFAULT_BATCH_MAX_MESSAGES = 100;
  public static final int DEFAULT_BATCH_MAX_BYTES = 1_000_000;
  public static final long DEFAULT_BATCH_FLUSH_MILLIS = 1_000L;

  private final HAClient client;
  private final CloseHandler closeHandler;

  private AmpsConnection(HAClient client, CloseHandler closeHandler) {
    this.client = Objects.requireNonNull(client, "client");
    this.closeHandler = closeHandler == null ? CloseHandler.NOOP : closeHandler;
  }

  public static AmpsConnection shared(HAClient client) {
    return new AmpsConnection(client, CloseHandler.NOOP);
  }

  public static AmpsConnection wrap(HAClient client) {
    return new AmpsConnection(client, CloseHandler.CLOSE_CLIENT);
  }

  public HAClient client() {
    return client;
  }

  public void publish(String topic, String payload) throws Exception {
    client.publish(topic, payload);
  }

  public void publish(String topic, String payload, long ttlMillis) throws Exception {
    client.publish(topic, payload, Math.toIntExact(ttlMillis));
  }

  public void publish(String topic, Map<String, Object> fields) throws Exception {
    client.publish(topic, encodeNvfix(fields));
  }

  public void publish(String topic, Map<String, Object> fields, long ttlMillis) throws Exception {
    client.publish(topic, encodeNvfix(fields), Math.toIntExact(ttlMillis));
  }

  public void publish(String topic, byte[] payload) throws Exception {
    publish(topic, payload, null);
  }

  public void publish(String topic, String payload, String options) throws Exception {
    String normalizedTopic = Strings.requireNonBlank(topic, "topic");
    String normalizedPayload = payload == null ? "" : payload;
    String normalizedOptions = Strings.trimToNull(options);
    if (normalizedOptions == null) {
      client.publish(normalizedTopic, normalizedPayload);
      return;
    }
    Command command = new Command(Message.Command.Publish);
    command.setTopic(normalizedTopic);
    command.setOptions(normalizedOptions);
    command.setData(normalizedPayload);
    try (MessageStream ignored = client.execute(command)) {
      // sync-ack publish path
    }
  }

  public void publish(String topic, byte[] payload, String options) throws Exception {
    String normalizedTopic = Strings.requireNonBlank(topic, "topic");
    byte[] bytes = payload == null ? new byte[0] : payload;
    String normalizedOptions = Strings.trimToNull(options);
    if (normalizedOptions == null) {
      client.publish(
          normalizedTopic.getBytes(StandardCharsets.UTF_8),
          0,
          normalizedTopic.getBytes(StandardCharsets.UTF_8).length,
          bytes,
          0,
          bytes.length);
      return;
    }
    Command command = new Command(Message.Command.Publish);
    command.setTopic(normalizedTopic);
    command.setOptions(normalizedOptions);
    command.setData(bytes, 0, bytes.length);
    try (MessageStream ignored = client.execute(command)) {
      // sync-ack publish path
    }
  }

  public Subscription sowAndSubscribe(SubscriptionSpec spec, SubscriptionListener listener)
      throws Exception {
    return subscribe(spec, listener);
  }

  public Subscription subscribe(SubscriptionSpec spec, SubscriptionListener listener)
      throws Exception {
    Objects.requireNonNull(spec, "spec");
    int commandType =
        spec.sowOnly()
            ? Message.Command.SOW
            : spec.sow() ? Message.Command.SOWAndSubscribe : Message.Command.Subscribe;
    String bookmark = spec.sowOnly() ? null : spec.bookmarkStart();
    return subscribeInternal(
        commandType, spec.topic(), spec.options(), spec.filter(), bookmark, spec.bookmarkFile(), listener);
  }

  public Subscription subscribe(SubscriptionSpec spec, Consumer<AmpsMessage> consumer)
      throws Exception {
    Objects.requireNonNull(consumer, "consumer");
    return subscribe(spec, (SubscriptionListener) consumer::accept);
  }

  public Subscription sowAndSubscribe(
      String topic, String options, String filter, SubscriptionListener listener) throws Exception {
    return subscribeInternal(
        Message.Command.SOWAndSubscribe, topic, options, filter, null, null, listener);
  }

  public Subscription subscribe(
      String topic, String options, String filter, SubscriptionListener listener) throws Exception {
    return subscribeInternal(Message.Command.Subscribe, topic, options, filter, null, null, listener);
  }

  @Override
  public void close() {
    try {
      closeHandler.close(client);
    } catch (Exception ignore) {
    }
  }

  public static String ensureTrailingNewline(String s) {
    Objects.requireNonNull(s, "s");
    return s.endsWith("\n") ? s : s + '\n';
  }

  private static String encodeNvfix(Map<String, Object> fields) {
    if (fields == null || fields.isEmpty()) {
      throw new IllegalArgumentException("fields is required");
    }
    StringBuilder builder = new StringBuilder();
    for (Map.Entry<String, Object> entry : new LinkedHashMap<>(fields).entrySet()) {
      String key = Strings.trimToNull(entry.getKey());
      if (key == null) {
        throw new IllegalArgumentException("nvfix key is required");
      }
      if (key.indexOf('=') >= 0) {
        throw new IllegalArgumentException("nvfix key contains '=': " + key);
      }
      if (key.indexOf(SOH) >= 0) {
        throw new IllegalArgumentException("nvfix key contains SOH: " + key);
      }
      Object rawValue = entry.getValue();
      if (rawValue == null) {
        continue;
      }
      String rendered = String.valueOf(rawValue);
      if (rendered.indexOf(SOH) >= 0) {
        throw new IllegalArgumentException("nvfix value contains SOH for key: " + key);
      }
      builder.append(key).append('=').append(rendered).append(SOH);
    }
    if (builder.isEmpty()) {
      throw new IllegalArgumentException("fields must contain at least one non-null value");
    }
    return builder.toString();
  }

  private Subscription subscribeInternal(
      int commandType,
      String topic,
      String options,
      String filter,
      String bookmarkStart,
      Path bookmarkFile,
      SubscriptionListener listener)
      throws Exception {
    Objects.requireNonNull(listener, "listener");
    String normalizedTopic = Strings.requireNonBlank(topic, "topic");
    Command command = new Command(commandType);
    command.setTopic(normalizedTopic);
    String normalizedOptions = Strings.trimToNull(options);
    if (normalizedOptions != null) {
      command.setOptions(normalizedOptions);
    }
    String normalizedFilter = Strings.trimToNull(filter);
    if (normalizedFilter != null) {
      command.setFilter(normalizedFilter);
    }
    String normalizedBookmark = Strings.trimToNull(bookmarkStart);
    if (normalizedBookmark != null) {
      command.setBookmark(normalizedBookmark);
    }
    CommandId commandId =
        client.executeAsync(
            command,
            message -> {
              if (message == null) {
                return;
              }
              listener.onMessage(new AmpsMessage(message));
            });
    return new Subscription(
        client, commandId, commandType, normalizedTopic, normalizedOptions, normalizedFilter);
  }

  public static final class Subscription implements AutoCloseable {
    private final HAClient client;
    private final CommandId commandId;
    private final int commandType;
    private final String topic;
    private final String options;
    private volatile String filter;
    private volatile boolean closed;

    private Subscription(
        HAClient client,
        CommandId commandId,
        int commandType,
        String topic,
        String options,
        String filter) {
      this.client = client;
      this.commandId = commandId;
      this.commandType = commandType;
      this.topic = topic;
      this.options = options;
      this.filter = filter;
    }

    @Override
    public void close() {
      closed = true;
      if (client == null || commandId == null) {
        return;
      }
      try {
        client.unsubscribe(commandId);
      } catch (Exception ignore) {
      }
      try {
        client.removeMessageHandler(commandId);
      } catch (Exception ignore) {
      }
    }

    public void replaceFilter(String nextFilter) throws Exception {
      if (client == null || commandId == null) {
        throw new IllegalStateException("Subscription handle is invalid");
      }
      if (closed) {
        throw new IllegalStateException("Subscription is closed");
      }
      if (!isReplaceSupported(commandType)) {
        throw new IllegalStateException(
            "Subscription replacement is unsupported for command type=" + commandType);
      }
      Command command = new Command(commandType);
      command.setTopic(topic);
      String replaceOptions = withReplaceOption(options);
      if (replaceOptions != null) {
        command.setOptions(replaceOptions);
      }
      String normalizedFilter = Strings.trimToNull(nextFilter);
      if (normalizedFilter != null) {
        command.setFilter(normalizedFilter);
      }
      command.setSubId(commandId);
      client.executeAsync(command, null);
      filter = normalizedFilter;
    }

    private static boolean isReplaceSupported(int commandType) {
      return commandType == Message.Command.Subscribe
          || commandType == Message.Command.SOWAndSubscribe
          || commandType == Message.Command.DeltaSubscribe
          || commandType == Message.Command.SOWAndDeltaSubscribe;
    }

    private static String withReplaceOption(String options) {
      LinkedHashSet<String> normalized = new LinkedHashSet<>();
      String source = Strings.trimToNull(options);
      if (source != null) {
        for (String token : source.split(",")) {
          String trimmed = Strings.trimToNull(token);
          if (trimmed != null) {
            normalized.add(trimmed);
          }
        }
      }
      boolean hasReplace = false;
      for (String token : normalized) {
        if ("replace".equalsIgnoreCase(token)) {
          hasReplace = true;
          break;
        }
      }
      if (!hasReplace) {
        normalized.add("replace");
      }
      if (normalized.isEmpty()) {
        return null;
      }
      return String.join(",", normalized);
    }
  }

  @FunctionalInterface
  public interface SubscriptionListener {
    void onMessage(AmpsMessage message);
  }

  public record SubscriptionSpec(
      boolean sow,
      boolean sowOnly,
      String topic,
      String options,
      String filter,
      String bookmarkStart,
      Path bookmarkFile
  ) {
    public SubscriptionSpec(boolean sow, String topic, String options, String filter) {
      this(sow, false, topic, options, filter, null, null);
    }

    public SubscriptionSpec(boolean sow, boolean sowOnly, String topic, String options, String filter) {
      this(sow, sowOnly, topic, options, filter, null, null);
    }

    public SubscriptionSpec {
      if (sowOnly) {
        sow = false;
      }
      topic = Strings.requireNonBlank(topic, "topic");
      options = Strings.trimToNull(options);
      filter = Strings.trimToNull(filter);
      bookmarkStart = Strings.trimToNull(bookmarkStart);
    }

    public static Builder builder() {
      return new Builder();
    }

    public static final class Builder {
      private boolean sow;
      private boolean sowOnly;
      private String topic;
      private String options;
      private String filter;
      private String bookmarkStart;
      private Path bookmarkFile;

      private Builder() {}

      public Builder withSowSubscribe(boolean value) {
        sow = value;
        if (value) {
          sowOnly = false;
        }
        return this;
      }

      public Builder withSow(boolean value) {
        sowOnly = value;
        if (value) {
          sow = false;
        }
        return this;
      }

      public Builder withSow() {
        return withSow(true);
      }

      public Builder withSow(Boolean value) {
        return withSow(Boolean.TRUE.equals(value));
      }

      public Builder withSowSubscribe(Boolean value) {
        sow = Boolean.TRUE.equals(value);
        if (Boolean.TRUE.equals(value)) {
          sowOnly = false;
        }
        return this;
      }

      public Builder withTopic(String value) {
        topic = value;
        return this;
      }

      public Builder withOptions(String value) {
        options = value;
        return this;
      }

      public Builder withFilter(String value) {
        filter = value;
        return this;
      }

      public Builder withBookmarkStart(String value) {
        bookmarkStart = value;
        return this;
      }

      public Builder withBookmarkFile(Path value) {
        bookmarkFile = value;
        return this;
      }

      public Builder withBookmarkFile(String value) {
        String normalized = Strings.trimToNull(value);
        bookmarkFile = normalized == null ? null : Path.of(normalized);
        return this;
      }

      public SubscriptionSpec build() {
        return new SubscriptionSpec(
            sow,
            sowOnly,
            topic,
            options,
            filter,
            bookmarkStart,
            bookmarkFile);
      }
    }
  }

  public enum BatchMode {
    STREAM,
    BATCH
  }

  public record BatchSpec(
      BatchMode mode,
      int batchMaxMessages,
      int batchMaxBytes,
      long batchFlushMillis
  ) {
    public BatchSpec {
      mode = mode == null ? BatchMode.STREAM : mode;
      batchMaxMessages =
          batchMaxMessages <= 0 ? DEFAULT_BATCH_MAX_MESSAGES : batchMaxMessages;
      batchMaxBytes = batchMaxBytes <= 0 ? DEFAULT_BATCH_MAX_BYTES : batchMaxBytes;
      batchFlushMillis =
          batchFlushMillis <= 0L ? DEFAULT_BATCH_FLUSH_MILLIS : batchFlushMillis;
    }

    public static BatchSpec defaults() {
      return new BatchSpec(
          BatchMode.STREAM,
          DEFAULT_BATCH_MAX_MESSAGES,
          DEFAULT_BATCH_MAX_BYTES,
          DEFAULT_BATCH_FLUSH_MILLIS);
    }
  }

  public static final class AmpsMessage {
    private final Message raw;

    private AmpsMessage(Message raw) {
      this.raw = Objects.requireNonNull(raw, "raw");
    }

    public String topic() {
      return raw.getTopic();
    }

    public String data() {
      return raw.getData();
    }

    public byte[] dataBytes() {
      Field data = raw.getDataRaw();
      if (data == null || data.isNull() || data.buffer == null || data.length <= 0) {
        String fallback = raw.getData();
        return fallback == null ? null : fallback.getBytes(StandardCharsets.UTF_8);
      }
      return Arrays.copyOfRange(data.buffer, data.position, data.position + data.length);
    }

    public String bookmark() {
      return raw.getBookmark();
    }

    public String options() {
      return raw.getOptions();
    }

    public Message raw() {
      return raw;
    }
  }

  @FunctionalInterface
  private interface CloseHandler {
    CloseHandler NOOP = client -> {};

    CloseHandler CLOSE_CLIENT =
        client -> {
          try {
            client.disconnect();
          } catch (Exception ignore) {
          }
          client.close();
        };

    void close(HAClient client) throws Exception;
  }

  public static final class BookmarkStoreLocal {
    private BookmarkStoreLocal() {}

    public static Path defaultBookmarkPath(String group, String subscriptionName) {
      String normalizedGroup = Strings.requireNonBlank(group, "group");
      String normalizedSubscription = Strings.requireNonBlank(subscriptionName, "subscriptionName");
      return Path.of("data", "lite", "bookmarks", normalizedGroup, normalizedSubscription + ".bookmark");
    }

    public static String subscriptionKey(String ingestName, String connectionName, String topic) {
      String normalizedIngest = sanitizeToken(Strings.requireNonBlank(ingestName, "ingestName"));
      String normalizedConnection =
          sanitizeToken(Strings.requireNonBlank(connectionName, "connectionName"));
      String normalizedTopic = sanitizeToken(Strings.requireNonBlank(topic, "topic"));
      return normalizedIngest + "__" + normalizedConnection + "__" + normalizedTopic;
    }

    public static Optional<String> load(Path path) {
      Objects.requireNonNull(path, "path");
      try {
        if (!Files.exists(path)) {
          return Optional.empty();
        }
        String value = Strings.trimToNull(Files.readString(path, StandardCharsets.UTF_8));
        return value == null ? Optional.empty() : Optional.of(value);
      } catch (Exception ex) {
        throw new IllegalStateException("Failed to load bookmark file: " + path, ex);
      }
    }

    public static void saveAtomic(Path path, String bookmark) {
      Objects.requireNonNull(path, "path");
      String normalizedBookmark = Strings.requireNonBlank(bookmark, "bookmark");
      try {
        Path parent = path.getParent();
        if (parent != null) {
          Files.createDirectories(parent);
        }
        Path temp =
            path.resolveSibling(path.getFileName().toString() + ".tmp." + Thread.currentThread().getId());
        Files.writeString(
            temp,
            normalizedBookmark,
            StandardCharsets.UTF_8,
            StandardOpenOption.CREATE,
            StandardOpenOption.TRUNCATE_EXISTING,
            StandardOpenOption.WRITE);
        try {
          Files.move(
              temp,
              path,
              StandardCopyOption.REPLACE_EXISTING,
              StandardCopyOption.ATOMIC_MOVE);
        } catch (AtomicMoveNotSupportedException ex) {
          Files.move(temp, path, StandardCopyOption.REPLACE_EXISTING);
        }
      } catch (Exception ex) {
        throw new IllegalStateException("Failed to persist bookmark file: " + path, ex);
      }
    }

    private static String sanitizeToken(String value) {
      StringBuilder normalized = new StringBuilder(value.length());
      for (int index = 0; index < value.length(); index++) {
        char current = value.charAt(index);
        if (Character.isLetterOrDigit(current) || current == '-' || current == '_' || current == '.') {
          normalized.append(current);
          continue;
        }
        normalized.append('_');
      }
      return normalized.toString();
    }
  }
