      return new GetRelatedAssetsResponse(
          requestId, ToolStatus.OK, null, assetId, depth, relationships, relationshipCount);
    } catch (RuntimeException ex) {
      logInvocation(
          "get_related_assets",
          requestId,
          accessContext.subject(),
          assetId,
          ToolStatus.ERROR,
          ex.getMessage());
      return new GetRelatedAssetsResponse(
          requestId,
          ToolStatus.ERROR,
          "Get related assets failed: " + ex.getMessage(),
          assetId,
          depth,
          Map.of(),
          0);
    }
  }

  @Override
  public QueryDatasetResponse queryDataset(QueryDatasetRequest request) {
    String requestId = requestId(request == null ? null : request.requestId());
    String datasetInput = normalize(request == null ? null : request.datasetId());
    if (datasetInput == null) {
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.INVALID_REQUEST,
          "datasetId is required",
          null,
          0,
          null,
          List.of(),
          List.of(),
          0,
          false);
    }
    CatalogAccessContext accessContext =
        accessContext(
            request == null ? null : request.subject(),
            request == null ? null : request.delegatedSubject(),
            request == null ? null : request.lane(),
            datasetInput);
    String canonicalDatasetId;
    try {
      canonicalDatasetId = DatasetId.parse(datasetInput).toString();
    } catch (RuntimeException ex) {
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.INVALID_REQUEST,
          "Invalid dataset id: " + datasetInput,
          datasetInput,
          0,
          null,
          List.of(),
          List.of(),
          0,
          false);
    }
    Optional<CatalogDetails> assetDetails = catalogService.getAsset(canonicalDatasetId, accessContext);
    if (assetDetails.isEmpty()) {
      ToolStatus status = assetExists(canonicalDatasetId) ? ToolStatus.DENIED : ToolStatus.NOT_FOUND;
      String message =
          status == ToolStatus.DENIED
              ? "Access denied for dataset: " + canonicalDatasetId
              : "Dataset not found: " + canonicalDatasetId;
      logInvocation("query_dataset", requestId, accessContext.subject(), canonicalDatasetId, status, message);
      return new QueryDatasetResponse(
          requestId, status, message, canonicalDatasetId, 0, null, List.of(), List.of(), 0, false);
    }

    CatalogDetails details = assetDetails.get();
    CatalogSummary summary = details.summary();
    if (summary == null || summary.type() != DefinitionType.DATASET) {
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.INVALID_REQUEST,
          "Asset is not a dataset: " + canonicalDatasetId,
          canonicalDatasetId,
          0,
          null,
          List.of(),
          List.of(),
          0,
          false);
    }
    if (!isReadToolAllowed(details.toolHints())) {
      String message = "Tool action class is not read-safe for dataset: " + canonicalDatasetId;
      logInvocation("query_dataset", requestId, accessContext.subject(), canonicalDatasetId, ToolStatus.DENIED, message);
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.DENIED,
          message,
          canonicalDatasetId,
          0,
          null,
          List.of(),
          List.of(),
          0,
          false);
    }

    int hintLimit =
        Math.max(
            1,
            Math.min(
                maxQueryLimit,
                details.toolHints() == null
                    ? DEFAULT_QUERY_LIMIT
                    : Math.max(1, details.toolHints().safeDefaultLimit())));
    int appliedLimit =
        Math.max(
            1,
            Math.min(
                Math.min(hintLimit, maxQueryLimit),
                request == null || request.limit() == null ? hintLimit : request.limit()));
    Duration hintTimeout =
        minDuration(
            parseDuration(
                details.toolHints() == null ? null : details.toolHints().safeDefaultTimeout(),
                DEFAULT_QUERY_TIMEOUT),
            maxQueryTimeout);
    Duration requestedTimeout = request == null ? null : request.timeout();
    Duration appliedTimeout =
        requestedTimeout == null || requestedTimeout.isZero() || requestedTimeout.isNegative()
            ? hintTimeout
            : minDuration(requestedTimeout, hintTimeout);
    if (appliedTimeout == null || appliedTimeout.isZero() || appliedTimeout.isNegative()) {
      appliedTimeout = DEFAULT_QUERY_TIMEOUT;
    }

    String filter;
    try {
      filter = validateFilter(request == null ? null : request.query());
    } catch (IllegalArgumentException ex) {
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.INVALID_REQUEST,
          ex.getMessage(),
          canonicalDatasetId,
          appliedLimit,
          appliedTimeout.toString(),
          List.of(),
          List.of(),
          0,
          false);
    }

    if (!queryPermits.tryAcquire()) {
      String message = "Tool query concurrency limit reached";
      logInvocation("query_dataset", requestId, accessContext.subject(), canonicalDatasetId, ToolStatus.TOO_BUSY, message);
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.TOO_BUSY,
          message,
          canonicalDatasetId,
          appliedLimit,
          appliedTimeout.toString(),
          List.of(),
          List.of(),
          0,
          false);
    }

    try {
      RowManager rowManager = rowManagerSupplier.get();
      if (rowManager == null) {
        String message = "No runtime data plane available for query_dataset";
        logInvocation("query_dataset", requestId, accessContext.subject(), canonicalDatasetId, ToolStatus.ERROR, message);
        return new QueryDatasetResponse(
            requestId,
            ToolStatus.ERROR,
            message,
            canonicalDatasetId,
            appliedLimit,
            appliedTimeout.toString(),
            List.of(),
            List.of(),
            0,
            false);
      }
      DatasetId datasetId = DatasetId.parse(canonicalDatasetId);
      String sql = buildDatasetPreviewSql(datasetId, filter, appliedLimit);
      RowResult result =
          runWithTimeout(
              () -> rowManager.querySql(sql),
              appliedTimeout);
      List<List<Object>> rows = result == null ? List.of() : result.rows();
      List<String> columns = result == null ? List.of() : result.columns();
      List<QueryDatasetColumn> schema = inferSchema(columns, rows);
      int rowCount = rows == null ? 0 : rows.size();
      boolean truncated = rowCount >= appliedLimit;
      logInvocation(
          "query_dataset",
          requestId,
          accessContext.subject(),
          canonicalDatasetId,
          ToolStatus.OK,
          "rows=" + rowCount + " truncated=" + truncated);
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.OK,
          null,
          canonicalDatasetId,
          appliedLimit,
          appliedTimeout.toString(),
          schema,
          rows,
          rowCount,
          truncated);
    } catch (TimeoutException ex) {
      String message = "query_dataset timeout after " + appliedTimeout;
      logInvocation("query_dataset", requestId, accessContext.subject(), canonicalDatasetId, ToolStatus.ERROR, message);
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.ERROR,
          message,
          canonicalDatasetId,
          appliedLimit,
          appliedTimeout.toString(),
          List.of(),
          List.of(),
          0,
          false);
    } catch (RuntimeException ex) {
      logInvocation(
          "query_dataset",
          requestId,
          accessContext.subject(),
          canonicalDatasetId,
          ToolStatus.ERROR,
          ex.getMessage());
      return new QueryDatasetResponse(
          requestId,
          ToolStatus.ERROR,
          "query_dataset failed: " + ex.getMessage(),
          canonicalDatasetId,
          appliedLimit,
          appliedTimeout.toString(),
          List.of(),
          List.of(),
          0,
          false);
    } finally {
      queryPermits.release();
    }
  }

  @Override
  public void close() {
    queryExecutor.shutdownNow();
  }

  private Map<RelationshipType, List<CatalogRelationshipView>> collectRelated(
      String rootAssetId,
      CatalogAccessContext accessContext,
      int depth,
      RelationshipType filterType) {
    LinkedHashMap<RelationshipType, List<CatalogRelationshipView>> aggregate = new LinkedHashMap<>();
    Set<String> seenEdges = new LinkedHashSet<>();
    Set<String> visitedAssets = new LinkedHashSet<>();
    ArrayDeque<TraversalNode> queue = new ArrayDeque<>();
    queue.add(new TraversalNode(rootAssetId, 0));
    visitedAssets.add(rootAssetId);

    while (!queue.isEmpty()) {
      TraversalNode current = queue.removeFirst();
      if (current.depth() >= depth) {
        continue;
      }
      Map<RelationshipType, List<CatalogRelationshipView>> relatedByType =
          catalogService.getRelatedAssets(current.assetId(), accessContext);
      for (Map.Entry<RelationshipType, List<CatalogRelationshipView>> entry : relatedByType.entrySet()) {
        if (filterType != null && filterType != entry.getKey()) {
          continue;
        }
        List<CatalogRelationshipView> relationships =
            entry.getValue() == null ? List.of() : entry.getValue();
        for (CatalogRelationshipView relationship : relationships) {
          if (relationship == null) {
            continue;
          }
          String edgeKey = edgeKey(relationship);
          if (!seenEdges.add(edgeKey)) {
            continue;
          }
          aggregate.computeIfAbsent(entry.getKey(), ignored -> new ArrayList<>()).add(relationship);
          String nextAssetId = resolveNextAssetId(current.assetId(), relationship);
          if (nextAssetId != null && visitedAssets.add(nextAssetId)) {
            queue.addLast(new TraversalNode(nextAssetId, current.depth() + 1));
          }
        }
      }
    }

    for (List<CatalogRelationshipView> relationships : aggregate.values()) {
      relationships.sort(
          Comparator.comparing(CatalogRelationshipView::fromAssetId)
              .thenComparing(CatalogRelationshipView::toAssetId)
              .thenComparing(view -> view.relationshipType().name()));
    }
    return Map.copyOf(aggregate);
  }

  private static int countRelationships(Map<RelationshipType, List<CatalogRelationshipView>> relationshipsByType) {
    int total = 0;
    if (relationshipsByType == null || relationshipsByType.isEmpty()) {
      return total;
    }
    for (List<CatalogRelationshipView> values : relationshipsByType.values()) {
      total += values == null ? 0 : values.size();
    }
    return total;
  }

  private String validateFilter(String query) {
    String normalized = normalize(query);
    if (normalized == null) {
      return null;
    }
    if (normalized.contains(";")) {
      throw new IllegalArgumentException("query must be a single read-only filter expression");
    }
    String lower = normalized.toLowerCase(Locale.ROOT);
    if (lower.startsWith("select") || lower.startsWith("with")) {
      throw new IllegalArgumentException(
          "query_dataset expects a filter expression, not a full SQL statement");
    }
    for (String forbidden : FORBIDDEN_FILTER_KEYWORDS) {
      if (containsToken(lower, forbidden)) {
        throw new IllegalArgumentException("query contains forbidden keyword: " + forbidden);
      }
    }
    return normalized;
  }

  private static String buildDatasetPreviewSql(DatasetId datasetId, String filter, int limit) {
    StringBuilder sql = new StringBuilder("select * from ").append(sqlToken(datasetId));
    String normalizedFilter = normalize(filter);
    if (normalizedFilter != null) {
      sql.append(" where ").append(normalizedFilter);
    }
    sql.append(" limit ").append(limit);
    return sql.toString();
  }

  private static String sqlToken(DatasetId datasetId) {
    Objects.requireNonNull(datasetId, "datasetId");
    return quoteIdentifier(datasetId.namespace()) + "." + quoteIdentifier(datasetId.name());
  }

  private static String quoteIdentifier(String identifier) {
    return "\"" + identifier.replace("\"", "\"\"") + "\"";
  }

  private static boolean containsToken(String text, String token) {
    int index = text.indexOf(token);
    while (index >= 0) {
      boolean leftBoundary = index == 0 || !Character.isLetterOrDigit(text.charAt(index - 1));
      int rightIndex = index + token.length();
      boolean rightBoundary =
          rightIndex >= text.length() || !Character.isLetterOrDigit(text.charAt(rightIndex));
      if (leftBoundary && rightBoundary) {
        return true;
      }
      index = text.indexOf(token, index + 1);
    }
    return false;
  }

  private boolean isReadToolAllowed(Semantic.ToolHints hints) {
    Semantic.ToolHints resolved =
        hints == null ? new Semantic.ToolHints(null, null, null, null, null, null) : hints;
    Semantic.ActionClass actionClass = resolved.actionClass();
    boolean allowedAction =
        actionClass == Semantic.ActionClass.READ || actionClass == Semantic.ActionClass.ANALYZE;
    boolean readOnly = Boolean.TRUE.equals(resolved.readOnly());
    boolean idempotent = Boolean.TRUE.equals(resolved.idempotent());
    return allowedAction && readOnly && idempotent;
  }

  private static List<QueryDatasetColumn> inferSchema(List<String> columns, List<List<Object>> rows) {
    if (columns == null || columns.isEmpty()) {
      return List.of();
    }
    List<QueryDatasetColumn> schema = new ArrayList<>(columns.size());
    for (int index = 0; index < columns.size(); index++) {
      String column = columns.get(index);
      String type = inferColumnType(rows, index);
      schema.add(new QueryDatasetColumn(column, type));
    }
    return List.copyOf(schema);
  }

  private static String inferColumnType(List<List<Object>> rows, int columnIndex) {
    if (rows == null || rows.isEmpty()) {
      return "UNKNOWN";
    }
    for (List<Object> row : rows) {
      if (row == null || columnIndex < 0 || columnIndex >= row.size()) {
        continue;
      }
      Object value = row.get(columnIndex);
      if (value == null) {
        continue;
      }
      if (value instanceof Integer || value instanceof Short || value instanceof Byte) {
        return "INTEGER";
      }
      if (value instanceof Long) {
        return "BIGINT";
      }
      if (value instanceof Float || value instanceof Double || value instanceof BigDecimal) {
        return "DECIMAL";
      }
      if (value instanceof Boolean) {
        return "BOOLEAN";
      }
      if (value instanceof LocalDate) {
        return "DATE";
      }
      if (value instanceof LocalDateTime || value instanceof OffsetDateTime || value instanceof ZonedDateTime) {
        return "TIMESTAMP";
      }
      if (value instanceof byte[]) {
        return "BINARY";
      }
      return "STRING";
    }
    return "UNKNOWN";
  }

  private <T> T runWithTimeout(Callable<T> task, Duration timeout) throws TimeoutException {
    Future<T> future = queryExecutor.submit(task);
    try {
      long timeoutMs = Math.max(1L, timeout.toMillis());
      return future.get(timeoutMs, TimeUnit.MILLISECONDS);
    } catch (TimeoutException ex) {
      future.cancel(true);
      throw ex;
    } catch (RuntimeException ex) {
      throw ex;
    } catch (Exception ex) {
      throw new IllegalStateException(ex.getMessage(), ex);
    }
  }

  private CatalogAccessContext accessContext(
      String subject,
      String delegatedSubject,
      String lane,
      String seedAssetId) {
    return new CatalogAccessContext(
        Strings.firstNonBlank(normalize(subject), defaultSubject, "app:tools/system"),
        Strings.firstNonBlank(normalize(delegatedSubject), defaultDelegatedSubject),
        Strings.firstNonBlank(normalize(lane), defaultLane, "dev"),
        normalize(seedAssetId));
  }

  private boolean assetExists(String assetId) {
    String normalized = normalize(assetId);
    if (normalized == null) {
      return false;
    }
    List<Definition> definitions = safeDefinitions();
    if (definitions.isEmpty()) {
      return false;
    }
    AssetLifecycleGraph graph = AssetLifecycleGraphBuilder.build(definitions);
    String resolved = graph.resolveAssetId(normalized);
    return resolved != null || graph.nodes().containsKey(normalized);
  }

  private List<Definition> safeDefinitions() {
    try {
      List<Definition> definitions = definitionsSupplier.get();
      return definitions == null ? List.of() : List.copyOf(definitions);
    } catch (RuntimeException ex) {
      return List.of();
    }
  }

  private void logInvocation(
      String toolName,
      String requestId,
      String subject,
      String assetId,
      ToolStatus status,
      String message) {
    LOGGER.info(
        "[ToolGateway] tool={} requestId={} subject={} assetId={} status={} message={}",
        toolName,
        requestId,
        Strings.firstNonBlank(subject, "<none>"),
        Strings.firstNonBlank(assetId, "<none>"),
        status,
        Strings.firstNonBlank(message, ""));
  }

  private static String resolveNextAssetId(String currentAssetId, CatalogRelationshipView relationship) {
    String from = normalize(relationship.fromAssetId());
    String to = normalize(relationship.toAssetId());
    if (from == null && to == null) {
      return null;
    }
    if (currentAssetId != null && currentAssetId.equals(from)) {
      return to;
    }
    if (currentAssetId != null && currentAssetId.equals(to)) {
      return from;
    }
    return to == null ? from : to;
  }

  private static String edgeKey(CatalogRelationshipView relationship) {
    return Strings.firstNonBlank(normalize(relationship.fromAssetId()), "")
        + "|"
        + Strings.firstNonBlank(normalize(relationship.toAssetId()), "")
        + "|"
        + relationship.relationshipType().name()
        + "|"
        + relationship.direction().name();
  }

  private static String requestId(String requestId) {
    return Strings.firstNonBlank(normalize(requestId), "tool-" + UUID.randomUUID());
  }

  private static String normalize(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }

  private static Duration parseDuration(String text, Duration fallback) {
    String normalized = normalize(text);
    if (normalized == null) {
      return fallback;
    }
    try {
      Duration parsed = Duration.parse(normalized);
      if (parsed.isNegative() || parsed.isZero()) {
        return fallback;
      }
      return parsed;
    } catch (RuntimeException ex) {
      return fallback;
    }
  }

  private static Duration minDuration(Duration first, Duration second) {
    if (first == null) {
      return second;
    }
    if (second == null) {
      return first;
    }
    return first.compareTo(second) <= 0 ? first : second;
  }

  private static int normalizePositive(Integer value, int fallback) {
    if (value == null || value <= 0) {
      return fallback;
    }
    return value;
  }
