  @Override
  public FlightInfo getFlightInfoStatement(
      FlightSql.CommandStatementQuery command,
      FlightProducer.CallContext context,
      FlightDescriptor descriptor) {
    RequestIdentity identity = resolveIdentity(context);
    return RequestIdentityResolver.callAsRequestIdentity(
        identity,
        () -> {
          String subject = identity.effectiveUserSubject();
          authorizeInvoke(subject, OPERATION_STATEMENT_EXECUTE, command == null ? null : command.getQuery());

          DatasetId datasetId = resolveDatasetId(context);
          installDatasetIfPresent(datasetId);
          String sql = rewriteDatasetPlaceholder(requireSql(command), datasetId);
          logSqlRequest(identity, OPERATION_STATEMENT_EXECUTE, datasetId, sql);
          ByteBuffer ipc = queryIpcSqlAsCaller(identity, sql, datasetId);
          IpcSummary summary = inspectIpc(ipc);

          String statementHandle = UUID.randomUUID().toString();
          statements.put(
              statementHandle, new PendingStatement(subject, summary.schema(), summary.rowCount(), ipc));

          FlightSql.TicketStatementQuery ticketMessage =
              FlightSql.TicketStatementQuery.newBuilder()
                  .setStatementHandle(ByteString.copyFromUtf8(statementHandle))
                  .build();
          Ticket ticket = new Ticket(ticketMessage.toByteArray());
          FlightDescriptor responseDescriptor =
              descriptor != null
                  ? descriptor
                  : FlightDescriptor.command(Any.pack(command).toByteArray());
          FlightEndpoint endpoint = new FlightEndpoint(ticket, location());
          return new FlightInfo(
              summary.schema(),
              responseDescriptor,
              List.of(endpoint),
              summary.rowCount(),
              summary.rowCount());
        });
  }

  @Override
  public SchemaResult getSchema(
      FlightProducer.CallContext context, FlightDescriptor descriptor) {
    if (descriptor == null || !descriptor.isCommand()) {
      return super.getSchema(context, descriptor);
    }
    Any packed;
    try {
      packed = Any.parseFrom(descriptor.getCommand());
    } catch (Exception ex) {
      return super.getSchema(context, descriptor);
    }
    if (!packed.is(FlightSql.CommandGetTables.class)) {
      return super.getSchema(context, descriptor);
    }
    FlightSql.CommandGetTables command;
    try {
      command = packed.unpack(FlightSql.CommandGetTables.class);
    } catch (InvalidProtocolBufferException ex) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription("Invalid Flight SQL GetTableSchema descriptor payload")
          .withCause(ex)
          .toRuntimeException();
    }
    return getTableSchema(command, context);
  }

  @Override
  public void getStream(
      FlightProducer.CallContext context,
      Ticket ticket,
      FlightProducer.ServerStreamListener listener) {
    Objects.requireNonNull(ticket, "ticket");
    if (FlightDatasetStreamTicketCodec.isStreamTicket(ticket)) {
      handleDatasetStream(context, ticket, listener);
      return;
    }
    if (dispatchMetadataTicket(context, ticket, listener)) {
      return;
    }
    if (dispatchStatementTicket(context, ticket, listener)) {
      return;
    }
    throw CallStatus.INVALID_ARGUMENT.withDescription("Invalid Flight SQL ticket").toRuntimeException();
  }

  @Override
  public void getStreamStatement(
      FlightSql.TicketStatementQuery ticket,
      FlightProducer.CallContext context,
      FlightProducer.ServerStreamListener listener) {
    RequestIdentity identity = resolveIdentity(context);
    RequestIdentityResolver.runAsRequestIdentity(
        identity,
        () -> {
          String caller = identity.effectiveUserSubject();
          authorizeInvoke(caller, OPERATION_STATEMENT_FETCH, null);

          String handle =
              ticket == null || ticket.getStatementHandle() == null
                  ? null
                  : Strings.trimToNull(ticket.getStatementHandle().toStringUtf8());
          if (handle == null) {
            throw CallStatus.INVALID_ARGUMENT
                .withDescription("Ticket statement handle is required")
                .toRuntimeException();
          }
          PendingStatement pending = statements.remove(handle);
          if (pending == null) {
            throw CallStatus.NOT_FOUND
                .withDescription("Unknown statement handle: " + handle)
                .toRuntimeException();
          }
          LOGGER.info("[FlightSQL][request] requestId={} operation={} subject={} statementHandle={} cachedRows={}", identity.requestId(), OPERATION_STATEMENT_FETCH, safeSubject(caller), handle, pending.rowCount());
          if (!Objects.equals(pending.callerSubject(), caller)) {
            throw CallStatus.UNAUTHORIZED
                .withDescription("Statement handle is not owned by caller")
                .toRuntimeException();
          }
          streamIpc(listener, pending.ipcBytes());
        });
  }

  private FlightInfo metadataFlightInfo(
      Schema schema, FlightDescriptor descriptor, com.google.protobuf.Message command) {
    Objects.requireNonNull(schema, "schema");
    Objects.requireNonNull(command, "command");
    byte[] commandBytes = Any.pack(command).toByteArray();
    FlightDescriptor responseDescriptor =
        descriptor != null ? descriptor : FlightDescriptor.command(commandBytes);
    Ticket ticket = new Ticket(commandBytes);
    FlightEndpoint endpoint = new FlightEndpoint(ticket, location());
    return new FlightInfo(schema, responseDescriptor, List.of(endpoint), -1L, -1L);
  }

  private boolean dispatchStatementTicket(
      FlightProducer.CallContext context,
      Ticket ticket,
      FlightProducer.ServerStreamListener listener) {
    FlightSql.TicketStatementQuery statementTicket;
    try {
      statementTicket = FlightSql.TicketStatementQuery.parseFrom(ticket.getBytes());
    } catch (Exception ex) {
      return false;
    }
    String handle =
        statementTicket == null || statementTicket.getStatementHandle() == null
            ? null
            : Strings.trimToNull(statementTicket.getStatementHandle().toStringUtf8());
    if (handle == null || !statements.containsKey(handle)) {
      return false;
    }
    getStreamStatement(statementTicket, context, listener);
    return true;
  }

  private boolean dispatchMetadataTicket(
      FlightProducer.CallContext context,
      Ticket ticket,
      FlightProducer.ServerStreamListener listener) {
    final Any packed;
    try {
      packed = Any.parseFrom(ticket.getBytes());
    } catch (Exception ex) {
      return false;
    }
    try {
      if (packed.is(FlightSql.CommandGetCatalogs.class)) {
        getStreamCatalogs(context, listener);
        return true;
      }
      if (packed.is(FlightSql.CommandGetDbSchemas.class)) {
        getStreamSchemas(packed.unpack(FlightSql.CommandGetDbSchemas.class), context, listener);
        return true;
      }
      if (packed.is(FlightSql.CommandGetTableTypes.class)) {
        getStreamTableTypes(context, listener);
        return true;
      }
      if (packed.is(FlightSql.CommandGetTables.class)) {
        getStreamTables(packed.unpack(FlightSql.CommandGetTables.class), context, listener);
        return true;
      }
    } catch (InvalidProtocolBufferException ex) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription("Invalid Flight SQL metadata ticket payload")
          .withCause(ex)
          .toRuntimeException();
    }
    return false;
  }

  private SchemaResult getTableSchema(
      FlightSql.CommandGetTables command, FlightProducer.CallContext context) {
    RequestIdentity identity = resolveIdentity(context);
    return RequestIdentityResolver.callAsRequestIdentity(
        identity,
        () -> {
          authorizeInvoke(identity.effectiveUserSubject(), OPERATION_METADATA_TABLE_SCHEMA, null);
          FlightSql.CommandGetTables resolved =
              command == null ? FlightSql.CommandGetTables.getDefaultInstance() : command;
          String catalog = validateExactPatternOrAll(resolved.getCatalog(), "catalog");
          String schemaPattern =
              validateExactPatternOrAll(
                  resolved.getDbSchemaFilterPattern(), "db_schema_name_pattern");
          String tablePattern =
              validateExactPatternOrAll(
                  resolved.getTableNameFilterPattern(), "table_name_pattern");
          if (schemaPattern == null || tablePattern == null) {
            throw CallStatus.INVALID_ARGUMENT
                .withDescription(
                    "GetTableSchema requires exact db_schema_name_pattern and table_name_pattern")
                .toRuntimeException();
          }
          Set<String> requestedTypes = validateRequestedTableTypes(resolved.getTableTypesList());
          List<DuckdbCatalogImpl.MetadataTable> rows =
              listTables(identity, catalog, schemaPattern, tablePattern, requestedTypes);
          if (rows.isEmpty()) {
            throw CallStatus.NOT_FOUND
                .withDescription("Table not found: " + schemaPattern + "." + tablePattern)
                .toRuntimeException();
          }
          if (rows.size() > 1) {
            List<DatasetId> distinctDatasets =
                rows.stream().map(DuckdbCatalogImpl.MetadataTable::datasetId).distinct().toList();
            if (distinctDatasets.size() > 1) {
              throw CallStatus.INVALID_ARGUMENT
                  .withDescription(
                      "Ambiguous table reference for GetTableSchema: "
                          + schemaPattern
                          + "."
                          + tablePattern)
                  .toRuntimeException();
            }
            return new SchemaResult(tableSchemaForDataset(distinctDatasets.get(0), true));
          }
          return new SchemaResult(tableSchemaForDataset(rows.get(0).datasetId(), true));
        });
  }

  private List<String> listSchemas(
      RequestIdentity identity, String catalog, String schemaPattern) {
    if (catalog != null && !matchesExactIgnoreCase(catalog, METADATA_CATALOG_NAME)) {
      return List.of();
    }
    LinkedHashSet<String> schemas = new LinkedHashSet<>();
    for (DuckdbCatalogImpl.MetadataTable table : listReadableTables(identity)) {
      if (schemaPattern != null && !matchesExactIgnoreCase(table.namespace(), schemaPattern)) {
        continue;
      }
      schemas.add(table.namespace());
    }
    return List.copyOf(schemas);
  }

  private List<DuckdbCatalogImpl.MetadataTable> listTables(
      RequestIdentity identity,
      String catalog,
      String schemaPattern,
      String tablePattern,
      Set<String> requestedTypes) {
    if (catalog != null && !matchesExactIgnoreCase(catalog, METADATA_CATALOG_NAME)) {
      LOGGER.info("[FlightSQL][metadata][tables] requestId={} serviceId={} subject={} catalog={} schemaPattern={} tablePattern={} types={} rowCount=0 tables=[] datasetIds=[]", identity.requestId(), serviceResourceId, safeSubject(identity.effectiveUserSubject()), catalog, schemaPattern, tablePattern, summarizeValues(requestedTypes, 10));
      return List.of();
    }
    List<DuckdbCatalogImpl.MetadataTable> rows = new ArrayList<>();
    for (DuckdbCatalogImpl.MetadataTable table : listReadableTables(identity)) {
      if (schemaPattern != null && !matchesExactIgnoreCase(table.namespace(), schemaPattern)) {
        continue;
      }
      if (tablePattern != null && !matchesExactIgnoreCase(table.name(), tablePattern)) {
        continue;
      }
      if (!requestedTypes.isEmpty()
          && !requestedTypes.contains(table.tableType().toUpperCase(Locale.ROOT))) {
        continue;
      }
      rows.add(table);
    }
    List<DuckdbCatalogImpl.MetadataTable> resolvedRows = List.copyOf(rows);
    List<String> tableRefs = new ArrayList<>(resolvedRows.size());
    LinkedHashSet<String> datasetIds = new LinkedHashSet<>();
    for (DuckdbCatalogImpl.MetadataTable row : resolvedRows) {
      if (row == null) {
        continue;
      }
      tableRefs.add(
          Strings.firstNonBlank(row.namespace(), "<null>")
              + "."
              + Strings.firstNonBlank(row.name(), "<null>")
              + "("
              + Strings.firstNonBlank(row.tableType(), "<null>")
              + ")");
      if (row.datasetId() != null) {
        datasetIds.add(row.datasetId().toString());
      }
    }
    LOGGER.info("[FlightSQL][metadata][tables] requestId={} serviceId={} subject={} catalog={} schemaPattern={} tablePattern={} types={} rowCount={} tables={} datasetIds={}", identity.requestId(), serviceResourceId, safeSubject(identity.effectiveUserSubject()), Strings.firstNonBlank(catalog, "<none>"), Strings.firstNonBlank(schemaPattern, "<none>"), Strings.firstNonBlank(tablePattern, "<none>"), summarizeValues(requestedTypes, 10), resolvedRows.size(), summarizeValues(tableRefs, 40), summarizeValues(datasetIds, 40));
    return resolvedRows;
  }

  private List<DuckdbCatalogImpl.MetadataTable> listReadableTables(RequestIdentity identity) {
    DuckdbCatalogImpl catalog = requireMetadataCatalog();
    RowManagerImpl rowManager = requireLocalRowManager();
    DataEntitlements entitlements = rowManager.dataEntitlementsForSqlPlanning();
    DataEntitlements.Identity entitlementsIdentity =
        new DataEntitlements.Identity(identity.effectiveUserSubject(), null);
    List<DuckdbCatalogImpl.MetadataTable> catalogTables = catalog.metadataTables();
    if (catalogTables.isEmpty()) {
      LOGGER.info("[FlightSQL][metadata][visibility] requestId={} serviceId={} subject={} catalogTableCount=0 visibleTableCount=0 deniedTableCount=0 deniedSamples=[]", identity.requestId(), serviceResourceId, safeSubject(identity.effectiveUserSubject()));
      return List.of();
    }
    List<DuckdbCatalogImpl.MetadataTable> visible = new ArrayList<>();
    int deniedCount = 0;
    List<String> deniedSamples = new ArrayList<>();
    for (DuckdbCatalogImpl.MetadataTable table : catalogTables) {
      DataEntitlements.Decision decision =
          entitlements.authorizeRead(entitlementsIdentity, table.datasetId().toString());
      if (decision != null && decision.allowed()) {
        visible.add(table);
      } else {
        deniedCount++;
        if (deniedSamples.size() < 5) {
          deniedSamples.add(
              table.datasetId()
                  + "("
                  + Strings.firstNonBlank(decision == null ? null : decision.reason(), "DENIED")
                  + ")");
        }
      }
    }
    LOGGER.info("[FlightSQL][metadata][visibility] requestId={} serviceId={} subject={} catalogTableCount={} visibleTableCount={} deniedTableCount={} deniedSamples={}", identity.requestId(), serviceResourceId, safeSubject(identity.effectiveUserSubject()), catalogTables.size(), visible.size(), deniedCount, summarizeValues(deniedSamples, 5));
    return List.copyOf(visible);
  }

  private static String validateExactPatternOrAll(String value, String label) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      return null;
    }
    if (containsWildcardPattern(normalized)) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription("unsupported pattern for " + label + ": " + normalized)
          .toRuntimeException();
    }
    return normalized;
  }

  private static Set<String> validateRequestedTableTypes(List<String> values) {
    if (values == null || values.isEmpty()) {
      return Set.of();
    }
    LinkedHashSet<String> normalized = new LinkedHashSet<>();
    for (String raw : values) {
      String type = Strings.trimToNull(raw);
      if (type == null) {
        continue;
      }
      if (containsWildcardPattern(type)) {
        throw CallStatus.INVALID_ARGUMENT
            .withDescription("unsupported pattern for table_type: " + type)
            .toRuntimeException();
      }
      String upper = type.toUpperCase(Locale.ROOT);
      if (!SUPPORTED_TABLE_TYPES.contains(upper)) {
        throw CallStatus.INVALID_ARGUMENT
            .withDescription("unsupported table_type: " + type)
            .toRuntimeException();
      }
      normalized.add(upper);
    }
    return normalized.isEmpty() ? Set.of() : Set.copyOf(normalized);
  }

  private static boolean containsWildcardPattern(String value) {
    if (value == null || value.isEmpty()) {
      return false;
    }
    return value.contains("%")
        || value.contains("*")
        || value.contains("?")
        || value.contains("[")
        || value.contains("]")
        || value.contains(".*");
  }

  private static boolean matchesExactIgnoreCase(String left, String right) {
    if (left == null || right == null) {
      return false;
    }
    return left.equalsIgnoreCase(right);
  }

  private VectorSchemaRoot buildCatalogsRoot(List<String> catalogs, Schema schema) {
    VectorSchemaRoot root = VectorSchemaRoot.create(schema, ipcReadAllocator);
    int rowCount = catalogs == null ? 0 : catalogs.size();
    VarCharVector catalogVector = (VarCharVector) root.getVector("catalog_name");
    catalogVector.allocateNew(Math.max(1, rowCount));
    for (int index = 0; index < rowCount; index++) {
      putUtf8(catalogVector, index, catalogs.get(index));
    }
    catalogVector.setValueCount(rowCount);
    root.setRowCount(rowCount);
    return root;
  }

  private VectorSchemaRoot buildTablesRoot(List<DuckdbCatalogImpl.MetadataTable> rows, Schema schema) {
    VectorSchemaRoot root = VectorSchemaRoot.create(schema, ipcReadAllocator);
    int rowCount = rows == null ? 0 : rows.size();
    VarCharVector catalogVector = (VarCharVector) root.getVector("catalog_name");
    VarCharVector schemaVector = (VarCharVector) root.getVector("db_schema_name");
    VarCharVector tableVector = (VarCharVector) root.getVector("table_name");
    VarCharVector typeVector = (VarCharVector) root.getVector("table_type");
    catalogVector.allocateNew(Math.max(1, rowCount));
    schemaVector.allocateNew(Math.max(1, rowCount));
    tableVector.allocateNew(Math.max(1, rowCount));
    typeVector.allocateNew(Math.max(1, rowCount));
    for (int index = 0; index < rowCount; index++) {
      DuckdbCatalogImpl.MetadataTable row = rows.get(index);
      putUtf8(catalogVector, index, METADATA_CATALOG_NAME);
      putUtf8(schemaVector, index, row.namespace());
      putUtf8(tableVector, index, row.name());
      putUtf8(typeVector, index, row.tableType());
    }
    catalogVector.setValueCount(rowCount);
    schemaVector.setValueCount(rowCount);
    tableVector.setValueCount(rowCount);
    typeVector.setValueCount(rowCount);
    root.setRowCount(rowCount);
    return root;
  }

  private VectorSchemaRoot buildTablesWithSchemaRoot(
      List<DuckdbCatalogImpl.MetadataTable> rows, Schema schema, boolean strictSchema) {
    VectorSchemaRoot root = VectorSchemaRoot.create(schema, ipcReadAllocator);
    int rowCount = rows == null ? 0 : rows.size();
    VarCharVector catalogVector = (VarCharVector) root.getVector("catalog_name");
    VarCharVector schemaVector = (VarCharVector) root.getVector("db_schema_name");
    VarCharVector tableVector = (VarCharVector) root.getVector("table_name");
    VarCharVector typeVector = (VarCharVector) root.getVector("table_type");
    VarBinaryVector tableSchemaVector = (VarBinaryVector) root.getVector("table_schema");
    catalogVector.allocateNew(Math.max(1, rowCount));
    schemaVector.allocateNew(Math.max(1, rowCount));
    tableVector.allocateNew(Math.max(1, rowCount));
    typeVector.allocateNew(Math.max(1, rowCount));
    tableSchemaVector.allocateNew();
    for (int index = 0; index < rowCount; index++) {
      DuckdbCatalogImpl.MetadataTable row = rows.get(index);
      putUtf8(catalogVector, index, METADATA_CATALOG_NAME);
      putUtf8(schemaVector, index, row.namespace());
      putUtf8(tableVector, index, row.name());
      putUtf8(typeVector, index, row.tableType());
      byte[] tableSchema = tableSchemaBytes(row.datasetId(), strictSchema);
      if (tableSchema == null) {
        tableSchemaVector.setNull(index);
      } else {
        tableSchemaVector.setSafe(index, tableSchema);
      }
    }
    catalogVector.setValueCount(rowCount);
    schemaVector.setValueCount(rowCount);
    tableVector.setValueCount(rowCount);
    typeVector.setValueCount(rowCount);
    tableSchemaVector.setValueCount(rowCount);
    root.setRowCount(rowCount);
    return root;
  }

  private void streamSingleBatch(
      FlightProducer.ServerStreamListener listener, VectorSchemaRoot root) {
    Objects.requireNonNull(listener, "listener");
    Objects.requireNonNull(root, "root");
    try (root) {
      listener.start(root);
      listener.putNext();
      listener.completed();
    } catch (RuntimeException ex) {
      throw ex;
    } catch (Exception ex) {
      throw CallStatus.INTERNAL
          .withDescription("Failed to stream Flight SQL metadata")
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private static void putUtf8(VarCharVector vector, int index, String value) {
    if (value == null) {
      vector.setNull(index);
      return;
    }
    vector.setSafe(index, value.getBytes(java.nio.charset.StandardCharsets.UTF_8));
  }

  private byte[] tableSchemaBytes(DatasetId datasetId, boolean required) {
    Schema schema = tableSchemaForDataset(datasetId, required);
    if (schema == null) {
      return null;
    }
    return serializeSchema(schema);
  }

  private Schema tableSchemaForDataset(DatasetId datasetId, boolean required) {
    DuckdbCatalogImpl catalog = requireMetadataCatalog();
    Optional<SchemaModel> model = catalog.metadataSchemaModel(datasetId);
    if (model.isEmpty()) {
      if (!required) {
        return null;
      }
      throw CallStatus.NOT_FOUND
          .withDescription("Schema definition not found for dataset " + datasetId)
          .toRuntimeException();
    }
    return toArrowSchema(model.get());
  }

  private static Schema toArrowSchema(SchemaModel model) {
    if (model == null || model.getFields() == null || model.getFields().isEmpty()) {
      throw CallStatus.NOT_FOUND
          .withDescription("Schema definition is empty for dataset")
          .toRuntimeException();
    }
    List<Field> fields = new ArrayList<>();
    for (SchemaModel.Field field : model.getFields()) {
      if (field == null) {
        continue;
      }
      fields.add(
          new Field(
              Strings.requireNonBlank(field.getName(), "field.name"),
              new FieldType(field.isNullable(), toArrowType(field.getType()), null, null),
              null));
    }
    if (fields.isEmpty()) {
      throw CallStatus.NOT_FOUND
          .withDescription("Schema definition is empty for dataset")
          .toRuntimeException();
    }
    return new Schema(List.copyOf(fields));
  }

  private static ArrowType toArrowType(SchemaModel.FieldType type) {
    if (type == null) {
      throw CallStatus.NOT_FOUND.withDescription("Schema field type is required").toRuntimeException();
    }
    return switch (type) {
      case STRING -> new ArrowType.Utf8();
      case BOOL -> new ArrowType.Bool();
      case INT32 -> new ArrowType.Int(32, true);
      case INT64 -> new ArrowType.Int(64, true);
      case FLOAT64 -> new ArrowType.FloatingPoint(FloatingPointPrecision.DOUBLE);
      case DATE -> new ArrowType.Date(DateUnit.DAY);
      case TIMESTAMP -> new ArrowType.Timestamp(TimeUnit.MICROSECOND, null);
      case BINARY -> new ArrowType.Binary();
    };
  }

  private static byte[] serializeSchema(Schema schema) {
    try (ByteArrayOutputStream out = new ByteArrayOutputStream()) {
      MessageSerializer.serialize(new WriteChannel(Channels.newChannel(out)), schema);
      return out.toByteArray();
    } catch (IOException ex) {
      throw CallStatus.INTERNAL
          .withDescription("Failed to serialize table schema")
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private RequestIdentity resolveIdentity(FlightProducer.CallContext context) {
    FlightSqlRequestHeadersMiddleware middleware =
        context == null ? null : context.getMiddleware(FlightSqlRequestHeadersMiddleware.KEY);
    FabricRequestHeaders baseHeaders =
        middleware == null ? FabricRequestHeaders.empty(null) : middleware.requestHeaders();
    String principal = kerberosPrincipal(context);
    FabricRequestHeaders headers = baseHeaders.withAuthenticatedPrincipal(principal);
    try {
      RequestIdentity identity =
          RequestIdentityResolver.resolve(headers, DiscoveryCache, runtimeEnvironment);
      logRequestEntry(identity, middleware);
      return identity;
    } catch (RequestIdentityResolver.ResolutionException ex) {
      throw toFlightRuntime(ex);
    }
  }

  private void logRequestEntry(
      RequestIdentity identity, FlightSqlRequestHeadersMiddleware middleware) {
    String datasetId = middleware == null ? null : middleware.datasetId();
    String callSummary = middleware == null ? null : middleware.callSummary();
    Map<String, String> headers = middleware == null ? Map.of() : middleware.headerSnapshot();
    LOGGER.info("[Request] transport=ARROW_FLIGHT_SQL call={} requestId={} caller={} user={} delegatedUser={} datasetId={} headers={}", Strings.firstNonBlank(callSummary, "<unknown>"), identity.requestId(), safeCaller(identity.callerAppSubject()), identity.effectiveUserSubject(), Strings.firstNonBlank(identity.delegatedUserSubject(), "<none>"), Strings.firstNonBlank(datasetId, "<none>"), headers);
  }

      RequestIdentity identity, String operation, DatasetId datasetId, String sql) {
    LOGGER.info("[FlightSQL][request] requestId={} operation={} subject={} datasetId={} sql={}", identity.requestId(), operation, safeSubject(identity.effectiveUserSubject()), datasetId == null ? "<none>" : datasetId, summarizeSql(sql));

  private static String summarizeSql(String sql) {
    String normalized = Strings.trimToNull(sql);
    if (normalized == null) {
      return "<empty>";
    }
    if (normalized.length() > 512) {
      return normalized.substring(0, 512) + "...";
    }
    return normalized;
  }

  private static String summarizeValues(Iterable<String> values, int max) {
    if (values == null) {
      return "[]";
    }
    int limit = Math.max(1, max);
    List<String> preview = new ArrayList<>();
    int total = 0;
    for (String value : values) {
      if (value == null) {
        continue;
      }
      total++;
      if (preview.size() < limit) {
        preview.add(value);
      }
    }
    if (total == 0) {
      return "[]";
    }
    if (total <= limit) {
      return preview.toString();
    }
    return preview + "...(" + (total - limit) + " more)";
  }

  private void authorizeInvoke(String callerSubject, String operationName, String requestPayload) {
    ServiceEntitlements.Decision decision =
        serviceEntitlements.authorize(
            new ServiceEntitlements.Identity(callerSubject, null),
            new ServiceEntitlements.Request(
                ServiceEntitlements.Transport.ARROW_FLIGHT_SQL,
                ServiceEntitlements.Action.INVOKE,
                serviceResourceId,
                operationName,
                null,
                requestPayload,
                Map.of(),
                Map.of()));
    if (decision == null || !decision.allowed()) {
      String reason = decision == null ? "DENIED" : Strings.firstNonBlank(decision.reason(), "DENIED");
      LOGGER.info("[FlightSQL][deny] serviceId={} operation={} subject={} reason={}", serviceResourceId, operationName, safeSubject(callerSubject), reason);
      throw CallStatus.UNAUTHORIZED
          .withDescription(
              reason
                  + " [serviceId="
                  + serviceResourceId
                  + ", operation="
                  + operationName
                  + ", subject="
                  + safeSubject(callerSubject)
                  + "]")
          .toRuntimeException();
    }
  }

  private void installDatasetIfPresent(DatasetId datasetId) {
    if (datasetId == null) {
      return;
    }
    if (!isAllowedDataset(datasetId)) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "Dataset is not allowed for this service datasetId="
                  + datasetId
                  + " serviceId="
                  + serviceResourceId)
          .toRuntimeException();
    }
    if (!(tabularManager instanceof TabularManagerImpl localTabularManager)) {
      return;
    }
    try {
      localTabularManager.installDataset(datasetId.toString());
    } catch (RuntimeException ex) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "Dataset definition not available for query datasetId="
                  + datasetId
                  + " serviceId="
                  + serviceResourceId)
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private ByteBuffer queryIpcSqlAsCaller(
      RequestIdentity identity, String sql, DatasetId datasetId) {
    String plannedSql = Strings.requireNonBlank(sql, "sql");
    String callerSubject = identity.effectiveUserSubject();
    try {
      return tabularManager.queryIpcBuffer(plannedSql);
    } catch (IllegalArgumentException ex) {
      String details = errorContext(callerSubject, datasetId);
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(ex.getMessage() + " [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    } catch (IllegalStateException ex) {
      throw mapQueryFailure(ex, callerSubject, datasetId);
    } catch (RuntimeException ex) {
      String details = errorContext(callerSubject, datasetId);
      LOGGER.error("[FlightSQL][query][unexpected] {}", details, ex);
      throw CallStatus.INTERNAL
          .withDescription("Query execution failed [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private static String rewriteDatasetPlaceholder(String sql, DatasetId datasetId) {
    if (datasetId == null) {
      return sql;
    }
    int first = sql.indexOf(DATASET_TOKEN_MARKER);
    if (first < 0) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "query.sql must contain placeholder "
                  + DATASET_TOKEN_MARKER
                  + " for dataset "
                  + datasetId)
          .toRuntimeException();
    }
    int second = sql.indexOf(DATASET_TOKEN_MARKER, first + DATASET_TOKEN_MARKER.length());
    if (second >= 0) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "query.sql must contain placeholder "
                  + DATASET_TOKEN_MARKER
                  + " exactly once for dataset "
                  + datasetId)
          .toRuntimeException();
    }
    return sql.replace(DATASET_TOKEN_MARKER, sqlToken(datasetId));
  }

  private static String datasetTokenMarker() {
    return "$" + "{DATASET}";

  private boolean isAllowedDataset(DatasetId datasetId) {
    if (datasetId == null) {
      return true;
    }
    return allowedDatasetIds.isEmpty() || allowedDatasetIds.contains(datasetId);
  }

  private static String sqlToken(DatasetId datasetId) {
    return quoteIdentifier(datasetId.group())
        + "."
        + quoteIdentifier(datasetId.namespace())
        + "."
        + quoteIdentifier(datasetId.name());
  }

  private static String quoteIdentifier(String identifier) {
    return "\"" + identifier.replace("\"", "\"\"") + "\"";
  }

  private RuntimeException mapQueryFailure(
      IllegalStateException ex, String callerSubject, DatasetId datasetId) {
    String message = Strings.trimToNull(ex.getMessage());
    String details = errorContext(callerSubject, datasetId);
    if (message == null) {
      LOGGER.error("[FlightSQL][query][unexpected] {} message=<empty>", details, ex);
      return CallStatus.INTERNAL
          .withDescription("Query execution failed [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    }
    String lower = message.toLowerCase();
    if (lower.contains("access denied for dataset")) {
      return CallStatus.UNAUTHORIZED
          .withDescription(message + " [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    }
    if (lower.contains("${dataset}")
        || lower.contains("dataset not found")
        || lower.contains("unknown table token")
        || lower.contains("ambiguous table token")) {
      return CallStatus.INVALID_ARGUMENT
          .withDescription(message + " [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    }
    if (lower.contains("duckdb sql failed")) {
      return CallStatus.INVALID_ARGUMENT
          .withDescription(message + " [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    }
    LOGGER.error("[FlightSQL][query][unexpected] {} message={}", details, message, ex);
    return CallStatus.INTERNAL
        .withDescription(message + " [" + details + "]")
        .withCause(ex)
        .toRuntimeException();
  }

  private IpcSummary inspectIpc(ByteBuffer ipc) {
    if (ipc == null || !ipc.hasRemaining()) {
      throw CallStatus.INTERNAL.withDescription("Query returned empty Arrow IPC payload").toRuntimeException();
    }
    try (ArrowStreamReader reader =
        new ArrowStreamReader(Bytes.asInputStream(ipc), ipcReadAllocator)) {
      VectorSchemaRoot root = reader.getVectorSchemaRoot();
      Schema schema = root.getSchema();
      long rowCount = 0L;
      while (reader.loadNextBatch()) {
        rowCount += root.getRowCount();
      }
      return new IpcSummary(schema, rowCount);
    } catch (Exception ex) {
      throw CallStatus.INTERNAL
          .withDescription("Failed to inspect Arrow IPC payload")
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private void streamIpc(FlightProducer.ServerStreamListener listener, ByteBuffer ipc) {
    Objects.requireNonNull(listener, "listener");
    try (ArrowStreamReader reader =
        new ArrowStreamReader(Bytes.asInputStream(ipc), ipcReadAllocator)) {
      VectorSchemaRoot root = reader.getVectorSchemaRoot();
      listener.start(root);
      while (reader.loadNextBatch()) {
        listener.putNext();
      }
      listener.completed();
    } catch (Exception ex) {
      throw CallStatus.INTERNAL
          .withDescription("Failed to stream Arrow IPC payload")
          .withCause(ex)
          .toRuntimeException();
    }
  }
