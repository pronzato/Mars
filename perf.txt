package org.pronzato.fabric.vaadin;

import com.vaadin.flow.component.html.H2;
import com.vaadin.flow.component.html.Paragraph;
import com.vaadin.flow.component.orderedlayout.VerticalLayout;
import com.vaadin.flow.router.Route;
import java.util.Map;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

@SpringBootApplication
@Route("")
public class FabricLiteVaadinApp extends VerticalLayout {

  public FabricLiteVaadinApp() {
    setPadding(true);
    setSpacing(true);
    add(
        new H2("fabric-lite-vaadin"),
        new Paragraph("Standalone Vaadin test page is running."));
  }

  public static void main(String[] args) {
    SpringApplication app = new SpringApplication(FabricLiteVaadinApp.class);
    app.setDefaultProperties(Map.of("server.port", "8080"));
    app.run(args);
  }
}
