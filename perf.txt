# Full npm registry URL probe script with progress logging
# - Scans package-lock.json files
# - Extracts registry tarball URLs
# - Probes each URL through your proxy
# - Logs progress and writes blocked list to disk

$ErrorActionPreference = "Continue"

# REQUIRED: set your authenticated proxy URL
$proxy = "http://USERNAME:PASSWORD@PROXY_HOST:PROXY_PORT"

# Optional output dir
$outDir = ".tmp"
New-Item -ItemType Directory -Force -Path $outDir | Out-Null

Write-Host "Finding package-lock.json files..."
$lockfiles = Get-ChildItem -Recurse -Filter package-lock.json |
  Where-Object { $_.FullName -notmatch "\\node_modules\\" }

if (-not $lockfiles) {
  Write-Host "No package-lock.json files found." -ForegroundColor Yellow
  exit 1
}

Write-Host "Lockfiles found: $($lockfiles.Count)"
$lockfiles | ForEach-Object { Write-Host "  - $($_.FullName)" }

Write-Host "Extracting registry tarball URLs..."
$allUrls =
  $lockfiles |
  ForEach-Object { Get-Content $_.FullName -Raw } |
  ForEach-Object {
    [regex]::Matches(
      $_,
      '"resolved"\s*:\s*"(?<u>https://registry\.npmjs\.org/[^"]+\.tgz)"'
    ) | ForEach-Object { $_.Groups["u"].Value }
  } |
  Sort-Object -Unique

if (-not $allUrls) {
  Write-Host "No registry tarball URLs found in lockfiles." -ForegroundColor Yellow
  exit 1
}

Write-Host "URLs to probe: $($allUrls.Count)"

$blocked = New-Object System.Collections.Generic.List[string]
$okCount = 0
$errCount = 0
$total = $allUrls.Count
$i = 0

foreach ($u in $allUrls) {
  $i++
  Write-Host ("[{0}/{1}] {2}" -f $i, $total, $u)

  $tmp = New-TemporaryFile
  try {
    # Fetch first byte only, still enough to detect block pages/errors
    curl.exe -sS -L --proxy $proxy --range 0-0 -o $tmp $u 2>$null | Out-Null
    $body = Get-Content $tmp -Raw -ErrorAction SilentlyContinue

    if ($body -match "GIS_Malicious_Block") {
      Write-Host "  -> BLOCKED (GIS)" -ForegroundColor Yellow
      $blocked.Add($u)
    } else {
      Write-Host "  -> OK" -ForegroundColor DarkGreen
      $okCount++
    }
  } catch {
    $msg = $_.Exception.Message
    Write-Host ("  -> ERROR: {0}" -f $msg) -ForegroundColor Red
    $blocked.Add($u)
    $errCount++
  } finally {
    Remove-Item $tmp -ErrorAction SilentlyContinue
  }
}

$blockedUnique = $blocked | Sort-Object -Unique
$blockedFile = Join-Path $outDir "blocked-full-list.txt"
$allFile = Join-Path $outDir "all-registry-urls.txt"

$allUrls | Set-Content -Path $allFile -Encoding UTF8
$blockedUnique | Set-Content -Path $blockedFile -Encoding UTF8

Write-Host ""
Write-Host "Done."
Write-Host "Total URLs:   $total"
Write-Host "OK:           $okCount"
Write-Host "Errors:       $errCount"
Write-Host "Blocked/Fail: $($blockedUnique.Count)"
Write-Host "All URLs:     $allFile"
Write-Host "Blocked list: $blockedFile"
