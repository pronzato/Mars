package org.pronzato.fabric.lite.config;

import com.crankuptheamps.client.Command;
import com.crankuptheamps.client.HAClient;
import com.crankuptheamps.client.Message;
import com.crankuptheamps.client.MessageStream;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;
import java.util.Locale;
import java.util.stream.Stream;
import org.pronzato.fabric.api.connectors.amps.factory.AmpsFactoryClient;
import org.pronzato.fabric.api.connectors.amps.factory.AmpsFactoryOptions;

public abstract class FabricConfig {
  protected static final ObjectMapper MAPPER = new ObjectMapper();
  protected static final String TOPIC = "/fabric/discovery";

  private final String discovery;
  private final String kerberosUsername;
  private final String kerberosKeytab;
  private final Path jsonRoot;

  protected FabricConfig(
      String discovery, String kerberosUsername, String kerberosKeytab, Path jsonRoot) {
    this.discovery = requireNonBlank(discovery, "discovery");
    this.kerberosUsername = requireNonBlank(kerberosUsername, "kerberosUsername");
    this.kerberosKeytab = requireNonBlank(kerberosKeytab, "kerberosKeytab");
    this.jsonRoot = jsonRoot == null ? Path.of("vault", "git", "dev") : jsonRoot;
  }

  protected final HAClient connect(String clientName) throws Exception {
    List<String> endpoints = parseDiscovery(discovery);
    if (endpoints.isEmpty()) {
      throw new IllegalArgumentException("No AMPS discovery endpoints configured.");
    }

    AmpsFactoryOptions options = new AmpsFactoryOptions();
    options.withKerberos(
        kerberos ->
            kerberos
                .enabled(true)
                .principal(kerberosUsername)
                .keytab(Path.of(kerberosKeytab).toAbsolutePath().toString()));

    HAClient client = AmpsFactoryClient.create(clientName, options);
    try {
      AmpsFactoryClient.connect(client, endpoints, options);
      return client;
    } catch (Exception ex) {
      try {
        client.close();
      } catch (Exception ignored) {
      }
      throw ex;
    }
  }

  protected final void clearTopic(HAClient client) throws Exception {
    Command command = new Command("sow_delete");
    command.setTopic(TOPIC);
    command.setFilter("1=1");
    try (MessageStream stream = client.execute(command)) {
      for (Message ignored : stream) {
        // consume stream to ensure command completes
      }
    }
  }

  protected final Path jsonRoot() {
    return jsonRoot;
  }

  protected final List<Path> listJsonFiles() throws IOException {
    if (Files.notExists(jsonRoot)) {
      return List.of();
    }
    try (Stream<Path> stream = Files.walk(jsonRoot)) {
      return stream
          .filter(Files::isRegularFile)
          .filter(path -> path.getFileName().toString().toLowerCase(Locale.ROOT).endsWith(".json"))
          .sorted(Comparator.comparing(Path::toString))
          .toList();
    }
  }

  protected final JsonNode parseOrThrow(String payload, String errorPrefix) {
    try {
      return MAPPER.readTree(payload);
    } catch (Exception ex) {
      throw new IllegalStateException(errorPrefix + " message=" + trim(ex.getMessage()), ex);
    }
  }

  protected final EnvelopeKey extractEnvelope(JsonNode node, String sowKey, String payload) {
    String group = pointerText(node, "/envelope/group");
    String type = pointerText(node, "/envelope/type");
    String name = pointerText(node, "/envelope/name");

    if (group != null && type != null && name != null) {
      return new EnvelopeKey(group, type, name);
    }

    String[] parts = parseCompositeSowKey(sowKey);
    if (parts != null) {
      String fromSowGroup = group != null ? group : parts[0];
      String fromSowType = type != null ? type : parts[1];
      String fromSowName = name != null ? name : parts[2];
      if (fromSowGroup != null && fromSowType != null && fromSowName != null) {
        return new EnvelopeKey(fromSowGroup, fromSowType, fromSowName);
      }
    }

    throw new IllegalStateException(
        "Missing envelope.group/type/name and unable to infer from sowKey="
            + trim(sowKey)
            + " payloadPrefix="
            + abbreviate(payload, 120));
  }

  /**
   * Default on-disk layout for legacy/other assets:
   *   <root>/<group>/<type>/<name>.json
   *
   * DATASET / METASET folder layout is handled by Import/Export, not here.
   */
  protected final Path toFilePath(EnvelopeKey key) {
    return jsonRoot
        .resolve(safeDirectoryName(key.group()))
        .resolve(safeDirectoryName(key.type()))
        .resolve(safeFileName(key.name()) + ".json");
  }

  protected static String trim(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }

  protected static boolean isBlank(String value) {
    return value == null || value.trim().isEmpty();
  }

  protected static String safeFileName(String value) {
    String trimmed = trim(value);
    if (trimmed == null) {
      return "unknown";
    }
    StringBuilder builder = new StringBuilder(trimmed.length());
    for (int index = 0; index < trimmed.length(); index++) {
      char ch = trimmed.charAt(index);
      if (isInvalidPathChar(ch)) {
        builder.append('_');
      } else {
        builder.append(ch);
      }
    }
    String normalized = trimTrailingDotsAndSpaces(builder.toString());
    return normalized.isBlank() ? "unknown" : normalized;
  }

  protected static String safeDirectoryName(String value) {
    return safeFileName(value).toLowerCase(Locale.ROOT);
  }

  private static List<String> parseDiscovery(String value) {
    String raw = trim(value);
    if (raw == null) {
      return List.of();
    }
    String[] parts = raw.split(",");
    List<String> endpoints = new ArrayList<>(parts.length);
    for (String part : parts) {
      String item = trim(part);
      if (item != null) {
        endpoints.add(normalizeEndpoint(item));
      }
    }
    return List.copyOf(endpoints);
  }

  private static String normalizeEndpoint(String value) {
    if (value.contains("://")) {
      return value;
    }
    int index = value.lastIndexOf(':');
    if (index <= 0 || index >= value.length() - 1) {
      throw new IllegalArgumentException(
          "Invalid discovery endpoint '" + value + "'. Expected host:port or full URI.");
    }
    String host = value.substring(0, index).trim();
    String portPart = value.substring(index + 1).trim();
    if (host.isEmpty()) {
      throw new IllegalArgumentException("Discovery endpoint host is required: " + value);
    }
    try {
      Integer.parseInt(portPart);
    } catch (NumberFormatException ex) {
      throw new IllegalArgumentException("Discovery endpoint port is invalid: " + value, ex);
    }
    return "tcp://" + host + ":" + portPart + "/amps/json";
  }

  private static String pointerText(JsonNode node, String pointer) {
    if (node == null) {
      return null;
    }
    JsonNode value = node.at(pointer);
    if (value == null || value.isMissingNode() || value.isNull()) {
      return null;
    }
    return trim(value.asText(null));
  }

  /**
   * Supports:
   * - group|TYPE|name
   * - group|DATASET|namespace|name
   *
   * Returns [group, type, name] for envelope inference.
   */
  private static String[] parseCompositeSowKey(String sowKey) {
    String normalized = trim(sowKey);
    if (normalized == null) {
      return null;
    }
    String[] parts = normalized.split("\\|", -1);
    if (parts.length < 3) {
      return null;
    }
    String group = trim(parts[0]);
    String type = trim(parts[1]);
    if (group == null || type == null) {
      return null;
    }

    String upper = type.toUpperCase(Locale.ROOT);
    if ("DATASET".equals(upper) && parts.length >= 4) {
      // group|DATASET|namespace|name
      String name = trim(parts[3]);
      if (name == null) {
        return null;
      }
      return new String[] {group, type, name};
    }

    // group|TYPE|name
    String name = trim(parts[2]);
    if (name == null) {
      return null;
    }
    return new String[] {group, type, name};
  }

  private static String abbreviate(String value, int max) {
    if (value == null) {
      return "";
    }
    String normalized = value.replace('\n', ' ').replace('\r', ' ');
    if (normalized.length() <= max) {
      return normalized;
    }
    return normalized.substring(0, Math.max(1, max)) + "...";
  }

  private static String requireNonBlank(String value, String field) {
    String trimmed = trim(value);
    if (trimmed == null) {
      throw new IllegalArgumentException(field + " is required");
    }
    return trimmed;
  }

  protected record EnvelopeKey(String group, String type, String name) {}

  private static boolean isInvalidPathChar(char ch) {
    if (ch < 32) {
      return true;
    }
    return switch (ch) {
      case '/', '\\', '<', '>', ':', '"', '|', '?', '*' -> true;
      default -> false;
    };
  }

  private static String trimTrailingDotsAndSpaces(String value) {
    if (value == null || value.isEmpty()) {
      return value;
    }
    int end = value.length();
    while (end > 0) {
      char ch = value.charAt(end - 1);
      if (ch == ' ' || ch == '.') {
        end--;
      } else {
        break;
      }
    }
    return end == value.length() ? value : value.substring(0, end);
  }
}













package org.pronzato.fabric.lite.config;

import org.pronzato.fabric.api.control.discovery.DiscoveryUrl;
import com.crankuptheamps.client.Command;
import com.crankuptheamps.client.HAClient;
import com.crankuptheamps.client.Message;
import com.crankuptheamps.client.MessageStream;
import com.fasterxml.jackson.databind.JsonNode;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.time.Duration;
import java.util.ArrayList;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Locale;
import java.util.Objects;
import java.util.Set;
import org.pronzato.fabric.api.control.discovery.codec.DiscoveryCodec;
import org.pronzato.fabric.api.control.discovery.definitions.Definition;

public final class FabricConfigImport extends FabricConfig {
  private FabricConfigImport(
      String discovery, String kerberosUsername, String kerberosKeytab, Path jsonRoot) {
    super(discovery, kerberosUsername, kerberosKeytab, jsonRoot);
  }

  public static void main(String[] args) {
    String discovery = DiscoveryUrl.URL;
    String kerberosUsername = "dfabric";
    String kerberosKeytab = "vault/kerberos/dfabric.keytab";
    Path jsonRoot = Path.of("vault", "git", "dev");

    boolean wipe = true;
    String envOverride = null;
    for (String arg : args == null ? new String[0] : args) {
      if ("--wipe".equals(arg)) {
        wipe = true;
      } else if (arg.startsWith("--env=")) {
        envOverride = trim(arg.substring("--env=".length()));
      } else if ("--help".equals(arg) || "-h".equals(arg)) {
        printUsage();
        return;
      } else {
        System.err.println("[LITE_CONFIG_IMPORT][ERROR] Unknown argument: " + arg);
        printUsage();
        System.exit(2);
      }
    }

    int code = 0;
    try {
      code = new FabricConfigImport(discovery, kerberosUsername, kerberosKeytab, jsonRoot).run(wipe, envOverride);
    } catch (Exception ex) {
      System.err.println("[LITE_CONFIG_IMPORT][ERROR] " + rootCauseMessage(ex));
      code = 1;
    }
    if (code != 0) {
      System.exit(code);
    }
  }

  private int run(boolean wipe, String envOverride) throws Exception {
    long started = System.nanoTime();
    ScanResult scan = scanInputFiles();
    List<Path> files = scan.files();
    List<ImportEntry> validated = new ArrayList<>(files.size());
    boolean hasValidationErrors = false;
    int duplicateKeys = 0;
    LinkedHashSet<String> seenKeys = new LinkedHashSet<>();

    System.out.println(
        "[LiteConfigImport][Scan] env="
            + scan.env()
            + " groups="
            + scan.groups()
            + " schemasFiles="
            + scan.schemasFiles()
            + " configFiles="
            + scan.configFiles());
    System.out.println(
        "[LITE_CONFIG_IMPORT][SCAN] root="
            + jsonRoot().toAbsolutePath().normalize()
            + " discoveredFiles="
            + files.size());

    for (Path file : files) {
      String rawPayload = Files.readString(file, StandardCharsets.UTF_8);
      String payload = stripUtf8Bom(rawPayload);
      if (!Objects.equals(rawPayload, payload)) {
        System.out.println("[LiteConfigImport][Sanitize] file=" + file + " strippedBom=true");
      }
      try {
        JsonNode node = MAPPER.readTree(payload);
        EnvelopeKey envelope = extractEnvelope(node, null, payload);
        String envelopeKey = toEnvelopeKey(envelope);
        String normalizedPayload = normalizePayload(node);

        boolean duplicate = !seenKeys.add(envelopeKey);
        if (duplicate) {
          duplicateKeys++;
        }
        validated.add(new ImportEntry(file, normalizedPayload, envelopeKey, duplicate));
        System.out.println(
            "[LiteConfigImport][Key] file="
                + file
                + " envelopeKey="
                + envelopeKey
                + " status="
                + (duplicate ? "DUPLICATE" : "NEW"));
      } catch (Exception ex) {
        hasValidationErrors = true;
        System.err.println(
            "[LITE_CONFIG_IMPORT][ERROR] invalid-json path="
                + file
                + " message="
                + rootCauseMessage(ex));
      }
    }

    if (hasValidationErrors) {
      System.err.println("[LITE_CONFIG_IMPORT][ABORT] JSON validation failed. Nothing published.");
      return 2;
    }

    String env = isBlank(envOverride) ? scan.env() : envOverride;
    System.out.println("[LITE_CONFIG_IMPORT][MODE] wipe=" + wipe + " topic=" + TOPIC + " env=" + env);
    int published = 0;

    LinkedHashSet<String> expectedEnvelopeKeys = new LinkedHashSet<>();
    for (ImportEntry entry : validated) {
      expectedEnvelopeKeys.add(entry.envelopeKey());
    }

    try (HAClient client = connect("fabric-lite-config-import")) {
      if (wipe) {
        System.out.println("[LITE_CONFIG_IMPORT][CLEAR] deleting all records from " + TOPIC);
        clearTopic(client);
      }

      for (ImportEntry entry : validated) {
        Command publish = new Command("publish");
        publish.setTopic(TOPIC);
        publish.setData(entry.payload());
        client.execute(publish);
        published++;
      }

      SowSnapshot sow = loadSowSnapshot(client);
      LinkedHashSet<String> missing = new LinkedHashSet<>(expectedEnvelopeKeys);
      missing.removeAll(sow.actualEnvelopeKeys());
      LinkedHashSet<String> extra = new LinkedHashSet<>(sow.actualEnvelopeKeys());
      extra.removeAll(expectedEnvelopeKeys);

      System.out.println(
          "[LITE_CONFIG_IMPORT][SOW] topic="
              + TOPIC
              + " sowRowCount="
              + sow.sowRowCount()
              + " envelopeKeysSample="
              + firstN(sow.actualEnvelopeKeys(), 20));
      System.out.println(
          "[LITE_CONFIG_IMPORT][DIFF] expectedKeys="
              + expectedEnvelopeKeys.size()
              + " actualKeys="
              + sow.actualEnvelopeKeys().size()
              + " missingKeys="
              + missing.size()
              + " missingSample="
              + firstN(missing, 20)
              + " extraKeys="
              + extra.size()
              + " extraSample="
              + firstN(extra, 20));
    }

    long durationMs = Duration.ofNanos(System.nanoTime() - started).toMillis();
    System.out.println(
        "[LITE_CONFIG_IMPORT][SUMMARY] files="
            + files.size()
            + " validJson="
            + validated.size()
            + " published="
            + published
            + " uniqueKeys="
            + seenKeys.size()
            + " duplicateKeys="
            + duplicateKeys
            + " wipe="
            + wipe
            + " durationMs="
            + durationMs);
    return 0;
  }

  private ScanResult scanInputFiles() throws Exception {
    Path root = jsonRoot().toAbsolutePath().normalize();
    if (Files.notExists(root)) {
      return new ScanResult(
          root.getFileName() == null ? "unknown" : root.getFileName().toString(),
          List.of(),
          0,
          0,
          List.of());
    }
    List<Path> groupDirs =
        Files.list(root)
            .filter(Files::isDirectory)
            .sorted()
            .toList();
    List<String> groups = new ArrayList<>();
    List<Path> files = new ArrayList<>();
    int schemasFiles = 0;
    int configFiles = 0;

    for (Path groupDir : groupDirs) {
      String group = groupDir.getFileName() == null ? null : groupDir.getFileName().toString();
      if (group == null || group.isBlank()) {
        continue;
      }
      groups.add(group);
      try (var stream = Files.walk(groupDir)) {
        List<Path> groupFiles =
            stream
                .filter(Files::isRegularFile)
                .filter(
                    path ->
                        path.getFileName()
                            .toString()
                            .toLowerCase(Locale.ROOT)
                            .endsWith(".json"))
                .sorted()
                .toList();
        files.addAll(groupFiles);
        for (Path file : groupFiles) {
          Path relative = groupDir.relativize(file);
          if (relative.getNameCount() > 1) {
            String top = relative.getName(0).toString();
            if ("schemas".equalsIgnoreCase(top)) {
              schemasFiles++;
            } else {
              configFiles++;
            }
          }
        }
      }
    }
    files.sort((left, right) -> left.toString().compareTo(right.toString()));
    String env = root.getFileName() == null ? "unknown" : root.getFileName().toString();
    return new ScanResult(env, List.copyOf(groups), schemasFiles, configFiles, List.copyOf(files));
  }

  private SowSnapshot loadSowSnapshot(HAClient client) throws Exception {
    LinkedHashSet<String> actualEnvelopeKeys = new LinkedHashSet<>();
    int rows = 0;
    try (MessageStream stream = client.sow(TOPIC)) {
      for (Message message : stream) {
        if (message == null || message.getCommand() != Message.Command.SOW) {
          continue;
        }
        String payload = message.getData();
        if (isBlank(payload)) {
          continue;
        }
        payload = stripUtf8Bom(payload);
        rows++;
        JsonNode node =
            parseOrThrow(payload, "SOW payload parse failed for key=" + trim(message.getSowKey()));
        EnvelopeKey envelope = extractEnvelope(node, null, payload);
        actualEnvelopeKeys.add(toEnvelopeKey(envelope));
      }
    }
    return new SowSnapshot(rows, Set.copyOf(actualEnvelopeKeys));
  }

  private static String toEnvelopeKey(EnvelopeKey envelope) {
    return envelope.group() + "|" + envelope.type() + "|" + envelope.name();
  }

  private static String rootCauseMessage(Throwable throwable) {
    if (throwable == null) {
      return "<unknown>";
    }
    Throwable root = throwable;
    while (root.getCause() != null && root.getCause() != root) {
      root = root.getCause();
    }
    String message = trim(root.getMessage());
    return message == null
        ? root.getClass().getSimpleName()
        : root.getClass().getSimpleName() + ": " + message;
  }

  private static List<String> firstN(Iterable<String> values, int max) {
    if (values == null || max <= 0) {
      return List.of();
    }
    List<String> out = new ArrayList<>();
    for (String value : values) {
      if (value == null) {
        continue;
      }
      out.add(value);
      if (out.size() >= max) {
        break;
      }
    }
    return List.copyOf(out);
  }

  private static String stripUtf8Bom(String value) {
    if (value == null || value.isEmpty()) {
      return value;
    }
    return value.charAt(0) == '\uFEFF' ? value.substring(1) : value;
  }

  private static String normalizePayload(JsonNode node) throws Exception {
    Definition decoded = DiscoveryCodec.decode(node);
    if (decoded == null) {
      return MAPPER.writerWithDefaultPrettyPrinter().writeValueAsString(node);
    }
    return DiscoveryCodec.mapper().writerWithDefaultPrettyPrinter().writeValueAsString(decoded);
  }

  private record ImportEntry(Path file, String payload, String envelopeKey, boolean duplicate) {}

  private record ScanResult(
      String env, List<String> groups, int schemasFiles, int configFiles, List<Path> files) {}

  private record SowSnapshot(int sowRowCount, Set<String> actualEnvelopeKeys) {}

  private static void printUsage() {
    System.out.println("Usage: import [--wipe] [--env=<dev|uat|prod|...>]");
  }
}













package org.pronzato.fabric.lite.config;

import org.pronzato.fabric.api.control.discovery.DiscoveryUrl;
import com.crankuptheamps.client.HAClient;
import com.crankuptheamps.client.Message;
import com.crankuptheamps.client.MessageStream;
import com.fasterxml.jackson.databind.JsonNode;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.StandardOpenOption;
import java.util.HashSet;
import java.util.Set;
import org.pronzato.fabric.api.control.discovery.codec.DiscoveryCodec;
import org.pronzato.fabric.api.control.discovery.definitions.Definition;

public final class FabricConfigExport extends FabricConfig {
  private FabricConfigExport(
      String discovery, String kerberosUsername, String kerberosKeytab, Path jsonRoot) {
    super(discovery, kerberosUsername, kerberosKeytab, jsonRoot);
  }

  public static void main(String[] args) {
    String discovery = DiscoveryUrl.URL;
    String kerberosUsername = "dfabric";
    String kerberosKeytab = "vault/kerberos/dfabric.keytab";
    Path jsonRoot = Path.of("vault", "git", "dev");

    for (String arg : args == null ? new String[0] : args) {
      if ("--help".equals(arg) || "-h".equals(arg)) {
        printUsage();
        return;
      }
      System.err.println("[LITE_CONFIG_EXPORT][ERROR] Unknown argument: " + arg);
      printUsage();
      System.exit(2);
    }

    int code = 0;
    try {
      code =
          new FabricConfigExport(discovery, kerberosUsername, kerberosKeytab, jsonRoot).run();
    } catch (Exception ex) {
      System.err.println("[LITE_CONFIG_EXPORT][ERROR] " + ex.getMessage());
      ex.printStackTrace(System.err);
      code = 1;
    }
    if (code != 0) {
      System.exit(code);
    }
  }

  private int run() throws Exception {
    if (Files.notExists(jsonRoot())) {
      Files.createDirectories(jsonRoot());
    }

    Set<Path> exported = new HashSet<>();
    int exportedCount = 0;

    try (HAClient client = connect("fabric-lite-config-export");
        MessageStream stream = client.sow(TOPIC)) {
      for (Message message : stream) {
        if (message == null || message.getCommand() != Message.Command.SOW) {
          continue;
        }
        String payload = message.getData();
        if (isBlank(payload)) {
          continue;
        }
        JsonNode node = parseOrThrow(payload, "SOW payload parse failed for key=" + message.getSowKey());
        EnvelopeKey key = extractEnvelope(node, null, payload);
        String prettyPayload = normalizePayload(node);

        Path target = toExportPath(key);
        Files.createDirectories(target.getParent());
        Files.writeString(
            target,
            prettyPayload,
            StandardCharsets.UTF_8,
            StandardOpenOption.CREATE,
            StandardOpenOption.TRUNCATE_EXISTING,
            StandardOpenOption.WRITE);

        exported.add(target.normalize().toAbsolutePath());
        exportedCount++;
      }
    }

    int deletedCount = 0;
    for (Path file : listJsonFiles()) {
      Path normalized = file.normalize().toAbsolutePath();
      if (!exported.contains(normalized)) {
        Files.deleteIfExists(file);
        deletedCount++;
        System.out.println("[LITE_CONFIG_EXPORT][DELETE] path=" + file);
      }
    }

    System.out.println(
        "[LITE_CONFIG_EXPORT][SUMMARY] exported=" + exportedCount + " deleted=" + deletedCount);
    return 0;
  }

  private static void printUsage() {
    System.out.println("Usage: export");
  }

  private Path toExportPath(EnvelopeKey key) {
    if (key == null) {
      throw new IllegalArgumentException("envelope key is required");
    }
    String normalizedName = key.name();
    if ("CONFIG".equalsIgnoreCase(key.type())) {
      normalizedName = stripConfigFilePrefix(normalizedName);
    }
    normalizedName = normalizeExportName(key.type(), normalizedName);
    EnvelopeKey normalizedKey = new EnvelopeKey(key.group(), key.type(), normalizedName);
    return toFilePath(normalizedKey);
  }

  private static String stripConfigFilePrefix(String name) {
    if (name == null) {
      return null;
    }
    if (name.regionMatches(true, 0, "config__", 0, "config__".length())) {
      return name.substring("config__".length());
    }
    return name;
  }

  private static String normalizeExportName(String type, String name) {
    if (name == null) {
      return null;
    }
    if (!"ENTITLEMENT".equalsIgnoreCase(type)) {
      return name;
    }
    if (name.startsWith("overrides/services/")) {
      return name.substring("overrides/".length());
    }
    if (name.startsWith("overrides/namespaces/")) {
      return name.substring("overrides/".length());
    }
    return name;
  }

  private static String normalizePayload(JsonNode node) throws Exception {
    Definition decoded = DiscoveryCodec.decode(node);
    if (decoded == null) {
      return MAPPER.writerWithDefaultPrettyPrinter().writeValueAsString(node);
    }
    return DiscoveryCodec.mapper().writerWithDefaultPrettyPrinter().writeValueAsString(decoded);
  }
}










package org.pronzato.fabric.lite.config;

import org.pronzato.fabric.api.control.discovery.DiscoveryUrl;
import com.crankuptheamps.client.HAClient;
import com.crankuptheamps.client.Message;
import com.crankuptheamps.client.MessageStream;
import com.fasterxml.jackson.databind.JsonNode;
import java.nio.file.Path;

public final class FabricConfigDump extends FabricConfig {
  private FabricConfigDump(
      String discovery, String kerberosUsername, String kerberosKeytab, Path jsonRoot) {
    super(discovery, kerberosUsername, kerberosKeytab, jsonRoot);
  }

  public static void main(String[] args) {
    String discovery = DiscoveryUrl.URL;
    String kerberosUsername = "dfabric";
    String kerberosKeytab = "vault/kerberos/dfabric.keytab";
    Path jsonRoot = Path.of("vault", "git", "dev");

    for (String arg : args == null ? new String[0] : args) {
      if ("--help".equals(arg) || "-h".equals(arg)) {
        printUsage();
        return;
      }
      System.err.println("[LITE_CONFIG_DUMP][ERROR] Unknown argument: " + arg);
      printUsage();
      System.exit(2);
    }

    int code = 0;
    try {
      code = new FabricConfigDump(discovery, kerberosUsername, kerberosKeytab, jsonRoot).run();
    } catch (Exception ex) {
      System.err.println("[LITE_CONFIG_DUMP][ERROR] " + ex.getMessage());
      ex.printStackTrace(System.err);
      code = 1;
    }
    if (code != 0) {
      System.exit(code);
    }
  }

  private int run() throws Exception {
    try (HAClient client = connect("fabric-lite-config-dump");
        MessageStream stream = client.sow(TOPIC)) {
      for (Message message : stream) {
        if (message == null || message.getCommand() != Message.Command.SOW) {
          continue;
        }
        String payload = message.getData();
        if (isBlank(payload)) {
          continue;
        }
        JsonNode node =
            parseOrThrow(payload, "SOW payload parse failed for key=" + message.getSowKey());
        EnvelopeKey key = extractEnvelope(node, message.getSowKey(), payload);
        System.out.println("[DUMP] group=" + key.group() + " type=" + key.type() + " name=" + key.name());
      }
    }
    return 0;
  }

  private static void printUsage() {
    System.out.println("Usage: dump");
  }
}
