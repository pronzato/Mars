public final class ConfigWorkflowService implements ConfigAdminService.Workflow {
  private static final Logger LOGGER = LoggerFactory.getLogger(ConfigWorkflowService.class);
  private static final int AUDIT_MAX = 100;
  private static final int HISTORY_MAX = ConfigHistoryUtil.DEFAULT_HISTORY_LIMIT;
  private static final String SCHEMA_FORMAT = "FABRIC_SCHEMA_CONFIG";

  private final Store store;

  public ConfigWorkflowService(Store store) {
    this.store = Objects.requireNonNull(store, "store");
  }

  @Override
  public DefinitionConfigChange workflow(String group, String configId, String layerId, int auditLimit) {
    String requiredGroup = required(group, "group");
    String requiredconfigId = required(configId, "configId");
    String requiredLayerId = required(layerId, "layerId");

    DefinitionConfigLayer published = published(requiredGroup, requiredconfigId, requiredLayerId);
    DefinitionConfigChange change = loadChange(requiredGroup, requiredconfigId, requiredLayerId);
    if (change == null) {
      return emptyChange(requiredGroup, requiredconfigId, requiredLayerId, published);
    }
    return clampAudit(change, auditLimit);
  }

  @Override
  public JsonNode draft(String group, String configId, String layerId) {
    DefinitionConfigChange change = requiredChange(group, configId, layerId);
    JsonNode draft = change.draftDeclared();
    if (draft == null || !draft.isObject()) {
      throw new WorkflowDraftNotFoundException("draft not found");
    }
    return draft.deepCopy();
  }

  @Override
  public DefinitionConfigChange propose(
      String group,
      String configId,
      String layerId,
      JsonNode declaredPayload,
      DefinitionConfigChange.ChangeMode mode,
      Instant effectiveAt,
      Instant expiresAt,
      String comment,
      String subject) {
    String requiredGroup = required(group, "group");
    String requiredconfigId = required(configId, "configId");
    String requiredLayerId = required(layerId, "layerId");
    String resolvedSubject = resolveSubject(subject);
    JsonNode draft = requireObject(declaredPayload, "declaredPayload").deepCopy();

    DefinitionConfigLayer published = published(requiredGroup, requiredconfigId, requiredLayerId);
    validateDraft(requiredGroup, published, draft);

    DefinitionConfigChange existing = loadChange(requiredGroup, requiredconfigId, requiredLayerId);
    DefinitionConfigChange.ChangeState from = stateOf(existing);
    DefinitionConfigChange.ChangeMode resolvedMode =
        mode == null ? DefinitionConfigChange.ChangeMode.IMMEDIATE : mode;
    Window window = validateWindow(resolvedMode, effectiveAt, expiresAt, Instant.now());
    String now = Instant.now().toString();
    String normalizedComment = Strings.trimToNull(comment);
    String baseCreatedAt = now;
    String baseCreatedBy = resolvedSubject;
    String previousPublishedConfigId = existing == null ? null : Strings.trimToNull(existing.lastPublishedConfigId());

    DefinitionConfigChange.ChangeEntry nextEntry =
        new DefinitionConfigChange.ChangeEntry(
            UUID.randomUUID().toString(),
            DefinitionConfigChange.ChangeState.PROPOSED,
            resolvedMode,
            window.effectiveAtUtc(),
            window.expiresAtUtc(),
            draft,
            Strings.trimToNull(published.configId()),
            Strings.trimToNull(published.effectiveConfigId()),
            previousPublishedConfigId,
            baseCreatedAt,
            baseCreatedBy,
            now,
            resolvedSubject,
            normalizedComment,
            null,
            appendAudit(
                activeAudit(existing),
                now,
                resolvedSubject,
                DefinitionConfigChange.DecisionAction.PROPOSE,
                normalizedComment,
                from,
                DefinitionConfigChange.ChangeState.PROPOSED,
                null));

    DefinitionConfigChange next =
        new DefinitionConfigChange(
            nextEnvelope(existing, requiredGroup, requiredconfigId, requiredLayerId, now, resolvedSubject),
            requiredGroup,
            requiredconfigId,
            requiredLayerId,
            schemaKey(published),
            nextEntry.state(),
            nextEntry.mode(),
            nextEntry.effectiveAtUtc(),
            nextEntry.expiresAtUtc(),
            nextEntry.draftDeclared(),
            nextEntry.baseConfigId(),
            nextEntry.baseEffectiveConfigId(),
            nextEntry.lastPublishedConfigId(),
            nextEntry.createdTimeUtc(),
            nextEntry.createdBySubject(),
            nextEntry.updatedTimeUtc(),
            nextEntry.updatedBySubject(),
            nextEntry.rationale(),
            nextEntry.decisionMessage(),
            nextEntry.audit(),
            prependEntry(existing, nextEntry));
    upsertChange(next);
    LOGGER.info("[ConfigWorkflow][Propose] group={} configId={} layerId={} mode={} effectiveAt={} expiresAt={} subject={}", requiredGroup, requiredconfigId, requiredLayerId, resolvedMode, window.effectiveAtUtc() == null ? "IMMEDIATE" : window.effectiveAtUtc(), window.expiresAtUtc() == null ? "NEVER" : window.expiresAtUtc(), resolvedSubject);
    return next;
  }

  @Override
  public DefinitionConfigChange approve(
      String group, String configId, String layerId, String comment, String subject) {
    DefinitionConfigChange approved =
        adminTransition(
            "APPROVE",
            group,
            configId,
            layerId,
            comment,
            subject,
            List.of(
                DefinitionConfigChange.ChangeState.PROPOSED,
                DefinitionConfigChange.ChangeState.DRAFT),
            DefinitionConfigChange.ChangeState.APPROVED,
            DefinitionConfigChange.DecisionAction.APPROVE);
    LOGGER.info("[ConfigWorkflow][Approve] group={} configId={} layerId={} effectiveAt={} expiresAt={} subject={}", approved.group(), approved.configId(), approved.layerId(), Strings.trimToNull(approved.effectiveAtUtc()) == null ? "IMMEDIATE" : approved.effectiveAtUtc(), Strings.trimToNull(approved.expiresAtUtc()) == null ? "NEVER" : approved.expiresAtUtc(), resolveSubject(subject));
    return approved;
  }

  @Override
  public DefinitionConfigChange reject(
      String group, String configId, String layerId, String comment, String subject) {
    return adminTransition(
        "REJECT",
        group,
        configId,
        layerId,
        comment,
        subject,
        List.of(
            DefinitionConfigChange.ChangeState.PROPOSED,
            DefinitionConfigChange.ChangeState.DRAFT,
            DefinitionConfigChange.ChangeState.APPROVED),
        DefinitionConfigChange.ChangeState.REJECTED,
        DefinitionConfigChange.DecisionAction.REJECT);
  }

  @Override
  public DefinitionConfigChange cancel(
      String group, String configId, String layerId, String comment, String subject) {
    String requiredGroup = required(group, "group");
    String requiredconfigId = required(configId, "configId");
    String requiredLayerId = required(layerId, "layerId");
    String resolvedSubject = resolveSubject(subject);

    DefinitionConfigChange existing =
        requiredChange(requiredGroup, requiredconfigId, requiredLayerId);
    DefinitionConfigChange.ChangeState existingState = stateOf(existing);
    boolean scheduledPending = isScheduledPending(existing);
    if (!List.of(
                DefinitionConfigChange.ChangeState.PROPOSED,
                DefinitionConfigChange.ChangeState.DRAFT,
                DefinitionConfigChange.ChangeState.APPROVED)
            .contains(existingState)
        && !scheduledPending) {
      throw new WorkflowStateException(
          "cancel requires PROPOSED, DRAFT, APPROVED, or a pending scheduled publish");
    }
    if (scheduledPending) {
      cancelScheduledEffective(requiredGroup, requiredconfigId, requiredLayerId, existing);
    }

    String now = Instant.now().toString();
    String normalizedComment = Strings.trimToNull(comment);
    DefinitionConfigChange.ChangeEntry currentEntry = activeEntry(existing);
    DefinitionConfigChange.ChangeEntry nextEntry =
        new DefinitionConfigChange.ChangeEntry(
            currentEntry == null ? UUID.randomUUID().toString() : currentEntry.changeId(),
            DefinitionConfigChange.ChangeState.NONE,
            scheduledPending
                ? DefinitionConfigChange.ChangeMode.SCHEDULED
                : DefinitionConfigChange.ChangeMode.IMMEDIATE,
            null,
            null,
            scheduledPending ? deepCopy(existing.draftDeclared()) : null,
            existing.baseConfigId(),
            existing.baseEffectiveConfigId(),
            existing.lastPublishedConfigId(),
            existing.createdTimeUtc(),
            existing.createdBySubject(),
            now,
            resolvedSubject,
            existing.rationale(),
            normalizedComment,
            appendAudit(
                activeAudit(existing),
                now,
                resolvedSubject,
                DefinitionConfigChange.DecisionAction.CANCEL,
                normalizedComment,
                existingState,
                DefinitionConfigChange.ChangeState.NONE,
                null));
    DefinitionConfigChange next =
        new DefinitionConfigChange(
            nextEnvelope(existing, requiredGroup, requiredconfigId, requiredLayerId, now, resolvedSubject),
            requiredGroup,
            requiredconfigId,
            requiredLayerId,
            existing.schemaKey(),
            nextEntry.state(),
            nextEntry.mode(),
            nextEntry.effectiveAtUtc(),
            nextEntry.expiresAtUtc(),
            nextEntry.draftDeclared(),
            nextEntry.baseConfigId(),
            nextEntry.baseEffectiveConfigId(),
            nextEntry.lastPublishedConfigId(),
            nextEntry.createdTimeUtc(),
            nextEntry.createdBySubject(),
            nextEntry.updatedTimeUtc(),
            nextEntry.updatedBySubject(),
            nextEntry.rationale(),
            nextEntry.decisionMessage(),
            nextEntry.audit(),
            replaceEntry(existing, nextEntry));
    upsertChange(next);
    return next;
  }

  @Override
  public DefinitionConfigLayer publish(
      String group,
      String configId,
      String layerId,
      DefinitionConfigChange.ChangeMode modeOverride,
      Instant effectiveAtOverride,
      Instant expiresAtOverride,
      String comment,
      String subject) {
    String requiredGroup = required(group, "group");
    String requiredconfigId = required(configId, "configId");
    String requiredLayerId = required(layerId, "layerId");
    String resolvedSubject = resolveSubject(subject);

    DefinitionConfigChange change = requiredChange(requiredGroup, requiredconfigId, requiredLayerId);
    DefinitionConfigChange.ChangeState changeState = stateOf(change);
    if (changeState != DefinitionConfigChange.ChangeState.APPROVED) {
      throw new WorkflowStateException("publish requires APPROVED state");
    }

    JsonNode draft = requireObject(change.draftDeclared(), "draftDeclared");
    DefinitionConfigLayer current = published(requiredGroup, requiredconfigId, requiredLayerId);
    validatePublish(requiredGroup, current, draft);

    DefinitionConfigChange.ChangeMode publishMode =
        modeOverride == null ? modeOf(change) : modeOverride;
    Instant publishedAt = Instant.now();
    Instant configuredEffectiveAt = parseInstantOrNull(change.effectiveAtUtc(), "effectiveAtUtc");
    Instant configuredExpiresAt = parseInstantOrNull(change.expiresAtUtc(), "expiresAtUtc");
    Instant publishEffectiveAt;
    Instant publishExpiresAt;
    if (publishMode == DefinitionConfigChange.ChangeMode.SCHEDULED) {
      publishEffectiveAt = effectiveAtOverride == null ? configuredEffectiveAt : effectiveAtOverride;
      if (publishEffectiveAt == null) {
        throw new IllegalArgumentException("effectiveAt is required for SCHEDULED publish");
      }
      publishExpiresAt = expiresAtOverride == null ? configuredExpiresAt : expiresAtOverride;
      validateWindow(DefinitionConfigChange.ChangeMode.SCHEDULED, publishEffectiveAt, publishExpiresAt, publishedAt);
    } else {
      if (effectiveAtOverride != null) {
        throw new IllegalArgumentException("effectiveAt must be null for IMMEDIATE publish");
      }
      publishEffectiveAt = null;
      publishExpiresAt = expiresAtOverride == null ? configuredExpiresAt : expiresAtOverride;
      validateWindow(DefinitionConfigChange.ChangeMode.IMMEDIATE, null, publishExpiresAt, publishedAt);
    }

    DefinitionConfigLayer updated =
        publishDefinition(
            current, draft.deepCopy(), publishMode, publishEffectiveAt, publishExpiresAt, publishedAt, resolvedSubject);
    store.upsert(updated);
    DefinitionConfigLayer persisted = published(requiredGroup, requiredconfigId, requiredLayerId);

    String now = publishedAt.toString();
    String publishEffectiveAtUtc = publishEffectiveAt == null ? null : publishEffectiveAt.toString();
    String publishExpiresAtUtc = publishExpiresAt == null ? null : publishExpiresAt.toString();
    String clearedEffectiveAtUtc =
        publishMode == DefinitionConfigChange.ChangeMode.SCHEDULED
            ? publishEffectiveAtUtc
            : publishExpiresAt == null ? null : now;
    String normalizedComment = Strings.trimToNull(comment);
    DefinitionConfigChange.ChangeEntry activeEntry = activeEntry(change);
    DefinitionConfigChange.ChangeEntry clearedEntry =
        new DefinitionConfigChange.ChangeEntry(
            activeEntry == null ? UUID.randomUUID().toString() : activeEntry.changeId(),
            DefinitionConfigChange.ChangeState.NONE,
            publishMode == DefinitionConfigChange.ChangeMode.SCHEDULED
                ? DefinitionConfigChange.ChangeMode.SCHEDULED
                : DefinitionConfigChange.ChangeMode.IMMEDIATE,
            clearedEffectiveAtUtc,
            publishExpiresAtUtc,
            publishMode == DefinitionConfigChange.ChangeMode.SCHEDULED ? draft.deepCopy() : null,
            change.baseConfigId(),
            change.baseEffectiveConfigId(),
            Strings.trimToNull(persisted.configId()),
            change.createdTimeUtc(),
            change.createdBySubject(),
            now,
            resolvedSubject,
            change.rationale(),
            normalizedComment,
            appendAudit(
                activeAudit(change),
                now,
                resolvedSubject,
                DefinitionConfigChange.DecisionAction.PUBLISH,
                normalizedComment,
                changeState,
                DefinitionConfigChange.ChangeState.NONE,
                Strings.trimToNull(persisted.configId())));
    DefinitionConfigChange cleared =
        new DefinitionConfigChange(
            nextEnvelope(change, requiredGroup, requiredconfigId, requiredLayerId, now, resolvedSubject),
            requiredGroup,
            requiredconfigId,
            requiredLayerId,
            change.schemaKey(),
            clearedEntry.state(),
            clearedEntry.mode(),
            clearedEntry.effectiveAtUtc(),
            clearedEntry.expiresAtUtc(),
            clearedEntry.draftDeclared(),
            clearedEntry.baseConfigId(),
            clearedEntry.baseEffectiveConfigId(),
            clearedEntry.lastPublishedConfigId(),
            clearedEntry.createdTimeUtc(),
            clearedEntry.createdBySubject(),
            clearedEntry.updatedTimeUtc(),
            clearedEntry.updatedBySubject(),
            clearedEntry.rationale(),
            clearedEntry.decisionMessage(),
            clearedEntry.audit(),
            replaceEntry(change, clearedEntry));
    upsertChange(cleared);
    LOGGER.info("[ConfigWorkflow][Publish] group={} configId={} layerId={} mode={} effectiveAt={} expiresAt={} subject={}", requiredGroup, requiredconfigId, requiredLayerId, publishMode, publishMode == DefinitionConfigChange.ChangeMode.IMMEDIATE ? "IMMEDIATE" : publishEffectiveAtUtc, publishExpiresAtUtc == null ? "NEVER" : publishExpiresAtUtc, resolvedSubject);
    return persisted;
  }

  private void cancelScheduledEffective(
      String group, String configId, String layerId, DefinitionConfigChange existing) {
    String effectiveAtUtc = Strings.trimToNull(existing == null ? null : existing.effectiveAtUtc());
    if (effectiveAtUtc == null) {
      throw new WorkflowStateException("pending scheduled publish effectiveAt is missing");
    }
    Instant effectiveAt;
    try {
      effectiveAt = Instant.parse(effectiveAtUtc);
    } catch (Exception ex) {
      throw new WorkflowStateException("pending scheduled publish effectiveAt is invalid");
    }
    if (!effectiveAt.isAfter(Instant.now())) {
      throw new WorkflowStateException("scheduled publish is already effective");
    }

    DefinitionConfigLayer current = published(group, configId, layerId);
    DefinitionConfigLayer payload = ConfigPayload(current);
    List<DefinitionConfigLayer.Effective> baseline = payload.effective();
    if (baseline == null || baseline.isEmpty()) {
      throw new WorkflowStateException("scheduled publish entry not found");
    }
    List<DefinitionConfigLayer.Effective> rewritten = new ArrayList<>();
    boolean removed = false;
    for (DefinitionConfigLayer.Effective entry : baseline) {
      if (entry == null) {
        continue;
      }
      if (effectiveAtUtc.equals(entry.effectiveTimeUtc())) {
        removed = true;
        continue;
      }
      rewritten.add(entry);
    }
    if (!removed) {
      throw new WorkflowStateException("scheduled publish entry not found");
    }
    DefinitionConfigLayer rewrittenPayload =
        new DefinitionConfigLayer(
            current.envelope(),
            payload.resourceId(),
            payload.parentId(),
            payload.shortName(),
            payload.configType(),
            payload.schemaKey(),
            payload.displayName(),
            deepCopy(payload.declared()),
            List.copyOf(rewritten),
            payload.history(),
            current.configId(),
            current.effectiveConfigId());
    DefinitionConfigLayer updated = rewrittenPayload;
    store.upsert(updated);
  }

  private DefinitionConfigChange adminTransition(
      String action,
      String group,
      String configId,
      String layerId,
      String comment,
      String subject,
      List<DefinitionConfigChange.ChangeState> fromStates,
      DefinitionConfigChange.ChangeState toState,
      DefinitionConfigChange.DecisionAction decisionAction) {
    String requiredGroup = required(group, "group");
    String requiredconfigId = required(configId, "configId");
    String requiredLayerId = required(layerId, "layerId");
    String resolvedSubject = resolveSubject(subject);

    DefinitionConfigChange existing =
        requiredChange(requiredGroup, requiredconfigId, requiredLayerId);
    DefinitionConfigChange.ChangeState fromState = stateOf(existing);
    if (!fromStates.contains(fromState)) {
      throw new WorkflowStateException(action + " not allowed from " + fromState);
    }

    String now = Instant.now().toString();
    String normalizedComment = Strings.trimToNull(comment);
    DefinitionConfigChange.ChangeEntry currentEntry = activeEntry(existing);
    DefinitionConfigChange.ChangeEntry nextEntry =
        new DefinitionConfigChange.ChangeEntry(
            currentEntry == null ? UUID.randomUUID().toString() : currentEntry.changeId(),
            toState,
            modeOf(existing),
            existing.effectiveAtUtc(),
            existing.expiresAtUtc(),
            existing.draftDeclared(),
            existing.baseConfigId(),
            existing.baseEffectiveConfigId(),
            existing.lastPublishedConfigId(),
            existing.createdTimeUtc(),
            existing.createdBySubject(),
            now,
            resolvedSubject,
            existing.rationale(),
            normalizedComment,
            appendAudit(
                activeAudit(existing),
                now,
                resolvedSubject,
                decisionAction,
                normalizedComment,
                fromState,
                toState,
                null));
    DefinitionConfigChange next =
        new DefinitionConfigChange(
            nextEnvelope(existing, requiredGroup, requiredconfigId, requiredLayerId, now, resolvedSubject),
            requiredGroup,
            requiredconfigId,
            requiredLayerId,
            existing.schemaKey(),
            nextEntry.state(),
            nextEntry.mode(),
            nextEntry.effectiveAtUtc(),
            nextEntry.expiresAtUtc(),
            nextEntry.draftDeclared(),
            nextEntry.baseConfigId(),
            nextEntry.baseEffectiveConfigId(),
            nextEntry.lastPublishedConfigId(),
            nextEntry.createdTimeUtc(),
            nextEntry.createdBySubject(),
            nextEntry.updatedTimeUtc(),
            nextEntry.updatedBySubject(),
            nextEntry.rationale(),
            nextEntry.decisionMessage(),
            nextEntry.audit(),
            replaceEntry(existing, nextEntry));
    upsertChange(next);
    return next;
  }

  private DefinitionConfigLayer publishDefinition(
      DefinitionConfigLayer current,
      JsonNode draft,
      DefinitionConfigChange.ChangeMode mode,
      Instant effectiveAt,
      Instant expiresAt,
      Instant publishedAt,
      String subject) {
    DefinitionConfigLayer payload = ConfigPayload(current);
    Instant snapshotEffectiveAt = mode == DefinitionConfigChange.ChangeMode.IMMEDIATE ? publishedAt : effectiveAt;
    DefinitionConfigLayer.History currentSnapshot =
        ConfigHistoryUtil.snapshotFromCurrent(current, mode.name(), snapshotEffectiveAt, subject);
    List<DefinitionConfigLayer.History> rewrittenHistory =
        ConfigHistoryUtil.appendSnapshot(payload.history(), currentSnapshot, HISTORY_MAX);

    DefinitionConfigLayer rewrittenPayload;
    if (mode == DefinitionConfigChange.ChangeMode.SCHEDULED) {
      List<DefinitionConfigLayer.Effective> effective =
          upsertScheduledEffective(payload.effective(), effectiveAt, expiresAt, draft);
      rewrittenPayload =
          new DefinitionConfigLayer(
              current.envelope(),
              payload.resourceId(),
              payload.parentId(),
              payload.shortName(),
              payload.configType(),
              payload.schemaKey(),
              payload.displayName(),
              deepCopy(payload.declared()),
              effective,
              rewrittenHistory,
              current.configId(),
              current.effectiveConfigId());
    } else {
      if (expiresAt != null) {
        List<DefinitionConfigLayer.Effective> effective =
            upsertScheduledEffective(payload.effective(), publishedAt, expiresAt, draft);
        rewrittenPayload =
            new DefinitionConfigLayer(
                current.envelope(),
                payload.resourceId(),
                payload.parentId(),
                payload.shortName(),
                payload.configType(),
                payload.schemaKey(),
                payload.displayName(),
                deepCopy(payload.declared()),
                effective,
                rewrittenHistory,
                current.configId(),
                current.effectiveConfigId());
      } else {
        rewrittenPayload =
            new DefinitionConfigLayer(
                current.envelope(),
                payload.resourceId(),
                payload.parentId(),
                payload.shortName(),
                payload.configType(),
                payload.schemaKey(),
                payload.displayName(),
                draft,
                ConfigHistoryUtil.copyEffective(payload.effective()),
                rewrittenHistory,
                current.configId(),
                current.effectiveConfigId());
      }
    }
    return rewrittenPayload;
  }

  private static List<DefinitionConfigLayer.Effective> upsertScheduledEffective(
      List<DefinitionConfigLayer.Effective> baseline,
      Instant effectiveAt,
      Instant expiresAt,
      JsonNode declaredPayload) {
    if (effectiveAt == null) {
      throw new IllegalArgumentException("effectiveAt is required for SCHEDULED publish");
    }
    String effectiveUtc = effectiveAt.toString();
    String expiresUtc = expiresAt == null ? null : expiresAt.toString();
    List<DefinitionConfigLayer.Effective> rewritten = new ArrayList<>();
    if (baseline != null) {
      for (DefinitionConfigLayer.Effective entry : baseline) {
        if (entry == null) {
          continue;
        }
        if (!effectiveUtc.equals(entry.effectiveTimeUtc())) {
          rewritten.add(entry);
        }
      }
    }
    rewritten.add(new DefinitionConfigLayer.Effective(effectiveUtc, expiresUtc, declaredPayload.deepCopy()));
    rewritten.sort(Comparator.comparing(entry -> Instant.parse(entry.effectiveTimeUtc())));
    return List.copyOf(rewritten);
  }

  private void validateDraft(String group, DefinitionConfigLayer published, JsonNode draftDeclared) {
    if (draftDeclared == null || !draftDeclared.isObject()) {
      throw new IllegalArgumentException("declaredPayload must be a JSON object");
    }
    resolveSchema(group, schemaKey(published));
  }

  private void validatePublish(String group, DefinitionConfigLayer published, JsonNode draftDeclared) {
    DefinitionSchema schema = resolveSchema(group, schemaKey(published));
    ObjectNode draft = requireObject(draftDeclared, "draftDeclared");
    for (String key : ConfigSchemaRequiredKeys.explicitRequiredKeys(schema)) {
      if (!draft.has(key)) {
        draft.set(key, DiscoveryCodec.mapper().nullNode());
      }
    }
  }

  private DefinitionSchema resolveSchema(String group, String schemaKey) {
    Definition definition = store.get(DefinitionType.SCHEMA, group, schemaKey);
    if (!(definition instanceof DefinitionSchema schema)) {
      throw new IllegalStateException("schema not found for group=" + group + " key=" + schemaKey);
    }
    if (schema.schemaType() != DefinitionSchema.SchemaType.CONFIG) {
      throw new IllegalStateException("schemaType must be CONFIG");
    }
    if (!SCHEMA_FORMAT.equals(schema.format())) {
      throw new IllegalStateException(
          "schema format must be " + SCHEMA_FORMAT + " but was " + schema.format());
    }
    return schema;
  }

  private DefinitionConfigLayer published(String group, String configId, String layerId) {
    Definition definition = store.get(DefinitionType.CONFIG_LAYER, group, configLayerDefinitionName(configId, layerId));
    if (definition instanceof DefinitionConfigLayer config) {
      return config;
    }
    throw new ConfigNotFoundException("config layer not found: " + configId + "/" + layerId);
  }

  private DefinitionConfigChange requiredChange(String group, String configId, String layerId) {
    String requiredGroup = required(group, "group");
    String requiredconfigId = required(configId, "configId");
    String requiredLayerId = required(layerId, "layerId");
    DefinitionConfigChange change = loadChange(requiredGroup, requiredconfigId, requiredLayerId);
    if (change == null) {
      throw new WorkflowDraftNotFoundException("workflow change not found");
    }
    return change;
  }

  private DefinitionConfigChange loadChange(String group, String configId, String layerId) {
    Definition definition = store.get(DefinitionType.CONFIG_CHANGE, group, changeName(configId, layerId));
    if (!(definition instanceof DefinitionConfigChange change)) {
      return null;
    }
    return normalizeChange(change);
  }

  private void upsertChange(DefinitionConfigChange change) {
    DefinitionConfigChange normalized = normalizeChange(change);
    if (normalized == null || normalized.envelope() == null) {
      throw new IllegalArgumentException("change envelope is required");
    }
    String expectedName = changeName(normalized.configId(), normalized.layerId());
    if (!expectedName.equals(normalized.envelope().name())) {
      throw new IllegalArgumentException("CONFIG_CHANGE name must be " + expectedName);
    }
    if (!Objects.equals(normalized.group(), normalized.envelope().group())) {
      throw new IllegalArgumentException("CONFIG_CHANGE group mismatch");
    }
    store.upsert(normalized);
  }

  private DefinitionConfigChange emptyChange(
      String group, String configId, String layerId, DefinitionConfigLayer published) {
    String now = Instant.now().toString();
    return new DefinitionConfigChange(
        new DefinitionEnvelope(
            group,
            DefinitionType.CONFIG_CHANGE,
            changeName(configId, layerId),
            null,
            true,
            "ACTIVE",
            now,
            resolveSubject(null),
            now,
            resolveSubject(null)),
        group,
        configId,
        layerId,
        schemaKey(published),
        DefinitionConfigChange.ChangeState.NONE,
        DefinitionConfigChange.ChangeMode.IMMEDIATE,
        null,
        null,
        null,
        Strings.trimToNull(published.configId()),
        Strings.trimToNull(published.effectiveConfigId()),
        null,
        null,
        null,
        null,
        null,
        null,
        null,
        List.of());
  }

  private DefinitionConfigChange clampAudit(DefinitionConfigChange change, int limitRaw) {
    int limit = limitRaw <= 0 ? 20 : Math.min(limitRaw, AUDIT_MAX);
    List<DefinitionConfigChange.AuditEntry> audit =
        change.audit() == null ? List.of() : change.audit();
    if (audit.size() <= limit) {
      return change;
    }
    DefinitionConfigChange.ChangeEntry activeEntry = activeEntry(change);
    List<DefinitionConfigChange.ChangeEntry> changes = change.changes();
    if (activeEntry != null && changes != null && !changes.isEmpty()) {
      DefinitionConfigChange.ChangeEntry clampedEntry =

        change.group(),

        change.state(),
        change.mode(),

  }

  private static DefinitionConfigChange normalizeChange(DefinitionConfigChange change) {
    if (change == null) {
      return null;
    }
    List<DefinitionConfigChange.ChangeEntry> normalizedEntries = normalizedEntries(change.changes());
    if (normalizedEntries.isEmpty()) {
      if (!hasLegacyEntryPayload(change)) {
        return new DefinitionConfigChange(
            change.envelope(),
            change.group(),
            change.configId(),
            change.layerId(),
            change.schemaKey(),
            change.state(),
            change.mode(),
            change.effectiveAtUtc(),
            change.expiresAtUtc(),

            change.lastPublishedConfigId(),
            change.createdTimeUtc(),
            change.createdBySubject(),
            change.updatedTimeUtc(),
            change.updatedBySubject(),
            change.rationale(),
            change.decisionMessage(),
            change.audit(),
            List.of());
      }
      DefinitionConfigChange.ChangeEntry legacy = legacyEntry(change);
      return projectChange(change, legacy, List.of(legacy));
    }
    DefinitionConfigChange.ChangeEntry latest = firstEntry(normalizedEntries);
    return projectChange(change, latest, normalizedEntries);
  }

  private static DefinitionConfigChange.ChangeEntry activeEntry(DefinitionConfigChange change) {
    if (change == null) {

      return latest;

    return hasLegacyEntryPayload(change) ? legacyEntry(change) : null;
  }

  private static List<DefinitionConfigChange.ChangeEntry> prependEntry(
      DefinitionConfigChange existing, DefinitionConfigChange.ChangeEntry nextEntry) {
    List<DefinitionConfigChange.ChangeEntry> entries = new ArrayList<>();
    if (nextEntry != null) {
      entries.add(nextEntry);
    }
    for (DefinitionConfigChange.ChangeEntry entry : normalizedEntries(existing == null ? null : existing.changes())) {
      if (entry != null) {
        entries.add(entry);
      }
    }
    return entries.isEmpty() ? List.of() : List.copyOf(entries);
  }

  private static List<DefinitionConfigChange.ChangeEntry> replaceEntry(
      DefinitionConfigChange existing, DefinitionConfigChange.ChangeEntry replacement) {
    if (replacement == null) {
      return normalizedEntries(existing == null ? null : existing.changes());
    }
    List<DefinitionConfigChange.ChangeEntry> entries =
        new ArrayList<>(normalizedEntries(existing == null ? null : existing.changes()));
    String replacementId = Strings.trimToNull(replacement.changeId());
    int replaceIndex = -1;
    if (replacementId != null) {
      for (int i = 0; i < entries.size(); i++) {
        DefinitionConfigChange.ChangeEntry current = entries.get(i);
        if (current != null && Objects.equals(replacementId, Strings.trimToNull(current.changeId()))) {
          replaceIndex = i;
          break;
        }
      }
    }
    if (replaceIndex < 0 && !entries.isEmpty()) {
      replaceIndex = 0;
    }
    if (replaceIndex < 0) {
      entries.add(replacement);
    } else {
      entries.set(replaceIndex, replacement);
    }
    return List.copyOf(entries);
  }

  private static boolean hasLegacyEntryPayload(DefinitionConfigChange change) {
    if (change == null) {

    if (change.state() != null && change.state() != DefinitionConfigChange.ChangeState.NONE) {
      return true;
    }
    if (change.draftDeclared() != null) {
      return true;

    if (change.audit() != null && !change.audit().isEmpty()) {
      return true;
    }
    if (Strings.trimToNull(change.createdTimeUtc()) != null || Strings.trimToNull(change.createdBySubject()) != null) {
      return true;
    }
    if (Strings.trimToNull(change.updatedTimeUtc()) != null || Strings.trimToNull(change.updatedBySubject()) != null) {

    if (Strings.trimToNull(change.rationale()) != null || Strings.trimToNull(change.decisionMessage()) != null) {
      return true;
    }
    return Strings.trimToNull(change.lastPublishedConfigId()) != null;
  }

  private static Window validateWindow(
      DefinitionConfigChange.ChangeMode mode, Instant effectiveAt, Instant expiresAt, Instant now) {
    DefinitionConfigChange.ChangeMode resolvedMode =
        mode == null ? DefinitionConfigChange.ChangeMode.IMMEDIATE : mode;
    Instant referenceNow = now == null ? Instant.now() : now;
    if (resolvedMode == DefinitionConfigChange.ChangeMode.SCHEDULED) {
      if (effectiveAt == null) {
        throw new IllegalArgumentException("effectiveAt is required for SCHEDULED mode");
      }
      if (expiresAt != null && !expiresAt.isAfter(effectiveAt)) {
        throw new IllegalArgumentException("expiresAt must be after effectiveAt for SCHEDULED mode");
      }
      return new Window(effectiveAt.toString(), expiresAt == null ? null : expiresAt.toString());
    }
    if (effectiveAt != null) {
      throw new IllegalArgumentException("effectiveAt must be null for IMMEDIATE mode");
    }
    if (expiresAt != null && !expiresAt.isAfter(referenceNow)) {
      throw new IllegalArgumentException("expiresAt must be after now for IMMEDIATE mode");
    }
    return new Window(null, expiresAt == null ? null : expiresAt.toString());
  }

  private static Instant parseInstantOrNull(String value, String field) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      return null;
    }
    try {
      return Instant.parse(normalized);
    } catch (Exception ex) {
      throw new IllegalArgumentException(field + " must be an ISO-8601 UTC instant", ex);
    }
  }

  private static boolean isScheduledPending(DefinitionConfigChange change) {
    if (change == null || modeOf(change) != DefinitionConfigChange.ChangeMode.SCHEDULED) {
      return false;
    }
    String effectiveAtUtc = Strings.trimToNull(change.effectiveAtUtc());
    if (effectiveAtUtc == null) {
      return false;
    }
    try {
      Instant effectiveAt = Instant.parse(effectiveAtUtc);
      Instant now = Instant.now();
      Instant expiresAt = parseInstantOrNull(change.expiresAtUtc(), "expiresAtUtc");
      if (expiresAt != null && !expiresAt.isAfter(effectiveAt)) {
        return false;
      }
      if (expiresAt != null && !now.isBefore(expiresAt)) {
        return false;
      }
      return effectiveAt.isAfter(now);
    } catch (Exception ignored) {
      return false;
    }
  }
