  public SessionImpl(SessionConfig config) {
    this.cfg = Objects.requireNonNull(config, "config");
    this.group = Strings.trimToNull(config.group());
    this.appName = Strings.trimToNull(config.appName());
    this.instance = SessionBootstrap.resolveInstanceName(config.instance());
    this.normalizedDiscoveryUrls = SessionBootstrap.normalizeDiscoveryUrls(config.discoveryUrls());
    this.shutdownHook = new Thread(this::closeQuietlyFromShutdownHook, "fabric-session-" + sessionId);
    this.serviceEntitlementsLogDecisions = config.entitlementsDecisionLogging();
    this.grpcRoutingLogSelections = config.grpcRoutingLogSelections();
    this.statusUserTag = config.statusUserTag();
    this.connectionManager =
        new ConnectionManagerImpl(
            this::discoveryCache, () -> group, () -> sessionContext, cfg, this::decryptS3Credential);
    this.securityManager =
        new SecurityManagerImpl(this::discoveryCache, () -> group, connectionManager::evictS3Connection);
    this.accessManager =
        new AccessManagerImpl(
            this, () -> entitlementsManager, this::discoveryCache, this::serviceStatusCache);
    this.recoveryManager =
        new RecoveryManagerImpl(
            this::dumpConfiguredRecoverySnapshot, this::currentDiscoveryRecoveryStatus);
    refreshSessionViews();
  }

  @Override
  public void start() throws Exception {
    synchronized (lifecycleLock) {
      if (closed.get()) {
        throw new IllegalStateException("Session is closed");
      }
      if (started.get()) {
        return;
      }

      long startupStartNanos = System.nanoTime();
      LOGGER.info(
          "[Session][StartupTiming] phase=start group={} app={} instance={} recoveryEnabled={}",
          cfg.group(),
          cfg.appName(),
          cfg.instance(),
          cfg.discoveryCacheRecoveryEnabled());

      long phaseStartNanos = System.nanoTime();
      SessionBootstrap.BootstrapResult result = bootstrapWithRecoveryFallback();
      LOGGER.info(
          "[Session][StartupTiming] phase=bootstrap elapsedMs={} mode={}",
          elapsedMillis(phaseStartNanos),
          discoveryRecoveryStatus.recoveryUsed() ? "recovered" : "live");

      phaseStartNanos = System.nanoTime();
      this.kerberosPropertyScope = result.kerberosPropertyScope();
      this.group = result.group();
      this.appName = result.appName();
      this.instance = result.instance();
      this.normalizedDiscoveryUrls = result.discoveryUrls();
      this.env = result.environment();
      this.region = result.region();
      this.site = result.site();
      this.configCacheAdmin = result.configCacheAdmin();
      this.discoveryUris = result.discoveryUris();
      this.controlPlaneConnection = result.controlPlaneConnection();
      this.runtimePlacementHost = resolveHostname();
      this.environmentDefinition = result.environmentDefinition();
      this.appDefinition = result.appDefinition();
      this.otelManager = createOtelManager(result.environmentDefinition(), result.appDefinition());
      LOGGER.info(
          "[Session][StartupTiming] phase=bootstrap-assign elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      Path configuredCheckpointPath = cfg.checkpointPath();
      if (configuredCheckpointPath != null) {
        Path resolvedCheckpointPath = configuredCheckpointPath.toAbsolutePath().normalize();
        this.checkpointPath = resolvedCheckpointPath;
        this.checkpointStore = CheckpointStores.local(resolvedCheckpointPath);
        LOGGER.info("[Checkpoint] usingConfiguredPath path={}", resolvedCheckpointPath);
      } else {
        this.checkpointPath = null;
        this.checkpointStore =
            CheckpointStores.noOp(
                "checkpointPath not configured for session "
                    + Strings.firstNonBlank(group, "<group>")
                    + "/"
                    + Strings.firstNonBlank(appName, "<app>"));
        LOGGER.info("[Checkpoint] disabled (no checkpointPath configured)");
      }
      LOGGER.info(
          "[Session][StartupTiming] phase=checkpoint elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      SessionImplManagers.StartedManagers startedManagers =
          SessionImplManagers.initialize(
              this,
              cfg,
              result,
              sessionId,
              checkpointStore,
              connectionManager,
              serviceEntitlementsLogDecisions,
              LOGGER);
      DiscoveryCache cache = startedManagers.discoveryCache();
      LOGGER.info(
          "[Session][StartupTiming] phase=managers.initialize elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      this.discoveryManager = startedManagers.discoveryManager();
      this.deleteManager = startedManagers.deleteManager();
      this.entitlementsManager = startedManagers.entitlementsManager();
      this.serviceEntitlements = startedManagers.serviceEntitlements();
      this.dataEntitlements = startedManagers.dataEntitlements();
      this.serviceManager = startedManagers.serviceManager();
      this.schemaRegistryManager = startedManagers.schemaRegistryManager();
      this.streamManager = startedManagers.streamManager();
      this.dataPlanes = startedManagers.dataPlanes();
      this.assetCatalogService = startedManagers.assetCatalogService();
      this.configsCore = startedManagers.configsCore();
      LOGGER.info(
          "[Session][StartupTiming] phase=managers.publish elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      registerShutdownHook();
      this.started.set(true);
      refreshSessionViews();
      LOGGER.info(
          "[Session][StartupTiming] phase=session-mark-started elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      SessionImplLifecycle.runDuplicateInstanceCheck(
          LOGGER,
          controlPlaneConnection,
          appDefinition,
          cfg,
          group,
          appName,
          instance,
          DEFAULT_DUPLICATE_HEARTBEAT_GRACE,
          DEFAULT_DUPLICATE_INSTANCE_CHECK_ATTEMPTS,
          DEFAULT_DUPLICATE_INSTANCE_CHECK_DELAY_MILLIS);
      LOGGER.info(
          "[Session][StartupTiming] phase=duplicate-instance-check elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      startStatusPublisher();
      LOGGER.info(
          "[Session][StartupTiming] phase=status-publisher-start elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      refreshSessionViews();
      LOGGER.info(
          "[Session][StartupTiming] phase=session-refresh-final elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      if (discoveryRecoveryStatus.recoveryUsed()) {
        LOGGER.info(
            "[Session][DiscoveryRecovery] startup=recovered sessionId={} app={}/{} instance={} snapshotPath={} definitions={}",
            sessionId,
            group,
            appName,
            instance,
            discoveryRecoveryStatus.snapshotPath(),
            discoveryRecoveryStatus.definitionCount());
      } else {
        LOGGER.info(
            "[Session][DiscoveryRecovery] startup=live sessionId={} app={}/{} instance={} endpoints={} definitions={}",
            sessionId,
            group,
            appName,
            instance,
            discoveryUris,
            cache == null ? 0 : cache.size());
      }
      LOGGER.info(
          "[Session][StartupTiming] phase=complete sessionId={} totalElapsedMs={} mode={}",
          sessionId,
          elapsedMillis(startupStartNanos),
          discoveryRecoveryStatus.recoveryUsed() ? "recovered" : "live");
      LOGGER.debug("L0 bootstrap complete sessionId={} app={}/{} instance={} env={} region={} site={} endpoints={} configCount={}", sessionId, group, appName, instance, env, region, site, discoveryUris, cache == null ? 0 : cache.size());
    }
  }

  @Override
  public void stop() {
    synchronized (lifecycleLock) {
      if (!started.get()
          && controlPlaneConnection == null
          && discoveryManager == null
          && statusScheduler == null
          && configsCore == null
          && dataPlanes == null) {
        return;
      }
      SessionImplShutdown.stopManagedServices(serviceManager, LOGGER);
      serviceManager = null;
      CloseSupport.closeQuietly(dataPlanes instanceof AutoCloseable closeable ? closeable : null);
      dataPlanes = null;
      CloseSupport.closeQuietly(streamManager);
      streamManager = null;
      assetCatalogService = null;
      CloseSupport.closeQuietly(configsCore);
      configsCore = null;
      CloseSupport.closeQuietly(connectionManager);
      schemaRegistryManager = null;
      CloseSupport.closeQuietly(statusPublisher);
      statusPublisher = null;
      CloseSupport.closeQuietly(entitlementsManager);
      entitlementsManager = null;
      CloseSupport.closeQuietly(accessManager);
      serviceEntitlements = null;
      dataEntitlements = null;
      CloseSupport.closeQuietly(deleteManager);
      deleteManager = null;
      CloseSupport.closeQuietly(otelManager);
      otelManager = null;
      CloseSupport.closeQuietly(discoveryManager);
      discoveryManager = null;
      SessionImplShutdown.closeControlConnection(controlPlaneConnection);
      controlPlaneConnection = null;
      runtimePlacementHost = null;
      environmentDefinition = null;
      appDefinition = null;
      configCacheAdmin = false;
      discoveryUris = List.of();
      discoveryRecoveryStatus = DiscoveryCacheRecoveryResult.notStarted();
      started.set(false);
      closeStatusScheduler();
      refreshSessionViews();
    }
  }

  @Override
  public void close() {
    synchronized (lifecycleLock) {
      if (!closed.compareAndSet(false, true)) {
        return;
      }
      unregisterShutdownHook();
      stop();
      CloseSupport.closeQuietly(kerberosPropertyScope);
      kerberosPropertyScope = null;
    }
  }

  public ScheduledExecutorService statusScheduler() {
    synchronized (lifecycleLock) {
      if (statusScheduler == null || statusScheduler.isShutdown()) {
        statusScheduler =
            Executors.newScheduledThreadPool(
                STATUS_SCHEDULER_THREADS,
                runnable -> {
                  Thread thread =
                      new Thread(
                          runnable,
                          "fabric-status-" + STATUS_SCHEDULER_THREAD_ID.incrementAndGet());
                  thread.setDaemon(true);
                  return thread;
                });
      }
      return statusScheduler;
    }
  }

  String decryptS3Credential(String alias, String encryptedValue) {
    SessionSecrets secrets = cfg.sessionSecrets();
    if (secrets == null) {
      throw new IllegalStateException(
          "S3 credentials are encrypted but no KEK was provided (alias="
              + Strings.firstNonBlank(alias, SessionSecrets.DEFAULT_KEK_ALIAS)
              + ")");
    }
    return secrets.decrypt(alias, encryptedValue);
  }

  private void dumpConfiguredRecoverySnapshot() {
    Path configuredPath = cfg.discoveryCacheRecoveryPath();
    if (configuredPath == null) {
      throw new IllegalStateException(
          "Discovery cache recovery path is not configured; set withRecoveryPath(...)");
    }
    dumpRecoverySnapshot(configuredPath);
  }

  private void dumpRecoverySnapshot(Path path) {
    DiscoveryCache cache = discoveryCache();
    if (cache == null) {
      throw new IllegalStateException("DiscoveryCache is not initialized");
    }
    DiscoveryCacheSnapshotManifest manifest =
        discoverySnapshotService.dump(
            cache,
            Objects.requireNonNull(path, "path"),
            new DiscoveryCacheSnapshotService.SnapshotMetadata(sessionId, group, appName, instance));
    LOGGER.info(
        "[Session][DiscoveryRecovery][Dump] path={} createdTimeUtc={} definitions={}",
        path.toAbsolutePath().normalize(),
        manifest.createdTimeUtc(),
        manifest.definitionCount());
  }

  private DiscoveryCacheRecoveryResult currentDiscoveryRecoveryStatus() {
    return discoveryRecoveryStatus;

  private SessionBootstrap.BootstrapResult bootstrapWithRecoveryFallback() throws Exception {
    LOGGER.info(
        "[Session][DiscoveryRecovery] config enabled={} path={} ttl={}",
        cfg.discoveryCacheRecoveryEnabled(),
        cfg.discoveryCacheRecoveryPath(),
        cfg.discoveryCacheRecoveryTtl());
    LOGGER.info(
        "[Session][DiscoveryRecovery] live-bootstrap-start app={}/{} instance={} discoveryUrls={}",
        cfg.group(),
        cfg.appName(),
        cfg.instance(),
        cfg.discoveryUrls());
    try {
      SessionBootstrap.BootstrapResult liveResult =
          SessionBootstrap.bootstrap(cfg, sessionId, kerberosPropertyScope);
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.live(
              cfg.discoveryCacheRecoveryEnabled(), "live DiscoveryCache bootstrap succeeded");
      LOGGER.info(
          "[Session][DiscoveryRecovery] live-bootstrap-succeeded app={}/{} instance={}",
          cfg.group(),
          cfg.appName(),
          cfg.instance());
      return liveResult;
    } catch (Exception liveFailure) {
      LOGGER.warn(
          "[Session][DiscoveryRecovery] live-bootstrap-failed app={}/{} instance={} message={}",
          cfg.group(),
          cfg.appName(),
          cfg.instance(),
          Strings.firstNonBlank(liveFailure.getMessage(), liveFailure.getClass().getSimpleName()));
      return recoverBootstrapOrThrow(liveFailure);
    }
  }

  private SessionBootstrap.BootstrapResult recoverBootstrapOrThrow(Exception liveFailure)
      throws Exception {
    if (!cfg.discoveryCacheRecoveryEnabled()) {
      LOGGER.info("[Session][DiscoveryRecovery] recovery-disabled; startup will fail if live bootstrap fails");
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.failed(
              false,
              null,
              null,
              null,
              null,
              "Recovery disabled and live bootstrap failed");
      throw liveFailure;
    }

    Path snapshotPath = cfg.discoveryCacheRecoveryPath();
    Duration snapshotTtl = cfg.discoveryCacheRecoveryTtl();
    if (snapshotPath == null) {
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.failed(
              true,
              null,
              null,
              null,
              snapshotTtl,
              "Recovery enabled but snapshot path is not configured");
      IllegalStateException error =
          new IllegalStateException(
              "DiscoveryCache recovery is enabled but no snapshot path is configured");
      error.addSuppressed(liveFailure);
      throw error;
    }

    LOGGER.info(
        "[Session][DiscoveryRecovery] recovery-attempt-enabled=true path={} ttl={}",
        snapshotPath.toAbsolutePath().normalize(),
        snapshotTtl);
    DiscoveryCache probeCache = null;
    DiscoveryCache runtimeCache = null;
    boolean published = false;
    try {
      DiscoveryCacheSnapshotService.LoadedSnapshot loaded =
          discoverySnapshotService.load(snapshotPath, snapshotTtl);
      String sessionGroup = Strings.trimToNull(cfg.group());
      probeCache =
          DiscoveryCache.recoveredSnapshotCache(
              sessionGroup, true, cfg.verbose(), loaded.definitions());
      SessionBootstrap.BootstrapResult recoveredResult =
          SessionBootstrap.bootstrapFromRecoveredSnapshot(
              cfg, sessionId, kerberosPropertyScope, probeCache);
      runtimeCache = probeCache;
      if (!recoveredResult.configCacheAdmin()) {
        runtimeCache =
            DiscoveryCache.recoveredSnapshotCache(
                recoveredResult.group(), false, cfg.verbose(), loaded.definitions());
        probeCache.close();
        probeCache = null;
      }
      publishDiscoveryManager(
          DiscoveryManager.recoveredSnapshot(
              runtimeCache, recoveredResult.group(), this::defaultCatalogSubject, () -> null));
      published = true;
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.recovered(
              loaded.snapshotPath(),
              loaded.createdTimeUtc(),
              loaded.age(),
              loaded.ttl(),
              loaded.manifest().definitionCount(),
              "Recovered DiscoveryCache from disk snapshot");
      LOGGER.info(
          "[Session][DiscoveryRecovery] recovery-accepted path={} createdTimeUtc={} age={} ttl={} definitions={}",
          loaded.snapshotPath(),
          loaded.createdTimeUtc(),
          loaded.age(),
          loaded.ttl(),
          loaded.manifest().definitionCount());
      return recoveredResult;
    } catch (Exception recoveryFailure) {
      CloseSupport.closeQuietly(probeCache);
      if (!published) {
        CloseSupport.closeQuietly(runtimeCache);
      }
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.failed(
              true,
              snapshotPath,
              null,
              null,
              snapshotTtl,
              Strings.firstNonBlank(
                  recoveryFailure.getMessage(), recoveryFailure.getClass().getSimpleName()));
      LOGGER.error(
          "[Session][DiscoveryRecovery] recovery-rejected path={} ttl={} reason={}",
          snapshotPath.toAbsolutePath().normalize(),
          snapshotTtl,
          Strings.firstNonBlank(
              recoveryFailure.getMessage(), recoveryFailure.getClass().getSimpleName()));
      IllegalStateException startupFailure =
          new IllegalStateException(
              "Live bootstrap failed and DiscoveryCache recovery failed: "
                  + Strings.firstNonBlank(
                      recoveryFailure.getMessage(),
                      recoveryFailure.getClass().getSimpleName()),
              recoveryFailure);
      startupFailure.addSuppressed(liveFailure);
      throw startupFailure;
    }
  }

  OtelManager createOtelManager(
      DefinitionEnvironment environment, DefinitionApp app) {
    if (!FabricOtelEnablement.isEnvironmentOtelEnabled(environment)
        || !FabricOtelEnablement.isAppOtelEnabled(app)) {
      return OtelManagerNoop.instance();
    }
    String defaultServiceName =
        Strings.firstNonBlank(
            app == null || app.envelope() == null ? null : app.envelope().name(),
            appName,
            "fabric-lite");
    return OtelManagerImpl.create(environment, app, defaultServiceName);
  }

  private void refreshSessionViews() {
    DiscoveryCache cache = discoveryCache();
    SessionContext context =
        new SessionContext(
            cfg.type(),
            group,
            appName,
            instance,
            env,
            region,
            site,
            sessionId,
            normalizedDiscoveryUrls,
            discoveryUris,
            configCacheAdmin,
            cache != null && cache.recovered(),
            cache == null ? 0 : cache.size(),
            statusPublisher == null ? 0 : statusPublisher.publishIntervalSeconds(),
            statusPublisher == null ? null : statusPublisher.intervalSource(),
            statusPublisher != null && statusPublisher.started(),
            statusPublisher == null ? null : statusPublisher.lastPublishedPayload(),
            statusPublisher == null ? 0L : statusPublisher.lastPublishedEpochMs(),
            statusPublisher == null ? 0L : statusPublisher.publishCount());
    this.sessionContext = context;
  }

  private void startStatusPublisher() {
    DiscoveryCache cache = discoveryCache();
    if (controlPlaneConnection == null || cache == null || !cache.recovered()) {
      return;
    }
    AppStatusPublisher publisher =
        new AppStatusPublisher(
            controlPlaneConnection,
            sessionContext,
            appDefinition,
            statusUserTag,
            otelManager.enabled(),
            statusScheduler(),
            (publishEpochMs, status, payload) -> {
              synchronized (lifecycleLock) {
                if (!started.get() || statusPublisher == null) {
                  return;
                }
                refreshSessionViews();
              }
            });
    statusPublisher = publisher;
    publisher.start();
  }

  String defaultCatalogSubject() {
    String resolvedGroup = Strings.trimToNull(group);
    String resolvedApp = Strings.trimToNull(appName);
    if (resolvedGroup == null || resolvedApp == null) {
      return "app:catalog/system";
    }
    return "app:" + resolvedGroup + "/" + resolvedApp;
  }

  private static long elapsedMillis(long startNanos) {
    return Math.max(0L, (System.nanoTime() - startNanos) / 1_000_000L);
  }

  private static long resolveCadenceSeconds(Integer configuredSeconds, long defaultSeconds) {
    if (configuredSeconds != null && configuredSeconds > 0) {
      return configuredSeconds.longValue();
    }
    return defaultSeconds;
  }
