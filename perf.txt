  public ConfigCodegen generate(
      ConfigSchemaDoc schemaDoc,
      String group,
      String schemaName,
      Language language,
      String basePackageOrModule) {
    Objects.requireNonNull(schemaDoc, "schemaDoc");
    String requiredGroup = requireValue(group, "group");
    String requiredSchemaName = requireValue(schemaName, "schemaName");
    Language requiredLanguage = Objects.requireNonNull(language, "language");
    String requiredBasePackageOrModule =
        resolveBasePackageOrModule(basePackageOrModule, requiredSchemaName);
    validateModuleName(requiredBasePackageOrModule);

    ConfigSchemaDoc normalized = normalizeSchemaDoc(schemaDoc);
    String className = resolveClassName(normalized.displayName(), requiredSchemaName);
    List<FieldModel> fields = buildFieldModels(normalized.fields());

    Map<String, String> files = new LinkedHashMap<>();
    if (requiredLanguage == Language.JAVA) {
      files.putAll(generateJavaFiles(requiredBasePackageOrModule, className, fields));
    } else {
      files.putAll(generatePythonFiles(requiredBasePackageOrModule, className, fields));
    }
    files.put(
        "README.md",
        emitReadme(
            normalized,
            requiredGroup,
            requiredSchemaName,
            requiredLanguage,
            requiredBasePackageOrModule,
            className,
            files));
    files.put("schema.json", prettyJson(ConfigSchemaCodec.toCanonicalJsonNode(normalized)));

    byte[] zipBytes = buildZip(files);
    String filename = buildFilename(requiredGroup, requiredSchemaName, requiredLanguage);
    Map<String, String> stats = new LinkedHashMap<>();
    stats.put("group", requiredGroup);
    stats.put("schema", requiredSchemaName);
    stats.put("language", requiredLanguage.name());
    stats.put("base", requiredBasePackageOrModule);
    stats.put("className", className);
    stats.put("files", Integer.toString(files.size()));
    stats.put("bytes", Integer.toString(zipBytes.length));
    return new ConfigCodegen(zipBytes, filename, stats);
  }

  private Map<String, String> generateJavaFiles(
      String basePackage, String className, List<FieldModel> fields) {
    String packagePath = basePackage.replace('.', '/');
    String baseDir = "src/main/java/" + packagePath + "/";
    Map<String, String> files = new LinkedHashMap<>();
    files.put(baseDir + className + ".java", emitJavaConfig(basePackage, className, fields));
    files.put(baseDir + className + "Defaults.java", emitJavaDefaults(basePackage, className, fields));
    files.put(baseDir + className + "Codec.java", emitJavaCodec(basePackage, className));
    files.put(baseDir + className + "Validator.java", emitJavaValidator(basePackage, className, fields));
    return files;
  }

  private Map<String, String> generatePythonFiles(
      String baseModule, String className, List<FieldModel> fields) {
    String modulePath = baseModule.replace('.', '/');
    String baseDir = "src/main/python/" + modulePath + "/";
    String baseFile = toSnakeCase(className);
    Map<String, String> files = new LinkedHashMap<>();
    files.put(baseDir + baseFile + ".py", emitPythonModel(className, fields));
    files.put(baseDir + baseFile + "_codec.py", emitPythonCodec(baseFile, className, fields));
    files.put(baseDir + baseFile + "_validator.py", emitPythonValidator(baseFile, className, fields));
    addPythonInitFiles(files, baseModule);
    return files;
  }

  private void addPythonInitFiles(Map<String, String> files, String baseModule) {
    if (baseModule == null || baseModule.isBlank()) {
      return;
    }
    String[] segments = baseModule.split("\\.");
    StringBuilder path = new StringBuilder("src/main/python/");
    for (String segment : segments) {
      if (segment == null || segment.isBlank()) {
        continue;
      }
      path.append(segment).append('/');
      files.put(path + "__init__.py", "");
    }
  }

  private String emitJavaConfig(String packageName, String className, List<FieldModel> fields) {
    boolean hasEnums = fields.stream().anyMatch(field -> field.enumTypeName() != null);
    boolean hasRequiredReference = fields.stream().anyMatch(FieldModel::requiredReference);
    boolean hasJsonNode = fields.stream().anyMatch(FieldModel::usesJsonNode);

    StringBuilder builder = new StringBuilder();
    builder.append("package ").append(packageName).append(";\n\n");
    builder.append("import com.fasterxml.jackson.annotation.JsonProperty;\n");
    if (hasEnums) {
      builder.append("import com.fasterxml.jackson.annotation.JsonCreator;\n");
      builder.append("import com.fasterxml.jackson.annotation.JsonValue;\n");
    }
    if (hasJsonNode) {
      builder.append("import com.fasterxml.jackson.databind.JsonNode;\n");
    }
    if (hasRequiredReference) {
      builder.append("import java.util.Objects;\n");
    }
    builder.append("\n");
    if (fields.isEmpty()) {
      builder.append("public record ").append(className).append("() {}\n");
      return builder.toString();
    }

    builder.append("public record ").append(className).append("(\n");
    for (int i = 0; i < fields.size(); i++) {
      FieldModel field = fields.get(i);
      builder
          .append("    @JsonProperty(\"")
          .append(escapeJava(field.jsonKey()))
          .append("\") ")
          .append(field.javaType())
          .append(" ")
          .append(field.propertyName());
      builder.append(i + 1 < fields.size() ? ",\n" : "\n");
    }
    builder.append(") {\n");
    if (hasRequiredReference) {
      builder.append("\n");
      builder.append("  public ").append(className).append(" {\n");
      for (FieldModel field : fields) {
        if (field.requiredReference()) {
          builder
              .append("    Objects.requireNonNull(")
              .append(field.propertyName())
              .append(", \"")
              .append(escapeJava(field.jsonKey()))
              .append(" is required\");\n");
        }
      }
      builder.append("  }\n");
    }
    for (FieldModel field : fields) {
      if (field.enumTypeName() != null) {
        builder.append("\n");
        builder.append(emitJavaEnum(field));
      }
    }
    builder.append("}\n");
    return builder.toString();
  }

  private String emitJavaEnum(FieldModel field) {
    StringBuilder builder = new StringBuilder();
    builder.append("  public enum ").append(field.enumTypeName()).append(" {\n");
    for (int i = 0; i < field.enumValues().size(); i++) {
      EnumValueModel enumValue = field.enumValues().get(i);
      builder
          .append("    ")
          .append(enumValue.constant())
          .append("(\"")
          .append(escapeJava(enumValue.raw()))
          .append("\")");
      builder.append(i + 1 < field.enumValues().size() ? ",\n" : ";\n");
    }
    builder.append("\n");
    builder.append("    private final String value;\n\n");
    builder
        .append("    ")
        .append(field.enumTypeName())
        .append("(String value) {\n")
        .append("      this.value = value;\n")
        .append("    }\n\n");
    builder
        .append("    @JsonValue\n")
        .append("    public String value() {\n")
        .append("      return value;\n")
        .append("    }\n\n");
    builder
        .append("    @JsonCreator\n")
        .append("    public static ")
        .append(field.enumTypeName())
        .append(" fromValue(String value) {\n")
        .append("      if (value == null) {\n")
        .append("        return null;\n")
        .append("      }\n")
        .append("      for (")
        .append(field.enumTypeName())
        .append(" candidate : values()) {\n")
        .append("        if (candidate.value.equals(value)) {\n")
        .append("          return candidate;\n")
        .append("        }\n")
        .append("      }\n")
        .append("      throw new IllegalArgumentException(\"Unsupported ")
        .append(escapeJava(field.enumTypeName()))
        .append(" value: \" + value);\n")
        .append("    }\n\n");
    builder
        .append("    public static boolean isValid(String value) {\n")
        .append("      if (value == null) {\n")
        .append("        return false;\n")
        .append("      }\n")
        .append("      for (")
        .append(field.enumTypeName())
        .append(" candidate : values()) {\n")
        .append("        if (candidate.value.equals(value)) {\n")
        .append("          return true;\n")
        .append("        }\n")
        .append("      }\n")
        .append("      return false;\n")
        .append("    }\n");
    builder.append("  }\n");
    return builder.toString();
  }

  private String emitJavaDefaults(String packageName, String className, List<FieldModel> fields) {
    boolean hasStringDefaults =
        fields.stream()
            .anyMatch(
                field ->
                    field.hasOptionalDefault()
                        && field.field().type() == FieldType.STRING
                        && field.enumTypeName() == null);
    boolean hasJsonNodeField = fields.stream().anyMatch(FieldModel::usesJsonNode);
    List<FieldModel> jsonDefaults =
        fields.stream().filter(this::needsJsonDefaultConstant).toList();

    StringBuilder builder = new StringBuilder();
    builder.append("package ").append(packageName).append(";\n\n");
    if (hasJsonNodeField) {
      builder.append("import com.fasterxml.jackson.databind.JsonNode;\n");
    }
    if (!jsonDefaults.isEmpty()) {
      builder.append("import com.fasterxml.jackson.databind.ObjectMapper;\n");
    }
    builder.append("\n");
    builder.append("public final class ").append(className).append("Defaults {\n\n");
    if (!jsonDefaults.isEmpty()) {
      builder.append("  private static final ObjectMapper MAPPER = new ObjectMapper();\n");
      for (FieldModel field : jsonDefaults) {
        builder
            .append("  private static final JsonNode ")
            .append(defaultConstantName(field))
            .append(" = readNode(\"")
            .append(escapeJava(field.field().defaultValue().toString()))
            .append("\");\n");
      }
      builder.append("\n");
    }
    builder
        .append("  private ")
        .append(className)
        .append("Defaults() {}\n\n")
        .append("  public static ")
        .append(className)
        .append(" apply(")
        .append(className)
        .append(" in) {\n")
        .append("    if (in == null) {\n")
        .append("      return null;\n")
        .append("    }\n");
    if (!fields.stream().anyMatch(FieldModel::hasOptionalDefault)) {
      builder.append("    return in;\n");
      builder.append("  }\n");
      builder.append("}\n");
      return builder.toString();
    }

    builder.append("    return new ").append(className).append("(\n");
    for (int i = 0; i < fields.size(); i++) {
      FieldModel field = fields.get(i);
      String expression = javaDefaultExpression(className, field);
      builder.append("        ").append(expression);
      builder.append(i + 1 < fields.size() ? ",\n" : "\n");
    }
    builder.append("    );\n");
    builder.append("  }\n");
    if (hasStringDefaults) {
      builder
          .append("\n")
          .append("  private static boolean isBlank(String value) {\n")
          .append("    return value == null || value.trim().isEmpty();\n")
          .append("  }\n");
    }
    if (!jsonDefaults.isEmpty()) {
      builder
          .append("\n")
          .append("  private static JsonNode readNode(String value) {\n")
          .append("    try {\n")
          .append("      return MAPPER.readTree(value);\n")
          .append("    } catch (Exception ex) {\n")
          .append("      throw new IllegalStateException(\"Failed to parse default JSON\", ex);\n")
          .append("    }\n")
          .append("  }\n");
    }
    builder.append("}\n");
    return builder.toString();
  }

  private String emitJavaCodec(String packageName, String className) {
    StringBuilder builder = new StringBuilder();
    builder.append("package ").append(packageName).append(";\n\n");
    builder.append("import com.fasterxml.jackson.databind.JsonNode;\n");
    builder.append("import com.fasterxml.jackson.databind.ObjectMapper;\n\n");
    builder.append("public final class ").append(className).append("Codec {\n\n");
    builder.append("  private static final ObjectMapper MAPPER = new ObjectMapper();\n\n");
    builder
        .append("  private ")
        .append(className)
        .append("Codec() {}\n\n")
        .append("  public static ")
        .append(className)
        .append(" fromJson(String json) {\n")
        .append("    try {\n")
        .append("      return MAPPER.readValue(json, ")
        .append(className)
        .append(".class);\n")
        .append("    } catch (Exception ex) {\n")
        .append("      throw new IllegalArgumentException(\"Invalid JSON for ")
        .append(className)
        .append(": \" + ex.getMessage(), ex);\n")
        .append("    }\n")
        .append("  }\n\n")
        .append("  public static String toJson(")
        .append(className)
        .append(" value) {\n")
        .append("    try {\n")
        .append("      return MAPPER.writeValueAsString(value);\n")
        .append("    } catch (Exception ex) {\n")
        .append("      throw new IllegalStateException(\"Failed to write ")
        .append(className)
        .append(" JSON\", ex);\n")
        .append("    }\n")
        .append("  }\n\n")
        .append("  public static ")
        .append(className)
        .append(" fromNode(JsonNode node) {\n")
        .append("    try {\n")
        .append("      return MAPPER.treeToValue(node, ")
        .append(className)
        .append(".class);\n")
        .append("    } catch (Exception ex) {\n")
        .append("      throw new IllegalArgumentException(\"Invalid JsonNode for ")
        .append(className)
        .append(": \" + ex.getMessage(), ex);\n")
        .append("    }\n")
        .append("  }\n\n")
        .append("  public static JsonNode toNode(")
        .append(className)
        .append(" value) {\n")
        .append("    return MAPPER.valueToTree(value);\n")
        .append("  }\n");
    builder.append("}\n");
    return builder.toString();
  }

  private String emitJavaValidator(String packageName, String className, List<FieldModel> fields) {
    boolean hasPattern =
        fields.stream()
            .anyMatch(
                field ->
                    field.field().type() == FieldType.STRING
                        && field.enumTypeName() == null
                        && field.field().pattern() != null
                        && !field.field().pattern().isBlank());
    boolean hasEnum = fields.stream().anyMatch(field -> field.enumTypeName() != null);

    StringBuilder builder = new StringBuilder();
    builder.append("package ").append(packageName).append(";\n\n");
    builder.append("import java.util.ArrayList;\n");
    builder.append("import java.util.List;\n");
    if (hasPattern) {
      builder.append("import java.util.regex.Pattern;\n");
    }
    if (hasEnum) {
      builder.append("import java.util.Set;\n");
    }
    builder.append("\n");
    builder.append("public final class ").append(className).append("Validator {\n\n");
    for (FieldModel field : fields) {
      if (field.enumTypeName() != null) {
        builder
            .append("  private static final Set<String> ")
            .append(enumAllowedConstantName(field))
            .append(" = Set.of(")
            .append(
                String.join(
                    ", ",
                    field.enumValues().stream()
                        .map(value -> "\"" + escapeJava(value.raw()) + "\"")
                        .toList()))
            .append(");\n");
      }
    }
    for (FieldModel field : fields) {
      String pattern = field.field().pattern();
      if (field.field().type() == FieldType.STRING
          && field.enumTypeName() == null
          && pattern != null
          && !pattern.isBlank()) {
        builder
            .append("  private static final Pattern ")
            .append(patternConstantName(field))
            .append(" = Pattern.compile(\"")
            .append(escapeJava(pattern))
            .append("\");\n");
      }
    }
    if (hasEnum || hasPattern) {
      builder.append("\n");
    }
    builder
        .append("  private ")
        .append(className)
        .append("Validator() {}\n\n")
        .append("  public static List<String> validate(")
        .append(className)
        .append(" in) {\n")
        .append("    List<String> issues = new ArrayList<>();\n")
        .append("    if (in == null) {\n")
        .append("      issues.add(\"config is required\");\n")
        .append("      return List.copyOf(issues);\n")
        .append("    }\n");
    for (FieldModel field : fields) {
      emitJavaValidatorChecks(builder, field);
    }
    builder.append("    return List.copyOf(issues);\n");
    builder.append("  }\n");
    builder.append("}\n");
    return builder.toString();
  }

  private void emitJavaValidatorChecks(StringBuilder builder, FieldModel field) {
    String accessor = "in." + field.propertyName() + "()";
    String key = escapeJava(field.jsonKey());
    if (field.requiredReference()) {
      builder
          .append("    if (")
          .append(accessor)
          .append(" == null) {\n")
          .append("      issues.add(\"")
          .append(key)
          .append(" is required\");\n")
          .append("    }\n");
    }
    if (field.enumTypeName() != null) {
      builder
          .append("    if (")
          .append(accessor)
          .append(" != null && !")
          .append(enumAllowedConstantName(field))
          .append(".contains(")
          .append(accessor)
          .append(".value())) {\n")
          .append("      issues.add(\"")
          .append(key)
          .append(" must be one of ")
          .append(escapeJava(field.enumValues().stream().map(EnumValueModel::raw).toList().toString()))
          .append("\");\n")
          .append("    }\n");
    }
    if (field.field().type() == FieldType.STRING
        && field.enumTypeName() == null
        && field.field().pattern() != null
        && !field.field().pattern().isBlank()) {
      builder
          .append("    if (")
          .append(accessor)
          .append(" != null && !")
          .append(patternConstantName(field))
          .append(".matcher(")
          .append(accessor)
          .append(").matches()) {\n")
          .append("      issues.add(\"")
          .append(key)
          .append(" does not match pattern ")
          .append(escapeJava(field.field().pattern()))
          .append("\");\n")
          .append("    }\n");
    }
    if (field.field().type() == FieldType.INTEGER) {
      emitJavaNumericCheck(builder, field, accessor, key, true);
    } else if (field.field().type() == FieldType.NUMBER) {
      emitJavaNumericCheck(builder, field, accessor, key, false);
    }
  }

  private void emitJavaNumericCheck(
      StringBuilder builder, FieldModel field, String accessor, String key, boolean integerType) {
    BigDecimal min = field.field().min();
    BigDecimal max = field.field().max();
    String minLiteral = min == null ? null : numericLiteral(min, integerType);
    String maxLiteral = max == null ? null : numericLiteral(max, integerType);
    if (minLiteral == null && maxLiteral == null) {
      return;
    }
    if (field.javaPrimitive()) {
      if (minLiteral != null) {
        builder
            .append("    if (")
            .append(accessor)
            .append(" < ")
            .append(minLiteral)
            .append(") {\n")
            .append("      issues.add(\"")
            .append(key)
            .append(" must be >= ")
            .append(formatNumber(min))
            .append("\");\n")
            .append("    }\n");
      }
      if (maxLiteral != null) {
        builder
            .append("    if (")
            .append(accessor)
            .append(" > ")
            .append(maxLiteral)
            .append(") {\n")
            .append("      issues.add(\"")
            .append(key)
            .append(" must be <= ")
            .append(formatNumber(max))
            .append("\");\n")
            .append("    }\n");
      }
      return;
    }
    if (minLiteral != null) {
      builder
          .append("    if (")
          .append(accessor)
          .append(" != null && ")
          .append(accessor)
          .append(" < ")
          .append(minLiteral)
          .append(") {\n")
          .append("      issues.add(\"")
          .append(key)
          .append(" must be >= ")
          .append(formatNumber(min))
          .append("\");\n")
          .append("    }\n");
    }
    if (maxLiteral != null) {
      builder
          .append("    if (")
          .append(accessor)
          .append(" != null && ")
          .append(accessor)
          .append(" > ")
          .append(maxLiteral)
          .append(") {\n")
          .append("      issues.add(\"")
          .append(key)
          .append(" must be <= ")
          .append(formatNumber(max))
          .append("\");\n")
          .append("    }\n");
    }
  }

  private String emitPythonModel(String className, List<FieldModel> fields) {
    boolean hasAny = fields.stream().anyMatch(field -> "Any".equals(field.pythonType()));
    boolean hasOptional = fields.stream().anyMatch(field -> !field.required());
    List<FieldModel> ordered = pythonFieldOrder(fields);

    StringBuilder builder = new StringBuilder();
    builder.append("from dataclasses import dataclass\n");
    if (hasAny || hasOptional) {
      List<String> imports = new ArrayList<>();
      if (hasAny) {
        imports.add("Any");
      }
      if (hasOptional) {
        imports.add("Optional");
      }
      builder.append("from typing import ").append(String.join(", ", imports)).append("\n");
    }
    builder.append("\n\n");
    builder.append("@dataclass\n");
    builder.append("class ").append(className).append(":\n");
    if (ordered.isEmpty()) {
      builder.append("    pass\n");
      return builder.toString();
    }
    for (FieldModel field : ordered) {
      String type = field.pythonType();
      if (!field.required()) {
        type = "Optional[" + type + "]";
      }
      builder.append("    ").append(field.propertyName()).append(": ").append(type);
      if (!field.required()) {
        builder.append(" = None");
      }
      if (field.enumTypeName() != null) {
        builder
            .append("  # allowed: ")
            .append(String.join(", ", field.enumValues().stream().map(EnumValueModel::raw).toList()));
      }
      builder.append("\n");
    }
    return builder.toString();
  }

  private String emitPythonCodec(String modelFile, String className, List<FieldModel> fields) {
    StringBuilder builder = new StringBuilder();
    builder.append("import json\n");
    builder.append("from typing import Any, Dict, Optional\n\n");
    builder.append("try:\n");
    builder.append("    from .").append(modelFile).append(" import ").append(className).append("\n");
    builder.append("except ImportError:\n");
    builder.append("    from ").append(modelFile).append(" import ").append(className).append("\n\n");
    builder.append("def from_dict(data: Dict[str, Any]) -> ").append(className).append(":\n");
    builder.append("    payload = {} if data is None else dict(data)\n");
    builder.append("    config = ").append(className).append("(\n");
    List<FieldModel> ordered = pythonFieldOrder(fields);
    for (FieldModel field : ordered) {
      builder
          .append("        ")
          .append(field.propertyName())
          .append("=payload.get(\"")
          .append(escapePython(field.jsonKey()))
          .append("\"),\n");
    }
    builder.append("    )\n");
    builder.append("    return apply_defaults(config)\n\n");
    builder.append("def from_json(data: str) -> ").append(className).append(":\n");
    builder.append("    return from_dict(json.loads(data))\n\n");
    builder.append("def to_dict(config: ").append(className).append(") -> Dict[str, Any]:\n");
    builder.append("    if config is None:\n");
    builder.append("        return {}\n");
    builder.append("    return {\n");
    for (FieldModel field : ordered) {
      builder
          .append("        \"")
          .append(escapePython(field.jsonKey()))
          .append("\": config.")
          .append(field.propertyName())
          .append(",\n");
    }
    builder.append("    }\n\n");
    builder.append("def to_json(config: ").append(className).append(") -> str:\n");
    builder.append("    return json.dumps(to_dict(config), indent=2, sort_keys=True)\n\n");
    builder
        .append("def apply_defaults(config: Optional[")
        .append(className)
        .append("]) -> Optional[")
        .append(className)
        .append("]:\n");
    builder.append("    if config is None:\n");
    builder.append("        return None\n");
    if (!fields.stream().anyMatch(FieldModel::hasOptionalDefault)) {
      builder.append("    return config\n");
      return builder.toString();
    }
    builder.append("    return ").append(className).append("(\n");
    for (FieldModel field : ordered) {
      builder
          .append("        ")
          .append(field.propertyName())
          .append("=")
          .append(pythonDefaultExpression(field))
          .append(",\n");
    }
    builder.append("    )\n");
    return builder.toString();
  }

  private String emitPythonValidator(String modelFile, String className, List<FieldModel> fields) {
    boolean hasPattern =
        fields.stream()
            .anyMatch(
                field ->
                    field.field().type() == FieldType.STRING
                        && field.enumTypeName() == null
                        && field.field().pattern() != null
                        && !field.field().pattern().isBlank());
    boolean hasEnum = fields.stream().anyMatch(field -> field.enumTypeName() != null);
    StringBuilder builder = new StringBuilder();
    if (hasPattern) {
      builder.append("import re\n");
    }
    builder.append("from typing import List, Optional\n\n");
    builder.append("try:\n");
    builder.append("    from .").append(modelFile).append(" import ").append(className).append("\n");
    builder.append("except ImportError:\n");
    builder.append("    from ").append(modelFile).append(" import ").append(className).append("\n\n");
    for (FieldModel field : fields) {
      if (field.enumTypeName() != null) {
        builder
            .append("_")
            .append(enumAllowedConstantName(field))
            .append(" = {")
            .append(
                String.join(
                    ", ",
                    field.enumValues().stream()
                        .map(value -> "'" + escapePython(value.raw()) + "'")
                        .toList()))
            .append("}\n");
      }
    }
    for (FieldModel field : fields) {
      String pattern = field.field().pattern();
      if (field.field().type() == FieldType.STRING
          && field.enumTypeName() == null
          && pattern != null
          && !pattern.isBlank()) {
        builder
            .append("_")
            .append(patternConstantName(field))
            .append(" = re.compile(r'")
            .append(escapePythonRegex(pattern))
            .append("')\n");
      }
    }
    if (hasEnum || hasPattern) {
      builder.append("\n");
    }
    builder
        .append("def validate(config: Optional[")
        .append(className)
        .append("]) -> List[str]:\n");
    builder.append("    issues: List[str] = []\n");
    builder.append("    if config is None:\n");
    builder.append("        issues.append('config is required')\n");
    builder.append("        return issues\n");
    for (FieldModel field : fields) {
      emitPythonValidatorChecks(builder, field);
    }
    builder.append("    return issues\n");
    return builder.toString();
  }

  private void emitPythonValidatorChecks(StringBuilder builder, FieldModel field) {
    String access = "config." + field.propertyName();
    String key = escapePython(field.jsonKey());
    if (field.required()) {
      builder
          .append("    if ")
          .append(access)
          .append(" is None:\n")
          .append("        issues.append('")
          .append(key)
          .append(" is required')\n");
    }
    if (field.enumTypeName() != null) {
      builder
          .append("    if ")
          .append(access)
          .append(" is not None and ")
          .append(access)
          .append(" not in _")
          .append(enumAllowedConstantName(field))
          .append(":\n")
          .append("        issues.append('")
          .append(key)
          .append(" must be one of ")
          .append(escapePython(field.enumValues().stream().map(EnumValueModel::raw).toList().toString()))
          .append("')\n");
    }
    if (field.field().type() == FieldType.STRING
        && field.enumTypeName() == null
        && field.field().pattern() != null
        && !field.field().pattern().isBlank()) {
      builder
          .append("    if ")
          .append(access)
          .append(" is not None and _")
          .append(patternConstantName(field))
          .append(".fullmatch(str(")
          .append(access)
          .append(")) is None:\n")
          .append("        issues.append('")
          .append(key)
          .append(" does not match pattern ")
          .append(escapePython(field.field().pattern()))
          .append("')\n");
    }
    if (field.field().type() == FieldType.INTEGER || field.field().type() == FieldType.NUMBER) {
      BigDecimal min = field.field().min();
      BigDecimal max = field.field().max();
      if (min != null) {
        builder
            .append("    if ")
            .append(access)
            .append(" is not None and ")
            .append(access)
            .append(" < ")
            .append(formatPythonNumber(min, field.field().type() == FieldType.INTEGER))
            .append(":\n")
            .append("        issues.append('")
            .append(key)
            .append(" must be >= ")
            .append(formatNumber(min))
            .append("')\n");
      }
      if (max != null) {
        builder
            .append("    if ")
            .append(access)
            .append(" is not None and ")
            .append(access)
            .append(" > ")
            .append(formatPythonNumber(max, field.field().type() == FieldType.INTEGER))
            .append(":\n")
            .append("        issues.append('")
            .append(key)
            .append(" must be <= ")
            .append(formatNumber(max))
            .append("')\n");
      }
    }
  }

  private String javaDefaultExpression(String className, FieldModel field) {
    String accessor = "in." + field.propertyName() + "()";
    if (!field.hasOptionalDefault()) {
      return accessor;
    }
    JsonNode defaultValue = field.field().defaultValue();
    if (defaultValue == null || defaultValue.isNull()) {
      return accessor;
    }
    if (field.enumTypeName() != null) {
      String enumValue = defaultValue.asText(null);
      if (enumValue == null) {
        return accessor;
      }
      String constant = resolveEnumConstant(field, enumValue);
      String enumRef =
          constant == null
              ? className + "." + field.enumTypeName() + ".fromValue(\"" + escapeJava(enumValue) + "\")"
              : className + "." + field.enumTypeName() + "." + constant;
      return "(" + accessor + " == null ? " + enumRef + " : " + accessor + ")";
    }
    if (field.field().type() == FieldType.STRING) {
      String literal = "\"" + escapeJava(defaultValue.asText()) + "\"";
      return "(isBlank(" + accessor + ") ? " + literal + " : " + accessor + ")";
    }
    if (field.field().type() == FieldType.INTEGER) {
      return "(" + accessor + " == null ? " + defaultValue.asInt() + " : " + accessor + ")";
    }
    if (field.field().type() == FieldType.NUMBER) {
      return "("
          + accessor
          + " == null ? "
          + numericLiteral(defaultValue.decimalValue(), false)
          + " : "
          + accessor
          + ")";
    }
    if (field.field().type() == FieldType.BOOLEAN) {
      return "(" + accessor + " == null ? " + (defaultValue.asBoolean() ? "true" : "false") + " : " + accessor + ")";
    }
    if (needsJsonDefaultConstant(field)) {
      String constant = defaultConstantName(field);
      return "(" + accessor + " == null ? " + constant + ".deepCopy() : " + accessor + ")";
    }
    return accessor;
  }

  private String pythonDefaultExpression(FieldModel field) {
    String accessor = "config." + field.propertyName();
    if (!field.hasOptionalDefault()) {
      return accessor;
    }
    JsonNode defaultValue = field.field().defaultValue();
    if (defaultValue == null || defaultValue.isNull()) {
      return accessor;
    }
    if (field.field().type() == FieldType.STRING) {
      String defaultLiteral = "'" + escapePython(defaultValue.asText()) + "'";
      return "(" + defaultLiteral + " if " + accessor + " is None or str(" + accessor + ").strip() == '' else " + accessor + ")";
    }
    if (field.field().type() == FieldType.INTEGER) {
      return "(" + defaultValue.asInt() + " if " + accessor + " is None else " + accessor + ")";
    }
    if (field.field().type() == FieldType.NUMBER) {
      return "(" + formatPythonNumber(defaultValue.decimalValue(), false) + " if " + accessor + " is None else " + accessor + ")";
    }
    if (field.field().type() == FieldType.BOOLEAN) {
      return "(" + (defaultValue.asBoolean() ? "True" : "False") + " if " + accessor + " is None else " + accessor + ")";
    }
    return accessor;
  }

  private boolean needsJsonDefaultConstant(FieldModel field) {
    if (!field.hasOptionalDefault()) {
      return false;
    }
    FieldType type = field.field().type();
    return type == FieldType.STRUCT || type == FieldType.ARRAY;
  }

  private byte[] buildZip(Map<String, String> files) {
    TreeMap<String, String> sorted = new TreeMap<>(files);
    try {
      ByteArrayOutputStream output = new ByteArrayOutputStream();
      try (ZipOutputStream zip = new ZipOutputStream(output, StandardCharsets.UTF_8)) {
        for (Map.Entry<String, String> entry : sorted.entrySet()) {
          String path = normalizeZipPath(entry.getKey());
          ZipEntry zipEntry = new ZipEntry(path);
          zipEntry.setTime(0L);
          zip.putNextEntry(zipEntry);
          byte[] bytes =
              entry.getValue() == null
                  ? new byte[0]
                  : entry.getValue().getBytes(StandardCharsets.UTF_8);
          zip.write(bytes);
          zip.closeEntry();
        }
      }
      return output.toByteArray();
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to build codegen ZIP", ex);
    }
  }

  private String normalizeZipPath(String value) {
    String path = value == null ? "" : value.trim();
    path = path.replace('\\', '/');
    while (path.startsWith("/")) {
      path = path.substring(1);
    }
    return path;
  }

  private ConfigSchemaDoc normalizeSchemaDoc(ConfigSchemaDoc doc) {
    List<Group> groups = new ArrayList<>(safeList(doc.groups()));
    groups.sort(
        Comparator.comparingInt((Group group) -> group.order() == null ? Integer.MAX_VALUE : group.order())
            .thenComparing(group -> group.id() == null ? "" : group.id()));

    List<Struct> structs = new ArrayList<>(safeList(doc.structs()));
    structs.sort(Comparator.comparing((Struct struct) -> struct.id() == null ? "" : struct.id()));
    List<Struct> normalizedStructs = new ArrayList<>();
    for (Struct struct : structs) {
      if (struct == null) {
        continue;
      }
      normalizedStructs.add(
          new Struct(
              struct.id(),
              struct.label(),
              struct.description(),
              sortFields(struct.fields())));
    }

    List<Field> fields = sortFields(doc.fields());
    LinkedHashSet<String> requiredKeys = new LinkedHashSet<>();
    for (String key : safeList(doc.requiredKeys())) {
      if (key != null && !key.isBlank()) {
        requiredKeys.add(key.trim());
      }
    }
    List<String> required = new ArrayList<>(requiredKeys);
    required.sort(String::compareTo);
    return new ConfigSchemaDoc(
        doc.displayName(),
        doc.description(),
        List.copyOf(groups),
        List.copyOf(normalizedStructs),
        List.copyOf(fields),
        List.copyOf(required));
  }

  private List<Field> sortFields(List<Field> fields) {
    List<Field> ordered = new ArrayList<>(safeList(fields));
    ordered.sort(
        Comparator.comparingInt((Field field) -> field.order() == null ? Integer.MAX_VALUE : field.order())
            .thenComparing(field -> field.key() == null ? "" : field.key()));
    List<Field> sorted = new ArrayList<>();
    for (Field field : ordered) {
      sorted.add(
          new Field(
              field.key(),
              field.label(),
              field.description(),
              field.groupId(),
              field.order(),
              field.type(),
              field.required(),
              field.editable(),
              field.enumValues(),
              field.defaultValue(),
              field.min(),
              field.max(),
              field.pattern(),
              sortFields(field.fields()),
              field.itemType(),
              field.itemsEnum(),
              field.structRef(),
              field.minItems(),
              field.maxItems()));
    }
    return List.copyOf(sorted);
  }

  private List<FieldModel> buildFieldModels(List<Field> fields) {
    List<FieldModel> models = new ArrayList<>();
    Set<String> usedProperties = new LinkedHashSet<>();
    Set<String> usedEnumTypes = new LinkedHashSet<>();
    for (Field field : safeList(fields)) {
      if (field == null) {
        continue;
      }
      String key = requireValue(field.key(), "field.key");
      String propertyName = allocateUnique(safeJavaIdentifier(toCamelCase(key)), usedProperties);
      boolean required = Boolean.TRUE.equals(field.required());

      List<EnumValueModel> enumValues = List.of();
      String enumTypeName = null;
      if (field.type() == FieldType.STRING && !safeList(field.enumValues()).isEmpty()) {
        enumTypeName = allocateUnique(toPascalCase(propertyName) + "Enum", usedEnumTypes);
        enumValues = buildEnumValues(field.enumValues());
      }
      TypeMapping mapping = resolveTypeMapping(field, required, enumTypeName);
      models.add(
          new FieldModel(
              field,
              key,
              propertyName,
              required,
              mapping.javaType(),
              mapping.javaPrimitive(),
              mapping.pythonType(),
              enumTypeName,
              enumValues));
    }
    return List.copyOf(models);
  }

  private List<EnumValueModel> buildEnumValues(List<String> values) {
    List<EnumValueModel> models = new ArrayList<>();
    Set<String> used = new LinkedHashSet<>();
    for (String value : safeList(values)) {
      if (value == null || value.isBlank()) {
        continue;
      }
      String constant = allocateEnumConstant(value, used);
      models.add(new EnumValueModel(value, constant));
    }
    return List.copyOf(models);
  }

  private TypeMapping resolveTypeMapping(Field field, boolean required, String enumTypeName) {
    if (enumTypeName != null) {
      return new TypeMapping(enumTypeName, false, "str");
    }
    FieldType type = field.type();
    if (type == null) {
      return new TypeMapping("JsonNode", false, "Any");
    }
    return switch (type) {
      case STRING -> new TypeMapping("String", false, "str");
      case INTEGER -> required ? new TypeMapping("int", true, "int") : new TypeMapping("Integer", false, "int");
      case NUMBER -> required ? new TypeMapping("double", true, "float") : new TypeMapping("Double", false, "float");
      case BOOLEAN ->
          required ? new TypeMapping("boolean", true, "bool") : new TypeMapping("Boolean", false, "bool");
      case STRUCT, ARRAY -> new TypeMapping("JsonNode", false, "Any");
    };
  }

  private List<FieldModel> pythonFieldOrder(List<FieldModel> fields) {
    List<FieldModel> required = new ArrayList<>();
    List<FieldModel> optional = new ArrayList<>();
    for (FieldModel field : fields) {
      if (field.required()) {
        required.add(field);
      } else {
        optional.add(field);
      }
    }
    List<FieldModel> ordered = new ArrayList<>(required.size() + optional.size());
    ordered.addAll(required);
    ordered.addAll(optional);
    return ordered;
  }

  private String resolveClassName(String displayName, String schemaName) {
    String base = displayName == null || displayName.isBlank() ? schemaName : displayName;
    String className = toPascalCase(base);
    if (className.isBlank()) {
      className = "Config";
    }
    if (!className.endsWith("Config")) {
      className = className + "Config";
    }
    if (!Character.isJavaIdentifierStart(className.charAt(0))) {
      className = "Config" + className;
    }
    return className;
  }

  private String resolveBasePackageOrModule(String value, String schemaName) {
    if (value == null || value.isBlank()) {
      return defaultBasePackage(schemaName);
    }
    return value.trim();
  }

  public static String defaultBasePackage(String schemaNameOrconfigId) {
    String token = schemaNameOrconfigId == null ? "" : schemaNameOrconfigId.trim();
    String lower = token.toLowerCase(Locale.ROOT);
    if (lower.startsWith(CONFIG_PREFIX)) {
      token = token.substring(CONFIG_PREFIX.length());
      lower = token.toLowerCase(Locale.ROOT);
    }
    if (lower.startsWith(CONFIG_PREFIX)) {
      token = token.substring(CONFIG_PREFIX.length());
    }
    if (token.isBlank()) {
      token = "config";
    }
    token = token.replaceAll("[^A-Za-z0-9]+", "");
    if (token.isBlank()) {
      token = "config";
    }
    if (Character.isDigit(token.charAt(0))) {
      token = "c" + token;
    }
    return DEFAULT_PACKAGE_ROOT + token.toLowerCase(Locale.ROOT);
  }

  private void validateModuleName(String value) {
    if (value == null || value.isBlank()) {
      throw new IllegalArgumentException("basePackageOrModule is required");
    }
    if (!MODULE_PATTERN.matcher(value).matches()) {
      throw new IllegalArgumentException("basePackageOrModule must be a dotted identifier path");
    }
  }

  private String formatNumber(BigDecimal value) {
    return value == null ? "" : value.stripTrailingZeros().toPlainString();
  }

  private String numericLiteral(BigDecimal value, boolean integer) {
    if (value == null) {
      return integer ? "0" : "0.0d";
    }
    if (integer) {
      return Integer.toString(value.intValue());
    }
    String text = value.stripTrailingZeros().toPlainString();
    if (!text.contains(".") && !text.contains("e") && !text.contains("E")) {
      text = text + ".0";
    }
    return text + "d";
  }

  private String formatPythonNumber(BigDecimal value, boolean integer) {
    if (value == null) {
      return integer ? "0" : "0.0";
    }
    if (integer) {
      return Integer.toString(value.intValue());
    }
    String text = value.stripTrailingZeros().toPlainString();
    if (!text.contains(".") && !text.contains("e") && !text.contains("E")) {
      text = text + ".0";
    }
    return text;
  }

  private String allocateEnumConstant(String rawValue, Set<String> used) {
    String normalized = rawValue == null ? "VALUE" : rawValue.trim();
    if (normalized.isEmpty()) {
      normalized = "VALUE";
    }
    normalized = normalized.replaceAll("[^A-Za-z0-9]+", "_").toUpperCase(Locale.ROOT);
    if (normalized.isEmpty()) {
      normalized = "VALUE";
    }
    if (Character.isDigit(normalized.charAt(0))) {
      normalized = "_" + normalized;
    }
    String candidate = normalized;
    int suffix = 2;
    while (used.contains(candidate)) {
      candidate = normalized + "_" + suffix;
      suffix++;
    }
    used.add(candidate);
    return candidate;
  }

  private String allocateUnique(String base, Set<String> used) {
    String candidate = base == null || base.isBlank() ? "field" : base;
    if (!used.contains(candidate)) {
      used.add(candidate);
      return candidate;
    }
    int suffix = 2;
    while (used.contains(candidate + suffix)) {
      suffix++;
    }
    String resolved = candidate + suffix;
    used.add(resolved);
    return resolved;
  }

  private String safeJavaIdentifier(String value) {
    String candidate = value == null || value.isBlank() ? "field" : value;
    StringBuilder builder = new StringBuilder();
    for (int i = 0; i < candidate.length(); i++) {
      char ch = candidate.charAt(i);
      if (i == 0 && !Character.isJavaIdentifierStart(ch)) {
        builder.append('_');
      }
      builder.append(Character.isJavaIdentifierPart(ch) ? ch : '_');
    }
    String resolved = builder.toString();
    if (JAVA_KEYWORDS.contains(resolved)) {
      resolved = resolved + "_";
    }
    return resolved;
  }

  private String toCamelCase(String raw) {
    if (raw == null || raw.isBlank()) {
      return "field";
    }
    String value = raw.trim();
    if (value.matches("[a-z][A-Za-z0-9]*")) {
      return value;
    }
    if (value.matches("[A-Z][A-Za-z0-9]*")) {
      return Character.toLowerCase(value.charAt(0)) + value.substring(1);
    }
    String[] tokens = value.split("[^A-Za-z0-9]+");
    StringBuilder builder = new StringBuilder();
    for (String token : tokens) {
      if (token == null || token.isBlank()) {
        continue;
      }
      if (builder.length() == 0) {
        builder.append(Character.toLowerCase(token.charAt(0)));
        if (token.length() > 1) {
          builder.append(token.substring(1));
        }
      } else {
        builder.append(Character.toUpperCase(token.charAt(0)));
        if (token.length() > 1) {
          builder.append(token.substring(1));
        }
      }
    }
    return builder.length() == 0 ? "field" : builder.toString();
  }

  private String toPascalCase(String raw) {
    if (raw == null || raw.isBlank()) {
      return "";
    }
    String[] tokens = raw.trim().split("[^A-Za-z0-9]+");
    StringBuilder builder = new StringBuilder();
    for (String token : tokens) {
      if (token == null || token.isBlank()) {
        continue;
      }
      builder.append(Character.toUpperCase(token.charAt(0)));
      if (token.length() > 1) {
        builder.append(token.substring(1));
      }
    }
    return builder.toString();
  }

  private String toSnakeCase(String value) {
    if (value == null || value.isBlank()) {
      return "config_layer";
    }
    String normalized = value.replaceAll("([a-z0-9])([A-Z])", "$1_$2");
    normalized = normalized.replaceAll("[^A-Za-z0-9]+", "_");
    normalized = normalized.replaceAll("_+", "_");
    normalized = normalized.replaceAll("^_", "");
    normalized = normalized.replaceAll("_$", "");
    return normalized.toLowerCase(Locale.ROOT);
  }

  private String upperIdentifier(String value) {
    if (value == null || value.isBlank()) {
      return "FIELD";
    }
    String normalized = value.replaceAll("[^A-Za-z0-9]+", "_").toUpperCase(Locale.ROOT);
    if (normalized.isEmpty()) {
      return "FIELD";
    }
    if (Character.isDigit(normalized.charAt(0))) {
      return "_" + normalized;
    }
    return normalized;
  }

  private String escapeJava(String value) {
    if (value == null) {
      return "";
    }
    return value.replace("\\", "\\\\").replace("\"", "\\\"");
  }

  private String escapePython(String value) {
    if (value == null) {
      return "";
    }
    return value.replace("\\", "\\\\").replace("'", "\\'");
  }

  private String escapePythonRegex(String value) {
    return escapePython(value);
  }

  private String slug(String value) {
    if (value == null || value.isBlank()) {
      return "unknown";
    }
    String normalized = value.trim().toLowerCase(Locale.ROOT);
    normalized = normalized.replaceAll("[^a-z0-9._-]+", "-");
    normalized = normalized.replaceAll("-+", "-");
    normalized = normalized.replaceAll("^-", "");
    normalized = normalized.replaceAll("-$", "");
    return normalized.isBlank() ? "unknown" : normalized;
  }

  private String requireValue(String value, String field) {
    if (value == null || value.isBlank()) {
      throw new IllegalArgumentException(field + " is required");
    }
    return value.trim();
  }

  private String nullToValue(String value, String fallback) {
    if (value == null || value.isBlank()) {
      return fallback;
    }
    return value;
  }

  private <T> List<T> safeList(List<T> list) {
    return list == null ? List.of() : list;
  }

  private record TypeMapping(String javaType, boolean javaPrimitive, String pythonType) {}

  private record EnumValueModel(String raw, String constant) {}

  private record FieldModel(
      Field field,
      String jsonKey,
      String propertyName,
      boolean required,
      String javaType,
      boolean javaPrimitive,
      String pythonType,
      String enumTypeName,
      List<EnumValueModel> enumValues) {

    private boolean usesJsonNode() {
      return "JsonNode".equals(javaType);
    }

    private boolean requiredReference() {
      return required && !javaPrimitive;
    }

    private boolean hasOptionalDefault() {
      return !required && field != null && field.defaultValue() != null && !field.defaultValue().isNull();
    }
  }
