package org.pronzato.fabric.api.services.grpc;

import io.grpc.ClientCall;
import io.grpc.ClientInterceptor;
import io.grpc.ManagedChannel;
import io.grpc.MethodDescriptor;
import io.grpc.netty.shaded.io.grpc.netty.GrpcSslContexts;
import io.grpc.netty.shaded.io.grpc.netty.NettyChannelBuilder;
import io.grpc.netty.shaded.io.netty.handler.ssl.SslContext;
import io.grpc.netty.shaded.io.netty.handler.ssl.SslContextBuilder;
import java.io.File;
import java.util.ArrayList;
import java.util.List;
import java.util.Objects;
import java.util.concurrent.ConcurrentHashMap;
import java.util.concurrent.ConcurrentMap;
import java.util.concurrent.TimeUnit;

public final class GrpcClientRuntime {
  private static final Logger LOGGER = LoggerFactory.getLogger(GrpcClientRuntime.class);
  private GrpcClientRuntime() {}

  public static ManagedChannel createChannel(
      Config cfg, List<ClientInterceptor> extraInterceptors) throws Exception {
    Objects.requireNonNull(cfg, "cfg");
    cfg.validate();

    String host = ServiceAdvertise.requireRoutableHost(cfg.host, "grpc target host");
    NettyChannelBuilder builder = NettyChannelBuilder.forAddress(host, cfg.port);

    if (cfg.tlsEnabled) {
      SslContextBuilder sslBuilder = GrpcSslContexts.forClient();
      if (cfg.trustCertPem != null) {
        sslBuilder.trustManager(cfg.trustCertPem);
      }
      SslContext context = sslBuilder.build();
      builder.sslContext(context);
    } else {
      if (cfg.kerberosEnabled && cfg.kerberosRequireTls) {
        throw new IllegalStateException("Kerberos requires TLS but grpc.tls.enabled=false");
      }
      builder.usePlaintext();
    }

    KerberosClient ownedKerberosContext = null;
    List<ClientInterceptor> interceptors = new ArrayList<>();
    if (cfg.kerberosEnabled) {
      ownedKerberosContext = startKerberos(cfg);
      String realm = KerberosSpn.resolveRealm(cfg.kerberosRealm, ownedKerberosContext.getSubject());
      String baseSpn = KerberosSpn.resolveSpn(cfg.kerberosSpn, host);
      String normalizedSpn = KerberosSpn.normalizeSpnWithRealm(baseSpn, realm);
      interceptors.add(
          new GrpcKerberosClientInterceptor(
              ownedKerberosContext, normalizedSpn, cfg.useDelegatedSubjectIfPresent));
    }

    if (extraInterceptors != null) {
      for (ClientInterceptor interceptor : extraInterceptors) {
        if (interceptor != null) {
          interceptors.add(interceptor);
        }
      }
    }
    if (!interceptors.isEmpty()) {
      builder.intercept(interceptors);
    }

    ManagedChannel channel = builder.build();
    if (ownedKerberosContext == null) {
      return channel;
    }
    return new ManagedChannelWithCleanup(channel, ownedKerberosContext);
  }

  public static ManagedChannel createRoutingChannel(
      RoutingConfig cfg, List<ClientInterceptor> extraInterceptors) throws Exception {
    Objects.requireNonNull(cfg, "cfg");
    cfg.validate();
    return new RoutingManagedChannel(cfg, extraInterceptors);
  }

  private static KerberosClient startKerberos(Config cfg) {
    KerberosConfig kerberosConfig =
        new KerberosConfig(
            KerberosConfig.DEFAULT_CONTEXT_NAME,
            true,
            cfg.kerberosPrincipal,
            cfg.kerberosKeytab.toPath(),
            null,
            KerberosKrb5Source.DEFAULT,
            cfg.kerberosRealm,
            false,
            true);
    return new KerberosClient(kerberosConfig).start();
  }

  public static final class RoutingConfig {
    public String serviceId;
    public ServiceHbCache statusCache;
    public ServiceInstanceResolver instanceResolver;
    public String serviceGroup;
    public String serviceName;
    public boolean tlsEnabled;
    public File trustCertPem;
    public boolean kerberosEnabled;
    public boolean kerberosRequireTls = true;
    public String kerberosSpn;
    public String kerberosPrincipal;
    public File kerberosKeytab;
    public String kerberosRealm;
    public boolean useDelegatedSubjectIfPresent;
    public boolean logRoutingSelections;
    public boolean closeInstanceResolverOnCleanup = true;
    public long shutdownTimeoutMillis = 2000L;
    public ServiceRoutingMode routingMode = ServiceRoutingMode.ROUND_ROBIN;
    public String defaultEffectiveSubject;
    public String callerAppSubject;
    public String routingKey;

    public void validate() {
      serviceId = Strings.trimToNull(serviceId);
      serviceGroup = Strings.trimToNull(serviceGroup);
      serviceName = Strings.trimToNull(serviceName);
      defaultEffectiveSubject = Strings.trimToNull(defaultEffectiveSubject);
      callerAppSubject = Strings.trimToNull(callerAppSubject);
      routingKey = Strings.trimToNull(routingKey);
      if (statusCache == null && instanceResolver == null) {
        throw new IllegalArgumentException("routing status cache is required");
      }
      if (instanceResolver != null) {
        if (serviceGroup == null) {
          throw new IllegalArgumentException("routing service group is required");
        }
        if (serviceName == null) {
          throw new IllegalArgumentException("routing service name is required");
        }
      }
      if (kerberosEnabled) {
        kerberosPrincipal = Strings.trimToNull(kerberosPrincipal);
        if (kerberosPrincipal == null) {
          throw new IllegalArgumentException(
              "Kerberos principal is required when grpc.kerberos.enabled=true");
        }
        if (kerberosKeytab == null) {
          throw new IllegalArgumentException(
              "Kerberos keytab is required when grpc.kerberos.enabled=true");
        }
        if (!kerberosKeytab.isFile()) {
          throw new IllegalArgumentException(
              "Kerberos keytab does not exist: " + kerberosKeytab.getAbsolutePath());
        }
        kerberosSpn = Strings.trimToNull(kerberosSpn);
        kerberosRealm = Strings.trimToNull(kerberosRealm);
      }
      if (routingMode == null) {
        routingMode = ServiceRoutingMode.ROUND_ROBIN;
      }
    }
  }

  public static final class Config {
    public String host;
    public int port;
    public boolean tlsEnabled;
    public File trustCertPem;
    public boolean kerberosEnabled;
    public boolean kerberosRequireTls = true;
    public String kerberosSpn;
    public String kerberosPrincipal;
    public File kerberosKeytab;
    public String kerberosRealm;
    public boolean useDelegatedSubjectIfPresent;

    public void validate() {
      host = Strings.trimToNull(host);
      if (host == null) {
        throw new IllegalArgumentException("grpc target host is required");
      }
      if (port <= 0) {
        throw new IllegalArgumentException("grpc target port must be > 0");
      }
      if (kerberosEnabled) {
        kerberosPrincipal = Strings.trimToNull(kerberosPrincipal);
        if (kerberosPrincipal == null) {
          throw new IllegalArgumentException(
              "Kerberos principal is required when grpc.kerberos.enabled=true");
        }
        if (kerberosKeytab == null) {
          throw new IllegalArgumentException(
              "Kerberos keytab is required when grpc.kerberos.enabled=true");
        }
        if (!kerberosKeytab.isFile()) {
          throw new IllegalArgumentException(
              "Kerberos keytab does not exist: " + kerberosKeytab.getAbsolutePath());
        }
        kerberosSpn = Strings.trimToNull(kerberosSpn);
        kerberosRealm = Strings.trimToNull(kerberosRealm);
      }
    }
  }

  private static final class RoutingManagedChannel extends ManagedChannel {
    private final RoutingConfig cfg;
    private final List<ClientInterceptor> interceptors;
    private final ConcurrentMap<String, ManagedChannel> channels = new ConcurrentHashMap<>();
    private final AtomicBoolean shutdown = new AtomicBoolean(false);
    private final AtomicBoolean cleanupDone = new AtomicBoolean(false);
    private final ServiceRoutingState routingState = new ServiceRoutingState();
    private volatile String lastRouteKey;

    private RoutingManagedChannel(RoutingConfig cfg, List<ClientInterceptor> extraInterceptors) {
      this.cfg = cfg;
      if (extraInterceptors == null || extraInterceptors.isEmpty()) {
        this.interceptors = List.of();
      } else {
        List<ClientInterceptor> normalized = new ArrayList<>();
        for (ClientInterceptor interceptor : extraInterceptors) {
          if (interceptor != null) {
            normalized.add(interceptor);
          }
        }
        this.interceptors = normalized.isEmpty() ? List.of() : List.copyOf(normalized);
      }
    }

    @Override
    public ManagedChannel shutdown() {
      if (shutdown.compareAndSet(false, true)) {
        for (ManagedChannel channel : channels.values()) {
          if (channel != null) {
            channel.shutdown();
          }
        }
      }
      if (channels.isEmpty()) {
        cleanupRouting();
      }
      return this;
    }

    @Override
    public boolean isShutdown() {
      return shutdown.get();
    }

    @Override
    public boolean isTerminated() {
      if (!shutdown.get()) {
        return false;
      }
      for (ManagedChannel channel : channels.values()) {
        if (channel != null && !channel.isTerminated()) {
          return false;
        }
      }
      cleanupRouting();
      return true;
    }

    @Override
    public ManagedChannel shutdownNow() {
      if (shutdown.compareAndSet(false, true)) {
        for (ManagedChannel channel : channels.values()) {
          if (channel != null) {
            channel.shutdownNow();
          }
        }
      }
      cleanupRouting();
      return this;
    }

    @Override
    public boolean awaitTermination(long timeout, TimeUnit unit) throws InterruptedException {
      long timeoutNanos = unit.toNanos(timeout);
      long start = System.nanoTime();
      for (ManagedChannel channel : channels.values()) {
        if (channel == null) {
          continue;
        }
        long elapsed = System.nanoTime() - start;
        long remaining = timeoutNanos - elapsed;
        if (remaining <= 0L) {
          return false;
        }
        if (!channel.awaitTermination(remaining, TimeUnit.NANOSECONDS)) {
          return false;
        }
      }
      cleanupRouting();
      return true;
    }

    @Override
    public <RequestT, ResponseT> ClientCall<RequestT, ResponseT> newCall(
        MethodDescriptor<RequestT, ResponseT> methodDescriptor, io.grpc.CallOptions callOptions) {
      if (shutdown.get()) {
        throw unavailable("Routing channel is closed");
      }
      ServiceRoutingContext routingContext = routingContext(methodDescriptor);
      ServiceHbCache.Target target =
          // Selection happens once per RPC; stream RPCs stay pinned to that selected endpoint.
          cfg.instanceResolver != null
              ? (cfg.routingMode == ServiceRoutingMode.ROUND_ROBIN
                  ? cfg.instanceResolver.chooseRoundRobin(
                      cfg.serviceGroup, cfg.serviceName, serviceId())
                  : cfg.instanceResolver.choose(
                      cfg.serviceGroup,
                      cfg.serviceName,
                      serviceId(),
                      cfg.routingMode,
                      routingContext,
                      routingState))
              : ServiceEndpointSelector.choose(
                  cfg.statusCache,
                  serviceId(),
                  cfg.routingMode,
                  routingContext,
                  routingState);
      if (target == null) {
        throw unavailable("No available instances for service: " + serviceId());
      }
      String host = ServiceAdvertise.requireRoutableHost(target.host, "grpc target host");
      int port = target.port;
      String key = host + ":" + port + "|" + target.tls + "|" + Strings.nullToEmpty(target.authorityOverride);
      ManagedChannel delegate;
      try {
        delegate =
            channels.computeIfAbsent(
                key, ignored -> createEndpointChannel(host, port, target.tls));
      } catch (RuntimeException ex) {
        if (ex.getCause() instanceof Exception cause) {
          throw new IllegalStateException(
              "Failed to create routed channel for service " + serviceId() + " endpoint " + host + ":" + port,
              cause);
        }
        throw ex;
      }
      if (cfg.logRoutingSelections) {
        LOGGER.info("[GrpcClient][routing] mode={} serviceId={} endpoint={} instance={} tls={} subject={} callerApp={} routingKey={} operation={}", cfg.routingMode, serviceId(), host + ":" + port, target.instanceId, target.tls, routingContext.effectiveSubject(), routingContext.callerAppSubject(), routingContext.routingKey(), routingContext.operation());
      } else if (!key.equals(lastRouteKey)) {
        lastRouteKey = key;
      }
      return delegate.newCall(methodDescriptor, callOptions);
    }

    private ServiceRoutingContext routingContext(MethodDescriptor<?, ?> methodDescriptor) {
      String operation = methodDescriptor == null ? null : methodDescriptor.getFullMethodName();
      String subject = resolveEffectiveSubject();
      CallerAppContext.ContextValue callerContext = CallerAppContext.get().orElse(null);
      String callerAppSubject =
          Strings.firstNonBlank(
              callerContext == null ? null : callerContext.callerAppSubject(),
              cfg.callerAppSubject);
      return new ServiceRoutingContext(
          serviceId(),
          "GRPC",
          subject,
          callerAppSubject,
          cfg.routingKey,
          callerContext == null ? null : callerContext.requestId(),
          operation);
    }

    private String resolveEffectiveSubject() {
      UserPrincipal principal = UserContext.get().orElse(null);
      String userSubject = principal == null ? null : Strings.trimToNull(principal.username());
      return Strings.firstNonBlank(userSubject, cfg.defaultEffectiveSubject);
    }

    @Override
    public String authority() {
      return "fabric-routing/" + serviceId();
    }

    private ManagedChannel createEndpointChannel(String host, int port, boolean routeTls) {
      Config endpoint = new Config();
      endpoint.host = host;
      endpoint.port = port;
      endpoint.tlsEnabled = cfg.tlsEnabled || routeTls;
      endpoint.trustCertPem = cfg.trustCertPem;
      endpoint.kerberosEnabled = cfg.kerberosEnabled;
      endpoint.kerberosRequireTls = cfg.kerberosRequireTls;
      endpoint.kerberosSpn = cfg.kerberosSpn;
      endpoint.kerberosPrincipal = cfg.kerberosPrincipal;
      endpoint.kerberosKeytab = cfg.kerberosKeytab;
      endpoint.kerberosRealm = cfg.kerberosRealm;
      endpoint.useDelegatedSubjectIfPresent = cfg.useDelegatedSubjectIfPresent;
      try {
        return createChannel(endpoint, interceptors);
      } catch (Exception ex) {
        throw new RuntimeException(ex);
      }
    }

    private void cleanupRouting() {
      if (!cleanupDone.compareAndSet(false, true)) {
        return;
      }
      try {
        if (cfg.statusCache != null) {
          cfg.statusCache.close();
        }
      } catch (Exception ignore) {
      }
      try {
        if (cfg.closeInstanceResolverOnCleanup && cfg.instanceResolver != null) {
          cfg.instanceResolver.close();
        }
      } catch (Exception ignore) {
      }
    }

    private String serviceId() {
      return Strings.isBlank(cfg.serviceId) ? "<unknown>" : cfg.serviceId;
    }

    private IllegalStateException unavailable(String message) {
      return new IllegalStateException(message);
    }
  }

  private static final class ManagedChannelWithCleanup extends ManagedChannel {
    private final ManagedChannel delegate;
    private final AutoCloseable cleanup;
    private final AtomicBoolean cleanupDone = new AtomicBoolean(false);

    private ManagedChannelWithCleanup(ManagedChannel delegate, AutoCloseable cleanup) {
      this.delegate = Objects.requireNonNull(delegate, "delegate");
      this.cleanup = Objects.requireNonNull(cleanup, "cleanup");
    }

    @Override
    public ManagedChannel shutdown() {
      delegate.shutdown();
      if (delegate.isTerminated()) {
        cleanup();
      }
      return this;
    }

    @Override
    public boolean isShutdown() {
      return delegate.isShutdown();
    }

    @Override
    public boolean isTerminated() {
      boolean terminated = delegate.isTerminated();
      if (terminated) {
        cleanup();
      }
      return terminated;
    }

    @Override
    public ManagedChannel shutdownNow() {
      delegate.shutdownNow();
      cleanup();
      return this;
    }

    @Override
    public boolean awaitTermination(long timeout, TimeUnit unit) throws InterruptedException {
      boolean terminated = delegate.awaitTermination(timeout, unit);
      if (terminated) {
        cleanup();
      }
      return terminated;
    }

    @Override
    public <RequestT, ResponseT> ClientCall<RequestT, ResponseT> newCall(
        MethodDescriptor<RequestT, ResponseT> methodDescriptor, io.grpc.CallOptions callOptions) {
      return delegate.newCall(methodDescriptor, callOptions);
    }

    @Override
    public String authority() {
      return delegate.authority();
    }

    private void cleanup() {
      if (!cleanupDone.compareAndSet(false, true)) {
        return;
      }
      try {
        cleanup.close();
      } catch (Exception ignore) {
      }
    }
  }
}
