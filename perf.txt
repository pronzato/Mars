public final class IngestAgentImpl implements IngestAgent {
  private static final Logger LOGGER = LoggerFactory.getLogger(IngestAgentImpl.class);

  static final long[] RESTART_BACKOFF_MILLIS = new long[] {1_000L, 5_000L, 15_000L};
  private static final String CONFIG_CONNECTION_NAME = "amps-fabric-discovery";

  private final Fabric.Session session;
  private final DiscoveryCache DiscoveryCache;
  private final String sessionGroup;
  private final IngestAgent.Options options;
  private final PumpFactory pumpFactory;
  private final Sleeper sleeper;
  private final ExecutorService retryExecutor;

  private final Object lock = new Object();
  private final Map<IngestKey, DefinitionIngestSource> desiredSources = new HashMap<>();
  private final Map<IngestKey, RunningPump> runningPumps = new HashMap<>();
  private final Map<IngestKey, DataStatusPublisher> statusPublishers = new HashMap<>();
  private final Map<IngestKey, Integer> restartAttempts = new HashMap<>();
  private final Set<IngestKey> restartInFlight = new HashSet<>();
  private final Set<IngestKey> onceCompleted = new HashSet<>();

  private final AtomicBoolean started = new AtomicBoolean(false);
  private final AtomicBoolean closed = new AtomicBoolean(false);
  private final AtomicBoolean initialSyncComplete = new AtomicBoolean(false);
  private final AtomicBoolean initialConfigRefreshPending = new AtomicBoolean(false);

  private volatile AmpsConnection.Subscription configSubscription;

  public IngestAgentImpl(Fabric.Session session, IngestAgent.Options options) {
    this(
        session,
        requireConfigCache(session),
        requireGroup(session),
        options == null ? IngestAgent.Options.defaults() : options,
        PumpFactory.DEFAULT,
        Sleeper.SYSTEM,
        newRetryExecutor());
  }

  IngestAgentImpl(
      Fabric.Session session,
      DiscoveryCache DiscoveryCache,
      String sessionGroup,
      IngestAgent.Options options,
      PumpFactory pumpFactory,
      Sleeper sleeper,
      ExecutorService retryExecutor) {
    this.session = Objects.requireNonNull(session, "session");
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.sessionGroup = Strings.requireNonBlank(sessionGroup, "sessionGroup");
    this.options = options == null ? IngestAgent.Options.defaults() : options;
    this.pumpFactory = Objects.requireNonNull(pumpFactory, "pumpFactory");
    this.sleeper = Objects.requireNonNull(sleeper, "sleeper");
    this.retryExecutor = Objects.requireNonNull(retryExecutor, "retryExecutor");
  }

  @Override
  public void start() throws Exception {
    if (closed.get()) {
      throw new IllegalStateException("IngestAgent is closed");
    }
    if (!session.started()) {
      throw new IllegalStateException("Session must be started before IngestAgent");
    }
    if (!started.compareAndSet(false, true)) {
      return;
    }

    subscribeConfig();
    try {
      List<DefinitionIngestSource> discovered =
          new ArrayList<>(DiscoveryCache.list(DefinitionIngestSource.class, sessionGroup));
      List<String> names = new ArrayList<>();
      for (DefinitionIngestSource source : discovered) {
        IngestKey key = keyOf(source == null ? null : source.envelope());
        if (key != null) {
          names.add(key.ingestName());
        }
      }
      LOGGER.info("[IngestAgent][start] discovered={} dryRun={} once={} ingests={}", names.size(), options.dryRun(), options.once(), names);
      for (DefinitionIngestSource source : discovered) {
        onSourceUpsert(source);
      }
    } finally {
      initialSyncComplete.set(true);
    }
    if (initialConfigRefreshPending.getAndSet(false)) {
      refreshFromCache();
    }
  }

  @Override
  public boolean started() {
    return started.get() && !closed.get();
  }

  @Override
  public void close() {
    if (!closed.compareAndSet(false, true)) {
      return;
    }
    started.set(false);

    AmpsConnection.Subscription activeSubscription = configSubscription;
    configSubscription = null;
    CloseSupport.closeQuietly(activeSubscription);

    synchronized (lock) {
      List<IngestKey> keys = new ArrayList<>(runningPumps.keySet());
      for (IngestKey key : keys) {
        stopRunningLocked(key, "shutdown");
      }
      for (DataStatusPublisher publisher : statusPublishers.values()) {
        CloseSupport.closeQuietly(publisher);
      }
      statusPublishers.clear();
      desiredSources.clear();
      restartAttempts.clear();
      restartInFlight.clear();
      onceCompleted.clear();
    }
    initialConfigRefreshPending.set(false);
    initialSyncComplete.set(false);
    retryExecutor.shutdownNow();
  }

  void onSourceUpsert(DefinitionIngestSource source) {
    IngestKey key = keyOf(source == null ? null : source.envelope());
    if (key == null || !isManagedGroup(key.group())) {
      return;
    }
    synchronized (lock) {
      desiredSources.put(key, source);
      onceCompleted.remove(key);
      reconcileLocked(key);
    }
  }

  void onSourceDelete(String group, String ingestName) {
    onSourceDelete(group, ingestName, "deleted");
  }

  private void onSourceDelete(String group, String ingestName, String reason) {
    IngestKey key = keyOf(group, ingestName);
    if (key == null || !isManagedGroup(key.group())) {
      return;
    }
    synchronized (lock) {
      desiredSources.remove(key);
      onceCompleted.remove(key);
      restartAttempts.remove(key);
      stopRunningLocked(key, reason);
      closeStatusPublisherLocked(key);
    }
  }

  private void subscribeConfig() throws Exception {
    String filter =
        "(/envelope/type='INGEST_SOURCE' AND /envelope/group='"
            + escapeFilterValue(sessionGroup)
            + "')";
    AmpsConnection.SubscriptionSpec spec =
        new AmpsConnection.SubscriptionSpec(true, DiscoveryTopics.DISCOVERY, "oof", filter);
    AmpsConnection connection = session.connections().amps(CONFIG_CONNECTION_NAME);
    configSubscription =
        connection.sowAndSubscribe(
            spec,
            (AmpsConnection.SubscriptionListener) message -> onConfigMessage(message.raw()));
  }

  private void onConfigMessage(Message raw) {
    if (raw == null) {
      return;
    }
    int command = raw.getCommand();
    if (command == Message.Command.GroupEnd) {
      return;
    }
    if (command != Message.Command.SOW
        && command != Message.Command.Publish
        && command != Message.Command.OOF) {
      return;
    }
    if (!initialSyncComplete.get()) {
      initialConfigRefreshPending.set(true);
      return;
    }

    String payload = raw.getData();
    if (Strings.isBlank(payload)) {
      if (command == Message.Command.OOF) {
        refreshFromCache();
      }
      return;
    }

    final Definition definition;
    try {
      definition = DiscoveryCodec.decode(payload);
    } catch (Exception ignore) {
      return;
    }
    if (!(definition instanceof DefinitionIngestSource source)) {
      return;
    }
    IngestKey key = keyOf(source.envelope());
    if (key == null || !isManagedGroup(key.group())) {
      return;
    }
    if (command == Message.Command.OOF) {
      onSourceDelete(key.group(), key.ingestName(), "deleted");
      return;
    }
    onSourceUpsert(source);
  }

  private void refreshFromCache() {
    List<DefinitionIngestSource> snapshot =
        new ArrayList<>(DiscoveryCache.list(DefinitionIngestSource.class, sessionGroup));
    Set<IngestKey> seen = new HashSet<>();
    for (DefinitionIngestSource source : snapshot) {
      IngestKey key = keyOf(source == null ? null : source.envelope());
      if (key == null || !isManagedGroup(key.group())) {
        continue;
      }
      seen.add(key);
      onSourceUpsert(source);
    }
    synchronized (lock) {
      List<IngestKey> existing = new ArrayList<>(desiredSources.keySet());
      for (IngestKey key : existing) {
        if (!seen.contains(key)) {
          desiredSources.remove(key);
          restartAttempts.remove(key);
          stopRunningLocked(key, "deleted");
          closeStatusPublisherLocked(key);
        }
      }
    }
  }

  private void reconcileLocked(IngestKey key) {
    DefinitionIngestSource source = desiredSources.get(key);
    if (source == null) {
      restartAttempts.remove(key);
      stopRunningLocked(key, "deleted");
      closeStatusPublisherLocked(key);
      return;
    }

    PipelineSpec desired = buildSpec(source, key);
    if (desired == null) {
      desiredSources.remove(key);
      restartAttempts.remove(key);
      stopRunningLocked(key, "disabled");
      closeStatusPublisherLocked(key);
      return;
    }
    DataStatusPublisher statusPublisher = ensureStatusPublisherLocked(key, desired);

    RunningPump running = runningPumps.get(key);
    if (running != null && running.spec().equals(desired)) {
      if (statusPublisher != null) {
        statusPublisher.publishOnline("ingest pipeline running");
      }
      return;
    }
    if (running != null) {
      stopRunningLocked(key, "reconfigure");
    }

    if (options.dryRun()) {
      logStart(desired, true);
      restartAttempts.remove(key);
      if (statusPublisher != null) {
        statusPublisher.publishOnline("dry-run active");
      }
      return;
    }

    startPumpLocked(key, desired);
  }

  private void startPumpLocked(IngestKey key, PipelineSpec spec) {
    try {
      applyRuntimeOptions(key);
      PumpController pump =
          pumpFactory.create(
              session,
              spec,
              error -> onPumpError(key, error),
              options.once() || spec.sowOnly() ? () -> onSowComplete(key) : null);
      pump.start();
      runningPumps.put(key, new RunningPump(spec, pump));
      restartAttempts.remove(key);
      DataStatusPublisher statusPublisher = statusPublishers.get(key);
      if (statusPublisher != null) {
        statusPublisher.putTag("restartAttempts", "0");
        statusPublisher.putTag("connectionName", spec.connectionName());
        statusPublisher.putTag("topic", spec.topic());
        statusPublisher.putTag("mode", spec.mode() == null ? null : spec.mode().name());
        statusPublisher.putTag("sow", Boolean.toString(spec.sow()));
        statusPublisher.putTag("sowOnly", Boolean.toString(spec.sowOnly()));
        statusPublisher.publishOnline("ingest pipeline running");
      }
      logStart(spec, false);
    } catch (Exception ex) {
      logError(key.ingestName(), spec.topic(), ex);
      DataStatusPublisher statusPublisher = statusPublishers.get(key);
      if (statusPublisher != null) {
        statusPublisher.publishError(
            "start failed: "
                + Strings.firstNonBlank(
                    Strings.trimToNull(ex.getMessage()), ex.getClass().getSimpleName()));
      }
      scheduleRestartLocked(key);
    }
  }

  private void applyRuntimeOptions(IngestKey key) {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    if (runtimeOptions == null) {
      return;
    }
    Integer maxRowsPerFile = runtimeOptions.parquetMaxRowsPerFile();
    Long maxIdleMillis = runtimeOptions.parquetMaxIdleMillis();
    if (maxRowsPerFile == null && maxIdleMillis == null) {
      return;
    }
    IngestManagerImpl ingestManager = resolveIngestManagerImpl(session);
    if (ingestManager == null) {
      throw new IllegalStateException(
          "Ingest runtime options require a local IngestManager implementation");
    }
    ingestManager.setParquetRollPolicy(key.ingestName(), maxRowsPerFile, maxIdleMillis);
  }

  private void onPumpError(IngestKey key, Throwable error) {
    synchronized (lock) {
      RunningPump running = runningPumps.get(key);
      if (closed.get() || running == null) {
        return;
      }
      logError(key.ingestName(), running.spec().topic(), error);
      DataStatusPublisher statusPublisher = statusPublishers.get(key);
      if (statusPublisher != null) {
        statusPublisher.publishError(
            "pipeline error: "
                + Strings.firstNonBlank(
                    Strings.trimToNull(error == null ? null : error.getMessage()),
                    error == null ? "unknown" : error.getClass().getSimpleName()));
      }
      stopRunningLocked(key, "error");
      scheduleRestartLocked(key);
    }
  }

  private void onSowComplete(IngestKey key) {
    synchronized (lock) {
      if (!runningPumps.containsKey(key) || !onceCompleted.add(key)) {
        return;
      }
      desiredSources.remove(key);
      restartAttempts.remove(key);
      stopRunningLocked(key, "sow-complete");
      closeStatusPublisherLocked(key);
    }
  }

  private void scheduleRestartLocked(IngestKey key) {
    if (closed.get() || options.dryRun() || !desiredSources.containsKey(key)) {
      return;
    }
    if (!restartInFlight.add(key)) {
      return;
    }
    retryExecutor.execute(() -> runRestartLoop(key));
  }

  private void runRestartLoop(IngestKey key) {
    try {
      while (!closed.get()) {
        long delayMs;
        synchronized (lock) {
          if (closed.get()
              || options.dryRun()
              || !desiredSources.containsKey(key)
              || runningPumps.containsKey(key)) {
            return;
          }
          int attempt = restartAttempts.getOrDefault(key, 0) + 1;
          restartAttempts.put(key, attempt);
          DataStatusPublisher statusPublisher = statusPublishers.get(key);
          if (statusPublisher != null) {
            statusPublisher.putTag("restartAttempts", Integer.toString(attempt));
          }
          delayMs = restartDelayMs(attempt);
        }

        try {
          sleeper.sleep(delayMs);
        } catch (InterruptedException ex) {
          Thread.currentThread().interrupt();
          return;
        }

        synchronized (lock) {
          if (closed.get()
              || options.dryRun()
              || !desiredSources.containsKey(key)
              || runningPumps.containsKey(key)) {
            return;
          }
          reconcileLocked(key);
          if (runningPumps.containsKey(key) || !desiredSources.containsKey(key)) {
            return;
          }
        }
      }
    } finally {
      synchronized (lock) {
        restartInFlight.remove(key);
      }
    }
  }

  private void stopRunningLocked(IngestKey key, String reason) {
    RunningPump running = runningPumps.remove(key);
    if (running == null) {
      return;
    }
    CloseSupport.closeQuietly(running.controller());
    LOGGER.info("[IngestAgent][stop] ingest={} reason={}", key.ingestName(), reason);
  }

  private PipelineSpec buildSpec(DefinitionIngestSource source, IngestKey key) {
    if (!isEnabled(source == null ? null : source.envelope())) {
      return null;
    }
    if (!options.allowIngest(key.ingestName())) {
      return null;
    }
    DefinitionIngest ingest = resolveEnabledIngest(key);
    if (ingest == null) {
      return null;
    }
    String datasetId =
        Strings.requireNonBlank(
            Strings.trimToNull(ingest.datasetId()),
            "Ingest datasetId is required: " + key.group() + "/" + key.ingestName());
    DatasetId.parse(datasetId);
    DefinitionIngestSource.AmpsSource amps = source.amps();
    if (amps == null) {
      return null;
    }

    String connectionName = Strings.trimToNull(amps.connectionName());
    String topic = Strings.trimToNull(amps.topic());
    if (connectionName == null) {
      throw new IllegalStateException(
          "IngestSource amps.connectionName is required: " + key.group() + "/" + key.ingestName());
    }
    if (topic == null) {
      throw new IllegalStateException(
          "IngestSource amps.topic is required: " + key.group() + "/" + key.ingestName());
    }

    IngestAmpsPump.PayloadFormat format;
    try {
      String normalizedFormat =
          Strings.requireNonBlank(source.format(), "ingestSource.format").toUpperCase(Locale.ROOT);
      format = IngestAmpsPump.PayloadFormat.valueOf(normalizedFormat);
    } catch (RuntimeException ex) {
      throw new IllegalStateException(
          "IngestSource format must be JSON or NVFIX: "
              + key.group()
              + "/"
              + key.ingestName()
              + " format="
              + source.format());
    }

    boolean sowOnly = Boolean.TRUE.equals(amps.sowOnly());
    boolean sow = !sowOnly && (options.once() || Boolean.TRUE.equals(amps.sow()));
    AmpsConnection.BatchMode mode =
        amps.mode() == DefinitionIngestSource.Mode.BATCH
            ? AmpsConnection.BatchMode.BATCH
            : AmpsConnection.BatchMode.STREAM;
    String connectionInstanceName = resolveAmpsConnectionInstance();
    String parquetFilenamePrefix = resolveParquetFilenamePrefix();
    boolean ampsLogIncoming = resolveAmpsLogIncoming();
    boolean ampsLogOutgoing = resolveAmpsLogOutgoing();
    UnaryOperator<String> ampsPayloadTransformer = resolveAmpsPayloadTransformer();
    BiFunction<String, Integer, String> ampsPayloadTransformerWithCommand =
        resolveAmpsPayloadTransformerWithCommand();
    boolean ampsPersistOofAsRows = resolveAmpsPersistOofAsRows();
    return new PipelineSpec(
        key.group(),
        key.ingestName(),
        format,
        connectionName,
        connectionInstanceName,
        parquetFilenamePrefix,
        ampsLogIncoming,
        ampsLogOutgoing,
        ampsPayloadTransformer,
        ampsPayloadTransformerWithCommand,
        ampsPersistOofAsRows,
        topic,
        sow,
        sowOnly,
        Strings.trimToNull(amps.options()),
        Strings.trimToNull(amps.filter()),
        datasetId,
        mode,
        sanitizePositiveInt(amps.batchMaxMessages(), AmpsConnection.DEFAULT_BATCH_MAX_MESSAGES),
        sanitizePositiveInt(amps.batchMaxBytes(), AmpsConnection.DEFAULT_BATCH_MAX_BYTES),
        sanitizePositiveLong(amps.batchFlushMillis(), AmpsConnection.DEFAULT_BATCH_FLUSH_MILLIS));
  }

  private DefinitionIngest resolveEnabledIngest(IngestKey key) {
    Optional<DefinitionIngest> ingest =
        DiscoveryCache.get(DefinitionIngest.class, key.group(), key.ingestName());
    if (ingest.isEmpty()) {
      return null;
    }
    return isEnabled(ingest.get().envelope()) ? ingest.get() : null;
  }

  private boolean isManagedGroup(String group) {
    return sessionGroup.equals(Strings.trimToNull(group));
  }

  private DataStatusPublisher ensureStatusPublisherLocked(IngestKey key, PipelineSpec spec) {
    DataStatusPublisher existing = statusPublishers.get(key);
    if (existing != null) {
      return existing;
    }
    DataStatusPublisher created = createStatusPublisher(key, spec);
    if (created == null) {
      return null;
    }
    statusPublishers.put(key, created);
    created.start();
    created.publishOnline("ingest pipeline opened");
    return created;
  }

  private void closeStatusPublisherLocked(IngestKey key) {
    DataStatusPublisher publisher = statusPublishers.remove(key);
    CloseSupport.closeQuietly(publisher);
  }

  private DataStatusPublisher createStatusPublisher(IngestKey key, PipelineSpec spec) {
    if (!(session instanceof SessionImpl sessionImpl)) {
      return null;
    }
    if (sessionImpl.controlPlaneConnection() == null || sessionImpl.context() == null) {
      return null;
    }
    SessionContext context = sessionImpl.context();
    String group = Strings.trimToNull(context.getGroup());
    String instance = Strings.trimToNull(context.getInstance());
    if (group == null || instance == null) {
      return null;
    }
    String statusName = "agent/" + key.ingestName();
    Map<String, String> tags = new HashMap<>();
    if (spec != null) {
      tags.put("datasetId", Strings.trimToNull(spec.datasetId()));
      tags.put("connectionName", Strings.trimToNull(spec.connectionName()));
      tags.put("topic", Strings.trimToNull(spec.topic()));
    }
    return new DataStatusPublisher(
        sessionImpl.controlPlaneConnection(),
        group,
        statusName,
        instance,
        DefinitionStatusData.ProducerType.INGEST_AGENT,
        "ingest-agent:" + key.group() + "/" + key.ingestName(),
        spec == null ? null : Strings.trimToNull(spec.datasetId()),
        sessionImpl.runtimePlacementHost(),
        Strings.trimToNull(context.getAppName()),
        Strings.trimToNull(context.getInstance()),
        Strings.trimToNull(context.getGroup()),
        tags,
        sessionImpl.dataStatusCadenceSeconds(),
        sessionImpl.statusScheduler());
  }

  private static boolean isEnabled(DefinitionEnvelope envelope) {
    if (envelope == null) {
      return false;
    }
    if (Boolean.FALSE.equals(envelope.enabled())) {
      return false;
    }
    String lifecycle = Strings.trimToNull(envelope.lifecycleStatus());
    return lifecycle == null || "ACTIVE".equalsIgnoreCase(lifecycle);
  }

  private static IngestKey keyOf(DefinitionEnvelope envelope) {
    if (envelope == null) {
      return null;
    }
    return keyOf(envelope.group(), envelope.name());
  }

  private static IngestKey keyOf(String group, String ingestName) {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedIngest = Strings.trimToNull(ingestName);
    if (normalizedGroup == null || normalizedIngest == null) {
      return null;
    }
    return new IngestKey(normalizedGroup, normalizedIngest);
  }

  private static long restartDelayMs(int attempt) {
    int idx = Math.max(0, attempt - 1);
    if (idx >= RESTART_BACKOFF_MILLIS.length) {
      idx = RESTART_BACKOFF_MILLIS.length - 1;
    }
    return RESTART_BACKOFF_MILLIS[idx];
  }

  private static int sanitizePositiveInt(Integer value, int defaultValue) {
    if (value == null || value <= 0) {
      return defaultValue;
    }
    return value;
  }

  private static long sanitizePositiveLong(Long value, long defaultValue) {
    if (value == null || value <= 0L) {
      return defaultValue;
    }
    return value;
  }

  private static String escapeFilterValue(String value) {
    return value == null ? "" : value.replace("'", "''");
  }


  private static void logStart(PipelineSpec spec, boolean dryRun) {
    boolean transformerEnabled =
        spec.ampsPayloadTransformer() != null || spec.ampsPayloadTransformerWithCommand() != null;
    if (dryRun) {
      LOGGER.info("[IngestAgent][start] ingest={} group={} connectionName={} connectionInstance={} topic={} format={} mode={} sowSubscribe={} sowOnly={} options={} filter={} logIncoming={} logOutgoing={} persistOofAsRows={} transformerEnabled={} dryRun=true", spec.ingestName(), spec.group(), spec.connectionName(), spec.connectionInstanceName(), spec.topic(), spec.format(), spec.mode(), spec.sow(), spec.sowOnly(), spec.options(), spec.filter(), spec.ampsLogIncoming(), spec.ampsLogOutgoing(), spec.ampsPersistOofAsRows(), transformerEnabled);
      return;
    }
    LOGGER.info("[IngestAgent][start] ingest={} group={} connectionName={} connectionInstance={} topic={} format={} mode={} sowSubscribe={} sowOnly={} options={} filter={} logIncoming={} logOutgoing={} persistOofAsRows={} transformerEnabled={}", spec.ingestName(), spec.group(), spec.connectionName(), spec.connectionInstanceName(), spec.topic(), spec.format(), spec.mode(), spec.sow(), spec.sowOnly(), spec.options(), spec.filter(), spec.ampsLogIncoming(), spec.ampsLogOutgoing(), spec.ampsPersistOofAsRows(), transformerEnabled);
  }

  private static void logError(String ingestName, String topic, Throwable error) {
    Throwable root = Exceptions.rootCause(error);
    if (root == null) {
      root = new IllegalStateException("unknown");
    }
    LOGGER.warn("[IngestAgent][error] ingest={} topic={} exceptionClass={} message={}", ingestName, topic, root.getClass().getName(), Strings.firstNonBlank(root.getMessage(), "unknown"));
  }

  private static DiscoveryCache requireConfigCache(Fabric.Session session) {
    SessionImpl impl = requireSessionImpl(session);
    DiscoveryCache cache = impl.discoveryCache();
    if (cache == null) {
      throw new IllegalStateException("Session DiscoveryCache is not available");
    }
    return cache;
  }

  private static String requireGroup(Fabric.Session session) {
    if (session == null || session.context() == null) {
      throw new IllegalStateException("Session context is not available");
    }
    return IdentityParsingSupport.requireGroup(session.context().getGroup(), "session group");
  }

  private static SessionImpl requireSessionImpl(Fabric.Session session) {
    Objects.requireNonNull(session, "session");
    if (session instanceof SessionImpl impl) {
      return impl;
    }
    throw new IllegalArgumentException(
        "IngestAgent requires SessionImpl-backed Fabric.Session");
  }

  private static ExecutorService newRetryExecutor() {
    return Executors.newCachedThreadPool(
        runnable -> {
          Thread thread = new Thread(runnable, "ingest-agent-retry");
          thread.setDaemon(true);
          return thread;
        });
  }

  interface Sleeper {
    Sleeper SYSTEM = Thread::sleep;

    void sleep(long millis) throws InterruptedException;
  }

  interface PumpFactory {
    PumpFactory DEFAULT =
        (session, spec, errorListener, sowCompleteListener) -> {
          IngestAmpsPump.Builder builder =
              IngestAmpsPump.builder(session)
                  .withIngestName(spec.ingestName())
                  .withIngestSourceName(spec.ingestName())
                  .withDatasetId(spec.datasetId())
                  .withConnectionName(spec.connectionName())
                  .withConnectionInstanceName(spec.connectionInstanceName())
                  .withParquetFilenamePrefix(spec.parquetFilenamePrefix())
                  .withLogIncomingMessages(spec.ampsLogIncoming())
                  .withLogOutgoingMessages(spec.ampsLogOutgoing())
                  .withPayloadTransformer(spec.ampsPayloadTransformer())
                  .withPayloadTransformerWithCommand(spec.ampsPayloadTransformerWithCommand())
                  .withPersistOofAsRows(spec.ampsPersistOofAsRows())
                  .withTopic(spec.topic())
                  .withSowSubscribe(spec.sow())
                  .withSow(spec.sowOnly())
                  .withOptions(spec.options())
                  .withFilter(spec.filter())
                  .withMode(spec.mode())
                  .withBatchMaxMessages(spec.batchMaxMessages())
                  .withBatchMaxBytes(spec.batchMaxBytes())
                  .withBatchFlushMillis(spec.batchFlushMillis())
                  .withPayloadFormat(spec.format())
                  .withErrorListener(errorListener)
                  .withSowCompleteListener(sowCompleteListener);
          IngestManagerImpl ingestManager = resolveIngestManagerImpl(session);
          if (ingestManager != null) {
            builder
                .withWriteAuthorizer(ingestManager::requireWrite)
                .withAmpsMessageHandler(ingestManager::appendFromAmps)
                .withCheckpointStore(ingestManager.checkpointStore());
          }
          return new IngestPumpController(builder.build());
        };

    PumpController create(
        Fabric.Session session,
        PipelineSpec spec,
        IngestAmpsPump.ErrorListener errorListener,
        Runnable sowCompleteListener);
  }

  interface PumpController extends AutoCloseable {
    void start() throws Exception;

    @Override
    void close();
  }

  private static final class IngestPumpController implements PumpController {
    private final IngestAmpsPump pump;

    private IngestPumpController(IngestAmpsPump pump) {
      this.pump = Objects.requireNonNull(pump, "pump");
    }

    @Override
    public void start() throws Exception {
      pump.start();
    }

    @Override
    public void close() {
      pump.close();
    }
  }

  private record IngestKey(String group, String ingestName) {}

  private record RunningPump(PipelineSpec spec, PumpController controller) {}

  record PipelineSpec(
      String group,
      String ingestName,
      IngestAmpsPump.PayloadFormat format,
      String connectionName,
      String connectionInstanceName,
      String parquetFilenamePrefix,
      boolean ampsLogIncoming,
      boolean ampsLogOutgoing,
      UnaryOperator<String> ampsPayloadTransformer,
      BiFunction<String, Integer, String> ampsPayloadTransformerWithCommand,
      boolean ampsPersistOofAsRows,
      String topic,
      boolean sow,
      boolean sowOnly,
      String options,
      String filter,
      String datasetId,
      AmpsConnection.BatchMode mode,
      int batchMaxMessages,
      int batchMaxBytes,
      long batchFlushMillis) {
  }

  private String resolveAmpsConnectionInstance() {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    if (runtimeOptions == null) {
      return null;
    }
    return Strings.trimToNull(runtimeOptions.ampsConnectionInstance());
  }

  private String resolveParquetFilenamePrefix() {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    if (runtimeOptions == null) {
      return null;
    }
    return Strings.trimToNull(runtimeOptions.parquetFilenamePrefix());
  }

  private boolean resolveAmpsLogIncoming() {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    return runtimeOptions != null && Boolean.TRUE.equals(runtimeOptions.ampsLogIncoming());
  }

  private boolean resolveAmpsLogOutgoing() {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    return runtimeOptions != null && Boolean.TRUE.equals(runtimeOptions.ampsLogOutgoing());
  }

  private UnaryOperator<String> resolveAmpsPayloadTransformer() {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    return runtimeOptions == null ? null : runtimeOptions.ampsPayloadTransformer();
  }

  private BiFunction<String, Integer, String> resolveAmpsPayloadTransformerWithCommand() {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    return runtimeOptions == null ? null : runtimeOptions.ampsPayloadTransformerWithCommand();
  }

  private boolean resolveAmpsPersistOofAsRows() {
    IngestManager.IngestRuntimeOptions runtimeOptions = options.ingestRuntimeOptions();
    return runtimeOptions != null && Boolean.TRUE.equals(runtimeOptions.ampsPersistOofAsRows());
  }

  private static IngestManagerImpl resolveIngestManagerImpl(Fabric.Session session) {
    if (session == null || session.data() == null) {
      return null;
    }
    DataManager direct = session.data().direct();
    IngestManager ingestManager = direct.ingest();
    if (ingestManager instanceof IngestManagerImpl local) {
      return local;
    }
    return null;
  }
