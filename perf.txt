package org.pronzato.fabric.api.configs.manager;

import com.fasterxml.jackson.databind.JsonNode;
import java.time.Instant;
import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.concurrent.RejectedExecutionException;
import java.util.concurrent.ScheduledFuture;
import java.util.concurrent.TimeUnit;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionConfigLayer;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionType;
import org.pronzato.fabric.api.configs.ConfigManager.ConfigSubscription;
import org.pronzato.fabric.api.configs.ConfigManager.ConfigUpdate;
import org.pronzato.fabric.api.configs.ConfigManager.ConfigUpdateListener;
import org.pronzato.fabric.api.configs.audit.ConfigEventSubscribed;
import org.pronzato.fabric.api.configs.audit.ConfigEventType;
import org.pronzato.fabric.api.configs.audit.ConfigEventUnsubscribed;
import org.pronzato.fabric.api.configs.audit.ConfigEventUpdateCallback;
import org.pronzato.fabric.api.configs.ConfigManager.ConfigLineage;
import org.pronzato.fabric.api.configs.model.ConfigResolved;
import org.pronzato.fabric.api.util.Strings;

final class ConfigManagerImplSubscription {
  private ConfigManagerImplSubscription() {}

  static ConfigSubscription subscribe(
      ConfigManagerImpl manager, String configId, String layerId, ConfigUpdateListener listener) {
    ConfigManagerImpl.Resolution resolved = manager.resolveId(configId, layerId);
    return subscribeInternal(manager, resolved.group(), resolved.configId(), resolved.layerId(), listener);
  }

  static ConfigSubscription subscribeInternal(
      ConfigManagerImpl manager,
      String group,
      String configId,
      String layerId,
      ConfigUpdateListener listener) {
    manager.ensureOpen();
    String requiredGroup = ConfigManagerImpl.required(group, "group");
    String requiredconfigId = ConfigManagerImpl.required(configId, "configId");
    String requiredLayerId = Strings.trimToNull(layerId);
    ConfigUpdateListener requiredListener = Objects.requireNonNull(listener, "listener");
    ConfigManagerImpl.AuditContext auditContext = manager.captureAuditContext();
    String subscriptionId = "config-sub-" + ConfigManagerImpl.SUBSCRIPTION_ID.incrementAndGet();

    ConfigManagerImpl.SubscriptionImpl subscription =
        new ConfigManagerImpl.SubscriptionImpl(
            manager,
            subscriptionId,
            requiredGroup,
            requiredconfigId,
            requiredLayerId,
            requiredListener,
            auditContext);

    AutoCloseable cacheHandle =
        manager.configView.onUpsert(
            DefinitionType.CONFIG_LAYER,
            (key, definition) -> {
              if (subscription.closed.get()) {
                return;
              }
              if (!(definition instanceof DefinitionConfigLayer config)) {
                ConfigManagerImpl.LOGGER.debug(
                    "[ConfigManager][Ignored] reason=definition-type type={} key={}",
                    key == null ? null : key.type(),
                    key);
                return;
              }
              ConfigManagerImpl.ParsedLayerIdentity identity =
                  manager.parseLayerIdentity(config, key == null ? null : key.group());
              if (identity == null) {
                ConfigManagerImpl.LOGGER.debug(
                    "[ConfigManager][Ignored] reason=identity-parse key={}", key);
                return;
              }
              if (!requiredGroup.equals(identity.group())
                  || !requiredconfigId.equals(identity.configId())) {
                ConfigManagerImpl.LOGGER.debug(
                    "[ConfigManager][Ignored] reason=filter-mismatch key={} configId={} layerId={}",
                    key,
                    identity.configId(),
                    identity.layerId());
                return;
              }
              if (requiredLayerId != null) {
                if (!layerAffectsTarget(
                    manager,
                    requiredGroup,
                    requiredconfigId,
                    requiredLayerId,
                    identity.layerId())) {
                  ConfigManagerImpl.LOGGER.debug(
                      "[ConfigManager][Ignored] reason=layer-not-in-chain key={} changedLayerId={} targetLayerId={}",
                      key,
                      identity.layerId(),
                      requiredLayerId);
                  return;
                }
                ConfigManagerImpl.LayerEntry targetLayer =
                    manager.requireLayer(requiredGroup, requiredconfigId, requiredLayerId);
                emitResolvedWithReschedule(
                    manager,
                    subscription,
                    targetLayer,
                    Instant.now(),
                    ConfigManagerImpl.CALLBACK_REASON_RESOLVED_CHAIN_CHANGE);
                return;
              }
              emitWithReschedule(
                  manager,
                  subscription,
                  config,
                  identity,
                  Instant.now(),
                  ConfigManagerImpl.CALLBACK_REASON_UPSERT);
            });

    subscription.cacheHandle = cacheHandle;
    manager.subscriptions.add(subscription);

    List<ConfigManagerImpl.LayerEntry> initialLayers =
        manager.sortLayers(manager.findConfigLayersOrEmpty(requiredGroup, requiredconfigId));
    if (requiredLayerId != null) {
      initialLayers =
          initialLayers.stream()
              .filter(entry -> requiredLayerId.equals(entry.identity().layerId()))
              .toList();
    }
    ConfigManagerImpl.LOGGER.info(
        "[ConfigManager][Subscribe] group={} configId={} layerId={} initialLayers={}",
        requiredGroup,
        requiredconfigId,
        requiredLayerId,
        initialLayers.size());
    Instant subscribedAt = Instant.now();
    ConfigLineage subscribedLineage =
        manager.resolveLineage(requiredGroup, requiredconfigId, requiredLayerId, subscribedAt);
    JsonNode subscribedChangedPayload = null;
    if (!initialLayers.isEmpty() && (requiredLayerId != null || initialLayers.size() == 1)) {
      subscribedChangedPayload =
          manager.deepCopy(manager.resolveDeclaredAt(initialLayers.getFirst().definition(), subscribedAt));
    }
    manager.publishAudit(
        new ConfigEventSubscribed(
            manager.envelope(
                ConfigEventType.CONFIG_SUBSCRIBED,
                requiredGroup,
                requiredconfigId,
                requiredLayerId,
                manager.schemaKey(initialLayers.isEmpty() ? null : initialLayers.getFirst().definition()),
                auditContext,
                subscribedAt),
            subscription.subscriptionId,
            requiredLayerId,
            subscribedLineage,
            subscribedChangedPayload));
    for (ConfigManagerImpl.LayerEntry layer : initialLayers) {
      if (requiredLayerId != null) {
        emitResolvedWithReschedule(
            manager, subscription, layer, Instant.now(), ConfigManagerImpl.CALLBACK_REASON_INITIAL);
      } else {
        emitWithReschedule(
            manager,
            subscription,
            layer.definition(),
            layer.identity(),
            Instant.now(),
            ConfigManagerImpl.CALLBACK_REASON_INITIAL);
      }
    }
    return subscription;
  }

  static void emitWithReschedule(
      ConfigManagerImpl manager,
      ConfigManagerImpl.SubscriptionImpl subscription,
      DefinitionConfigLayer config,
      ConfigManagerImpl.ParsedLayerIdentity identity,
      Instant asOfUtc,
      String callbackReason) {
    emitIfChanged(manager, subscription, config, identity, asOfUtc, callbackReason);
    scheduleTransitions(manager, subscription, config, identity);
  }

  static void emitResolvedWithReschedule(
      ConfigManagerImpl manager,
      ConfigManagerImpl.SubscriptionImpl subscription,
      ConfigManagerImpl.LayerEntry targetLayer,
      Instant asOfUtc,
      String callbackReason) {
    if (targetLayer == null) {
      return;
    }
    emitIfChanged(
        manager, subscription, targetLayer.definition(), targetLayer.identity(), asOfUtc, callbackReason);
    scheduleResolvedTransitions(
        manager, subscription, targetLayer, asOfUtc == null ? Instant.now() : asOfUtc);
  }

  static void emitIfChanged(
      ConfigManagerImpl manager,
      ConfigManagerImpl.SubscriptionImpl subscription,
      DefinitionConfigLayer config,
      ConfigManagerImpl.ParsedLayerIdentity identity,
      Instant asOfUtc,
      String callbackReason) {
    if (subscription.closed.get()) {
      return;
    }
    try {
      ConfigManagerImpl.UpdateSnapshot snapshot = manager.toUpdateSnapshotAt(config, identity, asOfUtc);
      ConfigUpdate update = snapshot.update();
      String dedupeKey = dedupeKey(update);
      String last = subscription.lastSeenByLayer.put(identity.layerId(), dedupeKey);
      if (Objects.equals(last, dedupeKey)) {
        return;
      }
      dispatch(manager, subscription, snapshot, callbackReason, last, dedupeKey);
    } catch (Exception ex) {
      ConfigManagerImpl.LOGGER.warn(
          "[ConfigManager][EmitIgnored] group={} configId={} layerId={} message={}",
          identity.group(),
          identity.configId(),
          identity.layerId(),
          ex.getMessage(),
          ex);
    }
  }

  static void scheduleTransitions(
      ConfigManagerImpl manager,
      ConfigManagerImpl.SubscriptionImpl subscription,
      DefinitionConfigLayer config,
      ConfigManagerImpl.ParsedLayerIdentity identity) {
    if (subscription.closed.get()) {
      return;
    }
    Instant now = Instant.now();
    List<Instant> transitions = transitionTimesAfter(config, now);
    subscription.cancelScheduled(identity.layerId());
    if (transitions.isEmpty()) {
      return;
    }
    for (Instant transition : transitions) {
      long delayMillis = Math.max(0L, transition.toEpochMilli() - now.toEpochMilli());
      try {
        ScheduledFuture<?> future =
            manager.scheduleExecutor.schedule(
                () -> onScheduledTransition(manager, subscription, identity, transition),
                delayMillis,
                TimeUnit.MILLISECONDS);
        subscription.addScheduled(identity.layerId(), future);
      } catch (RejectedExecutionException ignore) {
        // Session/client is closing; ignore.
      }
    }
  }

  static void scheduleResolvedTransitions(
      ConfigManagerImpl manager,
      ConfigManagerImpl.SubscriptionImpl subscription,
      ConfigManagerImpl.LayerEntry targetLayer,
      Instant now) {
    if (subscription.closed.get() || targetLayer == null) {
      return;
    }
    ConfigManagerImpl.ParsedLayerIdentity identity = targetLayer.identity();
    List<Instant> transitions = transitionTimesAfterResolvedChain(manager, targetLayer, now);
    subscription.cancelScheduled(identity.layerId());
    if (transitions.isEmpty()) {
      return;
    }
    for (Instant transition : transitions) {
      long delayMillis = Math.max(0L, transition.toEpochMilli() - now.toEpochMilli());
      try {
        ScheduledFuture<?> future =
            manager.scheduleExecutor.schedule(
                () -> onScheduledTransition(manager, subscription, identity, transition),
                delayMillis,
                TimeUnit.MILLISECONDS);
        subscription.addScheduled(identity.layerId(), future);
      } catch (RejectedExecutionException ignore) {
        // Session/client is closing; ignore.
      }
    }
  }

  static void onScheduledTransition(
      ConfigManagerImpl manager,
      ConfigManagerImpl.SubscriptionImpl subscription,
      ConfigManagerImpl.ParsedLayerIdentity identity,
      Instant transitionUtc) {
    if (subscription.closed.get() || manager.closed.get()) {
      return;
    }
    try {
      ConfigManagerImpl.LayerEntry layer =
          manager.requireLayer(identity.group(), identity.configId(), identity.layerId());
      emitIfChanged(
          manager,
          subscription,
          layer.definition(),
          layer.identity(),
          transitionUtc,
          ConfigManagerImpl.CALLBACK_REASON_SCHEDULED);
    } catch (IllegalArgumentException ignored) {
      // Config/layer disappeared between scheduling and execution; ignore.
    } finally {
      subscription.pruneScheduled(identity.layerId());
    }
  }

  static void dispatch(
      ConfigManagerImpl manager,
      ConfigManagerImpl.SubscriptionImpl subscription,
      ConfigManagerImpl.UpdateSnapshot snapshot,
      String callbackReason,
      String previousHash,
      String newHash) {
    ConfigUpdate update = snapshot.update();
    if (subscription.closed.get() || manager.closed.get()) {
      return;
    }
    try {
      manager.callbackExecutor.execute(
          () -> {
            if (subscription.closed.get() || manager.closed.get()) {
              return;
            }
            try {
              manager.publishAudit(
                  new ConfigEventUpdateCallback(
                      manager.envelope(
                          ConfigEventType.CONFIG_UPDATE_CALLBACK,
                          update.group(),
                          update.configName(),
                          update.layerId(),
                          update.schemaKey(),
                          subscription.auditContext,
                          Instant.now()),
                      subscription.subscriptionId,
                      callbackReason,
                      previousHash,
                      newHash,
                      update.configId(),
                      update.effectiveConfigId(),
                      update.updatedTimeUtc(),
                      snapshot.lineage(),
                      snapshot.changedPayload()));
              subscription.callbackCount.incrementAndGet();
              subscription.listener.onUpdate(update);
            } catch (Exception ex) {
              ConfigManagerImpl.LOGGER.warn(
                  "[ConfigManager][CallbackFailed] group={} configId={} layerId={} message={}",
                  update.group(),
                  update.configName(),
                  update.layerId(),
                  ex.getMessage(),
                  ex);
            }
          });
    } catch (RejectedExecutionException ignore) {
      // Session/client is closing; ignore.
    }
  }

  static boolean layerAffectsTarget(
      ConfigManagerImpl manager,
      String group,
      String configId,
      String targetLayerId,
      String changedLayerId) {
    if (targetLayerId == null || changedLayerId == null) {
      return false;
    }
    if (targetLayerId.equals(changedLayerId)) {
      return true;
    }
    try {
      if (manager.resolver == null) {
        return false;
      }
      ConfigResolved resolved = manager.resolver.resolve(group, configId, targetLayerId, Instant.now());
      ConfigLineage lineage = resolved == null ? null : resolved.lineage();
      List<String> resolvedLayers = lineage == null ? List.of() : lineage.resolvedLayers();
      return resolvedLayers.contains(changedLayerId);
    } catch (RuntimeException ex) {
      ConfigManagerImpl.LOGGER.debug(
          "[ConfigManager][ChainIgnored] group={} configId={} targetLayerId={} changedLayerId={} message={}",
          group,
          configId,
          targetLayerId,
          changedLayerId,
          ex.getMessage());
      return false;
    }
  }

  static List<Instant> transitionTimesAfterResolvedChain(
      ConfigManagerImpl manager, ConfigManagerImpl.LayerEntry targetLayer, Instant now) {
    if (targetLayer == null) {
      return List.of();
    }
    if (manager.resolver == null) {
      return transitionTimesAfter(targetLayer.definition(), now);
    }
    ConfigResolved resolved =
        manager.resolver.resolve(
            targetLayer.identity().group(),
            targetLayer.identity().configId(),
            targetLayer.identity().layerId(),
            now);
    List<String> resolvedLayers =
        resolved == null || resolved.lineage() == null || resolved.lineage().resolvedLayers() == null
            ? List.of()
            : resolved.lineage().resolvedLayers();
    if (resolvedLayers.isEmpty()) {
      return List.of();
    }
    Map<String, ConfigManagerImpl.LayerEntry> byLayerId = new LinkedHashMap<>();
    for (ConfigManagerImpl.LayerEntry entry :
        manager.findConfigLayers(targetLayer.identity().group(), targetLayer.identity().configId())) {
      byLayerId.put(entry.identity().layerId(), entry);
    }
    List<Instant> times = new ArrayList<>();
    for (String layerId : resolvedLayers) {
      ConfigManagerImpl.LayerEntry entry = byLayerId.get(layerId);
      if (entry == null) {
        continue;
      }
      times.addAll(transitionTimesAfter(entry.definition(), now));
    }
    if (times.isEmpty()) {
      return List.of();
    }
    times.sort(Instant::compareTo);
    List<Instant> distinct = new ArrayList<>(times.size());
    Instant last = null;
    for (Instant time : times) {
      if (!Objects.equals(last, time)) {
        distinct.add(time);
        last = time;
      }
    }
    return List.copyOf(distinct);
  }

  static List<Instant> transitionTimesAfter(DefinitionConfigLayer definition, Instant now) {
    DefinitionConfigLayer payload =
        definition == null  ? null : definition;
    List<DefinitionConfigLayer.Effective> effectiveEntries = payload == null ? null : payload.effective();
    if (effectiveEntries == null || effectiveEntries.isEmpty()) {
      return List.of();
    }
    List<Instant> times = new ArrayList<>();
    for (DefinitionConfigLayer.Effective entry : effectiveEntries) {
      if (entry == null) {
        continue;
      }
      String effectiveTimeUtc = Strings.trimToNull(entry.effectiveTimeUtc());
      if (effectiveTimeUtc != null) {
        try {
          Instant effectiveAt = Instant.parse(effectiveTimeUtc);
          if (effectiveAt.isAfter(now)) {
            times.add(effectiveAt);
          }
        } catch (Exception ex) {
          ConfigManagerImpl.LOGGER.debug(
              "[ConfigManager][ScheduleIgnored] reason=invalid-effectiveTimeUtc value={}",
              effectiveTimeUtc);
        }
      }
      String expiresAtUtc = Strings.trimToNull(entry.expiresAtUtc());
      if (expiresAtUtc != null) {
        try {
          Instant expiresAt = Instant.parse(expiresAtUtc);
          if (expiresAt.isAfter(now)) {
            times.add(expiresAt);
          }
        } catch (Exception ex) {
          ConfigManagerImpl.LOGGER.debug(
              "[ConfigManager][ScheduleIgnored] reason=invalid-expiresAtUtc value={}",
              expiresAtUtc);
        }
      }
    }
    if (times.isEmpty()) {
      return List.of();
    }
    times.sort(Instant::compareTo);
    List<Instant> distinct = new ArrayList<>(times.size());
    Instant last = null;
    for (Instant time : times) {
      if (!Objects.equals(last, time)) {
        distinct.add(time);
        last = time;
      }
    }
    return List.copyOf(distinct);
  }

  static String dedupeKey(ConfigUpdate update) {
    JsonNode declared = update.declared();
    return declared == null ? "" : ConfigManagerImplAudit.hashJson(ConfigManagerImpl.HASH_MAPPER, declared);
  }

  static void closeSubscription(ConfigManagerImpl.SubscriptionImpl subscription) {
    if (!subscription.closed.compareAndSet(false, true)) {
      return;
    }
    ConfigManagerImpl manager = subscription.manager;
    manager.subscriptions.remove(subscription);
    AutoCloseable handle = subscription.cacheHandle;
    subscription.cacheHandle = null;
    if (handle != null) {
      try {
        handle.close();
      } catch (Exception ignore) {
      }
    }
    for (String layer : subscription.scheduledByLayer.keySet()) {
      subscription.cancelScheduled(layer);
    }
    manager.publishAudit(
        new ConfigEventUnsubscribed(
            manager.envelope(
                ConfigEventType.CONFIG_UNSUBSCRIBED,
                subscription.group,
                subscription.configId,
                subscription.layerId,
                null,
                subscription.auditContext,
                Instant.now()),
            subscription.subscriptionId));
    ConfigManagerImpl.LOGGER.info(
        "[ConfigManager][Unsubscribe] subscriptionId={} group={} configId={} layerId={} callbacks={}",
        subscription.subscriptionId,
        subscription.group,
        subscription.configId,
        subscription.layerId,
        subscription.callbackCount.get());
  }
}
