package org.pronzato.fabric.api.data.format;

import com.fasterxml.jackson.databind.JsonNode;
import java.time.Instant;
import java.time.LocalDate;
import java.time.format.DateTimeParseException;
import java.util.ArrayList;
import java.util.Collections;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import java.util.TreeSet;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionIngest;
import org.pronzato.fabric.api.data.format.FormatDecoderArrowBuilder.FormatField;
import org.pronzato.fabric.api.data.format.FormatDecoderArrowBuilder.ValueKind;
import org.pronzato.fabric.api.util.Strings;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

final class SchemaEvolution {
  private static final Logger LOGGER = LoggerFactory.getLogger(SchemaEvolution.class);

  private SchemaEvolution() {}

  static Result resolveJson(
      String ingestId,
      DefinitionIngest.SchemaMode schemaMode,
      List<FormatField> expectedFields,
      List<Map<String, JsonNode>> inputRows) {
    return resolve(ingestId, schemaMode, expectedFields, inputRows, JsonAdapter.INSTANCE);
  }

  static Result resolveNvFix(
      String ingestId,
      DefinitionIngest.SchemaMode schemaMode,
      List<FormatField> expectedFields,
      List<Map<String, String>> inputRows) {
    return resolve(ingestId, schemaMode, expectedFields, inputRows, NvFixAdapter.INSTANCE);
  }

  private static <V> Result resolve(
      String ingestId,
      DefinitionIngest.SchemaMode schemaMode,
      List<FormatField> expectedFields,
      List<Map<String, V>> inputRows,
      ValueAdapter<V> adapter) {
    String normalizedIngestId = Strings.requireNonBlank(ingestId, "ingestId");
    DefinitionIngest.SchemaMode mode =
        schemaMode == null ? DefinitionIngest.SchemaMode.STRICT : schemaMode;
    Objects.requireNonNull(adapter, "adapter");

    if (expectedFields == null || expectedFields.isEmpty()) {
      throw new IllegalArgumentException("expectedFields are required");
    }
    if (inputRows == null || inputRows.isEmpty()) {
      throw new IllegalArgumentException(
          "Format decode payload did not contain any messages ingest=" + normalizedIngestId);
    }

    Map<String, FormatField> expectedByName = new LinkedHashMap<>();
    for (FormatField expectedField : expectedFields) {
      expectedByName.put(expectedField.name(), expectedField);
    }

    TreeSet<String> discoveredExtras = new TreeSet<>();
    TreeSet<String> observedInputFields = new TreeSet<>();
    for (Map<String, V> row : inputRows) {
      if (row == null) {
        continue;
      }
      for (String fieldName : row.keySet()) {
        observedInputFields.add(fieldName);
        if (!expectedByName.containsKey(fieldName)) {
          discoveredExtras.add(fieldName);
        }
      }
    }

    if (mode == DefinitionIngest.SchemaMode.STRICT && !discoveredExtras.isEmpty()) {
      throw new IllegalArgumentException(
          "Unexpected field(s) "
              + discoveredExtras
              + " ingest="
              + normalizedIngestId
              + " mode="
              + mode);
    }

    List<String> addedColumns = new ArrayList<>();
    List<FormatField> addedFields = new ArrayList<>();
    if (mode != DefinitionIngest.SchemaMode.STRICT) {
      int index = expectedFields.size();
      for (String fieldName : discoveredExtras) {
        ValueKind inferredKind = inferAddedValueKind(fieldName, inputRows, adapter);
        addedFields.add(
            new FormatField(
                index,
                fieldName,
                true,
                inferredKind.expectedTypeLabel(),
                inferredKind));
        addedColumns.add(fieldName);
        index++;
      }
    }

    List<FormatField> outputFields = new ArrayList<>(expectedFields.size() + addedFields.size());
    outputFields.addAll(expectedFields);
    outputFields.addAll(addedFields);

    TreeSet<String> defaultedFields = new TreeSet<>();
    TreeSet<String> widenedFields = new TreeSet<>();
    List<Object[]> outputRows = new ArrayList<>(inputRows.size());
    for (Map<String, V> inputRow : inputRows) {
      Map<String, V> row = inputRow == null ? Collections.emptyMap() : inputRow;
      Object[] values = new Object[outputFields.size()];

      for (FormatField field : expectedFields) {
        V rawValue = row.get(field.name());
        if (rawValue == null || adapter.isNull(rawValue)) {
          values[field.index()] =
              resolveMissingRequired(normalizedIngestId, mode, field, defaultedFields);
          continue;
        }
        ValueKind actualKind = adapter.inferActualKind(rawValue);
        if (mode == DefinitionIngest.SchemaMode.RELAXED
            && actualKind != null
            && actualKind != field.valueKind()
            && isAllowedWidening(actualKind, field.valueKind())) {
          widenedFields.add(field.name());
        }
        values[field.index()] =
            convertRequiredValue(normalizedIngestId, mode, field, rawValue, adapter);
      }

      for (FormatField addedField : addedFields) {
        V rawValue = row.get(addedField.name());
        if (rawValue == null || adapter.isNull(rawValue)) {
          values[addedField.index()] = null;
          continue;
        }
        values[addedField.index()] =
            convertAddedValue(normalizedIngestId, mode, addedField, rawValue, adapter);
      }

      outputRows.add(values);
    }

    if (mode != DefinitionIngest.SchemaMode.STRICT
        && (!addedColumns.isEmpty() || !defaultedFields.isEmpty() || !widenedFields.isEmpty())) {
      LOGGER.info("ingest={} mode={} addedColumns={} defaultedFields={} widenedFields={}", normalizedIngestId, mode, addedColumns, defaultedFields, widenedFields);
    }

    return new Result(
        List.copyOf(outputFields),
        List.copyOf(outputRows),
        List.copyOf(addedColumns),
        List.copyOf(defaultedFields),
        List.copyOf(widenedFields),
        List.copyOf(observedInputFields));
  }

  private static <V> Object resolveMissingRequired(
      String ingestId,
      DefinitionIngest.SchemaMode mode,
      FormatField field,
      Set<String> defaultedFields) {
    if (field.nullable()) {
      return null;
    }
    if (mode != DefinitionIngest.SchemaMode.RELAXED) {
      throw missingRequired(ingestId, mode, field.name());
    }
    Object defaultValue = defaultValueFor(field.valueKind());
    if (defaultValue == null) {
      throw missingRequired(ingestId, mode, field.name());
    }
    defaultedFields.add(field.name());
    return defaultValue;
  }

  private static Object defaultValueFor(ValueKind valueKind) {
    return switch (valueKind) {
      case UTF8 -> "";
      case INT32, DATE_DAY -> 0;
      case INT64, TIMESTAMP_MICROSECOND -> 0L;
      case FLOAT64 -> 0.0d;
      case BOOL -> false;
    };
  }

  private static <V> Object convertRequiredValue(
      String ingestId,
      DefinitionIngest.SchemaMode mode,
      FormatField field,
      V rawValue,
      ValueAdapter<V> adapter) {
    try {
      return adapter.convert(rawValue, field, ingestId);
    } catch (RuntimeException ex) {
      throw typeMismatch(
          ingestId, mode, field.name(), field.expectedType(), adapter.actualTypeLabel(rawValue), ex);
    }
  }

  private static <V> Object convertAddedValue(
      String ingestId,
      DefinitionIngest.SchemaMode mode,
      FormatField field,
      V rawValue,
      ValueAdapter<V> adapter) {
    try {
      return adapter.convert(rawValue, field, ingestId);
    } catch (RuntimeException ex) {
      if (field.valueKind() == ValueKind.UTF8) {
        return adapter.stringify(rawValue);
      }
      throw typeMismatch(
          ingestId, mode, field.name(), field.expectedType(), adapter.actualTypeLabel(rawValue), ex);
    }
  }

  private static boolean isAllowedWidening(ValueKind actual, ValueKind expected) {
    if (actual == ValueKind.INT32 && expected == ValueKind.INT64) {
      return true;
    }
    if ((actual == ValueKind.INT32 || actual == ValueKind.INT64) && expected == ValueKind.FLOAT64) {
      return true;
    }
    return isNumeric(actual) && expected == ValueKind.UTF8;
  }

  private static boolean isNumeric(ValueKind valueKind) {
    return valueKind == ValueKind.INT32
        || valueKind == ValueKind.INT64
        || valueKind == ValueKind.FLOAT64;
  }

  private static <V> ValueKind inferAddedValueKind(
      String fieldName, List<Map<String, V>> inputRows, ValueAdapter<V> adapter) {
    ValueKind merged = null;
    boolean unstable = false;
    for (Map<String, V> inputRow : inputRows) {
      if (inputRow == null) {
        continue;
      }
      V raw = inputRow.get(fieldName);
      if (raw == null || adapter.isNull(raw)) {
        continue;
      }
      ValueKind inferred = adapter.inferAddedKind(raw);
      if (inferred == null) {
        unstable = true;
        break;
      }
      if (merged == null) {
        merged = inferred;
        continue;
      }
      merged = mergeKinds(merged, inferred);
      if (merged == null) {
        unstable = true;
        break;
      }
    }
    if (unstable || merged == null) {
      return ValueKind.UTF8;
    }
    return merged;
  }

  private static ValueKind mergeKinds(ValueKind left, ValueKind right) {
    if (left == right) {
      return left;
    }
    if (isNumeric(left) && isNumeric(right)) {
      if (left == ValueKind.FLOAT64 || right == ValueKind.FLOAT64) {
        return ValueKind.FLOAT64;
      }
      if (left == ValueKind.INT64 || right == ValueKind.INT64) {
        return ValueKind.INT64;
      }
      return ValueKind.INT32;
    }
    return null;
  }

  private static IllegalArgumentException missingRequired(
      String ingestId, DefinitionIngest.SchemaMode mode, String fieldName) {
    return new IllegalArgumentException(
        "Missing required field '" + fieldName + "' ingest=" + ingestId + " mode=" + mode);
  }

  private static IllegalArgumentException typeMismatch(
      String ingestId,
      DefinitionIngest.SchemaMode mode,
      String fieldName,
      String expectedType,
      String actualType) {
    return typeMismatch(ingestId, mode, fieldName, expectedType, actualType, null);
  }

  private static IllegalArgumentException typeMismatch(
      String ingestId,
      DefinitionIngest.SchemaMode mode,
      String fieldName,
      String expectedType,
      String actualType,
      Throwable cause) {
    String message =
        "Type mismatch for field '"
            + fieldName
            + "' expected="
            + expectedType
            + " actual="
            + actualType
            + " ingest="
            + ingestId
            + " mode="
            + mode;
    return cause == null ? new IllegalArgumentException(message) : new IllegalArgumentException(message, cause);
  }

  private interface ValueAdapter<V> {
    boolean isNull(V value);

    ValueKind inferActualKind(V value);

    ValueKind inferAddedKind(V value);

    Object convert(V value, FormatField targetField, String ingestId);

    String stringify(V value);

    String actualTypeLabel(V value);
  }

  private enum JsonAdapter implements ValueAdapter<JsonNode> {
    INSTANCE;

    @Override
    public boolean isNull(JsonNode value) {
      return value == null || value.isNull();
    }

    @Override
    public ValueKind inferActualKind(JsonNode value) {
      if (value == null || value.isNull()) {
        return null;
      }
      if (value.isBoolean()) {
        return ValueKind.BOOL;
      }
      if (value.isIntegralNumber()) {
        long numeric = value.longValue();
        if (numeric >= Integer.MIN_VALUE && numeric <= Integer.MAX_VALUE) {
          return ValueKind.INT32;
        }
        return ValueKind.INT64;
      }
      if (value.isNumber()) {
        return ValueKind.FLOAT64;
      }
      if (value.isTextual()) {
        return ValueKind.UTF8;
      }
      return null;
    }

    @Override
    public ValueKind inferAddedKind(JsonNode value) {
      return inferActualKind(value);
    }

    @Override
    public Object convert(JsonNode value, FormatField targetField, String ingestId) {
      return FormatDecoderConversions.fromJson(value, targetField, ingestId);
    }

    @Override
    public String stringify(JsonNode value) {
      if (value == null || value.isNull()) {
        return null;
      }
      if (value.isTextual()) {
        return value.textValue();
      }
      return value.toString();
    }

    @Override
    public String actualTypeLabel(JsonNode value) {
      if (value == null || value.isNull()) {
        return "Null";
      }
      ValueKind kind = inferActualKind(value);
      if (kind != null) {
        return kind.expectedTypeLabel();
      }
      if (value.isArray()) {
        return "Array";
      }
      if (value.isObject()) {
        return "Object";
      }
      return "Unknown";
    }
  }

  private enum NvFixAdapter implements ValueAdapter<String> {
    INSTANCE;

    @Override
    public boolean isNull(String value) {
      return value == null;
    }

    @Override
    public ValueKind inferActualKind(String value) {
      return inferFromString(value);
    }

    @Override
    public ValueKind inferAddedKind(String value) {
      return inferFromString(value);
    }

    @Override
    public Object convert(String value, FormatField targetField, String ingestId) {
      return FormatDecoderConversions.fromString(value, targetField, ingestId);
    }

    @Override
    public String stringify(String value) {
      return value;
    }

    @Override
    public String actualTypeLabel(String value) {
      ValueKind inferred = inferFromString(value);
      if (inferred != null) {
        return inferred.expectedTypeLabel();
      }
      return "Utf8";
    }

    private static ValueKind inferFromString(String value) {
      String normalized = Strings.trimToNull(value);
      if (normalized == null) {
        return null;
      }
      if ("true".equalsIgnoreCase(normalized) || "false".equalsIgnoreCase(normalized)) {
        return ValueKind.BOOL;
      }
      if (looksLikeTimestamp(normalized)) {
        Long ignored = tryParseTimestampMicros(normalized);
        if (ignored != null) {
          return ValueKind.TIMESTAMP_MICROSECOND;
        }
      }
      if (looksLikeDate(normalized)) {
        Integer ignored = tryParseDateEpochDay(normalized);
        if (ignored != null) {
          return ValueKind.DATE_DAY;
        }
      }
      Long integerValue = tryParseLong(normalized);
      if (integerValue != null) {
        if (integerValue >= Integer.MIN_VALUE && integerValue <= Integer.MAX_VALUE) {
          return ValueKind.INT32;
        }
        return ValueKind.INT64;
      }
      Double floatingValue = tryParseDouble(normalized);
      if (floatingValue != null) {
        return ValueKind.FLOAT64;
      }
      return ValueKind.UTF8;
    }

    private static boolean looksLikeDate(String value) {
      return value.contains("-") && value.length() >= 8;
    }

    private static boolean looksLikeTimestamp(String value) {
      return value.contains("T") || value.endsWith("Z");
    }

    private static Integer tryParseDateEpochDay(String value) {
      try {
        LocalDate date = LocalDate.parse(value);
        return Math.toIntExact(date.toEpochDay());
      } catch (RuntimeException ex) {
        return null;
      }
    }

    private static Long tryParseTimestampMicros(String value) {
      try {
        Instant instant = Instant.parse(value);
        return Math.addExact(
            Math.multiplyExact(instant.getEpochSecond(), 1_000_000L), instant.getNano() / 1_000L);
      } catch (DateTimeParseException | ArithmeticException ex) {
        return null;
      }
    }

    private static Long tryParseLong(String value) {
      try {
        return Long.parseLong(value);
      } catch (NumberFormatException ex) {
        return null;
      }
    }

    private static Double tryParseDouble(String value) {
      try {
        return Double.parseDouble(value);
      } catch (NumberFormatException ex) {
        return null;
      }
    }
  }

  record Result(
      List<FormatField> outputFields,
      List<Object[]> outputRows,
      List<String> addedColumns,
      List<String> defaultedFields,
      List<String> widenedFields,
      List<String> observedInputFields) {}
}
