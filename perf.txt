package org.pronzato.fabric.api.configs.manager;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.NullNode;
import java.security.MessageDigest;
import java.util.ArrayList;
import java.util.List;
import java.util.Locale;
import java.util.Map;
import java.util.UUID;
import java.util.concurrent.atomic.AtomicLong;
import org.pronzato.fabric.api.configs.audit.ConfigEvent;
import org.pronzato.fabric.api.configs.audit.ConfigPublisher;
import org.pronzato.fabric.api.util.Strings;
import org.slf4j.Logger;

final class ConfigManagerImplAudit {
  private ConfigManagerImplAudit() {}

  static String resolveAuditSessionId(String sessionId, String callerAppName, String callerAppInstance) {
    String normalized = Strings.trimToNull(sessionId);
    if (normalized != null) {
      return normalized;
    }
    String app = Strings.firstNonBlank(Strings.trimToNull(callerAppName), "unknown-app");
    String instance = Strings.firstNonBlank(Strings.trimToNull(callerAppInstance), "default");
    return "app:" + app + "|" + instance + "|" + UUID.randomUUID();
  }

  static String resolveDelegatedUser(Map<String, Object> attributes) {
    if (attributes == null || attributes.isEmpty()) {
      return null;
    }
    return Strings.firstNonBlank(
        asString(attributes.get("delegatedUser")),
        asString(attributes.get("delegatedUserSubject")));
  }

  static String nextAuditRequestId(AtomicLong auditRequestSequence, String sessionId) {
    long value = auditRequestSequence.incrementAndGet();
    String prefix = Strings.firstNonBlank(sessionId, "config-audit");
    return prefix + ":" + value;
  }

  static void publishAudit(ConfigPublisher auditPublisher, Logger logger, ConfigEvent event) {
    try {
      auditPublisher.publish(event);
    } catch (Exception ex) {
      logger.debug(
          "[ConfigUsageAudit] publish ignored eventType={} message={}",
          event == null || event.envelope() == null ? null : event.envelope().eventType(),
          ex.toString());
    }
  }

  static String hashJson(ObjectMapper hashMapper, JsonNode node) {
    try {
      JsonNode canonical = canonicalize(hashMapper, node);
      byte[] payload = hashMapper.writeValueAsBytes(canonical);
      MessageDigest digest = MessageDigest.getInstance("SHA-256");
      return hex(digest.digest(payload));
    } catch (Exception ex) {
      throw new IllegalStateException("failed to hash config payload", ex);
    }
  }

  static JsonNode redactSensitiveAuditPayload(ObjectMapper mapper, JsonNode payload) {
    if (payload == null || payload.isNull() || payload.isMissingNode()) {
      return payload;
    }
    if (payload.isObject()) {
      com.fasterxml.jackson.databind.node.ObjectNode object = mapper.createObjectNode();
      payload
          .fields()
          .forEachRemaining(
              entry -> {
                String key = entry == null ? null : entry.getKey();
                JsonNode value = entry == null ? null : entry.getValue();
                if (isSensitiveAuditField(key)) {
                  object.put(Strings.firstNonBlank(Strings.trimToNull(key), "field"), "<redacted>");
                } else {
                  object.set(
                      Strings.firstNonBlank(Strings.trimToNull(key), "field"),
                      redactSensitiveAuditPayload(mapper, value));
                }
              });
      return object;
    }
    if (payload.isArray()) {
      com.fasterxml.jackson.databind.node.ArrayNode array = mapper.createArrayNode();
      for (JsonNode child : payload) {
        array.add(redactSensitiveAuditPayload(mapper, child));
      }
      return array;
    }
    return payload.deepCopy();
  }

  static JsonNode resolvedPayloadAsJsonNode(ObjectMapper mapper, Object payload) {
    if (payload == null) {
      return NullNode.getInstance();
    }
    if (payload instanceof JsonNode node) {
      return node;
    }
    return mapper.valueToTree(payload);
  }

  private static JsonNode canonicalize(ObjectMapper hashMapper, JsonNode node) {
    if (node == null || node.isMissingNode()) {
      return hashMapper.nullNode();
    }
    if (node.isObject()) {
      com.fasterxml.jackson.databind.node.ObjectNode sorted = hashMapper.createObjectNode();
      List<String> keys = new ArrayList<>();
      node.fieldNames().forEachRemaining(keys::add);
      keys.sort(String::compareTo);
      for (String key : keys) {
        sorted.set(key, canonicalize(hashMapper, node.get(key)));
      }
      return sorted;
    }
    if (node.isArray()) {
      com.fasterxml.jackson.databind.node.ArrayNode array = hashMapper.createArrayNode();
      for (JsonNode child : node) {
        array.add(canonicalize(hashMapper, child));
      }
      return array;
    }
    return node.deepCopy();
  }

  private static String hex(byte[] bytes) {
    if (bytes == null || bytes.length == 0) {
      return "";
    }
    StringBuilder out = new StringBuilder(bytes.length * 2);
    for (byte value : bytes) {
      out.append(String.format("%02x", value & 0xff));
    }
    return out.toString();
  }

  private static boolean isSensitiveAuditField(String key) {
    String normalized = Strings.trimToNull(key);
    if (normalized == null) {
      return false;
    }
    normalized = normalized.toLowerCase(Locale.ROOT).replaceAll("[^a-z0-9]", "");
    return normalized.contains("apikey")
        || normalized.contains("accesskey")
        || normalized.contains("secret")
        || normalized.contains("token")
        || normalized.contains("password")
        || normalized.contains("credential")
        || normalized.contains("privatekey")
        || normalized.contains("passwd");
  }

  private static String asString(Object value) {
    if (value == null) {
      return null;
    }
    return Strings.trimToNull(String.valueOf(value));
  }
}
