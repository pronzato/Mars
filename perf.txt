final class SessionBootstrap {
  private static final String PLATFORM_GROUP = "platform";
  private static final String ENVIRONMENT_NAME = "environment";
  private static final String DISCOVERY_CONNECTION_NAME = "amps-fabric-discovery";

  private SessionBootstrap() {}

  static BootstrapResult bootstrap(
      SessionImpl.SessionConfig cfg, String sessionId, SessionVmArgs existingKerberosScope)
      throws Exception {
    Objects.requireNonNull(cfg, "cfg");
    String normalizedGroup = Strings.trimToNull(cfg.group());
    if (normalizedGroup == null) {
      throw new IllegalStateException("CONFIG_GROUP_MISSING: group is required");
    }
    String normalizedAppName = Strings.trimToNull(cfg.appName());
    if (normalizedAppName == null) {
      throw new IllegalStateException("appName is required");
    }
    List<String> discovery = normalizeDiscoveryUrls(cfg.discoveryUrls());
    if (discovery.isEmpty()) {
      throw new IllegalStateException("discoveryUrls are required");
    }
    String resolvedInstance = resolveInstanceName(cfg.instance());
    List<String> discoveryUris = toDiscoveryUris(discovery);

    AmpsFactoryOptions options = AmpsFactoryOptions.fromSystemProperties();
    if (options == null) {
      options = new AmpsFactoryOptions();
    }
    configureKerberos(cfg, options);
    options.kerberos().cacheKeyPrefix("session:" + sessionId);

    SessionVmArgs kerberosScope = existingKerberosScope;
    boolean createdKerberosScope = false;
    if (kerberosScope == null) {
      kerberosScope = SessionVmArgs.applyKerberosOptions(options.kerberos());
      createdKerberosScope = kerberosScope != null;
    }

    HAClient createdClient = null;
    try {
      Placement placement;
      List<String> cpUris;
      boolean configCacheAdmin;
      DefinitionApp appDefinition = null;
      DefinitionEnvironment environmentDefinition = null;
      HAClient tempClient = null;
      try {
        String tmpClientId = normalizedAppName + "-" + resolvedInstance + "-" + java.util.UUID.randomUUID();
        tempClient = AmpsFactory.create(tmpClientId, options);
        AmpsFactory.connect(tempClient, discoveryUris, options);
        AmpsFactory.logon(tempClient, options);

        appDefinition =
            requireAppSnapshot(
                sowGetConfigRecord(tempClient, normalizedGroup, DefinitionType.APP, normalizedAppName),
                normalizedGroup,
                normalizedAppName,
                tempClient);
        if (Boolean.FALSE.equals(appDefinition.envelope().enabled())) {
          throw new IllegalStateException(
              "App "
                  + normalizedGroup
                  + "/"
                  + normalizedAppName
                  + " is disabled in metadata; refusing to start");
        }
        configCacheAdmin = Boolean.TRUE.equals(appDefinition.configCacheAdmin());

        environmentDefinition =
            requireEnvironmentRegistry(
                sowGetConfigRecord(
                    tempClient,
                    PLATFORM_GROUP,
                    DefinitionType.ENVIRONMENT,
                    ENVIRONMENT_NAME));

        DefinitionConnection discoveryConnection =
            requireDiscoveryConnection(
                sowGetConfigRecord(
                    tempClient,
                    PLATFORM_GROUP,
                    DefinitionType.CONNECTION,
                    DISCOVERY_CONNECTION_NAME));

        cpUris = resolveDiscoveryAmpsUris(discoveryConnection);
        placement = resolvePlacement(environmentDefinition, discovery);
      } finally {
        closeClient(tempClient);
      }

      String cpClientId =
          AmpsConfig.buildCpClientId(normalizedGroup, normalizedAppName, resolvedInstance);
      createdClient = AmpsFactory.create(cpClientId, options);
      AmpsFactory.connect(createdClient, cpUris, options);
      AmpsFactory.logon(createdClient, options);
      ControlConnection controlPlaneConnection =
          ControlConnection.of(createdClient, cpClientId);

      return new BootstrapResult(
          normalizedGroup,
          normalizedAppName,
          resolvedInstance,
          List.copyOf(discovery),
          placement.environment(),
          placement.region(),
          placement.site(),
          List.copyOf(cpUris),
          controlPlaneConnection,
          configCacheAdmin,
          environmentDefinition,
          appDefinition,
          kerberosScope);
    } catch (Exception ex) {
      closeClient(createdClient);
      if (createdKerberosScope) {
        CloseSupport.closeQuietly(kerberosScope);
      }
      throw ex;
    }
  }

  static String resolveInstanceName(String requested) {
    String trimmed = Strings.trimToNull(requested);
    return trimmed == null ? "instance1" : trimmed;
  }

  static BootstrapResult bootstrapFromRecoveredSnapshot(
      SessionImpl.SessionConfig cfg,
      String sessionId,
      SessionVmArgs kerberosPropertyScope,
      DiscoveryCache recoveredCache) {
    Objects.requireNonNull(cfg, "cfg");
    DiscoveryCache cache = Objects.requireNonNull(recoveredCache, "recoveredCache");

    String normalizedGroup = Strings.trimToNull(cfg.group());
    if (normalizedGroup == null) {
      throw new IllegalStateException("CONFIG_GROUP_MISSING: group is required");
    }
    String normalizedAppName = Strings.trimToNull(cfg.appName());
    if (normalizedAppName == null) {
      throw new IllegalStateException("appName is required");
    }
    List<String> discovery = normalizeDiscoveryUrls(cfg.discoveryUrls());
    if (discovery.isEmpty()) {
      throw new IllegalStateException("discoveryUrls are required");
    }
    String resolvedInstance = resolveInstanceName(cfg.instance());

    DefinitionApp appDefinition =
        cache
            .get(DefinitionApp.class, normalizedGroup, normalizedAppName)
            .orElseThrow(
                () ->
                    new IllegalStateException(
                        "Recovered snapshot is missing app definition "
                            + normalizedGroup
                            + "/"
                            + normalizedAppName));
    if (Boolean.FALSE.equals(appDefinition.envelope().enabled())) {
      throw new IllegalStateException(
          "Recovered app "
              + normalizedGroup
              + "/"
              + normalizedAppName
              + " is disabled in metadata; refusing to start");
    }
    boolean configCacheAdmin = Boolean.TRUE.equals(appDefinition.configCacheAdmin());

    DefinitionEnvironment environmentDefinition =
        cache
            .get(DefinitionEnvironment.class, PLATFORM_GROUP, ENVIRONMENT_NAME)
            .orElseThrow(
                () ->
                    new IllegalStateException(
                        "Recovered snapshot is missing environment registry definition platform/"
                            + ENVIRONMENT_NAME));
    Placement placement = resolvePlacement(environmentDefinition, discovery);

    List<String> cpUris =
        cache
            .get(DefinitionConnection.class, PLATFORM_GROUP, DISCOVERY_CONNECTION_NAME)
            .map(SessionBootstrap::resolveDiscoveryAmpsUris)
            .orElseGet(() -> toDiscoveryUris(discovery));

    return new BootstrapResult(
        normalizedGroup,
        normalizedAppName,
        resolvedInstance,
        List.copyOf(discovery),
        placement.environment(),
        placement.region(),
        placement.site(),
        List.copyOf(cpUris),
        null,
        configCacheAdmin,
        environmentDefinition,
        appDefinition,
        kerberosPropertyScope);
  }

  static List<String> normalizeDiscoveryUrls(List<String> urls) {
    if (urls == null || urls.isEmpty()) {
      return List.of();
    }
    List<String> cleaned = new ArrayList<>();
    for (String value : urls) {
      if (value == null) {
        continue;
      }
      String[] parts = value.split(",");
      for (String part : parts) {
        String candidate = Strings.trimToNull(part);
        if (candidate == null) {
          continue;
        }
        cleaned.add(AmpsConfig.requireHostPort(candidate, "discoveryUrl"));
      }
    }
    if (cleaned.isEmpty()) {
      return List.of();
    }
    LinkedHashSet<String> deduped = new LinkedHashSet<>(cleaned);
    return List.copyOf(deduped);
  }

  static void configureKerberos(SessionImpl.SessionConfig cfg, AmpsFactoryOptions options) {
    if (options == null) {
      return;
    }
    AmpsFactoryOptions.Kerberos kerberos = options.kerberos();
    if (Strings.isNotBlank(cfg.kerberosPrincipal()) || cfg.kerberosKeytab() != null) {
      kerberos.enabled(true);
    }
    if (Strings.isNotBlank(cfg.kerberosPrincipal())) {
      kerberos.principal(cfg.kerberosPrincipal());
    }
    Path keytab = cfg.kerberosKeytab();
    if (keytab != null) {
      kerberos.keytab(keytab.toAbsolutePath().toString());
    }
    maybeApplyClasspathKrb5Conf(kerberos);
  }

  private static void maybeApplyClasspathKrb5Conf(AmpsFactoryOptions.Kerberos kerberos) {
    if (kerberos == null || !kerberos.isEnabled()) {
      return;
    }
    if (Strings.isNotBlank(kerberos.getKrb5ConfPath())) {
      return;
    }
    byte[] existingData = kerberos.getKrb5ConfData();
    if (existingData != null && existingData.length > 0) {
      return;
    }
    try (InputStream input = SessionBootstrap.class.getClassLoader().getResourceAsStream("krb5.conf")) {
      if (input == null) {
        return;
      }
      byte[] data = input.readAllBytes();
      if (data.length > 0) {
        kerberos.krb5ConfData(data);
      }
    } catch (IOException ex) {
      throw new IllegalStateException("Failed to load classpath krb5.conf", ex);
    }
  }

  private static Optional<Definition> sowGetConfigRecord(
      HAClient client, String group, DefinitionType type, String name) throws Exception {
    if (client == null) {
      return Optional.empty();
    }
    String normalizedGroup = IdentityParsingSupport.requireGroup(group, "group");
    String normalizedName = IdentityParsingSupport.requireName(name, "name");
    DefinitionType requiredType = Objects.requireNonNull(type, "type");
    String filter =
        "/envelope/group='"
            + escapeFilterValue(normalizedGroup)
            + "' AND /envelope/name='"
            + escapeFilterValue(normalizedName)
            + "'";

    Command command = new Command(Message.Command.SOW);
    command.setTopic(DiscoveryTopics.DISCOVERY);
    command.setFilter(filter);

    try (MessageStream stream = client.execute(command)) {
      for (Message message : stream) {
        if (message.getCommand() != Message.Command.SOW) {
          continue;
        }
        String payload = message.getData();
        if (Strings.isBlank(payload)) {
          continue;
        }
        Definition definition = DiscoveryCodec.decode(payload);
        if (definitionType(definition) != requiredType) {
          continue;
        }
        return Optional.of(definition);
      }
    }
    return Optional.empty();
  }

  private static DefinitionApp requireAppSnapshot(
      Optional<Definition> record, String group, String appName, HAClient client) throws Exception {
    if (record.isPresent() && record.get() instanceof DefinitionApp appDefinition) {
      return appDefinition;
    }
    List<String> apps = listAppKeys(client, 50);
    String available = apps.isEmpty() ? "(none)" : String.join(", ", apps);
    throw new IllegalStateException(
        "App " + group + "/" + appName + " not found in metadata, available apps: " + available);
  }

  private static DefinitionEnvironment requireEnvironmentRegistry(Optional<Definition> record) {
    if (record.isPresent() && record.get() instanceof DefinitionEnvironment registry) {
      return registry;
    }
    throw new IllegalStateException(
        "Environment registry record not found: expected platform/environment. "
            + "Remediation: publish platform configuration definition environment.");
  }

  private static DefinitionConnection requireDiscoveryConnection(Optional<Definition> record) {
    if (record.isPresent() && record.get() instanceof DefinitionConnection connection) {
      return connection;
    }
    throw new IllegalStateException(
        "Connection platform/" + DISCOVERY_CONNECTION_NAME + " not found in metadata");
  }

  private static List<String> resolveDiscoveryAmpsUris(DefinitionConnection connection) {
    DefinitionConnection.ConnectionType type = connection.connectionType();
    if (type != null && type != DefinitionConnection.ConnectionType.AMPS) {
      throw new IllegalStateException("Discovery connection model must be AMPS (found " + type + ")");
    }
    DefinitionConnection.Amps amps = connection.amps();
    List<String> uris = amps == null || amps.uris() == null ? List.of() : amps.uris();
    if (uris.isEmpty()) {
      throw new IllegalStateException("Discovery connection model is missing amps.uris");
    }
    return normalizeBrokers(uris);
  }

  private static Placement resolvePlacement(
      DefinitionEnvironment registry, List<String> normalizedDiscovery) {
    Map<String, List<Placement>> index = buildEnvironmentRegistryIndex(registry);
    if (index.isEmpty()) {
      throw new IllegalStateException(
          "Environment registry endpoints missing. Expected payload.config.environments[].regions[].sites[].amps[].endpoints");
    }
    for (String endpoint : normalizedDiscovery) {
      List<Placement> placements = index.get(endpoint);
      if (placements == null || placements.isEmpty()) {
        continue;
      }
      if (placements.size() > 1) {
        throw new IllegalStateException(
            "ambiguous environment registry match for "
                + endpoint
                + " candidates="
                + describePlacements(placements));
      }
      return placements.get(0);
    }
    List<String> candidates = new ArrayList<>(index.keySet());
    candidates.sort(String::compareToIgnoreCase);
    throw new IllegalStateException(
        "no matching environment registry entry for discoveryUrls="
            + normalizedDiscovery
            + " candidates="
            + candidates);
  }

  private static Map<String, List<Placement>> buildEnvironmentRegistryIndex(
      DefinitionEnvironment registry) {
    Map<String, List<Placement>> index = new HashMap<>();
    if (registry == null || Strings.isEmpty(registry.environments())) {
      return index;
    }
    for (DefinitionEnvironment.Environment envSpec : registry.environments()) {
      if (envSpec == null || Strings.isEmpty(envSpec.regions())) {
        continue;
      }
      String env = normalizePlacementLabel(envSpec.environment());
      if (env == null) {
        continue;
      }
      for (DefinitionEnvironment.EnvironmentRegion regionSpec : envSpec.regions()) {
        if (regionSpec == null || Strings.isEmpty(regionSpec.sites())) {
          continue;
        }
        String region = normalizePlacementLabel(regionSpec.region());
        if (region == null) {
          continue;
        }
        for (DefinitionEnvironment.EnvironmentSite siteSpec : regionSpec.sites()) {
          if (siteSpec == null || Strings.isEmpty(siteSpec.amps())) {
            continue;
          }
          String site = normalizePlacementLabel(siteSpec.site());
          if (site == null) {
            continue;
          }
          Placement placement = new Placement(env, region, site);
          for (DefinitionEnvironment.EnvironmentAmps amps : siteSpec.amps()) {
            if (amps == null || Strings.isEmpty(amps.endpoints())) {
              continue;
            }
            for (String endpoint : amps.endpoints()) {
              String key = normalizeEndpointToHostPort(endpoint);
              if (key == null) {
                continue;
              }
              index.computeIfAbsent(key, ignored -> new ArrayList<>()).add(placement);
            }
          }
        }
      }
    }
    return index;
  }

  private static String normalizeEndpointToHostPort(String raw) {
    String trimmed = Strings.trimToNull(raw);
    if (trimmed == null) {
      return null;
    }
    try {
      return AmpsConfig.requireHostPort(trimmed, "endpoint");
    } catch (IllegalArgumentException ex) {
      return null;
    }
  }

  private static String normalizePlacementLabel(String value) {
    String trimmed = Strings.trimToNull(value);
    if (trimmed == null) {
      return null;
    }
    return trimmed.toUpperCase(Locale.ROOT);
  }

  private static String describePlacements(List<Placement> placements) {
    if (placements == null || placements.isEmpty()) {
      return "[]";
    }
    List<String> labels = new ArrayList<>();
    for (Placement placement : placements) {
      if (placement == null) {
        continue;
      }
      labels.add(placement.environment() + "/" + placement.region() + "/" + placement.site());
    }
    return labels.toString();
  }

  private static List<String> listAppKeys(HAClient client, int limit) throws Exception {
    if (client == null) {
      return List.of();
    }
    Command command = new Command(Message.Command.SOW);
    command.setTopic(DiscoveryTopics.DISCOVERY);
    command.setFilter("/envelope/type='APP'");

    LinkedHashSet<String> keys = new LinkedHashSet<>();
    try (MessageStream stream = client.execute(command)) {
      for (Message message : stream) {
        if (message.getCommand() != Message.Command.SOW) {
          continue;
        }
        String payload = message.getData();
        if (Strings.isBlank(payload)) {
          continue;
        }
        try {
          Definition definition = DiscoveryCodec.decode(payload);
          if (definitionType(definition) != DefinitionType.APP) {
            continue;
          }
          DefinitionEnvelope envelope = definitionEnvelope(definition);
          String key =
              (envelope == null ? null : envelope.group())
                  + "/"
                  + (envelope == null ? null : envelope.name());
          keys.add(key);
          if (keys.size() >= limit) {
            break;
          }
        } catch (Exception ignore) {
        }
      }
    }
    return List.copyOf(keys);
  }

  private static List<String> normalizeBrokers(List<String> brokers) {
    if (brokers == null) {
      throw new IllegalArgumentException("brokers are required");
    }
    List<String> cleaned = new ArrayList<>();
    for (String broker : brokers) {
      String trimmed = Strings.trimToNull(broker);
      if (trimmed != null) {
        cleaned.add(trimmed);
      }
    }
    if (cleaned.isEmpty()) {
      throw new IllegalArgumentException("brokers must contain at least one endpoint");
    }
    return List.copyOf(cleaned);
  }

  private static List<String> toDiscoveryUris(Collection<String> hostPorts) {
    if (hostPorts == null || hostPorts.isEmpty()) {
      return List.of();
    }
    List<String> uris = new ArrayList<>();
    for (String endpoint : hostPorts) {
      String trimmed = Strings.trimToNull(endpoint);
      if (trimmed != null) {
        uris.add("tcp://" + trimmed + "/amps/json");
      }
    }
    return List.copyOf(uris);
  }

  private static String escapeFilterValue(String value) {
    return value == null ? "" : value.replace("'", "''");
  }

  private static DefinitionType definitionType(Definition definition) {
    DefinitionEnvelope envelope = definitionEnvelope(definition);
    return envelope == null ? null : envelope.type();
  }

  private static DefinitionEnvelope definitionEnvelope(Definition definition) {
    return definition == null ? null : definition.envelope();
  }

  private static void closeClient(HAClient client) {
    if (client == null) {
      return;
    }
    try {
      client.disconnect();
    } catch (Exception ignore) {
    }
    try {
      client.close();
    } catch (Exception ignore) {
    }
  }


  record BootstrapResult(
      String group,
      String appName,
      String instance,
      List<String> discoveryUrls,
      String environment,
      String region,
      String site,
      List<String> discoveryUris,
      ControlConnection controlPlaneConnection,
      boolean configCacheAdmin,
      DefinitionEnvironment environmentDefinition,
      DefinitionApp appDefinition,
      SessionVmArgs kerberosPropertyScope) {}

  private record Placement(String environment, String region, String site) {}
}
