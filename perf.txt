  private DiscoveryManager(

  public void start() {
    synchronized (lifecycleLock) {
      if (discoveryCache != null) {
        return;
      }
      long managerStartNanos = System.nanoTime();
      long phaseStartNanos = System.nanoTime();
      DiscoveryCache createdCache =
          preloadedDiscoveryCache == null
              ? createDiscoveryCache(controlConnection)
              : startPreloadedDiscoveryCache(preloadedDiscoveryCache);
      discoveryCache = createdCache;
      LOGGER.info(
          "[Session][StartupTiming] phase=discovery-manager.discovery-cache elapsedMs={} recovered={} offline={}",
          elapsedMillis(phaseStartNanos),
          createdCache.recovered(),
          createdCache.offlineMode());

      phaseStartNanos = System.nanoTime();
      DiscoveryVisibilityManager visibilityManager =
          new DiscoveryVisibilityManager(
              createdCache,
              new DiscoveryScopeResolver(),
              () -> sessionGroup,
              subjectSupplier,
              delegatedSubjectSupplier);
      visibilityManager.start();
      discoveryVisibilityManager = visibilityManager;
      discoveryScope = visibilityManager.currentScope();
      discoveryScopeDiff = visibilityManager.lastDiff();
      LOGGER.info(
          "[Session][StartupTiming] phase=discovery-manager.visibility elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      appStatusCache = controlConnection == null ? null : createSharedAppStatusCache(controlConnection);
      LOGGER.info(
          "[Session][StartupTiming] phase=discovery-manager.app-status-cache elapsedMs={} enabled={}",
          elapsedMillis(phaseStartNanos),
          appStatusCache != null);
      phaseStartNanos = System.nanoTime();
      serviceStatusCache =
          controlConnection == null ? null : createSharedServiceStatusCache(controlConnection);
      LOGGER.info(
          "[Session][StartupTiming] phase=discovery-manager.service-status-cache elapsedMs={} enabled={}",
          elapsedMillis(phaseStartNanos),
          serviceStatusCache != null);
      phaseStartNanos = System.nanoTime();
      dataStatusCache = controlConnection == null ? null : createSharedDataStatusCache(controlConnection);
      if (controlConnection != null) {
        statusSubscription =
            createSharedStatusSubscription(
                controlConnection, appStatusCache, serviceStatusCache, dataStatusCache);
      }
      LOGGER.info(
          "[Session][StartupTiming] phase=discovery-manager.data-status-cache elapsedMs={} enabled={}",
          elapsedMillis(phaseStartNanos),
          dataStatusCache != null);
      LOGGER.info(
          "[Session][StartupTiming] phase=discovery-manager.complete elapsedMs={}",
          elapsedMillis(managerStartNanos));
    }
  }

    synchronized (lifecycleLock) {
      CloseSupport.closeQuietly(discoveryVisibilityManager);
      discoveryVisibilityManager = null;
      discoveryScope = null;
      discoveryScopeDiff = null;
      CloseSupport.closeQuietly(statusSubscription);
      statusSubscription = null;
      CloseSupport.closeQuietly(appStatusCache);
      appStatusCache = null;
      CloseSupport.closeQuietly(serviceStatusCache);
      serviceStatusCache = null;
      CloseSupport.closeQuietly(dataStatusCache);
      dataStatusCache = null;
      CloseSupport.closeQuietly(discoveryCache);
      discoveryCache = null;
    }
  }

  private DiscoveryCache createDiscoveryCache(ControlConnection connection) {
    DiscoveryCache created = new DiscoveryCache(connection, sessionGroup, configCacheAdmin, verbose);
    try {
      created.start();
      return created;
    } catch (Exception ex) {
      CloseSupport.closeQuietly(created);
      throw new IllegalStateException("Failed to initialize DiscoveryCache", ex);
    }
  }

  private static DiscoveryCache startPreloadedDiscoveryCache(DiscoveryCache preloaded) {
    try {
      preloaded.start();

      throw new IllegalStateException("Failed to initialize recovered DiscoveryCache", ex);
    }
  }

  private ControlConnection.Subscription createSharedStatusSubscription(
      ControlConnection connection,
      AppStatusCache appCache,
      ServiceStatusCache serviceCache,
      DataStatusCache dataCache) {
    Objects.requireNonNull(connection, "connection");
    Objects.requireNonNull(appCache, "appCache");
    Objects.requireNonNull(serviceCache, "serviceCache");
    Objects.requireNonNull(dataCache, "dataCache");
    ControlConnection.Subscription subscription = null;
    boolean success = false;
    try {
      subscription =
          connection.sowAndSubscribe(
              DiscoveryTopics.STATUS,
              "oof",
              STATUS_SUBSCRIPTION_FILTER,
              message -> {
                com.crankuptheamps.client.Message raw = message.raw();
                appCache.onStatusMessage(raw);
                serviceCache.onStatusMessage(raw);
                dataCache.onStatusMessage(raw);
              });
      Duration recoveryTimeout = Duration.ofSeconds(30);
      appCache.awaitRecovery(recoveryTimeout);
      serviceCache.awaitRecovery(recoveryTimeout);
      dataCache.awaitRecovery(recoveryTimeout);
      success = true;
      return subscription;
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to initialize shared status subscription", ex);
    } finally {
      if (!success) {
        CloseSupport.closeQuietly(subscription);
        CloseSupport.closeQuietly(appCache);
        CloseSupport.closeQuietly(serviceCache);
        CloseSupport.closeQuietly(dataCache);
      }
    }
  }

  private static long elapsedMillis(long startNanos) {
    return Math.max(0L, (System.nanoTime() - startNanos) / 1_000_000L);
  }

}

        LOGGER.info("[Checkpoint] usingConfiguredPath path={}", resolvedCheckpointPath);

        this.checkpointPath = null;
        this.checkpointStore =
            CheckpointStores.noOp(
                "checkpointPath not configured for session "
                    + Strings.firstNonBlank(group, "<group>")
                    + "/"
                    + Strings.firstNonBlank(appName, "<app>"));
        LOGGER.info("[Checkpoint] disabled (no checkpointPath configured)");
      }
      LOGGER.info(
          "[Session][StartupTiming] phase=checkpoint elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      SessionImplManagers.StartedManagers startedManagers =
          SessionImplManagers.initialize(
              this,
              cfg,
              result,
              sessionId,
              checkpointStore,
              connectionManager,
              serviceEntitlementsLogDecisions,
              LOGGER);
      DiscoveryCache cache = startedManagers.discoveryCache();
      LOGGER.info(
          "[Session][StartupTiming] phase=managers.initialize elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      this.discoveryManager = startedManagers.discoveryManager();
      this.deleteManager = startedManagers.deleteManager();
      this.entitlementsManager = startedManagers.entitlementsManager();
      this.serviceEntitlements = startedManagers.serviceEntitlements();
      this.dataEntitlements = startedManagers.dataEntitlements();
      this.serviceManager = startedManagers.serviceManager();
      this.schemaRegistryManager = startedManagers.schemaRegistryManager();
      this.streamManager = startedManagers.streamManager();
      this.dataPlanes = startedManagers.dataPlanes();
      this.assetCatalogService = startedManagers.assetCatalogService();
      this.configsCore = startedManagers.configsCore();
      LOGGER.info(
          "[Session][StartupTiming] phase=managers.publish elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      registerShutdownHook();
      this.started.set(true);
      refreshSessionViews();
      LOGGER.info(
          "[Session][StartupTiming] phase=session-mark-started elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      SessionImplLifecycle.runDuplicateInstanceCheck(
          LOGGER,
          controlPlaneConnection,
          appDefinition,
          cfg,
          group,
          appName,
          instance,
          DEFAULT_DUPLICATE_HEARTBEAT_GRACE,
          DEFAULT_DUPLICATE_INSTANCE_CHECK_ATTEMPTS,
          DEFAULT_DUPLICATE_INSTANCE_CHECK_DELAY_MILLIS);
      LOGGER.info(
          "[Session][StartupTiming] phase=duplicate-instance-check elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      startStatusPublisher();
      LOGGER.info(
          "[Session][StartupTiming] phase=status-publisher-start elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      phaseStartNanos = System.nanoTime();
      refreshSessionViews();
      LOGGER.info(
          "[Session][StartupTiming] phase=session-refresh-final elapsedMs={}",
          elapsedMillis(phaseStartNanos));

      if (discoveryRecoveryStatus.recoveryUsed()) {
        LOGGER.info(
            "[Session][DiscoveryRecovery] startup=recovered sessionId={} app={}/{} instance={} snapshotPath={} definitions={}",
            sessionId,
            group,
            appName,
            instance,
            discoveryRecoveryStatus.snapshotPath(),
            discoveryRecoveryStatus.definitionCount());
      } else {
        LOGGER.info(
            "[Session][DiscoveryRecovery] startup=live sessionId={} app={}/{} instance={} endpoints={} definitions={}",
            sessionId,
            group,
            appName,
            instance,
            discoveryUris,
            cache == null ? 0 : cache.size());
      }
      LOGGER.info(
          "[Session][StartupTiming] phase=complete sessionId={} totalElapsedMs={} mode={}",
          sessionId,
          elapsedMillis(startupStartNanos),
          discoveryRecoveryStatus.recoveryUsed() ? "recovered" : "live");
      LOGGER.debug("L0 bootstrap complete sessionId={} app={}/{} instance={} env={} region={} site={} endpoints={} configCount={}", sessionId, group, appName, instance, env, region, site, discoveryUris, cache == null ? 0 : cache.size());
    }
  }

  private SessionBootstrap.BootstrapResult bootstrapWithRecoveryFallback() throws Exception {
    LOGGER.info(
        "[Session][DiscoveryRecovery] config enabled={} path={} ttl={}",
        cfg.discoveryCacheRecoveryEnabled(),
        cfg.discoveryCacheRecoveryPath(),
        cfg.discoveryCacheRecoveryTtl());
    LOGGER.info(
        "[Session][DiscoveryRecovery] live-bootstrap-start app={}/{} instance={} discoveryUrls={}",
        cfg.group(),
        cfg.appName(),
        cfg.instance(),
        cfg.discoveryUrls());
    try {
      SessionBootstrap.BootstrapResult liveResult =
          SessionBootstrap.bootstrap(cfg, sessionId, kerberosPropertyScope);
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.live(
              cfg.discoveryCacheRecoveryEnabled(), "live DiscoveryCache bootstrap succeeded");
      LOGGER.info(
          "[Session][DiscoveryRecovery] live-bootstrap-succeeded app={}/{} instance={}",
          cfg.group(),
          cfg.appName(),
          cfg.instance());
      return liveResult;
    } catch (Exception liveFailure) {
      LOGGER.warn(
          "[Session][DiscoveryRecovery] live-bootstrap-failed app={}/{} instance={} message={}",
          cfg.group(),
          cfg.appName(),
          cfg.instance(),
          Strings.firstNonBlank(liveFailure.getMessage(), liveFailure.getClass().getSimpleName()));
      return recoverBootstrapOrThrow(liveFailure);
    }
  }

  private SessionBootstrap.BootstrapResult recoverBootstrapOrThrow(Exception liveFailure)
      throws Exception {
    if (!cfg.discoveryCacheRecoveryEnabled()) {
      LOGGER.info("[Session][DiscoveryRecovery] recovery-disabled; startup will fail if live bootstrap fails");
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.failed(
              false,
              null,
              null,
              null,
              null,
              "Recovery disabled and live bootstrap failed");
      throw liveFailure;
    }

    Path snapshotPath = cfg.discoveryCacheRecoveryPath();
    Duration snapshotTtl = cfg.discoveryCacheRecoveryTtl();
    if (snapshotPath == null) {
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.failed(
              true,
              null,
              null,
              null,
              snapshotTtl,
              "Recovery enabled but snapshot path is not configured");
      IllegalStateException error =
          new IllegalStateException(
              "DiscoveryCache recovery is enabled but no snapshot path is configured");
      error.addSuppressed(liveFailure);
      throw error;
    }

    LOGGER.info(
        "[Session][DiscoveryRecovery] recovery-attempt-enabled=true path={} ttl={}",
        snapshotPath.toAbsolutePath().normalize(),
        snapshotTtl);
    DiscoveryCache probeCache = null;
    DiscoveryCache runtimeCache = null;
    boolean published = false;
    try {
      DiscoveryCacheSnapshotService.LoadedSnapshot loaded =
          discoverySnapshotService.load(snapshotPath, snapshotTtl);
      String sessionGroup = Strings.trimToNull(cfg.group());
      probeCache =
          DiscoveryCache.recoveredSnapshotCache(
              sessionGroup, true, cfg.verbose(), loaded.definitions());
      SessionBootstrap.BootstrapResult recoveredResult =
          SessionBootstrap.bootstrapFromRecoveredSnapshot(
              cfg, sessionId, kerberosPropertyScope, probeCache);
      runtimeCache = probeCache;
      if (!recoveredResult.configCacheAdmin()) {
        runtimeCache =
            DiscoveryCache.recoveredSnapshotCache(
                recoveredResult.group(), false, cfg.verbose(), loaded.definitions());
        probeCache.close();
        probeCache = null;
      }
      publishDiscoveryManager(
          DiscoveryManager.recoveredSnapshot(
              runtimeCache, recoveredResult.group(), this::defaultCatalogSubject, () -> null));
      published = true;
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.recovered(
              loaded.snapshotPath(),
              loaded.createdTimeUtc(),
              loaded.age(),
              loaded.ttl(),
              loaded.manifest().definitionCount(),
              "Recovered DiscoveryCache from disk snapshot");
      LOGGER.info(
          "[Session][DiscoveryRecovery] recovery-accepted path={} createdTimeUtc={} age={} ttl={} definitions={}",
          loaded.snapshotPath(),
          loaded.createdTimeUtc(),
          loaded.age(),
          loaded.ttl(),
          loaded.manifest().definitionCount());
      return recoveredResult;
    } catch (Exception recoveryFailure) {
      CloseSupport.closeQuietly(probeCache);
      if (!published) {
        CloseSupport.closeQuietly(runtimeCache);
      }
      discoveryRecoveryStatus =
          DiscoveryCacheRecoveryResult.failed(
              true,
              snapshotPath,
              null,
              null,
              snapshotTtl,
              Strings.firstNonBlank(
                  recoveryFailure.getMessage(), recoveryFailure.getClass().getSimpleName()));
      LOGGER.error(
          "[Session][DiscoveryRecovery] recovery-rejected path={} ttl={} reason={}",
          snapshotPath.toAbsolutePath().normalize(),
          snapshotTtl,
          Strings.firstNonBlank(
              recoveryFailure.getMessage(), recoveryFailure.getClass().getSimpleName()));
      IllegalStateException startupFailure =
          new IllegalStateException(
              "Live bootstrap failed and DiscoveryCache recovery failed: "
                  + Strings.firstNonBlank(
                      recoveryFailure.getMessage(),
                      recoveryFailure.getClass().getSimpleName()),
              recoveryFailure);
      startupFailure.addSuppressed(liveFailure);
      throw startupFailure;
    }
  }

  private void publish(DefinitionStatusApp status, String reason) {
    Objects.requireNonNull(status, "status");
    DefinitionStatusApp normalized = normalizeStatus(status);
    String group = Strings.trimToNull(normalized.group());
    if (group == null) {
      throw new IllegalStateException("STATUS_GROUP_MISSING: group is required");
    }
    current = normalized;
    int attempt = 0;
    while (attempt < MAX_ATTEMPTS) {
      try {
        publishOnce(normalized, reason);
        return;
      } catch (Exception ex) {
        attempt++;
        if (attempt >= MAX_ATTEMPTS) {
          LOG.warn(
              "[AppStatusPublisher] failed to publish status for {}/{}/{} after {} attempts: {}",
              normalized.group(),
              normalized.name(),
              normalized.instance(),
              attempt,
              ex.toString());
        } else {
          try {
            Thread.sleep(50L * attempt);
          } catch (InterruptedException ie) {
            Thread.currentThread().interrupt();
            return;
          }
        }
      }
    }
  }

  private void publishHeartbeatSafe() {
    if (!running.get()) {
      return;
    }
    try {
      DefinitionStatusApp baseline = current;
      if (baseline == null) {
        return;
      }
      long now = clock.millis();
      DefinitionStatusApp heartbeat = withLastHeartbeatEpochMs(baseline, now);
      publish(heartbeat, "heartbeat");
    } catch (Exception ex) {
      LOG.warn("[AppStatusPublisher] heartbeat publish failed", ex);
    }
  }

  private void publishOnce(DefinitionStatusApp status, String reason) throws Exception {
    if (status == null) {
      return;
    }
    String payload = mapper.writeValueAsString(status);
    control.amps().publish(DiscoveryTopics.STATUS, payload);

    long publishedAt =
        status.lastHeartbeatEpochMs() == null
            ? clock.millis()
            : Math.max(0L, status.lastHeartbeatEpochMs());
    this.lastPayload = payload;
    this.lastPublishedEpochMs = publishedAt;
    long count = this.publishCount.incrementAndGet();
    LOG.info(
        "[AppStatusPublisher] published reason={} topic={} group={} name={} instance={} state={} heartbeatEpochMs={} count={}",
        Strings.firstNonBlank(reason, "unknown"),
        DiscoveryTopics.STATUS,
        Strings.firstNonBlank(status.group(), ""),
        Strings.firstNonBlank(status.name(), ""),
        Strings.firstNonBlank(status.instance(), ""),
        status.state() == null ? "UNKNOWN" : status.state().name(),
        publishedAt,
        count);
    if (tickListener != null) {
      try {
        tickListener.onPublished(publishedAt, status, payload);
      } catch (Exception ignore) {
      }
    }
  }

  private static final String TOPIC_STATUS = DiscoveryTopics.STATUS;
  private static final String FILTER_STATUS_DATA = "/type='STATUS_DATA'";
  private static final long DEFAULT_STATUS_CADENCE_SECONDS = 30L;
  private static final Duration DEFAULT_STALE_AFTER =
      resolveStaleAfter(DEFAULT_STATUS_CADENCE_SECONDS);
  private static final String UNKNOWN = "-";

  private final ControlConnection control;
  private final Clock clock;
  private final Duration staleAfter;
  private final ConcurrentHashMap<Key, StatusSnapshot> statuses = new ConcurrentHashMap<>();
  private final AtomicBoolean started = new AtomicBoolean(false);
  private final AtomicBoolean recovered = new AtomicBoolean(false);
  private final CountDownLatch recoveryLatch = new CountDownLatch(1);
  private volatile ControlConnection.Subscription subscription;

  public DataStatusCache(ControlConnection control) {
    this(control, Clock.systemUTC(), DEFAULT_STATUS_CADENCE_SECONDS);
  }

  public DataStatusCache(ControlConnection control, long statusCadenceSeconds) {
    this(control, Clock.systemUTC(), statusCadenceSeconds);
  }

  DataStatusCache(ControlConnection control, Clock clock, long statusCadenceSeconds) {
    this.control = Objects.requireNonNull(control, "control");
    this.clock = Objects.requireNonNull(clock, "clock");
    this.staleAfter = resolveStaleAfter(statusCadenceSeconds);
  }

  public void start() throws Exception {
    if (!started.compareAndSet(false, true)) {
      awaitRecovery(Duration.ofSeconds(30));
      return;
    }
    Command command = new Command(Message.Command.SOWAndSubscribe);
    command.setTopic(TOPIC_STATUS);
    command.setOptions("oof");
    command.setFilter(FILTER_STATUS_DATA);
    subscription =
        control
            .sowAndSubscribe(
                command.getTopic(),
                command.getOptions(),
                command.getFilter(),
                message -> handleMessage(message.raw()));
    awaitRecovery(Duration.ofSeconds(30));
    LOGGER.debug(
        "Data status cache ready topics={} enabledTypes={} instanceCount={}",
        List.of(TOPIC_STATUS),
        List.of("STATUS_DATA"),
        statuses.size());
  }

  public void startShared() {
    if (!started.compareAndSet(false, true)) {
      awaitRecovery(Duration.ofSeconds(30));
    }
  }

  public boolean recovered() {
    return recovered.get();
  }

  public int size() {
    return statuses.size();
  }

  public List<DataStatusRow> snapshot() {
    long nowMs = clock.millis();
    return statuses.entrySet().stream()
        .map(entry -> toRow(entry.getKey(), entry.getValue(), nowMs))
        .sorted(
            Comparator.comparing(
                    (DataStatusRow row) ->
                        Strings.trimToEmpty(row.group()), String.CASE_INSENSITIVE_ORDER)
                .thenComparing(
                    row -> Strings.trimToEmpty(row.name()), String.CASE_INSENSITIVE_ORDER)
                .thenComparing(
                    row -> Strings.trimToEmpty(row.instanceName()), String.CASE_INSENSITIVE_ORDER))
        .toList();
  }

  public void stop() {
    ControlConnection.Subscription id = subscription;
    if (id != null) {
      try {
        id.close();
      } catch (Exception ignore) {
      }
      subscription = null;
    }
    statuses.clear();
  }

  private DataStatusRow toRow(Key key, StatusSnapshot snapshot, long nowMs) {
    DefinitionStatusData status = snapshot.status();
    Long heartbeatMs = positive(status.lastHeartbeatEpochMs());
    Instant lastHeartbeatUtc = heartbeatMs == null ? null : Instant.ofEpochMilli(heartbeatMs);
    Instant lastSeenUtc = snapshot.receivedTimeUtc();
    String state = deriveState(status.state(), heartbeatMs, snapshot.receivedTimeUtc(), nowMs);
    String producerType =
        status.producerType() == null ? "UNKNOWN" : status.producerType().name();
    return new DataStatusRow(
        key.group(),
        key.name(),
        key.instanceName(),
        producerType,
        state,
        defaultText(status.producerId()),
        defaultText(status.datasetId()),
        defaultText(status.host()),
        defaultText(status.message()),
        lastHeartbeatUtc,
        lastSeenUtc);
  }

  private String deriveState(
      DefinitionStatusData.State rawState, Long heartbeatMs, Instant lastSeenUtc, long nowMs) {
    if (isOfflineState(rawState)) {
      return "OFFLINE";
    }
    if (isErrorState(rawState)) {
      return "ERROR";
    }
    if (isStale(heartbeatMs, lastSeenUtc, nowMs)) {
      return "STALE";
    }
    if (isOnlineState(rawState)) {
      return "ONLINE";
    }
    return "UNKNOWN";
  }

  private boolean isStale(Long heartbeatMs, Instant lastSeenUtc, long nowMs) {
    long referenceMs = heartbeatMs == null ? 0L : heartbeatMs;
    if (referenceMs <= 0L && lastSeenUtc != null) {
      referenceMs = lastSeenUtc.toEpochMilli();
    }
    if (referenceMs <= 0L) {
      return true;
    }
    return nowMs - referenceMs > staleAfter.toMillis();
  }

  private boolean isOnlineState(DefinitionStatusData.State state) {
    return switch (state == null ? DefinitionStatusData.State.UNKNOWN : state) {
      case ONLINE -> true;
      default -> false;
    };
  }

  private boolean isOfflineState(DefinitionStatusData.State state) {
    return switch (state == null ? DefinitionStatusData.State.UNKNOWN : state) {
      case OFFLINE -> true;
      default -> false;
    };
  }

  private boolean isErrorState(DefinitionStatusData.State state) {
    return switch (state == null ? DefinitionStatusData.State.UNKNOWN : state) {
      case ERROR -> true;
      default -> false;
    };
  }

  public void awaitRecovery(Duration timeout) {
    long millis = timeout == null ? 30000L : Math.max(1L, timeout.toMillis());
    try {
      boolean done = recoveryLatch.await(millis, TimeUnit.MILLISECONDS);
      if (!done) {
        throw new IllegalStateException(
            "DataStatusCache initial recovery did not complete (GroupEnd not observed)");
      }
    } catch (InterruptedException ex) {
      Thread.currentThread().interrupt();
      throw new IllegalStateException("Interrupted while waiting for DataStatusCache recovery", ex);
    }
  }

  private void handleMessage(Message message) {
    if (message == null) {
      return;
    }
    int command = message.getCommand();
    if (command == Message.Command.GroupEnd) {
      recovered.set(true);
      recoveryLatch.countDown();
      return;
    }
    if (command != Message.Command.SOW
        && command != Message.Command.Publish
        && command != Message.Command.OOF
        && command != Message.Command.SOWDelete) {
      return;
    }
    String payload = message.getData();
    if (Strings.isBlank(payload)) {
      return;
    }
    try {
      Definition definition = DiscoveryCodec.decode(payload);
      if (!(definition instanceof DefinitionStatusData status)) {
        return;
      }
      Key key = key(status.group(), status.name(), status.instance());
      if (key == null) {
        return;
      }
      if (command == Message.Command.OOF || command == Message.Command.SOWDelete) {
        statuses.remove(key);
        return;
      }
      statuses.put(key, new StatusSnapshot(status, Instant.now(clock)));
    } catch (Exception ex) {
      LOGGER.debug("Data status decode failed payload={} reason={}", payload, ex.toString());
    }
  }

  public void onStatusMessage(Message message) {
    handleMessage(message);
  }

  private static Key key(String group, String name, String instanceName) {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedName = Strings.trimToNull(name);
    String normalizedInstance = Strings.trimToNull(instanceName);
    if (normalizedGroup == null || normalizedName == null || normalizedInstance == null) {
      return null;
    }
    return new Key(normalizedGroup, normalizedName, normalizedInstance);
  }

  private static Long positive(Long value) {
    return value == null || value <= 0L ? null : value;
  }

  private static String defaultText(String value) {
    String normalized = Strings.trimToNull(value);
    return normalized == null ? UNKNOWN : normalized;
  }

  private static Duration resolveStaleAfter(long statusCadenceSeconds) {
    long cadence = statusCadenceSeconds <= 0L ? DEFAULT_STATUS_CADENCE_SECONDS : statusCadenceSeconds;
    long staleSeconds = Math.max(60L, cadence * 2L);
    return Duration.ofSeconds(staleSeconds);
  }

  private record Key(String group, String name, String instanceName) {}

  private record StatusSnapshot(DefinitionStatusData status, Instant receivedTimeUtc) {}
}

    }
    tags.put(normalizedKey, normalizedValue);
  }

  @Override
  public void close() {
    if (!open) {
      return;
    }
    open = false;
    publishSafe(DefinitionStatusData.State.OFFLINE, "stopped");
    ScheduledFuture<?> task = heartbeatTask;
    heartbeatTask = null;
    if (task != null) {
      task.cancel(true);
    }
    if (ownsScheduler) {
      ses.shutdownNow();
    }
  }

  private void publishSafe(DefinitionStatusData.State state, String detail) {
    try {
      publish(buildStatus(state, detail));
    } catch (Exception ex) {
      LOGGER.debug(
          "Data status publish failed group={} name={} instance={} reason={}",
          group,
          name,
          instance,
          ex.toString());
    }
  }

  private void publish(DefinitionStatusData status) throws Exception {
    if (status == null) {
      return;
    }
    String payload = om.writeValueAsString(status);
    if (STATUS_DEBUG) {
      LOGGER.debug("[DataStatus] publish topic={} payload={}", DiscoveryTopics.STATUS, payload);
    }
    control.amps().publish(DiscoveryTopics.STATUS, payload);
  }

  private DefinitionStatusData buildStatus(DefinitionStatusData.State state, String detail) {
    DefinitionEnvelope envelope =
        new DefinitionEnvelope(
            group, DefinitionType.STATUS_DATA, name, null, null, null, null, null, null, null);
    long now = System.currentTimeMillis();
    return new DefinitionStatusData(
        envelope,
        group,
        DefinitionType.STATUS_DATA,
        name,
        instance,
        producerType,
        normalizeState(state),
        producerId,
        datasetId,
        startEpochMs > 0L ? startEpochMs : null,
        now,
        Strings.trimToNull(detail),
        host,
        appName,
        appInstance,
        appGroup,
        tags.isEmpty() ? Map.of() : Map.copyOf(tags));
  }

  private static Map<String, String> sanitizeTags(Map<String, String> tags) {
    if (Strings.isEmpty(tags)) {
      return Map.of();
    }
    Map<String, String> normalized = new ConcurrentHashMap<>();
    for (Map.Entry<String, String> entry : tags.entrySet()) {
      if (entry == null) {
        continue;
      }
      String key = Strings.trimToNull(entry.getKey());
      String value = Strings.trimToNull(entry.getValue());
      if (key != null && value != null) {
        normalized.put(key, value);
      }
    }
    return normalized.isEmpty() ? Map.of() : Map.copyOf(normalized);
  }

  private static DefinitionStatusData.State normalizeState(DefinitionStatusData.State state) {
    if (state == null || state == DefinitionStatusData.State.UNKNOWN) {
      return DefinitionStatusData.State.ONLINE;
    }
    return state;
  }

  private String deriveState(
      DefinitionStatusService.State rawState, Long heartbeatMs, Instant lastSeenUtc, long nowMs) {
    if (isOfflineState(rawState)) {
      return "OFFLINE";
    }
    if (isStale(heartbeatMs, lastSeenUtc, nowMs)) {
      return "STALE";
    }
    if (isOnlineState(rawState)) {
      return "ONLINE";

  private boolean isStale(Long heartbeatMs, Instant lastSeenUtc, long nowMs) {
    long referenceMs = heartbeatMs == null ? 0L : heartbeatMs;
    if (referenceMs <= 0L && lastSeenUtc != null) {
      referenceMs = lastSeenUtc.toEpochMilli();
    }
    if (referenceMs <= 0L) {
      return true;
    }
    return nowMs - referenceMs > staleAfter.toMillis();
  }

  private boolean isOnlineState(DefinitionStatusService.State state) {
    return switch (state == null ? DefinitionStatusService.State.UNKNOWN : state) {
      case ONLINE -> true;
      default -> false;

  private boolean isOfflineState(DefinitionStatusService.State state) {
    return switch (state == null ? DefinitionStatusService.State.UNKNOWN : state) {

  public void awaitRecovery(Duration timeout) {
    long millis = timeout == null ? 30000L : Math.max(1L, timeout.toMillis());
    try {
      boolean done = recoveryLatch.await(millis, TimeUnit.MILLISECONDS);
      if (!done) {
        throw new IllegalStateException(
            "ServiceStatusCache initial recovery did not complete (GroupEnd not observed)");
      }
    } catch (InterruptedException ex) {
      Thread.currentThread().interrupt();
      throw new IllegalStateException("Interrupted while waiting for ServiceStatusCache recovery", ex);
    }
  }

  private void handleMessage(Message message) {
    if (message == null) {
      return;
    }
    int command = message.getCommand();
    if (command == Message.Command.GroupEnd) {
      recovered.set(true);
      recoveryLatch.countDown();
      return;
    }
    if (command != Message.Command.SOW
        && command != Message.Command.Publish
        && command != Message.Command.OOF
        && command != Message.Command.SOWDelete) {
      return;
    }
    String payload = message.getData();
    if (Strings.isBlank(payload)) {
      return;
    }
    try {
      Definition definition = DiscoveryCodec.decode(payload);
      if (!(definition instanceof DefinitionStatusService status)) {
        return;
      }
      Key key = key(status.group(), status.name(), status.instance());
      if (key == null) {
        return;
      }
      if (command == Message.Command.OOF || command == Message.Command.SOWDelete) {
        statuses.remove(key);
        return;
      }
      statuses.put(key, new StatusSnapshot(status, Instant.now(clock)));
    } catch (Exception ex) {
      LOGGER.warn("Ignoring malformed service status message command={} reason={}", command, ex.toString());
    }
  }

  private static String renderUptimeFromStart(Long startEpochMs, long nowMs) {
    if (startEpochMs == null || startEpochMs <= 0L || nowMs < startEpochMs) {
      return null;
    }
    return formatDuration(nowMs - startEpochMs);
  }

  private static String formatPercent(String value) {
    if (Strings.isBlank(value)) {
      return null;
    }
    String trimmed = value.trim();
    if (trimmed.endsWith("%")) {
      return trimmed;
    }
    return trimmed + "%";
  }

  private static String formatBytes(Long bytes) {
    if (bytes == null || bytes < 0) {
      return null;
    }
    String[] units = {"B", "KiB", "MiB", "GiB", "TiB"};
    double scaled = bytes;
    int unit = 0;
    while (scaled >= 1024.0 && unit < units.length - 1) {
      scaled /= 1024.0;
      unit++;
    }
    return unit == 0
        ? bytes + " B"
        : String.format(java.util.Locale.ROOT, "%.1f %s", scaled, units[unit]);
  }

  private static String formatDuration(Long millis) {
    if (millis == null || millis <= 0L) {
      return null;
    }
    long totalSeconds = millis / 1000L;
    if (totalSeconds <= 0L) {
      return "0s";
    }
    long days = totalSeconds / 86_400;
    long remainder = totalSeconds % 86_400;
    long hours = remainder / 3_600;
    remainder %= 3_600;
    long minutes = remainder / 60;
    long seconds = remainder % 60;
    List<String> parts = new java.util.ArrayList<>(4);
    if (days > 0) {
      parts.add(days + "d");
    }
    if (hours > 0) {
      parts.add(hours + "h");
    }
    if (minutes > 0) {
      parts.add(minutes + "m");
    }
    if (seconds > 0 && parts.isEmpty()) {
      parts.add(seconds + "s");
    }
    return parts.isEmpty() ? "0s" : String.join(" ", parts);
  }

  private static Long parseLong(String value) {
    if (Strings.isBlank(value)) {
      return null;
    }
    try {
      return Long.parseLong(value.trim());
    } catch (NumberFormatException ignore) {
      return null;
    }
  }

    return value == null || value <= 0L ? null : value;

    }
    return null;
  }

        firstNonBlank(Strings.trimToNull(attrs.get("tlsTermination")), tlsRequired ? "PASSTHROUGH" : null);
    Boolean kerberosEnabled = parseBoolean(attrs.get("kerberosEnabled"));
    if (kerberosEnabled == null) {
      kerberosEnabled = false;
    }
    String kerberosSpn = Strings.trimToNull(attrs.get("kerberosSpn"));
    String derivedSpn = Strings.trimToNull(attrs.get("derivedSpn"));
    Long startEpochMs = serviceStartEpochMs > 0L ? serviceStartEpochMs : null;
    DefinitionStatusService.State state = null;
    if (status != null) {
      try {
        state = DefinitionStatusService.State.valueOf(status);
      } catch (IllegalArgumentException ignore) {
        state = DefinitionStatusService.State.ONLINE;
      }
    }
    if (state == null) {
      state = DefinitionStatusService.State.ONLINE;
    }
    if (state == DefinitionStatusService.State.UNKNOWN) {
      state = DefinitionStatusService.State.ONLINE;
    }
    return new DefinitionStatusService(
        envelope,
        normalizedGroup,
        DefinitionType.STATUS_SERVICE,
        normalizedName,
        normalizedInstance,
        state,
        nowMs,
        startEpochMs,
        null,
        transport,
        uri,
        advertiseHost,
        advertisePort,
        bindHost,
        bindPort,
        tlsEnabled,
        tlsTermination,
        kerberosEnabled,
        kerberosSpn,
        derivedSpn,
        Strings.trimToNull(serviceHost),
        port,
        tlsRequired,
        Strings.trimToNull(routeHost),
        routePort,
        Strings.trimToNull(routeAuthority),
        Strings.trimToNull(region),
        Strings.trimToNull(serviceAppName),
        Strings.trimToNull(serviceAppInstanceId),
        Strings.trimToNull(serviceAppGroup),
        tagsSnapshot,
        sessionId,
        sessionKey);
  }

  private DefinitionHbService buildServiceHeartbeat() {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedName = Strings.trimToNull(serviceName);
    String normalizedInstance = Strings.trimToNull(serviceInstanceId);
    if (normalizedGroup == null || normalizedName == null || normalizedInstance == null) {
      return null;
    }
    DefinitionEnvelope envelope = statusEnvelope(DefinitionType.HEARTBEAT_SERVICE, normalizedName);
    if (envelope == null) {
      return null;
    }
    Integer port = servicePort > 0 ? servicePort : null;
    String transport = resolveTransport();
    String uri = resolveUri(port);
    Map<String, String> tagsSnapshot = attrs.isEmpty() ? Map.of() : Map.copyOf(attrs);
    String advertiseHost = Strings.trimToNull(serviceHost);
    Integer advertisePort = port;
    String bindHost = Strings.trimToNull(attrs.get("bindHost"));
    Integer bindPort = parseInteger(attrs.get("bindPort"));
    Boolean tlsEnabled = tlsRequired;
    String tlsTermination =
        firstNonBlank(Strings.trimToNull(attrs.get("tlsTermination")), tlsRequired ? "PASSTHROUGH" : null);
    Boolean kerberosEnabled = parseBoolean(attrs.get("kerberosEnabled"));
    if (kerberosEnabled == null) {
      kerberosEnabled = false;
    }
    String kerberosSpn = Strings.trimToNull(attrs.get("kerberosSpn"));
    String derivedSpn = Strings.trimToNull(attrs.get("derivedSpn"));
    long nowMs = System.currentTimeMillis();
    return new DefinitionHbService(
        envelope,
        normalizedGroup,
        DefinitionType.HEARTBEAT_SERVICE,
        normalizedName,
        normalizedInstance,
        transport,
        uri,
        advertiseHost,
        advertisePort,
        bindHost,
        bindPort,
        tlsEnabled,
        tlsTermination,
        kerberosEnabled,
        kerberosSpn,
        derivedSpn,
        Strings.trimToNull(serviceHost),
        port,
        tlsRequired,
        Strings.trimToNull(routeHost),
        routePort,
        Strings.trimToNull(routeAuthority),
        Strings.trimToNull(region),
        nowMs,
        tagsSnapshot);
  }

  private DefinitionEnvelope statusEnvelope(DefinitionType type, String name) {
    String normalizedGroup = Strings.trimToNull(group);
    String normalizedName = Strings.trimToNull(name);
    if (normalizedGroup == null || normalizedName == null || type == null) {
      return null;
    }
    return new DefinitionEnvelope(
        normalizedGroup,
        type,
        normalizedName,
        null,
        null,
        null,
        null,
        null,
        null,
        null);
  }

  private String resolveTransport() {
    if (Strings.isEmpty(attrs)) {
      return null;
    }
    return firstNonBlank(attrs.get("transport"), attrs.get("protocol"), attrs.get("serviceType"));
  }

  private String resolveUri(Integer port) {
    String authority = firstNonBlank(routeAuthority, routeHost, serviceHost);
    if (authority == null) {
      return null;
    }
    if (authority.contains(":")) {
      return authority;
    }
    int resolvedPort = port == null ? 0 : port;
    if (resolvedPort <= 0 && routePort != null) {
      resolvedPort = routePort;
    }
    return resolvedPort > 0 ? authority + ":" + resolvedPort : authority;
  }

  private static String firstNonBlank(String... values) {
    if (values == null) {
      return null;
    }
    for (String value : values) {
      if (Strings.isNotBlank(value)) {
        return value.trim();
      }
    }
    return null;
  }

  private static Integer parseInteger(String value) {
    if (Strings.isBlank(value)) {
      return null;
    }
    try {
      return Integer.parseInt(value.trim());
    } catch (NumberFormatException ignore) {
      return null;
    }
  }

  private static Boolean parseBoolean(String value) {
    if (Strings.isBlank(value)) {
      return null;
    }
    String normalized = value.trim();
    if ("true".equalsIgnoreCase(normalized) || "yes".equalsIgnoreCase(normalized)) {
      return true;
    }
    if ("false".equalsIgnoreCase(normalized) || "no".equalsIgnoreCase(normalized)) {
      return false;
    }
    return null;
  }

  private static String valueOrUnknown(String value) {
    return Strings.isBlank(value) ? "<unknown>" : value;
  }
