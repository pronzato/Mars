package org.pronzato.fabric.api.control.discovery.lifecycle;

import java.util.ArrayDeque;
import java.util.ArrayList;
import java.util.Comparator;
import java.util.LinkedHashMap;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionType;

public final class DeletePlanner {
  private DeletePlanner() {}

  public static DeletePlan plan(AssetLifecycleGraph graph, String rootAssetIdOrToken) {
    return plan(graph, rootAssetIdOrToken, DeleteOptions.defaults());
  }

  public static DeletePlan plan(
      AssetLifecycleGraph graph, String rootAssetIdOrToken, DeleteOptions options) {
    DeleteOptions resolvedOptions = options == null ? DeleteOptions.defaults() : options;
    String normalizedRootToken = trim(rootAssetIdOrToken);
    String resolvedRoot = resolveRootAssetId(graph, normalizedRootToken);

    List<DeleteBlocker> blockers = new ArrayList<>();
    LinkedHashSet<String> assetsToDelete = new LinkedHashSet<>();
    LinkedHashSet<String> assetsToDetach = new LinkedHashSet<>();
    LinkedHashSet<String> assetsRetained = new LinkedHashSet<>();
    LinkedHashSet<String> storageToDelete = new LinkedHashSet<>();
    LinkedHashSet<String> historyToDelete = new LinkedHashSet<>();
    LinkedHashSet<String> warnings = new LinkedHashSet<>();
    LinkedHashSet<String> manualActions = new LinkedHashSet<>();

    if (graph == null) {
      blockers.add(
          new DeleteBlocker(
              normalizedRootToken,
              "<graph>",
              "GRAPH_UNAVAILABLE",
              "Lifecycle graph is required for delete planning."));
      return emptyPlan(normalizedRootToken, resolvedOptions, blockers);
    }
    if (resolvedRoot == null || !graph.hasAsset(resolvedRoot)) {
      blockers.add(
          new DeleteBlocker(
              normalizedRootToken,
              "<graph>",
              "ROOT_NOT_FOUND",
              "Root asset was not found in lifecycle graph."));
      return emptyPlan(normalizedRootToken, resolvedOptions, blockers);
    }

    ArrayDeque<String> queue = new ArrayDeque<>();
    assetsToDelete.add(resolvedRoot);
    queue.add(resolvedRoot);
    while (!queue.isEmpty()) {
      String current = queue.removeFirst();
      for (AssetLifecycleEdge edge : graph.outgoing(current)) {
        String target = edge.toAssetId();
        if (target == null || target.equals(current)) {
          continue;
        }
        DefinitionType targetType =
            graph.node(target).map(AssetLifecycleNode::type).orElse(DefinitionType.UNKNOWN);
        if (targetType == DefinitionType.UNKNOWN && isInternalAssetId(target)) {
          warnings.add(
              "Compatibility: unresolved dependency type for internal asset "
                  + target
                  + "; dependency retained");
          assetsToDetach.add(target);
          assetsRetained.add(target);
          if (edge.required() && !resolvedOptions.force()) {
            blockers.add(
                new DeleteBlocker(
                    current,
                    target,
                    "UNKNOWN_DEPENDENCY_TYPE",
                    "Dependency type is UNKNOWN for internal asset; cannot safely cascade delete."));
          }
          continue;
        }
        switch (edge.deleteBehavior()) {
          case CASCADE_SOFT, CASCADE_HARD -> {
            boolean cascadeAllowed =
                resolvedOptions.deleteOwnedDependencies()
                    && (edge.ownershipType() == OwnershipType.OWNED
                        || edge.ownershipType() == OwnershipType.DERIVED);
            boolean cascadeTypeAllowed = resolvedOptions.allowsCascadeType(targetType);
            if (!cascadeTypeAllowed) {
              warnings.add(
                  "Cascade suppressed by options for "
                      + targetType.name().toLowerCase()
                      + " dependency "
                      + target);
              assetsToDetach.add(target);
              assetsRetained.add(target);
              continue;
            }
            if (!cascadeAllowed) {
              if (edge.required() && !resolvedOptions.force()) {
                blockers.add(
                    new DeleteBlocker(
                        current,
                        target,
                        "DEPENDENCY_NOT_OWNED",
                        "Dependency cannot be cascaded because it is not owned."));
              } else {
                assetsToDetach.add(target);
                assetsRetained.add(target);
              }
              continue;
            }
            if (isStorageOrExternalTarget(graph, target, edge)) {
              if (resolvedOptions.deleteData()
                  && (resolvedOptions.hardDeleteData()
                      || edge.deleteBehavior() == DeleteBehavior.CASCADE_HARD)) {
                storageToDelete.add(target);
              } else {
                assetsRetained.add(target);
                manualActions.add("Manual data deletion required for " + target);
              }
              if (resolvedOptions.deleteHistory()
                  && edge.retentionBehavior() == RetentionBehavior.DELETE_HISTORY) {
                historyToDelete.add(target);
              }
              continue;
            }
            if (assetsToDelete.add(target)) {
              queue.addLast(target);
            }
          }
          case DETACH -> {
            assetsToDetach.add(target);
            assetsRetained.add(target);
          }
          case BLOCK -> {
            if (!resolvedOptions.force()) {
              blockers.add(
                  new DeleteBlocker(
                      current, target, "BLOCKING_DEPENDENCY", "Dependency blocks delete cascade."));
            } else {
              warnings.add("Force delete bypassed blocker: " + current + " -> " + target);
            }
            assetsRetained.add(target);
          }
          case MANUAL -> {
            manualActions.add("Manual delete action required: " + current + " -> " + target);
            assetsRetained.add(target);
          }
        }
      }
    }

    addIncomingDependencyBlockers(graph, assetsToDelete, assetsToDetach, assetsRetained, blockers, warnings, resolvedOptions);
    if (resolvedOptions.deleteOrphanedSchemas()) {
      addOrphanedAssetsByType(graph, assetsToDelete, DefinitionType.SCHEMA, warnings);
    }
    if (resolvedOptions.deleteOrphanedEntitlements()) {
      addOrphanedAssetsByType(graph, assetsToDelete, DefinitionType.ENTITLEMENT, warnings);
    }
    populateStorageAndHistoryEffects(
        graph,
        assetsToDelete,
        resolvedOptions,
        storageToDelete,
        historyToDelete,
        assetsRetained,
        manualActions);

    // Final normalization: an asset cannot be both scheduled for delete and retained/detached.
    assetsToDetach.removeAll(assetsToDelete);
    assetsRetained.removeAll(assetsToDelete);

    List<String> deleteList = sortStrings(assetsToDelete);
    List<String> detachList = sortStrings(assetsToDetach);
    List<String> retainedList = sortStrings(assetsRetained);
    List<String> storageList = sortStrings(storageToDelete);
    List<String> historyList = sortStrings(historyToDelete);
    List<String> warningList = sortStrings(warnings);
    List<String> manualList = sortStrings(manualActions);
    List<String> order = buildDeleteExecutionOrder(graph, assetsToDelete);
    blockers.sort(
        Comparator.comparing(DeleteBlocker::assetId)
            .thenComparing(DeleteBlocker::blockedByAssetId)
            .thenComparing(DeleteBlocker::code));

    return new DeletePlan(
        resolvedRoot,
        resolvedOptions,
        deleteList,
        detachList,
        retainedList,
        storageList,
        historyList,
        blockers,
        warningList,
        order,
        manualList);
  }

  private static void addIncomingDependencyBlockers(
      AssetLifecycleGraph graph,
      Set<String> assetsToDelete,
      Set<String> assetsToDetach,
      Set<String> assetsRetained,
      List<DeleteBlocker> blockers,
      Set<String> warnings,
      DeleteOptions options) {
    for (String deleting : assetsToDelete) {
      for (AssetLifecycleEdge incoming : graph.incoming(deleting)) {
        String dependent = incoming.fromAssetId();
        if (dependent == null || assetsToDelete.contains(dependent)) {
          continue;
        }
        boolean blocking = incoming.deleteBehavior() == DeleteBehavior.BLOCK || incoming.required();
        if (blocking && !options.force()) {
          blockers.add(
              new DeleteBlocker(
                  deleting,
                  dependent,
                  "INCOMING_DEPENDENT",
                  "External dependent asset still references asset slated for delete."));
        } else if (blocking) {
          warnings.add("Force delete bypassed incoming dependency: " + dependent + " -> " + deleting);
        } else if (incoming.deleteBehavior() == DeleteBehavior.DETACH) {
          assetsToDetach.add(dependent);
        }
        assetsRetained.add(dependent);
      }
    }
  }

  private static void addOrphanedAssetsByType(
      AssetLifecycleGraph graph, Set<String> assetsToDelete, DefinitionType type, Set<String> warnings) {
    boolean changed = true;
    while (changed) {
      changed = false;
      for (AssetLifecycleNode node : graph.nodes().values()) {
        if (node == null || node.type() != type || assetsToDelete.contains(node.assetId())) {
          continue;
        }
        if (isOrphanAfterDelete(graph, node.assetId(), assetsToDelete)) {
          assetsToDelete.add(node.assetId());
          warnings.add("Deleting orphaned " + type.name().toLowerCase() + ": " + node.assetId());
          changed = true;
        }
      }
    }
  }

  private static boolean isOrphanAfterDelete(
      AssetLifecycleGraph graph, String candidateAssetId, Set<String> assetsToDelete) {
    for (AssetLifecycleEdge incoming : graph.incoming(candidateAssetId)) {
      String from = incoming.fromAssetId();
      if (from == null || from.equals(candidateAssetId) || assetsToDelete.contains(from)) {
        continue;
      }
      if (isInternalAssetId(from)) {
        return false;
      }
    }
    return true;
  }

  private static void populateStorageAndHistoryEffects(
      AssetLifecycleGraph graph,
      Set<String> assetsToDelete,
      DeleteOptions options,
      Set<String> storageToDelete,
      Set<String> historyToDelete,
      Set<String> assetsRetained,
      Set<String> manualActions) {
    for (String deleting : assetsToDelete) {
      for (AssetLifecycleEdge edge : graph.outgoing(deleting)) {
        String target = edge.toAssetId();
        if (target == null) {
          continue;
        }
        if (edge.relationshipType() == RelationshipType.STORES_IN || target.startsWith("storage:")) {
          if (options.deleteData() && (options.hardDeleteData() || edge.deleteBehavior() == DeleteBehavior.CASCADE_HARD)) {
            storageToDelete.add(target);
          } else {
            assetsRetained.add(target);
            manualActions.add("Data retained unless hard delete is requested: " + target);
          }
        }
        if (options.deleteHistory() && edge.retentionBehavior() == RetentionBehavior.DELETE_HISTORY) {
          historyToDelete.add(target);
        }
      }
    }
  }

  private static List<String> buildDeleteExecutionOrder(
      AssetLifecycleGraph graph, Set<String> selectedAssets) {
    Map<String, Integer> indegree = new LinkedHashMap<>();
    Map<String, LinkedHashSet<String>> edges = new LinkedHashMap<>();
    for (String asset : selectedAssets) {
      indegree.put(asset, 0);
      edges.put(asset, new LinkedHashSet<>());
    }
    for (String dependent : selectedAssets) {
      for (AssetLifecycleEdge edge : graph.outgoing(dependent)) {
        String dependency = edge.toAssetId();
        if (!selectedAssets.contains(dependency) || dependent.equals(dependency)) {
          continue;
        }
        if (edges.get(dependent).add(dependency)) {
          indegree.put(dependency, indegree.get(dependency) + 1);
        }
      }
    }

    ArrayDeque<String> queue = new ArrayDeque<>();
    List<String> roots = new ArrayList<>();
    for (Map.Entry<String, Integer> entry : indegree.entrySet()) {
      if (entry.getValue() == 0) {
        roots.add(entry.getKey());
      }
    }
    roots.sort(String::compareToIgnoreCase);
    queue.addAll(roots);

    List<String> order = new ArrayList<>(selectedAssets.size());
    while (!queue.isEmpty()) {
      String next = queue.removeFirst();
      order.add(next);
      List<String> dependencies = new ArrayList<>(edges.getOrDefault(next, new LinkedHashSet<>()));
      dependencies.sort(String::compareToIgnoreCase);
      for (String dependency : dependencies) {
        int remaining = indegree.get(dependency) - 1;
        indegree.put(dependency, remaining);
        if (remaining == 0) {
          queue.addLast(dependency);
        }
      }
    }
    if (order.size() < selectedAssets.size()) {
      List<String> unresolved = new ArrayList<>();
      for (String asset : selectedAssets) {
        if (!order.contains(asset)) {
          unresolved.add(asset);
        }
      }
      unresolved.sort(String::compareToIgnoreCase);
      order.addAll(unresolved);
    }
    return List.copyOf(order);
  }

  private static boolean isStorageOrExternalTarget(
      AssetLifecycleGraph graph, String targetAssetId, AssetLifecycleEdge edge) {
    if (targetAssetId.startsWith("storage:")
        || targetAssetId.startsWith("external:")
        || edge.ownershipType() == OwnershipType.EXTERNAL) {
      return true;
    }
    DefinitionType type = graph.node(targetAssetId).map(AssetLifecycleNode::type).orElse(DefinitionType.UNKNOWN);
    return type == DefinitionType.UNKNOWN && !isInternalAssetId(targetAssetId);
  }

  private static boolean isInternalAssetId(String assetId) {
    String value = trim(assetId);
    if (value == null) {
      return false;
    }
    return value.startsWith("dataset:")
        || value.startsWith("service:")
        || value.startsWith("stream:")
        || value.startsWith("schema:")
        || value.startsWith("config:")
        || value.startsWith("connection:")
        || value.startsWith("ingest:")
        || value.startsWith("ingest-source:")
        || value.startsWith("entitlement:");
  }

  private static DeletePlan emptyPlan(
      String rootAssetId, DeleteOptions options, List<DeleteBlocker> blockers) {
    return new DeletePlan(
        rootAssetId,
        options,
        List.of(),
        List.of(),
        List.of(),
        List.of(),
        List.of(),
        blockers,
        List.of(),
        List.of(),
        List.of());
  }

  private static String resolveRootAssetId(AssetLifecycleGraph graph, String rootAssetIdOrToken) {
    if (graph == null || rootAssetIdOrToken == null || rootAssetIdOrToken.isBlank()) {
      return null;
    }
    if (graph.hasAsset(rootAssetIdOrToken)) {
      return rootAssetIdOrToken;
    }
    return graph.resolveAssetId(rootAssetIdOrToken);
  }

  private static List<String> sortStrings(Set<String> values) {
    List<String> sorted = new ArrayList<>(values);
    sorted.sort(String::compareToIgnoreCase);
    return List.copyOf(sorted);
  }

  private static String trim(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }
}
