public final class ParquetManagerImpl implements ParquetManager {
  private static final Pattern PARTITION_PATTERN = Pattern.compile("(^|/)date=([0-9]{8})(/|$)");
  private static final String DEFAULT_PARQUET_GLOB = "**/*.parquet";

  private final RowManagerImpl rowManager;
  private final DiscoveryCache DiscoveryCache;
  private final Fabric.Session session;
  private final String group;
  private final CheckpointStore checkpointStore;
  private final ParquetCompactionPlanner compactionPlanner;
  private final ParquetRetentionPlanner retentionPlanner;

  public ParquetManagerImpl(
      RowManager manager, DiscoveryCache DiscoveryCache, String group, Fabric.Session session) {
    this(
        manager,
        DiscoveryCache,
        group,
        session,
        CheckpointStores.failFast(
            "ParquetManager checkpoint store is not configured. Configure session with withCheckpointPath(...)"));
  }

  public ParquetManagerImpl(
      RowManager manager,
      DiscoveryCache DiscoveryCache,
      String group,
      Fabric.Session session,
      CheckpointStore checkpointStore) {
    this(
        manager,
        DiscoveryCache,
        group,
        session,
        checkpointStore,
        new ParquetCompactionPlanner(),
        new ParquetRetentionPlanner());
  }

  ParquetManagerImpl(
      RowManager manager,
      DiscoveryCache DiscoveryCache,
      String group,
      Fabric.Session session,
      CheckpointStore checkpointStore,
      ParquetCompactionPlanner compactionPlanner,
      ParquetRetentionPlanner retentionPlanner) {
    if (!(manager instanceof RowManagerImpl localManager)) {
      throw new IllegalStateException(
          "Parquet maintenance requires DIRECT mode local RowManager implementation");
    }
    this.rowManager = localManager;
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.group = Strings.requireNonBlank(group, "group");
    this.session = session;
    this.checkpointStore = Objects.requireNonNull(checkpointStore, "checkpointStore");
    this.compactionPlanner = Objects.requireNonNull(compactionPlanner, "compactionPlanner");
    this.retentionPlanner = Objects.requireNonNull(retentionPlanner, "retentionPlanner");
  }

  @Override
  public ParquetMaintenanceResult compactDataset(String datasetId, ParquetCompactionOptions options) {
    return compact(datasetId, null, options, "compact-dataset");
  }

  @Override
  public ParquetMaintenanceResult compactPartition(
      String datasetId, String partitionSpec, ParquetCompactionOptions options) {
    String normalized = Strings.requireNonBlank(partitionSpec, "partitionSpec");
    return compact(datasetId, normalized, options, "compact-partition");
  }

  @Override
  public ParquetMaintenanceResult applyRetention(String datasetId, ParquetRetentionPolicy policy) {
    long started = System.currentTimeMillis();
    DatasetContext context = resolveDataset(datasetId);
    DataWriteAuthorization.requireWrite(
        rowManager.dataEntitlementsForSqlPlanning(),
        rowManager.effectiveIdentityForSqlPlanning(),
        context.datasetId().toString(),
        "PARQUET_RETENTION",
        null);
    List<String> warnings = new ArrayList<>();
    List<String> files = listParquetFiles(context);
    S3Connection s3Connection = context.s3() ? resolveS3Connection(context) : null;
    Map<String, List<String>> filesByPartition = partitionToFiles(files);
    List<String> partitions = new ArrayList<>(filesByPartition.keySet());
    partitions.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
    List<String> partitionsToDelete = retentionPlanner.partitionsToDelete(partitions, policy);

    int deleted = 0;
    long bytes = 0L;
    for (String partition : partitionsToDelete) {
      List<String> partitionFiles = filesByPartition.getOrDefault(partition, List.of());
      for (String file : partitionFiles) {
        bytes += context.s3() ? s3FileSize(s3Connection, file, warnings) : localFileSize(file);
      }
      if (!policy.dryRun()) {
        if (context.s3()) {
          deleted += deleteS3Files(s3Connection, partitionFiles, warnings);
        } else {
          for (String file : partitionFiles) {
            if (deleteLocalFile(file)) {
              deleted++;
            }
          }
        }
      }
    }
    long durationMs = Math.max(0L, System.currentTimeMillis() - started);
    writeCheckpoint(
        context.datasetId().toString(),
        "retention",
        Map.of(
            "partitionsDeleted", Integer.toString(partitionsToDelete.size()),
            "filesDeleted", Integer.toString(deleted),
            "bytes", Long.toString(Math.max(0L, bytes)),
            "durationMs", Long.toString(durationMs)));
    return new ParquetMaintenanceResult(
        0L,
        bytes,
        files.size(),
        0,
        deleted,
        durationMs,
        List.copyOf(warnings));
  }

  @Override
  public ParquetMaintenanceResult optimizeDataset(String datasetId, ParquetCompactionOptions options) {
    ParquetMaintenanceResult compactResult = compact(datasetId, null, options, "optimize");
    writeCheckpoint(
        DatasetId.parse(datasetId).toString(),
        "optimize",
        Map.of(
            "filesRead", Integer.toString(compactResult.filesRead()),
            "filesWritten", Integer.toString(compactResult.filesWritten()),
            "durationMs", Long.toString(compactResult.durationMs())));
    return compactResult;
  }

  private ParquetMaintenanceResult compact(
      String datasetId,
      String partitionSpec,
      ParquetCompactionOptions options,
      String checkpointScope) {
    long started = System.currentTimeMillis();
    DatasetContext context = resolveDataset(datasetId);
    DataWriteAuthorization.requireWrite(
        rowManager.dataEntitlementsForSqlPlanning(),
        rowManager.effectiveIdentityForSqlPlanning(),
        context.datasetId().toString(),
        "PARQUET_COMPACT",
        null);
    List<String> allFiles = listParquetFiles(context);
    List<String> filtered = filterPartition(allFiles, partitionSpec);
    ParquetCompactionOptions resolved =
        options == null ? ParquetCompactionOptions.defaults() : options;
    List<List<String>> groups = compactionPlanner.plan(filtered, resolved);
    List<String> warnings = new ArrayList<>();
    S3Connection s3Connection = context.s3() ? resolveS3Connection(context) : null;

    long rows = 0L;
    long bytes = 0L;
    int filesRead = 0;
    int filesWritten = 0;
    int filesDeleted = 0;

    for (int index = 0; index < groups.size(); index++) {
      List<String> groupFiles = groups.get(index);
      if (groupFiles.size() < resolved.minFilesPerGroup()) {
        continue;
      }
      filesRead += groupFiles.size();
      for (String file : groupFiles) {
        bytes += context.s3() ? s3FileSize(s3Connection, file, warnings) : localFileSize(file);
      }
      rows += countRows(groupFiles, context);

      String outputFile = outputFilePath(context, groupFiles, index);
      if (!resolved.dryRun()) {
        writeCompactedFile(groupFiles, outputFile, context);
      }
      filesWritten++;
      if (!resolved.dryRun()) {
        if (context.s3()) {
          filesDeleted += deleteS3Files(s3Connection, groupFiles, warnings);
        } else {
          for (String source : groupFiles) {
            if (deleteLocalFile(source)) {
              filesDeleted++;
            }
          }
        }
      }
    }

    long durationMs = Math.max(0L, System.currentTimeMillis() - started);
    writeCheckpoint(
        context.datasetId().toString(),
        checkpointScope,
        Map.of(
            "groups", Integer.toString(groups.size()),
            "filesRead", Integer.toString(filesRead),
            "filesWritten", Integer.toString(filesWritten),
            "filesDeleted", Integer.toString(filesDeleted),
            "durationMs", Long.toString(durationMs)));
    return new ParquetMaintenanceResult(
        rows, bytes, filesRead, filesWritten, filesDeleted, durationMs, List.copyOf(warnings));
  }

  private int deleteS3Files(S3Connection connection, List<String> files, List<String> warnings) {
    if (connection == null || Strings.isEmpty(files)) {
      return 0;
    }
    try {
      var result = connection.admin().deleteObjects(files);
      for (String failedKey : result.failedKeys()) {
        warnings.add("S3 delete failed for file=" + failedKey);
      }
      return result.deletedCount();
    } catch (Exception ex) {
      warnings.add("S3 delete failed for " + files.size() + " file(s): " + ex.getMessage());
      return 0;
    }
  }

  private long s3FileSize(S3Connection connection, String file, List<String> warnings) {
    if (connection == null || Strings.isBlank(file)) {
      return 0L;
    }
    try {
      return Math.max(0L, connection.admin().headObject(file).sizeBytes());
    } catch (Exception ex) {
      warnings.add("S3 headObject failed for file=" + file + " error=" + ex.getMessage());
      return 0L;
    }
  }

  private S3Connection resolveS3Connection(DatasetContext context) {
    if (!context.s3()) {
      return null;
    }
    if (session == null) {
      throw new IllegalStateException("S3 parquet maintenance requires a session");
    }
    String connectionName =
        Strings.requireNonBlank(
            context.connectionName(),
            "dataset.parquet.connectionName is required for s3 dataset " + context.datasetId());
    return session.connections().s3(connectionName);
  }

  private long countRows(List<String> files, DatasetContext context) {
    if (files.isEmpty()) {
      return 0L;
    }
    return rowManager.withDuckdbSession(
        session -> {
          configureS3IfNeeded(session, context);
          String sql =
              "select count(*) from read_parquet(" + parquetArraySql(files) + ", hive_partitioning=true)";
          try (java.sql.Statement statement = session.createStatement();
              ResultSet result = statement.executeQuery(sql)) {
            return result.next() ? result.getLong(1) : 0L;
          } catch (SQLException ex) {
            throw new IllegalStateException("Failed to count parquet rows: " + ex.getMessage(), ex);
          }
        });
  }

  private void writeCompactedFile(List<String> files, String outputFile, DatasetContext context) {
    rowManager.withDuckdbSession(
        session -> {
          configureS3IfNeeded(session, context);
          if (!context.s3()) {
            Path outputPath = Path.of(outputFile).toAbsolutePath().normalize();
            Path parent = outputPath.getParent();
            if (parent != null) {
              try {
                Files.createDirectories(parent);
              } catch (Exception ex) {
                throw new IllegalStateException("Failed to create output directory: " + parent, ex);
              }
            }
          }
          String sql =
              "copy (select * from read_parquet("
                  + parquetArraySql(files)
                  + ", hive_partitioning=true)) to '"
                  + escapeSql(outputFile)
                  + "' (format parquet, compression snappy)";
          session.exec(sql);
        });
  }

  private List<String> listParquetFiles(DatasetContext context) {
    if (!context.s3()) {
      return listLocalParquetFiles(context.rootUri(), context.glob());
    }
    return rowManager.withDuckdbSession(
        session -> {
          configureS3IfNeeded(session, context);
          String sql =
              "select file from glob('"
                  + escapeSql(appendGlob(normalizeUri(context.rootUri()), context.glob()))
                  + "') order by file";
          List<String> files = new ArrayList<>();
          try (java.sql.Statement statement = session.createStatement();
              ResultSet result = statement.executeQuery(sql)) {
            while (result.next()) {
              String file = Strings.trimToNull(result.getString(1));
              if (file != null && file.toLowerCase(Locale.ROOT).endsWith(".parquet")) {
                files.add(file);
              }
            }
          } catch (SQLException ex) {
            throw new IllegalStateException("Failed to list parquet files from s3: " + ex.getMessage(), ex);
          }
          files.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
          return List.copyOf(files);
        });
  }

  private List<String> listLocalParquetFiles(String rootUri, String glob) {
    try {
      Path root = Path.of(rootUri).toAbsolutePath().normalize();
      if (!Files.exists(root)) {
        return List.of();
      }
      List<String> files = new ArrayList<>();
      try (java.util.stream.Stream<Path> stream = Files.walk(root)) {
        stream
            .filter(Files::isRegularFile)
            .filter(path -> path.getFileName().toString().toLowerCase(Locale.ROOT).endsWith(".parquet"))
            .forEach(path -> files.add(path.toString().replace('\\', '/')));
      }
      files.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
      return List.copyOf(files);
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to list local parquet files: root=" + rootUri, ex);
    }
  }

  private List<String> filterPartition(List<String> files, String partitionSpec) {
    if (partitionSpec == null) {
      return files;
    }
    String needle = "/" + partitionSpec + "/";
    List<String> filtered = new ArrayList<>();
    for (String file : files) {
      String normalized = file.replace('\\', '/');
      if (normalized.contains(needle)) {
        filtered.add(normalized);
      }
    }
    filtered.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
    return List.copyOf(filtered);
  }

  private String outputFilePath(DatasetContext context, List<String> sourceFiles, int groupIndex) {
    String root = normalizeUri(context.rootUri());
    String signature =
        Integer.toHexString(
            Objects.hash(sourceFiles.get(0), sourceFiles.get(sourceFiles.size() - 1), sourceFiles.size()));
    String name = String.format(Locale.ROOT, "compact_%03d_%s.parquet", groupIndex + 1, signature);
    return appendPath(root, "_compact/" + name);
  }

  private Map<String, List<String>> partitionToFiles(List<String> files) {
    LinkedHashMap<String, List<String>> grouped = new LinkedHashMap<>();
    for (String file : files) {
      Matcher matcher = PARTITION_PATTERN.matcher(file.replace('\\', '/'));
      if (!matcher.find()) {
        continue;
      }
      String partition = "date=" + matcher.group(2);
      grouped.computeIfAbsent(partition, ignored -> new ArrayList<>()).add(file);
    }
    LinkedHashMap<String, List<String>> ordered = new LinkedHashMap<>();
    List<String> partitions = new ArrayList<>(grouped.keySet());
    partitions.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
    for (String partition : partitions) {
      List<String> partitionFiles = new ArrayList<>(grouped.get(partition));
      partitionFiles.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
      ordered.put(partition, List.copyOf(partitionFiles));
    }
    return Map.copyOf(ordered);
  }

  private DatasetContext resolveDataset(String datasetIdValue) {
    DatasetId datasetId = DatasetId.parse(Strings.requireNonBlank(datasetIdValue, "datasetId"));
    DefinitionDataset definition = resolveDefinition(datasetId);
    DefinitionDataset.ParquetBinding parquet =
        Optional.ofNullable(definition.binding())
            .map(DefinitionDataset.Binding::parquet)
            .orElseThrow(() -> new IllegalStateException("Dataset is not parquet: " + datasetId));
    String uri = Strings.requireNonBlank(parquet.uri(), "dataset.parquet.uri");
    return new DatasetContext(
        datasetId,
        uri,
        normalizeGlob(parquet.glob()),
        isS3Uri(uri),
        Strings.trimToNull(parquet.connectionName()));
  }

  private DefinitionDataset resolveDefinition(DatasetId datasetId) {
    for (DefinitionDataset candidate : DiscoveryCache.list(DefinitionDataset.class, datasetId.group())) {
      if (candidate == null || candidate.envelope() == null) {
        continue;
      }
      DefinitionEnvelope envelope = candidate.envelope();
      if (!datasetId.group().equals(envelope.group())) {
        continue;
      }
      if (!datasetId.namespace().equals(candidate.namespace())) {
        continue;
      }
      if (envelope.name() == null || !envelope.name().equals(datasetId.namespace() + "/" + datasetId.name())) {
        continue;
      }
      return candidate;
    }
    throw new IllegalStateException("Dataset definition not found: " + datasetId);
  }

  private void configureS3IfNeeded(DuckdbSession sessionHandle, DatasetContext context) {
    if (!context.s3()) {
      return;
    }
    resolveS3Connection(context).admin().configureDuckdbSession(sessionHandle);
  }

  private void writeCheckpoint(String datasetId, String scope, Map<String, String> payload) {
    checkpointStore.save(
        CheckpointRecord.of(
            new CheckpointKey(group, CheckpointType.PARQUET_MAINT, datasetId, scope),
            payload == null ? Map.of("updatedUtc", Instant.now().toString()) : payload));
  }

  private static String parquetArraySql(List<String> files) {
    List<String> quoted = new ArrayList<>(files.size());
    for (String file : files) {
      quoted.add("'" + escapeSql(file) + "'");
    }
    return "[" + String.join(", ", quoted) + "]";
  }

  private static boolean deleteLocalFile(String file) {
    try {
      return Files.deleteIfExists(Path.of(file));
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to delete local parquet file: " + file, ex);
    }
  }

  private static long localFileSize(String file) {
    if (file == null || isS3Uri(file)) {
      return 0L;
    }
    try {
      Path path = Path.of(file);
      if (!Files.exists(path)) {
        return 0L;
      }
      return Files.size(path);
    } catch (Exception ex) {
      return 0L;
    }
  }

  private static String normalizeGlob(String value) {
    return Strings.firstNonBlank(Strings.trimToNull(value), DEFAULT_PARQUET_GLOB);
  }

  private static String appendGlob(String base, String glob) {
    return appendPath(base, glob.startsWith("/") ? glob.substring(1) : glob);
  }

  private static String appendPath(String base, String suffix) {
    String normalizedBase = normalizeUri(base);
    String normalizedSuffix = Strings.requireNonBlank(suffix, "suffix").replace('\\', '/');
    if (normalizedBase.endsWith("/")) {
      return normalizedBase + normalizedSuffix;
    }
    return normalizedBase + "/" + normalizedSuffix;
  }

  private static String normalizeUri(String value) {
    return Strings.requireNonBlank(value, "value").replace('\\', '/');
  }

  private static boolean isS3Uri(String value) {
    return value != null
        && (value.regionMatches(true, 0, "s3://", 0, "s3://".length())
            || value.regionMatches(true, 0, "s3a://", 0, "s3a://".length()));
  }

  private static String escapeSql(String value) {
    return value.replace("'", "''");
  }

  private record DatasetContext(
      DatasetId datasetId,
      String rootUri,
      String glob,
      boolean s3,
      String connectionName) {}
}
