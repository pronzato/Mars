package org.pronzato.fabric.api.control.entitlements.simple.services;

import java.util.LinkedHashSet;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import java.util.concurrent.atomic.AtomicReference;
import org.pronzato.fabric.api.control.entitlements.core.EntitlementsScopeType;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementSubjectCandidates;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementProviderSelector;
import org.pronzato.fabric.api.control.entitlements.provider.ResolveRequest;
import org.pronzato.fabric.api.control.entitlements.provider.SubjectContext;
import org.pronzato.fabric.api.control.entitlements.core.services.ServiceEntitlements;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public final class ServiceEntitlementsSimple implements ServiceEntitlements {
  private static final Logger LOGGER = LoggerFactory.getLogger(ServiceEntitlementsSimple.class);
  private static final String PLATFORM_GROUP = "platform";

  private final AtomicReference<Map<String, Set<String>>> membershipIndexRef;
  private final AtomicReference<Map<String, ServiceEntitlementsSimpleManager.Rules>> entitlementsIndexRef;
  private final EntitlementProviderSelector providerSelector;
  private final boolean logDecisions;

  ServiceEntitlementsSimple(
      AtomicReference<Map<String, Set<String>>> membershipIndexRef,
      AtomicReference<Map<String, ServiceEntitlementsSimpleManager.Rules>> entitlementsIndexRef,
      boolean logDecisions) {
    this(membershipIndexRef, entitlementsIndexRef, null, logDecisions);
  }

  ServiceEntitlementsSimple(
      AtomicReference<Map<String, Set<String>>> membershipIndexRef,
      AtomicReference<Map<String, ServiceEntitlementsSimpleManager.Rules>> entitlementsIndexRef,
      EntitlementProviderSelector providerSelector,
      boolean logDecisions) {
    this.membershipIndexRef = Objects.requireNonNull(membershipIndexRef, "membershipIndexRef");
    this.entitlementsIndexRef = Objects.requireNonNull(entitlementsIndexRef, "entitlementsIndexRef");
    this.providerSelector = providerSelector;
    this.logDecisions = logDecisions;
  }

  @Override
  public Decision authorize(Identity identity, Request request) {
    Objects.requireNonNull(identity, "identity");
    Objects.requireNonNull(request, "request");
    Objects.requireNonNull(request.action(), "request.action is required");

    String subject = requireIdentitySubject(identity);
    String resourceId = ServiceEntitlements.requireAssetResourceId(request.serviceResourceId());

    SubjectContext subjectContext;
    try {
      subjectContext = resolveSubjectContext(identity, request, resourceId, subject);
    } catch (RuntimeException ex) {
      return new Decision(false, "DENY SUBJECT_PROVIDER_UNAVAILABLE: " + safeReason(ex), null);
    }

    String resolvedSubject = subjectContext.subjectId();
    Set<String> subjectCandidates = directSubjectCandidates(resolvedSubject);
    Set<String> membershipCandidates = membershipSubjectCandidates(resolvedSubject);
    String operationName = trimToNull(request.operationName());
    Set<String> subjectGroups = subjectContext.effectiveGroups();
    boolean platformAdmin = isPlatformAdmin(subjectGroups);

    ServiceEntitlementsSimpleManager.Rules rules = entitlementsIndexRef.get().get(resourceId);
    if (rules == null) {
      Decision decision =
          platformAdmin
              ? new Decision(true, "ALLOW platform admin default (no entitlements configured)", null)
              : new Decision(false, "ABSTAIN no entitlements for resourceId", null);
      logPlatformAdminDefault(resourceId, operationName, resolvedSubject, decision);
      if (isAbstain(decision)) {
        logAbstain(
            resourceId,
            operationName,
            resolvedSubject,
            membershipCandidates,
            null,
            null,
            decision.reason());
      }
      return decision;
    }

    ServiceEntitlementsSimpleManager.ActionSlot slot = actionSlot(request.action());
    ServiceEntitlementsSimpleManager.RuleSet ruleSet = rules.forAction(slot);
    MatchResult denyMatch =
        match(ruleSet == null ? List.of() : ruleSet.denyBindings(), subjectCandidates, subjectGroups, operationName);
    MatchResult allowMatch =
        match(ruleSet == null ? List.of() : ruleSet.allowBindings(), subjectCandidates, subjectGroups, operationName);

    Decision decision;
    if (denyMatch.matched()) {
      decision = new Decision(false, "DENY matched rules", null);
    } else if (allowMatch.matched()) {
      decision = new Decision(true, "ALLOW matched rules", null);
    } else if (platformAdmin) {
      decision = new Decision(true, "ALLOW platform admin default", null);
    } else if (allowMatch.subjectMatched()) {
      decision = new Decision(false, "ABSTAIN no matching policy (operation mismatch)", null);
    } else {
      decision = new Decision(false, "ABSTAIN no matching policy", null);
    }

    logPlatformAdminDefault(resourceId, operationName, resolvedSubject, decision);
    if (isAbstain(decision)) {
      logAbstain(resourceId, operationName, resolvedSubject, membershipCandidates, rules, ruleSet, decision.reason());
    }

    if (logDecisions) {
      LOGGER.info("[Entitlements][SIMPLE] action={} resourceId={} subject={} allowed={} matched={}", request.action().name(), resourceId, resolvedSubject, decision.allowed(), !isAbstain(decision));
    }
    return decision;
  }

  private SubjectContext resolveSubjectContext(
      Identity identity, Request request, String resourceId, String subject) {
    String effectiveSubject = effectiveSubject(identity, subject);
    if (providerSelector == null) {
      Set<String> membershipCandidates = membershipSubjectCandidates(effectiveSubject);
      Set<String> groups = groupsForCandidates(membershipCandidates);
      return new SubjectContext(effectiveSubject, groups, groups, Map.of(), false);
    }

    EntitlementProviderSelector.Selection selectedProvider =
        providerSelector.select(EntitlementsScopeType.SERVICE, resourceId);

    Map<String, String> requestAttributes = new java.util.LinkedHashMap<>();
    requestAttributes.putAll(request.metadata());
    requestAttributes.putAll(request.bindings());
    String operation = trimToNull(request.operationName());
    if (operation != null) {
      requestAttributes.put("operationName", operation);
    }
    return selectedProvider.provider().resolve(
        new ResolveRequest(
            identity.subject(),
            identity.delegatedSubject(),
            EntitlementsScopeType.SERVICE.name(),
            resourceId,
            requestAttributes));
  }

  private void logPlatformAdminDefault(
      String resourceId, String operationName, String subject, Decision decision) {
    if (decision == null || !decision.allowed()) {
      return;
    }
    String reason = trimToNull(decision.reason());
    if (reason == null || !reason.toUpperCase().startsWith("ALLOW PLATFORM ADMIN DEFAULT")) {
      return;
    }
    LOGGER.info("[Entitlements][SIMPLE][PLATFORM_DEFAULT_ALLOW] scopeType={} scopeId={} operation={} subject={} reason={}", ServiceEntitlements.scopeType(), resourceId, operationName, subject, reason);
  }

  private void logAbstain(
      String resourceId,
      String operationName,
      String subject,
      Set<String> subjectCandidates,
      ServiceEntitlementsSimpleManager.Rules rules,
      ServiceEntitlementsSimpleManager.RuleSet ruleSet,
      String reason) {
    int policyCount = rules == null ? 0 : rules.policyCount();
    String bindingSummary = summarizeBindings(ruleSet == null ? List.of() : ruleSet.bindingSummaries());
    LOGGER.warn("[Entitlements][SIMPLE][ABSTAIN] scopeType={} scopeId={} operation={} subject={} subjectCandidates={} policyCount={} bindings={} reason={}", ServiceEntitlements.scopeType(), resourceId, operationName, subject, subjectCandidates, policyCount, bindingSummary, reason);
  }

  private static boolean isAbstain(Decision decision) {
    if (decision == null || decision.allowed()) {
      return false;
    }
    String reason = trimToNull(decision.reason());
    return reason != null && reason.toUpperCase().startsWith("ABSTAIN");
  }

  public boolean isMemberOfGroup(String subject, String groupId) {
    String normalizedSubject = trimToNull(subject);
    String normalizedGroup = normalizeGroup(groupId);
    if (normalizedSubject == null || normalizedGroup == null) {
      return false;
    }
    Set<String> groups = groupsForCandidates(membershipSubjectCandidates(normalizedSubject));
    if (groups.isEmpty()) {
      return false;
    }
    for (String candidate : groups) {
      if (normalizedGroup.equals(normalizeGroup(candidate))) {
        return true;
      }
    }
    return false;
  }

  public boolean hasRules(String resourceId) {
    try {
      String canonical = ServiceEntitlements.requireAssetResourceId(resourceId);
      return entitlementsIndexRef.get().containsKey(canonical);
    } catch (RuntimeException ex) {
      return false;
    }
  }

  public Set<String> groupsForSubject(String subject) {
    String normalizedSubject = trimToNull(subject);
    if (normalizedSubject == null) {
      return Set.of();
    }
    return Set.copyOf(groupsForCandidates(membershipSubjectCandidates(normalizedSubject)));
  }

  private static ServiceEntitlementsSimpleManager.ActionSlot actionSlot(Action action) {
    if (action == Action.START) {
      return ServiceEntitlementsSimpleManager.ActionSlot.START;
    }
    if (action == Action.READ) {
      return ServiceEntitlementsSimpleManager.ActionSlot.READ;
    }
    return ServiceEntitlementsSimpleManager.ActionSlot.INVOKE;
  }

  private static MatchResult match(
      List<ServiceEntitlementsSimpleManager.BindingRule> bindings,
      Set<String> subjects,
      Set<String> groups,
      String operationName) {
    if (bindings == null || bindings.isEmpty()) {
      return MatchResult.NO_MATCH;
    }
    boolean subjectMatched = false;
    for (ServiceEntitlementsSimpleManager.BindingRule binding : bindings) {
      if (binding == null || !subjectMatches(binding, subjects, groups)) {
        continue;
      }
      subjectMatched = true;
      if (operationMatches(binding.methodsAllow(), operationName)) {
        return MatchResult.MATCHED;
      }
    }
    return subjectMatched ? MatchResult.SUBJECT_ONLY : MatchResult.NO_MATCH;
  }

  private static boolean subjectMatches(
      ServiceEntitlementsSimpleManager.BindingRule binding,
      Set<String> subjects,
      Set<String> groups) {
    if (binding.group()) {
      if (groups == null || groups.isEmpty()) {
        return false;
      }
      String normalizedPolicyGroup = normalizeGroup(binding.subject());
      if (normalizedPolicyGroup == null) {
        return false;
      }
      for (String group : groups) {
        if (normalizedPolicyGroup.equals(normalizeGroup(group))) {
          return true;
        }
      }
      return false;
    }
    return subjects != null && subjects.contains(binding.subject());
  }

  private static boolean operationMatches(Set<String> methodsAllow, String operationName) {
    if (methodsAllow == null || methodsAllow.isEmpty()) {
      return true;
    }
    return operationName != null && methodsAllow.contains(operationName);
  }

  private Set<String> groupsForCandidates(Set<String> subjectCandidates) {
    if (subjectCandidates == null || subjectCandidates.isEmpty()) {
      return Set.of();
    }
    Map<String, Set<String>> membershipIndex = membershipIndexRef.get();
    if (membershipIndex == null || membershipIndex.isEmpty()) {
      return Set.of();
    }
    LinkedHashSet<String> groups = new LinkedHashSet<>();
    for (String candidate : subjectCandidates) {
      Set<String> candidateGroups = membershipIndex.get(candidate);
      if (candidateGroups != null && !candidateGroups.isEmpty()) {
        groups.addAll(candidateGroups);
      }
    }
    return groups.isEmpty() ? Set.of() : Set.copyOf(groups);
  }

  private static Set<String> directSubjectCandidates(String subject) {
    String normalized = trimToNull(subject);
    if (normalized == null) {
      return Set.of();
    }
    return Set.of(normalized);
  }

  private static Set<String> membershipSubjectCandidates(String subject) {
    return EntitlementSubjectCandidates.membershipSubjectCandidates(subject);
  }

  private static String normalizeGroup(String value) {
    String normalized = trimToNull(value);
    if (normalized == null) {
      return null;
    }
    if (normalized.startsWith("group:") && normalized.length() > "group:".length()) {
      return normalized.substring("group:".length());
    }
    return normalized;
  }

  private static boolean isPlatformAdmin(Set<String> groups) {
    if (groups == null || groups.isEmpty()) {
      return false;
    }
    for (String group : groups) {
      String normalized = normalizeGroup(group);
      if (normalized != null && PLATFORM_GROUP.equalsIgnoreCase(normalized)) {
        return true;
      }
    }
    return false;
  }

  private static String summarizeBindings(List<ServiceEntitlementsSimpleManager.BindingSummary> bindings) {
    if (bindings == null || bindings.isEmpty()) {
      return "[]";
    }
    StringBuilder out = new StringBuilder("[");
    for (int i = 0; i < bindings.size(); i++) {
      ServiceEntitlementsSimpleManager.BindingSummary binding = bindings.get(i);
      if (i > 0) {
        out.append(", ");
      }
      out.append("{policyId=")
          .append(nullToDash(binding.policyId()))
          .append(", name=")
          .append(nullToDash(binding.policyName()))
          .append(", ")
          .append(binding.group() ? "group=" : "subject=")
          .append(nullToDash(binding.subject()))
          .append(", methodsAllow=")
          .append(binding.methodsAllow().isEmpty() ? "[]" : binding.methodsAllow())
          .append("}");
    }
    out.append(']');
    return out.toString();
  }

  private static String nullToDash(String value) {
    return value == null ? "-" : value;
  }

  private static String trimToNull(String value) {
    if (value == null) {
      return null;
    }
    String normalized = value.trim();
    return normalized.isEmpty() ? null : normalized;
  }

  private static String safeReason(RuntimeException error) {
    if (error == null) {
      return "unknown";
    }
    String message = trimToNull(error.getMessage());
    return message == null ? error.getClass().getSimpleName() : message;
  }

  private static String requireIdentitySubject(Identity identity) {
    String subject = identity.subject();
    if (subject == null || subject.isBlank()) {
      throw new IllegalArgumentException("identity.subject is required");
    }
    return subject;
  }

  private static String effectiveSubject(Identity identity, String subject) {
    if (identity.delegatedSubject() == null) {
      return subject;
    }
    if (identity.delegatedSubject().isBlank()) {
      throw new IllegalArgumentException("identity.delegatedSubject is blank");
    }
    return identity.delegatedSubject();
  }

  private enum MatchResult {
    MATCHED(true, true),
    SUBJECT_ONLY(false, true),
    NO_MATCH(false, false);

    private final boolean matched;
    private final boolean subjectMatched;

    MatchResult(boolean matched, boolean subjectMatched) {
      this.matched = matched;
      this.subjectMatched = subjectMatched;
    }

    private boolean matched() {
      return matched;
    }

    private boolean subjectMatched() {
      return subjectMatched;
    }
  }
}
