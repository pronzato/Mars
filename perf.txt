      throw new IllegalArgumentException(
          "RemoteTabularManager requires DataOptions service.enabled=true");

        configured.serviceOptions() == null

  boolean check() {
    RemoteFlightClientSupport.ServiceEndpoint endpoint = resolveQueryService();
    OtelManager otelRuntime = resolveClientRuntime(endpoint, true);
    Attributes otelAttrs = otelAttributes(endpoint, "arrow-flight-sql");
    long otelStartedAt = System.nanoTime();
    Span otelSpan =
        FabricOtelSupport.startSpan(
            otelRuntime, "fabric.flight.sql.client.check", SpanKind.CLIENT, otelAttrs);
    Throwable otelError = null;
    try (RemoteFlightClientSupport.ClientResources client =
        RemoteFlightClientSupport.openClient(
            endpoint, options, sessionKerberosPrincipal, sessionKerberosKeytab)) {
      CallOption[] callOptions =
          RemoteFlightClientSupport.buildCallOptions(
              callerApp, environment, region, site, client, null);
      for (FlightInfo ignored : client.flightClient().listFlights(Criteria.ALL, callOptions)) {
        break;
      }
      return true;
    } catch (Exception ex) {
      otelError = ex;
      return false;
    } finally {
      FabricOtelSupport.endSpan(otelRuntime, otelSpan, otelAttrs, otelStartedAt, otelError);
    }
  }

  @Override
  public ByteBuffer queryIpcBuffer(DataQuery query) {
    if (closed.get()) {
      throw new IllegalStateException("Remote tabular manager is closed");
    }
    Objects.requireNonNull(query, "query");
    int limit = query.limit() == null ? -1 : query.limit();
    ByteBuffer buffer = queryIpcBuffer(query.sql(), limit);
    assert buffer.isDirect();
    assert buffer.position() == 0;
    return buffer;
  }

  @Override
  public ByteBuffer queryIpcBuffer(String sql, int limit) {
    if (closed.get()) {
      throw new IllegalStateException("Remote tabular manager is closed");
    }
    String plannedSql = Strings.requireNonBlank(sql, "sql");
    Integer resolvedLimit = null;
    if (limit > 0) {
      resolvedLimit = limit;
    } else if (limit == 0) {
      throw new IllegalArgumentException("limit must be > 0");
    }
    if (resolvedLimit != null) {
      plannedSql = stripTrailingSemicolons(plannedSql) + " limit " + resolvedLimit;
    }

    RemoteFlightClientSupport.ServiceEndpoint endpoint = resolveQueryService();
    OtelManager otelRuntime = resolveClientRuntime(endpoint, true);
    Attributes otelAttrs =
        otelAttributesWithStatementType(
            endpoint, "arrow-flight-sql", statementType(plannedSql));
    long otelStartedAt = System.nanoTime();
    Span otelSpan =
        FabricOtelSupport.startSpan(
            otelRuntime, "fabric.flight.sql.client.query", SpanKind.CLIENT, otelAttrs);
    Throwable otelError = null;
    try (RemoteFlightClientSupport.ClientResources client =
            RemoteFlightClientSupport.openClient(
                endpoint, options, sessionKerberosPrincipal, sessionKerberosKeytab);
        FlightSqlClient sqlClient = new FlightSqlClient(client.flightClient())) {
      CallOption[] callOptions =
          RemoteFlightClientSupport.buildCallOptions(
              callerApp, environment, region, site, client, null);
      FlightInfo info = sqlClient.execute(plannedSql, callOptions);
      ByteBuffer buffer = toArrowIpcBuffer(client.flightClient(), client.allocator(), info, callOptions);
      assert buffer.isDirect();
      assert buffer.position() == 0;
      return buffer;
    } catch (Exception ex) {
      otelError = ex;
      String kerberosDetails =
          RemoteFlightClientSupport.kerberosAuthFailureDetails(
              endpoint, ex, sessionKerberosPrincipal);
      String endpointContext = RemoteFlightClientSupport.endpointSecurityContext(endpoint);
      String clientDiagnostics =
          RemoteFlightClientSupport.clientDiagnostics(
              endpoint, options, sessionKerberosPrincipal, sessionKerberosKeytab, null);
      String detailSuffix = kerberosDetails == null ? "" : " " + kerberosDetails;
      LOGGER.error("[RemoteTabularManager][queryIpcSql][failed] {} sql={} diagnostics={} throwableChain={}", endpointContext, summarizeSqlForLogs(plannedSql), clientDiagnostics, RemoteFlightClientSupport.summarizeThrowable(ex), ex);
      throw new IllegalStateException(
          "Remote query failed ["
              + endpointContext
              + "] sql="
              + plannedSql
              + ": "
              + ex.getMessage()
              + detailSuffix,
          ex);
    } finally {
      FabricOtelSupport.endSpan(otelRuntime, otelSpan, otelAttrs, otelStartedAt, otelError);
    }
  }

  @Override
  public TabularStreamSubscription subscribeDataset(
      org.pronzato.fabric.api.data.dataset.DatasetId datasetId, TabularStreamOptions options) {
    if (closed.get()) {
      throw new IllegalStateException("Remote tabular manager is closed");
    }
    Objects.requireNonNull(datasetId, "datasetId");
    TabularStreamOptions resolved = options == null ? TabularStreamOptions.defaults() : options;
    RemoteFlightClientSupport.ServiceEndpoint endpoint = resolveQueryService();
    OtelManager otelRuntime = resolveClientRuntime(endpoint, true);
    Attributes otelAttrs = otelAttributes(endpoint, "arrow-flight-sql");
    long otelStartedAt = System.nanoTime();
    Span otelSpan =
        FabricOtelSupport.startSpan(
            otelRuntime, "fabric.flight.sql.client.subscribe", SpanKind.CLIENT, otelAttrs);
    Throwable otelError = null;
    RemoteFlightClientSupport.ClientResources client = null;
    try {
      client =
          RemoteFlightClientSupport.openClient(
              endpoint, this.options, sessionKerberosPrincipal, sessionKerberosKeytab);
      boolean includeDelegatedUser = !Boolean.TRUE.equals(resolved.publicAccess());
      CallOption[] callOptions =
          RemoteFlightClientSupport.buildCallOptions(
              callerApp,
              environment,
              region,
              site,
              client,
              datasetId.toString(),
              includeDelegatedUser);
      Ticket ticket = FlightDatasetStreamTicketCodec.encode(toStreamTicket(datasetId, resolved));
      FlightStream stream = client.flightClient().getStream(ticket, callOptions);
      return new RemoteTabularStreamSubscription(client, stream);
    } catch (Exception ex) {
      otelError = ex;
      if (client != null) {
        client.close();
      }
      String endpointContext = RemoteFlightClientSupport.endpointSecurityContext(endpoint);
      throw new IllegalStateException(
          "Remote stream subscription failed ["
              + endpointContext
              + "] dataset="
              + datasetId
              + ": "
              + ex.getMessage(),
          ex);
    } finally {
      FabricOtelSupport.endSpan(otelRuntime, otelSpan, otelAttrs, otelStartedAt, otelError);
    }
  }

  @Override
  public void close() {
    if (!closed.compareAndSet(false, true)) {
      return;
    }
    ipcDecodeAllocator.close();
  }

  private OtelManager resolveClientRuntime(
      RemoteFlightClientSupport.ServiceEndpoint endpoint, boolean flightSql) {
    OtelManager runtime = session.telemetry();
    DefinitionService service = endpoint == null ? null : endpoint.definition();
    if (flightSql) {
      return FabricOtelEnablement.isArrowFlightSqlClientTracingEnabled(service)
          ? runtime
          : null;
    }
    return FabricOtelEnablement.isArrowFlightClientTracingEnabled(service)
        ? runtime
        : null;
  }

  private Attributes otelAttributes(
      RemoteFlightClientSupport.ServiceEndpoint endpoint, String transport) {
    String endpointAddress =
        endpoint == null ? null : endpoint.connectHost() + ":" + endpoint.port();
    String serviceName = endpoint == null ? null : endpoint.serviceName();
    String serviceGroup = endpoint == null ? null : endpoint.group();
    return FabricOtelAttributes.common(
        serviceGroup,
        callerApp == null ? null : callerApp.name(),
        serviceName,
        transport,
        callerApp == null ? null : callerApp.asSubject(),
        endpointAddress);
  }

  private Attributes otelAttributesWithStatementType(
      RemoteFlightClientSupport.ServiceEndpoint endpoint, String transport, String statementType) {
    Attributes base = otelAttributes(endpoint, transport);
    String normalizedType = Strings.trimToNull(statementType);
    if (normalizedType == null) {
      return base;
    }
    return Attributes.builder().putAll(base).put(FABRIC_STATEMENT_TYPE, normalizedType).build();
  }

  private RemoteFlightClientSupport.ServiceEndpoint resolveQueryService() {

  }

  private ServiceRoutingContext currentRoutingContext() {
    UserPrincipal principal = UserContext.get().orElse(null);
    CallerAppContext.ContextValue callerContext = CallerAppContext.get().orElse(null);
    String effectiveSubject =
        Strings.firstNonBlank(
            principal == null ? null : Strings.trimToNull(principal.username()),
            sessionKerberosPrincipal);
    String callerAppSubject =
        Strings.firstNonBlank(
            callerContext == null ? null : callerContext.callerAppSubject(),
            callerApp == null ? null : callerApp.asSubject());
    return new ServiceRoutingContext(
        null,
        "ARROW_FLIGHT_SQL",
        effectiveSubject,
        callerAppSubject,
        options.routingKey(),
        callerContext == null ? null : callerContext.requestId(),
        null);
  }

  private static String stripTrailingSemicolons(String sql) {
    String value = Strings.requireNonBlank(sql, "sql");
    while (value.endsWith(";")) {
      value = value.substring(0, value.length() - 1).trim();
    }
    return value;
  }

  private static String summarizeSqlForLogs(String sql) {
    String normalized = Strings.trimToNull(sql);
    if (normalized == null) {
      return "<empty>";
    }
    if (normalized.length() > 512) {
      return normalized.substring(0, 512) + "...";
    }
    return normalized;
  }

  private static String statementType(String sql) {
    String normalized = Strings.trimToNull(sql);
    if (normalized == null) {
      return null;
    }
    int firstSpace = normalized.indexOf(' ');
    String token = firstSpace < 0 ? normalized : normalized.substring(0, firstSpace);
    return Strings.trimToNull(token == null ? null : token.toUpperCase(java.util.Locale.ROOT));
  }

  private static FlightDatasetStreamTicket toStreamTicket(
      org.pronzato.fabric.api.data.dataset.DatasetId datasetId, TabularStreamOptions options) {
    FlightDatasetStreamTicket.Builder builder =
        FlightDatasetStreamTicket.builder()
            .withDatasetId(datasetId)
            .withBatchMaxRows(options.batchMaxRows())
            .withFlushMillis(options.flushMillis())
            .withSubscriptionId(options.subscriptionId())
            .withFilterSql(options.filterSql());
    if (options.publicAccess() != null) {
      builder.withPublic(options.publicAccess());
    }
    if (options.start() == TabularStreamOptions.Start.BEGINNING) {
      builder.withStart(FlightDatasetStreamTicket.Start.BEGINNING);
    } else if (options.start() == TabularStreamOptions.Start.RESUME) {
      builder.withStart(FlightDatasetStreamTicket.Start.RESUME);
    } else if (options.start() == TabularStreamOptions.Start.FROM_SEQUENCE) {
      builder.withFromSequence(options.fromSequence() == null ? 0L : options.fromSequence());
    } else {
      builder.withStart(FlightDatasetStreamTicket.Start.LATEST);
    }
    return builder.build();
  }

  private static List<TabularTableMetadata> readTableMetadataRows(
      FlightClient client, FlightInfo info, CallOption[] callOptions) throws Exception {
    if (info == null) {
      return List.of();
    }
    List<TabularTableMetadata> rows = new ArrayList<>();
    for (FlightEndpoint endpoint : info.getEndpoints()) {
      if (endpoint == null || endpoint.getTicket() == null) {
        continue;
      }
      try (FlightStream stream = client.getStream(endpoint.getTicket(), callOptions)) {
        while (stream.next()) {
          VectorSchemaRoot root = stream.getRoot();
          for (int row = 0; row < root.getRowCount(); row++) {
            rows.add(
                new TabularTableMetadata(
                    stringValue(root, "catalog_name", row),
                    stringValue(root, "db_schema_name", row),
                    stringValue(root, "table_name", row),
                    stringValue(root, "table_type", row)));
          }
        }
      }
    }
    rows.sort(
        Comparator.comparing(TabularTableMetadata::schemaName, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(TabularTableMetadata::schemaName)
            .thenComparing(TabularTableMetadata::tableName, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(TabularTableMetadata::tableName)
            .thenComparing(TabularTableMetadata::tableType, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(TabularTableMetadata::tableType));
    return List.copyOf(rows);
  }

  private static List<String> readSchemaNames(
      FlightClient client, FlightInfo info, CallOption[] callOptions) throws Exception {
    if (info == null) {
      return List.of();
    }
    LinkedHashSet<String> names = new LinkedHashSet<>();
    for (FlightEndpoint endpoint : info.getEndpoints()) {
      if (endpoint == null || endpoint.getTicket() == null) {
        continue;
      }
      try (FlightStream stream = client.getStream(endpoint.getTicket(), callOptions)) {
        while (stream.next()) {
          VectorSchemaRoot root = stream.getRoot();
          for (int row = 0; row < root.getRowCount(); row++) {
            String value = stringValue(root, "db_schema_name", row);
            if (value != null) {
              names.add(value);
            }
          }
        }
      }
    }
    List<String> sorted = new ArrayList<>(names);
    sorted.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
    return List.copyOf(sorted);
  }

  private static byte[] readSingleTableSchema(
      FlightClient client,
      FlightInfo info,
      CallOption[] callOptions,
      org.pronzato.fabric.api.data.dataset.DatasetId datasetId)
      throws Exception {
    List<byte[]> schemas = new ArrayList<>();
    if (info == null) {
      throw new IllegalStateException("Remote table schema not found: " + datasetId);
    }
    for (FlightEndpoint endpoint : info.getEndpoints()) {
      if (endpoint == null || endpoint.getTicket() == null) {
        continue;
      }
      try (FlightStream stream = client.getStream(endpoint.getTicket(), callOptions)) {
        while (stream.next()) {
          VectorSchemaRoot root = stream.getRoot();
          for (int row = 0; row < root.getRowCount(); row++) {
            Object value = root.getVector("table_schema").getObject(row);
            if (value instanceof byte[] bytes && bytes.length > 0) {
              schemas.add(bytes);
            }
          }
        }
      }
    }
    if (schemas.isEmpty()) {
      throw new IllegalStateException("Remote table schema not found: " + datasetId);
    }
    return schemas.get(0);
  }

  private static String stringValue(VectorSchemaRoot root, String fieldName, int row) {
    if (root == null || root.getVector(fieldName) == null) {
      return null;
    }
    Object value = root.getVector(fieldName).getObject(row);
    if (value == null) {
      return null;
    }
    if (value instanceof Text text) {
      return text.toString();
    }
    return String.valueOf(value);
  }

  private static final class RemoteTabularStreamSubscription implements TabularStreamSubscription {
    private final RemoteFlightClientSupport.ClientResources client;
    private final FlightStream stream;
    private final AtomicBoolean closed = new AtomicBoolean(false);

    private RemoteTabularStreamSubscription(
        RemoteFlightClientSupport.ClientResources client, FlightStream stream) {
      this.client = Objects.requireNonNull(client, "client");
      this.stream = Objects.requireNonNull(stream, "stream");
    }

    @Override
    public FlightStream stream() {
      return stream;
    }

    @Override
    public void close() {
      if (!closed.compareAndSet(false, true)) {
        return;
      }
      try {
        stream.close();
      } catch (Exception ignore) {
      }
      client.close();
    }
  }

  private static ByteBuffer toArrowIpcBuffer(
      FlightClient client, BufferAllocator allocator, FlightInfo info, CallOption[] callOptions) {
    if (info == null) {
      throw new IllegalStateException("Remote query returned no FlightInfo");
    }
    List<FlightEndpoint> endpoints =
        info.getEndpoints() == null ? List.of() : List.copyOf(info.getEndpoints());
    Schema schema = info.getSchema();
    try (Bytes.DirectBufferOutputStream out = Bytes.directOutputStream()) {
      VectorSchemaRoot writeRoot = null;
      VectorLoader loader = null;
      ArrowStreamWriter writer = null;
      boolean started = false;
      try {
        for (FlightEndpoint endpoint : endpoints) {
          if (endpoint == null || endpoint.getTicket() == null) {
            continue;
          }
          try {
            try (FlightStream stream = client.getStream(endpoint.getTicket(), callOptions)) {
              VectorSchemaRoot streamRoot = stream.getRoot();
              if (schema == null && streamRoot != null) {
                schema = streamRoot.getSchema();
              }
              if (!started) {
                Schema effectiveSchema = schema;
                if (effectiveSchema == null && streamRoot != null) {
                  effectiveSchema = streamRoot.getSchema();
                }
                if (effectiveSchema == null) {
                  throw new IllegalStateException("Remote query response schema is required");
                }
                writeRoot = VectorSchemaRoot.create(effectiveSchema, allocator);
                loader = new VectorLoader(writeRoot);
                writer = new ArrowStreamWriter(writeRoot, null, out);
                writer.start();
                started = true;
              }
              if (streamRoot == null) {
                continue;
              }
              while (stream.next()) {
                try (ArrowRecordBatch batch = new VectorUnloader(streamRoot).getRecordBatch()) {
                  loader.load(batch);
                  writer.writeBatch();
                }
              }
            }
          } catch (Exception ex) {
            throw new IllegalStateException("Failed to read remote query endpoint stream", ex);
          }
        }

        if (!started) {
          if (schema == null) {
            throw new IllegalStateException("Remote query response schema is required");
          }
          writeRoot = VectorSchemaRoot.create(schema, allocator);
          writer = new ArrowStreamWriter(writeRoot, null, out);
          writer.start();
        }
        writer.end();
        ByteBuffer buffer = out.toDirectByteBuffer();
        assert buffer.isDirect();
        assert buffer.position() == 0;
        return buffer;
      } finally {
        if (writer != null) {
          try {
            writer.close();
          } catch (Exception ignore) {
          }
        }
        if (writeRoot != null) {
          try {
            writeRoot.close();
          } catch (Exception ignore) {
          }
        }
      }
    } catch (IOException ex) {
      throw new IllegalStateException("Failed to serialize remote Arrow IPC stream", ex);
    }
  }
}
