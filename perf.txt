import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Comparator;
import java.util.LinkedHashMap;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import java.util.concurrent.atomic.AtomicReference;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryCache;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryListener;
import org.pronzato.fabric.api.control.discovery.definitions.Definition;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEntitlement;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEnvelope;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionPlatform;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionType;
import org.pronzato.fabric.api.control.entitlements.core.services.ServiceEntitlements;
import org.pronzato.fabric.api.control.entitlements.provider.DefaultEntitlementProviderRegistry;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementProviderRegistry;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementProviderSelector;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementSubjectProvider;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementSubjectProviderPlatform;
import org.pronzato.fabric.api.util.Strings;

public final class ServiceEntitlementsSimpleManager implements AutoCloseable {
  private static final String PLATFORM_GROUP = "platform";
  private static final String PLATFORM_NAME = "platform";

  private final DiscoveryCache DiscoveryCache;
  private final String sessionGroup;
  private final EntitlementSubjectProvider provider;
  private final DiscoveryListener listener;
  private final ServiceEntitlementsSimple serviceEntitlements;
  private final AtomicReference<Map<String, Set<String>>> membershipIndexRef =
      new AtomicReference<>(Map.of());
  private final AtomicReference<Map<String, Rules>> entitlementsIndexRef =
      new AtomicReference<>(Map.of());
  private final AtomicReference<String> platformModelVersionKeyRef = new AtomicReference<>("");
  private final AtomicReference<String> entitlementsVersionKeyRef = new AtomicReference<>("");
  private final Object rebuildLock = new Object();

  public ServiceEntitlementsSimpleManager(DiscoveryCache DiscoveryCache, String sessionGroup, boolean logDecisions) {
    this(
        DiscoveryCache,
        sessionGroup,
        logDecisions,
        new DefaultEntitlementProviderRegistry(DiscoveryCache));
  }

  public ServiceEntitlementsSimpleManager(
      DiscoveryCache DiscoveryCache,
      String sessionGroup,
      boolean logDecisions,
      EntitlementSubjectProvider provider) {
    this(
        DiscoveryCache,
        sessionGroup,
        logDecisions,
        new DefaultEntitlementProviderRegistry(provider));
  }

  public ServiceEntitlementsSimpleManager(
      DiscoveryCache DiscoveryCache,
      String sessionGroup,
      boolean logDecisions,
      EntitlementProviderRegistry providerRegistry) {
    this(
        DiscoveryCache,
        sessionGroup,
        logDecisions,
        providerRegistry,
        new EntitlementProviderSelector(DiscoveryCache, providerRegistry));
  }

  public ServiceEntitlementsSimpleManager(
      DiscoveryCache DiscoveryCache,
      String sessionGroup,
      boolean logDecisions,
      EntitlementProviderRegistry providerRegistry,
      EntitlementProviderSelector providerSelector) {
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.sessionGroup = sessionGroup;
    Objects.requireNonNull(providerRegistry, "providerRegistry");
    Objects.requireNonNull(providerSelector, "providerSelector");
    this.provider = Objects.requireNonNull(providerRegistry.defaultProvider(), "providerRegistry.defaultProvider");
    this.serviceEntitlements =
        new ServiceEntitlementsSimple(
            membershipIndexRef, entitlementsIndexRef, providerSelector, logDecisions);
    this.listener =
        new DiscoveryListener() {
          @Override
          public void onUpsert(Definition definition) {
            handleUpsert(definition);
          }

          @Override
          public void onDelete(DefinitionEnvelope envelope) {
            handleDelete(envelope);
          }
        };

    this.DiscoveryCache.addListener(listener);
    try {
      rebuildMembershipIndex(true);
      rebuildEntitlementsIndex(true);
    } catch (RuntimeException ex) {
      this.DiscoveryCache.removeListener(listener);
      throw ex;
    }
  }

  public ServiceEntitlementsSimple serviceEntitlements() {
    return serviceEntitlements;
  }

  @Override
  public void close() {
    DiscoveryCache.removeListener(listener);
  }

  private void handleUpsert(Definition definition) {
    if (definition instanceof DefinitionPlatform platformModel
        && isPlatformModel(platformModel.envelope())) {
      rebuildMembershipIndex(false);
      return;
    }
    if (definition instanceof DefinitionEntitlement entitlement && isInScope(entitlement.envelope())) {
      rebuildEntitlementsIndex(false);
    }
  }

  private void handleDelete(DefinitionEnvelope envelope) {
    if (envelope == null || envelope.type() == null) {
      return;
    }
    if (envelope.type() == DefinitionType.PLATFORM && isPlatformModel(envelope)) {
      rebuildMembershipIndex(false);
      return;
    }
    if (envelope.type() == DefinitionType.ENTITLEMENT && isInScope(envelope)) {
      rebuildEntitlementsIndex(false);
    }
  }

  private void rebuildMembershipIndex(boolean required) {
    synchronized (rebuildLock) {
      String nextVersionKey = provider.versionKey();
      if ("MISSING".equals(nextVersionKey)) {
        if (required) {
          throw new IllegalStateException(
              "Missing required platform model " + PLATFORM_GROUP + "/" + PLATFORM_NAME);
        }
        membershipIndexRef.set(Map.of());
        platformModelVersionKeyRef.set(nextVersionKey);
        return;
      }
      if (nextVersionKey.equals(platformModelVersionKeyRef.get())) {
        return;
      }
      membershipIndexRef.set(membershipIndexFromProvider(required));
      platformModelVersionKeyRef.set(nextVersionKey);
    }
  }

  private Map<String, Set<String>> membershipIndexFromProvider(boolean required) {
    if (provider instanceof EntitlementSubjectProviderPlatform platformProvider) {
      return platformProvider.membershipIndex();
    }
    return Map.of();
  }

  private void rebuildEntitlementsIndex(boolean required) {
    synchronized (rebuildLock) {
      List<DefinitionEntitlement> assetEntitlements = collectAssetEntitlements();
      if (assetEntitlements.isEmpty()) {
        entitlementsIndexRef.set(Map.of());
        entitlementsVersionKeyRef.set("MISSING");
        return;
      }

      String nextVersionKey = entitlementsVersionKey(assetEntitlements);
      if (nextVersionKey.equals(entitlementsVersionKeyRef.get())) {
        return;
      }

      entitlementsIndexRef.set(compileRulesIndex(assetEntitlements));
      entitlementsVersionKeyRef.set(nextVersionKey);
    }
  }

  private List<DefinitionEntitlement> collectAssetEntitlements() {
    List<DefinitionEntitlement> all = DiscoveryCache.list(DefinitionEntitlement.class);
    List<DefinitionEntitlement> assetEntitlements = new ArrayList<>();
    for (DefinitionEntitlement entitlement : all) {
      if (entitlement == null || !enabled(entitlement.envelope() == null ? null : entitlement.envelope().enabled())) {
        continue;
      }
      if (!isInScope(entitlement.envelope())) {
        continue;
      }
      if (canonicalAssetScopeId(entitlement.scopeId()) == null) {
        continue;
      }
      assetEntitlements.add(entitlement);
    }
    return List.copyOf(assetEntitlements);
  }

  static Map<String, Rules> compileRulesIndex(List<DefinitionEntitlement> definitions) {
    Map<String, MutableRules> mutable = new LinkedHashMap<>();
    for (DefinitionEntitlement entitlement : definitions) {
      String scopeId = canonicalAssetScopeId(entitlement.scopeId());
      if (scopeId == null) {
        continue;
      }
      MutableRules target =
          mutable.computeIfAbsent(scopeId, ignored -> new MutableRules());
      appendPolicyRules(target, entitlement.policies());
      appendRelationshipRules(target, entitlement.relationships(), scopeId);
    }

    Map<String, Rules> built = new LinkedHashMap<>();
    for (Map.Entry<String, MutableRules> entry : mutable.entrySet()) {
      built.put(entry.getKey(), entry.getValue().freeze());
    }
    return Map.copyOf(built);
  }

  private static void appendPolicyRules(
      MutableRules rules, List<DefinitionEntitlement.PolicyDocument> policies) {
    if (rules == null || policies == null) {
      return;
    }
    for (DefinitionEntitlement.PolicyDocument policy : policies) {
      if (policy == null || !enabled(policy.enabled()) || policy.bindings() == null) {
        continue;
      }
      String policyId = normalize(policy.policyId());
      String policyName = normalize(policy.name());
      rules.addPolicy(policyId, policyName);
      for (DefinitionEntitlement.PolicyBinding binding : policy.bindings()) {
        if (binding == null || !enabled(binding.enabled())) {
          continue;
        }
        List<ActionSlot> slots = actionSlots(binding.action());
        if (slots.isEmpty()) {
          continue;
        }
        SubjectSlot subjectSlot = subjectSlot(binding.subject());
        if (subjectSlot == null) {
          continue;
        }
        Set<String> methodsAllow = methodsAllow(binding.obligations());
        BindingSummary summary =
            new BindingSummary(policyId, policyName, subjectSlot.id(), subjectSlot.group(), methodsAllow);
        rules.addBindingSummary(slots, summary);
        String effect = binding.effect();
        if ("DENY".equalsIgnoreCase(effect)) {
          for (ActionSlot slot : slots) {
            rules.addDeny(slot, subjectSlot, methodsAllow);
          }
          continue;
        }
        if ("ALLOW".equalsIgnoreCase(effect)) {
          for (ActionSlot slot : slots) {
            rules.addAllow(slot, subjectSlot, methodsAllow);
          }
        }
      }
    }
  }

  private static void appendRelationshipRules(
      MutableRules rules,
      List<DefinitionEntitlement.RelationshipTuple> relationships,
      String resourceId) {
    if (rules == null || relationships == null || resourceId == null) {
      return;
    }
    for (DefinitionEntitlement.RelationshipTuple tuple : relationships) {
      if (tuple == null) {
        continue;
      }
      String tupleResourceId = canonicalAssetScopeId(tuple.resourceId());
      String tupleResourceSubject = canonicalAssetScopeId(tuple.resourceSubject());
      if (!Objects.equals(resourceId, tupleResourceId) && !Objects.equals(resourceId, tupleResourceSubject)) {
        continue;
      }
      SubjectSlot subjectSlot = subjectSlot(tuple.subject());
      if (subjectSlot == null) {
        continue;
      }
      String relation = tuple.relation();
      if (relation == null) {
        continue;
      }
      if ("DENY".equalsIgnoreCase(relation) || "service.deny".equalsIgnoreCase(relation)) {
        rules.addDeny(ActionSlot.START, subjectSlot, Set.of());
        rules.addDeny(ActionSlot.INVOKE, subjectSlot, Set.of());
        continue;
      }
      if ("ALLOW".equalsIgnoreCase(relation) || "service.allow".equalsIgnoreCase(relation)) {
        rules.addAllow(ActionSlot.START, subjectSlot, Set.of());
        rules.addAllow(ActionSlot.INVOKE, subjectSlot, Set.of());
        continue;
      }
      for (ActionSlot slot : actionSlots(relation)) {
        rules.addAllow(slot, subjectSlot, Set.of());
      }
    }
  }

  static List<ActionSlot> actionSlots(String action) {
    if (action == null) {
      return List.of();
    }
    if ("*".equals(action)) {
      return List.of(ActionSlot.START, ActionSlot.INVOKE, ActionSlot.READ);
    }
    if ("service.start".equalsIgnoreCase(action) || "START".equalsIgnoreCase(action)) {
      return List.of(ActionSlot.START);
    }
    if ("service.invoke".equalsIgnoreCase(action)
        || "INVOKE".equalsIgnoreCase(action)
        || "WRITE".equalsIgnoreCase(action)) {
      return List.of(ActionSlot.INVOKE);
    }
    if ("service.read".equalsIgnoreCase(action) || "READ".equalsIgnoreCase(action)) {
      return List.of(ActionSlot.READ);
    }
    return List.of();
  }

  private static SubjectSlot subjectSlot(String subject) {
    String normalized = normalize(subject);
    if (normalized == null) {
      return null;
    }
    if (normalized.startsWith("group:")) {
      return new SubjectSlot(normalized, true);
    }
    return new SubjectSlot(normalized, false);
  }

  private static Set<String> methodsAllow(DefinitionEntitlement.ObligationSet obligations) {
    if (obligations == null || obligations.methodsAllow() == null || obligations.methodsAllow().isEmpty()) {
      return Set.of();
    }
    LinkedHashSet<String> normalized = new LinkedHashSet<>();
    for (String method : obligations.methodsAllow()) {
      String value = normalize(method);
      if (value != null) {
        normalized.add(value);
      }
    }
    return normalized.isEmpty() ? Set.of() : Set.copyOf(normalized);
  }

  private static String normalize(String value) {
    return Strings.trimToNull(value);
  }

  private static boolean isPlatformModel(DefinitionEnvelope envelope) {
    if (envelope == null) {
      return false;
    }
    return Objects.equals(PLATFORM_GROUP, envelope.group()) && Objects.equals(PLATFORM_NAME, envelope.name());
  }

  private boolean isInScope(DefinitionEnvelope envelope) {
    return envelope != null && normalize(envelope.group()) != null;
  }

  private static String canonicalAssetScopeId(String scopeId) {
    if (scopeId == null) {
      return null;
    }
    try {
      return ServiceEntitlements.requireAssetResourceId(scopeId);
    } catch (RuntimeException ex) {
      return null;
    }
  }

  private static String entitlementsVersionKey(List<DefinitionEntitlement> definitions) {
    if (definitions == null || definitions.isEmpty()) {
      return "EMPTY";
    }
    List<String> keys = new ArrayList<>(definitions.size());
    for (DefinitionEntitlement definition : definitions) {
      if (definition == null) {
        continue;
      }
      DefinitionEnvelope envelope = definition.envelope();
      String group = envelope == null ? null : envelope.group();
      String name = envelope == null ? null : envelope.name();
      String uuid = envelope == null ? null : envelope.uuid();
      String updated = envelope == null ? null : envelope.updatedTimeUtc();
      keys.add(
          String.valueOf(group)
              + "/"
              + String.valueOf(name)
              + "|"
              + String.valueOf(uuid)
              + "|"
              + String.valueOf(updated)
              + "|"
              + definition.hashCode());
    }
    keys.sort(Comparator.naturalOrder());
    return String.join(";", keys);
  }

  private static boolean enabled(Boolean value) {
    return value == null || value.booleanValue();
  }

  enum ActionSlot {
    START,
    INVOKE,
    READ
  }

  static final class Rules {
    private final RuleSet start;
    private final RuleSet invoke;
    private final RuleSet read;
    private final int policyCount;

    private Rules(RuleSet start, RuleSet invoke, RuleSet read, int policyCount) {
      this.start = start;
      this.invoke = invoke;
      this.read = read;
      this.policyCount = policyCount;
    }

    RuleSet forAction(ActionSlot actionSlot) {
      return switch (actionSlot) {
        case START -> start;
        case INVOKE -> invoke;
        case READ -> read;
      };
    }

    int policyCount() {
      return policyCount;
    }
  }

  static final class RuleSet {
    private final Set<String> allowSubjects;
    private final Set<String> allowGroups;
    private final Set<String> denySubjects;
    private final Set<String> denyGroups;
    private final List<BindingRule> allowBindings;
    private final List<BindingRule> denyBindings;
    private final List<BindingSummary> bindingSummaries;

    private RuleSet(
        Set<String> allowSubjects,
        Set<String> allowGroups,
        Set<String> denySubjects,
        Set<String> denyGroups,
        List<BindingRule> allowBindings,
        List<BindingRule> denyBindings,
        List<BindingSummary> bindingSummaries) {
      this.allowSubjects = allowSubjects;
      this.allowGroups = allowGroups;
      this.denySubjects = denySubjects;
      this.denyGroups = denyGroups;
      this.allowBindings = allowBindings;
      this.denyBindings = denyBindings;
      this.bindingSummaries = bindingSummaries;
    }

    Set<String> allowSubjects() {
      return allowSubjects;
    }

    Set<String> allowGroups() {
      return allowGroups;
    }

    Set<String> denySubjects() {
      return denySubjects;
    }

    Set<String> denyGroups() {
      return denyGroups;
    }

    List<BindingRule> allowBindings() {
      return allowBindings;
    }

    List<BindingRule> denyBindings() {
      return denyBindings;
    }

    List<BindingSummary> bindingSummaries() {
      return bindingSummaries;
    }
  }

  private static final class MutableRules {
    private final MutableRuleSet start = new MutableRuleSet();
    private final MutableRuleSet invoke = new MutableRuleSet();
    private final MutableRuleSet read = new MutableRuleSet();
    private final Set<String> policies = new LinkedHashSet<>();

    private void addAllow(ActionSlot actionSlot, SubjectSlot subject, Set<String> methodsAllow) {
      target(actionSlot).addAllow(subject, methodsAllow);
    }

    private void addDeny(ActionSlot actionSlot, SubjectSlot subject, Set<String> methodsAllow) {
      target(actionSlot).addDeny(subject, methodsAllow);
    }

    private void addPolicy(String policyId, String policyName) {
      policies.add((policyId == null ? "-" : policyId) + "|" + (policyName == null ? "-" : policyName));
    }

    private void addBindingSummary(List<ActionSlot> actionSlots, BindingSummary summary) {
      if (actionSlots == null || actionSlots.isEmpty() || summary == null) {
        return;
      }
      for (ActionSlot actionSlot : actionSlots) {
        target(actionSlot).addBindingSummary(summary);
      }
    }

    private MutableRuleSet target(ActionSlot actionSlot) {
      return switch (actionSlot) {
        case START -> start;
        case INVOKE -> invoke;
        case READ -> read;
      };
    }

    private Rules freeze() {
      return new Rules(start.freeze(), invoke.freeze(), read.freeze(), policies.size());
    }
  }

  private static final class MutableRuleSet {
    private final Set<String> allowSubjects = new LinkedHashSet<>();
    private final Set<String> allowGroups = new LinkedHashSet<>();
    private final Set<String> denySubjects = new LinkedHashSet<>();
    private final Set<String> denyGroups = new LinkedHashSet<>();
    private final List<BindingRule> allowBindings = new ArrayList<>();
    private final List<BindingRule> denyBindings = new ArrayList<>();
    private final List<BindingSummary> bindingSummaries = new ArrayList<>();

    private void addAllow(SubjectSlot subject, Set<String> methodsAllow) {
      allowBindings.add(new BindingRule(subject.id(), subject.group(), methodsAllow));
      if (subject.group()) {
        allowGroups.add(subject.id());
      } else {
        allowSubjects.add(subject.id());
      }
    }

    private void addDeny(SubjectSlot subject, Set<String> methodsAllow) {
      denyBindings.add(new BindingRule(subject.id(), subject.group(), methodsAllow));
      if (subject.group()) {
        denyGroups.add(subject.id());
      } else {
        denySubjects.add(subject.id());
      }
    }

    private void addBindingSummary(BindingSummary summary) {
      bindingSummaries.add(summary);
    }

    private RuleSet freeze() {
      return new RuleSet(
          immutable(allowSubjects),
          immutable(allowGroups),
          immutable(denySubjects),
          immutable(denyGroups),
          immutableList(allowBindings),
          immutableList(denyBindings),
          immutableList(bindingSummaries));
    }
  }

  private static Set<String> immutable(Collection<String> values) {
    if (values == null || values.isEmpty()) {
      return Set.of();
    }
    return Collections.unmodifiableSet(new LinkedHashSet<>(values));
  }

  private static <T> List<T> immutableList(Collection<T> values) {
    if (values == null || values.isEmpty()) {
      return List.of();
    }
    return Collections.unmodifiableList(new ArrayList<>(values));
  }

  static final class BindingRule {
    private final String subject;
    private final boolean group;
    private final Set<String> methodsAllow;

    private BindingRule(String subject, boolean group, Set<String> methodsAllow) {
      this.subject = subject;
      this.group = group;
      this.methodsAllow = methodsAllow == null || methodsAllow.isEmpty() ? Set.of() : Set.copyOf(methodsAllow);
    }

    String subject() {
      return subject;
    }

    boolean group() {
      return group;
    }

    Set<String> methodsAllow() {
      return methodsAllow;
    }
  }

  static final class BindingSummary {
    private final String policyId;
    private final String policyName;
    private final String subject;
    private final boolean group;
    private final Set<String> methodsAllow;

    private BindingSummary(
        String policyId, String policyName, String subject, boolean group, Set<String> methodsAllow) {
      this.policyId = policyId;
      this.policyName = policyName;
      this.subject = subject;
      this.group = group;
      this.methodsAllow = methodsAllow == null || methodsAllow.isEmpty() ? Set.of() : Set.copyOf(methodsAllow);
    }

    String policyId() {
      return policyId;
    }

    String policyName() {
      return policyName;
    }

    String subject() {
      return subject;
    }

    boolean group() {
      return group;
    }

    Set<String> methodsAllow() {
      return methodsAllow;
    }
  }

  private record SubjectSlot(String id, boolean group) {}
}
