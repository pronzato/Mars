  public IngestManagerImpl(
      RowManager rowManager,
      DiscoveryCache DiscoveryCache,
      String group,
      Fabric.Session session,
      ScheduledExecutorService rollScheduler,
      CheckpointStore checkpointStore) {
    this.rowManager = Objects.requireNonNull(rowManager, "rowManager");
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.group = Strings.requireNonBlank(group, "session group");
    this.serviceMode = rowManager instanceof RemoteRowManager;
    this.runtimeMode = serviceMode ? MODE_SERVICE : MODE_DIRECT;
    this.rollScheduler = rollScheduler;
    this.schemaManager =
        session == null
            ? new SchemaManagerImpl(() -> this.DiscoveryCache, () -> this.group, () -> null)
            : Objects.requireNonNull(session.schemas(), "session.schemas");
    this.modelMapper = new IngestModelMapper();
    this.plugins = FabricPlugins.system();
    this.session = session;
    DeleteManager sessionDeleteManager =
        session instanceof SessionImpl sessionImpl ? sessionImpl.deletes() : null;
    this.deleteManager =
        sessionDeleteManager != null
            ? sessionDeleteManager
            : new DeleteManager(
                () -> this.DiscoveryCache.allDefinitions(),
                (type, definitionGroup, definitionName) ->
                    this.DiscoveryCache.delete(type, definitionGroup, definitionName));
    this.checkpointStore = Objects.requireNonNull(checkpointStore, "checkpointStore");
    this.parquetManager =
        rowManager instanceof RowManagerImpl
            ? new ParquetManagerImpl(rowManager, DiscoveryCache, group, session, checkpointStore)
            : new ParquetMaintenanceManagerUnsupported(
                "Parquet maintenance requires DIRECT mode local data manager");
  }

  @Override
  public SchemaModel sampleSchemaFromAmps(SchemaSampleOptions options) {
    SchemaSampleOptions resolvedOptions =
        options == null ? SchemaSampleOptions.defaults() : options;
    if (session == null) {
      throw new IllegalStateException("AMPS schema sampling requires a session-backed ingest manager");
    }
    String connectionName =
        Strings.requireNonBlank(resolvedOptions.connectionName(), "options.connectionName");
    String topic = Strings.requireNonBlank(resolvedOptions.topic(), "options.topic");
    IngestFormat format =
        resolvedOptions.format() == null ? IngestFormat.NVFIX : resolvedOptions.format();
    List<String> samples =
        collectAmpsPayloadSamples(
            connectionName,
            topic,
            resolvedOptions.ampsOptions(),
            resolvedOptions.ampsFilter(),
            resolvedOptions.ampsSow(),
            resolvedOptions.maxSamples(),
            resolvedOptions.ampsWaitMillis());
    if (samples.isEmpty()) {
      throw new IllegalStateException(
          "No payload samples received from AMPS connection="
              + connectionName
              + " topic="
              + topic);
    }
    SchemaModel sampled =
        sampleSchema(
            new SchemaSampleRequest(
                format,
                samples,
                resolvedOptions));
    return toRegistrationSchema(sampled);
  }

  private List<String> collectAmpsPayloadSamples(
      String connectionName,
      String topic,
      String options,
      String filter,
      boolean sow,
      int maxSamples,
      long waitMillis) {
    List<String> samples = new ArrayList<>();
    Object monitor = new Object();
    AtomicBoolean done = new AtomicBoolean(false);
    AmpsConnection.SubscriptionSpec spec =
        new AmpsConnection.SubscriptionSpec(sow, topic, options, filter);
    try (AmpsConnection amps = session.connections().amps(connectionName);
        AmpsConnection.Subscription ignored =
            amps.subscribe(
                spec,
                (AmpsConnection.SubscriptionListener)
                    message -> {
                  if (message == null || done.get()) {
                    return;
                  }
                  Message raw = message.raw();
                  if (raw != null && raw.getCommand() == Message.Command.GroupEnd && sow) {
                    synchronized (monitor) {
                      done.set(true);
                      monitor.notifyAll();
                    }
                    return;
                  }
                  String payload = Strings.trimToNull(message.data());
                  if (payload == null) {
                    return;
                  }
                  synchronized (monitor) {
                    if (done.get()) {
                      return;
                    }
                    if (samples.size() >= maxSamples) {
                      done.set(true);
                      monitor.notifyAll();
                      return;
                    }
                    samples.add(payload);
                    if (samples.size() >= maxSamples) {
                      done.set(true);
                    }
                    monitor.notifyAll();
                  }
                })) {
      long deadline = System.currentTimeMillis() + Math.max(1_000L, waitMillis);
      synchronized (monitor) {
        while (!done.get() && samples.size() < maxSamples) {
          long remaining = deadline - System.currentTimeMillis();
          if (remaining <= 0L) {
            break;
          }
          monitor.wait(Math.min(500L, remaining));
        }
      }
    } catch (Exception ex) {
      if (ex instanceof InterruptedException) {
        Thread.currentThread().interrupt();
      }
      throw new IllegalStateException(
          "Failed to sample AMPS payloads connection="
              + connectionName
              + " topic="
              + topic,
          ex);
    }
    LOGGER.info("[Ingest][SchemaSample][AMPS] connectionName={} topic={} requested={} received={} sow={} waitMillis={}", connectionName, topic, maxSamples, samples.size(), sow, waitMillis);
    return List.copyOf(samples);
  }

  private SchemaModel toRegistrationSchema(SchemaModel schema) {
    if (schema == null || schema.getFields() == null || schema.getFields().isEmpty()) {
      return schema;
    }
    List<SchemaModel.Field> nullableFields = new ArrayList<>(schema.getFields().size());
    for (SchemaModel.Field field : schema.getFields()) {
      if (field == null) {
        continue;
      }
      nullableFields.add(field.toBuilder().withNullable(true).build());
    }
    return schema.toBuilder().withFields(nullableFields).build();
  }

  @Override
  public void delete(String ingestName, IngestAdmin.DeleteOptions options) {
    String normalizedName = Strings.requireNonBlank(ingestName, "ingestName");
    IngestAdmin.DeleteOptions resolvedOptions =
        options == null ? IngestAdmin.DeleteOptions.ingestOnly() : options;
    resolveIngestDefinition(normalizedName);

    stop(normalizedName);
    activeTableDescriptors.remove(normalizedName);
    parquetRollPolicies.remove(normalizedName);
    parquetFilenamePrefixes.remove(normalizedName);
    parquetRollBuffers.remove(normalizedName);
    runtimes.remove(normalizedName);

    DeleteResult result =
        deleteManager.delete(
            "ingest:" + group + "/" + normalizedName,
            toLifecycleDeleteOptions(resolvedOptions));
    if (result.outcome() == DeleteOutcome.BLOCKED) {
      throw new IllegalStateException(
          "Delete blocked for ingest " + normalizedName + ": " + result.blockers());
    }
    if (result.outcome() == DeleteOutcome.PARTIAL) {
      throw new IllegalStateException(
          "Delete partially completed for ingest "
              + normalizedName
              + ": "
              + result.warnings());
    }
  }

  private org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOptions toLifecycleDeleteOptions(
      IngestAdmin.DeleteOptions options) {
    List<DefinitionType> cascadeTypes = new ArrayList<>();
    cascadeTypes.add(DefinitionType.INGEST_SOURCE);
    if (options != null && options.deleteDatasetDefinition()) {
      cascadeTypes.add(DefinitionType.DATASET);
    }
    return new org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOptions(
        true,
        options != null && options.deleteSchemaDefinition(),
        false,
        false,
        false,
        false,
        true,
        false,
        List.copyOf(cascadeTypes));
  }

  @Override
  public Optional<IngestModel> getModel(String ingestName) {
    String normalizedName = Strings.requireNonBlank(ingestName, "ingestName");
    Optional<DefinitionIngest> ingest = DiscoveryCache.get(DefinitionIngest.class, group, normalizedName);
    if (ingest.isEmpty()) {
      return Optional.empty();
    }
    Optional<DefinitionIngestSource> source =
        DiscoveryCache.get(DefinitionIngestSource.class, group, normalizedName);
    SchemaModel schema = resolveSchemaModel(ingest.get());
    IngestModel model = modelMapper.fromDefinitions(ingest.get(), source.orElse(null), schema);
    DefinitionDataset dataset = resolveDatasetDefinition(ingest.get().datasetId());
    if (dataset != null) {
      IngestTargetModel target = modelMapper.fromDefinitionDataset(dataset);
      if (target != null) {
        if (target instanceof IngestTargetParquetModel parquetTarget) {
          String clientPath =
              toClientVisibleS3Path(parquetTarget.parquetPath(), parquetTarget.connectionName());
          if (!Objects.equals(clientPath, parquetTarget.parquetPath())) {
            target = parquetTarget.toBuilder().withParquetPath(clientPath).build();
          }
        }
        model = model.toBuilder().withTarget(target).build();
      }
    }
    return Optional.of(model);
  }

  @Override
  public List<IngestModel> listModels() {
    List<DefinitionIngest> definitions = new ArrayList<>(DiscoveryCache.list(DefinitionIngest.class, group));
    definitions.sort(
        Comparator.comparing(
            definition ->
                Strings.requireNonBlank(
                    definition.envelope() == null ? null : definition.envelope().name(),
                    "ingest.envelope.name"),
            String.CASE_INSENSITIVE_ORDER));
    List<IngestModel> models = new ArrayList<>(definitions.size());
    for (DefinitionIngest definition : definitions) {
      String name =
          Strings.requireNonBlank(
              definition.envelope() == null ? null : definition.envelope().name(),
              "ingest.envelope.name");
      getModel(name).ifPresent(models::add);
    }
    return List.copyOf(models);
  }

  @Override
  public IngestRuntime open(String ingestName) {
    String normalizedName = Strings.requireNonBlank(ingestName, "ingestName");
    DefinitionIngest ingestDefinition = resolveIngestDefinition(normalizedName);
    DefinitionIngestSource sourceDefinition =
        DiscoveryCache
            .get(DefinitionIngestSource.class, group, normalizedName)
            .orElseThrow(
                () ->
                    new IllegalStateException("IngestSource not found for ingest: " + group + "/" + normalizedName));
    DatasetId datasetId =
        DatasetId.parse(Strings.requireNonBlank(ingestDefinition.datasetId(), "ingest.datasetId"));
    prepareActiveDataset(ingestDefinition, datasetId);
    boolean apiSource = sourceDefinition.amps() == null && sourceDefinition.external() == null;
    String source = resolveSourceKind(sourceDefinition, apiSource);
    IngestRuntimeImpl runtime = new IngestRuntimeImpl(normalizedName, datasetId, apiSource, source);
    IngestRuntimeImpl existing = runtimes.putIfAbsent(normalizedName, runtime);
    if (existing != null && !existing.closed()) {
      LOGGER.info("[IngestRuntime] open ingestName={} mode={} source={} bytes=0", normalizedName, runtimeMode, existing.sourceKind());
      return existing;
    }
    if (existing != null) {
      runtimes.put(normalizedName, runtime);
    }
    LOGGER.info("[IngestRuntime] open ingestName={} mode={} source={} bytes=0", normalizedName, runtimeMode, source);
    return runtime;
  }

  @Override
  public IngestRuntime start(String ingestName) {
    IngestRuntime runtime = open(ingestName);
    if (runtime instanceof IngestRuntimeImpl impl) {
      LOGGER.info("[IngestRuntime] start ingestName={} mode={} source={} bytes=0", impl.ingestName, runtimeMode, impl.sourceKind());
      impl.startManagedSource();
    }
    return runtime;
  }

  @Override
  public IngestRuntime start(String ingestName, IngestManager.IngestRuntimeOptions options) {
    String normalizedName = Strings.requireNonBlank(ingestName, "ingestName");
    IngestManager.IngestRuntimeOptions resolved =
        options == null ? IngestManager.IngestRuntimeOptions.builder().build() : options;
    if (options != null) {
      setParquetRollPolicy(
          normalizedName, resolved.parquetMaxRowsPerFile(), resolved.parquetMaxIdleMillis());
      setParquetFilenamePrefix(normalizedName, resolved.parquetFilenamePrefix());
    }
    return start(normalizedName);
  }

  @Override
  public void stop(String ingestName) {
    String normalizedName = Strings.requireNonBlank(ingestName, "ingestName");
    IngestRuntimeImpl runtime = runtimes.get(normalizedName);
    if (runtime != null) {
      LOGGER.info("[IngestRuntime] stop ingestName={} mode={} source={} bytes=0", normalizedName, runtimeMode, runtime.sourceKind());
      runtime.stop();
      if (runtime.closed()) {
        runtimes.remove(normalizedName, runtime);
      }
    }
    flushPendingParquetBuffer(normalizedName, "runtime-stop");
  }

  public void appendFromAmps(
      String ingestName,
      String payloadText,
      IngestAmpsPump.PayloadFormat payloadFormat,
      int ampsCommand,
      IngestAmpsPump.OofHandling oofHandling,
      String parquetFilenamePrefix) {
    String priorPrefix = parquetFilenamePrefixOverride.get();
    String resolvedPrefix = Strings.trimToNull(parquetFilenamePrefix);
    if (resolvedPrefix == null) {
      parquetFilenamePrefixOverride.remove();
    } else {
      parquetFilenamePrefixOverride.set(resolvedPrefix);
    }
    try {
      handleAmpsMessage(ingestName, payloadText, payloadFormat, ampsCommand, oofHandling, true);
    } finally {
      if (priorPrefix == null) {
        parquetFilenamePrefixOverride.remove();
      } else {
        parquetFilenamePrefixOverride.set(priorPrefix);
      }
    }
  }

  private void handleAmpsMessage(
      String ingestName,
      String payloadText,
      IngestAmpsPump.PayloadFormat payloadFormat,
      int ampsCommand,
      IngestAmpsPump.OofHandling oofHandling,
      boolean enforceWriteGate) {
    IngestPayload payload =
        payloadFormat == IngestAmpsPump.PayloadFormat.JSON
            ? new IngestPayloadJson(payloadText)
            : new IngestPayloadNvFix(payloadText);
    if (ampsCommand == com.crankuptheamps.client.Message.Command.OOF
        || ampsCommand == com.crankuptheamps.client.Message.Command.SOWDelete) {
      DuckDbActiveMutation mutation =
          oofHandling == IngestAmpsPump.OofHandling.MARK_AS_DELETED
              ? DuckDbActiveMutation.MARK_AS_DELETED
              : DuckDbActiveMutation.DELETE;
      append(ingestName, payload, mutation, enforceWriteGate);
      markRuntimeMessageIngested(ingestName);
      return;
    }
    append(ingestName, payload, DuckDbActiveMutation.UPSERT, enforceWriteGate);
    markRuntimeMessageIngested(ingestName);
  }

  private static String ensureSubscribeOption(String options, String optionToken) {
    String normalizedToken = Strings.requireNonBlank(optionToken, "optionToken");
    String normalizedOptions = Strings.trimToNull(options);
    if (normalizedOptions == null) {
      return normalizedToken;
    }
    for (String token : normalizedOptions.split(";")) {
      String normalized = Strings.trimToNull(token);
      if (normalized != null && normalizedToken.equalsIgnoreCase(normalized)) {
        return normalizedOptions;
      }
    }
    return normalizedOptions + ";" + normalizedToken;
  }

  private static String removeSubscribeOption(String options, String optionToken) {
    String normalizedToken = Strings.requireNonBlank(optionToken, "optionToken");
    String normalizedOptions = Strings.trimToNull(options);
    if (normalizedOptions == null) {
      return null;
    }
    List<String> retained = new ArrayList<>();
    for (String token : normalizedOptions.split(";")) {
      String normalized = Strings.trimToNull(token);
      if (normalized == null) {
        continue;
      }
      if (!normalizedToken.equalsIgnoreCase(normalized)) {
        retained.add(normalized);
      }
    }
    if (retained.isEmpty()) {
      return null;
    }
    return String.join(";", retained);
  }

  private static String resolveAmpsSubscribeOptions(DefinitionIngestSource.AmpsSource amps) {
    boolean sowEnabled = Boolean.TRUE.equals(amps.sow()) || Boolean.TRUE.equals(amps.sowOnly());
    String options = removeSubscribeOption(amps.options(), "no_bookmark");
    options = removeSubscribeOption(options, "nobookmark");
    if (!sowEnabled) {
      return removeSubscribeOption(options, "oof");
    }
    if (amps.oofHandling() == null) {
      return options;
    }
    return ensureSubscribeOption(options, "oof");
  }

  private static boolean resolveAmpsBookmarkResume(DefinitionIngestSource.AmpsSource amps) {
    return !hasSubscribeOption(amps.options(), "no_bookmark")
        && !hasSubscribeOption(amps.options(), "nobookmark");
  }

  private static boolean hasSubscribeOption(String options, String optionToken) {
    String normalizedToken = Strings.requireNonBlank(optionToken, "optionToken");
    String normalizedOptions = Strings.trimToNull(options);
    if (normalizedOptions == null) {
      return false;
    }
    for (String token : normalizedOptions.split(";")) {
      String normalized = Strings.trimToNull(token);
      if (normalized != null && normalizedToken.equalsIgnoreCase(normalized)) {
        return true;
      }
    }
    return false;
  }

  @Override

    for (IngestRuntimeImpl runtime : runtimes.values()) {

    runtimes.clear();
    parquetRollBuffers.clear();
    parquetRollPolicies.clear();
    parquetFilenamePrefixes.clear();
    parquetFilenamePrefixOverride.remove();
    CloseSupport.closeQuietly(schemaIpcAllocator);
    CloseSupport.closeQuietly(rollIpcAllocator);
  }

  private DataStatusPublisher createDataStatusPublisher(
      String statusName,
      DefinitionStatusData.ProducerType producerType,
      String producerId,
      String datasetId,
      Map<String, String> tags) {
    if (!(session instanceof SessionImpl sessionImpl)) {
      return null;
    }
    if (sessionImpl.controlPlaneConnection() == null) {
      return null;
    }
    SessionContext context = sessionImpl.context();
    if (context == null) {
      return null;
    }
    String resolvedGroup = Strings.trimToNull(context.getGroup());
    String resolvedName = Strings.trimToNull(statusName);
    String resolvedInstance = Strings.trimToNull(context.getInstance());
    if (resolvedGroup == null || resolvedName == null || resolvedInstance == null) {
      return null;
    }
    return new DataStatusPublisher(
        sessionImpl.controlPlaneConnection(),
        resolvedGroup,
        resolvedName,
        resolvedInstance,
        producerType,
        producerId,
        datasetId,
        sessionImpl.runtimePlacementHost(),
        Strings.trimToNull(context.getAppName()),
        Strings.trimToNull(context.getInstance()),
        Strings.trimToNull(context.getGroup()),
        tags,
        sessionImpl.dataStatusCadenceSeconds(),
        sessionImpl.statusScheduler());
  }

    String normalized = Strings.trimToNull(ingestName);
    if (normalized == null) {

    IngestRuntimeImpl runtime = runtimes.get(normalized);
    if (runtime != null) {

    }
  }

  private final class IngestRuntimeImpl implements IngestRuntime {
    private final String ingestName;
    private final DatasetId datasetId;
    private final boolean apiSource;
    private final boolean managedSource;
    private final String sourceKind;
    private final DataStatusPublisher statusPublisher;
    private final AtomicLong messagesIngested = new AtomicLong(0L);
    private final AtomicReference<Throwable> failure = new AtomicReference<>();
    private final java.util.concurrent.atomic.AtomicBoolean closed = new java.util.concurrent.atomic.AtomicBoolean(false);
    private volatile IngestAmpsPump pump;
    private volatile AutoCloseable sourceHandle;

    private IngestRuntimeImpl(
        String ingestName, DatasetId datasetId, boolean apiSource, String sourceKind) {
      this.ingestName = ingestName;
      this.datasetId = datasetId;
      this.apiSource = apiSource;
      this.managedSource = !apiSource && !serviceMode;
      this.sourceKind = Strings.requireNonBlank(sourceKind, "sourceKind");
      this.statusPublisher =
          createDataStatusPublisher(
              "runtime/" + ingestName,
              DefinitionStatusData.ProducerType.INGEST_RUNTIME,
              "ingest-runtime:" + group + "/" + ingestName,
              datasetId.toString(),
              Map.of(
                  "sourceKind", this.sourceKind,
                  "runtimeMode", runtimeMode,
                  "managedSource", Boolean.toString(this.managedSource)));
      if (statusPublisher != null) {
        statusPublisher.start();
        statusPublisher.publishOnline(
            managedSource ? "runtime opened (managed source)" : "runtime opened (api source)");
        statusPublisher.putTag("messagesIngested", "0");
      }
    }

    @Override
    public void append(IngestPayload payload) {
      if (closed()) {
        throw new IllegalStateException("IngestRuntime is closed: " + ingestName);
      }
      if (!apiSource) {
        throw new IllegalStateException("append(payload) is supported for API source only: " + ingestName);
      }
      try {
        LOGGER.info("[IngestRuntime] append ingestName={} mode={} source={} bytes={}", ingestName, runtimeMode, sourceKind, payloadBytes(payload));
        IngestManagerImpl.this.append(ingestName, payload);
        recordMessageIngested();
      } catch (RuntimeException ex) {
        failure.compareAndSet(null, ex);
        if (statusPublisher != null) {
          statusPublisher.publishError(
              "append failed: "
                  + Strings.firstNonBlank(
                      Strings.trimToNull(ex.getMessage()), ex.getClass().getSimpleName()));
        }
        throw ex;
      }
    }

    @Override
    public Optional<Throwable> failure() {
      return Optional.ofNullable(failure.get());
    }

    @Override
    public long queryCount() {
      RowResult result;
      try {
        result = rowManager.querySql(COUNT_SQL_PREFIX + sqlToken(datasetId), 1);
      } catch (RuntimeException ex) {
        if (isMissingActiveTable(ex)) {
          return 0L;
        }
        throw ex;
      }
      if (result.rows().isEmpty() || result.rows().get(0).isEmpty()) {
        return 0L;
      }
      Object count = result.rows().get(0).get(0);
      if (count instanceof Number number) {
        return number.longValue();
      }
      String text = Strings.trimToNull(String.valueOf(count));
      return text == null ? 0L : Long.parseLong(text);
    }

    private static boolean isMissingActiveTable(Throwable error) {
      Throwable current = error;
      while (current != null) {
        String message = Strings.trimToNull(current.getMessage());
        if (message != null) {
          String normalized = message.toLowerCase(Locale.ROOT);
          if (normalized.contains("table with name")
              && normalized.contains("does not exist")) {
            return true;
          }
        }
        current = current.getCause();
      }
      return false;
    }

    private static String sqlToken(DatasetId datasetId) {
      return quoteIdentifier(datasetId.namespace()) + "." + quoteIdentifier(datasetId.name());
    }

    private static String quoteIdentifier(String identifier) {
      return "\"" + identifier.replace("\"", "\"\"") + "\"";
    }

    @Override
    public IngestRuntimeStatus status() {
      Throwable currentFailure = failure.get();
      String lastError =
          currentFailure == null
              ? null
              : Strings.firstNonBlank(
                  Strings.trimToNull(currentFailure.getMessage()),
                  currentFailure.getClass().getSimpleName());
      IngestAmpsPump active = pump;
      return new IngestRuntimeStatus(
          queryCount(),
          lastError,
          active == null ? null : active.lastCheckpoint(),
          active == null ? null : active.lastCheckpointFlushUtc());
    }

    @Override
    public void stop() {
      if (!managedSource) {
        if (!closed.compareAndSet(false, true)) {
          return;
        }
        CloseSupport.closeQuietly(statusPublisher);
        removeRuntime(ingestName, this);
        return;
      }
      if (!closed.compareAndSet(false, true)) {
        return;
      }
      IngestAmpsPump active = pump;
      pump = null;
      if (active != null) {
        active.close();
      }
      AutoCloseable managed = sourceHandle;
      sourceHandle = null;
      CloseSupport.closeQuietly(managed);
      CloseSupport.closeQuietly(statusPublisher);
      removeRuntime(ingestName, this);
    }

    private boolean closed() {
      return closed.get();
    }

    private void recordMessageIngested() {
      long ingested = messagesIngested.incrementAndGet();
      if (statusPublisher != null) {
        statusPublisher.putTag("messagesIngested", Long.toString(ingested));
      }
    }

    private String sourceKind() {
      return sourceKind;
    }

    private synchronized void startManagedSource() {
      if (!managedSource) {
        return;
      }
      if (closed()) {
        throw new IllegalStateException("IngestRuntime is closed: " + ingestName);
      }
      if (pump != null || sourceHandle != null) {
        return;
      }
      if (statusPublisher != null) {
        statusPublisher.publishOnline("starting managed source");
      }
      DefinitionIngestSource source =
          DiscoveryCache
              .get(DefinitionIngestSource.class, group, ingestName)
              .orElseThrow(
                  () ->
                      new IllegalStateException("IngestSource not found for ingest: " + group + "/" + ingestName));
      if (source.amps() != null) {
        startAmpsSource(source);
        if (statusPublisher != null) {
          statusPublisher.publishOnline("managed AMPS source running");
        }
        return;
      }
      if (source.odbc() != null) {
        startOdbcSource(source.odbc());
        if (statusPublisher != null) {
          statusPublisher.publishOnline("managed ODBC source running");
        }
        return;
      }
      if (source.external() != null) {
        startExternalSource(source);
        if (statusPublisher != null) {
          statusPublisher.publishOnline("managed external source running");
        }
        return;
      }
      throw new IllegalStateException(
          "Unsupported managed ingest source for " + group + "/" + ingestName);
    }

    private void startAmpsSource(DefinitionIngestSource source) {
      if (session == null) {
        throw new IllegalStateException("AMPS runtime requires session-backed ingest manager");
      }
      DefinitionIngestSource.AmpsSource amps =
          Objects.requireNonNull(source.amps(), "ingestSource.amps");
      String format = Strings.requireNonBlank(source.format(), "ingestSource.format");
      IngestAmpsPump.PayloadFormat payloadFormat = payloadFormat(format, ingestName);

      IngestAmpsPump created =
          IngestAmpsPump.builder(session)
              .withIngestName(ingestName)
              .withIngestSourceName(ingestName)
              .withDatasetId(datasetId.toString())
              .withConnectionName(Strings.requireNonBlank(amps.connectionName(), "amps.connectionName"))
              .withTopic(Strings.requireNonBlank(amps.topic(), "amps.topic"))
              .withSowSubscribe(amps.sow())
              .withSow(amps.sowOnly())
              .withOptions(resolveAmpsSubscribeOptions(amps))
              .withBookmarkResume(resolveAmpsBookmarkResume(amps))
              .withFilter(amps.filter())
              .withMode(
                  amps.mode() == DefinitionIngestSource.Mode.BATCH
                      ? AmpsConnection.BatchMode.BATCH
                      : AmpsConnection.BatchMode.STREAM)
              .withBatchMaxMessages(
                  sanitizePositiveInt(amps.batchMaxMessages(), AmpsConnection.DEFAULT_BATCH_MAX_MESSAGES))
              .withBatchMaxBytes(
                  sanitizePositiveInt(amps.batchMaxBytes(), AmpsConnection.DEFAULT_BATCH_MAX_BYTES))
              .withBatchFlushMillis(
                  sanitizePositiveLong(amps.batchFlushMillis(), AmpsConnection.DEFAULT_BATCH_FLUSH_MILLIS))
              .withPayloadFormat(payloadFormat)
              .withOofHandling(
                  amps.oofHandling() == DefinitionIngestSource.OofHandling.MARK_AS_DELETED
                      ? IngestAmpsPump.OofHandling.MARK_AS_DELETED
                      : IngestAmpsPump.OofHandling.DELETE_EVENT)
              .withWriteAuthorizer(
                  (datasetIdArg, operation) ->
                      IngestManagerImpl.this.enforceWrite(datasetIdArg, operation, null))
              .withAmpsMessageHandler(
                  (ingestNameArg,
                      payloadText,
                      payloadFormatArg,
                      ampsCommand,
                      oofHandling,
                      parquetFilenamePrefix) ->
                      IngestManagerImpl.this.handleAmpsMessage(
                          ingestNameArg,
                          payloadText,
                          payloadFormatArg,
                          ampsCommand,
                          oofHandling,
                          false))
              .withErrorListener(
                  error -> {
                    failure.compareAndSet(null, error);
                    if (statusPublisher != null && error != null) {
                      statusPublisher.publishError(
                          "pump error: "
                              + Strings.firstNonBlank(
                                  Strings.trimToNull(error.getMessage()),
                                  error.getClass().getSimpleName()));
                    }
                  })
              .withCheckpointStore(checkpointStore)
              .build();
      try {
        created.start();
        pump = created;
      } catch (Exception ex) {
        failure.compareAndSet(null, ex);
        if (statusPublisher != null) {
          statusPublisher.publishError(
              "managed AMPS source failed: "
                  + Strings.firstNonBlank(
                      Strings.trimToNull(ex.getMessage()), ex.getClass().getSimpleName()));
        }
        throw new IllegalStateException("Failed to start ingest AMPS runtime: " + ingestName, ex);
      }
    }

    private void startExternalSource(DefinitionIngestSource source) {
      if (session == null) {
        throw new IllegalStateException("External source runtime requires session-backed ingest manager");
      }
      DefinitionIngestSource.ExternalSource external =
          Objects.requireNonNull(source.external(), "ingestSource.external");
      String providerId =
          Strings.requireNonBlank(external.sourceType(), "ingestSource.external.sourceType");
      if (OdbcRuntimeGuard.isOdbcType(providerId)) {
        startOdbcSource(toOdbcSource(external));
        return;
      }
      IngestSourceProvider provider = plugins.findIngestSourceProvider(providerId);
      if (provider == null) {
        throw new IllegalStateException(
            "No ingest source provider registered for external source type '" + providerId + "'");
      }
      if (!provider.supportsRuntime()) {
        throw new IllegalStateException(
            "Ingest source provider '" + provider.providerId() + "' does not support runtime execution");
      }
      try {
        AutoCloseable handle =
            provider.startRuntime(
                new IngestSourceRuntimeContext(
                    session,
                    ingestName,
                    source,
                    external,
                    new IngestSourceRuntimeContext.Sink() {
                      @Override
                      public void appendJson(String payload) {
                        IngestManagerImpl.this.append(ingestName, new IngestPayloadJson(payload));
                      }

                      @Override
                      public void appendNvfix(String payload) {
                        IngestManagerImpl.this.append(ingestName, new IngestPayloadNvFix(payload));
                      }

                      @Override
                      public void appendIpc(byte[] arrowIpcStreamBytes) {
                        IngestManagerImpl.this.appendIpc(ingestName, arrowIpcStreamBytes);
                      }
                    }));
        sourceHandle = handle == null ? () -> {} : handle;
      } catch (RuntimeException ex) {
        failure.compareAndSet(null, ex);
        if (statusPublisher != null) {
          statusPublisher.publishError(
              "managed external source failed: "
                  + Strings.firstNonBlank(
                      Strings.trimToNull(ex.getMessage()), ex.getClass().getSimpleName()));
        }
        throw ex;
      } catch (Exception ex) {
        failure.compareAndSet(null, ex);
        if (statusPublisher != null) {
          statusPublisher.publishError(
              "managed external source failed: "
                  + Strings.firstNonBlank(
                      Strings.trimToNull(ex.getMessage()), ex.getClass().getSimpleName()));
        }
        throw new IllegalStateException(
            "Failed to start ingest external runtime: " + ingestName + " provider=" + provider.providerId(),
            ex);
      }
    }

    private void startOdbcSource(DefinitionIngestSource.OdbcSource odbcSource) {
      if (!(rowManager instanceof RowManagerImpl localRowManager)) {
        throw OdbcSessionSupport.ingestSourceFailed(
            ingestName,
            odbcSource == null ? null : odbcSource.connectionName(),
            odbcSource == null ? null : odbcSource.databaseType(),
            "direct_mode_required",
            null);
      }
      DefinitionIngestSource.OdbcSource source =
          Objects.requireNonNull(odbcSource, "ingestSource.odbc");
      String requestedConnectionName =
          Strings.requireNonBlank(source.connectionName(), "ingestSource.odbc.connectionName");
      DefinitionConnection.Odbc.DatabaseType requestedDatabaseType = source.databaseType();
      try {
        byte[] arrowIpcStreamBytes =
            localRowManager.withDuckdbSession(
                session -> {
                  OdbcSessionSupport.OdbcConnectionSpec connection =
                      OdbcSessionSupport.resolveConnection(
                          DiscoveryCache,
                          group,
                          requestedConnectionName,
                          requestedDatabaseType,
                          ingestName);
                  String remoteSql =
                      OdbcSessionSupport.ingestSourceSql(source, ingestName, connection.connectionName());
                  try (OdbcSessionSupport.OdbcQueryResult queryResult =
                      OdbcSessionSupport.openQueryResult(session, connection, remoteSql)) {
                    return TabularManagerImpl.toArrowIpcBytes(
                        queryResult.resultSet(), datasetId.toString(), schemaIpcAllocator);
                  } catch (SQLException ex) {
                    throw OdbcSessionSupport.ingestSourceFailed(
                        ingestName,
                        connection.connectionName(),
                        connection.databaseType(),
                        ex.getMessage(),
                        ex);
                  } catch (RuntimeException ex) {
                    if (isDeterministicOdbcFailure(ex)) {
                      throw ex;
                    }
                    throw OdbcSessionSupport.ingestSourceFailed(
                        ingestName,
                        connection.connectionName(),
                        connection.databaseType(),
                        ex.getMessage(),
                        ex);
                  }
                });
        if (arrowIpcStreamBytes != null && arrowIpcStreamBytes.length > 0) {
          IngestManagerImpl.this.appendIpc(ingestName, arrowIpcStreamBytes);
        }
        sourceHandle = () -> {};
      } catch (RuntimeException ex) {
        failure.compareAndSet(null, ex);
        if (statusPublisher != null) {
          statusPublisher.publishError(
              "managed ODBC source failed: "
                  + Strings.firstNonBlank(
                      Strings.trimToNull(ex.getMessage()), ex.getClass().getSimpleName()));
        }
        if (isDeterministicOdbcFailure(ex)) {
          throw ex;
        }
        throw OdbcSessionSupport.ingestSourceFailed(
            ingestName, requestedConnectionName, requestedDatabaseType, ex.getMessage(), ex);
      }
    }

    private static boolean isDeterministicOdbcFailure(RuntimeException ex) {
      String message = Strings.trimToNull(ex == null ? null : ex.getMessage());
      return message != null && message.startsWith("ODBC_");
    }

    private static DefinitionIngestSource.OdbcSource toOdbcSource(
        DefinitionIngestSource.ExternalSource external) {
      Map<String, String> options = external.options();
      String connectionName =
          Strings.firstNonBlank(
              Strings.trimToNull(external.reference()),
              Strings.trimToNull(options == null ? null : options.get("connectionName")));
      String tableName = Strings.trimToNull(options == null ? null : options.get("tableName"));
      String querySql = Strings.trimToNull(options == null ? null : options.get("querySql"));
      DefinitionConnection.Odbc.DatabaseType databaseType =
          parseDatabaseType(options == null ? null : options.get("databaseType"));
      return new DefinitionIngestSource.OdbcSource(
          connectionName,
          databaseType,
          tableName,
          querySql,
          options == null ? Map.of() : Map.copyOf(options));
    }

    private static DefinitionConnection.Odbc.DatabaseType parseDatabaseType(String value) {
      String normalized = Strings.trimToNull(value);
      if (normalized == null) {
        return DefinitionConnection.Odbc.DatabaseType.OTHER;
      }
      try {
        return DefinitionConnection.Odbc.DatabaseType.valueOf(normalized.toUpperCase(Locale.ROOT));
      } catch (IllegalArgumentException ex) {
        return DefinitionConnection.Odbc.DatabaseType.OTHER;
      }
    }

    private static IngestAmpsPump.PayloadFormat payloadFormat(String format, String ingestName) {
      String normalized = Strings.requireNonBlank(format, "format").toUpperCase(Locale.ROOT);
      return switch (normalized) {
        case FORMAT_JSON -> IngestAmpsPump.PayloadFormat.JSON;
        case FORMAT_NVFIX -> IngestAmpsPump.PayloadFormat.NVFIX;
        default ->
            throw new IllegalStateException(
                "IngestSource format must be JSON or NVFIX: " + ingestName + " format=" + format);
      };
    }
  }

  private static int sanitizePositiveInt(Integer value, int defaultValue) {
    if (value == null || value <= 0) {
      return defaultValue;
    }
    return value;
  }

  private static long sanitizePositiveLong(Long value, long defaultValue) {
    if (value == null || value <= 0L) {
      return defaultValue;
    }
    return value;
  }

  static int resolveMaxRowsPerFile(Integer value) {
    if (value == null) {
      return DEFAULT_PARQUET_MAX_ROWS_PER_FILE;
    }
    if (value <= 0) {
      throw new IllegalArgumentException("maxRowsPerFile must be > 0");
    }
    return value;
  }

  static long resolveMaxIdleMillis(Long value) {
    if (value == null) {
      return DEFAULT_PARQUET_MAX_IDLE_MILLIS;
    }
    if (value <= 0L) {
      throw new IllegalArgumentException("maxIdleMillis must be > 0");
    }
    return value;
  }

  static final class ParquetRollBuffer {
    final Object monitor = new Object();
    final List<byte[]> pendingIpcBatches = new ArrayList<>();
    DatasetId datasetId;
    ParquetRollPolicy policy;
    String parquetFilenamePrefix;
    long pendingRows;
    long lastAppendMillis;
    Schema schema;

    ParquetRollBuffer(
        DatasetId datasetId, ParquetRollPolicy policy, String parquetFilenamePrefix) {
      this.datasetId = Objects.requireNonNull(datasetId, "datasetId");
      this.policy = Objects.requireNonNull(policy, "policy");
      this.parquetFilenamePrefix = Strings.trimToNull(parquetFilenamePrefix);
    }
  }

  record ParquetRollPolicy(int maxRowsPerFile, long maxIdleMillis) {
    static ParquetRollPolicy disabled() {
      return new ParquetRollPolicy(0, 0L);
    }

    boolean enabled() {
      return maxRowsPerFile > 0 || maxIdleMillis > 0L;
    }
  }
