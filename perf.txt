package org.pronzato.fabric.api.services.flight.streaming;

import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Map;
import java.nio.ByteBuffer;
import java.util.Objects;
import java.util.OptionalLong;
import java.util.UUID;
import java.util.concurrent.ArrayBlockingQueue;
import java.util.concurrent.BlockingQueue;
import java.util.concurrent.TimeUnit;
import org.pronzato.fabric.api.data.dataset.DatasetId;
import org.pronzato.fabric.api.data.manager.DuckdbActiveChangeLog;
import org.pronzato.fabric.api.data.manager.RowManagerImpl;
import org.pronzato.fabric.api.data.duckdb.DuckdbNotifier;
import org.pronzato.fabric.api.util.Strings;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public final class DatasetStreamHub {
  private static final Logger LOGGER = LoggerFactory.getLogger(DatasetStreamHub.class);
  private static final int SUBSCRIBER_QUEUE_CAPACITY = 32;
  private static final int CHANGE_READ_LIMIT = 1024;
  private static final long IDLE_STOP_MILLIS = 2_000L;
  private static final long AWAIT_MAX_MILLIS = 1_000L;

  private final DatasetId datasetId;
  private final String accessMode;
  private final List<String> primaryKeyColumns;
  private final RowManagerImpl rowManager;
  private final DuckdbNotifier changeNotifier;
  private final ArrowBatchPublisher batchPublisher;
  private final DatasetStreamHubRegistry registry;
  private final Object lock = new Object();
  private final LinkedHashMap<String, SubscriberState> subscribers = new LinkedHashMap<>();

  private volatile Thread producerThread;
  private volatile boolean closed;
  private volatile boolean stopRequested;
  private volatile RuntimeException fatalFailure;
  private volatile long lastActivityMillis = System.currentTimeMillis();

  private long knownSeq;
  private String publicFilterSql;
  private int publicBatchMaxRows;
  private int publicFlushMillis;
  private long publicCursorSeq;
  private long publicPendingMaxSeq;
  private long publicDeadlineMillis;
  private final LinkedHashSet<String> publicPendingKeys = new LinkedHashSet<>();

  public DatasetStreamHub(
      DatasetId datasetId,
      String accessMode,
      List<String> primaryKeyColumns,
      RowManagerImpl rowManager,
      DuckdbNotifier changeNotifier,
      ArrowBatchPublisher batchPublisher,
      DatasetStreamHubRegistry registry) {
    this.datasetId = Objects.requireNonNull(datasetId, "datasetId");
    this.accessMode = normalizeAccessMode(accessMode);
    this.primaryKeyColumns = List.copyOf(Objects.requireNonNull(primaryKeyColumns, "primaryKeyColumns"));
    this.rowManager = Objects.requireNonNull(rowManager, "rowManager");
    this.changeNotifier = Objects.requireNonNull(changeNotifier, "changeNotifier");
    this.batchPublisher = Objects.requireNonNull(batchPublisher, "batchPublisher");
    this.registry = Objects.requireNonNull(registry, "registry");
    this.knownSeq = Math.max(0L, rowManager.duckDbActiveChangeMaxSeq(datasetId));
  }

  public SubscriberHandle addSubscriber(StreamSubscriptionContext context) {
    Objects.requireNonNull(context, "context");
    if (!datasetId.equals(context.datasetId())) {
      throw new IllegalArgumentException("Hub dataset mismatch: " + context.datasetId());
    }
    if (!accessMode.equals(context.accessMode())) {
      throw new IllegalArgumentException(
          "STREAM_ACCESS_MODE_MISMATCH dataset="
              + datasetId
              + " expected="
              + accessMode
              + " requested="
              + context.accessMode());
    }
    synchronized (lock) {
      if (closed) {
        throw new IllegalStateException("Dataset stream hub is closed for " + datasetId);
      }
      if (fatalFailure != null) {
        throw fatalFailure;
      }
      String token = "sub-" + UUID.randomUUID();
      long startingSeq = resolveStartingSeqLocked(context);
      SubscriberState state =
          new SubscriberState(
              token,
              context,
              new ArrayBlockingQueue<>(SUBSCRIBER_QUEUE_CAPACITY),
              Math.max(0L, startingSeq));
      if (context.isPublic()) {
        configurePublicContractLocked(context);
      }
      subscribers.put(token, state);
      stopRequested = false;
      lastActivityMillis = System.currentTimeMillis();
      startProducerLocked();
      return new SubscriberHandle(this, token, state.queue, state.lastSeq);
    }
  }

  public void removeSubscriber(String token) {
    if (Strings.trimToNull(token) == null) {
      return;
    }
    synchronized (lock) {
      SubscriberState removed = subscribers.remove(token);
      if (removed == null) {
        return;
      }
      removed.closed = true;
      removed.queue.offer(Delivery.terminal(removed.lastSeq));
      if (subscribers.isEmpty()) {
        stopRequested = true;
        lastActivityMillis = System.currentTimeMillis();
      }
    }
  }

  public boolean isIdle() {
    synchronized (lock) {
      return subscribers.isEmpty() && (producerThread == null || !producerThread.isAlive());
    }
  }

  public void close() {
    Thread thread;
    synchronized (lock) {
      closed = true;
      stopRequested = true;
      thread = producerThread;
      for (SubscriberState state : subscribers.values()) {
        state.closed = true;
        state.queue.offer(Delivery.terminal(state.lastSeq));
      }
      subscribers.clear();
    }
    if (thread != null) {
      thread.interrupt();
    }
  }

  private long resolveStartingSeqLocked(StreamSubscriptionContext context) {
    if (context.isPublic() && !subscribers.isEmpty()) {
      return publicCursorSeq;
    }
    return context.startingSeq();
  }

  private void configurePublicContractLocked(StreamSubscriptionContext context) {
    if (subscribers.isEmpty()) {
      publicFilterSql = context.filterSql();
      publicBatchMaxRows = context.batchMaxRows();
      publicFlushMillis = context.flushMillis();
      publicCursorSeq = context.startingSeq();
      publicPendingMaxSeq = publicCursorSeq;
      publicDeadlineMillis = 0L;
      publicPendingKeys.clear();
      return;
    }
    if (!Objects.equals(publicFilterSql, context.filterSql())
        || publicBatchMaxRows != context.batchMaxRows()
        || publicFlushMillis != context.flushMillis()) {
      throw new IllegalArgumentException(
          "PUBLIC_STREAM_OPTIONS_MISMATCH dataset="
              + datasetId
              + " expected(batchMaxRows="
              + publicBatchMaxRows
              + ", flushMillis="
              + publicFlushMillis
              + ", filterSql="
              + Strings.firstNonBlank(publicFilterSql, "<none>")
              + ")");
    }
  }

  private void startProducerLocked() {
    if (producerThread != null && producerThread.isAlive()) {
      return;
    }
    Thread thread =
        Thread.ofPlatform()
            .daemon()
            .name("flight-stream-hub-" + datasetId.toString().replace(':', '_').replace('/', '_'))
            .start(this::runProducerLoop);
    producerThread = thread;
  }

  private void runProducerLoop() {
    try {
      while (!Thread.currentThread().isInterrupted()) {
        if (closed) {
          return;
        }
        List<SubscriberState> snapshot = snapshotSubscribers();
        if (snapshot.isEmpty()) {
          if (stopRequested) {
            return;
          }
          if (System.currentTimeMillis() - lastActivityMillis >= IDLE_STOP_MILLIS) {
            return;
          }
          awaitWithoutChanges();
          continue;
        }
        if ("PUBLIC".equals(accessMode)) {
          producePublic(snapshot);
        } else {
          producePrivate(snapshot);
        }
      }
    } catch (RuntimeException ex) {
      fatalFailure = ex;
      LOGGER.warn("[FlightStreamHub][fatal] dataset={} mode={} message={}", datasetId, accessMode, ex.getMessage(), ex);
      failAllSubscribers(ex);
    } finally {
      synchronized (lock) {
        producerThread = null;
      }
      registry.evict(datasetId, this);
    }
  }

  private void producePublic(List<SubscriberState> snapshot) {
    long now = System.currentTimeMillis();
    long timeout = publicAwaitTimeout(now);
    awaitForNextKnownSeq(publicCursorSeq, timeout);
    if (publicCursorSeq < knownSeq) {
      List<DuckdbActiveChangeLog.ChangeEvent> changes = readChangesAfter(publicCursorSeq);
      if (!changes.isEmpty()) {
        for (DuckdbActiveChangeLog.ChangeEvent change : changes) {
          if (change.seq() <= publicCursorSeq) {
            continue;
          }
          if (publicPendingKeys.add(change.pk()) && publicPendingKeys.size() == 1) {
            publicDeadlineMillis = now + Math.max(1L, publicFlushMillis);
          }
          publicPendingMaxSeq = Math.max(publicPendingMaxSeq, change.seq());
          if (publicPendingKeys.size() >= publicBatchMaxRows) {
            flushPublic(snapshot);
          }
        }
      }
    }
    if (!publicPendingKeys.isEmpty() && System.currentTimeMillis() >= publicDeadlineMillis) {
      flushPublic(snapshot);
    }
  }

  private void producePrivate(List<SubscriberState> snapshot) {
    long minSeq = minSubscriberSeq(snapshot);
    long now = System.currentTimeMillis();
    long timeout = privateAwaitTimeout(snapshot, now);
    awaitForNextKnownSeq(minSeq, timeout);
    if (minSeq < knownSeq) {
      List<DuckdbActiveChangeLog.ChangeEvent> changes = readChangesAfter(minSeq);
      if (!changes.isEmpty()) {
        for (DuckdbActiveChangeLog.ChangeEvent change : changes) {
          for (SubscriberState state : snapshot) {
            if (state.closed || change.seq() <= state.lastSeq) {
              continue;
            }
            if (state.pendingKeys.add(change.pk()) && state.pendingKeys.size() == 1) {
              state.deadlineMillis = System.currentTimeMillis() + Math.max(1L, state.context.flushMillis());
            }
            state.pendingMaxSeq = Math.max(state.pendingMaxSeq, change.seq());
            if (state.pendingKeys.size() >= state.context.batchMaxRows()) {
              flushPrivate(state);
            }
          }
        }
      }
    }
    now = System.currentTimeMillis();
    for (SubscriberState state : snapshot) {
      if (!state.pendingKeys.isEmpty() && now >= state.deadlineMillis) {
        flushPrivate(state);
      }
    }
  }

  private long publicAwaitTimeout(long now) {
    if (publicPendingKeys.isEmpty()) {
      return AWAIT_MAX_MILLIS;
    }
    return Math.max(1L, Math.min(AWAIT_MAX_MILLIS, publicDeadlineMillis - now));
  }

  private long privateAwaitTimeout(List<SubscriberState> states, long now) {
    long deadline = Long.MAX_VALUE;
    for (SubscriberState state : states) {
      if (state.pendingKeys.isEmpty()) {
        continue;
      }
      deadline = Math.min(deadline, state.deadlineMillis);
    }
    if (deadline == Long.MAX_VALUE) {
      return AWAIT_MAX_MILLIS;
    }
    return Math.max(1L, Math.min(AWAIT_MAX_MILLIS, deadline - now));
  }

  private long minSubscriberSeq(List<SubscriberState> states) {
    long min = Long.MAX_VALUE;
    for (SubscriberState state : states) {
      min = Math.min(min, state.lastSeq);
    }
    return min == Long.MAX_VALUE ? 0L : min;
  }

  private void awaitWithoutChanges() {
    awaitForNextKnownSeq(knownSeq, Math.min(250L, AWAIT_MAX_MILLIS));
  }

  private void awaitForNextKnownSeq(long afterSeq, long timeoutMillis) {
    OptionalLong nextSeq =
        changeNotifier.awaitNextSeq(
            datasetId.toString(),
            Math.max(0L, afterSeq),
            Math.max(1L, timeoutMillis),
            () -> Thread.currentThread().isInterrupted() || closed || stopRequested);
    if (nextSeq.isPresent()) {
      knownSeq = Math.max(knownSeq, nextSeq.getAsLong());
    }
  }

  private List<DuckdbActiveChangeLog.ChangeEvent> readChangesAfter(long afterSeq) {
    List<DuckdbActiveChangeLog.ChangeEvent> changes =
        rowManager.readDuckDbActiveChanges(datasetId, Math.max(0L, afterSeq), CHANGE_READ_LIMIT);
    if (!changes.isEmpty()) {
      knownSeq = Math.max(knownSeq, changes.get(changes.size() - 1).seq());
      return changes;
    }
    knownSeq = Math.max(knownSeq, rowManager.duckDbActiveChangeMaxSeq(datasetId));
    return List.of();
  }

  private void flushPublic(List<SubscriberState> snapshot) {
    if (publicPendingKeys.isEmpty()) {
      return;
    }
    List<String> keys = List.copyOf(publicPendingKeys);
    long seq = publicPendingMaxSeq;
    ByteBuffer ipc =
        batchPublisher.buildBatchIpc(
            new StreamSubscriptionContext(
                datasetId,
                "PUBLIC",
                null,
                snapshot.isEmpty() ? null : snapshot.get(0).context.callerAppSubject(),
                snapshot.isEmpty() ? null : snapshot.get(0).context.requestMetadata(),
                snapshot.isEmpty() ? "default" : snapshot.get(0).context.subscriptionId(),
                snapshot.isEmpty() ? null : snapshot.get(0).context.startMode(),
                publicCursorSeq,
                publicBatchMaxRows,
                publicFlushMillis,
                publicFilterSql,
                null),
            primaryKeyColumns,
            keys);
    Delivery delivery = Delivery.data(seq, ipc);
    for (SubscriberState state : snapshot) {
      if (state.closed) {
        continue;
      }
      enqueuePublic(state, delivery);
      state.lastSeq = Math.max(state.lastSeq, seq);
    }
    publicCursorSeq = Math.max(publicCursorSeq, seq);
    publicPendingKeys.clear();
    publicPendingMaxSeq = publicCursorSeq;
    publicDeadlineMillis = 0L;
  }

  private void flushPrivate(SubscriberState state) {
    if (state.closed || state.pendingKeys.isEmpty()) {
      return;
    }
    List<String> keys = List.copyOf(state.pendingKeys);
    long seq = state.pendingMaxSeq;
    ByteBuffer ipc = batchPublisher.buildBatchIpc(state.context, primaryKeyColumns, keys);
    enqueuePrivate(state, Delivery.data(seq, ipc));
    state.lastSeq = Math.max(state.lastSeq, seq);
    state.pendingKeys.clear();
    state.pendingMaxSeq = state.lastSeq;
    state.deadlineMillis = 0L;
  }

  private void enqueuePublic(SubscriberState state, Delivery delivery) {
    if (state.queue.offer(delivery)) {
      return;
    }
    state.queue.poll();
    state.queue.offer(delivery);
  }

  private void enqueuePrivate(SubscriberState state, Delivery delivery) {
    try {
      while (!state.closed && !closed && !Thread.currentThread().isInterrupted()) {
        if (stopRequested) {
          return;
        }
        if (state.queue.offer(delivery, 100L, TimeUnit.MILLISECONDS)) {
          return;
        }
      }
    } catch (InterruptedException ex) {
      Thread.currentThread().interrupt();
      return;
    }
  }

  private void failAllSubscribers(RuntimeException failure) {
    synchronized (lock) {
      for (SubscriberState state : subscribers.values()) {
        state.closed = true;
        state.queue.offer(Delivery.failure(state.lastSeq, failure));
      }
      subscribers.clear();
      lastActivityMillis = System.currentTimeMillis();
    }
  }

  private List<SubscriberState> snapshotSubscribers() {
    synchronized (lock) {
      if (subscribers.isEmpty()) {
        return List.of();
      }
      return new ArrayList<>(subscribers.values());
    }
  }

  private static String normalizeAccessMode(String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null || "PRIVATE".equalsIgnoreCase(normalized)) {
      return "PRIVATE";
    }
    if ("PUBLIC".equalsIgnoreCase(normalized)) {
      return "PUBLIC";
    }
    throw new IllegalArgumentException("accessMode must be PUBLIC or PRIVATE");
  }

  public static final class SubscriberHandle implements AutoCloseable {
    private final DatasetStreamHub hub;
    private final String token;
    private final BlockingQueue<Delivery> queue;
    private final long startSeq;
    private volatile boolean closed;

    private SubscriberHandle(
        DatasetStreamHub hub, String token, BlockingQueue<Delivery> queue, long startSeq) {
      this.hub = hub;
      this.token = token;
      this.queue = queue;
      this.startSeq = startSeq;
    }

    public Delivery poll(long timeoutMillis) throws InterruptedException {
      return queue.poll(Math.max(1L, timeoutMillis), TimeUnit.MILLISECONDS);
    }

    public long startSeq() {
      return startSeq;
    }

    @Override
    public void close() {
      if (closed) {
        return;
      }
      closed = true;
      hub.removeSubscriber(token);
    }
  }

  public record Delivery(long lastSeq, ByteBuffer ipc, RuntimeException failure, boolean terminal) {
    static Delivery data(long lastSeq, ByteBuffer ipc) {
      return new Delivery(lastSeq, ipc, null, false);
    }

    static Delivery failure(long lastSeq, RuntimeException failure) {
      return new Delivery(lastSeq, null, failure, true);
    }

    static Delivery terminal(long lastSeq) {
      return new Delivery(lastSeq, null, null, true);
    }
  }

  private static final class SubscriberState {
    private final String token;
    private final StreamSubscriptionContext context;
    private final BlockingQueue<Delivery> queue;
    private long lastSeq;
    private long pendingMaxSeq;
    private long deadlineMillis;
    private boolean closed;
    private final LinkedHashSet<String> pendingKeys = new LinkedHashSet<>();

    private SubscriberState(
        String token, StreamSubscriptionContext context, BlockingQueue<Delivery> queue, long startSeq) {
      this.token = token;
      this.context = context;
      this.queue = queue;
      this.lastSeq = Math.max(0L, startSeq);
      this.pendingMaxSeq = this.lastSeq;
    }
  }
}
