  static List<ConfigInfo> listConfigs(ConfigManagerImpl manager) {
    manager.ensureOpen();
    String requiredGroup = manager.requiredSessionGroup();
    Map<String, List<ConfigManagerImpl.LayerEntry>> byConfig = manager.groupByConfig(requiredGroup);
    List<String> configIds = new ArrayList<>(byConfig.keySet());
    configIds.sort(String::compareTo);
    List<ConfigInfo> results = new ArrayList<>(configIds.size());
    for (String configId : configIds) {
      List<ConfigManagerImpl.LayerEntry> layers = manager.sortLayers(byConfig.get(configId));
      ConfigManagerImpl.LayerEntry representative = manager.representativeLayer(layers);
      String displayName = null;
      String schemaKey = null;
      Definition metadata = manager.configView.get(DefinitionType.CONFIG, requiredGroup, configId);
      if (metadata instanceof DefinitionConfig config) {
        displayName = Strings.trimToNull(config.id());
        schemaKey = Strings.trimToNull(config.schemaRef());
      } else if (representative != null) {
        DefinitionConfigLayer payload = representative.definition();
        schemaKey = Strings.trimToNull(payload == null ? null : payload.schemaKey());
      }
      results.add(new ConfigInfo(configId, displayName, schemaKey));
    }
    return List.copyOf(results);
  }

  static List<String> listLayers(ConfigManagerImpl manager, String configId) {
    manager.ensureOpen();
    String requiredGroup = manager.requiredSessionGroup();
    String requiredconfigId = manager.required(configId, "configId");
    List<ConfigManagerImpl.LayerEntry> layers =
        manager.sortLayers(manager.findConfigLayers(requiredGroup, requiredconfigId));
    List<String> layerIds = new ArrayList<>(layers.size());
    for (ConfigManagerImpl.LayerEntry layer : layers) {
      layerIds.add(layer.identity().layerId());
    }
    return List.copyOf(layerIds);
  }

  static <T> T resolve(
      ConfigManagerImpl manager, String configId, String layerId, ConfigDecoder<T> decoder) {
    ConfigDecoder<T> requiredDecoder = Objects.requireNonNull(decoder, "decoder");
    ConfigManagerImpl.Resolution resolvedId = manager.resolveId(configId, layerId);
    ConfigManagerImpl.LayerEntry layer =
        manager.requireLayer(resolvedId.group(), resolvedId.configId(), resolvedId.layerId());
    ConfigManagerImpl.AuditContext auditContext = manager.captureAuditContext();
    Instant resolvedAt = Instant.now();
    JsonNode resolved;
    ConfigResolved resolvedAudit;
    if (manager.resolver != null) {
      ConfigResolved resolvedConfig =
          manager.resolver.resolve(
              resolvedId.group(), resolvedId.configId(), resolvedId.layerId(), resolvedAt);
      resolved = manager.deepCopy(manager.resolvedPayloadAsJsonNode(resolvedConfig.payload()));
      resolvedAudit = resolvedConfig;
    } else {
      resolved = manager.deepCopy(manager.resolveDeclaredAt(layer.definition(), resolvedAt));
      resolvedAudit = manager.fallbackResolved(resolvedId, layer, resolved, resolvedAt);
    }
    T decoded = manager.decode(resolvedId, layer, resolved, resolvedAt, null, requiredDecoder);
    manager.publishAudit(
        new ConfigEventResolved(
            manager.envelope(
                ConfigEventType.CONFIG_RESOLVED,
                resolvedId.group(),
                resolvedId.configId(),
                resolvedId.layerId(),
                manager.schemaKey(layer.definition()),
                auditContext,
                resolvedAt),
            "resolve",
            resolvedAt.toString(),
            null,
            resolvedAudit == null ? null : resolvedAudit.updatedTimeUtc(),
            resolvedAudit == null ? null : resolvedAudit.lineage(),
            manager.redactSensitiveAuditPayload(manager.deepCopy(resolved))));
    return decoded;
  }

  static ConfigCodegen generate(
      ConfigManagerImpl manager,
      String configId,
      ConfigManager.Language language,
      String basePackageOrModule) {
    String group = manager.requiredSessionGroup();
    String requiredconfigId = manager.required(configId, "configId");
    List<ConfigManagerImpl.LayerEntry> layers = manager.sortLayers(manager.findConfigLayers(group, requiredconfigId));
    ConfigManagerImpl.LayerEntry representative = manager.representativeLayer(layers);
    if (representative == null || representative.identity() == null) {
      throw new IllegalArgumentException("config not found: " + requiredconfigId);
    }
    ConfigManagerImpl.Resolution resolvedId =
        new ConfigManagerImpl.Resolution(group, requiredconfigId, representative.identity().layerId());
    return manager.generateForLayer(resolvedId, representative, language, basePackageOrModule);
  }

  static ConfigCodegen generate(
      ConfigManagerImpl manager,

    ConfigManagerImpl.Resolution resolvedId = manager.resolveId(configId, layerId);
    ConfigManagerImpl.LayerEntry layer =
        manager.requireLayer(resolvedId.group(), resolvedId.configId(), resolvedId.layerId());
    return manager.generateForLayer(resolvedId, layer, language, basePackageOrModule);
  }

  static List<ConfigHistoryEntry> listHistory(
      ConfigManagerImpl manager, String group, String configId, String layerId, int limit) {
    manager.ensureOpen();
    String requiredGroup = manager.required(group, "group");
    String requiredconfigId = manager.required(configId, "configId");
    String requiredLayerId = manager.required(layerId, "layerId");
    int boundedLimit =
        limit <= 0
            ? ConfigHistoryUtil.DEFAULT_HISTORY_LIMIT
            : Math.min(limit, ConfigHistoryUtil.DEFAULT_HISTORY_LIMIT);
    ConfigManagerImpl.LayerEntry layer = manager.requireLayer(requiredGroup, requiredconfigId, requiredLayerId);
    DefinitionConfigLayer payload = layer.definition();
    List<DefinitionConfigLayer.History> history = payload == null ? null : payload.history();
    if (history == null || history.isEmpty()) {
      return List.of();
    }
    List<ConfigHistoryEntry> results = new ArrayList<>();
    for (DefinitionConfigLayer.History snapshot : history) {
      if (snapshot == null) {
        continue;
      }
      List<String> keys = manager.topLevelKeys(snapshot.declared());
      results.add(
          new ConfigHistoryEntry(
              requiredGroup,
              requiredconfigId,
              requiredLayerId,
              Strings.trimToNull(snapshot.configId()),
              Strings.trimToNull(snapshot.updatedTimeUtc()),
              Strings.trimToNull(snapshot.updatedBy()),
              Strings.trimToNull(snapshot.mode()),
              Strings.trimToNull(snapshot.effectiveAtUtc()),
              keys.size(),
              keys));
      if (results.size() >= boundedLimit) {
        break;
      }
    }
    return List.copyOf(results);
  }

  static ConfigSnapshot getSnapshot(
      ConfigManagerImpl manager,
      String group,
      String configId,
      String layerId,
      String snapshotConfigId) {
    manager.ensureOpen();
    String requiredGroup = manager.required(group, "group");
    String requiredconfigId = manager.required(configId, "configId");
    String requiredLayerId = manager.required(layerId, "layerId");
    String requiredConfigId = manager.required(snapshotConfigId, "snapshotConfigId");
    ConfigManagerImpl.LayerEntry layer = manager.requireLayer(requiredGroup, requiredconfigId, requiredLayerId);
    DefinitionConfigLayer current = layer.definition();
    if (requiredConfigId.equals(Strings.trimToNull(current == null ? null : current.configId()))) {
      DefinitionEnvelope envelope = current == null ? null : current.envelope();
      DefinitionConfigLayer payload = current;
      return new ConfigSnapshot(
          requiredGroup,
          requiredconfigId,
          requiredLayerId,
          requiredConfigId,
          Strings.trimToNull(envelope == null ? null : envelope.updatedTimeUtc()),
          Strings.trimToNull(envelope == null ? null : envelope.updatedBy()),
          manager.deepCopy(payload == null ? null : payload.declared()),
          manager.copyEffective(payload == null ? null : payload.effective()));
    }
    DefinitionConfigLayer payload = current;
    List<DefinitionConfigLayer.History> history = payload == null ? null : payload.history();
    DefinitionConfigLayer.History snapshot = manager.findHistorySnapshot(history, requiredConfigId);
    if (snapshot == null) {
      throw new IllegalArgumentException(
          "history snapshot not found for config "
              + requiredconfigId
              + "/"
              + requiredLayerId
              + " configId="
              + requiredConfigId);
    }
    return new ConfigSnapshot(
        requiredGroup,
        requiredconfigId,
        requiredLayerId,
        Strings.trimToNull(snapshot.configId()),
        Strings.trimToNull(snapshot.updatedTimeUtc()),
        Strings.trimToNull(snapshot.updatedBy()),
        manager.deepCopy(snapshot.declared()),
        ConfigHistoryUtil.copyEffective(snapshot.effective()));
  }

  static ConfigUpdate rollback(ConfigManagerImpl manager, ConfigRollbackRequest request) {
    manager.ensureOpen();
    if (request == null) {
      throw new IllegalArgumentException("request is required");
    }
    String group = manager.required(request.group(), "group");
    String configId = manager.required(request.configId(), "configId");
    String layerId = manager.required(request.layerId(), "layerId");
    String targetConfigId = manager.required(request.targetConfigId(), "targetConfigId");
    ConfigRollbackRequest.Mode mode = Objects.requireNonNull(request.mode(), "mode");
    Instant effectiveAt = request.effectiveAt();
    if (mode == ConfigRollbackRequest.Mode.IMMEDIATE && effectiveAt != null) {
      throw new IllegalArgumentException("effectiveAt must be null when mode is IMMEDIATE");
    }
    if (mode == ConfigRollbackRequest.Mode.SCHEDULED && effectiveAt == null) {
      throw new IllegalArgumentException("effectiveAt is required when mode is SCHEDULED");
    }
    ConfigManagerImpl.LayerEntry layer = manager.requireLayer(group, configId, layerId);
    DefinitionConfigLayer existing = layer.definition();
    DefinitionConfigLayer payload = existing;
    if (payload == null) {
      throw new IllegalStateException("config payload is required");
    }
    DefinitionConfigLayer.History target =
        manager.findHistorySnapshot(payload.history(), targetConfigId);
    if (target == null) {
      throw new IllegalArgumentException(
          "rollback target configId not found in history: "
              + configId
              + "/"
              + layerId
              + " targetConfigId="
              + targetConfigId);
    }
    JsonNode targetDeclared = target.declared();
    if (targetDeclared == null || !targetDeclared.isObject()) {
      throw new IllegalStateException(
          "rollback target declared payload must be a JSON object for configId=" + targetConfigId);
    }
    DefinitionConfigLayer.History currentSnapshot =
        ConfigHistoryUtil.snapshotFromCurrent(existing, mode.name(), effectiveAt, "config-client");
    List<DefinitionConfigLayer.History> rewrittenHistory =
        ConfigHistoryUtil.appendSnapshot(
            payload.history(), currentSnapshot, ConfigHistoryUtil.DEFAULT_HISTORY_LIMIT);
    JsonNode rewrittenDeclared;
    List<DefinitionConfigLayer.Effective> rewrittenEffective;
    if (mode == ConfigRollbackRequest.Mode.SCHEDULED) {
      rewrittenDeclared = manager.deepCopy(payload.declared());
      rewrittenEffective = manager.upsertScheduledEffective(payload.effective(), effectiveAt, null, targetDeclared);
    } else {
      rewrittenDeclared = manager.deepCopy(targetDeclared);
      rewrittenEffective = ConfigHistoryUtil.copyEffective(target.effective());
    }
    DefinitionConfigLayer rewrittenPayload =
        new DefinitionConfigLayer(
            existing.envelope(),
            payload.resourceId(),
            payload.parentId(),
            payload.shortName(),
            payload.configType(),
            payload.schemaKey(),
            payload.displayName(),
            rewrittenDeclared,
            rewrittenEffective,
            rewrittenHistory,
            existing.configId(),
            existing.effectiveConfigId());
    DefinitionConfigLayer updated =
        rewrittenPayload;
    manager.configView.upsert(updated);
    return manager.toUpdateAt(updated, layer.identity(), Instant.now());
  }
