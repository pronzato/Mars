package org.pronzato.fabric.api.connectors.amps.factory;

import java.time.Duration;
import java.util.Arrays;
import java.util.Base64;
import java.util.Objects;
import java.util.function.Consumer;
import org.pronzato.fabric.api.control.security.kerberos.KerberosClient;
import org.pronzato.fabric.api.util.Strings;

public final class AmpsFactoryOptions {
  private final Kerberos kerberos = new Kerberos();
  private final Connection connection = new Connection();

  public Kerberos kerberos() {
    return kerberos;
  }

  public Connection connection() {
    return connection;
  }

  public AmpsFactoryOptions withKerberos(Consumer<Kerberos> customizer) {
    Objects.requireNonNull(customizer, "customizer");
    customizer.accept(kerberos);
    return this;
  }

  public AmpsFactoryOptions withConnection(Consumer<Connection> customizer) {
    Objects.requireNonNull(customizer, "customizer");
    customizer.accept(connection);
    return this;
  }

  public AmpsFactoryOptions copy() {
    AmpsFactoryOptions copy = new AmpsFactoryOptions();
    copy.kerberos.copyFrom(this.kerberos);
    copy.connection.copyFrom(this.connection);
    return copy;
  }

  static AmpsFactoryOptions merge(AmpsFactoryOptions options) {
    return options == null ? new AmpsFactoryOptions() : options;
  }

  public static AmpsFactoryOptions fromSystemProperties() {
    AmpsFactoryOptions options = new AmpsFactoryOptions();
    Kerberos kerb = options.kerberos();
    kerb.enabled = Boolean.parseBoolean(System.getProperty("amps.kerberos.enabled", "false"));
    kerb.principal = System.getProperty("amps.kerberos.principal", kerb.principal);
    kerb.keytab = System.getProperty("amps.kerberos.keytab", kerb.keytab);
    kerb.realm = System.getProperty("amps.kerberos.realm", kerb.realm);
    kerb.kdc = System.getProperty("amps.kerberos.kdc", kerb.kdc);
    kerb.mechanism = System.getProperty("amps.kerberos.mechanism", kerb.mechanism);
    kerb.service = System.getProperty("amps.kerberos.service", kerb.service);
    kerb.saslHost = System.getProperty("amps.kerberos.saslHost", kerb.saslHost);
    kerb.authorizationId =
        System.getProperty("amps.kerberos.authorizationId", kerb.authorizationId);
    kerb.qop = System.getProperty("amps.kerberos.qop", kerb.qop);
    kerb.logonOptions = System.getProperty("amps.kerberos.logonOptions", kerb.logonOptions);
    kerb.debug = Boolean.parseBoolean(System.getProperty("amps.kerberos.debug", "false"));
    String autoRefreshProp = System.getProperty("amps.kerberos.autoRefresh");
    if (autoRefreshProp != null) {
      kerb.autoRefresh = Boolean.parseBoolean(autoRefreshProp);
    }
    String refreshPeriodProp = System.getProperty("amps.kerberos.refreshPeriod");
    if (Strings.isNotBlank(refreshPeriodProp)) {
      kerb.refreshPeriod = parseDuration(refreshPeriodProp, kerb.refreshPeriod);
    }
    String krb5ConfPathProp = System.getProperty("amps.kerberos.krb5ConfPath");
    if (Strings.isNotBlank(krb5ConfPathProp)) {
      kerb.krb5ConfPath = krb5ConfPathProp;
    }
    String krb5ConfBase64Prop = System.getProperty("amps.kerberos.krb5ConfDataBase64");
    if (Strings.isNotBlank(krb5ConfBase64Prop)) {
      try {
        kerb.krb5ConfData = Base64.getDecoder().decode(krb5ConfBase64Prop.trim());
        kerb.krb5ConfPath = null;
      } catch (IllegalArgumentException ex) {
        throw new IllegalStateException("Invalid base64 for amps.kerberos.krb5ConfDataBase64", ex);
      }
    }
    return options;
  }

  public static final class Connection {
    private Duration connectTimeout = Duration.ofSeconds(5);
    private Duration initialDelay = Duration.ofMillis(200);
    private Duration maximumDelay = Duration.ofSeconds(2);
    private Duration maximumRetryTime = Duration.ofSeconds(15);
    private double backoffExponent = 2.0d;
    private double jitter = 0.25d;

    public Duration connectTimeout() {
      return connectTimeout;
    }

    public Connection connectTimeout(Duration value) {
      if (value != null && !value.isNegative() && !value.isZero()) {
        connectTimeout = value;
      }
      return this;
    }

    public Duration initialDelay() {
      return initialDelay;
    }

    public Connection initialDelay(Duration value) {
      if (value != null && !value.isNegative()) {
        initialDelay = value;
      }
      return this;
    }

    public Duration maximumDelay() {
      return maximumDelay;
    }

    public Connection maximumDelay(Duration value) {
      if (value != null && !value.isNegative() && !value.isZero()) {
        maximumDelay = value;
      }
      return this;
    }

    public Duration maximumRetryTime() {
      return maximumRetryTime;
    }

    public Connection maximumRetryTime(Duration value) {
      if (value != null && !value.isNegative() && !value.isZero()) {
        maximumRetryTime = value;
      }
      return this;
    }

    public double backoffExponent() {
      return backoffExponent;
    }

    public Connection backoffExponent(double value) {
      if (value > 1.0d) {
        backoffExponent = value;
      }
      return this;
    }

    public double jitter() {
      return jitter;
    }

    public Connection jitter(double value) {
      if (value >= 0.0d && value <= 1.0d) {
        jitter = value;
      }
      return this;
    }

    void copyFrom(Connection source) {
      if (source == null) {
        return;
      }
      this.connectTimeout = source.connectTimeout;
      this.initialDelay = source.initialDelay;
      this.maximumDelay = source.maximumDelay;
      this.maximumRetryTime = source.maximumRetryTime;
      this.backoffExponent = source.backoffExponent;
      this.jitter = source.jitter;
    }
  }

  public static final class Kerberos {
    private boolean enabled;
    private String principal;
    private String keytab;
    private byte[] keytabBytes;
    private String realm;
    private String kdc;
    private String mechanism = "GSSAPI";
    private String service = "AMPS";
    private String saslHost;
    private String authorizationId;
    private String qop;
    private String logonOptions;
    private boolean debug;
    private boolean autoRefresh = true;
    private Duration refreshPeriod = Duration.ofHours(6);
    private String krb5ConfPath;
    private byte[] krb5ConfData;
    private String cacheKeyPrefix;
    private KerberosClient runtimeContext;

    public Kerberos enabled(boolean value) {
      enabled = value;
      return this;
    }

    public Kerberos principal(String value) {
      principal = value;
      return this;
    }

    public Kerberos keytab(String value) {
      keytab = value;
      return this;
    }

    public Kerberos keytabBytes(byte[] data) {
      keytabBytes = data == null ? null : data.clone();
      keytab = null;
      return this;
    }

    public Kerberos realm(String value) {
      realm = value;
      return this;
    }

    public Kerberos kdc(String value) {
      kdc = value;
      return this;
    }

    public Kerberos mechanism(String value) {
      if (Strings.isNotBlank(value)) {
        mechanism = value;
      }
      return this;
    }

    public Kerberos service(String value) {
      if (Strings.isNotBlank(value)) {
        service = value;
      }
      return this;
    }

    public Kerberos saslHost(String value) {
      saslHost = Strings.isBlank(value) ? null : value;
      return this;
    }

    public Kerberos authorizationId(String value) {
      authorizationId = value;
      return this;
    }

    public Kerberos qop(String value) {
      qop = value;

    }

    public Kerberos logonOptions(String value) {

    }

    public Kerberos debug(boolean value) {

    }

    public Kerberos autoRefresh(boolean value) {
      autoRefresh = value;
      return this;
    }

    public Kerberos refreshPeriod(Duration value) {
      if (value != null && !value.isNegative() && !value.isZero()) {
        refreshPeriod = value;
      }
      return this;
    }

    public Kerberos krb5ConfPath(String value) {
      krb5ConfPath = value;
      return this;
    }

    public Kerberos krb5ConfData(byte[] value) {
      krb5ConfData = value == null ? null : value.clone();
      return this;
    }

    public Kerberos cacheKeyPrefix(String value) {
      cacheKeyPrefix = value;
      return this;
    }

    public Kerberos runtimeContext(KerberosClient value) {
      runtimeContext = value;
      return this;
    }

    public boolean isEnabled() {
      return enabled;
    }

    public String getPrincipal() {
      return principal;
    }

    public String getKeytab() {
      return keytab;
    }

    public byte[] getKeytabBytes() {
      return keytabBytes == null ? null : keytabBytes.clone();
    }

    public String realm() {
      return realm;
    }

    public String kdc() {
      return kdc;
    }

    public String getKrb5ConfPath() {
      return krb5ConfPath;
    }

    public byte[] getKrb5ConfData() {
      return krb5ConfData == null ? null : krb5ConfData.clone();
    }

    public boolean debugEnabled() {
      return debug;
    }

    public boolean hasKrb5Config() {
      return Strings.isNotBlank(krb5ConfPath) || (krb5ConfData != null && krb5ConfData.length > 0);
    }

    public KerberosClient runtimeContext() {
      return runtimeContext;
    }

    boolean enabled() {
      return enabled;
    }

    String mechanism() {
      return mechanism;
    }

    String service() {
      return service;
    }

    String saslHost() {
      return saslHost;
    }

    String authorizationId() {
      return authorizationId;
    }

    String qop() {
      return qop;
    }

    String logonOptions() {
      return Strings.isBlank(logonOptions) ? null : logonOptions;
    }

    boolean debug() {
      return debug;
    }

    boolean autoRefreshEnabled() {
      return autoRefresh;
    }

    Duration refreshPeriod() {
      return refreshPeriod;
    }

    String loginCacheKey() {
      String base =
          String.join(
              "|",
              valueOrEmpty(principal),
              valueOrEmpty(keytab),
              hashBytes(keytabBytes),
              valueOrEmpty(realm),
              valueOrEmpty(kdc),
              valueOrEmpty(krb5ConfPath),
              hashBytes(krb5ConfData));
      String prefix = valueOrEmpty(cacheKeyPrefix);
      if (prefix.isEmpty()) {
        return base;
      }
      return prefix + "|" + base;
    }

    void validate() {
      if (!enabled) {
        return;
      }
      if (runtimeContext != null) {
        return;
      }
      if (Strings.isBlank(principal)) {
        throw new IllegalStateException(
            "Kerberos principal must be provided when Kerberos is enabled.");
      }
      boolean hasPath = Strings.isNotBlank(keytab);
      boolean hasBytes = keytabBytes != null && keytabBytes.length > 0;
      if (hasPath == hasBytes) {
        throw new IllegalStateException(
            "Kerberos keytab is required when Kerberos is enabled. Configure exactly one of keytab(...) or keytabBytes(...).");
      }
    }

    void copyFrom(Kerberos source) {
      if (source == null) {
        return;
      }
      this.enabled = source.enabled;
      this.principal = source.principal;
      this.keytab = source.keytab;
      this.keytabBytes = source.keytabBytes == null ? null : source.keytabBytes.clone();
      this.realm = source.realm;
      this.kdc = source.kdc;
      this.mechanism = source.mechanism;
      this.service = source.service;
      this.saslHost = source.saslHost;
      this.authorizationId = source.authorizationId;
      this.qop = source.qop;
      this.logonOptions = source.logonOptions;
      this.debug = source.debug;
      this.autoRefresh = source.autoRefresh;
      this.refreshPeriod = source.refreshPeriod;
      this.krb5ConfPath = source.krb5ConfPath;
      this.krb5ConfData = source.krb5ConfData == null ? null : source.krb5ConfData.clone();
      this.cacheKeyPrefix = source.cacheKeyPrefix;
      this.runtimeContext = source.runtimeContext;
    }

    private static String hashBytes(byte[] value) {
      if (value == null || value.length == 0) {
        return "";
      }
      return Integer.toString(Arrays.hashCode(value));
    }
  }

  private static Duration parseDuration(String value, Duration fallback) {
    if (value == null) {
      return fallback;
    }
    String trimmed = value.trim();
    if (trimmed.isEmpty()) {
      return fallback;
    }
    try {
      if (trimmed.startsWith("P")) {
        return Duration.parse(trimmed);
      }
      char last = trimmed.charAt(trimmed.length() - 1);
      if (Character.isLetter(last)) {
        long amount = Long.parseLong(trimmed.substring(0, trimmed.length() - 1));
        switch (Character.toUpperCase(last)) {
          case 'H':
            return Duration.ofHours(amount);
          case 'M':
            return Duration.ofMinutes(amount);
          case 'S':
            return Duration.ofSeconds(amount);
          default:
            return fallback;
        }
      }
      return Duration.ofSeconds(Long.parseLong(trimmed));
    } catch (Exception ex) {
      return fallback;
    }
  }

  private static String valueOrEmpty(String value) {
    return value == null ? "" : value;
  }
}
