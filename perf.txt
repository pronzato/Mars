              "S3 credentials are encrypted but no KEK was provided (alias=" + normalizedAlias + ")");
        });
  }

    ConnectionOptions normalizedOptions = normalizeOptions(options);
    CacheKey key =
        new CacheKey(
            ConnectionKind.AMPS, normalizedName, normalizeCacheOptions(normalizedOptions));
    return getOrCreate(key, () -> createAmpsConnection(normalizedName, normalizedOptions));
  }

      throw new IllegalArgumentException("maxInflight is only supported for AMPS connections");

        new CacheKey(
            ConnectionKind.HTTP, normalizedName, normalizeCacheOptions(normalizedOptions));
    return getOrCreate(key, () -> createHttpConnection(normalizedName, normalizedOptions));
  }

  @Override
  public <H> H get(ConnectionRequest<H> request) throws Exception {
    Objects.requireNonNull(request, "request");
    Class<H> handleType = Objects.requireNonNull(request.handleType(), "request.handleType");

    Object builder = request.builder();
    String connectionName = Strings.trimToNull(request.connectionName());
    if (connectionName == null) {
      connectionName = extractConnectionName(builder);
    }
    String normalizedName = requireConnectionName(connectionName);
    ConnectionOptions options = normalizeOptions(request.options());

    DefinitionConnection model = resolveDefinitionConnection(normalizedName, null);
    String providerId =
        Strings.firstNonBlank(
            Strings.trimToNull(request.providerId()),
            Strings.trimToNull(model.providerId()),
            defaultProviderId(model.connectionType()));
    if (providerId == null) {
      throw new IllegalStateException(
          "Connection '" + scopedName(normalizedName) + "' does not define a provider id");
    }

    String normalizedProviderId = providerId.trim().toLowerCase(Locale.ROOT);
    if ("odbc".equals(normalizedProviderId)) {
      OdbcRuntimeGuard.failConnectionRuntime(normalizedName, model.odbc());
    }
    if ("amps".equals(normalizedProviderId)) {
      if (!handleType.isAssignableFrom(AmpsConnection.class)) {
        throw new IllegalArgumentException(
            "AMPS provider handle type "
                + AmpsConnection.class.getName()
                + " is incompatible with requested "
                + handleType.getName());
      }
      return handleType.cast(amps(normalizedName, options));
    }
    if ("http".equals(normalizedProviderId)) {
      if (!handleType.isAssignableFrom(HttpConnection.class)) {
        throw new IllegalArgumentException(
            "HTTP provider handle type "
                + HttpConnection.class.getName()
                + " is incompatible with requested "
                + handleType.getName());
      }
      return handleType.cast(http(normalizedName, options));
    }
    if ("s3".equals(normalizedProviderId)) {
      if (!handleType.isAssignableFrom(S3Connection.class)) {
        throw new IllegalArgumentException(
            "S3 provider handle type "
                + S3Connection.class.getName()
                + " is incompatible with requested "
                + handleType.getName());
      }
      return handleType.cast(s3(normalizedName));
    }

    ConnectionProvider provider = plugins.findConnectionProvider(providerId);
    if (provider == null) {
      throw new IllegalStateException(
          "No connection provider registered for '" + providerId + "'");
    }

    Class<?> providerHandleType = provider.handleType();
    if (providerHandleType == null) {
      throw new IllegalStateException(
          "Connection provider '" + provider.providerId() + "' reports no handle type");
    }
    if (!handleType.isAssignableFrom(providerHandleType)) {
      throw new IllegalArgumentException(
          "Connection provider '"
              + provider.providerId()
              + "' returns "
              + providerHandleType.getName()
              + " which is incompatible with requested "
              + handleType.getName());
    }

    Class<?> builderType = provider.builderType();
    if (builder != null && builderType != null && builderType != Object.class && !builderType.isInstance(builder)) {
      throw new IllegalArgumentException(
          "Connection builder type "
              + builder.getClass().getName()
              + " is not supported by provider '"
              + provider.providerId()
              + "' (expects "
              + builderType.getName()
              + ")");
    }

    ConnectionProviderContext context =
        new ConnectionProviderContext(
            this,
            model,
            builder,
            connectionName,
            normalizedName,
            options,
            sessionContextSupplier.get(),
            Strings.trimToNull(groupSupplier.get()));
    Object handle = provider.open(context);
    if (handle == null) {
      throw new IllegalStateException(
          "Connection provider '" + provider.providerId() + "' returned null");
    }
    if (!handleType.isInstance(handle)) {
      throw new IllegalStateException(
          "Connection provider '"
              + provider.providerId()
              + "' returned "
              + handle.getClass().getName()
              + " which is incompatible with "
              + handleType.getName());
    }
    return handleType.cast(handle);
  }

  public void evictS3Connection(String connectionName) {
    String normalizedName = requireConnectionName(connectionName);
    CacheKey key = new CacheKey(ConnectionKind.S3, normalizedName, NO_OPTIONS);
    AutoCloseable removed = cache.remove(key);
    closeCachedHandle(removed);
  }

  @Override
  public void close() {
    List<AutoCloseable> handles = new ArrayList<>(cache.values());
    cache.clear();
    for (AutoCloseable handle : handles) {
      closeCachedHandle(handle);
    }
  }

  private AmpsConnection createAmpsConnection(String connectionName, ConnectionOptions options) {
    DefinitionConnection definition =
        resolveDefinitionConnection(connectionName, DefinitionConnection.ConnectionType.AMPS);
    List<String> endpoints = resolveAmpsEndpoints(definition, options);
    String clientName = resolveAmpsClientName(connectionName, options);
    AmpsFactoryOptions factoryOptions = buildAmpsFactoryOptions(connectionName, definition, options);

    HAClient client = null;
    try {
      client = AmpsFactory.create(clientName, factoryOptions);
      AmpsFactory.connect(client, endpoints, factoryOptions);
      AmpsFactory.logon(client, factoryOptions);
      // Return a shared handle so user close() does not tear down the cached connection.
      return AmpsConnection.shared(client);
    } catch (Exception ex) {
      closeHaClient(client);
      throw new IllegalStateException("Failed to open AMPS connection '" + connectionName + "'", ex);
    }
  }

  private HttpConnection createHttpConnection(String connectionName, ConnectionOptions options) {
    DefinitionConnection definition =
        resolveDefinitionConnection(connectionName, DefinitionConnection.ConnectionType.HTTP);
    DefinitionConnection.Http http =
        Objects.requireNonNull(
            definition.http(),
            "Connection '" + scopedName(connectionName) + "' is missing http configuration");
    DefinitionConnection.Http.Auth auth =
        http.auth() == null ? DefinitionConnection.Http.Auth.NONE : http.auth();
    if (auth != DefinitionConnection.Http.Auth.NONE) {
      throw new IllegalArgumentException(
          "Connection '"
              + scopedName(connectionName)
              + "' has unsupported HTTP auth mode "
              + auth
              + " (only NONE is supported)");
    }

    URI baseUri = resolveHttpBaseUri(connectionName, definition);
    HttpClient.Builder builder = HttpClient.newBuilder();
    if (options.timeout() != null) {
      builder.connectTimeout(options.timeout());
    }
    HttpClient client = builder.build();
    return new HttpConnection(baseUri, client, options.timeout());
  }

  private S3Connection createS3Connection(String connectionName) {
    DefinitionConnection definition =
        resolveDefinitionConnection(connectionName, DefinitionConnection.ConnectionType.S3);
    DefinitionConnection.S3 s3 =
        Objects.requireNonNull(
            definition.s3(),
            "Connection '" + scopedName(connectionName) + "' is missing s3 configuration");
    ensureNoPlaintextS3Credentials(connectionName, definition, s3);

    String region = Strings.requireNonBlank(s3.region(), "s3.region");
    String bucket = Strings.requireNonBlank(s3.bucket(), "s3.bucket");
    String accessKeyIdEnc = Strings.trimToNull(s3.accessKeyIdEnc());
    String secretAccessKeyEnc = Strings.trimToNull(s3.secretAccessKeyEnc());
    String sessionTokenEnc = Strings.trimToNull(s3.sessionTokenEnc());
    if (accessKeyIdEnc == null || secretAccessKeyEnc == null) {
      throw new IllegalStateException(
          "S3 connection '" + scopedName(connectionName) + "' is missing encrypted credentials");
    }
    String encFormat = Strings.firstNonBlank(s3.encFormat(), SecretCrypto.ENCRYPTION_FORMAT);
    if (!SecretCrypto.ENCRYPTION_FORMAT.equals(encFormat)) {
      throw new IllegalArgumentException("Unsupported S3 credential encryption format: " + encFormat);
    }
    String kekAlias = Strings.firstNonBlank(s3.kekAlias(), SessionSecrets.DEFAULT_KEK_ALIAS);
    String accessKeyId = decryptS3Credential(kekAlias, accessKeyIdEnc);
    String secretAccessKey = decryptS3Credential(kekAlias, secretAccessKeyEnc);
    String sessionToken = sessionTokenEnc == null ? null : decryptS3Credential(kekAlias, sessionTokenEnc);

    return new S3Connection(
            connectionName,
            region,
            bucket,
            Strings.trimToNull(s3.endpoint()),
            s3.pathStyleAccess(),
            Strings.trimToNull(s3.basePrefix()),
            s3.attributes(),
            accessKeyId,
            secretAccessKey,
            sessionToken)
        .shared();
  }

      return new OdbcSessionSupport.OdbcConnectionHealth(
          false,
          normalizedName,
          DefinitionConnection.Odbc.DatabaseType.OTHER,
          null,
          false,
          false,
          null,
          Strings.firstNonBlank(Strings.trimToNull(ex.getMessage()), "ODBC_CONNECTION_TEST_FAILED"),
          null,
          null,

          List.of(),
          false);
    }
    try (DuckdbEngine engine = new DuckdbEngine();

      return new OdbcSessionSupport.OdbcConnectionHealth(
          false,
          spec.connectionName(),
          spec.databaseType(),
          null,
          false,
          false,
          null,
          Strings.firstNonBlank(Strings.trimToNull(ex.getMessage()), "ODBC_CONNECTION_TEST_FAILED"),
          null,

          0L,

          false);
    }
  }

  private String decryptS3Credential(String kekAlias, String encryptedValue) {
    try {
      return secretDecryptor.decrypt(kekAlias, encryptedValue);
    } catch (IllegalStateException ex) {
      throw ex;
    } catch (RuntimeException ex) {
      throw new IllegalStateException(
          "Unable to decrypt S3 credential with provided KEK (alias=" + kekAlias + ")");
    }
  }

  private static void ensureNoPlaintextS3Credentials(
      String connectionName, DefinitionConnection definition, DefinitionConnection.S3 s3) {
    if (hasAnyPlaintextCredential(definition.properties()) || hasAnyPlaintextCredential(s3.attributes())) {
      throw new IllegalArgumentException(
          "Plaintext S3 credentials are not supported for connection '" + connectionName + "'");
    }
  }

  private static boolean hasAnyPlaintextCredential(Map<String, String> map) {
    if (map == null || map.isEmpty()) {
      return false;
    }
    return hasCredentialValue(map, "accessKeyId")
        || hasCredentialValue(map, "secretAccessKey")
        || hasCredentialValue(map, "sessionToken")
        || hasCredentialValue(map, "aws_access_key_id")
        || hasCredentialValue(map, "aws_secret_access_key")
        || hasCredentialValue(map, "aws_session_token")
        || hasCredentialValue(map, "s3_access_key_id")
        || hasCredentialValue(map, "s3_secret_access_key")
        || hasCredentialValue(map, "s3_session_token");
  }

  private static boolean hasCredentialValue(Map<String, String> map, String key) {
    return Strings.trimToNull(map.get(key)) != null;
  }

  private DefinitionConnection resolveDefinitionConnection(
      String connectionName, DefinitionConnection.ConnectionType expectedType) {
    DiscoveryCache cache = configCacheSupplier.get();
    if (cache == null) {
      throw new IllegalStateException("Session DiscoveryCache is not available");
    }
    String group = Strings.trimToNull(groupSupplier.get());
    if (group == null) {
      throw new IllegalStateException("Session group is not resolved");
    }
    Optional<DefinitionConnection> optional = cache.get(DefinitionConnection.class, group, connectionName);
    if (optional.isEmpty()) {
      optional = cache.get(DefinitionConnection.class, PLATFORM_GROUP, connectionName);
    }
    if (optional.isEmpty()) {
      throw new IllegalArgumentException(
          "Connection '"
              + group
              + "/"
              + connectionName
              + "' or '"
              + PLATFORM_GROUP
              + "/"
              + connectionName
              + "' was not found");
    }
    DefinitionConnection definition = optional.get();
    ensureConnectionEnabled(group, connectionName, definition);
    if (expectedType != null) {
      DefinitionConnection.ConnectionType actualType = definition.connectionType();
      if (actualType != expectedType) {
        throw new IllegalArgumentException(
            "Connection '"
                + group
                + "/"
                + connectionName
                + "' has type "
                + actualType
                + ", expected "
                + expectedType);
      }
    }
    return definition;
  }

  private OdbcSessionSupport.OdbcConnectionSpec resolveOdbcConnectionSpec(String connectionName) {
    DefinitionConnection definition =
        resolveDefinitionConnection(connectionName, DefinitionConnection.ConnectionType.ODBC);
    DefinitionConnection.Odbc odbc =
        Objects.requireNonNull(
            definition.odbc(),
            "Connection '" + scopedName(connectionName) + "' is missing ODBC configuration");
    String group = Strings.trimToNull(groupSupplier.get());
    String normalizedGroup = group == null ? PLATFORM_GROUP : group;
    return OdbcSessionSupport.resolveConnection(
        configCacheSupplier.get(),
        normalizedGroup,
        connectionName,
        odbc.databaseType(),
        "connection:" + scopedName(connectionName));
  }

  private URI resolveHttpBaseUri(String connectionName, DefinitionConnection definition) {
    DefinitionConnection.Http http = definition.http();
    String endpoint =
        firstNonBlank(
            http == null ? null : http.endpoints(), definition.properties(), "endpoint", "url", "baseUrl");
    if (endpoint == null) {
      throw new IllegalArgumentException(
          "HTTP connection '" + scopedName(connectionName) + "' is missing endpoints");
    }

    boolean httpsRequired =
        (http != null && http.transport() == DefinitionConnection.Http.Transport.HTTPS)
            || (http != null && http.tls() != null && http.tls().enabled());
    URI uri = parseHttpEndpoint(endpoint, httpsRequired, connectionName);
    return ensureTrailingSlash(uri);
  }

  private AmpsFactoryOptions buildAmpsFactoryOptions(
      String connectionName, DefinitionConnection definition, ConnectionOptions options) {
    AmpsFactoryOptions factoryOptions = AmpsFactoryOptions.fromSystemProperties();
    if (factoryOptions == null) {
      factoryOptions = new AmpsFactoryOptions();
    }
    configureKerberos(factoryOptions);
    SessionContext context = sessionContextSupplier.get();
    String sessionId = context == null ? null : Strings.trimToNull(context.getSessionId());
    if (sessionId != null) {
      factoryOptions.kerberos().cacheKeyPrefix("session:" + sessionId + ":connection:" + connectionName);
    }
    if (options.timeout() != null) {
      factoryOptions.connection().connectTimeout(options.timeout());
    }
    DefinitionConnection.Amps amps = definition.amps();
    if (amps != null && amps.kerberosEnabled()) {
      factoryOptions.kerberos().enabled(true);
      String service = Strings.trimToNull(amps.kerberosService());
      if (service != null) {
        factoryOptions.kerberos().service(service);
      }
      String host = Strings.trimToNull(amps.kerberosHost());
      if (host != null) {
        factoryOptions.kerberos().saslHost(host);
      }
    }
    return factoryOptions;
  }

  private void configureKerberos(AmpsFactoryOptions options) {
    if (options == null) {
      return;
    }
    AmpsFactoryOptions.Kerberos kerberos = options.kerberos();
    if (Strings.isNotBlank(sessionConfig.kerberosPrincipal()) || sessionConfig.kerberosKeytab() != null) {
      kerberos.enabled(true);
    }
    if (Strings.isNotBlank(sessionConfig.kerberosPrincipal())) {
      kerberos.principal(sessionConfig.kerberosPrincipal());
    }
    Path keytab = sessionConfig.kerberosKeytab();
    if (keytab != null) {
      kerberos.keytab(keytab.toAbsolutePath().toString());
    }
    maybeApplyClasspathKrb5Conf(kerberos);
  }

  private static void maybeApplyClasspathKrb5Conf(AmpsFactoryOptions.Kerberos kerberos) {
    if (kerberos == null || !kerberos.isEnabled()) {
      return;
    }
    if (Strings.isNotBlank(kerberos.getKrb5ConfPath())) {
      return;
    }
    byte[] existingData = kerberos.getKrb5ConfData();
    if (existingData != null && existingData.length > 0) {
      return;
    }
    try (InputStream input = ConnectionManagerImpl.class.getClassLoader().getResourceAsStream("krb5.conf")) {
      if (input == null) {
        return;
      }
      byte[] data = input.readAllBytes();
      if (data.length > 0) {
        kerberos.krb5ConfData(data);
      }
    } catch (IOException ignore) {
      // keep default kerberos config lookup if classpath file cannot be loaded
    }
  }

  private List<String> resolveAmpsEndpoints(DefinitionConnection definition, ConnectionOptions options) {
    DefinitionConnection.Amps amps = definition == null ? null : definition.amps();
    String pinnedInstance = options == null ? null : Strings.trimToNull(options.pinnedInstance());
    if (pinnedInstance != null) {
      if (amps == null || amps.instances() == null || amps.instances().isEmpty()) {
        throw new IllegalArgumentException(
            "AMPS connection '"
                + scopedName(definition == null || definition.envelope() == null ? null : definition.envelope().name())
                + "' does not define instances; cannot pin to instance '"
                + pinnedInstance
                + "'");
      }
      DefinitionConnection.Amps.AmpsInstanceSpec matched = null;
      List<String> names = new ArrayList<>();
      for (DefinitionConnection.Amps.AmpsInstanceSpec instance : amps.instances()) {
        if (instance == null) {
          continue;
        }
        String name = Strings.trimToNull(instance.name());
        if (name != null) {
          names.add(name);
        }
        if (name != null && name.equalsIgnoreCase(pinnedInstance)) {
          matched = instance;
        }
      }
      if (matched == null) {
        throw new IllegalArgumentException(
            "AMPS connection '"
                + scopedName(definition == null || definition.envelope() == null ? null : definition.envelope().name())
                + "' does not define instance '"
                + pinnedInstance
                + "'. Available instances="
                + names);
      }
      List<String> matchedUris = normalizedList(matched.uris());
      if (!matchedUris.isEmpty()) {
        return matchedUris;
      }
      throw new IllegalArgumentException(
          "AMPS connection instance '"
              + Strings.trimToNull(matched.name())
              + "' for connection '"
              + scopedName(definition == null || definition.envelope() == null ? null : definition.envelope().name())
              + "' has no URIs configured");
    }

    List<String> uris = normalizedList(amps == null ? null : amps.uris());
    if (!uris.isEmpty()) {
      return uris;
    }
    String fromProperties =
        firstNonBlank(null, definition.properties(), "uris", "uri", "endpoints", "endpoint", "brokers");
    if (fromProperties == null) {
      throw new IllegalArgumentException(
          "AMPS connection '"
              + scopedName(definition.envelope() == null ? null : definition.envelope().name())
              + "' is missing endpoints");
    }
    List<String> resolved = new ArrayList<>();
    for (String value : fromProperties.split(",")) {
      String trimmed = Strings.trimToNull(value);
      if (trimmed != null) {
        resolved.add(trimmed);
      }
    }
    if (resolved.isEmpty()) {
      throw new IllegalArgumentException(
          "AMPS connection '"
              + scopedName(definition.envelope() == null ? null : definition.envelope().name())
              + "' is missing endpoints");
    }
    return List.copyOf(resolved);
  }

  private String resolveAmpsClientName(String connectionName, ConnectionOptions options) {
    String explicit = Strings.trimToNull(options == null ? null : options.clientName());
    if (explicit != null) {
      return explicit;
    }
    SessionContext context = Objects.requireNonNull(sessionContextSupplier.get(), "session context");
    String group = IdentityParsingSupport.requireGroup(groupSupplier.get());
    String app = Strings.requireNonBlank(context.getAppName(), "session.appName");
    String instance = Strings.requireNonBlank(context.getInstance(), "session.instance");
    String base =
        AmpsConfig.buildCpClientId(group, app, instance);
    String token = sanitizeClientToken(connectionName);
    String pinnedInstance = Strings.trimToNull(options == null ? null : options.pinnedInstance());
    if (pinnedInstance != null) {
      token = token + "-inst-" + sanitizeClientToken(pinnedInstance);
    }
    return base + "-conn-" + token;
  }

  private String scopedName(String connectionName) {
    String group = IdentityParsingSupport.requireGroup(groupSupplier.get());
    String name = IdentityParsingSupport.requireName(connectionName, "connectionName");
    return group + "/" + name;
  }

  private static String defaultProviderId(DefinitionConnection.ConnectionType connectionType) {
    if (connectionType == null) {
      return null;
    }
    return switch (connectionType) {
      case AMPS -> "amps";
      case HTTP -> "http";
      case S3 -> "s3";
      case UNKNOWN -> null;
      default -> connectionType.name().toLowerCase(Locale.ROOT);
    };
  }

  private static ConnectionOptions normalizeOptions(ConnectionOptions options) {
    return options == null ? ConnectionOptions.defaults() : options;
  }

  private static String extractConnectionName(Object builder) {
    if (builder == null) {
      return null;
    }
    for (String methodName : List.of("connectionName", "name", "getConnectionName", "getName")) {
      String value = invokeNameMethod(builder, methodName);
      if (value != null) {
        return value;
      }
    }
    return null;
  }

  private static String invokeNameMethod(Object target, String methodName) {
    try {
      Method method = target.getClass().getMethod(methodName);
      if (method.getParameterCount() != 0 || method.getReturnType() != String.class) {
        return null;
      }
      return Strings.trimToNull((String) method.invoke(target));
    } catch (ReflectiveOperationException ex) {
      return null;
    }
  }

  private static String requireConnectionName(String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      throw new IllegalArgumentException("connectionName is required");
    }
    return normalized;
  }

  private static URI ensureTrailingSlash(URI uri) {
    String path = Strings.trimToNull(uri.getPath());
    if (path == null) {
      path = "/";
    } else if (!path.endsWith("/")) {
      path = path + "/";
    }
    try {
      return new URI(uri.getScheme(), uri.getUserInfo(), uri.getHost(), uri.getPort(), path, null, null);
    } catch (URISyntaxException ex) {
      throw new IllegalArgumentException("Invalid HTTP endpoint URI: " + uri, ex);
    }
  }

  private static URI parseHttpEndpoint(String endpoint, boolean httpsRequired, String connectionName) {
    String normalized = Strings.trimToNull(endpoint);
    if (normalized == null) {
      throw new IllegalArgumentException("HTTP endpoint is required for connection '" + connectionName + "'");
    }

    URI parsed;
    try {
      if (normalized.contains("://")) {
        parsed = URI.create(normalized);
      } else {
        String scheme = httpsRequired ? "https://" : "http://";
        parsed = URI.create(scheme + normalized);
      }
    } catch (IllegalArgumentException ex) {
      throw new IllegalArgumentException(
          "Invalid HTTP endpoint for connection '" + connectionName + "': " + endpoint, ex);
    }

    String scheme = Strings.trimToNull(parsed.getScheme());
    if (scheme == null || parsed.getHost() == null) {
      throw new IllegalArgumentException(
          "HTTP endpoint must be an absolute URI for connection '" + connectionName + "'");
    }
    if (httpsRequired && "http".equalsIgnoreCase(scheme)) {
      try {
        parsed =
            new URI(
                "https",
                parsed.getUserInfo(),
                parsed.getHost(),
                parsed.getPort(),
                parsed.getPath(),
                parsed.getQuery(),
                parsed.getFragment());
      } catch (URISyntaxException ex) {
        throw new IllegalArgumentException(
            "Invalid HTTPS endpoint for connection '" + connectionName + "'", ex);
      }
    }
    if (!"http".equalsIgnoreCase(parsed.getScheme()) && !"https".equalsIgnoreCase(parsed.getScheme())) {
      throw new IllegalArgumentException(
          "HTTP endpoint must use http or https scheme for connection '" + connectionName + "'");
    }
    return parsed;
  }

  private static String sanitizeClientToken(String value) {
    String normalized = Strings.requireNonBlank(value, "connection token");
    StringBuilder builder = new StringBuilder(normalized.length());
    for (int i = 0; i < normalized.length(); i++) {
      char ch = normalized.charAt(i);
      boolean allowed =
          (ch >= 'a' && ch <= 'z')
              || (ch >= 'A' && ch <= 'Z')
              || (ch >= '0' && ch <= '9')
              || ch == '-'
              || ch == '_'
              || ch == '.';
      builder.append(allowed ? ch : '-');
    }
    String token = Strings.trimToNull(builder.toString());
    if (token == null) {
      throw new IllegalArgumentException("connection token must contain at least one non-blank character");
    }
    return token;
  }

  private static CacheOptions normalizeCacheOptions(ConnectionOptions options) {
    Duration timeout = options.timeout();
    long timeoutMillis = timeout == null ? -1L : timeout.toMillis();
    return new CacheOptions(
        options.clientName(),
        timeoutMillis,
        options.maxInflight(),
        Strings.trimToNull(options.pinnedInstance()));
  }

  private static void ensureConnectionEnabled(
      String group, String connectionName, DefinitionConnection connection) {
    DefinitionEnvelope envelope = connection == null ? null : connection.envelope();
    if (Boolean.FALSE.equals(envelope == null ? null : envelope.enabled())) {
      throw new IllegalArgumentException("Connection '" + group + "/" + connectionName + "' is disabled");
    }
    LifecycleStatus status = lifecycleStatus(envelope);
    if (status != LifecycleStatus.ACTIVE) {
      throw new IllegalArgumentException(
          "Connection '"
              + group
              + "/"
              + connectionName
              + "' is not active (lifecycleStatus="
              + status
              + ")");
    }
  }

  private static LifecycleStatus lifecycleStatus(DefinitionEnvelope envelope) {
    if (envelope == null) {
      return LifecycleStatus.ACTIVE;
    }
    String status = Strings.trimToNull(envelope.lifecycleStatus());
    if (status == null) {
      return LifecycleStatus.ACTIVE;
    }
    try {
      return LifecycleStatus.valueOf(status.trim().toUpperCase(Locale.ROOT));
    } catch (IllegalArgumentException ex) {
      throw new IllegalArgumentException("Unsupported lifecycleStatus '" + status + "'", ex);
    }
  }

  private static List<String> normalizedList(List<String> values) {
    if (values == null || values.isEmpty()) {
      return List.of();
    }
    List<String> result = new ArrayList<>(values.size());
    for (String value : values) {
      String trimmed = Strings.trimToNull(value);
      if (trimmed != null) {
        result.add(trimmed);
      }
    }
    return result.isEmpty() ? List.of() : List.copyOf(result);
  }

  private static String firstNonBlank(
      List<String> preferredValues, Map<String, String> fallbackMap, String... fallbackKeys) {
    if (preferredValues != null) {
      for (String value : preferredValues) {
        String trimmed = Strings.trimToNull(value);
        if (trimmed != null) {
          return trimmed;
        }
      }
    }
    if (fallbackMap == null || fallbackMap.isEmpty() || fallbackKeys == null) {
      return null;
    }
    for (String key : fallbackKeys) {
      String value = Strings.trimToNull(fallbackMap.get(key));
      if (value != null) {
        return value;
      }
    }
    return null;
  }

  private static void closeHaClient(HAClient client) {
    if (client == null) {
      return;
    }
    try {
      client.disconnect();
    } catch (Exception ignore) {
    }
    try {
      client.close();
    } catch (Exception ignore) {
    }
  }

  @SuppressWarnings("unchecked")
  private <T extends AutoCloseable> T getOrCreate(CacheKey key, ThrowingSupplier<T> supplier) {
    AutoCloseable existing = cache.get(key);
    if (existing != null) {
      return (T) existing;
    }
    T created;
    try {
      created = supplier.get();
    } catch (RuntimeException ex) {
      throw ex;
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to create connection", ex);
    }
    AutoCloseable prior = cache.putIfAbsent(key, created);
    if (prior != null) {
      closeCachedHandle(created);
      return (T) prior;
    }
    return created;
  }

  private static void closeCachedHandle(AutoCloseable handle) {
    if (handle == null) {
      return;
    }
    if (handle instanceof AmpsConnection ampsConnection) {
      closeHaClient(ampsConnection.client());
      return;
    }
    if (handle instanceof S3Connection s3Connection) {
      CloseSupport.closeQuietly(s3Connection.admin());
      return;
    }
    CloseSupport.closeQuietly(handle);
  }
