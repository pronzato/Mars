package org.pronzato.fabric.lite.patch;

import java.io.BufferedReader;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.nio.file.StandardOpenOption;

public final class PatchInstaller {
  private static final String DEFAULT_PATCH_RELATIVE = "docs/patches/patch.txt";

  private PatchInstaller() {}

  public static void main(String[] args) {
    if (args.length > 2) {
      System.err.println(
          "Usage: java org.pronzato.fabric.lite.patch.PatchInstaller [patch.txt] [repoRoot]");
      System.exit(2);
    }

    Path cwd = Paths.get("").toAbsolutePath().normalize();
    Path repoRoot;
    Path patchPath;

    try {
      if (args.length == 0) {
        repoRoot = findRepoRoot(cwd);
        patchPath = repoRoot.resolve(DEFAULT_PATCH_RELATIVE).normalize();
      } else if (args.length == 1) {
        repoRoot = findRepoRoot(cwd);
        patchPath = resolveFrom(cwd, args[0]);
      } else {
        patchPath = resolveFrom(cwd, args[0]);
        repoRoot = resolveFrom(cwd, args[1]);
      }
    } catch (Exception e) {
      System.err.println("Patch initialization failed: " + e.getMessage());
      System.exit(2);
      return;
    }

    try {
      runInstaller(patchPath, repoRoot);
    } catch (Exception e) {
      System.err.println("Patch installation failed: " + e.getMessage());
      System.exit(1);
    }
  }

  private static void runInstaller(Path patchPath, Path repoRoot) throws IOException {
    if (!Files.exists(patchPath)) {
      throw new IllegalArgumentException("Patch file does not exist: " + patchPath);
    }
    if (!Files.isDirectory(repoRoot)) {
      throw new IllegalArgumentException("Repo root is not a directory: " + repoRoot);
    }

    int writes = 0;
    int deletes = 0;
    int deleteSkipped = 0;
    int errors = 0;
    boolean noChanges = false;

    try (BufferedReader reader = Files.newBufferedReader(patchPath, StandardCharsets.UTF_8)) {
      String line;
      int lineNo = 0;
      boolean started = false;

      while ((line = reader.readLine()) != null) {
        lineNo++;
        if (line.isBlank()) {
          continue;
        }
        if ("NO_CHANGES".equals(line)) {
          noChanges = true;
          break;
        }
        if ("BEGIN_FILE".equals(line)) {
          started = true;
          PatchEntry entry = readEntry(reader, lineNo);
          lineNo = entry.lastLineNumber;
          Path target = resolveTarget(repoRoot, entry.path);

          try {
            if (entry.action == Action.WRITE) {
              Path parent = target.getParent();
              if (parent != null) {
                Files.createDirectories(parent);
              }
              Files.writeString(
                  target,
                  entry.content,
                  StandardCharsets.UTF_8,
                  StandardOpenOption.CREATE,
                  StandardOpenOption.TRUNCATE_EXISTING);
              writes++;
              System.out.println("WRITE  " + entry.path);
            } else {
              if (Files.exists(target)) {
                Files.delete(target);
                deletes++;
                System.out.println("DELETE " + entry.path);
              } else {
                deleteSkipped++;
                System.out.println("SKIP_DELETE_MISSING " + entry.path);
              }
            }
          } catch (Exception e) {
            errors++;
            System.err.println("ERROR applying " + entry.action + " for " + entry.path + ": " + e.getMessage());
          }
          continue;
        }

        if (!started) {
          continue;
        }
        throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected BEGIN_FILE or EOF");
      }
    }

    if (noChanges) {
      System.out.println("Patch contains NO_CHANGES. Nothing to apply.");
    }

    System.out.println();
    System.out.println("Patch apply summary");
    System.out.println("  repoRoot            : " + repoRoot);
    System.out.println("  patchFile           : " + patchPath);
    System.out.println("  files written       : " + writes);
    System.out.println("  files deleted       : " + deletes);
    System.out.println("  delete skipped      : " + deleteSkipped);
    System.out.println("  errors              : " + errors);

    if (errors > 0) {
      throw new IllegalStateException("Patch completed with " + errors + " error(s).");
    }
  }

  private static Path findRepoRoot(Path start) {
    Path current = start;
    while (current != null) {
      Path candidatePatch = current.resolve(DEFAULT_PATCH_RELATIVE).normalize();
      if (Files.exists(candidatePatch)) {
        return current;
      }
      current = current.getParent();
    }
    throw new IllegalArgumentException(
        "Could not find repo root containing " + DEFAULT_PATCH_RELATIVE + " starting from " + start);
  }

  private static Path resolveFrom(Path base, String pathText) {
    Path path = Paths.get(pathText);
    if (!path.isAbsolute()) {
      path = base.resolve(path);
    }
    return path.toAbsolutePath().normalize();
  }

  private static PatchEntry readEntry(BufferedReader reader, int beginLineNo) throws IOException {
    int lineNo = beginLineNo;

    String actionLine = reader.readLine();
    lineNo++;
    String pathLine = reader.readLine();
    lineNo++;
    String moduleLine = reader.readLine();
    lineNo++;
    String packageLine = reader.readLine();
    lineNo++;

    String actionValue = parseField(actionLine, "ACTION", lineNo - 3);
    String pathValue = parseField(pathLine, "PATH", lineNo - 2);
    parseField(moduleLine, "MODULE", lineNo - 1); // parsed for validation; not used by installer
    parseField(packageLine, "PACKAGE", lineNo); // parsed for validation; not used by installer

    Action action;
    if ("WRITE".equals(actionValue)) {
      action = Action.WRITE;
    } else if ("DELETE".equals(actionValue)) {
      action = Action.DELETE;
    } else {
      throw new IllegalArgumentException("Invalid ACTION at line " + (lineNo - 3) + ": " + actionValue);
    }

    StringBuilder content = new StringBuilder();
    if (action == Action.WRITE) {
      String contentStart = reader.readLine();
      lineNo++;
      if (!"CONTENT_START".equals(contentStart)) {
        throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected CONTENT_START");
      }

      String line;
      while ((line = reader.readLine()) != null) {
        lineNo++;
        if ("CONTENT_END".equals(line)) {
          break;
        }
        content.append(line).append('\n');
      }
      if (content.length() > 0) {
        content.setLength(content.length() - 1);
      }
      if (line == null) {
        throw new IllegalArgumentException("Malformed patch: missing CONTENT_END for " + pathValue);
      }
    }

    String endFile = reader.readLine();
    lineNo++;
    if (!"END_FILE".equals(endFile)) {
      throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected END_FILE");
    }

    return new PatchEntry(action, pathValue, content.toString(), lineNo);
  }

  private static String parseField(String line, String key, int lineNo) {
    String prefix = key + ":";
    if (line == null || !line.startsWith(prefix)) {
      throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected " + key + ":");
    }
    return line.substring(prefix.length()).trim();
  }

  private static Path resolveTarget(Path repoRoot, String relativePath) {
    String normalizedRelative = relativePath.replace('\\', '/');
    Path target = repoRoot.resolve(normalizedRelative).normalize();
    if (!target.startsWith(repoRoot)) {
      throw new IllegalArgumentException("Unsafe path outside repo root: " + relativePath);
    }
    return target;
  }

  private enum Action {
    WRITE,
    DELETE
  }

  private static final class PatchEntry {
    private final Action action;
    private final String path;
    private final String content;
    private final int lastLineNumber;

    private PatchEntry(Action action, String path, String content, int lastLineNumber) {
      this.action = action;
      this.path = path;
      this.content = content;
      this.lastLineNumber = lastLineNumber;
    }
  }
}

package org.pronzato.fabric.lite.patch;

import java.io.BufferedWriter;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.time.Instant;
import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

public final class PatchGenerator {
  private static final String DEFAULT_PATCH_RELATIVE = "docs/patches/patch.txt";
  private static final String PATCH_FORMAT = "FABRIC_PATCH_V1";

  private PatchGenerator() {}

  public static void main(String[] args) {
    if (args.length > 2) {
      System.err.println(
          "Usage: java org.pronzato.fabric.lite.patch.PatchGenerator [patch.txt] [repoRoot]");
      System.exit(2);
    }

    Path cwd = Paths.get("").toAbsolutePath().normalize();
    Path repoRoot;
    Path patchPath;
    try {
      if (args.length == 0) {
        repoRoot = findRepoRoot(cwd);
        patchPath = repoRoot.resolve(DEFAULT_PATCH_RELATIVE).normalize();
      } else if (args.length == 1) {
        repoRoot = findRepoRoot(cwd);
        patchPath = resolveFrom(cwd, args[0]);
      } else {
        patchPath = resolveFrom(cwd, args[0]);
        repoRoot = resolveFrom(cwd, args[1]);
      }

      generatePatch(repoRoot, patchPath);
    } catch (Exception e) {
      System.err.println("Patch generation failed: " + e.getMessage());
      System.exit(1);
    }
  }

  private static void generatePatch(Path repoRoot, Path patchPath) throws IOException {
    if (!Files.isDirectory(repoRoot)) {
      throw new IllegalArgumentException("Repo root is not a directory: " + repoRoot);
    }

    String baselineSha = runGit(repoRoot, "rev-parse", "HEAD").trim();
    List<String> diffLines = runGitLines(repoRoot, "diff", "--name-status", "HEAD", "--", "*.java");
    List<String> untrackedLines = runGitLines(repoRoot, "ls-files", "--others", "--exclude-standard", "--", "*.java");

    Set<String> writeSet = new HashSet<>();
    Set<String> deleteSet = new HashSet<>();

    for (String line : diffLines) {
      if (line == null || line.isBlank()) {
        continue;
      }
      String[] parts = line.split("\t");
      if (parts.length < 2) {
        continue;
      }
      String status = parts[0].trim();
      if (status.startsWith("D")) {
        deleteSet.add(normalizePath(parts[parts.length - 1]));
      } else if (status.startsWith("R") && parts.length >= 3) {
        // Rename: delete old path, write new path.
        deleteSet.add(normalizePath(parts[1]));
        writeSet.add(normalizePath(parts[2]));
      } else {
        writeSet.add(normalizePath(parts[parts.length - 1]));
      }
    }

    for (String line : untrackedLines) {
      if (line == null || line.isBlank()) {
        continue;
      }
      writeSet.add(normalizePath(line));
    }

    List<PatchEntry> entries = new ArrayList<>();
    for (String path : writeSet) {
      entries.add(PatchEntry.write(path));
    }
    for (String path : deleteSet) {
      if (!writeSet.contains(path)) {
        entries.add(PatchEntry.delete(path));
      }
    }

    entries.sort((a, b) -> {
      int c1 = a.module.compareTo(b.module);
      if (c1 != 0) {
        return c1;
      }
      int c2 = a.packageName.compareTo(b.packageName);
      if (c2 != 0) {
        return c2;
      }
      int c3 = a.fileName.compareTo(b.fileName);
      if (c3 != 0) {
        return c3;
      }
      return a.action.compareTo(b.action);
    });

    int writeCount = (int) entries.stream().filter(e -> e.action == Action.WRITE).count();
    int deleteCount = (int) entries.stream().filter(e -> e.action == Action.DELETE).count();

    Path parent = patchPath.getParent();
    if (parent != null) {
      Files.createDirectories(parent);
    }

    try (BufferedWriter writer = Files.newBufferedWriter(patchPath, StandardCharsets.UTF_8)) {
      writer.write("PATCH_FORMAT: " + PATCH_FORMAT);
      writer.newLine();
      writer.write("BASELINE_SHA: " + baselineSha);
      writer.newLine();
      writer.write("GENERATED_AT_UTC: " + Instant.now().toString());
      writer.newLine();
      writer.write("TOTAL_FILES: " + entries.size());
      writer.newLine();
      writer.write("WRITE_COUNT: " + writeCount);
      writer.newLine();
      writer.write("DELETE_COUNT: " + deleteCount);
      writer.newLine();
      writer.newLine();

      if (entries.isEmpty()) {
        writer.write("NO_CHANGES");
        writer.newLine();
      } else {
        for (PatchEntry entry : entries) {
          writer.write("BEGIN_FILE");
          writer.newLine();
          writer.write("ACTION: " + entry.action.name());
          writer.newLine();
          writer.write("PATH: " + entry.path);
          writer.newLine();
          writer.write("MODULE: " + entry.module);
          writer.newLine();
          writer.write("PACKAGE: " + entry.packageName);
          writer.newLine();

          if (entry.action == Action.WRITE) {
            Path sourceFile = repoRoot.resolve(entry.path).normalize();
            if (!Files.isRegularFile(sourceFile)) {
              throw new IllegalArgumentException("WRITE entry source file missing: " + entry.path);
            }
            String content = Files.readString(sourceFile, StandardCharsets.UTF_8);
            writer.write("CONTENT_START");
            writer.newLine();
            writer.write(content);
            if (!content.endsWith("\n") && !content.endsWith("\r")) {
              writer.newLine();
            }
            writer.write("CONTENT_END");
            writer.newLine();
          }

          writer.write("END_FILE");
          writer.newLine();
          writer.newLine();
        }
      }
    }

    System.out.println("Generated patch: " + patchPath);
    System.out.println("Baseline SHA    : " + baselineSha);
    System.out.println("WRITE entries   : " + writeCount);
    System.out.println("DELETE entries  : " + deleteCount);
    System.out.println("TOTAL entries   : " + entries.size());
  }

  private static String normalizePath(String pathText) {
    return pathText.trim().replace('\\', '/');
  }

  private static String runGit(Path repoRoot, String... args) throws IOException {
    Process process = startGit(repoRoot, args);
    try {
      String output = new String(process.getInputStream().readAllBytes(), StandardCharsets.UTF_8);
      String error = new String(process.getErrorStream().readAllBytes(), StandardCharsets.UTF_8);
      int exit = process.waitFor();
      if (exit != 0) {
        throw new IOException("git command failed (" + String.join(" ", args) + "): " + error.trim());
      }
      return output;
    } catch (InterruptedException e) {
      Thread.currentThread().interrupt();
      throw new IOException("git command interrupted", e);
    }
  }

  private static List<String> runGitLines(Path repoRoot, String... args) throws IOException {
    String output = runGit(repoRoot, args);
    String[] lines = output.split("\\R");
    List<String> result = new ArrayList<>();
    for (String line : lines) {
      if (!line.isBlank()) {
        result.add(line);
      }
    }
    return result;
  }

  private static Process startGit(Path repoRoot, String... args) throws IOException {
    List<String> command = new ArrayList<>();
    command.add("git");
    for (String arg : args) {
      command.add(arg);
    }
    ProcessBuilder pb = new ProcessBuilder(command);
    pb.directory(repoRoot.toFile());
    return pb.start();
  }

  private static Path findRepoRoot(Path start) {
    Path current = start;
    while (current != null) {
      Path gitDir = current.resolve(".git");
      Path patchFile = current.resolve(DEFAULT_PATCH_RELATIVE);
      if (Files.exists(gitDir) || Files.exists(patchFile)) {
        return current;
      }
      current = current.getParent();
    }
    throw new IllegalArgumentException(
        "Could not find repo root (missing .git and " + DEFAULT_PATCH_RELATIVE + ") from " + start);
  }

  private static Path resolveFrom(Path base, String pathText) {
    Path path = Paths.get(pathText);
    if (!path.isAbsolute()) {
      path = base.resolve(path);
    }
    return path.toAbsolutePath().normalize();
  }

  private enum Action {
    WRITE,
    DELETE
  }

  private static final class PatchEntry {
    private final Action action;
    private final String path;
    private final String module;
    private final String packageName;
    private final String fileName;

    private PatchEntry(Action action, String path) {
      this.action = action;
      this.path = path;
      this.module = module(path);
      this.packageName = packageName(path);
      this.fileName = fileName(path);
    }

    private static PatchEntry write(String path) {
      return new PatchEntry(Action.WRITE, path);
    }

    private static PatchEntry delete(String path) {
      return new PatchEntry(Action.DELETE, path);
    }

    private static String module(String path) {
      int slash = path.indexOf('/');
      return slash > 0 ? path.substring(0, slash) : "";
    }

    private static String packageName(String path) {
      String norm = path.replace('\\', '/');
      String marker = "/java/";
      int javaIdx = norm.indexOf(marker);
      if (javaIdx >= 0) {
        int start = javaIdx + marker.length();
        int lastSlash = norm.lastIndexOf('/');
        if (lastSlash > start) {
          return norm.substring(start, lastSlash).replace('/', '.');
        }
      }
      int slash = norm.lastIndexOf('/');
      return slash > 0 ? norm.substring(0, slash) : "";
    }

    private static String fileName(String path) {
      int slash = path.lastIndexOf('/');
      return slash >= 0 ? path.substring(slash + 1) : path;
    }
  }
}

package org.pronzato.fabric.lite.patch;

import java.io.BufferedReader;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;

public final class PatchInstaller {
  private static final String DEFAULT_PATCH_RELATIVE = "docs/patches/patch.txt";

  private PatchInstaller() {}

  public static void main(String[] args) {
    if (args.length > 2) {
      System.err.println(
          "Usage: java org.pronzato.fabric.lite.patch.PatchInstaller [patch.txt] [repoRoot]");
      System.exit(2);
    }

    Path cwd = Paths.get("").toAbsolutePath().normalize();
    Path repoRoot;
    Path patchPath;

    try {
      if (args.length == 0) {
        repoRoot = findRepoRoot(cwd);
        patchPath = repoRoot.resolve(DEFAULT_PATCH_RELATIVE).normalize();
      } else if (args.length == 1) {
        repoRoot = findRepoRoot(cwd);
        patchPath = resolveFrom(cwd, args[0]);
      } else {
        patchPath = resolveFrom(cwd, args[0]);
        repoRoot = resolveFrom(cwd, args[1]);
      }
    } catch (Exception e) {
      System.err.println("Patch initialization failed: " + e.getMessage());
      System.exit(2);
      return;
    }

    try {
      runInstaller(patchPath, repoRoot);
    } catch (Exception e) {
      System.err.println("Patch installation failed: " + e.getMessage());
      System.exit(1);
    }
  }

  private static void runInstaller(Path patchPath, Path repoRoot) throws IOException {
    if (!Files.exists(patchPath)) {
      throw new IllegalArgumentException("Patch file does not exist: " + patchPath);
    }
    if (!Files.isDirectory(repoRoot)) {
      throw new IllegalArgumentException("Repo root is not a directory: " + repoRoot);
    }

    int writes = 0;
    int deletes = 0;
    int deleteSkipped = 0;
    int errors = 0;
    boolean noChanges = false;

    try (BufferedReader reader = Files.newBufferedReader(patchPath, StandardCharsets.UTF_8)) {
      String line;
      int lineNo = 0;
      boolean started = false;

      while ((line = reader.readLine()) != null) {
        lineNo++;
        if (line.isBlank()) {
          continue;
        }
        if ("NO_CHANGES".equals(line)) {
          noChanges = true;
          break;
        }
        if ("BEGIN_FILE".equals(line)) {
          started = true;
          PatchEntry entry = readEntry(reader, lineNo);
          lineNo = entry.lastLineNumber;
          Path target = resolveTarget(repoRoot, entry.path);

          try {
            if (entry.action == Action.WRITE) {
              Path parent = target.getParent();
              if (parent != null) {
                Files.createDirectories(parent);
              }
              Files.writeString(
                  target,
                  entry.content,
                  StandardCharsets.UTF_8,
                  StandardOpenOption.CREATE,
                  StandardOpenOption.TRUNCATE_EXISTING);
              writes++;
              System.out.println("WRITE  " + entry.path);
            } else {
              if (Files.exists(target)) {
                Files.delete(target);
                deletes++;
                System.out.println("DELETE " + entry.path);
              } else {
                deleteSkipped++;
                System.out.println("SKIP_DELETE_MISSING " + entry.path);
              }
            }
          } catch (Exception e) {
            errors++;
            System.err.println("ERROR applying " + entry.action + " for " + entry.path + ": " + e.getMessage());
          }
          continue;
        }

        if (!started) {
          continue;
        }
        throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected BEGIN_FILE or EOF");
      }
    }

    if (noChanges) {
      System.out.println("Patch contains NO_CHANGES. Nothing to apply.");
    }

    System.out.println();
    System.out.println("Patch apply summary");
    System.out.println("  repoRoot            : " + repoRoot);
    System.out.println("  patchFile           : " + patchPath);
    System.out.println("  files written       : " + writes);
    System.out.println("  files deleted       : " + deletes);
    System.out.println("  delete skipped      : " + deleteSkipped);
    System.out.println("  errors              : " + errors);

    if (errors > 0) {
      throw new IllegalStateException("Patch completed with " + errors + " error(s).");
    }
  }

  private static Path findRepoRoot(Path start) {
    Path current = start;
    while (current != null) {
      Path candidatePatch = current.resolve(DEFAULT_PATCH_RELATIVE).normalize();
      if (Files.exists(candidatePatch)) {
        return current;
      }
      current = current.getParent();
    }
    throw new IllegalArgumentException(
        "Could not find repo root containing " + DEFAULT_PATCH_RELATIVE + " starting from " + start);
  }

  private static Path resolveFrom(Path base, String pathText) {
    Path path = Paths.get(pathText);
    if (!path.isAbsolute()) {
      path = base.resolve(path);
    }
    return path.toAbsolutePath().normalize();
  }

  private static PatchEntry readEntry(BufferedReader reader, int beginLineNo) throws IOException {
    int lineNo = beginLineNo;

    String actionLine = reader.readLine();
    lineNo++;
    String pathLine = reader.readLine();
    lineNo++;
    String moduleLine = reader.readLine();
    lineNo++;
    String packageLine = reader.readLine();
    lineNo++;

    String actionValue = parseField(actionLine, "ACTION", lineNo - 3);
    String pathValue = parseField(pathLine, "PATH", lineNo - 2);
    parseField(moduleLine, "MODULE", lineNo - 1); // parsed for validation; not used by installer
    parseField(packageLine, "PACKAGE", lineNo); // parsed for validation; not used by installer

    Action action;
    if ("WRITE".equals(actionValue)) {
      action = Action.WRITE;
    } else if ("DELETE".equals(actionValue)) {
      action = Action.DELETE;
    } else {
      throw new IllegalArgumentException("Invalid ACTION at line " + (lineNo - 3) + ": " + actionValue);
    }

    StringBuilder content = new StringBuilder();
    if (action == Action.WRITE) {
      String contentStart = reader.readLine();
      lineNo++;
      if (!"CONTENT_START".equals(contentStart)) {
        throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected CONTENT_START");
      }

      String line;
      while ((line = reader.readLine()) != null) {
        lineNo++;
        if ("CONTENT_END".equals(line)) {
          break;
        }
        content.append(line).append('\n');
      }
      if (content.length() > 0) {
        content.setLength(content.length() - 1);
      }
      if (line == null) {
        throw new IllegalArgumentException("Malformed patch: missing CONTENT_END for " + pathValue);
      }
    }

    String endFile = reader.readLine();
    lineNo++;
    if (!"END_FILE".equals(endFile)) {
      throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected END_FILE");
    }

    return new PatchEntry(action, pathValue, content.toString(), lineNo);
  }

  private static String parseField(String line, String key, int lineNo) {
    String prefix = key + ":";
    if (line == null || !line.startsWith(prefix)) {
      throw new IllegalArgumentException("Malformed patch at line " + lineNo + ": expected " + key + ":");
    }
    return line.substring(prefix.length()).trim();
  }

  private static Path resolveTarget(Path repoRoot, String relativePath) {
    String normalizedRelative = relativePath.replace('\\', '/');
    Path target = repoRoot.resolve(normalizedRelative).normalize();
    if (!target.startsWith(repoRoot)) {
      throw new IllegalArgumentException("Unsafe path outside repo root: " + relativePath);
    }
    return target;
  }

  private enum Action {
    WRITE,
    DELETE
  }

  private static final class PatchEntry {
    private final Action action;
    private final String path;
    private final String content;
    private final int lastLineNumber;

    private PatchEntry(Action action, String path, String content, int lastLineNumber) {
      this.action = action;
      this.path = path;
      this.content = content;
      this.lastLineNumber = lastLineNumber;
    }
  }

  private static void warmupConnection(
      StitchRuntimeContext runtimeContext, String connectionName) {
    long startedAt = System.nanoTime();
    LOGGER.info("[Stitch][ConnectionWarmup] start connection={}", connectionName);
    try {
      AutoCloseable handle =
          runtimeContext
              .session()
              .connections()
              .get(
                  ConnectionRequest.builder(AutoCloseable.class)
                      .withConnectionName(connectionName)
                      .build());
      String type = touchWarmupHandle(handle);
      long elapsedMs = (System.nanoTime() - startedAt) / 1_000_000L;
      LOGGER.info(
          "[Stitch][ConnectionWarmup] success connection={} type={} elapsedMs={}",
          connectionName,
          type,
          elapsedMs);
    } catch (Exception error) {
      long elapsedMs = (System.nanoTime() - startedAt) / 1_000_000L;
      LOGGER.warn(
          "[Stitch][ConnectionWarmup] failed connection={} elapsedMs={} reason={}",
          connectionName,
          elapsedMs,
          error.getMessage(),
          error);
    }
  }

  private static String touchWarmupHandle(AutoCloseable handle) {
    if (handle == null) {
      throw new IllegalStateException("Connection warmup resolved to null handle");
    }
    if (handle instanceof S3Connection s3) {
      // Touch metadata so lazy connector initialization happens during warmup.
      s3.bucket();
      return "S3";
    }
    if (handle instanceof HttpConnection http) {
      http.baseUri();
      return "HTTP";
    }
    if (handle instanceof AmpsConnection amps) {
      amps.client();
      return "AMPS";
    }
    if (handle instanceof OdbcConnection) {
      return "ODBC";
    }
    return handle.getClass().getSimpleName();
  }

  private static SsoConfig buildSsoConfig(StitchSettings settings) {
    StitchSettings.Sso sso = Objects.requireNonNull(settings, "settings").sso();
    SsoConfig.Spoof spoof =
        SsoConfig.Spoof.builder()
            .withEnabled(sso.spoofEnabled())
            .withAllowEditing(sso.spoofAllowEditing())
            .withCookieName(sso.spoofCookieName())
            .withDefaultUsername(sso.spoofDefaultUsername())
            .withCookieMaxAgeSeconds(sso.spoofCookieMaxAgeSeconds())
            .withAttributeResolver(
                username ->
                    sso.spoofGroups().isEmpty() ? Map.of() : Map.of("groups", List.copyOf(sso.spoofGroups())))
            .build();
    return SsoConfig.builder()
        .withEnabled(sso.enabled())
        .withMode(sso.mode())
        .withForwardedHeadersEnabled(sso.forwardedHeadersEnabled())
        .withOidc(sso.oidc())
        .withSpoof(spoof)
        .build();
  }

    return session;

final class StitchVaadinBundleBootstrap {
  private static final Logger LOGGER = LoggerFactory.getLogger(StitchVaadinBundleBootstrap.class);
  private static final String PROD_BUNDLE_RELATIVE_PATH = "src/main/bundles/prod.bundle";
  private static final String INDEX_RELATIVE_PATH = "webapp/index.html";

  private StitchVaadinBundleBootstrap() {}

  static void ensureOnExplodedClasspath(Class<?> appClass) {
    Path classesRoot = resolveExplodedClassesRoot(appClass);
    if (classesRoot == null) {
      return;
    }
    Path vaadinRoot = classesRoot.resolve("META-INF").resolve("VAADIN");
    Path indexHtml = vaadinRoot.resolve(INDEX_RELATIVE_PATH);
    if (Files.isRegularFile(indexHtml)) {
      return;
    }
    Path bundlePath = resolveProdBundlePath(classesRoot);
    if (bundlePath == null) {
      throw new IllegalStateException(
          "Missing Vaadin production bundle and classpath index.html for "
              + appClass.getName()
              + ". Expected "
              + PROD_BUNDLE_RELATIVE_PATH
              + ". Run Maven process-resources first.");
    }
    try {
      unzipBundle(bundlePath, vaadinRoot);
    } catch (IOException error) {
      throw new UncheckedIOException(
          "Unable to unpack Vaadin bundle from " + bundlePath + " to " + vaadinRoot, error);
    }
    LOGGER.info("[Stitch][VaadinBundle] unpacked bundle={} destination={}", bundlePath, vaadinRoot);
  }

  private static Path resolveExplodedClassesRoot(Class<?> appClass) {
    try {
      Path location =
          Path.of(appClass.getProtectionDomain().getCodeSource().getLocation().toURI())
              .toAbsolutePath()
              .normalize();
      if (!Files.isDirectory(location)) {
        return null;
      }
      return isExplodedClassesDirectory(location) ? location : null;
    } catch (URISyntaxException | NullPointerException error) {
      return null;
    }
  }

  private static boolean isExplodedClassesDirectory(Path location) {
    String normalized = location.toString().replace('\\', '/').toLowerCase(Locale.ROOT);
    return normalized.endsWith("/target/classes")
        || normalized.contains("/target/classes/")
        || normalized.endsWith("/build/classes/java/main")
        || normalized.contains("/build/classes/java/main/");
  }

  private static Path resolveProdBundlePath(Path classesRoot) {
    for (Path root : candidateRoots(classesRoot)) {
      Path candidate = root.resolve(PROD_BUNDLE_RELATIVE_PATH).normalize();
      if (Files.isRegularFile(candidate)) {
        return candidate;
      }
    }
    return null;
  }

  private static List<Path> candidateRoots(Path classesRoot) {
    List<Path> candidates = new ArrayList<>();
    Path current = classesRoot;
    for (int depth = 0; depth < 8 && current != null; depth++) {
      candidates.add(current);
      current = current.getParent();
    }
    Path cwd = Path.of("").toAbsolutePath().normalize();
    candidates.add(cwd);
    if (cwd.getParent() != null) {
      candidates.add(cwd.getParent());
      if (cwd.getParent().getParent() != null) {
        candidates.add(cwd.getParent().getParent());
      }
    }
    return candidates;
  }

  private static void unzipBundle(Path bundlePath, Path destinationRoot) throws IOException {
    Files.createDirectories(destinationRoot);
    try (ZipInputStream zip = new ZipInputStream(Files.newInputStream(bundlePath))) {
      ZipEntry entry;
      while ((entry = zip.getNextEntry()) != null) {
        Path outputPath = destinationRoot.resolve(entry.getName()).normalize();
        if (!outputPath.startsWith(destinationRoot)) {
          throw new IOException("Invalid bundle entry path: " + entry.getName());
        }
        if (entry.isDirectory()) {
          Files.createDirectories(outputPath);
          continue;
        }
        Path parent = outputPath.getParent();
        if (parent != null) {
          Files.createDirectories(parent);
        }
        Files.copy(zip, outputPath, StandardCopyOption.REPLACE_EXISTING);
      }
    }
  }
}

    if (fromToken != null) {

      if (dashboard == null || dashboard.envelope() == null) {
        continue;

        return Optional.of(

      }

    if (normalized == null) {

      String name = Strings.trimToNull(normalized.substring(separator + 1));

  private void enforceVisibility(DefinitionDashboard dashboard) {
    DefinitionDashboard.DashboardVisibility visibility =
        dashboard.visibility() == null
            ? DefinitionDashboard.DashboardVisibility.GROUP
            : dashboard.visibility();
    if (visibility == DefinitionDashboard.DashboardVisibility.PUBLIC) {
      return;
    }
    if (visibility == DefinitionDashboard.DashboardVisibility.GROUP) {
      String contextGroup = Strings.trimToNull(StitchGroupSelectionContext.current());
      String dashboardGroup =
          Strings.trimToNull(dashboard.envelope() == null ? null : dashboard.envelope().group());
      if (contextGroup == null || dashboardGroup == null || contextGroup.equalsIgnoreCase(dashboardGroup)) {
        return;

      throw new IllegalStateException(
          "Dashboard '"
              + dashboardGroup
              + "/"
              + dashboard.envelope().name()
              + "' is scoped to group "
              + dashboardGroup
              + ".");

    String owner = normalizeSubject(owner(dashboard));
    if (subject != null && subject.equals(owner)) {
      return;

    throw new IllegalStateException(
        "Dashboard '" + dashboard.envelope().group() + "/" + dashboard.envelope().name() + "' is user-scoped.");

        Strings.trimToNull(dashboard.envelope().createdBy()));

    String normalized = Strings.trimToNull(value);

  private static Map<String, String> queryContext(BeforeEvent event) {
    if (event == null || event.getLocation() == null) {
      return Map.of();

    Map<String, List<String>> parameters = event.getLocation().getQueryParameters().getParameters();
    if (parameters == null || parameters.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, String> context = new LinkedHashMap<>();
    for (Map.Entry<String, List<String>> entry : parameters.entrySet()) {
      if (entry == null) {
        continue;
      }
      String key = Strings.trimToNull(entry.getKey());
      if (key == null || !key.startsWith("ctx_")) {
        continue;
      }
      String value = first(entry.getValue());
      if (value == null) {
        continue;
      }
      context.put(key.substring("ctx_".length()), value);
    }
    return context.isEmpty() ? Map.of() : Map.copyOf(context);
  }

  private static String firstQuery(BeforeEvent event, String key) {
    if (event == null || event.getLocation() == null || Strings.trimToNull(key) == null) {
      return null;
    }
    Map<String, List<String>> parameters = event.getLocation().getQueryParameters().getParameters();
    if (parameters == null) {
      return null;
    }
    return first(parameters.get(key));
  }

  private static String first(List<String> values) {
    if (values == null || values.isEmpty()) {
      return null;
    }
    return Strings.trimToNull(values.get(0));
  }

    scheduleRefreshPolicies();
  }

  private static String buildSubtitle(StitchDashboardDescriptor descriptor) {
    if (descriptor == null) {
      return "";
    }
    String base = Strings.firstNonBlank(descriptor.subtitle(), descriptor.description(), "");
    String runtime =
        descriptor.visibility().name() + " / " + descriptor.executionMode().name();
    StitchDashboardDescriptor.DashboardSemantic semantic = descriptor.semantic();
    if (semantic == null) {
      if (Strings.trimToNull(base) == null) {
        return runtime;
      }
      return base + " | " + runtime;
    }
    String semanticDomain =
        Strings.firstNonBlank(
            Strings.trimToNull(semantic.domain()),
            Strings.trimToNull(semantic.subdomain()));
    if (Strings.trimToNull(semanticDomain) == null) {
      if (Strings.trimToNull(base) == null) {
        return runtime;
      }
      return base + " | " + runtime;
    }
    if (Strings.trimToNull(base) == null) {
      return semanticDomain + " | " + runtime;
    }
    return base + " | " + semanticDomain + " | " + runtime;
  }

  private void rebuildFilters(StitchDashboardDescriptor selected) {
    filtersBar.clear();
    advancedFields.removeAll();
    appliedAdvancedFilters = Map.of();
    activeSharedFilters = Map.of();
    sharedFilters.clear();

    boolean hasAdvanced = false;
    for (StitchDashboardDescriptor.SharedFilter descriptor : selected.sharedFilters()) {
      if (descriptor == null) {
        continue;
      }
      DashboardFilterField field = createFilterField(descriptor);
      SharedFilterBinding binding = new SharedFilterBinding(descriptor, field);
      sharedFilters.put(normalizeKey(descriptor.id()), binding);
      if (descriptor.advanced()) {
        hasAdvanced = true;
        advancedFields.add(wrapAdvancedField(descriptor.label(), field.component()));
      } else {
        filtersBar.withFilter(field.component());
      }
      field.onChange(
          () -> {
            if (!descriptor.advanced()) {
              applyTopFilters();
            }
          });
    }

    filtersBar.withRightAction(clearLinksAction).withRightAction(refreshAction);
    advancedPanel.setVisible(hasAdvanced);

    if (!runtimeState.filters().isEmpty()) {
      applyContextToFields(runtimeState.filters());
    }
    if (!runtimeState.advancedFilters().isEmpty()) {
      applyContextToFields(runtimeState.advancedFilters());
    }
    Set<String> consumedKeys = applyContextToFields(dashboardContext);
    dashboardContext = stripConsumedContext(dashboardContext, consumedKeys);
    appliedAdvancedFilters = collectFilterValues(true);
    applyTopFilters();
  }

  private void rebuildWidgets(StitchDashboardDescriptor selected) {
    widgets.clear();
    rows.removeAll();

    if (buildSectionsLayout(selected)) {
      return;
    }

    for (StitchDashboardDescriptor.LayoutRow layoutRow : selected.layoutRows()) {
      if (layoutRow == null || layoutRow.widgetIds().isEmpty()) {
        continue;
      }
      HorizontalLayout row = new HorizontalLayout();
      row.setPadding(false);
      row.setSpacing(false);
      row.setWidthFull();
      row.getStyle().set("gap", resolveLayoutGap(selected.layout()));
      row.addClassName("stitch-dashboard-row");

      for (String widgetId : layoutRow.widgetIds()) {
        StitchDashboardDescriptor.Widget descriptor = selected.widget(widgetId);
        if (descriptor == null) {
          continue;
        }
        WidgetController controller = new WidgetController(descriptor);
        widgets.put(normalizeKey(descriptor.id()), controller);
        Component sectionComponent = controller.section();
        row.add(sectionComponent);
        row.setFlexGrow(Math.max(1.0d, descriptor.layout().colSpan()), sectionComponent);
      }

      if (row.getComponentCount() > 0) {
        rows.add(row);
      }
    }
  }

  private boolean buildSectionsLayout(StitchDashboardDescriptor selected) {
    if (selected == null || selected.layout() == null || selected.layout().sections().isEmpty()) {
      return false;
    }
    for (StitchDashboardDescriptor.LayoutSection section : selected.layout().sections()) {
      if (section == null || section.placements().isEmpty()) {
        continue;
      }
      VerticalLayout sectionContainer = new VerticalLayout();
      sectionContainer.setPadding(false);
      sectionContainer.setSpacing(false);
      sectionContainer.setWidthFull();
      sectionContainer.getStyle().set("gap", "0.45rem");
      sectionContainer.addClassName("stitch-dashboard-layout-section");

      String title = Strings.trimToNull(section.title());
      if (title != null) {
        Span heading = new Span(title);
        heading.addClassName("stitch-dashboard-layout-section-title");
        sectionContainer.add(heading);
      }

      Div sectionGrid = new Div();
      sectionGrid.setWidthFull();
      sectionGrid.addClassName("stitch-dashboard-layout-grid");
      Integer sectionColumns = section.columns();
      Integer dashboardColumns = selected.layout().columns();
      int columns =
          Math.max(
              1,
              sectionColumns != null
                  ? sectionColumns
                  : (dashboardColumns == null ? 1 : dashboardColumns));
      sectionGrid.getStyle().set("display", "grid");
      sectionGrid.getStyle().set("grid-template-columns", "repeat(" + columns + ", minmax(0, 1fr))");
      sectionGrid.getStyle().set("gap", resolveLayoutGap(selected.layout()));

      for (StitchDashboardDescriptor.LayoutPlacement placement : section.placements()) {
        if (placement == null || Strings.trimToNull(placement.widgetId()) == null) {
          continue;
        }
        StitchDashboardDescriptor.Widget descriptor = selected.widget(placement.widgetId());
        if (descriptor == null) {
          LOGGER.warn(
              "[StitchDashboard] layout section references missing widget id={} dashboard={}",
              placement.widgetId(),
              selected.id());
          continue;
        }
        WidgetController controller = new WidgetController(descriptor);
        widgets.put(normalizeKey(descriptor.id()), controller);
        Component sectionComponent = controller.section();
        StitchDashboardDescriptor.WidgetLayout resolved = resolveWidgetLayout(descriptor, placement);
        sectionComponent.getStyle().set("grid-column", "span " + resolved.colSpan());
        sectionComponent.getStyle().set("grid-row", "span " + resolved.rowSpan());
        sectionGrid.add(sectionComponent);
      }
      if (sectionGrid.getChildren().findAny().isPresent()) {
        sectionContainer.add(sectionGrid);
        rows.add(sectionContainer);
      }
    }
    return !widgets.isEmpty();
  }

  private StitchDashboardDescriptor.WidgetLayout resolveWidgetLayout(
      StitchDashboardDescriptor.Widget descriptor, StitchDashboardDescriptor.LayoutPlacement placement) {
    StitchDashboardDescriptor.WidgetLayout widgetLayout =
        descriptor == null || descriptor.layout() == null
            ? StitchDashboardDescriptor.WidgetLayout.singleCell()
            : descriptor.layout();
    if (placement == null) {
      return widgetLayout;
    }
    Integer placementCol = placement.colSpan();
    Integer placementRow = placement.rowSpan();
    if (placementCol != null
        && widgetLayout.colSpan() != null
        && !placementCol.equals(widgetLayout.colSpan())) {
      LOGGER.warn(
          "[StitchDashboard] widget layout colSpan conflict dashboard={} widget={} sectionPlacement={} widgetLayout={} (section placement wins)",
          dashboard == null ? null : dashboard.id(),
          descriptor == null ? null : descriptor.id(),
          placementCol,
          widgetLayout.colSpan());
    }
    if (placementRow != null
        && widgetLayout.rowSpan() != null
        && !placementRow.equals(widgetLayout.rowSpan())) {
      LOGGER.warn(
          "[StitchDashboard] widget layout rowSpan conflict dashboard={} widget={} sectionPlacement={} widgetLayout={} (section placement wins)",
          dashboard == null ? null : dashboard.id(),
          descriptor == null ? null : descriptor.id(),
          placementRow,
          widgetLayout.rowSpan());
    }
    return new StitchDashboardDescriptor.WidgetLayout(
        placementCol == null ? widgetLayout.colSpan() : placementCol,
        placementRow == null ? widgetLayout.rowSpan() : placementRow);
  }

  private static String resolveLayoutGap(StitchDashboardDescriptor.DashboardLayout layout) {
    String gap = layout == null ? null : Strings.trimToNull(layout.gap());
    if (gap == null) {
      return "0.75rem";
    }
    return switch (gap.toLowerCase(Locale.ROOT)) {
      case "xs" -> "0.25rem";
      case "sm" -> "0.45rem";
      case "md" -> "0.75rem";
      case "lg" -> "1rem";
      case "xl" -> "1.25rem";
      default -> gap;
    };
  }

  private void applyTopFilters() {
    LinkedHashMap<String, String> combined = new LinkedHashMap<>(collectFilterValues(false));
    combined.putAll(appliedAdvancedFilters);
    activeSharedFilters = combined.isEmpty() ? Map.of() : Map.copyOf(combined);
    persistRuntimeState();
    refreshAll(RefreshTrigger.FILTER_CHANGE, false);
  }

  private void applyAdvancedFilters() {
    appliedAdvancedFilters = collectFilterValues(true);
    LinkedHashMap<String, String> combined = new LinkedHashMap<>(collectFilterValues(false));
    combined.putAll(appliedAdvancedFilters);
    activeSharedFilters = combined.isEmpty() ? Map.of() : Map.copyOf(combined);
    persistRuntimeState();
    refreshAll(RefreshTrigger.FILTER_CHANGE, false);
  }

  private void clearAdvancedFilters() {
    for (SharedFilterBinding binding : sharedFilters.values()) {
      if (binding == null || !binding.descriptor().advanced()) {
        continue;
      }
      binding.field().clear();
    }
    appliedAdvancedFilters = Map.of();
    LinkedHashMap<String, String> combined = new LinkedHashMap<>(collectFilterValues(false));
    activeSharedFilters = combined.isEmpty() ? Map.of() : Map.copyOf(combined);
    persistRuntimeState();
    refreshAll(RefreshTrigger.FILTER_CHANGE, false);
  }

  private void clearLinkedFilters() {
    linkContextByWidget.clear();
    persistRuntimeState();
    refreshAll(RefreshTrigger.LINK_EVENT, false);
  }

  private Set<String> applyContextToFields(Map<String, String> context) {
    if (context == null || context.isEmpty()) {
      return Set.of();
    }
    LinkedHashSet<String> consumed = new LinkedHashSet<>();
    for (SharedFilterBinding binding : sharedFilters.values()) {
      if (binding == null) {
        continue;
      }
      consumed.addAll(binding.field().applyContext(context));
    }
    return consumed.isEmpty() ? Set.of() : Set.copyOf(consumed);
  }

  private static Map<String, String> stripConsumedContext(Map<String, String> context, Set<String> consumed) {
    if (context == null || context.isEmpty() || consumed == null || consumed.isEmpty()) {
      return context == null ? Map.of() : Map.copyOf(context);
    }
    LinkedHashMap<String, String> remaining = new LinkedHashMap<>();
    for (Map.Entry<String, String> entry : context.entrySet()) {
      if (entry == null) {
        continue;
      }
      String key = Strings.trimToNull(entry.getKey());
      String value = Strings.trimToNull(entry.getValue());
      if (key == null || value == null || consumed.contains(normalizeKey(key))) {
        continue;
      }
      remaining.put(key, value);
    }
    return remaining.isEmpty() ? Map.of() : Map.copyOf(remaining);
  }

  private Map<String, String> collectFilterValues(boolean advancedOnly) {
    LinkedHashMap<String, String> collected = new LinkedHashMap<>();
    for (SharedFilterBinding binding : sharedFilters.values()) {
      if (binding == null) {
        continue;
      }
      if (advancedOnly && !binding.descriptor().advanced()) {
        continue;
      }
      if (!advancedOnly && binding.descriptor().advanced()) {
        continue;
      }
      collected.putAll(binding.field().valueMap());
    }
    return collected.isEmpty() ? Map.of() : Map.copyOf(collected);
  }

  private DashboardFilterField createFilterField(StitchDashboardDescriptor.SharedFilter descriptor) {
    String filterId = descriptor.id();
    String label = Strings.firstNonBlank(Strings.trimToNull(descriptor.label()), filterId);
    List<String> resolvedOptions = resolveFilterOptions(descriptor);
    return switch (descriptor.type()) {
      case MULTI_SELECT -> new MultiSelectFilterField(filterId, label, resolvedOptions, descriptor.defaultValues());
      case TEXT -> new TextFilterField(filterId, label, descriptor.defaultValue());
      case TOGGLE -> new ToggleFilterField(filterId, label, descriptor.defaultValue());
      case NUMBER -> new NumberFilterField(filterId, label, descriptor.defaultValue());
      case DATE -> new DateFilterField(filterId, label, descriptor.defaultValue());
      case TIME -> new TimeFilterField(filterId, label, descriptor.defaultValue());
      case DATE_TIME_RANGE -> new DateTimeRangeFilterField(filterId, label);
      case SELECT -> {
        if (isGroupFilter(descriptor)) {
          yield new GroupFilterField(filterId, runtimeContext, descriptor.defaultValue());
        }
        yield new SelectFilterField(filterId, label, resolvedOptions, descriptor.defaultValue());
      }
    };
  }

  private static boolean isGroupFilter(StitchDashboardDescriptor.SharedFilter descriptor) {
    if (descriptor == null) {
      return false;
    }
    return "group".equalsIgnoreCase(Strings.firstNonBlank(descriptor.id(), ""))
        && (descriptor.options() == null || descriptor.options().isEmpty());
  }

  private List<String> resolveFilterOptions(StitchDashboardDescriptor.SharedFilter descriptor) {
    if (descriptor == null) {
      return List.of();
    }
    if (descriptor.options() != null && !descriptor.options().isEmpty()) {
      return descriptor.options();
    }
    if (Strings.trimToNull(descriptor.sourceRef()) == null) {
      return List.of();
    }
    try {
      return loadFilterOptionsFromSource(descriptor);
    } catch (Exception error) {
      LOGGER.warn(
          "[StitchDashboard] unable to load filter options for filter={} source={} message={}",
          descriptor.id(),
          descriptor.sourceRef(),
          error.getMessage());
      return List.of();
    }
  }

  private List<String> loadFilterOptionsFromSource(StitchDashboardDescriptor.SharedFilter descriptor) {
    if (descriptor == null || Strings.trimToNull(descriptor.sourceRef()) == null) {
      return List.of();
    }
    String table = sourceTable(descriptor.sourceRef());
    if (Strings.trimToNull(table) == null) {
      return List.of();
    }
    String valueField =
        Strings.firstNonBlank(
            Strings.trimToNull(descriptor.sourceValueField()),
            Strings.trimToNull(descriptor.id()));
    if (Strings.trimToNull(valueField) == null) {
      return List.of();
    }
    String labelField =
        Strings.firstNonBlank(
            Strings.trimToNull(descriptor.sourceLabelField()),
            valueField);
    String valueColumn = quotedIdentifier(valueField);
    String labelColumn = quotedIdentifier(labelField);
    String sql =
        "select distinct "
            + valueColumn
            + " as value, "
            + labelColumn
            + " as label from "
            + table
            + " where "
            + valueColumn
            + " is not null order by label asc limit 250";
    RowResult result = runtimeContext.session().data().direct().rows().querySql(sql, 250);
    if (result == null || result.rows().isEmpty()) {
      return List.of();
    }
    int valueIndex = columnIndex(result.columns(), "value", 0);
    int labelIndex = columnIndex(result.columns(), "label", valueIndex);
    LinkedHashSet<String> values = new LinkedHashSet<>();
    for (List<Object> row : result.rows()) {
      if (row == null || row.isEmpty()) {
        continue;
      }
      String label = rowValue(row, labelIndex);
      String value = rowValue(row, valueIndex);
      String option = Strings.firstNonBlank(label, value);
      if (option != null) {
        values.add(option);
      }
    }
    return values.isEmpty() ? List.of() : List.copyOf(values);
  }

  private static String sourceTable(String sourceRef) {
    String normalized = Strings.trimToNull(sourceRef);
    if (normalized == null) {
      return null;
    }
    if (normalized.startsWith("dataset:")) {
      DatasetId datasetId = DatasetId.parse(normalized);
      return qualifiedTable(datasetId.group(), datasetId.namespace(), datasetId.name());
    }
    if (normalized.startsWith("stream:")) {
      String value = normalized.substring("stream:".length());
      String[] parts = value.split("/");
      if (parts.length == 3) {
        return qualifiedTable(parts[0], parts[1], parts[2]);
      }
      return null;
    }
    if (normalized.contains("/")) {
      String[] parts = normalized.split("/");
      if (parts.length == 3) {
        return qualifiedTable(parts[0], parts[1], parts[2]);
      }
      if (parts.length == 2) {
        return quotedIdentifier(parts[0]) + "." + quotedIdentifier(parts[1]);
      }
      return null;
    }
    if (normalized.contains(".")) {
      String[] parts = normalized.split("\\.");
      if (parts.length == 2) {
        return quotedIdentifier(parts[0]) + "." + quotedIdentifier(parts[1]);
      }
      if (parts.length == 3) {
        return qualifiedTable(parts[0], parts[1], parts[2]);
      }
    }
    return quotedIdentifier("default") + "." + quotedIdentifier(normalized);
  }

  private static String qualifiedTable(String group, String namespace, String name) {
    return quotedIdentifier(Strings.requireNonBlank(group, "group"))
        + "."
        + quotedIdentifier(Strings.requireNonBlank(namespace, "namespace"))
        + "."
        + quotedIdentifier(Strings.requireNonBlank(name, "name"));
  }

  private static String quotedIdentifier(String value) {
    String identifier = Strings.requireNonBlank(value, "identifier");
    return "\"" + identifier.replace("\"", "\"\"") + "\"";
  }

  private static int columnIndex(List<String> columns, String key, int fallback) {
    if (columns == null || columns.isEmpty()) {
      return Math.max(0, fallback);
    }
    String normalizedKey = Strings.trimToNull(key);
    if (normalizedKey == null) {
      return Math.max(0, fallback);
    }
    for (int index = 0; index < columns.size(); index++) {
      String candidate = Strings.trimToNull(columns.get(index));
      if (candidate != null && candidate.equalsIgnoreCase(normalizedKey)) {
        return index;
      }
    }
    return Math.min(Math.max(0, fallback), Math.max(0, columns.size() - 1));
  }

  private static String rowValue(List<Object> row, int index) {
    if (row == null || index < 0 || index >= row.size()) {
      return null;
    }
    Object value = row.get(index);
    return Strings.trimToNull(value == null ? null : String.valueOf(value));
  }

  private void refreshAll(RefreshTrigger trigger, boolean force) {
    for (WidgetController controller : widgets.values()) {
      if (controller == null) {
        continue;
      }
      controller.refresh(trigger, force);
    }
  }

  private void disposeWidgets() {
    for (WidgetController controller : widgets.values()) {
      if (controller == null) {
        continue;
      }
      controller.dispose();
    }
    widgets.clear();
  }

  private void scheduleRefreshPolicies() {
    cancelScheduledRefreshes();
    for (WidgetController controller : widgets.values()) {
      if (controller == null) {
        continue;
      }
      if (!controller.scheduledRefreshEnabled()) {
        continue;
      }
      StitchDashboardDescriptor.RefreshPolicy policy = controller.policy();
      if (policy == null) {
        continue;
      }
      if (policy.mode() != StitchDashboardDescriptor.RefreshMode.INTERVAL
          && policy.mode() != StitchDashboardDescriptor.RefreshMode.LIVE) {
        continue;
      }
      long seconds =
          Math.max(
              1L,
              policy.intervalSeconds() == null
                  ? policy.mode() == StitchDashboardDescriptor.RefreshMode.LIVE
                      ? LIVE_DEFAULT_SECONDS
                      : INTERVAL_DEFAULT_SECONDS
                  : policy.intervalSeconds());
      ScheduledFuture<?> scheduled =
          scheduler.scheduleAtFixedRate(
              () ->
                  getUI()
                      .ifPresent(
                          ui ->
                              ui.access(
                                  () ->
                                      controller.refresh(
                                          policy.mode() == StitchDashboardDescriptor.RefreshMode.LIVE
                                              ? RefreshTrigger.LIVE
                                              : RefreshTrigger.INTERVAL,
                                          false))),
              seconds,
              seconds,
              TimeUnit.SECONDS);
      scheduledRefreshes.add(scheduled);
    }
  }

  private void cancelScheduledRefreshes() {

  private void handleWidgetEvent(
      WidgetController source,
      StitchDashboardDescriptor.EventType eventType,
      Map<String, String> payload) {
    if (source == null || eventType == null || dashboard == null) {
      return;
    }
    applyLinks(source, eventType, payload);
    applyDrilldowns(source, eventType, payload);
  }

  private void applyLinks(
      WidgetController source, StitchDashboardDescriptor.EventType eventType, Map<String, String> payload) {
    if (dashboard.links().isEmpty()) {
      return;
    }
    boolean stateChanged = false;
    for (StitchDashboardDescriptor.Link link : dashboard.links()) {
      if (link == null || !equalsKey(link.sourceWidgetId(), source.widget().id())) {
        continue;
      }
      if (link.event() != eventType) {
        continue;
      }
      for (String targetWidgetId : link.targetWidgetIds()) {
        WidgetController target = widgets.get(normalizeKey(targetWidgetId));
        if (target == null) {
          continue;
        }
        Map<String, String> targetContext =
            linkContextByWidget.computeIfAbsent(target.widget().id(), ignored -> new LinkedHashMap<>());
        Map<String, String> mapped = mapPayloadForLink(link, payload);
        if (mapped.isEmpty()) {
          removeLinkKeys(link, targetContext);
          stateChanged = true;
        } else {
          targetContext.putAll(mapped);
          stateChanged = true;
        }
        target.refresh(RefreshTrigger.LINK_EVENT, false);
      }
    }
    if (stateChanged) {
      persistRuntimeState();
    }
  }

  private static Map<String, String> mapPayloadForLink(
      StitchDashboardDescriptor.Link link, Map<String, String> payload) {
    if (payload == null || payload.isEmpty() || link == null) {
      return Map.of();
    }
    if (link.payloadToTargetParameter() == null || link.payloadToTargetParameter().isEmpty()) {
      return Map.copyOf(payload);
    }
    LinkedHashMap<String, String> mapped = new LinkedHashMap<>();
    for (Map.Entry<String, String> entry : link.payloadToTargetParameter().entrySet()) {
      if (entry == null) {
        continue;
      }
      String sourceKey = Strings.trimToNull(entry.getKey());
      String targetKey = Strings.trimToNull(entry.getValue());
      if (sourceKey == null || targetKey == null) {
        continue;
      }
      String value = payload.get(sourceKey);
      if (Strings.trimToNull(value) != null) {
        mapped.put(targetKey, value);
      }
    }
    return mapped.isEmpty() ? Map.of() : Map.copyOf(mapped);
  }

  private static void removeLinkKeys(StitchDashboardDescriptor.Link link, Map<String, String> context) {
    if (link == null || context == null || context.isEmpty()) {
      return;
    }
    if (link.payloadToTargetParameter() == null || link.payloadToTargetParameter().isEmpty()) {
      context.clear();
      return;
    }
    for (String key : link.payloadToTargetParameter().values()) {
      if (Strings.trimToNull(key) != null) {
        context.remove(key);
      }
    }
  }

  private void applyDrilldowns(
      WidgetController source, StitchDashboardDescriptor.EventType eventType, Map<String, String> payload) {
    StitchDashboardDescriptor.Widget widget = source.widget();
    if (widget == null || widget.drilldowns().isEmpty()) {
      return;
    }
    for (StitchDashboardDescriptor.Drilldown drilldown : widget.drilldowns()) {
      if (drilldown == null || drilldown.event() != eventType) {
        continue;
      }
      Map<String, String> context = mapPayloadForDrilldown(drilldown, payload);
      if (drilldown.includeSharedFilters() && !activeSharedFilters.isEmpty()) {
        LinkedHashMap<String, String> merged = new LinkedHashMap<>(activeSharedFilters);
        merged.putAll(context);
        context = Map.copyOf(merged);
      }
      if (drilldown.type() == StitchDashboardDescriptor.DrilldownType.FILTER) {
        applyFilterDrilldown(context);
      } else if (drilldown.type() == StitchDashboardDescriptor.DrilldownType.DASHBOARD) {
        openDashboardDrilldown(drilldown.target(), context);
      } else {
        openDialogDrilldown(widget.title(), drilldown.target(), context);
      }
    }
  }

  private void applyFilterDrilldown(Map<String, String> context) {
    if (context == null || context.isEmpty()) {
      return;
    }
    Set<String> consumed = applyContextToFields(context);
    if (consumed.isEmpty()) {
      return;
    }
    applyTopFilters();
  }

  private static Map<String, String> mapPayloadForDrilldown(
      StitchDashboardDescriptor.Drilldown drilldown, Map<String, String> payload) {
    if (drilldown == null) {
      return Map.of();
    }
    if (payload == null || payload.isEmpty()) {
      return Map.of();
    }
    if (drilldown.payloadToParameter() == null || drilldown.payloadToParameter().isEmpty()) {
      return Map.copyOf(payload);
    }
    LinkedHashMap<String, String> mapped = new LinkedHashMap<>();
    for (Map.Entry<String, String> mapping : drilldown.payloadToParameter().entrySet()) {
      if (mapping == null) {
        continue;
      }
      String payloadKey = Strings.trimToNull(mapping.getKey());
      String targetKey = Strings.trimToNull(mapping.getValue());
      if (payloadKey == null || targetKey == null) {
        continue;
      }
      String value = payload.get(payloadKey);
      if (Strings.trimToNull(value) != null) {
        mapped.put(targetKey, value);
      }
    }
    return mapped.isEmpty() ? Map.of() : Map.copyOf(mapped);
  }

  private void openDashboardDrilldown(String targetDashboardId, Map<String, String> context) {
    String target = Strings.trimToNull(targetDashboardId);
    if (target == null) {

    }
    if (dashboardNavigationHandler != null) {

      return;

  private static VaadinIcon resolveIcon(String iconName) {

    if (value == null) {
      return VaadinIcon.PLAY;
    }

  }

  private Map<String, String> resolveActionParameters(
      StitchDashboardDescriptor.Action action, Map<String, String> payload, Map<String, String> linked) {
    if (action == null || action.parameters().isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, String> resolved = new LinkedHashMap<>();
    for (Map.Entry<String, String> entry : action.parameters().entrySet()) {
      if (entry == null) {
        continue;
      }
      String key = Strings.trimToNull(entry.getKey());
      String value = resolveActionValue(entry.getValue(), payload, activeSharedFilters, linked);
      if (key != null && value != null) {
        resolved.put(key, value);
      }
    }
    if (action.includeSharedFilters() && !activeSharedFilters.isEmpty()) {
      for (Map.Entry<String, String> shared : activeSharedFilters.entrySet()) {
        if (shared == null) {
          continue;
        }
        String key = Strings.trimToNull(shared.getKey());
        String value = Strings.trimToNull(shared.getValue());
        if (key != null && value != null && !resolved.containsKey(key)) {
          resolved.put(key, value);
        }
      }
    }
    return resolved.isEmpty() ? Map.of() : Map.copyOf(resolved);
  }

  private static String resolveActionValue(
      String raw,
      Map<String, String> payload,
      Map<String, String> sharedFilters,
      Map<String, String> linkedFilters) {
    String value = Strings.trimToNull(raw);
    if (value == null) {
      return null;
    }
    if (value.startsWith("{{") && value.endsWith("}}") && value.length() > 4) {
      String token = Strings.trimToNull(value.substring(2, value.length() - 2));
      if (token == null) {
        return null;
      }
      String resolved =
          Strings.firstNonBlank(
              payload == null ? null : payload.get(token),
              sharedFilters == null ? null : sharedFilters.get(token),
              linkedFilters == null ? null : linkedFilters.get(token));
      return Strings.trimToNull(resolved);
    }
    return value;
  }

  private static void showActionNotification(String message, boolean success) {

    notification.setPosition(Notification.Position.BOTTOM_START);

    } else {
      notification.addThemeVariants(NotificationVariant.LUMO_ERROR);
    }
    notification.setDuration(2400);
  }

  private static String normalizeDisplay(Object value) {
    String normalized = Strings.trimToNull(value == null ? null : String.valueOf(value));
    return normalized == null ? "" : normalized;
  }

  private static String formatColumnValue(
      Object value, StitchDashboardDescriptor.ColumnSpec column) {
    if (column == null || Strings.trimToNull(column.format()) == null) {
      return normalizeDisplay(value);
    }
    String format = column.format().trim().toLowerCase(Locale.ROOT);
    if ("currency".equals(format) && value instanceof Number number) {
      return String.format(Locale.US, "$%,.2f", number.doubleValue());
    }
    if (("int".equals(format) || "integer".equals(format)) && value instanceof Number number) {
      return String.format(Locale.US, "%,d", number.longValue());
    }
    if ("number".equals(format) && value instanceof Number number) {
      return String.format(Locale.US, "%,.4f", number.doubleValue());
    }
    return normalizeDisplay(value);
  }

  private static String normalizeKey(String value) {
    String normalized = Strings.trimToNull(value);
    return normalized == null ? null : normalized.toLowerCase(Locale.ROOT);
  }

  private static boolean equalsKey(String left, String right) {
    String l = normalizeKey(left);
    String r = normalizeKey(right);
    return l != null && l.equals(r);
  }

  private static Map<String, String> normalizeMap(Map<String, String> values) {
    if (values == null || values.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, String> normalized = new LinkedHashMap<>();
    for (Map.Entry<String, String> entry : values.entrySet()) {
      if (entry == null) {
        continue;
      }
      String key = Strings.trimToNull(entry.getKey());
      String value = Strings.trimToNull(entry.getValue());
      if (key != null && value != null) {
        normalized.put(key, value);
      }
    }
    return normalized.isEmpty() ? Map.of() : Map.copyOf(normalized);
  }

  private static List<StitchDashboardDescriptor.ColumnSpec> normalizeColumnSpecs(
      StitchDashboardDescriptor.Widget widget, List<Map<String, Object>> rows) {
    if (widget != null && widget.columnSpecs() != null && !widget.columnSpecs().isEmpty()) {
      List<StitchDashboardDescriptor.ColumnSpec> visible = new ArrayList<>();
      for (StitchDashboardDescriptor.ColumnSpec spec : widget.columnSpecs()) {
        if (spec != null && !Boolean.FALSE.equals(spec.visible())) {
          visible.add(spec);
        }
      }
      if (!visible.isEmpty()) {
        return List.copyOf(visible);
      }
    }
    if (rows == null || rows.isEmpty()) {
      return List.of();
    }
    LinkedHashSet<String> columns = new LinkedHashSet<>();
    for (Map<String, Object> row : rows) {
      if (row != null) {
        columns.addAll(row.keySet());
      }
      if (!columns.isEmpty()) {
        break;
      }
    }
    if (columns.isEmpty()) {
      return List.of();
    }
    List<StitchDashboardDescriptor.ColumnSpec> derived = new ArrayList<>();
    for (String column : columns) {
      if (Strings.trimToNull(column) == null) {
        continue;
      }
      derived.add(new StitchDashboardDescriptor.ColumnSpec(column, column, null, Boolean.TRUE, null));
    }
    return derived.isEmpty() ? List.of() : List.copyOf(derived);
  }

  private static Map<String, String> payloadFromRow(
      StitchDashboardDescriptor.Widget widget, Map<String, Object> row) {
    if (row == null || row.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, String> payload = new LinkedHashMap<>();
    for (Map.Entry<String, Object> entry : row.entrySet()) {
      if (entry == null) {
        continue;
      }
      String key = Strings.trimToNull(entry.getKey());
      String value = Strings.trimToNull(String.valueOf(entry.getValue()));
      if (key != null && value != null) {
        payload.put(key, value);
      }
    }
    if (widget != null) {
      addAliasPayload(payload, "key", widget.keyField(), row);
      addAliasPayload(payload, "label", widget.labelField(), row);
      addAliasPayload(payload, "value", widget.valueField(), row);
      addAliasPayload(payload, "tone", widget.toneField(), row);
    }
    return payload.isEmpty() ? Map.of() : Map.copyOf(payload);
  }

  private static void addAliasPayload(
      Map<String, String> payload, String alias, String fieldName, Map<String, Object> row) {
    String field = Strings.trimToNull(fieldName);
    if (field == null || row == null) {
      return;
    }
    Object value = row.get(field);
    String normalized = Strings.trimToNull(value == null ? null : String.valueOf(value));
    if (normalized != null) {
      payload.put(alias, normalized);
    }
  }

    private void refresh(RefreshTrigger trigger, boolean force) {
      if (liveStreamWidget) {
        refreshLiveStream(trigger, force);
        return;
      }
      if (!force && !shouldRefresh(trigger, policy)) {
        return;
      }
      try {
        List<Map<String, Object>> rows = loadRowsForWidget(widget.id());
        if (widget.type() == StitchDashboardDescriptor.WidgetType.KPI_TILES) {
          renderTiles(rows);
        } else if (widget.type() == StitchDashboardDescriptor.WidgetType.BAR_CHART) {
          renderBars(rows);
        } else if (widget.type() == StitchDashboardDescriptor.WidgetType.ACTION_PANEL) {
          renderActionPanel();
        } else {
          renderTable(rows);
        }
      } catch (RuntimeException error) {
        String message =
            Strings.firstNonBlank(
                Strings.trimToNull(error.getMessage()),
                "Widget failed to load data.");
        LOGGER.warn(
            "[StitchDashboard] widget data refresh failed dashboard={} widget={} message={}",
            dashboard == null ? null : dashboard.id(),
            widget.id(),
            message);
        renderWidgetError(message);
      }
    }

    private void refreshLiveStream(RefreshTrigger trigger, boolean force) {
      if (!force && (trigger == RefreshTrigger.INTERVAL || trigger == RefreshTrigger.LIVE)) {
        return;
      }
      if (streamController == null) {
        bindStreamWidget(true);
        return;
      }
      if (force || trigger == RefreshTrigger.FILTER_CHANGE || trigger == RefreshTrigger.LINK_EVENT) {
        bindStreamWidget(isStreamActive(streamController.state()));
        return;
      }
      if (trigger == RefreshTrigger.MANUAL) {
        if (isStreamActive(streamController.state())) {
          streamController.restart();
        } else {
          streamController.start();
        }
      }
    }

      handleWidgetEvent(this, StitchDashboardDescriptor.EventType.ITEM_SELECTED, payload);
    }

    private void bindStreamWidget(boolean autoStart) {
      Map<String, String> linked = linkedFilters(widget.id());
      Map<String, String> parameters = resolveWidgetParameters(widget, source, linked);
      String streamId = resolveStreamId(source, parameters);
      if (Strings.trimToNull(streamId) == null) {
        renderLiveStreamError("Live stream widget is missing streamId/ref.");
        closeStreamBinding();
        return;
      }

      closeStreamBinding();
      try {
        switch (liveStreamRenderMode) {
          case EVENT -> bindEventStream(streamId, parameters, autoStart);
          case TABLE -> bindTableStream(streamId, parameters, autoStart);
          case KPI -> bindKpiStream(streamId, parameters, autoStart);
        }
      } catch (Exception error) {
        renderLiveStreamError(
            Strings.firstNonBlank(
                Strings.trimToNull(error.getMessage()), "Unable to initialize live stream widget."));
      }
    }

    private void bindEventStream(String streamId, Map<String, String> parameters, boolean autoStart) {
      StitchStreamsProvider.StreamEventContext context = streamsProvider.loadEventContext(streamId).orElse(null);
      if (context == null || !context.viewSupported()) {
        renderLiveStreamError(
            context == null
                ? "Stream not found for live event widget."
                : Strings.firstNonBlank(
                    Strings.trimToNull(context.viewUnavailableReason()),
                    "Live event view is not supported for this stream."));
        return;
      }
      int maxEvents = intParameter(parameters, 500, "streamMaxRows", "maxRows", "limit");
      eventLog.setMaxEvents(maxEvents);
      StitchRealtimeSubscriptionController<StitchStreamsProvider.StreamEvent> controller =
          new StitchRealtimeSubscriptionController<>(
              streamsProvider.eventSourceFor(context),
              streamsProvider.defaultEventPolicy(),
              streamsProvider.defaultEventScope(context),
              getUI().orElse(null));
      streamController = controller;
      streamBinding = eventLog.bindController(controller);
      if (autoStart || boolParameter(parameters, true, "streamAutoStart", "autoStart")) {
        controller.start();
      }
    }

    private void bindTableStream(String streamId, Map<String, String> parameters, boolean autoStart) {
      StitchStreamsProvider.StreamTabularContext context = streamsProvider.loadTabularContext(streamId).orElse(null);
      if (context == null || !context.viewSupported()) {
        renderLiveStreamError(
            context == null
                ? "Stream not found for live table widget."
                : Strings.firstNonBlank(
                    Strings.trimToNull(context.viewUnavailableReason()),
                    "Live table view is not supported for this stream."));
        return;
      }
      int maxRows = intParameter(parameters, 500, "streamMaxRows", "maxRows", "limit");
      configureLiveTableColumns(context.schemaColumns());
      StitchRealtimeSubscriptionController<
              StitchLiveTableDelta<StitchStreamsProvider.TabularStreamRow, String>>
          controller =
              new StitchRealtimeSubscriptionController<>(
                  streamsProvider.tabularSourceFor(context, new StitchStreamsProvider.TabularViewOptions(maxRows)),
                  streamsProvider.defaultTabularPolicy(),
                  streamsProvider.defaultTabularScope(context),
                  getUI().orElse(null));
      streamController = controller;
      streamBinding =
          liveTable.bindController(
              controller,
              updates -> {
                ensureLiveTableColumnsFromUpdates(updates);
                liveTable.refreshGrid();
              });
      if (autoStart || boolParameter(parameters, true, "streamAutoStart", "autoStart")) {
        controller.start();
      }
    }

    private void bindKpiStream(String streamId, Map<String, String> parameters, boolean autoStart)
        throws Exception {
      StitchStreamsProvider.StreamTabularContext context = streamsProvider.loadTabularContext(streamId).orElse(null);
      if (context == null || !context.viewSupported()) {
        renderLiveStreamError(
            context == null
                ? "Stream not found for live KPI widget."
                : Strings.firstNonBlank(
                    Strings.trimToNull(context.viewUnavailableReason()),
                    "Live KPI requires tabular stream support."));
        return;
      }
      int maxRows = intParameter(parameters, 100, "streamMaxRows", "maxRows", "limit");
      StitchRealtimeSubscriptionController<
              StitchLiveTableDelta<StitchStreamsProvider.TabularStreamRow, String>>
          controller =
              new StitchRealtimeSubscriptionController<>(
                  streamsProvider.tabularSourceFor(context, new StitchStreamsProvider.TabularViewOptions(maxRows)),
                  streamsProvider.defaultTabularPolicy(),
                  streamsProvider.defaultTabularScope(context),
                  getUI().orElse(null));
      streamController = controller;
      streamBinding =
          controller.addListener(
              new StitchRealtimeSubscriptionController.Listener<>() {
                @Override
                public void onUpdates(
                    List<StitchRealtimeUpdate<StitchLiveTableDelta<StitchStreamsProvider.TabularStreamRow, String>>>
                        updates) {
                  if (updates == null || updates.isEmpty()) {
                    return;
                  }
                  for (StitchRealtimeUpdate<StitchLiveTableDelta<StitchStreamsProvider.TabularStreamRow, String>>
                      update : updates) {
                    if (update == null || update.payload() == null || update.payload().rows().isEmpty()) {
                      continue;
                    }
                    StitchStreamsProvider.TabularStreamRow row =
                        update.payload().rows().get(update.payload().rows().size() - 1);
                    updateLiveKpi(row);
                  }
                }
              });
      if (autoStart || boolParameter(parameters, true, "streamAutoStart", "autoStart")) {
        controller.start();
      }
    }

    private void ensureLiveTableColumnsFromUpdates(
        List<StitchRealtimeUpdate<StitchLiveTableDelta<StitchStreamsProvider.TabularStreamRow, String>>> updates) {
      if (!visibleLiveColumns.isEmpty() || updates == null || updates.isEmpty()) {
        return;
      }
      for (StitchRealtimeUpdate<StitchLiveTableDelta<StitchStreamsProvider.TabularStreamRow, String>> update :
          updates) {
        if (update == null || update.payload() == null || update.payload().rows().isEmpty()) {
          continue;
        }
        StitchStreamsProvider.TabularStreamRow row = update.payload().rows().get(0);
        if (row == null || row.values().isEmpty()) {
          continue;
        }
        configureLiveTableColumns(new ArrayList<>(row.values().keySet()));
        return;
      }
    }

    private void configureLiveTableColumns(List<String> schemaColumns) {
      List<String> requested =
          widget.columns() != null && !widget.columns().isEmpty()
              ? widget.columns()
              : (schemaColumns == null ? List.of() : schemaColumns);
      List<String> columns = normalizeStringList(requested);
      if (columns.equals(visibleLiveColumns)) {
        return;
      }
      visibleLiveColumns = columns;
      Grid<StitchStreamsProvider.TabularStreamRow> grid = liveTable.resultsView().grid();
      grid.removeAllColumns();
      grid.addColumn(
              row ->
                  row == null || row.eventTimeUtc() == null ? "-" : row.eventTimeUtc().toString())
          .setHeader("EVENT")
          .setAutoWidth(true)
          .setFlexGrow(0);
      grid.addColumn(row -> row == null ? "-" : Strings.firstNonBlank(row.operation(), "-"))
          .setHeader("OP")
          .setAutoWidth(true)
          .setFlexGrow(0);
      for (String column : columns) {
        if (Strings.trimToNull(column) == null) {
          continue;
        }
        String key = column;
        grid.addColumn(
                row ->
                    row == null || row.values().isEmpty()
                        ? ""
                        : normalizeDisplay(row.values().get(key)))
            .setHeader(key.toUpperCase(Locale.ROOT))
            .setAutoWidth(true)
            .setFlexGrow(1);
      }
    }

    private void emitLiveRowSelection(StitchStreamsProvider.TabularStreamRow row) {
      if (row == null) {
        return;
      }
      Map<String, String> payload = payloadFromLiveRow(row);
      handleWidgetEvent(this, StitchDashboardDescriptor.EventType.ROW_SELECTED, payload);
      handleWidgetEvent(this, StitchDashboardDescriptor.EventType.ITEM_SELECTED, payload);
    }

    private void emitEventSelection(StitchStreamsProvider.StreamEvent row) {
      if (row == null) {
        return;
      }
      LinkedHashMap<String, String> payload = new LinkedHashMap<>();
      payload.put("streamId", Strings.firstNonBlank(row.streamId(), "-"));
      payload.put("topic", Strings.firstNonBlank(row.topic(), "-"));
      payload.put("summary", Strings.firstNonBlank(row.summary(), "-"));
      payload.put("payload", Strings.firstNonBlank(row.payloadPreview(), "-"));
      if (row.eventTimeUtc() != null) {
        payload.put("eventTimeUtc", row.eventTimeUtc().toString());
      }
      Map<String, String> immutable = payload.isEmpty() ? Map.of() : Map.copyOf(payload);
      handleWidgetEvent(this, StitchDashboardDescriptor.EventType.ROW_SELECTED, immutable);
      handleWidgetEvent(this, StitchDashboardDescriptor.EventType.ITEM_SELECTED, immutable);
    }

    private Map<String, String> payloadFromLiveRow(StitchStreamsProvider.TabularStreamRow row) {
      LinkedHashMap<String, String> payload = new LinkedHashMap<>();
      if (row.values() != null) {
        for (Map.Entry<String, Object> entry : row.values().entrySet()) {
          if (entry == null) {
            continue;
          }
          String key = Strings.trimToNull(entry.getKey());
          String value = Strings.trimToNull(entry.getValue() == null ? null : String.valueOf(entry.getValue()));
          if (key != null && value != null) {
            payload.put(key, value);
          }
        }
      }
      payload.put("streamId", Strings.firstNonBlank(row.streamId(), "-"));
      payload.put("topic", Strings.firstNonBlank(row.topic(), "-"));
      payload.put("operation", Strings.firstNonBlank(row.operation(), "-"));
      return payload.isEmpty() ? Map.of() : Map.copyOf(payload);
    }

    private void updateLiveKpi(StitchStreamsProvider.TabularStreamRow row) {
      if (liveKpiValue == null || liveKpiSubtitle == null || row == null || row.values() == null) {
        return;
      }
      String valueField = Strings.firstNonBlank(widget.valueField(), "value");
      String labelField = Strings.firstNonBlank(widget.labelField(), "label");
      Object value = row.values().get(valueField);
      Object label = row.values().get(labelField);
      liveKpiValue.setText(Strings.firstNonBlank(normalizeDisplay(value), "0"));
      liveKpiSubtitle.setText(
          Strings.firstNonBlank(
              Strings.trimToNull(label == null ? null : String.valueOf(label)),
              Strings.trimToNull(widget.subtitle()),
              Strings.firstNonBlank(widget.title(), "Live KPI")));
    }

    private void renderLiveStreamError(String message) {
      String text =
          Strings.firstNonBlank(Strings.trimToNull(message), "Live stream widget failed to start.");
      if (eventLog != null) {
        eventLog.resultsView().setEmptyState("Live Stream Unavailable", text);
        eventLog.resultsView().setItems(List.of());
        eventLog.resultsView().setEmpty(true);
        eventLog.statusBar().setMessage(text);
      }
      if (liveTable != null) {
        liveTable.resultsView().setEmptyState("Live Stream Unavailable", text);
        liveTable.resultsView().setItems(List.of());
        liveTable.resultsView().setEmpty(true);
        liveTable.statusBar().setMessage(text);
      }
      if (liveKpiValue != null && liveKpiSubtitle != null) {
        liveKpiValue.setText("-");
        liveKpiSubtitle.setText(text);
      }
    }

    private void renderWidgetError(String message) {
      String text = Strings.firstNonBlank(Strings.trimToNull(message), "Widget failed to load data.");
      if (tableResults != null) {
        tableResults.setItems(List.of());
        tableResults.setEmptyState("Widget Error", text);
        tableResults.setEmpty(true);
        return;
      }
      if (tileLayout != null) {
        tileLayout.removeAll();
        Div error = new Div();
        error.setText(text);
        error.addClassName("stitch-dashboard-widget-empty");
        tileLayout.add(error);
        return;
      }
      if (chartLayout != null) {
        chartLayout.removeAll();
        Div error = new Div();
        error.setText(text);
        error.addClassName("stitch-dashboard-widget-empty");
        chartLayout.add(error);
        return;
      }
      if (actionLayout != null) {
        actionLayout.removeAll();
        Div error = new Div();
        error.setText(text);
        error.addClassName("stitch-dashboard-widget-empty");
        actionLayout.add(error);
        return;
      }
      if (liveStreamWidget) {
        renderLiveStreamError(text);
      }
    }

    private boolean isLiveStreamSource(StitchDashboardDescriptor.Source value) {
      if (value == null || value.type() != StitchDashboardDescriptor.SourceType.STREAM) {
        return false;
      }
      String mode =
          value.parameters() == null ? null : Strings.trimToNull(value.parameters().get("sourceMode"));
      if (mode == null) {
        return true;
      }
      if ("LIVE".equals(mode)) {
        return true;
      }
      if ("SNAPSHOT".equals(mode)) {
        return false;
      }
      throw new IllegalStateException(
          "Unsupported sourceMode '" + mode + "' for stream widget '" + widget.id() + "'");
    }

    private LiveStreamRenderMode resolveLiveStreamRenderMode(
        StitchDashboardDescriptor.Widget widgetValue, StitchDashboardDescriptor.Source sourceValue) {
      if (!isLiveStreamSource(sourceValue)) {
        return null;
      }
      if (widgetValue.type() == StitchDashboardDescriptor.WidgetType.KPI_TILES) {
        return LiveStreamRenderMode.KPI;
      }
      StitchDashboardDescriptor.StreamMode streamMode = widgetValue.streamMode();
      if (streamMode == StitchDashboardDescriptor.StreamMode.HYBRID) {
        throw new IllegalStateException(
            "Unsupported streamMode HYBRID for widget '" + widgetValue.id() + "'");
      }
      if (streamMode == StitchDashboardDescriptor.StreamMode.EVENT
          || widgetValue.type() == StitchDashboardDescriptor.WidgetType.STREAM_LOG) {
        return LiveStreamRenderMode.EVENT;
      }
      return LiveStreamRenderMode.TABLE;
    }

    private List<String> normalizeStringList(List<String> values) {
      if (values == null || values.isEmpty()) {
        return List.of();
      }
      LinkedHashSet<String> normalized = new LinkedHashSet<>();
      for (String value : values) {
        String item = Strings.trimToNull(value);
        if (item != null) {
          normalized.add(item);
        }
      }
      return normalized.isEmpty() ? List.of() : List.copyOf(normalized);
    }

    private void renderTable(List<Map<String, Object>> rows) {
      List<Map<String, Object>> safeRows = rows == null ? List.of() : rows;
      List<StitchDashboardDescriptor.ColumnSpec> nextColumns = normalizeColumnSpecs(widget, safeRows);
      if (!nextColumns.equals(visibleColumns)) {
        visibleColumns = nextColumns;
        tableResults.grid().removeAllColumns();
        for (StitchDashboardDescriptor.ColumnSpec column : visibleColumns) {
          if (column == null || Strings.trimToNull(column.field()) == null) {
            continue;
          }
          String field = column.field();
          var configuredColumn =
              tableResults
                  .grid()
                  .addColumn(item -> formatColumnValue(item == null ? null : item.get(field), column))
                  .setHeader(
                      Strings.firstNonBlank(
                          Strings.trimToNull(column.title()), field.toUpperCase(Locale.ROOT)))
                  .setAutoWidth(true)
                  .setFlexGrow(column.width() == null ? 1 : 0);
          if (column.width() != null) {
            configuredColumn.setWidth(column.width() + "px");
          }
        }
        addRowActionsColumn();
      }
      tableResults.setItems(safeRows);
      tableResults.setEmpty(safeRows.isEmpty());
    }

      if (liveStreamWidget) {
        Button start = new Button("START");
        start.addThemeVariants(ButtonVariant.LUMO_SMALL, ButtonVariant.LUMO_PRIMARY);
        start.addClickListener(event -> bindStreamWidget(true));
        Button stop = new Button("STOP");
        stop.addThemeVariants(ButtonVariant.LUMO_SMALL, ButtonVariant.LUMO_TERTIARY);
        stop.addClickListener(
            event -> {
              if (streamController != null) {
                streamController.stop();
              }
            });
        Button clear = new Button("CLEAR");
        clear.addThemeVariants(ButtonVariant.LUMO_SMALL, ButtonVariant.LUMO_TERTIARY_INLINE);
        clear.addClickListener(
            event -> {
              if (eventLog != null) {
                eventLog.clearEvents();
              }
              if (liveTable != null) {
                liveTable.applySnapshot(List.of());
                liveTable.clearHighlights();
              }
              if (liveKpiValue != null) {
                liveKpiValue.setText("0");
              }
            });
        section.addSectionAction(start);
        section.addSectionAction(stop);
        section.addSectionAction(clear);
      } else if (widget.type() != StitchDashboardDescriptor.WidgetType.ACTION_PANEL) {
        Button refresh = new Button("REFRESH");
        refresh.addThemeVariants(ButtonVariant.LUMO_SMALL, ButtonVariant.LUMO_TERTIARY_INLINE);
        refresh.addClickListener(event -> refresh(RefreshTrigger.MANUAL, true));
        section.addSectionAction(refresh);
      }

      for (StitchDashboardDescriptor.Action action : widget.actions()) {
        if (action == null || action.scope() != StitchDashboardDescriptor.ActionScope.WIDGET) {
          continue;
        }
        if (widget.type() == StitchDashboardDescriptor.WidgetType.ACTION_PANEL) {
          continue;
        }
        Button button = createActionButton(action);
        button.addClickListener(event -> runAction(action, Map.of(), button));
        section.addSectionAction(button);
      }
    }

    private void addRowActionsColumn() {
      List<StitchDashboardDescriptor.Action> rowActions = new ArrayList<>();
      for (StitchDashboardDescriptor.Action action : widget.actions()) {
        if (action != null && action.scope() == StitchDashboardDescriptor.ActionScope.ROW) {
          rowActions.add(action);
        }
      }
      if (rowActions.isEmpty()) {
        return;
      }
      StitchRowActions<Map<String, Object>> stitchRowActions = new StitchRowActions<>();
      for (StitchDashboardDescriptor.Action action : rowActions) {
        stitchRowActions.addAction(
            StitchRowActions.Action.<Map<String, Object>>icon(
                resolveIcon(action.icon()),
                action.label(),
                row -> runAction(action, payloadFromRow(widget, row), null)));
      }
      stitchRowActions.addColumn(tableResults.grid());
    }

    private void renderActionPanel() {
      actionLayout.removeAll();
      List<StitchDashboardDescriptor.Action> widgetActions = new ArrayList<>();
      for (StitchDashboardDescriptor.Action action : widget.actions()) {
        if (action != null && action.scope() == StitchDashboardDescriptor.ActionScope.WIDGET) {
          widgetActions.add(action);
        }
      }
      if (widgetActions.isEmpty()) {
        Div empty = new Div();
        empty.setText(widget.emptyMessage());
        empty.addClassName("stitch-dashboard-widget-empty");
        actionLayout.add(empty);
        return;
      }
      for (StitchDashboardDescriptor.Action action : widgetActions) {
        Button button = createActionButton(action);
        button.addClickListener(event -> runAction(action, Map.of(), button));
        actionLayout.add(button);
      }
    }

    private Button createActionButton(StitchDashboardDescriptor.Action action) {
      Button button = new Button(Strings.firstNonBlank(action.label(), "ACTION"));
      button.addThemeVariants(ButtonVariant.LUMO_SMALL);
      button.addClassName("stitch-dashboard-action-button");
      button.addClassName("stitch-dashboard-action-tone-" + action.tone().name().toLowerCase(Locale.ROOT));
      VaadinIcon icon = resolveIcon(action.icon());
      if (icon != null) {
        button.setIcon(icon.create());
      }
      button.getElement().setAttribute("aria-label", action.label());
      return button;
    }

    private void runAction(
        StitchDashboardDescriptor.Action action, Map<String, String> payload, Button sourceButton) {
      if (action == null) {
        return;
      }
      Runnable invoke = () -> invokeActionNow(action, payload, sourceButton);
      if (action.confirmationRequired()) {
        String message =
            Strings.firstNonBlank(
                action.confirmationMessage(),
                "Run action " + action.label() + "?");
        StitchDialog dialog = new StitchDialog("Confirm Action");
        dialog.addToBody(new Paragraph(message));
        dialog.addPrimaryAction(
            "RUN",
            () -> {
              dialog.close();
              invoke.run();
            });
        dialog.addSecondaryAction("CANCEL", dialog::close);
        dialog.open();
      } else {
        invoke.run();
      }
    }

    private void invokeActionNow(
        StitchDashboardDescriptor.Action action, Map<String, String> payload, Button sourceButton) {
      String previousText = null;
      if (sourceButton != null) {
        previousText = sourceButton.getText();
        sourceButton.setEnabled(false);
        sourceButton.setText("RUNNING...");
      }
      try {
        Map<String, String> linked = linkedFilters(widget.id());
        Map<String, String> parameters = resolveActionParameters(action, payload, linked);
        StitchDashboardActionRequest request =
            new StitchDashboardActionRequest(
                runtimeContext,
                dashboard,
                widget,
                action,
                payload,
                activeSharedFilters,
                linked,
                parameters);
        StitchDashboardActionResult result = actions.execute(request);
        if (result.success()) {
          String message = Strings.firstNonBlank(action.successMessage(), result.message(), "Action completed.");
          showActionNotification(message, true);
          if (result.refreshRequested()) {
            refreshAll(RefreshTrigger.MANUAL, true);
          }
        } else {
          String message = Strings.firstNonBlank(action.failureMessage(), result.message(), "Action failed.");
          showActionNotification(message, false);
        }
      } finally {
        if (sourceButton != null) {
          sourceButton.setEnabled(true);
          sourceButton.setText(previousText == null ? action.label() : previousText);
        }
      }
    }

    private void renderTiles(List<Map<String, Object>> rows) {
      tileLayout.removeAll();

        Div empty = new Div();

      tiles.setSpacing(false);
      tiles.setWidthFull();
      tiles.getStyle().set("gap", "0.55rem");
      tiles.getStyle().set("flex-wrap", "wrap");
      tiles.addClassName("stitch-dashboard-kpi-tiles");

      for (Map<String, Object> row : rows) {
        String label = resolveTileValue(row, widget.labelField(), "label", "name", "key");
        String value = resolveTileValue(row, widget.valueField(), "value", "count", "total");
        String tone = resolveTileValue(row, widget.toneField(), "tone", "status");

        Button tile = new Button();
        tile.addThemeVariants(ButtonVariant.LUMO_TERTIARY);
        tile.addClassName("stitch-dashboard-kpi-tile");
        if (Strings.trimToNull(tone) != null) {
          tile.addClassName("stitch-dashboard-kpi-tone-" + tone.trim().toLowerCase(Locale.ROOT));

        content.setSpacing(false);

        content.setAlignItems(Alignment.START);
        content.addClassName("stitch-dashboard-kpi-content");

        labelSpan.addClassName("stitch-dashboard-kpi-label");
        Span valueSpan = new Span(Strings.firstNonBlank(Strings.trimToNull(value), "0"));

        tile.getElement().setAttribute("aria-label", labelSpan.getText() + " " + valueSpan.getText());

              handleWidgetEvent(this, StitchDashboardDescriptor.EventType.ITEM_SELECTED, payload);

      }

    }

    private void renderBars(List<Map<String, Object>> rows) {
      chartLayout.removeAll();
      StitchDashboardDescriptor.ChartType chartType =
          widget.chartType() == null ? StitchDashboardDescriptor.ChartType.BAR : widget.chartType();
      if (chartType != StitchDashboardDescriptor.ChartType.BAR) {
        String message =
            "Unsupported chartType '" + chartType + "' for widget '" + widget.id() + "'.";
        LOGGER.warn("[StitchDashboard] {}", message);
        Div unsupported = new Div();
        unsupported.setText(message);
        unsupported.addClassName("stitch-dashboard-widget-empty");
        chartLayout.add(unsupported);
        return;
      }
      if (rows == null || rows.isEmpty()) {
        Div empty = new Div();
        empty.setText(widget.emptyMessage());
        empty.addClassName("stitch-dashboard-widget-empty");
        chartLayout.add(empty);
        return;
      }

      long maxValue = 0L;
      for (Map<String, Object> row : rows) {
        String value = resolveTileValue(row, widget.valueField(), "value", "count", "total");
        long numeric = parseLong(value);
        if (numeric > maxValue) {
          maxValue = numeric;
        }
      }
      long safeMax = maxValue <= 0 ? 1 : maxValue;

      for (Map<String, Object> row : rows) {
        String label = resolveTileValue(row, widget.labelField(), "label", "status", "key");
        String value = resolveTileValue(row, widget.valueField(), "value", "count", "total");
        String tone = resolveTileValue(row, widget.toneField(), "tone", "status");
        long numeric = parseLong(value);
        int widthPercent = (int) Math.max(4, Math.min(100, Math.round((numeric * 100.0) / safeMax)));

        Button bar = new Button();
        bar.addThemeVariants(ButtonVariant.LUMO_TERTIARY_INLINE);
        bar.addClassName("stitch-dashboard-chart-bar");
        if (Strings.trimToNull(tone) != null) {

    private String resolveTileValue(
        Map<String, Object> row, String preferredField, String... fallbacks) {
      if (row == null || row.isEmpty()) {
        return null;
      }
      String preferred = Strings.trimToNull(preferredField);
      if (preferred != null) {
        String value = Strings.trimToNull(String.valueOf(row.get(preferred)));
        if (value != null) {
          return value;
        }
      }
      if (fallbacks != null) {
        for (String fallback : fallbacks) {
          String key = Strings.trimToNull(fallback);
          if (key == null) {
            continue;
          }
          String value = Strings.trimToNull(String.valueOf(row.get(key)));
          if (value != null) {
            return value;
          }
        }
      }
      return null;
    }

    private static long parseLong(String value) {
      String normalized = Strings.trimToNull(value);
      if (normalized == null) {
        return 0L;
      }
      try {
        return Long.parseLong(normalized);
      } catch (NumberFormatException ignore) {
        return 0L;
      }
    }

  private static int intParameter(
      Map<String, String> parameters, int fallback, String... keys) {
    if (parameters == null || parameters.isEmpty() || keys == null || keys.length == 0) {
      return Math.max(1, fallback);
    }
    for (String key : keys) {
      String value = Strings.trimToNull(parameters.get(key));
      if (value == null) {
        continue;
      }
      try {
        return Math.max(1, Integer.parseInt(value));
      } catch (NumberFormatException ignore) {
        // continue
      }
    }
    return Math.max(1, fallback);
  }

  private static boolean boolParameter(
      Map<String, String> parameters, boolean fallback, String... keys) {
    if (parameters == null || parameters.isEmpty() || keys == null || keys.length == 0) {
      return fallback;
    }
    for (String key : keys) {
      String value = Strings.trimToNull(parameters.get(key));
      if (value == null) {
        continue;
      }
      if ("true".equalsIgnoreCase(value) || "1".equals(value)) {
        return true;
      }
      if ("false".equalsIgnoreCase(value) || "0".equals(value)) {
        return false;
      }
    }
    return fallback;
  }

  private List<Map<String, Object>> loadRowsForWidget(String widgetId) {
    if (dashboard == null) {

    }
    StitchDashboardDescriptor.Widget widget = dashboard.widget(widgetId);
    if (widget == null) {
      return List.of();
    }
    Map<String, String> linked = linkedFilters(widget.id());
    return dataSources.query(dashboard, widget, activeSharedFilters, linked);
  }

  private Map<String, String> linkedFilters(String widgetId) {
    LinkedHashMap<String, String> merged = new LinkedHashMap<>();
    if (dashboardContext != null && !dashboardContext.isEmpty()) {
      merged.putAll(dashboardContext);
    }
    if (widgetId != null) {
      Map<String, String> byWidget = linkContextByWidget.get(widgetId);
      if (byWidget != null && !byWidget.isEmpty()) {
        merged.putAll(byWidget);
      }
    }
    return merged.isEmpty() ? Map.of() : Map.copyOf(merged);
  }

  private DashboardRuntimeState resolveRuntimeState(StitchDashboardDescriptor descriptor) {
    String key = runtimeStateKey(descriptor);
    if (key == null) {
      return DashboardRuntimeState.empty();
    }
    return RUNTIME_STATE.getOrDefault(key, DashboardRuntimeState.empty());
  }

  private void persistRuntimeState() {
    if (dashboard == null) {
      return;
    }
    String key = runtimeStateKey(dashboard);
    if (key == null) {
      return;
    }
    DashboardRuntimeState snapshot =
        new DashboardRuntimeState(
            collectFilterValues(false),
            collectFilterValues(true),
            copyLinkContext(linkContextByWidget));
    runtimeState = snapshot;
    RUNTIME_STATE.put(key, snapshot);
  }

  private static String runtimeStateKey(StitchDashboardDescriptor descriptor) {
    if (descriptor == null) {
      return null;
    }
    String dashboardId = normalizeKey(descriptor.id());
    if (dashboardId == null) {
      return null;
    }
    if (descriptor.executionMode() == StitchDashboardDescriptor.DashboardExecutionMode.PER_USER) {
      String subject = normalizeKey(StitchIdentityContext.currentSubject());
      return dashboardId + "|user|" + Strings.firstNonBlank(subject, "anonymous");
    }
    return dashboardId + "|shared";
  }

  private static Map<String, Map<String, String>> copyLinkContext(
      Map<String, Map<String, String>> source) {
    if (source == null || source.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, Map<String, String>> copied = new LinkedHashMap<>();
    for (Map.Entry<String, Map<String, String>> entry : source.entrySet()) {
      if (entry == null) {
        continue;
      }
      String widgetId = Strings.trimToNull(entry.getKey());
      Map<String, String> context = normalizeMap(entry.getValue());
      if (widgetId != null && !context.isEmpty()) {
        copied.put(widgetId, context);
      }
    }
    return copied.isEmpty() ? Map.of() : Map.copyOf(copied);
  }

  private StitchDashboardDescriptor.RefreshPolicy resolvePolicy(
      StitchDashboardDescriptor.Widget widget) {
    StitchDashboardDescriptor.RefreshPolicy widgetPolicy = widget.refreshPolicy();
    if (widgetPolicy != null) {
      return widgetPolicy;
    }
    if (dashboard != null && dashboard.refreshPolicy() != null) {
      return dashboard.refreshPolicy();
    }
    return StitchDashboardDescriptor.RefreshPolicy.defaultPolicy();
  }

  private static boolean shouldRefresh(
      RefreshTrigger trigger, StitchDashboardDescriptor.RefreshPolicy policy) {
    if (policy == null || policy.mode() == null) {
      return trigger == RefreshTrigger.MANUAL || trigger == RefreshTrigger.FILTER_CHANGE;
    }
    return switch (policy.mode()) {
      case MANUAL -> trigger == RefreshTrigger.MANUAL;
      case ON_FILTER_CHANGE -> trigger == RefreshTrigger.MANUAL || trigger == RefreshTrigger.FILTER_CHANGE;
      case ON_LINK_EVENT ->
          trigger == RefreshTrigger.MANUAL
              || trigger == RefreshTrigger.LINK_EVENT
              || trigger == RefreshTrigger.FILTER_CHANGE;
      case INTERVAL ->
          trigger == RefreshTrigger.MANUAL
              || trigger == RefreshTrigger.INTERVAL
              || trigger == RefreshTrigger.FILTER_CHANGE
              || trigger == RefreshTrigger.LINK_EVENT;
      case LIVE ->
          trigger == RefreshTrigger.MANUAL
              || trigger == RefreshTrigger.LIVE
              || trigger == RefreshTrigger.FILTER_CHANGE
              || trigger == RefreshTrigger.LINK_EVENT;
    };
  }

  private static final class SelectFilterField implements DashboardFilterField {
    private final String key;
    private final Select<String> field = new Select<>();

        String key, String label, List<String> options, String defaultValue) {
      this.key = Objects.requireNonNull(key, "key");
      field.setWidth("12.5rem");
      field.addClassName("stitch-dashboard-filter");
      field.addClassName("stitch-dashboard-filter-select");
      field.setAriaLabel(label);

      List<String> entries = new ArrayList<>();
      entries.add(FILTER_ALL);
      if (options != null) {
        for (String option : options) {
          String normalized = Strings.trimToNull(option);
          if (normalized != null && !FILTER_ALL.equalsIgnoreCase(normalized)) {
            entries.add(normalized);
          }
        }
      }
      field.setItems(entries);
      field.setItemLabelGenerator(value -> FILTER_ALL.equals(value) ? "All" : value);
      String candidate = Strings.trimToNull(defaultValue);
      field.setValue(candidate == null ? FILTER_ALL : candidate);
      if (!entries.contains(field.getValue())) {
        field.setValue(FILTER_ALL);
      }
    }

    @Override
    public Component component() {
      return field;
    }

    @Override
    public Map<String, String> valueMap() {
      String value = Strings.trimToNull(field.getValue());
      if (value == null || FILTER_ALL.equalsIgnoreCase(value)) {
        return Map.of();
      }
      return Map.of(key, value);
    }

    @Override
    public Set<String> applyContext(Map<String, String> context) {
      if (context == null || context.isEmpty()) {
        return Set.of();
      }
      String value = context.get(key);
      if (Strings.trimToNull(value) == null) {
        return Set.of();
      }
      List<String> items = field.getListDataView().getItems().toList();
      for (String item : items) {
        if (item != null && item.equalsIgnoreCase(value.trim())) {
          field.setValue(item);
          return Set.of(normalizeKey(key));
        }
      }
      return Set.of();
    }

    @Override
    public void clear() {
      field.setValue(FILTER_ALL);
    }

    @Override
    public Registration onChange(Runnable callback) {
      return field.addValueChangeListener(event -> callback.run());
    }
  }

        values.put(keyFrom, from.getValue().toString());

      if (to.getValue() != null) {
        values.put(keyTo, to.getValue().toString());
      }
      return values.isEmpty() ? Map.of() : Map.copyOf(values);

    public Set<String> applyContext(Map<String, String> context) {

        return Set.of();

      if (fromValue != null) {
        from.setValue(fromValue);
        consumed.add(normalizeKey(keyFrom));
      }
      LocalDateTime toValue = parseDateTime(context.get(keyTo));
      if (toValue != null) {
        to.setValue(toValue);
        consumed.add(normalizeKey(keyTo));
      }

    }

    public void clear() {
      from.clear();
      to.clear();
    }

    @Override

      Registration left = from.addValueChangeListener(event -> callback.run());
      Registration right = to.addValueChangeListener(event -> callback.run());
      return () -> {

        right.remove();
      };

class NoNodePolicyGuardTest {
  private static final List<String> BANNED_ANNOTATIONS =
      List.of("@JsModule", "@JavaScript", "@NpmPackage", "@CssImport");
  private static final Path MODULE_ROOT = Path.of("").toAbsolutePath().normalize();

  @Test
  void noBannedFrontendAnnotationsInMainJava() throws IOException {
    Path sourceRoot = MODULE_ROOT.resolve("src/main/java");
    if (!Files.isDirectory(sourceRoot)) {
      return;
    }
    List<String> violations = new ArrayList<>();
    try (Stream<Path> files = Files.walk(sourceRoot)) {
      files
          .filter(Files::isRegularFile)
          .filter(path -> path.toString().endsWith(".java"))
          .forEach(
              path -> {
                String content = read(path);
                for (String banned : BANNED_ANNOTATIONS) {
                  if (content.contains(banned)) {
                    violations.add(relative(path) + " contains " + banned);
                  }
                }
              });
    }
    assertTrue(
        violations.isEmpty(),
        () -> "Banned frontend annotation usage detected:\n" + String.join("\n", violations));
  }

  @Test
  void noThemeComponentsDirectory() throws IOException {
    Path themeRoot = MODULE_ROOT.resolve("src/main/resources/META-INF/resources/frontend/themes");
    if (!Files.exists(themeRoot)) {
      return;
    }
    List<String> violations;
    try (Stream<Path> paths = Files.walk(themeRoot)) {
      violations =
          paths
              .filter(Files::isDirectory)
              .filter(path -> "components".equals(String.valueOf(path.getFileName())))
              .map(NoNodePolicyGuardTest::relative)
              .sorted()
              .toList();
    }
    assertTrue(
        violations.isEmpty(),
        () -> "Banned theme components directories detected:\n" + String.join("\n", violations));
  }

  private static String read(Path path) {
    try {
      return Files.readString(path);
    } catch (IOException error) {
      throw new UncheckedIOException("Unable to read " + path, error);
    }
  }

  private static String relative(Path path) {
    return MODULE_ROOT.relativize(path).toString().replace('\\', '/');
  }

  public static StitchDashboardCatalog catalog(StitchRuntimeContext runtimeContext) {
    Objects.requireNonNull(runtimeContext, "runtimeContext");
    StitchDashboardCatalog catalog = StitchDashboardDefinitionSupport.loadCatalog(runtimeContext, DEFAULT_GROUP);
    if (catalog.isEmpty()) {
      throw new IllegalStateException(
          "No DefinitionDashboard records were discovered for group '" + DEFAULT_GROUP + "'.");
    }
    if (catalog.dashboard(DASHBOARD_ORDERS) == null) {
      throw new IllegalStateException(
          "Required DefinitionDashboard '"
              + DASHBOARD_ORDERS
              + "' was not found in discovery group '"
              + DEFAULT_GROUP
              + "'.");
    }
    if (catalog.dashboard(DASHBOARD_POSITIONS) == null) {
      throw new IllegalStateException(
          "Required DefinitionDashboard '"
              + DASHBOARD_POSITIONS
              + "' was not found in discovery group '"
              + DEFAULT_GROUP
              + "'.");
    }
    return catalog;
  }

  public static StitchDashboardRegistry registry(StitchRuntimeContext runtimeContext) {
    StitchDashboardCatalog catalog = catalog(runtimeContext);
    return StitchDashboardRegistry.builder()
        .registerCatalog(catalog, StitchDashboardRegistry.SourceType.CONFIG)
        .build();
  }

  public static StitchDashboardDataSources dataSources(StitchRuntimeContext runtimeContext) {
    Objects.requireNonNull(runtimeContext, "runtimeContext");
    ensureDashboardDatasetsInstalled(runtimeContext);
    return StitchDashboardDataSources.builder()
        .registerResource(RESOURCE_ORDERS_KPIS, request -> loadOrdersKpis(runtimeContext, request))
        .registerResource(RESOURCE_ORDERS_STATUS, request -> loadOrdersStatusChart(runtimeContext, request))
        .registerResource(RESOURCE_ORDERS_RECENT, request -> loadRecentOrders(runtimeContext, request))
        .registerResource(RESOURCE_ORDERS_EVENTS, request -> loadRecentOrderEvents(runtimeContext, request))
        .registerResource(RESOURCE_POSITIONS_DESK, request -> loadDeskExposure(runtimeContext, request))
        .registerResource(RESOURCE_OPS_EXCEPTIONS, request -> loadOpsExceptions(runtimeContext, request))
        .register(
            StitchDashboardDescriptor.SourceType.DATASET_QUERY,
            request -> loadDefinitionSource(runtimeContext, request))
        .register(
            StitchDashboardDescriptor.SourceType.MATERIALIZED_VIEW,
            request -> loadDefinitionSource(runtimeContext, request))
        .register(
            StitchDashboardDescriptor.SourceType.STREAM,
            request -> loadDefinitionSource(runtimeContext, request))
        .register(
            StitchDashboardDescriptor.SourceType.SERVICE_QUERY,
            request -> loadDefinitionSource(runtimeContext, request))
        .registerDefaultIdentitySources()
        .build();
  }

        .build();
  }

  private static List<Map<String, Object>> loadDefinitionSource(
      StitchRuntimeContext runtimeContext, StitchDashboardDataRequest request) {
    if (request == null || request.source() == null) {
      return List.of();
    }
    Map<String, String> parameters = request.parameters();
    StitchDashboardDescriptor.SourceType sourceType = request.source().type();
    String query = Strings.firstNonBlank(parameters.get("query"), parameters.get("sql"));
    if (sourceType == StitchDashboardDescriptor.SourceType.SERVICE_QUERY
        && Strings.trimToNull(query) == null) {
      throw new IllegalStateException(
          "service source requires query for dashboard="
              + (request.dashboard() == null ? null : request.dashboard().id())
              + " widget="
              + (request.widget() == null ? null : request.widget().id()));
    }
    if (sourceType == StitchDashboardDescriptor.SourceType.STREAM) {
      throw new IllegalStateException(
          "stream source is live-only and cannot run through snapshot query handler for widget="
              + (request.widget() == null ? null : request.widget().id()));
    }
    if (sourceType == StitchDashboardDescriptor.SourceType.SERVICE_QUERY
        && hasSharedFilterBindings(request)) {
      throw new IllegalStateException(
          "service source filter binding is not supported in demo runtime for widget="
              + (request.widget() == null ? null : request.widget().id()));
    }
    int limit = resolveLimit(request);
    if (Strings.trimToNull(query) != null) {
      String adjustedQuery = applyQueryFiltersIfSupported(runtimeContext, request, query);
      return normalizeWidgetRows(request, querySql(runtimeContext, adjustedQuery, limit));
    }

    String resource =
        Strings.firstNonBlank(
            request.source().resourceId(),
            parameters.get("ref"));
    String table = resolveSourceTable(resource);
    if (table == null) {
      LOGGER.warn(
          "[StitchDemoDashboard] unresolved source table dashboard={} widget={} source={}",
          request.dashboard() == null ? null : request.dashboard().id(),
          request.widget() == null ? null : request.widget().id(),
          resource);
      return List.of();
    }

    Set<String> columns = resolveColumns(runtimeContext, table);
    String where = whereClause(effectiveFilters(request), columns);
    String orderBy = orderByClause(columns, "updated_at", "created_at", "order_id");
    String sql =
        "select * from "
            + table
            + where
            + (Strings.trimToNull(orderBy) == null ? "" : orderBy)
            + " limit "
            + Math.max(1, limit);
    return normalizeWidgetRows(request, querySql(runtimeContext, sql, limit));
  }

  private static List<Map<String, Object>> loadOrdersKpis(
      StitchRuntimeContext runtimeContext, StitchDashboardDataRequest request) {
    String table = resolveOrdersTable(runtimeContext);
    Set<String> columns = ordersColumns(runtimeContext);
    String statusField = firstExisting(columns, "status");
    String quantityField = firstExisting(columns, "quantity");
    String where = whereClause(effectiveFilters(request), columns);
    String sql =
        "select "
            + "count(*) as total_orders, "
            + conditionalStatusSum(statusField, "('OPEN','PARTIAL')", "open_orders")
            + ", "
            + conditionalStatusSum(statusField, "('FILLED')", "filled_orders")
            + ", "
            + conditionalStatusSum(statusField, "('REJECTED')", "rejected_orders")
            + ", "
            + conditionalStatusSum(statusField, "('CANCELLED')", "cancelled_orders")
            + ", "
            + conditionalQuantitySum(quantityField, "total_quantity")
            + "from "
            + table
            + where;

    List<Map<String, Object>> rows = querySql(runtimeContext, sql, 1);
    if (rows.isEmpty()) {
      return List.of();
    }
    Map<String, Object> row = rows.get(0);
    long totalOrders = longValue(row.get("total_orders"));
    long openOrders = longValue(row.get("open_orders"));
    long filledOrders = longValue(row.get("filled_orders"));
    long rejectedOrders = longValue(row.get("rejected_orders"));
    long cancelledOrders = longValue(row.get("cancelled_orders"));
    long totalQuantity = longValue(row.get("total_quantity"));

    return List.of(
        metric("TOTAL", "Total Orders", totalOrders, "neutral"),
        metric("OPEN", "Open Orders", openOrders, "warning"),
        metric("FILLED", "Filled Orders", filledOrders, "success"),
        metric("REJECTED", "Rejected Orders", rejectedOrders, "error"),
        metric("CANCELLED", "Cancelled Orders", cancelledOrders, "neutral"),
        metric("QUANTITY", "Total Quantity", totalQuantity, "neutral"));
  }

  private static List<Map<String, Object>> loadOrdersStatusChart(
      StitchRuntimeContext runtimeContext, StitchDashboardDataRequest request) {
    String table = resolveOrdersTable(runtimeContext);
    Set<String> columns = ordersColumns(runtimeContext);
    String statusField = firstExisting(columns, "status");
    if (statusField == null) {
      return List.of();
    }
    String where = whereClause(effectiveFilters(request), columns);
    String sql =
        "select upper(" + statusField + ") as status, count(*) as value "
            + "from "
            + table
            + where
            + " group by upper(" + statusField + ") "
            + " order by value desc, status asc";

    List<Map<String, Object>> rows = querySql(runtimeContext, sql, 16);
    if (rows.isEmpty()) {
      return List.of();
    }

    List<Map<String, Object>> chart = new ArrayList<>();
    for (Map<String, Object> row : rows) {
      if (row == null || row.isEmpty()) {
        continue;
      }
      String status = normalize(row.get("status"));
      if (status == null) {
        continue;
      }
      long value = longValue(row.get("value"));
      chart.add(
          Map.of(
              "status", status,
              "label", status,
              "value", value,
              "tone", toneForStatus(status)));
    }
    return chart.isEmpty() ? List.of() : List.copyOf(chart);
  }

  private static List<Map<String, Object>> loadDeskExposure(
      StitchRuntimeContext runtimeContext, StitchDashboardDataRequest request) {
    String table = resolveOrdersTable(runtimeContext);
    Set<String> columns = ordersColumns(runtimeContext);
    String deskField = firstExisting(columns, "desk");
    if (deskField == null) {
      return List.of();
    }
    String where = whereClause(effectiveFilters(request), columns);
    String quantityField = firstExisting(columns, "quantity");
    String sql =
        "select "
            + deskField
            + " as desk, "
            + "count(*) as value, "
            + "coalesce(sum("
            + (quantityField == null ? "0" : quantityField)
            + "),0) as quantity "
            + "from "
            + table
            + where
            + " group by "
            + deskField
            + " order by value desc, desk asc";
    List<Map<String, Object>> rows = querySql(runtimeContext, sql, 40);
    if (rows.isEmpty()) {
      return List.of();
    }
    List<Map<String, Object>> projected = new ArrayList<>();
    for (Map<String, Object> row : rows) {
      if (row == null || row.isEmpty()) {
        continue;
      }
      String desk = normalize(row.get("desk"));
      if (desk == null) {
        continue;
      }
      projected.add(
          Map.of(
              "desk", desk,
              "label", desk,
              "value", longValue(row.get("value")),
              "quantity", longValue(row.get("quantity")),
              "tone", "neutral"));
    }
    return projected.isEmpty() ? List.of() : List.copyOf(projected);
  }

                + " else 'UPDATE' end as event_type";

  private static StitchDashboardActionResult onOrderStatusAction(
      String status, StitchDashboardActionRequest request) {
    String nextStatus = Strings.requireNonBlank(status, "status");
    String orderId =
        Strings.firstNonBlank(
            request == null ? null : request.parameters().get("order_id"),
            request == null ? null : request.parameters().get("orderId"),
            request == null ? null : request.payload().get("order_id"),
            request == null ? null : request.payload().get("orderId"));
    if (Strings.trimToNull(orderId) == null) {
      return StitchDashboardActionResult.failure("Missing order_id for action.");
    }
    ORDER_STATUS_OVERRIDES.put(orderId, nextStatus);
    return StitchDashboardActionResult.success("Order " + orderId + " -> " + nextStatus, true);
  }

  private static List<Map<String, Object>> applyStatusOverrides(List<Map<String, Object>> rows) {
    if (rows == null || rows.isEmpty() || ORDER_STATUS_OVERRIDES.isEmpty()) {
      return rows == null ? List.of() : rows;
    }
    List<Map<String, Object>> updated = new ArrayList<>();
    boolean changed = false;
    for (Map<String, Object> row : rows) {
      if (row == null || row.isEmpty()) {
        continue;
      }
      String orderId =
          firstNonBlank(normalize(row.get("order_id")), normalize(row.get("orderId")));
      String override = orderId == null ? null : ORDER_STATUS_OVERRIDES.get(orderId);
      if (override == null) {
        updated.add(row);
        continue;
      }
      LinkedHashMap<String, Object> mutable = new LinkedHashMap<>(row);
      mutable.put("status", override);
      if (mutable.containsKey("event_type")) {
        mutable.put("event_type", "ACTION_UPDATE");
      }
      updated.add(Map.copyOf(mutable));
      changed = true;
    }
    return changed ? List.copyOf(updated) : rows;
  }

  }

  private static String resolveOrdersTable(StitchRuntimeContext runtimeContext) {
    ensureDashboardDatasetsInstalled(runtimeContext);
    String cached = Strings.trimToNull(cachedOrdersTable);
    if (cached != null) {
      Set<String> columns = resolveColumns(runtimeContext, cached);
      if (!columns.isEmpty()) {
        cachedOrdersColumns = columns;
        return cached;
      }
    }

    String fallbackTable = null;
    Set<String> fallbackColumns = Set.of();
    for (String candidate : ORDERS_DATASET_CANDIDATES) {
      String table = toTable(candidate);
      Set<String> columns = resolveColumns(runtimeContext, table);
      if (columns.isEmpty()) {
        continue;
      }
      if (supportsOrdersDashboard(columns)) {
        cachedOrdersTable = table;
        cachedOrdersColumns = columns;
        LOGGER.info("[StitchDemoDashboard] using orders dataset table={} columns={}", table, columns.size());
        return table;
      }
      if (fallbackTable == null) {
        fallbackTable = table;
        fallbackColumns = columns;
      }
    }

    if (fallbackTable != null) {
      cachedOrdersTable = fallbackTable;
      cachedOrdersColumns = fallbackColumns;
      LOGGER.warn(
          "[StitchDemoDashboard] using fallback orders table={} columns={}",
          fallbackTable,
          fallbackColumns.size());
      return fallbackTable;
    }

    String defaultFallback = quote(DEFAULT_GROUP) + "." + quote(DEFAULT_NAMESPACE) + "." + quote("orders");
    cachedOrdersTable = defaultFallback;
    cachedOrdersColumns = Set.of();
    LOGGER.warn("[StitchDemoDashboard] orders dataset not found; defaulting to table={}", defaultFallback);
    return defaultFallback;
  }

  private static Set<String> ordersColumns(StitchRuntimeContext runtimeContext) {
    resolveOrdersTable(runtimeContext);
    Set<String> columns = cachedOrdersColumns;
    return columns == null ? Set.of() : columns;
  }

  private static Set<String> resolveColumns(StitchRuntimeContext runtimeContext, String table) {
    ensureDashboardDatasetsInstalled(runtimeContext);
    String queryTable = Strings.trimToNull(table);
    if (queryTable == null) {
      return Set.of();
    }
    try {
      RowResult result = runtimeContext.session().data().direct().rows().querySql("select * from " + queryTable + " limit 1", 1);
      return normalizeColumns(result == null ? null : result.columns());
    } catch (Exception ignore) {
      return Set.of();
    }
  }

  private static void ensureDashboardDatasetsInstalled(StitchRuntimeContext runtimeContext) {
    if (dashboardDatasetsInstalled || runtimeContext == null || runtimeContext.session() == null) {
      return;
    }
    synchronized (DATASET_INSTALL_LOCK) {
      if (dashboardDatasetsInstalled || runtimeContext.session() == null) {
        return;
      }
      TabularManager tabularManager = runtimeContext.session().data().direct().tabular();
      if (!(tabularManager instanceof TabularManagerImpl directTabularManager)) {
        throw new IllegalStateException(
            "Unsupported tabular manager for dashboard dataset install: "
                + (tabularManager == null ? "null" : tabularManager.getClass().getName()));
      }
      LinkedHashSet<String> datasetIds = new LinkedHashSet<>(DASHBOARD_DATASET_ALLOWLIST);
      int installedCount = 0;
      for (String datasetId : datasetIds) {
        try {
          directTabularManager.installDataset(datasetId);
          installedCount++;
        } catch (Exception error) {
          throw new IllegalStateException(
              "Failed to install required dashboard dataset: " + datasetId,
              error);
        }
      }
      dashboardDatasetsInstalled = true;
      LOGGER.info(
          "[StitchDemoDashboard] installed dashboard datasets for cross-group access installed={} requested={}",
          installedCount,
          datasetIds.size());
    }
  }

  private static Set<String> normalizeColumns(List<String> columns) {
    if (columns == null || columns.isEmpty()) {
      return Set.of();
    }
    LinkedHashSet<String> normalized = new LinkedHashSet<>();
    for (String column : columns) {
      String value = normalize(column);
      if (value != null) {
        normalized.add(value.toLowerCase(Locale.ROOT));
      }
    }
    return normalized.isEmpty() ? Set.of() : Set.copyOf(normalized);
  }

  private static boolean supportsOrdersDashboard(Set<String> columns) {
    if (columns == null || columns.isEmpty()) {
      return false;
    }
    return hasColumn(columns, "order_id")
        && hasColumn(columns, "status")
        && (hasColumn(columns, "updated_at") || hasColumn(columns, "created_at"));
  }

  private static boolean hasColumn(Set<String> columns, String column) {
    if (columns == null || columns.isEmpty()) {
      return false;
    }
    String key = normalize(column);
    return key != null && columns.contains(key.toLowerCase(Locale.ROOT));
  }

  private static String firstExisting(Set<String> columns, String... names) {
    if (columns == null || columns.isEmpty() || names == null || names.length == 0) {
      return null;
    }
    for (String name : names) {
      String key = normalize(name);
      if (key == null) {
        continue;
      }
      String normalized = key.toLowerCase(Locale.ROOT);
      if (columns.contains(normalized)) {
        return normalized;
      }
    }
    return null;
  }

  private static String selectExpr(Set<String> columns, String column) {
    String key = normalize(column);
    if (key == null) {
      return "null";
    }
    return hasColumn(columns, key) ? key : "null as " + key;
  }

  private static String orderByClause(Set<String> columns, String... preferredColumns) {
    String orderColumn = firstExisting(columns, preferredColumns);
    return orderColumn == null ? "" : " order by " + orderColumn + " desc ";
  }

  private static String conditionalStatusSum(String statusField, String statuses, String alias) {
    String name = Strings.requireNonBlank(alias, "alias");
    String statusColumn = normalize(statusField);
    if (statusColumn == null) {
      return "0 as " + name;
    }
    String valueSet = Strings.requireNonBlank(statuses, "statuses");
    return "sum(case when upper(" + statusColumn + ") in " + valueSet + " then 1 else 0 end) as " + name;
  }

  private static String conditionalQuantitySum(String quantityField, String alias) {
    String name = Strings.requireNonBlank(alias, "alias");
    String quantityColumn = normalize(quantityField);
    if (quantityColumn == null) {
      return "0 as " + name;
    }
    return "coalesce(sum(" + quantityColumn + "), 0) as " + name;
  }

  private static List<Map<String, Object>> querySql(
      StitchRuntimeContext runtimeContext, String sql, int limit) {
    try {
      RowResult result = runtimeContext.session().data().direct().rows().querySql(sql, Math.max(1, limit));
      return toRows(result);
    } catch (Exception error) {
      LOGGER.warn("[StitchDemoDashboard] SQL query failed: {}", sql, error);
      return List.of();
    }
  }

  private static int resolveLimit(StitchDashboardDataRequest request) {
    int configured = parsePositiveInt(request == null ? null : request.parameters().get("limit"));
    if (configured > 0) {
      return configured;
    }
    if (request == null || request.widget() == null) {
      return 200;
    }
    StitchDashboardDescriptor.WidgetType type = request.widget().type();
    if (type == StitchDashboardDescriptor.WidgetType.KPI_TILES) {
      return 32;
    }
    if (type == StitchDashboardDescriptor.WidgetType.BAR_CHART) {
      return 128;
    }
    if (type == StitchDashboardDescriptor.WidgetType.STREAM_LOG) {
      return 500;
    }
    return 200;
  }

  private static int parsePositiveInt(String value) {
    String text = Strings.trimToNull(value);
    if (text == null) {
      return -1;
    }
    try {
      int parsed = Integer.parseInt(text);
      return parsed > 0 ? parsed : -1;
    } catch (NumberFormatException ignore) {
      return -1;
    }
  }

  private static String resolveSourceTable(String resourceIdentifier) {
    String value = Strings.trimToNull(resourceIdentifier);
    if (value == null) {
      return null;
    }
    if (value.startsWith("dataset:")) {
      try {
        return toTable(value);
      } catch (Exception error) {
        LOGGER.warn("[StitchDemoDashboard] unable to parse dataset source: {}", value, error);
        return null;
      }
    }
    if (value.startsWith("stream:")) {
      String withoutPrefix = value.substring("stream:".length());
      String[] parts = withoutPrefix.split("/");
      if (parts.length == 3) {
        return quote(parts[0]) + "." + quote(parts[1]) + "." + quote(parts[2]);
      }
      return null;
    }
    if (value.contains("/")) {
      try {
        return toTable(value);
      } catch (Exception error) {
        LOGGER.warn("[StitchDemoDashboard] unable to parse source identifier: {}", value, error);
        return null;
      }
    }
    return quote(DEFAULT_GROUP) + "." + quote(DEFAULT_NAMESPACE) + "." + quote(value);
  }

  private static List<Map<String, Object>> normalizeWidgetRows(
      StitchDashboardDataRequest request, List<Map<String, Object>> rows) {
    if (rows == null || rows.isEmpty()) {
      return List.of();
    }
    if (request == null || request.widget() == null) {
      return rows;
    }
    if (request.widget().type() == StitchDashboardDescriptor.WidgetType.KPI_TILES
        && rows.size() == 1) {
      Map<String, Object> row = rows.get(0);
      if (row == null || row.isEmpty()) {
        return rows;
      }
      String labelField = Strings.trimToNull(request.widget().labelField());
      if (labelField != null && row.containsKey(labelField)) {
        return rows;
      }
      LinkedHashMap<String, Object> normalized = new LinkedHashMap<>(row);
      normalized.put("label", request.widget().title());
      return List.of(Map.copyOf(normalized));
    }
    return rows;
  }

  private static List<Map<String, Object>> toRows(RowResult result) {
    if (result == null || result.columns() == null || result.columns().isEmpty() || result.rows().isEmpty()) {
      return List.of();
    }
    List<Map<String, Object>> rows = new ArrayList<>();
    List<String> columns = result.columns();
    for (List<Object> values : result.rows()) {
      if (values == null || values.isEmpty()) {
        continue;
      }
      LinkedHashMap<String, Object> row = new LinkedHashMap<>();
      int size = Math.min(columns.size(), values.size());
      for (int i = 0; i < size; i++) {
        String column = Strings.trimToNull(columns.get(i));
        if (column == null) {
          continue;
        }
        Object value = values.get(i);
        if (value != null) {
          row.put(column, value);
        }
      }
      if (!row.isEmpty()) {
        rows.add(Map.copyOf(row));
      }
    }
    return rows.isEmpty() ? List.of() : List.copyOf(rows);
  }

  private static String whereClause(Map<String, String> parameters, Set<String> columns) {
    if (parameters == null || parameters.isEmpty()) {
      return "";
    }
    List<String> clauses = new ArrayList<>();

    String group = normalize(parameters.get("group"));
    if (group != null && !"all".equalsIgnoreCase(group) && !DEFAULT_GROUP.equalsIgnoreCase(group)) {
      clauses.add("1 = 0");
    }

    addExact(clauses, columns, "desk", normalize(parameters.get("desk")));
    addContains(clauses, columns, "account_id", normalize(parameters.get("account_id")));
    addContains(clauses, columns, "symbol", normalize(parameters.get("symbol")));
    addExact(
        clauses,
        columns,
        firstNonBlank(firstExisting(columns, "trading_region"), firstExisting(columns, "region")),
        normalize(parameters.get("region")));
    addExact(clauses, columns, "status", normalize(parameters.get("status")));
    addExact(clauses, columns, "order_type", normalize(parameters.get("order_type")));
    addQuantityMin(clauses, columns, normalize(parameters.get("min_quantity")));
    if ("true".equalsIgnoreCase(normalize(parameters.get("exceptions_only")))) {
      addStatusSet(clauses, columns, "REJECTED|CANCELLED");
    }

    String orderId = firstNonBlank(normalize(parameters.get("orderId")), normalize(parameters.get("order_id")));
    addExact(clauses, columns, "order_id", orderId);

    String createdDate = normalize(parameters.get("created_date"));
    if (createdDate != null) {
      String dateField = firstNonBlank(firstExisting(columns, "created_date"), firstExisting(columns, "date"));
      addDateEquals(clauses, dateField, createdDate);
    }
    String asOfDate = normalize(parameters.get("as_of_date"));
    if (asOfDate != null) {
      String dateField = firstNonBlank(firstExisting(columns, "as_of_date"), firstExisting(columns, "date"));
      addDateEquals(clauses, dateField, asOfDate);
    }

    String createdFrom = normalize(parameters.get("created_at_from"));
    String createdField =
        firstNonBlank(
            firstExisting(columns, "created_at"),
            firstExisting(columns, "updated_at"),
            firstExisting(columns, "date"));
    if (createdFrom != null) {
      addTimeRangeLower(clauses, createdField, createdFrom);
    }
    String createdTo = normalize(parameters.get("created_at_to"));
    if (createdTo != null) {
      addTimeRangeUpper(clauses, createdField, createdTo);
    }

    if (clauses.isEmpty()) {
      return "";
    }
    return " where " + String.join(" and ", clauses);
  }

  private static Map<String, String> effectiveFilters(StitchDashboardDataRequest request) {
    if (request == null || request.parameters() == null || request.parameters().isEmpty()) {
      return Map.of();
    }
    return request.parameters();
  }

  private static void addExact(List<String> clauses, Set<String> columns, String field, String value) {
    if (clauses == null || Strings.trimToNull(field) == null || Strings.trimToNull(value) == null) {
      return;
    }
    String key = normalize(field);
    if (key == null || !hasColumn(columns, key)) {
      return;
    }
    if (value.contains("|")) {
      List<String> values = new ArrayList<>();
      for (String part : value.split("\\|")) {
        String normalized = normalize(part);
        if (normalized != null) {
          values.add("'" + escape(normalized).toUpperCase(Locale.ROOT) + "'");
        }
      }
      if (!values.isEmpty()) {
        clauses.add("upper(" + key + ") in (" + String.join(", ", values) + ")");
      }
      return;
    }
    clauses.add("upper(" + key + ") = upper('" + escape(value) + "')");
  }

  private static void addContains(List<String> clauses, Set<String> columns, String field, String value) {
    if (clauses == null || Strings.trimToNull(field) == null || Strings.trimToNull(value) == null) {
      return;
    }
    String key = normalize(field);
    if (key == null || !hasColumn(columns, key)) {
      return;
    }
    clauses.add(key + " ilike '%" + escape(value) + "%'");
  }

  private static void addStatusSet(List<String> clauses, Set<String> columns, String value) {
    addExact(clauses, columns, "status", value);
  }

  private static void addQuantityMin(List<String> clauses, Set<String> columns, String value) {
    String quantityField = firstExisting(columns, "quantity");
    String boundary = normalize(value);
    if (clauses == null || quantityField == null || boundary == null) {
      return;
    }
    try {
      Double.parseDouble(boundary);
      clauses.add("try_cast(" + quantityField + " as double) >= " + boundary);
    } catch (NumberFormatException ignore) {
      // ignore invalid dashboard control value
    }
  }

  private static void addTimeRangeLower(List<String> clauses, String field, String value) {
    String key = normalize(field);
    String boundary = normalize(value);
    if (clauses == null || key == null || boundary == null) {
      return;
    }
    clauses.add("try_cast(" + key + " as timestamp) >= try_cast('" + escape(boundary) + "' as timestamp)");
  }

  private static void addTimeRangeUpper(List<String> clauses, String field, String value) {
    String key = normalize(field);
    String boundary = normalize(value);
    if (clauses == null || key == null || boundary == null) {
      return;
    }
    clauses.add("try_cast(" + key + " as timestamp) <= try_cast('" + escape(boundary) + "' as timestamp)");
  }

  private static void addDateEquals(List<String> clauses, String field, String value) {
    String key = normalize(field);
    String boundary = normalize(value);
    if (clauses == null || key == null || boundary == null) {
      return;
    }
    clauses.add("try_cast(" + key + " as date) = try_cast('" + escape(boundary) + "' as date)");
  }

  private static boolean hasSharedFilterBindings(StitchDashboardDataRequest request) {
    if (request == null || request.widget() == null || request.widget().filterBindings() == null) {
      return false;
    }
    if (request.parameters() == null || request.parameters().isEmpty()) {
      return false;
    }
    for (Map.Entry<String, String> binding : request.widget().filterBindings().entrySet()) {
      if (binding == null) {
        continue;
      }
      String parameter = Strings.trimToNull(binding.getValue());
      if (parameter != null && request.parameters().containsKey(parameter)) {
        return true;
      }
    }
    return false;
  }

  private static String applyQueryFiltersIfSupported(
      StitchRuntimeContext runtimeContext, StitchDashboardDataRequest request, String query) {
    String normalizedQuery = Strings.requireNonBlank(query, "query").trim();
    if (normalizedQuery.endsWith(";")) {
      normalizedQuery = normalizedQuery.substring(0, normalizedQuery.length() - 1).trim();
    }
    Map<String, String> filters = effectiveFilters(request);
    if (filters.isEmpty()) {
      return normalizedQuery;
    }
    String resource =
        Strings.firstNonBlank(
            request == null || request.source() == null ? null : request.source().resourceId(),
            request == null || request.parameters() == null ? null : request.parameters().get("ref"));
    String table = resolveSourceTable(resource);
    if (Strings.trimToNull(table) == null) {
      LOGGER.warn(
          "[StitchDemoDashboard] query source filter injection skipped (unresolved table) dashboard={} widget={}",
          request == null || request.dashboard() == null ? null : request.dashboard().id(),
          request == null || request.widget() == null ? null : request.widget().id());
      return normalizedQuery;
    }
    String where = whereClause(filters, resolveColumns(runtimeContext, table));
    if (Strings.trimToNull(where) == null) {
      return normalizedQuery;
    }
    String lower = normalizedQuery.toLowerCase(Locale.ROOT);
    if (lower.contains(" where ")) {
      LOGGER.warn(
          "[StitchDemoDashboard] query source already has WHERE; dashboard filters were not injected dashboard={} widget={}",
          request == null || request.dashboard() == null ? null : request.dashboard().id(),
          request == null || request.widget() == null ? null : request.widget().id());
      return normalizedQuery;
    }
    int insert = clauseIndex(lower, " group by ", " order by ", " limit ");
    if (insert < 0) {
      return normalizedQuery + where;
    }
    return normalizedQuery.substring(0, insert) + where + normalizedQuery.substring(insert);
  }

  private static int clauseIndex(String text, String... clauses) {
    if (Strings.trimToNull(text) == null || clauses == null || clauses.length == 0) {
      return -1;
    }
    int index = -1;
    for (String clause : clauses) {
      if (Strings.trimToNull(clause) == null) {
        continue;
      }
      int candidate = text.indexOf(clause);
      if (candidate >= 0 && (index < 0 || candidate < index)) {
        index = candidate;
      }
    }
    return index;
  }

  private static String toTable(String datasetIdentifier) {
    String value = Strings.requireNonBlank(datasetIdentifier, "datasetIdentifier");
    if (value.contains(":")) {
      DatasetId datasetId = DatasetId.parse(value);
      return quote(datasetId.group()) + "." + quote(datasetId.namespace()) + "." + quote(datasetId.name());
    }
    if (value.contains("/")) {
      String[] parts = value.split("/");
      if (parts.length == 3) {
        return quote(parts[0]) + "." + quote(parts[1]) + "." + quote(parts[2]);
      }
    }
    return quote(DEFAULT_GROUP) + "." + quote(DEFAULT_NAMESPACE) + "." + quote(value);
  }

  private static String quote(String value) {
    String identifier = Strings.requireNonBlank(value, "identifier");
    return "\"" + identifier.replace("\"", "\"\"") + "\"";
  }

  private static String escape(String value) {
    return Strings.requireNonBlank(value, "value").replace("'", "''");
  }

  private static long longValue(Object value) {
    if (value instanceof Number number) {
      return number.longValue();
    }
    String text = normalize(value);
    if (text == null) {
      return 0L;
    }
    try {
      return Long.parseLong(text);
    } catch (NumberFormatException ignore) {
      return 0L;
    }
  }

  private static String toneForStatus(String status) {
    String normalized = normalize(status);
    if (normalized == null) {
      return "neutral";
    }
    return switch (normalized.toUpperCase(Locale.ROOT)) {
      case "FILLED" -> "success";
      case "OPEN", "PARTIAL" -> "warning";
      case "REJECTED", "FAILED" -> "error";
      case "CANCELLED" -> "neutral";
      default -> "neutral";
    };
  }

  private static String normalize(Object value) {
    return Strings.trimToNull(value == null ? null : String.valueOf(value));
  }

  private static String firstNonBlank(String... values) {
    if (values == null || values.length == 0) {
      return null;
    }
    for (String value : values) {
      String normalized = Strings.trimToNull(value);
      if (normalized != null) {
        return normalized;
      }
    }
    return null;
  }
