  SchemaManagerImpl(

      Supplier<DeleteManager> deleteManagerSupplier,
      SampleSourceFactory sampleSourceFactory,

  @Override
  public SchemaSampleResult sample(String ingestName, SchemaSampleOptions options) {
    String normalizedIngestName = Strings.requireNonBlank(ingestName, "ingestName");
    SchemaSampleOptions resolvedOptions = options == null ? SchemaSampleOptions.defaults() : options;
    DiscoveryCache DiscoveryCache = requireConfigCache();
    String group = requireSessionGroup();
    DiscoveryCache
        .get(DefinitionIngest.class, group, normalizedIngestName)
        .orElseThrow(
            () ->
                new IllegalArgumentException(
                    "Unknown ingest: " + group + "/" + normalizedIngestName));
    DefinitionIngestSource source =
        DiscoveryCache
            .get(DefinitionIngestSource.class, group, normalizedIngestName)
            .orElseThrow(
                () ->
                    new IllegalStateException(
                        "IngestSource not found for ingest: " + group + "/" + normalizedIngestName));
    String resolvedFormat = resolveFormat(source, resolvedOptions);
    String ingestId = group + "/" + normalizedIngestName;

    List<PayloadSample> samples;
    try (SampleSource sampleSource =
        sampleSourceFactory.create(ingestId, source, resolvedOptions)) {
      samples = collect(sampleSource, resolvedOptions.maxMessages(), resolvedOptions.maxBytes());
    } catch (RuntimeException ex) {
      throw ex;
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to close sample source ingest=" + ingestId, ex);
    }
    if (samples.isEmpty()) {
      throw new IllegalStateException("No payload samples were collected ingest=" + ingestId);
    }

    return inferFromSamples(
        normalizedIngestName, ingestId, resolvedFormat, source, samples, resolvedOptions.allowNulls());
  }

  @Override
  public SchemaSampleResult samplePayloads(
      IngestFormat format, Iterable<String> payloads, SchemaSampleOptions options) {
    IngestFormat resolvedFormat = Objects.requireNonNull(format, "format");
    Objects.requireNonNull(payloads, "payloads");
    SchemaSampleOptions resolvedOptions = options == null ? SchemaSampleOptions.defaults() : options;
    String formatToken = normalizeFormat(resolvedFormat.name());
    return samplePayloads(
        PAYLOADS_INGEST_NAME,
        new DefinitionIngestSource(null, null, PAYLOADS_INGEST_NAME, formatToken, null, null, null),
        payloads,
        resolvedOptions);
  }

  public SchemaSampleResult samplePayloadsConfigured(
      String ingestName,
      IngestFormat format,
      String jsonMode,
      String jsonRootField,
      String nvFixPairDelimiter,
      String nvFixKvDelimiter,
      Boolean nvFixTrimKeys,
      Boolean nvFixTrimValues,
      Iterable<String> payloads,
      SchemaSampleOptions options) {
    String normalizedIngestName = Strings.requireNonBlank(ingestName, "ingestName");
    IngestFormat resolvedFormat = Objects.requireNonNull(format, "format");
    String formatToken = normalizeFormat(resolvedFormat.name());
    DefinitionIngestSource.JsonConfig json =
        resolvedFormat == IngestFormat.JSON
            ? new DefinitionIngestSource.JsonConfig(jsonMode, jsonRootField)
            : null;
    DefinitionIngestSource.NvFixConfig nvfix =
        resolvedFormat == IngestFormat.NVFIX
            ? new DefinitionIngestSource.NvFixConfig(
                nvFixPairDelimiter,
                nvFixKvDelimiter,
                nvFixTrimKeys,
                nvFixTrimValues)
            : null;
    return samplePayloads(
        normalizedIngestName,
        new DefinitionIngestSource(
            null,
            null,
            normalizedIngestName,
            formatToken,
            json,
            nvfix,
            null),
        payloads,
        options);
  }

  SchemaSampleResult samplePayloads(
      String ingestName,
      DefinitionIngestSource source,
      Iterable<String> payloads,
      SchemaSampleOptions options) {
    String resolvedIngestName = Strings.requireNonBlank(ingestName, "ingestName");
    Objects.requireNonNull(payloads, "payloads");
    SchemaSampleOptions resolvedOptions = options == null ? SchemaSampleOptions.defaults() : options;
    String formatToken = resolveFormat(source, resolvedOptions);
    List<PayloadSample> samples = collect(payloads, resolvedOptions.maxMessages(), resolvedOptions.maxBytes());
    if (samples.isEmpty()) {
      throw new IllegalStateException("No payload samples were provided");
    }
    String ingestId = "inline/" + resolvedIngestName;
    return inferFromSamples(
        resolvedIngestName,
        ingestId,
        formatToken,
        source,
        samples,
        resolvedOptions.allowNulls());
  }

  @Override
  public Optional<SchemaModel> get(String group, String schemaName) {
    String normalizedGroup = Strings.requireNonBlank(group, "group");
    String normalizedName = Strings.requireNonBlank(schemaName, "schemaName");
    DiscoveryCache cache = requireConfigCache();
    return cache
        .get(DefinitionSchema.class, normalizedGroup, normalizedName)
        .map(SchemaModelMapper::fromDefinition);
  }

  @Override
  public List<SchemaModel> listVersions(String schemaName) {
    String group = requireSessionGroup();
    return listVersions(group, schemaName);
  }

  @Override
  public Optional<SchemaModel> getVersion(String schemaName, int versionNum) {
    if (versionNum <= 0) {
      throw new IllegalArgumentException("versionNum must be > 0");
    }
    for (SchemaModel version : listVersions(schemaName)) {
      Integer candidate = version.getVersionNum();
      if (candidate != null && candidate == versionNum) {
        return Optional.of(version);
      }
    }
    return Optional.empty();
  }

  @Override
  public Optional<SchemaModel> getVersion(String schemaName, String versionToken) {
    String normalizedToken = Strings.trimToNull(versionToken);
    if (normalizedToken == null) {
      return Optional.empty();
    }
    Integer parsedVersionNum = parseVersionToken(normalizedToken);
    if (parsedVersionNum != null) {
      return getVersion(schemaName, parsedVersionNum);
    }
    for (SchemaModel version : listVersions(schemaName)) {
      String name = Strings.trimToNull(version.getName());
      if (normalizedToken.equals(name)) {
        return Optional.of(version);
      }
      String versionName = Strings.trimToNull(version.getVersionName());
      if (versionName != null && normalizedToken.equalsIgnoreCase(versionName)) {
        return Optional.of(version);
      }
    }
    return Optional.empty();
  }

  @Override
  public List<SchemaModel> list() {
    DiscoveryCache cache = requireConfigCache();
    String group = requireSessionGroup();
    return sortSchemas(filterCanonicalSchemas(mapSchemas(cache.listSchemas(group))));
  }

  @Override
  public List<SchemaModel> list(SchemaType type) {
    DiscoveryCache cache = requireConfigCache();
    String group = requireSessionGroup();
    SchemaType normalizedType = Objects.requireNonNull(type, "schemaType");
    return sortSchemas(
        filterCanonicalSchemas(
            mapSchemas(cache.listSchemas(group, SchemaModelMapper.toDefinitionType(normalizedType)))));
  }

  @Override

    return get(schemaName).isPresent();
  }

  @Override
  public void delete(String schemaName) {
    delete(requireSessionGroup(), schemaName);
  }

  @Override
  public void delete(String group, String schemaName) {
    String normalizedGroup = Strings.requireNonBlank(group, "group");
    String normalizedName = Strings.requireNonBlank(schemaName, "schemaName");
    DiscoveryCache cache = requireConfigCache();
    DeleteManager deleteManager = deleteManagerSupplier.get();
    if (deleteManager == null) {
      deleteManager =
          new DeleteManager(
              () -> new ArrayList<>(cache.allDefinitions()),
              (type, definitionGroup, definitionName) ->
                  cache.delete(type, definitionGroup, definitionName));
    }
    DeleteResult result =
        deleteManager.delete(
            "schema:" + normalizedGroup + "/" + normalizedName,
            new org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOptions(
                false,
                false,
                false,
                false,
                false,
                false,
                true,
                false,
                List.of()));
    if (result.outcome() == DeleteOutcome.BLOCKED) {
      throw new IllegalStateException(
          "Delete blocked for schema "
              + normalizedGroup
              + "/"
              + normalizedName
              + ": "
              + result.blockers());
    }
    if (result.outcome() == DeleteOutcome.PARTIAL) {
      throw new IllegalStateException(
          "Delete partially completed for schema "
              + normalizedGroup
              + "/"
              + normalizedName
              + ": "
              + result.warnings());
    }
  }

  private SchemaModel register(
      String group,
      String schemaName,
      SchemaType schemaType,
      SchemaModel schema) {
    String normalizedGroup = Strings.requireNonBlank(group, "group");
    String normalizedName = Strings.requireNonBlank(schemaName, "schemaName");
    SchemaType normalizedType = Objects.requireNonNull(schemaType, "schemaType");
    Objects.requireNonNull(schema, "schema");
    DiscoveryCache cache = requireConfigCache();
    SchemaModel normalizedModel =
        schema.toBuilder()
            .withGroup(normalizedGroup)
            .withName(normalizedName)
            .withType(normalizedType)
            .build();
    Optional<DefinitionSchema> existing =
        cache.get(DefinitionSchema.class, normalizedGroup, normalizedName);
    SchemaModel existingModel = null;
    DefinitionSchema existingDefinition = null;
    if (existing.isPresent()) {
      existingDefinition = existing.get();
      existingModel = SchemaModelMapper.fromDefinition(existingDefinition);
      enforceCompatibleUpdate(normalizedGroup, normalizedName, existingModel, normalizedModel);
    }
    SchemaModel modelToPersist = normalizedModel;
    if (existingModel != null) {
      int previousVersion = normalizedVersionNum(existingModel.getVersionNum());
      int nextVersion = previousVersion + 1;
      modelToPersist =
          normalizedModel.toBuilder()
              .withVersionNum(nextVersion)
              .withVersionName(versionName(nextVersion))
              .build();
    }
    String now = Instant.now().toString();
    DefinitionSchema canonicalDefinition = SchemaModelMapper.toDefinition(modelToPersist, now, SYSTEM_USER);
    if (existingDefinition != null
        && Objects.equals(
            Strings.trimToNull(existingDefinition.hash()),
            Strings.trimToNull(canonicalDefinition.hash()))) {
      return existingModel == null ? SchemaModelMapper.fromDefinition(existingDefinition) : existingModel;
    }

    DefinitionSchema versionedDefinition = null;
    if (existingDefinition != null) {
      String versionedName = nextVersionedSchemaName(cache, normalizedGroup, normalizedName);
      versionedDefinition = SchemaModelMapper.toVersionedDefinition(existingDefinition, versionedName);
    }
    try (DiscoveryCache.ConfigCacheTx tx =
        cache.beginTx("schema:" + normalizedGroup + "/" + normalizedName)) {
      if (versionedDefinition != null) {
        cache.upsertTx(tx, versionedDefinition);
      }
      cache.upsertTx(tx, canonicalDefinition);
      cache.commitTx(tx);
    }
    return SchemaModelMapper.fromDefinition(canonicalDefinition);
  }

  private List<SchemaModel> listVersions(String group, String schemaName) {
    String normalizedGroup = Strings.requireNonBlank(group, "group");
    String normalizedSchemaName = Strings.requireNonBlank(schemaName, "schemaName");
    DiscoveryCache cache = requireConfigCache();
    List<SchemaModel> versions = new ArrayList<>();
    for (SchemaModel model : mapSchemas(cache.listSchemas(normalizedGroup))) {
      String candidateName = Strings.trimToNull(model == null ? null : model.getName());
      if (!matchesVersionFamily(normalizedSchemaName, candidateName)) {
        continue;
      }
      versions.add(model);
    }
    versions.sort(
        Comparator.comparingInt((SchemaModel model) -> normalizedVersionNum(model.getVersionNum()))
            .thenComparing(SchemaManagerImpl::schemaName, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(SchemaManagerImpl::schemaName));
    return List.copyOf(versions);
  }

  private static void enforceCompatibleUpdate(
      String group, String schemaName, SchemaModel existing, SchemaModel updated) {
    if (existing == null || updated == null) {
      return;
    }
    SchemaType existingType = existing.getType();
    SchemaType updatedType = updated.getType();
    if (existingType != null && updatedType != null && existingType != updatedType) {
      throw schemaUpdateRejected(
          group,
          schemaName,
          "SCHEMA_TYPE_CHANGE",
          "_schema",
          existingType.name(),
          updatedType.name());
    }
    SchemaCompatibilityRules.requireCompatible(
        "schemaName=" + schemaName + " schemaType=" + updatedType + " group=" + group,
        updated.getCompatibility(),
        SchemaViews.fromSchemaModel(existing),
        SchemaViews.fromSchemaModel(updated));
  }

  private static IllegalArgumentException schemaUpdateRejected(
      String group,
      String schemaName,
      String reason,
      String field,
      String from,
      String to) {
    return new IllegalArgumentException(
        "Schema update rejected group="
            + group
            + " name="
            + schemaName
            + " reason="
            + reason
            + " field="
            + field
            + " from="
            + from
            + " to="
            + to);
  }

  private List<PayloadSample> collect(SampleSource source, int maxMessages, int maxBytes) {
    Objects.requireNonNull(source, "source");
    Iterator<PayloadSample> iterator = source.iterator();
    List<PayloadSample> result = new ArrayList<>();
    int totalBytes = 0;
    while (iterator != null && iterator.hasNext()) {
      PayloadSample sample = iterator.next();
      if (sample == null || sample.payload() == null) {
        continue;
      }
      if (result.size() >= maxMessages) {
        break;
      }
      int payloadBytes = sample.payload().getBytes(StandardCharsets.UTF_8).length;
      if (totalBytes + payloadBytes > maxBytes) {
        break;
      }
      totalBytes += payloadBytes;
      result.add(sample);
    }
    return List.copyOf(result);
  }

  private List<PayloadSample> collect(Iterable<String> payloads, int maxMessages, int maxBytes) {
    Objects.requireNonNull(payloads, "payloads");
    List<PayloadSample> result = new ArrayList<>();
    int totalBytes = 0;
    long now = Instant.now().toEpochMilli();
    int sampleIndex = 0;
    for (String payload : payloads) {
      if (result.size() >= maxMessages) {
        break;
      }
      PayloadSample sample =
          new PayloadSample(
              PAYLOADS_INGEST_NAME,
              payload,
              now + sampleIndex);
      int payloadBytes = sample.payload().getBytes(StandardCharsets.UTF_8).length;
      if (totalBytes + payloadBytes > maxBytes) {
        break;
      }
      totalBytes += payloadBytes;
      result.add(sample);
      sampleIndex++;
    }
    return List.copyOf(result);
  }

  private SchemaSampleResult inferFromSamples(
      String ingestName,
      String ingestId,
      String format,
      DefinitionIngestSource source,
      List<PayloadSample> samples,
      boolean allowNulls) {
    FormatSampler sampler = formatSamplerFactory.create(format, source);
    FormatSamplerResult parsed = sampler.parseSamples(samples, new FormatSamplerOptions(ingestId, allowNulls));
    if (parsed.rows().isEmpty()) {
      throw new IllegalStateException(
          "Unable to parse sampled payloads ingest="
              + ingestId
              + " format="
              + format
              + " parseErrors="
              + parsed.parseErrorCount());
    }

    SchemaInference.InferredSchema inferred = SchemaInference.infer(parsed.rows(), allowNulls);
    SchemaSampleDiagnostics diagnostics = mergeDiagnostics(inferred, parsed);
    String recommendedSchemaJson = recommendedSchemaJson(inferred.fields());

    return new SchemaSampleResult(
        ingestName,
        format,
        inferred.fields(),
        recommendedSchemaJson,
        diagnostics,
        samples.size(),
        sampledBytes(samples));
  }

  private SampleSource createDefaultSampleSource(
      String ingestId, DefinitionIngestSource source, SchemaSampleOptions options) {
    DefinitionIngestSource.AmpsSource amps = source == null ? null : source.amps();
    if (amps == null) {
      throw new IllegalArgumentException(
          "Unsupported sample source transport ingest=" + ingestId + " transport=NONE");
    }
    ConnectionManager connectionManager = connectionManagerSupplier.get();
    if (connectionManager == null) {
      throw new IllegalStateException("Session ConnectionManager is not available");
    }
    return new SampleSourceAmps(connectionManager, ingestId, amps, options);
  }

  private FormatSampler createDefaultFormatSampler(String format, DefinitionIngestSource source) {
    if (FORMAT_JSON.equals(format)) {
      return new FormatSamplerJson(source == null ? null : source.json());
    }
    if (FORMAT_NVFIX.equals(format)) {
      return new FormatSamplerNvFix(source == null ? null : source.nvfix());
    }
    throw new IllegalArgumentException("Unsupported ingest source format: " + format);
  }

  private static int sampledBytes(List<PayloadSample> samples) {
    int total = 0;
    if (samples == null) {
      return 0;
    }
    for (PayloadSample sample : samples) {
      if (sample == null || sample.payload() == null) {
        continue;
      }
      total += sample.payload().getBytes(StandardCharsets.UTF_8).length;
    }
    return total;
  }

  private static String resolveFormat(
      DefinitionIngestSource source, SchemaSampleOptions options) {
    String requested = Strings.trimToNull(options == null ? null : options.overrideFormat());
    String configured = Strings.trimToNull(source == null ? null : source.format());
    String format = requested == null ? configured : requested;
    if (format == null) {
      throw new IllegalArgumentException("ingestSource.format is required");
    }
    return normalizeFormat(format);
  }

  private static String normalizeFormat(String format) {
    String normalized = Strings.requireNonBlank(format, "format").toUpperCase(Locale.ROOT);
    if (!FORMAT_JSON.equals(normalized) && !FORMAT_NVFIX.equals(normalized)) {
      throw new IllegalArgumentException("Unsupported format: " + format);
    }
    return normalized;
  }

  private static SchemaSampleDiagnostics mergeDiagnostics(
      SchemaInference.InferredSchema inferred, FormatSamplerResult parsed) {
    Set<String> parseFailures = new TreeSet<>();
    if (parsed != null) {
      parseFailures.addAll(parsed.parseFailures());
    }
    if (inferred != null) {
      parseFailures.addAll(inferred.parseFailures());
    }
    int parseErrorCount = parseFailures.size();
    if (parsed != null && parsed.parseErrorCount() > parseErrorCount) {
      parseErrorCount = parsed.parseErrorCount();
    }
    if (inferred != null && inferred.parseErrorCount() > parseErrorCount) {
      parseErrorCount = inferred.parseErrorCount();
    }
    if (inferred == null) {
      return new SchemaSampleDiagnostics(
          List.of(), List.of(), List.of(), List.copyOf(parseFailures), parseErrorCount);
    }
    return new SchemaSampleDiagnostics(
        sortedDistinct(inferred.missingKeysSeen()),
        sortedDistinct(inferred.typeConflicts()),
        sortedDistinct(inferred.widenedFields()),
        List.copyOf(parseFailures),
        parseErrorCount);
  }

  private static String recommendedSchemaJson(List<InferredField> fields) {
    ObjectNode root = CANONICAL_MAPPER.createObjectNode();
    ArrayNode fieldNodes = root.putArray("fields");
    List<InferredField> ordered = fields == null ? List.of() : fields;
    for (InferredField field : ordered) {
      if (field == null) {
        continue;
      }
      ObjectNode node = fieldNodes.addObject();
      node.put("name", field.name());
      node.put("type", field.arrowTypeToken());
      node.put("nullable", field.nullable());
    }
    try {
      return CANONICAL_MAPPER.writeValueAsString(root);
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to render inferred schema JSON", ex);
    }
  }

  private static List<String> sortedDistinct(List<String> values) {
    if (values == null || values.isEmpty()) {
      return List.of();
    }
    TreeSet<String> deduped = new TreeSet<>();
    for (String value : values) {
      String normalized = Strings.trimToNull(value);
      if (normalized != null) {
        deduped.add(normalized);
      }
    }
    return List.copyOf(deduped);
  }

  }
