  @Override
  public CatalogSearchResult search(CatalogSearchQuery query) {
    return search(query, null);
  }

  @Override
  public CatalogSearchResult search(CatalogSearchQuery query, CatalogAccessContext accessContext) {
    CatalogSearchQuery effectiveQuery = query == null ? CatalogSearchQuery.defaults() : query;
    CatalogAccessContext effectiveAccess = resolveAccessContext(accessContext, effectiveQuery.seedAssetId());
    CatalogSnapshot snapshot = loadSnapshot(effectiveAccess.lane());
    Map<String, Boolean> visibility = resolveVisibility(snapshot, effectiveAccess);

    List<ScoredAsset> scored = new ArrayList<>();
    for (AssetRecord record : snapshot.records().values()) {
      if (!visibility.getOrDefault(record.node().assetId(), false)) {
        continue;
      }
      if (!matchesFilters(record, effectiveQuery)) {
        continue;
      }
      Score score = scoreAsset(record, effectiveQuery, snapshot);
      if (score == null) {
        continue;
      }
      scored.add(new ScoredAsset(record, score.total()));
    }

    scored.sort(
        Comparator.comparingDouble(ScoredAsset::score)
            .reversed()
            .thenComparing(scoredAsset -> scoredAsset.record().node().assetId()));

    int limit = effectiveQuery.limit() == null ? CatalogSearchQuery.DEFAULT_LIMIT : effectiveQuery.limit();
    int boundedLimit = Math.max(1, Math.min(CatalogSearchQuery.MAX_LIMIT, limit));
    List<CatalogSummary> results = new ArrayList<>();
    for (int index = 0; index < scored.size() && index < boundedLimit; index++) {
      results.add(toSummary(scored.get(index), snapshot));
    }

    return new CatalogSearchResult(results, buildFacets(scored), scored.size());
  }

  @Override
  public Optional<CatalogDetails> getAsset(String assetId) {
    return getAsset(assetId, null);
  }

  @Override
  public Optional<CatalogDetails> getAsset(String assetId, CatalogAccessContext accessContext) {
    String requested = normalize(assetId);
    if (requested == null) {
      return Optional.empty();
    }
    CatalogAccessContext effectiveAccess = resolveAccessContext(accessContext, null);
    CatalogSnapshot snapshot = loadSnapshot(effectiveAccess.lane());
    String resolved = snapshot.graph().resolveAssetId(requested);
    if (resolved == null) {
      resolved = snapshot.records().containsKey(requested) ? requested : null;
    }
    if (resolved == null) {
      return Optional.empty();
    }

    Map<String, Boolean> visibility = resolveVisibility(snapshot, effectiveAccess);
    if (!visibility.getOrDefault(resolved, false)) {
      return Optional.empty();
    }
    AssetRecord record = snapshot.records().get(resolved);
    if (record == null) {
      return Optional.empty();
    }

    Map<RelationshipType, List<CatalogRelationshipView>> relationshipsByType =
        buildRelationshipGroups(resolved, snapshot, visibility);
    Map<RelationshipType, Integer> relationshipCounts = relationshipCounts(relationshipsByType);
    CatalogSummary summary = toSummary(new ScoredAsset(record, 0d), snapshot);
    return Optional.of(
        new CatalogDetails(
            summary,
            record.semantic().businessMeaning(),
            record.semantic().aliases(),
            record.semantic().keywords(),
            record.semantic().ownership(),
            record.semantic().dataShape(),
            record.semantic().freshness(),
            record.semantic().trust(),
            record.semantic().usage(),
            record.semantic().toolHints(),
            record.completenessClass(),
            record.completenessScore(),
            relationshipsByType,
            relationshipCounts));
  }

  @Override
  public Map<RelationshipType, List<CatalogRelationshipView>> getRelatedAssets(
      String assetId, CatalogAccessContext accessContext) {
    String requested = normalize(assetId);
    if (requested == null) {
      return Map.of();
    }
    CatalogAccessContext effectiveAccess = resolveAccessContext(accessContext, null);
    CatalogSnapshot snapshot = loadSnapshot(effectiveAccess.lane());
    String resolved = snapshot.graph().resolveAssetId(requested);
    if (resolved == null) {
      resolved = snapshot.records().containsKey(requested) ? requested : null;
    }
    if (resolved == null) {
      return Map.of();
    }

    Map<String, Boolean> visibility = resolveVisibility(snapshot, effectiveAccess);
    if (!visibility.getOrDefault(resolved, false)) {
      return Map.of();
    }
    return buildRelationshipGroups(resolved, snapshot, visibility);
  }

  private CatalogAccessContext resolveAccessContext(CatalogAccessContext context, String seedAssetId) {
    String subject =
        Strings.firstNonBlank(
            normalize(context == null ? null : context.subject()), defaultSubject, "app:catalog/system");
    String delegated =
        Strings.firstNonBlank(
            normalize(context == null ? null : context.delegatedSubject()),
            defaultDelegatedSubject);
    String lane =
        Strings.firstNonBlank(
            normalize(context == null ? null : context.lane()), defaultLane, "dev");
    String seed =
        Strings.firstNonBlank(
            normalize(context == null ? null : context.seedAssetId()),
            normalize(seedAssetId));
    return new CatalogAccessContext(subject, delegated, lane, seed);
  }

  private CatalogSnapshot loadSnapshot(String lane) {
    List<Definition> definitions = safeDefinitions();
    Map<String, Semantic> configSemanticByKey = indexConfigSemantic(definitions);
    AssetLifecycleGraph graph = AssetLifecycleGraphBuilder.build(definitions);
    Map<String, Definition> definitionByAssetId = mapDefinitionsToAssetIds(graph, definitions);
    Map<String, SemanticAssetAudit> auditsByAssetId =
        buildSemanticAuditMap(definitionByAssetId, SemanticValidationPolicy.fromTier(lane));

    List<String> sortedAssetIds = new ArrayList<>(graph.nodes().keySet());
    sortedAssetIds.sort(String::compareTo);
    LinkedHashMap<String, AssetRecord> records = new LinkedHashMap<>();
    for (String assetId : sortedAssetIds) {
      AssetLifecycleNode node = graph.nodes().get(assetId);
      Definition definition = definitionByAssetId.get(assetId);
      Semantic semantic = semantic(definition, configSemanticByKey);
      if (semantic == null) {
        semantic = syntheticSemantic(node);
      }
      SemanticAssetAudit audit = auditsByAssetId.get(assetId);
      SemanticCompletenessClass completenessClass =
          audit == null ? SemanticCompletenessClass.WEAK : audit.classification();
      int completenessScore = audit == null ? 0 : audit.score();
      records.put(assetId, new AssetRecord(node, definition, semantic, completenessClass, completenessScore));
    }
    return new CatalogSnapshot(graph, records);
  }

  private static Map<String, SemanticAssetAudit> buildSemanticAuditMap(
      Map<String, Definition> definitionByAssetId, SemanticValidationPolicy policy) {
    if (definitionByAssetId.isEmpty()) {
      return Map.of();
    }
    List<AuditInput> inputs = new ArrayList<>();
    for (Map.Entry<String, Definition> entry : definitionByAssetId.entrySet()) {
      if (entry.getValue() != null) {
        inputs.add(new AuditInput(entry.getKey(), entry.getValue()));
      }
    }
    if (inputs.isEmpty()) {
      return Map.of();
    }
    Map<String, SemanticAssetAudit> auditsByAssetId = new LinkedHashMap<>();
    for (SemanticAssetAudit assetAudit : SemanticCompletenessAuditor.auditAssets(inputs, policy).assets()) {
      if (assetAudit == null || normalize(assetAudit.source()) == null) {
        continue;
      }
      auditsByAssetId.put(assetAudit.source().trim(), assetAudit);
    }
    return Map.copyOf(auditsByAssetId);
  }

  private static Map<String, Definition> mapDefinitionsToAssetIds(
      AssetLifecycleGraph graph, List<Definition> definitions) {
    if (graph == null || graph.nodes().isEmpty() || definitions == null || definitions.isEmpty()) {
      return Map.of();
    }
    Map<String, List<Definition>> indexed = indexDefinitions(definitions);
    LinkedHashMap<String, Definition> resolved = new LinkedHashMap<>();
    for (AssetLifecycleNode node : graph.nodes().values()) {
      if (node == null) {
        continue;
      }
      Definition definition = resolveDefinition(node, indexed);
      if (definition != null) {
        resolved.put(node.assetId(), definition);
      }
    }
    return Map.copyOf(resolved);
  }

  private static Map<String, List<Definition>> indexDefinitions(List<Definition> definitions) {
    LinkedHashMap<String, List<Definition>> indexed = new LinkedHashMap<>();
    for (Definition definition : definitions) {
      DefinitionEnvelope envelope = definition == null ? null : definition.envelope();
      if (envelope == null || envelope.type() == null) {
        continue;
      }
      String key = definitionKey(envelope.type(), envelope.group(), envelope.name());
      indexed.computeIfAbsent(key, ignored -> new ArrayList<>()).add(definition);
    }
    return indexed;
  }

  private static Definition resolveDefinition(AssetLifecycleNode node, Map<String, List<Definition>> indexed) {
    List<Definition> candidates =
        indexed.get(definitionKey(node.type(), node.group(), node.name()));
    if (candidates == null || candidates.isEmpty()) {
      return null;
    }
    if (candidates.size() == 1) {
      return candidates.getFirst();
    }
    String nodeNamespace = normalize(node.namespace());
    if (nodeNamespace == null) {
      return candidates.getFirst();
    }
    for (Definition candidate : candidates) {
      if (nodeNamespace.equalsIgnoreCase(defaultNamespace(candidate))) {
        return candidate;
      }
    }
    return candidates.getFirst();
  }

  private static String definitionKey(DefinitionType type, String group, String name) {
    return CatalogServiceImplIds.definitionKey(type, group, name);
  }

  private static String defaultNamespace(Definition definition) {
    return CatalogServiceImplIds.defaultNamespace(definition);
  }

  private Map<String, Boolean> resolveVisibility(
      CatalogSnapshot snapshot, CatalogAccessContext accessContext) {
    LinkedHashMap<String, Boolean> visibility = new LinkedHashMap<>();
    for (AssetRecord record : snapshot.records().values()) {
      String assetId = record.node().assetId();
      if (PRIMARY_ENTITLED_TYPES.contains(record.node().type())) {
        visibility.put(assetId, authorizePrimary(record, accessContext));
      }
    }

    for (AssetRecord record : snapshot.records().values()) {
      String assetId = record.node().assetId();
      if (visibility.containsKey(assetId)) {
        continue;
      }
      Set<String> linkedPrimaryAssets = linkedPrimaryAssets(snapshot, assetId);
      if (linkedPrimaryAssets.isEmpty()) {
        visibility.put(assetId, true);
        continue;
      }
      boolean allowed = false;
      for (String linkedPrimary : linkedPrimaryAssets) {
        if (visibility.getOrDefault(linkedPrimary, false)) {
          allowed = true;
          break;
        }
      }
      visibility.put(assetId, allowed);
    }
    return Map.copyOf(visibility);
  }

  private Set<String> linkedPrimaryAssets(CatalogSnapshot snapshot, String assetId) {
    LinkedHashSet<String> linked = new LinkedHashSet<>();
    for (AssetLifecycleEdge edge : snapshot.graph().outgoing(assetId)) {
      AssetRecord related = snapshot.records().get(edge.toAssetId());
      if (related != null && PRIMARY_ENTITLED_TYPES.contains(related.node().type())) {
        linked.add(edge.toAssetId());
      }
    }
    for (AssetLifecycleEdge edge : snapshot.graph().incoming(assetId)) {
      AssetRecord related = snapshot.records().get(edge.fromAssetId());
      if (related != null && PRIMARY_ENTITLED_TYPES.contains(related.node().type())) {
        linked.add(edge.fromAssetId());
      }
    }
    return Set.copyOf(linked);
  }

  private boolean authorizePrimary(AssetRecord record, CatalogAccessContext accessContext) {
    return switch (record.node().type()) {
      case DATASET -> authorizeDataset(record, accessContext);
      case SERVICE -> authorizeService(record, accessContext);
      case CONFIG_LAYER -> authorizeConfig(record, accessContext);
      default -> true;
    };
  }

  private boolean authorizeDataset(AssetRecord record, CatalogAccessContext accessContext) {
    String datasetId = normalize(record.node().assetId());
    if (datasetId == null || !datasetId.startsWith("dataset:")) {
      return false;
    }
    try {
      DataEntitlements.Decision decision =
          dataEntitlements.authorizeRead(
              new DataEntitlements.Identity(accessContext.subject(), accessContext.delegatedSubject()),
              datasetId);
      return decision != null && decision.allowed();
    } catch (RuntimeException ex) {
      return false;
    }
  }

  private boolean authorizeService(AssetRecord record, CatalogAccessContext accessContext) {
    String serviceResourceId = null;
    if (record.definition() instanceof DefinitionService service
        && Strings.isNotBlank(service.resourceId())) {
      serviceResourceId = normalize(service.resourceId());
    }
    if (serviceResourceId == null) {
      serviceResourceId = normalize(record.node().assetId());
    }
    if (serviceResourceId == null) {
      return false;
    }
    try {
      serviceResourceId = ServiceEntitlements.requireServiceResourceId(serviceResourceId);
      ServiceEntitlements.Decision decision =
          serviceEntitlements.authorize(
              new ServiceEntitlements.Identity(accessContext.subject(), accessContext.delegatedSubject()),
              new ServiceEntitlements.Request(
                  ServiceEntitlements.Transport.GRPC,
                  ServiceEntitlements.Action.READ,
                  serviceResourceId,
                  "catalog.read",
                  null,
                  null,
                  Map.of(),
                  Map.of("surface", "semantic-catalog")));
      return decision != null && decision.allowed();
    } catch (RuntimeException ex) {
      return false;
    }
  }

  private boolean authorizeConfig(AssetRecord record, CatalogAccessContext accessContext) {
    String configResourceId = configResourceId(record);
    if (configResourceId == null) {
      return false;
    }
    try {
      configResourceId = ServiceEntitlements.requireConfigResourceId(configResourceId);
      ServiceEntitlements.Decision decision =
          serviceEntitlements.authorize(
              new ServiceEntitlements.Identity(accessContext.subject(), accessContext.delegatedSubject()),
              new ServiceEntitlements.Request(
                  ServiceEntitlements.Transport.GRPC,
                  ServiceEntitlements.Action.READ,
                  configResourceId,
                  "catalog.read.config",
                  null,
                  null,
                  Map.of(),
                  Map.of("surface", "semantic-catalog")));
      return decision != null && decision.allowed();
    } catch (RuntimeException ex) {
      return false;
    }
  }

  private static String configResourceId(AssetRecord record) {
    if (record == null || record.node() == null) {
      return null;
    }
    String assetId = normalize(record.node().assetId());
    if (assetId != null && assetId.startsWith("config:")) {
      return assetId;
    }
    if (record.definition() instanceof DefinitionConfigLayer config) {
      String resourceId = normalize(config.resourceId());
      if (resourceId != null && resourceId.startsWith("config:")) {
        return resourceId;
      }
      String configId = normalize(config.configId());
      if (configId != null && configId.startsWith("config:")) {
        return configId;
      }
      String group = normalize(config.envelope() == null ? null : config.envelope().group());
      String logicalId = normalize(config.configId());
      if (group != null && logicalId != null) {
        return ServiceEntitlements.canonicalConfigResourceId(group, logicalId);
      }
    }
    return null;
  }

  private static boolean matchesFilters(AssetRecord record, CatalogSearchQuery query) {
    if (record == null || query == null) {
      return false;
    }
    Semantic semantic = record.semantic();
    if (!matchesType(record, query.type())) {
      return false;
    }
    if (query.domain() != null && !query.domain().equalsIgnoreCase(normalize(semantic.domain()))) {
      return false;
    }
    if (query.owner() != null) {
      String ownerId = semantic.ownership() == null ? null : normalize(semantic.ownership().ownerId());
      if (!query.owner().equalsIgnoreCase(ownerId)) {
        return false;
      }
    }
    if (query.trust() != null && !matchesTrustFilter(semantic.trust(), query.trust())) {
      return false;
    }
    if (query.freshness() != null && !matchesFreshnessFilter(semantic.freshness(), query.freshness())) {
      return false;
    }
    if (!query.tags().isEmpty() && !matchesTagsFilter(semantic.tags(), query.tags())) {
      return false;
    }
    return true;
  }

  private static boolean matchesType(AssetRecord record, String typeFilter) {
    if (typeFilter == null) {
      return true;
    }
    return typeFilter.equalsIgnoreCase(record.node().type().name());
  }

  private static boolean matchesTagsFilter(List<String> assetTags, List<String> filterTags) {
    if (filterTags == null || filterTags.isEmpty()) {
      return true;
    }
    Set<String> normalizedAssetTags = toNormalizedSet(assetTags);
    for (String filterTag : filterTags) {
      if (!normalizedAssetTags.contains(normalize(filterTag))) {
        return false;
      }
    }
    return true;
  }

  private static boolean matchesTrustFilter(Semantic.Trust trust, String trustFilter) {
    if (trustFilter == null) {
      return true;
    }
    if (trust == null) {
      return false;
    }
    String normalizedFilter = trustFilter.toLowerCase(Locale.ROOT);
    String certification = trust.certification() == null ? "" : trust.certification().name().toLowerCase(Locale.ROOT);
    String quality = trust.qualityStatus() == null ? "" : trust.qualityStatus().name().toLowerCase(Locale.ROOT);
    return certification.contains(normalizedFilter) || quality.contains(normalizedFilter);
  }

  private static boolean matchesFreshnessFilter(Semantic.Freshness freshness, String freshnessFilter) {
    if (freshnessFilter == null) {
      return true;
    }
    if (freshness == null) {
      return false;
    }
    String normalizedFilter = freshnessFilter.toLowerCase(Locale.ROOT);
    String updateCadence = normalize(freshness.updateCadence());
    String freshnessSla = normalize(freshness.freshnessSla());
    String latencyClass = normalize(freshness.latencyClass());
    return (updateCadence != null && updateCadence.toLowerCase(Locale.ROOT).contains(normalizedFilter))
        || (freshnessSla != null && freshnessSla.toLowerCase(Locale.ROOT).contains(normalizedFilter))
        || (latencyClass != null && latencyClass.toLowerCase(Locale.ROOT).contains(normalizedFilter));
  }

  private Score scoreAsset(AssetRecord record, CatalogSearchQuery query, CatalogSnapshot snapshot) {
    List<String> tokens = tokenize(query.query());
    double textScore = textScore(record, tokens);
    if (!tokens.isEmpty() && textScore <= 0d) {
      return null;
    }
    double completenessScore = record.completenessScore() / 100d;
    double trustScore = trustScore(record.semantic().trust());
    double freshnessScore = freshnessScore(record.semantic().freshness());
    double domainRelevance = domainRelevance(record.semantic(), query.domain());
    double proximityScore = proximityScore(query.seedAssetId(), record.node().assetId(), snapshot.graph());
    double relationshipImportance = relationshipImportance(record.node().assetId(), snapshot.graph());

    double total =
        (textScore * 0.45d)
            + (completenessScore * 0.20d)
            + (trustScore * 0.15d)
            + (freshnessScore * 0.10d)
            + (domainRelevance * 0.05d)
            + (proximityScore * 0.03d)
            + (relationshipImportance * 0.02d);
    return new Score(round(total * 100d));
  }

  private static double textScore(AssetRecord record, List<String> tokens) {
    if (tokens == null || tokens.isEmpty()) {
      return 0.20d;
    }
    Semantic semantic = record.semantic();
    double total = 0d;
    for (String token : tokens) {
      double best = 0d;
      best = Math.max(best, containsScore(semantic.title(), token, 1.00d));
      best = Math.max(best, containsScore(semantic.description(), token, 0.75d));
      best = Math.max(best, containsScore(semantic.businessMeaning(), token, 0.75d));
      best = Math.max(best, containsScore(semantic.domain(), token, 0.60d));
      best = Math.max(best, containsScore(semantic.subdomain(), token, 0.60d));
      best = Math.max(best, containsScore(record.node().assetId(), token, 0.60d));
      best = Math.max(best, containsScore(record.node().name(), token, 0.60d));
      best = Math.max(best, listContainsScore(semantic.tags(), token, 0.85d));
      best = Math.max(best, listContainsScore(semantic.aliases(), token, 0.85d));
      best = Math.max(best, listContainsScore(semantic.keywords(), token, 0.85d));
      total += best;
    }
    return Math.max(0d, Math.min(1d, total / tokens.size()));
  }

  private static double containsScore(String value, String token, double score) {
    String normalizedValue = normalize(value);
    if (normalizedValue == null || token == null) {
      return 0d;
    }
    return normalizedValue.toLowerCase(Locale.ROOT).contains(token) ? score : 0d;
  }

  private static double listContainsScore(List<String> values, String token, double score) {
    if (values == null || values.isEmpty()) {
      return 0d;
    }
    for (String value : values) {
      if (containsScore(value, token, score) > 0d) {
        return score;
      }
    }
    return 0d;
  }

  private static double trustScore(Semantic.Trust trust) {
    if (trust == null) {
      return 0.35d;
    }
    double certificationScore =
        switch (trust.certification() == null ? Semantic.Certification.NONE : trust.certification()) {
          case CERTIFIED -> 1.00d;
          case CANDIDATE -> 0.80d;
          case NONE -> 0.40d;
          case DEPRECATED -> 0.20d;
        };
    double qualityScore =
        switch (trust.qualityStatus() == null ? Semantic.QualityStatus.UNKNOWN : trust.qualityStatus()) {
          case GOOD -> 1.00d;
          case WARNING -> 0.60d;
          case BAD -> 0.20d;
          case UNKNOWN -> 0.40d;
        };
    return (certificationScore + qualityScore) / 2d;
  }

  private static double freshnessScore(Semantic.Freshness freshness) {
    if (freshness == null) {
      return 0.35d;
    }
    double score = 0d;
    if (!SemanticMetadataValidator.isPlaceholderText(freshness.updateCadence())) {
      score += 0.45d;
    }
    if (!SemanticMetadataValidator.isPlaceholderText(freshness.freshnessSla())) {
      score += 0.35d;
    }
    if (!SemanticMetadataValidator.isPlaceholderText(freshness.latencyClass())) {
      score += 0.20d;
    }
    return Math.max(0d, Math.min(1d, score));
  }

  private static double domainRelevance(Semantic semantic, String domainFilter) {
    if (semantic == null) {
      return 0.25d;
    }
    if (domainFilter == null) {
      return 0.50d;
    }
    String domain = normalize(semantic.domain());
    if (domain != null && domainFilter.equalsIgnoreCase(domain)) {
      return 1.00d;
    }
    String subdomain = normalize(semantic.subdomain());
    if (subdomain != null && domainFilter.equalsIgnoreCase(subdomain)) {
      return 0.80d;
    }
    return 0.10d;
  }

  private static double proximityScore(String seedAssetId, String targetAssetId, AssetLifecycleGraph graph) {
    String seed = normalize(seedAssetId);
    String target = normalize(targetAssetId);
    if (seed == null || target == null || graph == null) {
      return 0d;
    }
    String resolvedSeed = graph.resolveAssetId(seed);
    if (resolvedSeed == null) {
      resolvedSeed = seed;
    }
    if (resolvedSeed.equals(target)) {
      return 1.00d;
    }
    if (isDirectlyRelated(resolvedSeed, target, graph)) {
      return 0.85d;
    }
    if (isWithinTwoHops(resolvedSeed, target, graph)) {
      return 0.45d;
    }
    return 0d;
  }

  private static boolean isDirectlyRelated(String source, String target, AssetLifecycleGraph graph) {
    for (AssetLifecycleEdge edge : graph.outgoing(source)) {
      if (target.equals(edge.toAssetId())) {
        return true;
      }
    }
    for (AssetLifecycleEdge edge : graph.incoming(source)) {
      if (target.equals(edge.fromAssetId())) {
        return true;
      }
    }
    return false;
  }

  private static boolean isWithinTwoHops(String source, String target, AssetLifecycleGraph graph) {
    ArrayDeque<String> queue = new ArrayDeque<>();
    Map<String, Integer> depth = new LinkedHashMap<>();
    queue.add(source);
    depth.put(source, 0);
    while (!queue.isEmpty()) {
      String current = queue.removeFirst();
      int currentDepth = depth.getOrDefault(current, 0);
      if (currentDepth >= 2) {
        continue;
      }
      for (String next : neighbors(current, graph)) {
        if (depth.containsKey(next)) {
          continue;
        }
        int nextDepth = currentDepth + 1;
        if (next.equals(target) && nextDepth <= 2) {
          return true;
        }
        depth.put(next, nextDepth);
        queue.addLast(next);
      }
    }
    return false;
  }

  private static Set<String> neighbors(String assetId, AssetLifecycleGraph graph) {
    LinkedHashSet<String> neighbors = new LinkedHashSet<>();
    for (AssetLifecycleEdge edge : graph.outgoing(assetId)) {
      neighbors.add(edge.toAssetId());
    }
    for (AssetLifecycleEdge edge : graph.incoming(assetId)) {
      neighbors.add(edge.fromAssetId());
    }
    return Set.copyOf(neighbors);
  }

  private static double relationshipImportance(String assetId, AssetLifecycleGraph graph) {
    int relationshipCount = graph.outgoing(assetId).size() + graph.incoming(assetId).size();
    if (relationshipCount <= 0) {
      return 0d;
    }
    return Math.min(1d, relationshipCount / 10d);
  }

  private static CatalogSearchFacet buildFacets(List<ScoredAsset> scoredAssets) {
    if (scoredAssets == null || scoredAssets.isEmpty()) {
      return CatalogSearchFacet.empty();
    }
    LinkedHashMap<String, Integer> typeCounts = new LinkedHashMap<>();
    LinkedHashMap<String, Integer> domainCounts = new LinkedHashMap<>();
    LinkedHashMap<String, Integer> tagCounts = new LinkedHashMap<>();
    LinkedHashMap<String, Integer> ownerCounts = new LinkedHashMap<>();
    LinkedHashMap<String, Integer> trustCounts = new LinkedHashMap<>();
    LinkedHashMap<String, Integer> freshnessCounts = new LinkedHashMap<>();

    for (ScoredAsset scored : scoredAssets) {
      AssetRecord record = scored.record();
      Semantic semantic = record.semantic();
      increment(typeCounts, record.node().type().name());
      increment(domainCounts, semantic.domain());
      for (String tag : semantic.tags()) {
        increment(tagCounts, tag);
      }
      if (semantic.ownership() != null) {
        increment(ownerCounts, semantic.ownership().ownerId());
      }
      if (semantic.trust() != null && semantic.trust().certification() != null) {
        increment(trustCounts, semantic.trust().certification().name());
      }
      if (semantic.freshness() != null) {
        increment(freshnessCounts, semantic.freshness().updateCadence());
      }
    }
    return new CatalogSearchFacet(
        typeCounts, domainCounts, tagCounts, ownerCounts, trustCounts, freshnessCounts);
  }

  private CatalogSummary toSummary(ScoredAsset scoredAsset, CatalogSnapshot snapshot) {
    AssetRecord record = scoredAsset.record();
    Semantic semantic = record.semantic();
    int relationshipCount =
        snapshot.graph().outgoing(record.node().assetId()).size()
            + snapshot.graph().incoming(record.node().assetId()).size();
    Semantic.Ownership ownership = semantic.ownership();
    Semantic.Trust trust = semantic.trust();
    Semantic.Freshness freshness = semantic.freshness();
    return new CatalogSummary(
        record.node().assetId(),
        record.node().type(),
        record.node().group(),
        record.node().name(),
        record.node().namespace(),
        semantic.title(),
        semantic.description(),
        semantic.domain(),
        semantic.subdomain(),
        semantic.tags(),
        ownership == null ? null : ownership.ownerId(),
        trust == null ? Semantic.Certification.NONE : trust.certification(),
        trust == null ? Semantic.QualityStatus.UNKNOWN : trust.qualityStatus(),
        freshness == null ? null : freshness.updateCadence(),
        record.completenessClass(),
        record.completenessScore(),
        relationshipCount,
        scoredAsset.score());
  }

  private Map<RelationshipType, List<CatalogRelationshipView>> buildRelationshipGroups(
      String rootAssetId, CatalogSnapshot snapshot, Map<String, Boolean> visibility) {
    LinkedHashMap<RelationshipType, List<CatalogRelationshipView>> grouped = new LinkedHashMap<>();
    for (AssetLifecycleEdge edge : snapshot.graph().outgoing(rootAssetId)) {
      if (!isVisible(visibility, edge.toAssetId())) {
        continue;
      }
      CatalogRelationshipView relationship = toRelationshipView(edge, CatalogRelationshipDirection.OUTGOING, snapshot);
      grouped.computeIfAbsent(relationship.relationshipType(), ignored -> new ArrayList<>()).add(relationship);
    }
    for (AssetLifecycleEdge edge : snapshot.graph().incoming(rootAssetId)) {
      if (!isVisible(visibility, edge.fromAssetId())) {
        continue;
      }
      CatalogRelationshipView relationship = toRelationshipView(edge, CatalogRelationshipDirection.INCOMING, snapshot);
      grouped.computeIfAbsent(relationship.relationshipType(), ignored -> new ArrayList<>()).add(relationship);
    }

    LinkedHashMap<RelationshipType, List<CatalogRelationshipView>> sorted = new LinkedHashMap<>();
    for (RelationshipType type : RelationshipType.values()) {
      List<CatalogRelationshipView> values = grouped.get(type);
      if (values == null || values.isEmpty()) {
        continue;
      }
      values.sort(
          Comparator.comparing(CatalogRelationshipView::direction)
              .thenComparing(CatalogRelationshipView::fromAssetId)
              .thenComparing(CatalogRelationshipView::toAssetId));
      sorted.put(type, List.copyOf(values));
    }
    return Map.copyOf(sorted);
  }

  private static Map<RelationshipType, Integer> relationshipCounts(
      Map<RelationshipType, List<CatalogRelationshipView>> grouped) {
    if (grouped == null || grouped.isEmpty()) {
      return Map.of();
    }
    EnumMap<RelationshipType, Integer> counts = new EnumMap<>(RelationshipType.class);
    for (Map.Entry<RelationshipType, List<CatalogRelationshipView>> entry : grouped.entrySet()) {
      counts.put(entry.getKey(), entry.getValue() == null ? 0 : entry.getValue().size());
    }
    return Map.copyOf(counts);
  }

  private CatalogRelationshipView toRelationshipView(
      AssetLifecycleEdge edge, CatalogRelationshipDirection direction, CatalogSnapshot snapshot) {
    String relatedAssetId =
        direction == CatalogRelationshipDirection.OUTGOING ? edge.toAssetId() : edge.fromAssetId();
    AssetLifecycleNode relatedNode = snapshot.graph().nodes().get(relatedAssetId);
    ParsedAssetId parsed = relatedNode == null ? parseAssetId(relatedAssetId) : null;
    DefinitionType relatedType = relatedNode == null ? parsed.type() : relatedNode.type();
    String relatedGroup = relatedNode == null ? parsed.group() : relatedNode.group();
    String relatedName = relatedNode == null ? parsed.name() : relatedNode.name();
    String relatedNamespace = relatedNode == null ? parsed.namespace() : relatedNode.namespace();

    return new CatalogRelationshipView(
        edge.fromAssetId(),
        edge.toAssetId(),
        direction,
        edge.relationshipType(),
        edge.ownershipType(),
        edge.promotionBehavior(),
        edge.deleteBehavior(),
        edge.required(),
        edge.shared(),
        edge.notes(),
        edge.targetLocator(),
        relatedType,
        relatedGroup,
        relatedName,
        relatedNamespace);
  }

  private static boolean isVisible(Map<String, Boolean> visibility, String assetId) {
    if (visibility == null || visibility.isEmpty()) {
      return true;
    }
    if (!visibility.containsKey(assetId)) {
      return true;
    }
    return visibility.getOrDefault(assetId, false);
  }

  private static ParsedAssetId parseAssetId(String assetId) {
    CatalogServiceImplIds.ParsedAssetId parsed = CatalogServiceImplIds.parseAssetId(assetId);
    return new ParsedAssetId(parsed.type(), parsed.group(), parsed.name(), parsed.namespace());
  }

  private static List<Definition> safeDefinitionsCopy(List<Definition> value) {
    if (value == null || value.isEmpty()) {
      return List.of();
    }
    List<Definition> copy = new ArrayList<>();
    for (Definition definition : value) {
      if (definition != null) {
        copy.add(definition);
      }
    }
    return List.copyOf(copy);
  }

  private List<Definition> safeDefinitions() {
    try {
      return safeDefinitionsCopy(definitionsSupplier.get());
    } catch (RuntimeException ex) {
      return List.of();
    }
  }

  private static Map<String, Semantic> indexConfigSemantic(List<Definition> definitions) {
    if (definitions == null || definitions.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, Semantic> indexed = new LinkedHashMap<>();
    for (Definition definition : definitions) {
      if (!(definition instanceof DefinitionConfig configMeta)
          || configMeta.semantic() == null) {
        continue;
      }
      String group =
          Strings.firstNonBlank(
              normalize(configMeta.envelope() == null ? null : configMeta.envelope().group()),
              normalize(configMeta.group()));
      String configSetId = normalizeConfigSetId(configMeta.configId());
      if (group == null || configSetId == null) {
        continue;
      }
      indexed.put(configSemanticKey(group, configSetId), configMeta.semantic());
    }
    return indexed.isEmpty() ? Map.of() : Map.copyOf(indexed);
  }

  private static Semantic semantic(
      Definition definition, Map<String, Semantic> configSemanticByKey) {
    if (definition instanceof DefinitionDataset dataset) {
      return dataset.semantic();
    }
    if (definition instanceof DefinitionService service) {
      return service.semantic();
    }
    if (definition instanceof DefinitionStream stream) {
      return stream.semantic();
    }
    if (definition instanceof DefinitionIngest ingest) {
      return ingest.semantic();
    }
    if (definition instanceof DefinitionIngestSource ingestSource) {
      return ingestSource.semantic();
    }
    if (definition instanceof DefinitionSchema schema) {
      return schema.semantic();
    }
    if (definition instanceof DefinitionConfigLayer config) {
      String group = normalize(config.envelope() == null ? null : config.envelope().group());
      String configSetId = normalizeConfigSetId(config.configId());
      if (group != null && configSetId != null && configSemanticByKey != null) {
        Semantic fromMeta = configSemanticByKey.get(configSemanticKey(group, configSetId));
        if (fromMeta != null) {
          return fromMeta;
        }
      }
      return null;
    }
    return null;
  }

  private static String configSemanticKey(String group, String configSetId) {
    return group + "|" + configSetId;
  }

  private static String normalizeConfigSetId(String value) {
    String normalized = normalize(value);
    if (normalized == null) {
      return null;
    }
    if (!normalized.startsWith("config:")) {
      return normalized;
    }
    String payload = normalized.substring("config:".length());
    String[] parts = payload.split("/");
    if (parts.length < 2) {
      return null;
    }
    return normalize(parts[1]);
  }

  private static Semantic.Type toSemanticType(DefinitionType definitionType) {
    return switch (definitionType) {
      case DATASET -> Semantic.Type.DATASET;
      case SERVICE -> Semantic.Type.SERVICE;
      case STREAM -> Semantic.Type.STREAM;
      case INGEST -> Semantic.Type.INGEST;
      case INGEST_SOURCE -> Semantic.Type.INGEST_SOURCE;
      case SCHEMA -> Semantic.Type.SCHEMA;
      case CONFIG_LAYER -> Semantic.Type.CONFIG;
      default -> Semantic.Type.DOCUMENT;
    };
  }

  private static Set<String> toNormalizedSet(List<String> values) {
    if (values == null || values.isEmpty()) {
      return Set.of();
    }
    LinkedHashSet<String> normalized = new LinkedHashSet<>();
    for (String value : values) {
      String cleaned = normalize(value);
      if (cleaned != null) {
        normalized.add(cleaned);
      }
    }
    return Set.copyOf(normalized);
  }

  private static List<String> tokenize(String text) {
    String normalized = normalize(text);
    if (normalized == null) {
      return List.of();
    }
    LinkedHashSet<String> tokens = new LinkedHashSet<>();
    for (String token : normalized.toLowerCase(Locale.ROOT).split("[^a-z0-9:_./-]+")) {
      if (!token.isBlank()) {
        tokens.add(token);
      }
    }
    if (tokens.isEmpty()) {
      return List.of();
    }
    return List.copyOf(tokens);
  }

  private static void increment(Map<String, Integer> map, String value) {
    String normalized = normalize(value);
    if (normalized == null) {
      return;
    }
    map.merge(normalized, 1, Integer::sum);
  }

  private static String normalize(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }

  private static double round(double value) {
    return Math.round(value * 100d) / 100d;
  }
