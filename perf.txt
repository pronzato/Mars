  static void prepareDuckDbActive(RowManagerImpl self, DatasetId datasetId) {
    Objects.requireNonNull(self, "self");
    Objects.requireNonNull(datasetId, "datasetId");
    DefinitionDataset definition = self.resolveDatasetDefinition(datasetId);
    if (definition.binding() == null || definition.binding().duckdb() == null) {
      return;
    }
    DefinitionDataset.DuckdbBinding binding = definition.binding().duckdb();
    if (!self.isDuckDbActive(binding)) {
      return;
    }
    if (Boolean.TRUE.equals(binding.memory())) {
      synchronized (self.sharedSessionLockInternal()) {
        ensureDuckDbActiveInitialized(
            self, self.requireSharedSession(), datasetId.toString(), binding.relationName(), binding);
      }
      return;
    }
    try (DuckdbSession session = openDuckDbActiveSession(self, binding)) {
      ensureDuckDbActiveInitialized(self, session, datasetId.toString(), binding.relationName(), binding);
    }
  }

  static long duckDbActiveChangeMaxSeq(RowManagerImpl self, DatasetId datasetId) {
    Objects.requireNonNull(self, "self");
    Objects.requireNonNull(datasetId, "datasetId");
    RowManagerImpl.ActiveDatasetBinding active = resolveActiveDatasetBinding(self, datasetId);
    return withActiveDatasetSession(
        self,
        active,
        session -> {
          ensureDuckDbActiveChangeTable(session, active.datasetId());
          return queryDuckDbActiveChangeMaxSeq(session, active.datasetId(), active.canonicalDatasetId());
        });
  }

  static List<DuckdbActiveChangeLog.ChangeEvent> readDuckDbActiveChanges(
      RowManagerImpl self, DatasetId datasetId, long afterSeq, int limit) {
    Objects.requireNonNull(self, "self");
    Objects.requireNonNull(datasetId, "datasetId");
    if (afterSeq < 0L) {
      throw new IllegalArgumentException("afterSeq must be >= 0");
    }
    int resolvedLimit = limit <= 0 ? 1 : limit;
    RowManagerImpl.ActiveDatasetBinding active = resolveActiveDatasetBinding(self, datasetId);
    return withActiveDatasetSession(
        self,
        active,
        session -> {
          ensureDuckDbActiveChangeTable(session, active.datasetId());
          String sql =
              "select seq, pk, op from "
                  + DuckdbActiveChangeLog.quoteIdentifier(
                      DuckdbActiveChangeLog.tableName(active.datasetId()))
                  + " where seq > ? order by seq limit ?";
          try (PreparedStatement statement = session.prepareStatement(sql)) {
            statement.setLong(1, afterSeq);
            statement.setInt(2, resolvedLimit);
            boolean hasResult = statement.execute();
            if (!hasResult) {
              return List.of();
            }
            List<DuckdbActiveChangeLog.ChangeEvent> changes = new ArrayList<>();
            try (ResultSet resultSet = statement.getResultSet()) {
              while (resultSet.next()) {
                changes.add(
                    new DuckdbActiveChangeLog.ChangeEvent(
                        Math.max(0L, resultSet.getLong(1)),
                        RowManagerImpl.trimToNull(resultSet.getString(2)),
                        RowManagerImpl.trimToNull(resultSet.getString(3))));
              }
            }
            return List.copyOf(changes);
          } catch (SQLException ex) {
            throw new IllegalStateException(
                "Failed to read DUCKDB_ACTIVE changes for " + active.canonicalDatasetId(), ex);
          }
        });
  }

  static void appendArrowIpcToDuckdbActive(
      RowManagerImpl self,
      String canonicalDatasetId,
      String relationName,
      DefinitionDataset.DuckdbBinding binding,
      byte[] arrowIpcStreamBytes,
      DuckDbActiveMutation mutation) {
    String tableName = RowManagerImpl.requireNonBlank(relationName, "Dataset DUCKDB relationName is required");
    List<String> primaryKey = normalizePrimaryKey(binding.primaryKey(), canonicalDatasetId);

    try (ArrowStreamReader reader =
        new ArrowStreamReader(new ByteArrayInputStream(arrowIpcStreamBytes), self.ipcAllocatorInternal())) {
      if (Boolean.TRUE.equals(binding.memory())) {
        synchronized (self.sharedSessionLockInternal()) {
          appendArrowIpcToDuckdbActiveInSession(
              self,
              self.requireSharedSession(),
              canonicalDatasetId,
              tableName,
              binding,
              primaryKey,
              reader,
              mutation);
        }
      } else {
        try (DuckdbSession session = openDuckDbActiveSession(self, binding)) {
          appendArrowIpcToDuckdbActiveInSession(
              self, session, canonicalDatasetId, tableName, binding, primaryKey, reader, mutation);
        }
      }
    } catch (IllegalStateException ex) {
      throw ex;
    } catch (RuntimeException | IOException ex) {
      throw new IllegalStateException(
          "Arrow IPC append failed for dataset " + canonicalDatasetId + ": " + ex.getMessage(), ex);
    }
  }

  static void appendArrowIpcToDuckdbActiveInSession(
      RowManagerImpl self,
      DuckdbSession session,
      String canonicalDatasetId,
      String tableName,
      DefinitionDataset.DuckdbBinding binding,
      List<String> primaryKey,
      ArrowStreamReader reader,
      DuckDbActiveMutation mutation)
      throws IOException {
    ensureDuckDbActiveInitialized(self, session, canonicalDatasetId, tableName, binding);

    VectorSchemaRoot root = reader.getVectorSchemaRoot();
    List<Field> fields = root.getSchema().getFields();
    if (fields == null || fields.isEmpty()) {
      throw new IllegalStateException("Arrow IPC stream schema is empty for dataset " + canonicalDatasetId);
    }
    List<RowManagerImpl.AppendColumn> sortedColumns = self.sortedAppendColumns(fields, canonicalDatasetId);
    validatePrimaryKeyColumns(primaryKey, sortedColumns, canonicalDatasetId, tableName);
    ensureDuckDbActiveTable(self, session, canonicalDatasetId, tableName, sortedColumns, binding);

    String tempTableName = self.nextTempTableName("active_");
    String createTempSql = RowManagerImplSql.buildCreateActiveTempTableSql(tempTableName, sortedColumns);
    String insertSql = RowManagerImplSql.buildInsertActiveTempSql(tempTableName, sortedColumns);
    boolean tempCreated = false;
    RuntimeException failure = null;
    try {
      session.exec(createTempSql);
      tempCreated = true;
      long insertedRows =
          insertArrowRowsIntoActiveTemp(self, session, insertSql, sortedColumns, root, reader, canonicalDatasetId);
      if (insertedRows <= 0L) {
        return;
      }
      applyDuckDbActiveMutation(
          session, canonicalDatasetId, tableName, tempTableName, sortedColumns, primaryKey, mutation);
      DatasetId datasetId = DatasetId.parse(canonicalDatasetId);
      long latestSeq = appendDuckDbActiveChangeLog(session, datasetId, tempTableName, primaryKey, mutation);
      self.changeNotifierInternal().notifyChanged(datasetId.toString(), latestSeq);
      writeDuckDbActiveHistory(
          self, session, canonicalDatasetId, tableName, tempTableName, sortedColumns, mutation, binding);
    } catch (RuntimeException ex) {
      failure = ex;
      throw ex;
    } finally {
      if (tempCreated) {
        try {
          session.exec("drop table if exists " + RowManagerImpl.quoteIdentifier(tempTableName));
        } catch (RuntimeException cleanupEx) {
          if (failure != null) {
            failure.addSuppressed(cleanupEx);
          } else {
            throw cleanupEx;
          }
        }
      }
    }
  }

  static DuckdbSession openDuckDbActiveSession(
      RowManagerImpl self, DefinitionDataset.DuckdbBinding binding) {
    if (Boolean.TRUE.equals(binding.memory())) {
      throw new IllegalStateException(
          "DUCKDB_ACTIVE memory=true uses shared in-process DuckDB session and must not open a file-backed session");
    }
    String databasePath = resolveDuckDbActiveDatabasePath(binding);
    return self.engineInternal().openSession(databasePath);
  }

  static String resolveDuckDbActiveDatabasePath(DefinitionDataset.DuckdbBinding binding) {
    String configuredPath = RowManagerImpl.trimToNull(binding.databasePath());
    if (Boolean.FALSE.equals(binding.memory())) {
      return RowManagerImpl.requireNonBlank(configuredPath, "DUCKDB_ACTIVE databasePath is required");
    }
    if (configuredPath != null) {
      return configuredPath;
    }
    Path defaultPath =
        Path.of("data", "lite", "duckdb-active", sanitizeFilename(binding.relationName()) + ".duckdb")
            .toAbsolutePath()
            .normalize();
    try {
      Path parent = defaultPath.getParent();
      if (parent != null) {
        Files.createDirectories(parent);
      }
    } catch (IOException ex) {
      throw new IllegalStateException(
          "Failed to initialize DUCKDB_ACTIVE database directory for " + binding.relationName(), ex);
    }
    return defaultPath.toString();
  }

  static void ensureDuckDbActiveInitialized(
      RowManagerImpl self,
      DuckdbSession session,
      String canonicalDatasetId,
      String tableName,
      DefinitionDataset.DuckdbBinding binding) {
    if (!Boolean.TRUE.equals(binding.memory())) {
      return;
    }
    String historyPath = RowManagerImpl.trimToNull(binding.parquetHistoryPath());
    if (historyPath == null) {
      return;
    }
    if (self.activeDatasetsInitializedInternal().contains(canonicalDatasetId)) {
      return;
    }
    synchronized (self.activeDatasetsInitializedInternal()) {
      if (self.activeDatasetsInitializedInternal().contains(canonicalDatasetId)) {
        return;
      }
      Path root = RowManagerImpl.normalizePath(historyPath).toAbsolutePath().normalize();
      if (Files.exists(root)) {
        List<Path> historyFiles = listActiveHistoryFiles(root);
        for (Path historyFile : historyFiles) {
          replayDuckDbActiveHistoryFile(self, session, canonicalDatasetId, tableName, binding, historyFile);
        }
      }
      self.activeDatasetsInitializedInternal().add(canonicalDatasetId);
    }
  }

  static List<Path> listActiveHistoryFiles(Path root) {
    if (root == null || !Files.exists(root)) {
      return List.of();
    }
    try (var stream = Files.walk(root)) {
      return stream
          .filter(Files::isRegularFile)
          .filter(path -> path.getFileName().toString().toLowerCase(Locale.ROOT).endsWith(".parquet"))
          .sorted(Comparator.comparing(path -> path.toAbsolutePath().normalize().toString()))
          .toList();
    } catch (IOException ex) {
      throw new IllegalStateException("Failed to enumerate DUCKDB_ACTIVE history files under " + root, ex);
    }
  }

  static void replayDuckDbActiveHistoryFile(
      RowManagerImpl self,
      DuckdbSession session,
      String canonicalDatasetId,
      String tableName,
      DefinitionDataset.DuckdbBinding binding,
      Path historyFile) {
    String tempTableName = self.nextTempTableName("active_replay_");
    String sqlPath = historyFile.toAbsolutePath().normalize().toString().replace('\\', '/');
    boolean tempCreated = false;
    RuntimeException failure = null;
    try {
      session.exec(
          "create temp table "
              + RowManagerImpl.quoteIdentifier(tempTableName)
              + " as select * from read_parquet('"
              + RowManagerImpl.escapeSqlLiteral(sqlPath)
              + "')");
      tempCreated = true;
      List<RowManagerImpl.SqlColumn> replayColumns =
          sortedSqlColumns(readTableColumns(session, tempTableName, "_fabric_op"));
      if (replayColumns.isEmpty()) {
        return;
      }
      List<String> primaryKey = normalizePrimaryKey(binding.primaryKey(), canonicalDatasetId);
      ensureDuckDbActiveTable(
          self,
          session,
          canonicalDatasetId,
          tableName,
          toAppendColumns(replayColumns),
          binding);
      DuckDbActiveMutation mutation = readReplayMutation(session, tempTableName);
      applyDuckDbActiveMutation(
          session,
          canonicalDatasetId,
          tableName,
          tempTableName,
          toAppendColumns(replayColumns),
          primaryKey,
          mutation);
    } catch (RuntimeException ex) {
      failure = ex;
      throw ex;
    } finally {
      if (tempCreated) {
        try {
          session.exec("drop table if exists " + RowManagerImpl.quoteIdentifier(tempTableName));
        } catch (RuntimeException cleanupEx) {
          if (failure != null) {
            failure.addSuppressed(cleanupEx);
          } else {
            throw cleanupEx;
          }
        }
      }
    }
  }

  static DuckDbActiveMutation readReplayMutation(
      DuckdbSession session, String tempTableName) {
    List<RowManagerImpl.SqlColumn> columns = readTableColumns(session, tempTableName);
    boolean hasOp = columns.stream().anyMatch(column -> "_fabric_op".equalsIgnoreCase(column.name()));
    if (!hasOp) {
      return DuckDbActiveMutation.UPSERT;
    }
    String sql =
        "select coalesce(max(cast("
            + RowManagerImpl.quoteIdentifier("_fabric_op")
            + " as varchar)),'UPSERT') from "
            + RowManagerImpl.quoteIdentifier(tempTableName);
    try (PreparedStatement statement = session.prepareStatement(sql)) {
      boolean hasResult = statement.execute();
      if (!hasResult) {
        return DuckDbActiveMutation.UPSERT;
      }
      try (ResultSet resultSet = statement.getResultSet()) {
        if (!resultSet.next()) {
          return DuckDbActiveMutation.UPSERT;
        }
        String raw = RowManagerImpl.trimToNull(resultSet.getString(1));
        if (raw == null) {
          return DuckDbActiveMutation.UPSERT;
        }
        return DuckDbActiveMutation.valueOf(raw.toUpperCase(Locale.ROOT));
      }
    } catch (SQLException ex) {
      throw new IllegalStateException("Failed to resolve DUCKDB_ACTIVE replay mutation", ex);
    }
  }

  static List<String> normalizePrimaryKey(List<String> primaryKey, String canonicalDatasetId) {
    if (primaryKey == null || primaryKey.isEmpty()) {
      throw new IllegalStateException(
          "DUCKDB_ACTIVE primary key is required for dataset " + canonicalDatasetId);
    }
    List<String> normalized = new ArrayList<>(primaryKey.size());
    for (String key : primaryKey) {
      String candidate = RowManagerImpl.trimToNull(key);
      if (candidate != null && !normalized.contains(candidate)) {
        normalized.add(candidate);
      }
    }
    if (normalized.isEmpty()) {
      throw new IllegalStateException(
          "DUCKDB_ACTIVE primary key is required for dataset " + canonicalDatasetId);
    }
    return List.copyOf(normalized);
  }

  static List<RowManagerImpl.SqlColumn> sortedSqlColumns(List<RowManagerImpl.SqlColumn> columns) {
    List<RowManagerImpl.SqlColumn> sorted = new ArrayList<>(columns);
    sorted.sort(
        Comparator.comparing(
                (RowManagerImpl.SqlColumn column) -> column.name(), String.CASE_INSENSITIVE_ORDER)
            .thenComparing(RowManagerImpl.SqlColumn::name));
    return List.copyOf(sorted);
  }

  static List<RowManagerImpl.AppendColumn> toAppendColumns(
      List<RowManagerImpl.SqlColumn> sqlColumns) {
    List<RowManagerImpl.AppendColumn> result = new ArrayList<>(sqlColumns.size());
    int index = 0;
    for (RowManagerImpl.SqlColumn column : sqlColumns) {
      Field field = new Field(column.name(), FieldType.nullable(new ArrowType.Utf8()), null);
      result.add(new RowManagerImpl.AppendColumn(field, index++, column.duckdbType(), Types.VARCHAR));
    }
    return List.copyOf(result);
  }

  static void validatePrimaryKeyColumns(
      List<String> primaryKey,
      List<RowManagerImpl.AppendColumn> columns,
      String canonicalDatasetId,
      String tableName) {
    List<String> names = columns.stream().map(column -> column.field().getName()).toList();
    for (String key : primaryKey) {
      if (!names.contains(key)) {
        throw new IllegalStateException(
            "DUCKDB_ACTIVE primary key field '"
                + key
                + "' is missing from payload for dataset "
                + canonicalDatasetId
                + " table="
                + tableName);
      }
    }
  }

  static void ensureDuckDbActiveTable(
      RowManagerImpl self,
      DuckdbSession session,
      String canonicalDatasetId,
      String tableName,
      List<RowManagerImpl.AppendColumn> columns,
      DefinitionDataset.DuckdbBinding binding) {
    boolean tableExists = tableExists(session, tableName);
    if (!tableExists) {
      if (Boolean.FALSE.equals(binding.createIfMissing())) {
        throw new IllegalStateException(
            "DUCKDB_ACTIVE table does not exist and createIfMissing=false for dataset "
                + canonicalDatasetId
                + " table="
                + tableName);
      }
      String sql = RowManagerImplSql.buildCreateActiveTableSql(tableName, columns);
      session.exec(sql);
      return;
    }
    Map<String, String> existingTypes = readTableTypeMap(session, tableName);
    for (RowManagerImpl.AppendColumn column : columns) {
      String name = column.field().getName();
      String existingType = existingTypes.get(name.toLowerCase(Locale.ROOT));
      if (existingType == null) {
        session.exec(
            "alter table "
                + RowManagerImpl.quoteIdentifier(tableName)
                + " add column "
                + RowManagerImpl.quoteIdentifier(name)
                + " "
                + column.duckdbType());
        continue;
      }
      if (!normalizeDuckdbType(existingType).equals(normalizeDuckdbType(column.duckdbType()))) {
        throw new IllegalStateException(
            "DUCKDB_ACTIVE schema mismatch for dataset "
                + canonicalDatasetId
                + " table="
                + tableName
                + " column="
                + name
                + " expectedType="
                + existingType
                + " actualType="
                + column.duckdbType());
      }
    }
  }

  static void applyDuckDbActiveMutation(
      DuckdbSession session,
      String canonicalDatasetId,
      String tableName,
      String tempTableName,
      List<RowManagerImpl.AppendColumn> columns,
      List<String> primaryKey,
      DuckDbActiveMutation mutation) {
    switch (mutation) {
      case UPSERT -> {
        boolean hasDeleteMark = tableHasColumn(session, tableName, RowManagerImpl.ACTIVE_DELETE_MARK_COLUMN);
        boolean hasSequence = tableHasColumn(session, tempTableName, "__fabric_seq");
        String sql =
            RowManagerImplSql.buildDuckDbActiveMergeSql(
                tableName,
                tempTableName,
                columns,
                primaryKey,
                hasDeleteMark,
                hasSequence,
                RowManagerImpl.ACTIVE_DELETE_MARK_COLUMN);
        session.exec(sql);
      }
      case DELETE -> {
        String sql = RowManagerImplSql.buildDuckDbActiveDeleteSql(tableName, tempTableName, primaryKey);
        session.exec(sql);
      }
      case MARK_AS_DELETED -> {
        if (!tableHasColumn(session, tableName, RowManagerImpl.ACTIVE_DELETE_MARK_COLUMN)) {
          session.exec(
              "alter table "
                  + RowManagerImpl.quoteIdentifier(tableName)
                  + " add column "
                  + RowManagerImpl.quoteIdentifier(RowManagerImpl.ACTIVE_DELETE_MARK_COLUMN)
                  + " BOOLEAN DEFAULT FALSE");
        }
        String sql =
            RowManagerImplSql.buildDuckDbActiveMarkDeletedSql(
                tableName, tempTableName, primaryKey, RowManagerImpl.ACTIVE_DELETE_MARK_COLUMN);
        session.exec(sql);
      }
    }
  }

  static void writeDuckDbActiveHistory(
      RowManagerImpl self,
      DuckdbSession session,
      String canonicalDatasetId,
      String tableName,
      String tempTableName,
      List<RowManagerImpl.AppendColumn> columns,
      DuckDbActiveMutation mutation,
      DefinitionDataset.DuckdbBinding binding) {
    String historyRootText = RowManagerImpl.trimToNull(binding.parquetHistoryPath());
    if (historyRootText == null) {
      return;
    }
    Path historyRoot = RowManagerImpl.normalizePath(historyRootText).toAbsolutePath().normalize();
    Path writeRoot = historyRoot;
    if (columns.stream().anyMatch(column -> column.field().getType() instanceof ArrowType.Date)) {
      String partition = RowManagerImpl.PARTITION_DATE_FORMAT.format(LocalDate.now(ZoneOffset.UTC));
      writeRoot = historyRoot.resolve("date=" + partition);
    }
    try {
      Files.createDirectories(writeRoot);
    } catch (IOException ex) {
      throw new IllegalStateException(
          "Failed to create DUCKDB_ACTIVE history directory for dataset "
              + canonicalDatasetId
              + ": "
              + writeRoot,
          ex);
    }
    long sequence = self.nextActiveHistoryFileSequence();
    String fileName =
        sanitizeFilename(tableName)
            + "_"
            + RowManagerImpl.FILE_TIMESTAMP_FORMAT.format(Instant.now())
            + "_"
            + String.format(Locale.ROOT, "%08d", sequence)
            + ".parquet";
    Path targetFile = writeRoot.resolve(fileName).toAbsolutePath().normalize();
    String selectColumns =
        columns.stream()
            .map(column -> RowManagerImpl.quoteIdentifier(column.field().getName()))
            .reduce((left, right) -> left + ", " + right)
            .orElse("*");
    String sql =
        "copy (select "
            + selectColumns
            + ", '"
            + mutation.name()
            + "' as "
            + RowManagerImpl.quoteIdentifier("_fabric_op")
            + " from "
            + RowManagerImpl.quoteIdentifier(tempTableName)
            + ") to '"
            + RowManagerImpl.escapeSqlLiteral(targetFile.toString().replace('\\', '/'))
            + "' (format parquet, compression snappy)";
    session.exec(sql);
  }

  static String sanitizeFilename(String value) {
    String input = RowManagerImpl.trimToNull(value);
    if (input == null) {
      return "dataset";
    }
    StringBuilder builder = new StringBuilder(input.length());
    for (int index = 0; index < input.length(); index++) {
      char ch = input.charAt(index);
      if (Character.isLetterOrDigit(ch) || ch == '-' || ch == '_' || ch == '.') {
        builder.append(ch);
      } else {
        builder.append('_');
      }
    }
    String normalized = RowManagerImpl.trimToNull(builder.toString());
    return normalized == null ? "dataset" : normalized;
  }

  static void ensureDuckDbActiveChangeTable(DuckdbSession session, DatasetId datasetId) {
    String sql =
        "create table if not exists "
            + DuckdbActiveChangeLog.quoteIdentifier(DuckdbActiveChangeLog.tableName(datasetId))
            + " (seq BIGINT, ts TIMESTAMP, pk VARCHAR, op VARCHAR)";
    session.exec(sql);
  }

  static long appendDuckDbActiveChangeLog(
      DuckdbSession session,
      DatasetId datasetId,
      String tempTableName,
      List<String> primaryKey,
      DuckDbActiveMutation mutation) {
    ensureDuckDbActiveChangeTable(session, datasetId);
    String changeTable = DuckdbActiveChangeLog.quoteIdentifier(DuckdbActiveChangeLog.tableName(datasetId));
    String pkExpr = DuckdbActiveChangeLog.primaryKeyTextExpression(primaryKey, "src");
    String source =
        "select __fabric_pk, max(__fabric_seq) as __fabric_ord from (select "
            + pkExpr
            + " as __fabric_pk, __fabric_seq from "
            + RowManagerImpl.quoteIdentifier(tempTableName)
            + " as src) as staged where __fabric_pk is not null group by __fabric_pk";
    String sql =
        "insert into "
            + changeTable
            + " (seq, ts, pk, op) select base.base_seq + row_number() over (order by source.__fabric_ord, source.__fabric_pk), current_timestamp, source.__fabric_pk, "
            + DuckdbActiveChangeLog.quoteSqlString(mutation.name())
            + " from ("
            + source
            + ") as source cross join (select coalesce(max(seq), 0) as base_seq from "
            + changeTable
            + ") as base";
    session.exec(sql);
    return queryDuckDbActiveChangeMaxSeq(session, datasetId, datasetId.toString());
  }

  static long queryDuckDbActiveChangeMaxSeq(
      DuckdbSession session, DatasetId datasetId, String canonicalDatasetId) {
    String sql =
        "select coalesce(max(seq), 0) from "
            + DuckdbActiveChangeLog.quoteIdentifier(DuckdbActiveChangeLog.tableName(datasetId));
    try (PreparedStatement statement = session.prepareStatement(sql)) {
      boolean hasResult = statement.execute();
      if (!hasResult) {
        return 0L;
      }
      try (ResultSet resultSet = statement.getResultSet()) {
        if (!resultSet.next()) {
          return 0L;
        }
        return Math.max(0L, resultSet.getLong(1));
      }
    } catch (SQLException ex) {
      throw new IllegalStateException(
          "Failed to query DUCKDB_ACTIVE change max seq for " + canonicalDatasetId, ex);
    }
  }

  static long insertArrowRowsIntoActiveTemp(
      RowManagerImpl self,
      DuckdbSession session,
      String insertSql,
      List<RowManagerImpl.AppendColumn> columns,
      VectorSchemaRoot root,
      ArrowStreamReader reader,
      String canonicalDatasetId) {
    long insertedRows = 0L;
    long sequence = 0L;
    try (PreparedStatement statement = session.prepareStatement(insertSql)) {
      while (reader.loadNextBatch()) {
        int rowCount = root.getRowCount();
        if (rowCount <= 0) {
          continue;
        }
        for (int rowIndex = 0; rowIndex < rowCount; rowIndex++) {
          for (int index = 0; index < columns.size(); index++) {
            RowManagerImpl.AppendColumn column = columns.get(index);
            FieldVector vector = root.getVector(column.vectorIndex());
            if (vector == null || vector.isNull(rowIndex)) {
              statement.setNull(index + 1, column.jdbcNullType());
              continue;
            }
            Object value =
                RowManagerImpl.toJdbcValue(
                    column.field().getType(), vector.getObject(rowIndex), column.field().getName());
            if (value == null) {
              statement.setNull(index + 1, column.jdbcNullType());
            } else if (value instanceof byte[] bytes) {
              statement.setBytes(index + 1, bytes);
            } else if (value instanceof Timestamp timestamp) {
              statement.setTimestamp(index + 1, timestamp);
            } else {
              statement.setObject(index + 1, value);
            }
          }
          statement.setLong(columns.size() + 1, sequence++);
          statement.addBatch();
          insertedRows++;
        }
        statement.executeBatch();
        statement.clearBatch();
      }
    } catch (SQLException | IOException ex) {
      throw new IllegalStateException(
          "Failed to append DUCKDB_ACTIVE rows for dataset " + canonicalDatasetId, ex);
    }
    return insertedRows;
  }

  static boolean tableExists(DuckdbSession session, String tableName) {
    String sql =
        "select count(*) from information_schema.tables where table_schema = current_schema() and lower(table_name) = lower('"
            + RowManagerImpl.escapeSqlLiteral(tableName)
            + "')";
    try (PreparedStatement statement = session.prepareStatement(sql)) {
      boolean hasResult = statement.execute();
      if (!hasResult) {
        return false;
      }
      try (ResultSet resultSet = statement.getResultSet()) {
        if (!resultSet.next()) {
          return false;
        }
        return resultSet.getLong(1) > 0L;
      }
    } catch (SQLException ex) {
      throw new IllegalStateException("Failed to inspect table existence for " + tableName, ex);
    }
  }

  static boolean tableHasColumn(DuckdbSession session, String tableName, String columnName) {
    return readTableTypeMap(session, tableName).containsKey(columnName.toLowerCase(Locale.ROOT));
  }

  static Map<String, String> readTableTypeMap(DuckdbSession session, String tableName) {
    Map<String, String> types = new LinkedHashMap<>();
    for (RowManagerImpl.SqlColumn column : readTableColumns(session, tableName)) {
      types.put(column.name().toLowerCase(Locale.ROOT), column.duckdbType());
    }
    return types;
  }

  static List<RowManagerImpl.SqlColumn> readTableColumns(DuckdbSession session, String tableName) {
    return readTableColumns(session, tableName, null);
  }

  static List<RowManagerImpl.SqlColumn> readTableColumns(
      DuckdbSession session, String tableName, String excludedColumnName) {
    String excluded = RowManagerImpl.trimToNull(excludedColumnName);
    String sql = "pragma table_info('" + RowManagerImpl.escapeSqlLiteral(tableName) + "')";
    List<RowManagerImpl.SqlColumn> columns = new ArrayList<>();
    try (PreparedStatement statement = session.prepareStatement(sql)) {
      boolean hasResult = statement.execute();
      if (!hasResult) {
        return List.of();
      }
      try (ResultSet resultSet = statement.getResultSet()) {
        while (resultSet.next()) {
          String name = RowManagerImpl.trimToNull(resultSet.getString("name"));
          String type = RowManagerImpl.trimToNull(resultSet.getString("type"));
          if (name == null || type == null) {
            continue;
          }
          if (excluded != null && excluded.equalsIgnoreCase(name)) {
            continue;
          }
          columns.add(new RowManagerImpl.SqlColumn(name, type));
        }
        return List.copyOf(columns);
      }
    } catch (SQLException ex) {
      throw new IllegalStateException("Failed to read DuckDB table columns for " + tableName, ex);
    }
  }

  static String normalizeDuckdbType(String type) {
    String normalized = RowManagerImpl.trimToNull(type);
    if (normalized == null) {
      return "";
    }
    return normalized.toUpperCase(Locale.ROOT).replace(" ", "");
  }

  static RowManagerImpl.ActiveDatasetBinding resolveActiveDatasetBinding(
      RowManagerImpl self, DatasetId datasetId) {
    DefinitionDataset definition = self.resolveDatasetDefinition(datasetId);
    DefinitionDataset.DuckdbBinding binding = definition.binding() == null ? null : definition.binding().duckdb();
    if (!self.isDuckDbActive(binding)) {
      throw new IllegalStateException("Dataset is not DUCKDB_ACTIVE: " + datasetId);
    }
    String relationName =
        RowManagerImpl.requireNonBlank(binding.relationName(), "Dataset DUCKDB relationName is required");
    List<String> primaryKey = normalizePrimaryKey(binding.primaryKey(), datasetId.toString());
    return new RowManagerImpl.ActiveDatasetBinding(
        datasetId, datasetId.toString(), relationName, binding, primaryKey);
  }

  static <T> T withActiveDatasetSession(
      RowManagerImpl self,
      RowManagerImpl.ActiveDatasetBinding active,
      Function<DuckdbSession, T> callback) {
    Objects.requireNonNull(active, "active");
    Objects.requireNonNull(callback, "callback");
    if (Boolean.TRUE.equals(active.binding().memory())) {
      synchronized (self.sharedSessionLockInternal()) {
        DuckdbSession session = self.requireSharedSession();
        ensureDuckDbActiveInitialized(
            self, session, active.canonicalDatasetId(), active.relationName(), active.binding());
        return callback.apply(session);
      }
    }
    try (DuckdbSession session = openDuckDbActiveSession(self, active.binding())) {
      return callback.apply(session);
    }
  }
