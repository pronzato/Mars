package org.pronzato.fabric.api.data.manager;

import com.fasterxml.jackson.databind.JsonNode;
import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.util.ArrayList;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Locale;
import java.util.Map;
import org.apache.arrow.memory.BufferAllocator;
import org.apache.arrow.vector.ipc.ArrowStreamReader;
import org.apache.arrow.vector.types.DateUnit;
import org.apache.arrow.vector.types.FloatingPointPrecision;
import org.apache.arrow.vector.types.pojo.ArrowType;
import org.apache.arrow.vector.types.pojo.Field;
import org.apache.arrow.vector.types.pojo.Schema;
import org.pronzato.fabric.api.control.discovery.codec.DiscoveryCodec;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionIngest;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionSchema;
import org.pronzato.fabric.api.util.Strings;

final class IngestManagerImplSchema {
  private IngestManagerImplSchema() {}

  static ParsedExpectedSchema parseExpectedSchema(
      DefinitionSchema expectedSchemaDefinition, String ingestId, String outputSchemaId) {
    String schemaText =
        Strings.trimToNull(expectedSchemaDefinition == null ? null : expectedSchemaDefinition.schema());
    if (schemaText == null) {
      throw new IllegalStateException(
          "Expected schema payload is empty for ingest="
              + ingestId
              + " expectedSchema="
              + outputSchemaId);
    }
    try {
      JsonNode root = DiscoveryCodec.mapper().readTree(schemaText);
      JsonNode fields = root.path("fields");
      if (!fields.isArray()) {
        fields = root.path("columns");
      }
      if (!fields.isArray() || fields.isEmpty()) {
        throw new IllegalStateException(
            "Expected schema has no fields for ingest="
                + ingestId
                + " expectedSchema="
                + outputSchemaId);
      }
      List<ExpectedField> orderedFields = new ArrayList<>();
      Map<String, ExpectedField> byName = new LinkedHashMap<>();
      int index = 0;
      for (JsonNode field : fields) {
        String name = Strings.trimToNull(field.path("name").asText(null));
        if (name == null) {
          throw new IllegalStateException(
              "Expected schema field name is required at index "
                  + index
                  + " for ingest="
                  + ingestId
                  + " expectedSchema="
                  + outputSchemaId);
        }
        String type = expectedTypeLabel(field);
        boolean nullable = field.path("nullable").asBoolean(true);
        if (byName.containsKey(name)) {
          throw new IllegalStateException(
              "Expected schema has duplicate field name '"
                  + name
                  + "' for ingest="
                  + ingestId
                  + " expectedSchema="
                  + outputSchemaId);
        }
        ExpectedField expectedField = new ExpectedField(name, type, nullable);
        orderedFields.add(expectedField);
        byName.put(name, expectedField);
        index++;
      }
      return new ParsedExpectedSchema(List.copyOf(orderedFields), Map.copyOf(byName));
    } catch (IOException ex) {
      throw new IllegalStateException(
          "Expected schema payload is not valid JSON for ingest="
              + ingestId
              + " expectedSchema="
              + outputSchemaId,
          ex);
    }
  }

  static Schema readArrowSchema(
      byte[] arrowIpcStreamBytes, String ingestId, BufferAllocator schemaIpcAllocator) {
    if (arrowIpcStreamBytes == null || arrowIpcStreamBytes.length == 0) {
      throw new IllegalStateException("Arrow IPC stream bytes are required for ingest=" + ingestId);
    }
    try (ArrowStreamReader reader =
        new ArrowStreamReader(new ByteArrayInputStream(arrowIpcStreamBytes), schemaIpcAllocator)) {
      Schema schema = reader.getVectorSchemaRoot().getSchema();
      if (schema == null || schema.getFields() == null || schema.getFields().isEmpty()) {
        throw new IllegalStateException("Arrow IPC schema is empty for ingest=" + ingestId);
      }
      return schema;
    } catch (IOException | RuntimeException ex) {
      throw new IllegalStateException(
          "Failed to decode Arrow IPC schema for ingest=" + ingestId + ": " + ex.getMessage(), ex);
    }
  }

  static SchemaComparison compareSchemas(
      ParsedExpectedSchema expected, Schema actualSchema, DefinitionIngest.SchemaMode schemaMode) {
    List<ActualField> actualOrdered = new ArrayList<>();
    Map<String, ActualField> actualByName = new LinkedHashMap<>();
    for (Field field : actualSchema.getFields()) {
      String name = Strings.trimToNull(field == null ? null : field.getName());
      if (name == null) {
        throw new IllegalStateException("Arrow IPC schema contains blank field name");
      }
      if (actualByName.containsKey(name)) {
        throw new IllegalStateException("Arrow IPC schema contains duplicate field name '" + name + "'");
      }
      ActualField actualField = new ActualField(name, actualTypeLabel(field.getType()), field.isNullable());
      actualOrdered.add(actualField);
      actualByName.put(name, actualField);
    }

    List<String> missing = new ArrayList<>();
    List<String> extra = new ArrayList<>();
    List<String> typeMismatches = new ArrayList<>();
    List<String> nullabilityMismatches = new ArrayList<>();
    List<String> appendOnlyViolations = new ArrayList<>();
    List<String> nonNullableExtraFields = new ArrayList<>();

    for (ExpectedField expectedField : expected.orderedFields()) {
      ActualField actualField = actualByName.get(expectedField.name());
      if (actualField == null) {
        missing.add(expectedField.name());
        continue;
      }
      if (!expectedField.type().equals(actualField.type())) {
        typeMismatches.add(
            expectedField.name()
                + "(expected="
                + expectedField.type()
                + ", actual="
                + actualField.type()
                + ")");
      }
      if (expectedField.nullable() != actualField.nullable()) {
        nullabilityMismatches.add(
            expectedField.name()
                + "(expected="
                + nullableLabel(expectedField.nullable())
                + ", actual="
                + nullableLabel(actualField.nullable())
                + ")");
      }
    }

    for (ActualField actualField : actualOrdered) {
      if (!expected.byName().containsKey(actualField.name())) {
        extra.add(actualField.name());
      }
    }
    if (schemaMode != DefinitionIngest.SchemaMode.STRICT) {
      int expectedCount = expected.orderedFields().size();
      int compared = Math.min(expectedCount, actualOrdered.size());
      for (int i = 0; i < compared; i++) {
        String expectedName = expected.orderedFields().get(i).name();
        String actualName = actualOrdered.get(i).name();
        if (!expectedName.equals(actualName)) {
          appendOnlyViolations.add(
              "index=" + i + "(expected=" + expectedName + ", actual=" + actualName + ")");
        }
      }
      for (int i = expectedCount; i < actualOrdered.size(); i++) {
        ActualField actualField = actualOrdered.get(i);
        if (expected.byName().containsKey(actualField.name())) {
          appendOnlyViolations.add(
              "index=" + i + "(expectedExtraColumn, actualExpectedField=" + actualField.name() + ")");
          continue;
        }
        if (!actualField.nullable()) {
          nonNullableExtraFields.add(actualField.name());
        }
      }
    }

    return new SchemaComparison(
        expected.orderedFields().size(),
        actualOrdered.size(),
        List.copyOf(missing),
        List.copyOf(extra),
        List.copyOf(typeMismatches),
        List.copyOf(nullabilityMismatches),
        List.copyOf(appendOnlyViolations),
        List.copyOf(nonNullableExtraFields),
        schemaMode);
  }

  static String formatSchemaMismatch(
      String ingestId,
      DefinitionIngest.SchemaMode schemaMode,
      String expectedSchemaId,
      Schema actualSchema,
      List<String> missingFields,
      List<String> extraFields,
      List<String> typeMismatches,
      List<String> nullabilityMismatches,
      List<String> appendOnlyViolations,
      List<String> nonNullableExtraFields,
      int expectedFieldCount,
      int actualFieldCount) {
    StringBuilder message = new StringBuilder();
    message
        .append("Ingest schema mismatch ingest=")
        .append(ingestId)
        .append(" mode=")
        .append(schemaMode)
        .append(" expectedSchema=")
        .append(expectedSchemaId)
        .append(" actualSchema=")
        .append(actualSchema)
        .append(" diff: fieldCount(expected=")
        .append(expectedFieldCount)
        .append(", actual=")
        .append(actualFieldCount)
        .append(")");
    if (!missingFields.isEmpty()) {
      message.append(" missingFields=").append(missingFields);
    }
    if (!extraFields.isEmpty()) {
      message.append(" extraFields=").append(extraFields);
    }
    if (!typeMismatches.isEmpty()) {
      message.append(" typeMismatches=").append(typeMismatches);
    }
    if (!nullabilityMismatches.isEmpty()) {
      message.append(" nullabilityMismatches=").append(nullabilityMismatches);
    }
    if (!appendOnlyViolations.isEmpty()) {
      message.append(" appendOnlyViolations=").append(appendOnlyViolations);
    }
    if (!nonNullableExtraFields.isEmpty()) {
      message.append(" nonNullableExtraFields=").append(nonNullableExtraFields);
    }
    return message.toString();
  }

  private static String expectedTypeLabel(JsonNode field) {
    String typeToken = Strings.trimToNull(field.path("type").asText(null));
    if (typeToken == null) {
      throw new IllegalStateException("Expected schema field type is required");
    }
    String normalized = typeToken.toUpperCase(Locale.ROOT).trim();
    if (normalized.startsWith("DECIMAL(")) {
      return normalized.replace(" ", "");
    }
    if ("DECIMAL".equals(normalized) || "NUMERIC".equals(normalized)) {
      Integer precision = field.hasNonNull("precision") ? field.get("precision").asInt() : null;
      Integer scale = field.hasNonNull("scale") ? field.get("scale").asInt() : null;
      if (precision != null || scale != null) {
        return "DECIMAL(" + (precision == null ? 0 : precision) + "," + (scale == null ? 0 : scale) + ")";
      }
      return "DECIMAL";
    }
    return switch (normalized) {
      case "BOOL", "BOOLEAN" -> "BOOL";
      case "STRING", "VARCHAR", "CHAR", "TEXT" -> "STRING";
      case "BINARY", "BYTES", "VARBINARY" -> "BINARY";
      case "INT8", "TINYINT" -> "INT8";
      case "INT16", "SMALLINT" -> "INT16";
      case "INT", "INTEGER", "INT32" -> "INT32";
      case "BIGINT", "LONG", "INT64" -> "INT64";
      case "UINT8" -> "UINT8";
      case "UINT16" -> "UINT16";
      case "UINT32" -> "UINT32";
      case "UINT64" -> "UINT64";
      case "FLOAT16", "HALF" -> "FLOAT16";
      case "FLOAT", "REAL", "FLOAT32" -> "FLOAT32";
      case "DOUBLE", "FLOAT64" -> "FLOAT64";
      case "DATE", "DATE32" -> "DATE32";
      case "DATE64" -> "DATE64";
      case "TIMESTAMP" -> "TIMESTAMP";
      case "TIME", "TIME32" -> "TIME32";
      case "TIME64" -> "TIME64";
      default -> normalized;
    };
  }

  private static String actualTypeLabel(ArrowType type) {
    if (type instanceof ArrowType.Int intType) {
      int width = intType.getBitWidth();
      boolean signed = intType.getIsSigned();
      if (signed) {
        return switch (width) {
          case 8 -> "INT8";
          case 16 -> "INT16";
          case 32 -> "INT32";
          case 64 -> "INT64";
          default -> "INT" + width;
        };
      }
      return switch (width) {
        case 8 -> "UINT8";
        case 16 -> "UINT16";
        case 32 -> "UINT32";
        case 64 -> "UINT64";
        default -> "UINT" + width;
      };
    }
    if (type instanceof ArrowType.FloatingPoint floatingPoint) {
      FloatingPointPrecision precision = floatingPoint.getPrecision();
      return switch (precision) {
        case HALF -> "FLOAT16";
        case SINGLE -> "FLOAT32";
        case DOUBLE -> "FLOAT64";
      };
    }
    if (type instanceof ArrowType.Decimal decimal) {
      return "DECIMAL(" + decimal.getPrecision() + "," + decimal.getScale() + ")";
    }
    if (type instanceof ArrowType.Bool) {
      return "BOOL";
    }
    if (type instanceof ArrowType.Utf8
        || type instanceof ArrowType.LargeUtf8
        || type instanceof ArrowType.Utf8View) {
      return "STRING";
    }
    if (type instanceof ArrowType.Binary
        || type instanceof ArrowType.LargeBinary
        || type instanceof ArrowType.BinaryView
        || type instanceof ArrowType.FixedSizeBinary) {
      return "BINARY";
    }
    if (type instanceof ArrowType.Date dateType) {
      if (dateType.getUnit() == DateUnit.MILLISECOND) {
        return "DATE64";
      }
      return "DATE32";
    }
    if (type instanceof ArrowType.Timestamp) {
      return "TIMESTAMP";
    }
    if (type instanceof ArrowType.Time timeType) {
      return timeType.getBitWidth() <= 32 ? "TIME32" : "TIME64";
    }
    return type == null ? "UNKNOWN" : type.toString().toUpperCase(Locale.ROOT);
  }

  private static String nullableLabel(boolean nullable) {
    return nullable ? "NULLABLE" : "NOT_NULL";
  }

  record ParsedExpectedSchema(List<ExpectedField> orderedFields, Map<String, ExpectedField> byName) {}

  private record ExpectedField(String name, String type, boolean nullable) {}

  private record ActualField(String name, String type, boolean nullable) {}

  record SchemaComparison(
      int expectedFieldCount,
      int actualFieldCount,
      List<String> missingFields,
      List<String> extraFields,
      List<String> typeMismatches,
      List<String> nullabilityMismatches,
      List<String> appendOnlyViolations,
      List<String> nonNullableExtraFields,
      DefinitionIngest.SchemaMode schemaMode) {
    boolean matches() {
      if (schemaMode == DefinitionIngest.SchemaMode.STRICT) {
        return expectedFieldCount == actualFieldCount
            && missingFields.isEmpty()
            && extraFields.isEmpty()
            && typeMismatches.isEmpty()
            && nullabilityMismatches.isEmpty();
      }
      return missingFields.isEmpty()
          && typeMismatches.isEmpty()
          && nullabilityMismatches.isEmpty()
          && appendOnlyViolations.isEmpty()
          && nonNullableExtraFields.isEmpty();
    }
  }
}
