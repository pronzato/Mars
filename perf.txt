  int resolvedServiceHeartbeatTtlSeconds() {
    long heartbeatSeconds = resolvedServiceHeartbeatSeconds();
    long minTtl = heartbeatSeconds * 3L;
    long resolved = Math.max(DEFAULT_TTL_SECONDS, minTtl);
    return (int) Math.min(Integer.MAX_VALUE, resolved);
  }

  @Override
  public ServiceClient client(String name, ServiceRunOptions options) throws Exception {
    DefinitionService service = requireDefinition(name);
    if (isArrowFlightIngestService(service)) {
      throw unsupported("Arrow Flight ingest client");
    }
    if (isArrowFlightSqlService(service)) {
      throw unsupported("Arrow Flight SQL client");
    }
    requireGrpc(service, name);
    requireGrpcRoutingMode(service, "client");
    ManagedChannel channel = createGrpcChannel(service, options);
    String group = resolveGroup(service);
    return new LocalServiceClient(name, group, channel);
  }

  @Override
  public ServiceHandle start(String name, ServiceRunOptions options, Object implementation)
      throws Exception {
    DefinitionService service = requireDefinition(name);
    RunningService running;
    if (isArrowFlightIngestService(service)) {
      running = startArrowFlightIngestServerInternal(service, options, implementation);
    } else if (isArrowFlightSqlService(service)) {
      running = startArrowFlightSqlServerInternal(service, options, implementation);
    } else {
      requireGrpc(service, name);
      requireGrpcRoutingMode(service, "server");
      BindableService bindableService = toBindableService(name, implementation);
      running = startGrpcServer(service, bindableService, options);
    }
    LocalServiceHandle handle = new LocalServiceHandle(running);
    String serviceKey = runningServiceKey(normalizeServiceName(name), running.instanceId());
    LocalServiceHandle previous = runningServices.put(serviceKey, handle);
    if (previous != null) {
      previous.close();
      throw new IllegalStateException(
          "Service already started: " + name + " instance=" + running.instanceId());
    }
    return handle;
  }

  @Override
  public void stop(String name) throws Exception {
    String normalized = normalizeServiceName(name);
    if (normalized == null) {
      return;
    }
    List<String> keysToRemove = new ArrayList<>();
    for (String key : runningServices.keySet()) {
      if (isServiceKeyForName(key, normalized)) {
        keysToRemove.add(key);
      }
    }
    for (String key : keysToRemove) {
      LocalServiceHandle handle = runningServices.remove(key);
      if (handle != null) {
        handle.close();
      }
    }
  }

  @Override
  public void stopAll() {
    List<LocalServiceHandle> snapshot = new ArrayList<>(runningServices.values());
    runningServices.clear();
    for (LocalServiceHandle handle : snapshot) {
      handle.close();
    }
  }

  List<ServerInterceptor> buildServerInterceptors(
      DefinitionService def,
      String serviceId,
      DefinitionService.Grpc grpc,
      ServiceRunOptions options) {
    List<ServerInterceptor> interceptors = new ArrayList<>();
    OtelManager otelManager = sessionOtelManager();
    if (otelManager.enabled() && FabricOtelEnablement.isGrpcServerTracingEnabled(def)) {
      interceptors.add(
          FabricGrpcOtelInterceptors.server(
              otelManager, resolveGroup(def), currentAppName(), resolveServiceName(def)));
    }
    DefinitionService.IdentityMode identityMode = resolveGrpcIdentityMode(grpc);
    interceptors.add(
        new ServerInterceptor() {
          @Override
          public <ReqT, RespT> ServerCall.Listener<ReqT> interceptCall(
              ServerCall<ReqT, RespT> call,
              Metadata headers,
              ServerCallHandler<ReqT, RespT> next) {
            String operationName =
                call.getMethodDescriptor() == null
                    ? null
                    : call.getMethodDescriptor().getFullMethodName();
            Map<String, String> headerSnapshot = snapshotGrpcHeaders(headers);
            RequestIdentity identity;
            try {
              String principal = resolveGrpcAuthenticatedPrincipal(identityMode, headers);
              if (principal == null && identityMode == DefinitionService.IdentityMode.APP) {
                call.close(
                    Status.UNAUTHENTICATED.withDescription(
                        "Missing caller app headers for grpc.identityMode=APP"),
                    new Metadata());
                return new ServerCall.Listener<>() {};
              }
              FabricRequestHeaders requestHeaders = FabricRequestHeaders.fromGrpc(headers, principal);
              identity =
                  RequestIdentityResolver.resolve(
                      requestHeaders, session.discoveryCache(), resolveRuntimeEnvironment());
            } catch (FabricRequestHeaders.ParseException ex) {
              call.close(Status.UNAUTHENTICATED.withDescription(ex.getMessage()), new Metadata());
              return new ServerCall.Listener<>() {};
            } catch (RequestIdentityResolver.ResolutionException ex) {
              call.close(statusFor(ex).withDescription(ex.getMessage()), new Metadata());
              return new ServerCall.Listener<>() {};
            }
            LOGGER.info("[Request] transport=GRPC serviceId={} operation={} requestId={} caller={} user={} delegatedUser={} headers={}", serviceId, Strings.firstNonBlank(operationName, "<unknown>"), identity.requestId(), safeCaller(identity.callerAppSubject()), identity.effectiveUserSubject(), Strings.firstNonBlank(identity.delegatedUserSubject(), "<none>"), headerSnapshot);

            String subject = identity.effectiveUserSubject();
            if (Strings.isNotBlank(subject)) {
              FabricOtelSupport.setCurrentSpanAttributes(
                  Attributes.builder()
                      .put(FabricOtelAttributes.FABRIC_SUBJECT, subject)
                      .build());
            }
            ServiceEntitlements.Decision decision =
                authorizeInvoke(subject, serviceId, operationName);
            if (decision == null || !decision.allowed()) {
              String reason = decisionReason(decision);
              LOGGER.warn("[ServiceManager][gRPC][server-deny] serviceId={} operation={} subject={} reason={}", serviceId, operationName, subject, reason);
              call.close(Status.PERMISSION_DENIED.withDescription(reason), new Metadata());
              return new ServerCall.Listener<>() {};
            }
            ServerCall.Listener<ReqT> delegate =
                RequestIdentityResolver.callAsRequestIdentity(
                    identity, () -> next.startCall(call, headers));
            return scopedServerListener(delegate, identity);
          }
        });
    interceptors.addAll(extractServerInterceptors(options));
    return interceptors;
  }

  List<ClientInterceptor> buildClientInterceptors(
      DefinitionService def, String serviceId, ServiceRunOptions options) {
    List<ClientInterceptor> interceptors = new ArrayList<>();
    OtelManager otelManager = sessionOtelManager();
    if (otelManager.enabled() && FabricOtelEnablement.isGrpcClientTracingEnabled(def)) {
      interceptors.add(
          FabricGrpcOtelInterceptors.client(
              otelManager, resolveGroup(def), currentAppName(), resolveServiceName(def)));
    }
    String appSubject = sessionAppSubject();
    String userSubject = sessionKerberosPrincipal();
    String primarySubject = Strings.firstNonBlank(appSubject, userSubject);
    CallerAppIdentity callerApp = currentCallerApp();
    FabricRequestHeaders.RequestMetadata staticMetadata = currentRequestMetadataTemplate();
    interceptors.add(
        new ClientInterceptor() {
          @Override
          public <ReqT, RespT> ClientCall<ReqT, RespT> interceptCall(
              io.grpc.MethodDescriptor<ReqT, RespT> method,
              io.grpc.CallOptions callOptions,
              io.grpc.Channel next) {
            String operationName = method == null ? null : method.getFullMethodName();
            String normalizedOperation = Strings.firstNonBlank(operationName, "<unknown>");
            ServiceEntitlements.Decision primaryDecision =
                authorizeInvoke(primarySubject, serviceId, operationName);
            ServiceEntitlements.Decision decision = primaryDecision;
            if (shouldFallbackToUserSubject(primaryDecision, primarySubject, appSubject, userSubject)) {
              ServiceEntitlements.Decision fallbackDecision =
                  authorizeInvoke(userSubject, serviceId, operationName);
              decision = fallbackDecision;
              LOGGER.info("[ServiceManager][gRPC][client-auth] serviceId={} operation={} primarySubject={} primaryAllowed={} primaryReason={} fallbackSubject={} fallbackAllowed={} fallbackReason={} finalSubject={} finalAllowed={} finalReason={}", serviceId, normalizedOperation, primarySubject, decisionAllowed(primaryDecision), decisionReason(primaryDecision), userSubject, decisionAllowed(fallbackDecision), decisionReason(fallbackDecision), userSubject, decisionAllowed(decision), decisionReason(decision));
            } else if (decision == null || !decision.allowed()) {
              LOGGER.warn("[ServiceManager][gRPC][client-auth] serviceId={} operation={} subject={} allowed={} reason={}", serviceId, normalizedOperation, primarySubject, decisionAllowed(decision), decisionReason(decision));
            }
            if (decision == null || !decision.allowed()) {
              throw new IllegalArgumentException(decisionReason(decision));
            }
            ClientCall<ReqT, RespT> delegate = next.newCall(method, callOptions);
            return new ForwardingClientCall.SimpleForwardingClientCall<>(delegate) {
              @Override
              public void start(Listener<RespT> responseListener, Metadata headers) {
                Metadata metadata = headers == null ? new Metadata() : headers;
                FabricRequestHeaders requestHeaders =
                    FabricRequestHeaders.outbound(
                        callerApp,
                        null,
                        new FabricRequestHeaders.RequestMetadata(
                            Strings.firstNonBlank(
                                staticMetadata.requestId(), FabricRequestHeaders.generateRequestId()),
                            Strings.firstNonBlank(
                                staticMetadata.traceId(), UUID.randomUUID().toString()),
                            staticMetadata.env(),
                            staticMetadata.region(),
                            staticMetadata.site()));
                requestHeaders.applyToGrpcMetadata(metadata);
                super.start(responseListener, metadata);
              }
            };
          }
        });
    interceptors.addAll(extractClientInterceptors(options));
    return interceptors;
  }

  @SuppressWarnings("unchecked")
  static Map<
          FlightServerMiddleware.Key<?>,
          FlightServerMiddleware.Factory<? extends FlightServerMiddleware>>
      mergeFlightMiddlewares(
          ServiceRunOptions options,
          Map<
                  FlightServerMiddleware.Key<?>,
                  FlightServerMiddleware.Factory<? extends FlightServerMiddleware>>
              defaults) {
    LinkedHashMap<
            FlightServerMiddleware.Key<?>,
            FlightServerMiddleware.Factory<? extends FlightServerMiddleware>>
        merged = new LinkedHashMap<>();
    if (defaults != null && !defaults.isEmpty()) {
      merged.putAll(defaults);
    }
    if (options == null || Strings.isEmpty(options.interceptors())) {
      return Map.copyOf(merged);
    }
    for (Object candidate : options.interceptors()) {
      if (candidate == null) {
        continue;
      }
      if (candidate instanceof Map.Entry<?, ?> entry) {
        Object key = entry.getKey();
        Object value = entry.getValue();
        if (key instanceof FlightServerMiddleware.Key<?> middlewareKey
            && value instanceof FlightServerMiddleware.Factory<?> middlewareFactory) {
          merged.put(
              middlewareKey,
              (FlightServerMiddleware.Factory<? extends FlightServerMiddleware>) middlewareFactory);
          continue;
        }
      }
      throw new IllegalArgumentException(
          "Unsupported Arrow Flight middleware registration type: "
              + candidate.getClass().getName()
              + " (expected Map.Entry<FlightServerMiddleware.Key, FlightServerMiddleware.Factory>)");
    }
    return Map.copyOf(merged);
  }

  private static final class LocalServiceClient implements ServiceClient {
    private final String serviceName;
    private final String group;
    private final ManagedChannel channel;

    private LocalServiceClient(String serviceName, String group, ManagedChannel channel) {
      this.serviceName = serviceName;
      this.group = group;
      this.channel = Objects.requireNonNull(channel, "channel");
    }

    @Override
    public String serviceName() {
      return serviceName;
    }

    @Override
    public String group() {
      return group;
    }

    @Override
    public <T> T as(Class<T> type) {
      Objects.requireNonNull(type, "type");
      if (type.isAssignableFrom(channel.getClass())) {
        return type.cast(channel);
      }
      if (type == ManagedChannel.class || type == Channel.class) {
        return type.cast(channel);
      }
      Class<?> owner = type.getEnclosingClass();
      if (owner != null) {
        for (String factory : STUB_FACTORIES) {
          T stub = invokeFactory(owner, factory, channel, type);
          if (stub != null) {
            return stub;
          }
        }
      }
      for (String factory : STUB_FACTORIES) {
        T stub = invokeFactory(type, factory, channel, type);
        if (stub != null) {
          return stub;
        }
      }
      throw new IllegalArgumentException("Unsupported gRPC stub type: " + type.getName());
    }

    @Override
    public void close() {
      channel.shutdown();
      try {
        if (!channel.awaitTermination(2, TimeUnit.SECONDS)) {
          channel.shutdownNow();
          channel.awaitTermination(1, TimeUnit.SECONDS);
        }
      } catch (InterruptedException ex) {
        Thread.currentThread().interrupt();
        channel.shutdownNow();
      }
    }

    private static <T> T invokeFactory(
        Class<?> factoryOwner, String methodName, Channel channel, Class<T> expectedType) {
      Method method;
      try {
        method = factoryOwner.getMethod(methodName, Channel.class);
      } catch (NoSuchMethodException ex) {
        return null;
      }
      if (!Modifier.isStatic(method.getModifiers())) {
        return null;
      }
      if (!expectedType.isAssignableFrom(method.getReturnType())) {
        return null;
      }
      try {
        Object value = method.invoke(null, channel);
        return expectedType.cast(value);
      } catch (Exception ex) {
        throw new IllegalArgumentException(
            "Unable to create gRPC stub via "
                + factoryOwner.getName()
                + "."
                + methodName
                + "(Channel)",
            ex);
      }
    }
  }
