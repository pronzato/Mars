  @Override
  public void register(StreamModel model) {
    StreamModel normalized = normalizeModel(Objects.requireNonNull(model, "model"));
    StreamModel associated = associateSchema(normalized);
    String group = requireGroup();
    String scoped = scopedName(associated.namespace(), associated.name());
    DefinitionStream existing = requireCache().get(DefinitionStream.class, group, scoped).orElse(null);
    requireCache().upsert(toDefinition(group, associated, existing));
  }

  @Override
  public void delete(String streamName, DeleteOptions options) {
    StreamLookup lookup = parseLookup(streamName);
    String group = requireGroup();
    String scoped = scopedName(lookup.namespace(), lookup.name());
    DefinitionStream definition =
        requireCache().get(DefinitionStream.class, group, scoped)
            .orElseThrow(() -> new IllegalArgumentException("Unknown stream: " + group + "/" + scoped));

    String streamAssetId =
        Strings.firstNonBlank(
            Strings.trimToNull(definition.streamId()),
            "stream:" + group + "/" + lookup.namespace() + "/" + lookup.name());
    DeleteOptions resolved = options == null ? DeleteOptions.streamOnly() : options;
    org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOptions deleteOptions =
        new org.pronzato.fabric.api.control.discovery.lifecycle.DeleteOptions(
            true,
            resolved.deleteSchemaDefinition(),
            false,
            false,
            false,
            false,
            true,
            false,
            List.of());
    stopRuntime(lookup.namespace(), lookup.name());
    DeleteResult result = requireDeleteManager().delete(streamAssetId, deleteOptions);
    if (result.outcome() == DeleteOutcome.BLOCKED) {
      throw new IllegalStateException("Delete blocked for stream " + streamAssetId + ": " + result.blockers());
    }
    if (result.outcome() == DeleteOutcome.PARTIAL) {
      throw new IllegalStateException("Delete partially completed for stream " + streamAssetId + ": " + result.warnings());
    }
  }

  @Override
  public Optional<StreamModel> getModel(String streamName) {
    StreamLookup lookup = parseLookup(streamName);
    String group = requireGroup();
    String scoped = scopedName(lookup.namespace(), lookup.name());
    return requireCache().get(DefinitionStream.class, group, scoped).map(this::fromDefinition);
  }

  @Override
  public List<StreamModel> listModels() {
    String group = requireGroup();
    List<StreamModel> models = new ArrayList<>();
    for (DefinitionStream definition : requireCache().list(DefinitionStream.class, group)) {
      if (definition == null || definition.envelope() == null) {
        continue;
      }
      models.add(fromDefinition(definition));
    }
    models.sort(
        Comparator.comparing(StreamModel::namespace, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(StreamModel::name, String.CASE_INSENSITIVE_ORDER)
            .thenComparing(StreamModel::name));
    return List.copyOf(models);
  }

  @Override
  public Stream start(String streamName, StreamRunOptions options) {
    StreamModel model = getModel(streamName).orElseThrow(() -> new IllegalArgumentException("Unknown stream: " + streamName));
    String key = runtimeKey(model.namespace(), model.name());
    StreamRuntime existing = runtimes.get(key);
    if (existing != null) {
      return existing;
    }
    StreamRunOptions resolved = options == null ? StreamRunOptions.defaults() : options;
    ConnectionOptions connectionOptions =
        ConnectionOptions.builder().withPinnedInstance(Strings.trimToNull(resolved.ampsConnectionInstance())).build();
    AmpsConnection connection = requireConnectionManager().amps(model.connectionName(), connectionOptions);
    StreamRuntime created = new StreamRuntime(model, connection, () -> runtimes.remove(key));
    StreamRuntime prior = runtimes.putIfAbsent(key, created);
    if (prior != null) {
      created.close();
      return prior;
    }
    return created;
  }

  @Override
  public Stream runtime(String streamName) {
    StreamLookup lookup = parseLookup(streamName);
    return runtimes.get(runtimeKey(lookup.namespace(), lookup.name()));
  }

  @Override
  public List<Stream> runtimes() {
    return List.copyOf(runtimes.values());
  }

  @Override
  public void close() {
    List<StreamRuntime> snapshot = List.copyOf(runtimes.values());
    for (StreamRuntime runtime : snapshot) {
      runtime.close();
    }
    runtimes.clear();
  }

  private StreamModel normalizeModel(StreamModel model) {
    String group = requireGroup();
    String name = Strings.requireNonBlank(model.name(), "stream.name");
    String namespace = Strings.requireNonBlank(model.namespace(), "stream.namespace");
    String streamId = Strings.firstNonBlank(model.streamId(), StreamModel.buildStreamId(group, namespace, name));
    StreamModel.Transport transport = model.transport() == null ? StreamModel.Transport.AMPS : model.transport();
    if (transport != StreamModel.Transport.AMPS) {
      throw new IllegalArgumentException("Only AMPS stream transport is supported");
    }
    StreamModel.Serde serde = model.serde() == null ? StreamModel.Serde.JSON : model.serde();
    StreamModel.Address address = Objects.requireNonNull(model.address(), "stream.address");
    String topic = Strings.requireNonBlank(address.topic(), "stream.address.topic");
    String connectionName = Strings.requireNonBlank(model.connectionName(), "stream.connectionName");
    return model.toBuilder()
        .withName(name)
        .withNamespace(namespace)
        .withStreamId(streamId)
        .withTransport(transport)
        .withSerde(serde)
        .withAddress(new StreamModel.Address(topic, address.channel(), address.streamId()))
        .withConnectionName(connectionName)
        .withLifecycleStatus(Strings.firstNonBlank(model.lifecycleStatus(), DEFAULT_LIFECYCLE_STATUS))
        .build();
  }

  private StreamModel associateSchema(StreamModel model) {
    if (model.serde() == StreamModel.Serde.BYTES) {
      return model;
    }
    String group = requireGroup();
    SchemaManager schemas = requireSchemaManager();
    String schemaName = Strings.firstNonBlank(model.schemaName(), parseSchemaNameFromKey(model.schemaKey()), model.name() + "-stream");
    SchemaModel schemaModel = model.schema();
    if (schemaModel != null) {
      SchemaCompatibility compatibility = toSchemaCompatibility(model.compatibilityMode());
      schemaModel =
          schemaModel.toBuilder()
              .withGroup(group)
              .withName(schemaName)
              .withType(SchemaType.TRANSPORT)
              .withCompatibility(compatibility)
              .build();
      schemaModel = schemas.register(schemaName, SchemaType.TRANSPORT, schemaModel);
    } else {
      schemaModel = schemas.get(group, schemaName).orElseGet(() -> schemas.get(schemaName).orElse(null));
    }
    String schemaKey = "schema:" + group + "/" + schemaName;
    String schemaId = resolveSchemaUuid(group, schemaName);
    return model.toBuilder()
        .withSchemaName(schemaName)
        .withSchemaKey(schemaKey)
        .withSchemaId(Strings.firstNonBlank(schemaId, model.schemaId()))
        .withSchema(schemaModel)
        .build();
  }

  private DefinitionStream toDefinition(String group, StreamModel model, DefinitionStream existing) {
    String now = Instant.now().toString();
    String scoped = scopedName(model.namespace(), model.name());
    DefinitionEnvelope previous = existing == null ? null : existing.envelope();
    String createdAt = previous == null ? now : Strings.firstNonBlank(previous.createdTimeUtc(), now);
    String createdBy = previous == null ? SYSTEM_ACTOR : Strings.firstNonBlank(previous.createdBy(), SYSTEM_ACTOR);
    String uuid = Strings.firstNonBlank(model.uuid(), previous == null ? null : previous.uuid(), model.streamId());
    DefinitionEnvelope envelope =
        new DefinitionEnvelope(
            group,
            DefinitionType.STREAM,
            scoped,
            uuid,
            model.enabled(),
            Strings.firstNonBlank(model.lifecycleStatus(), DEFAULT_LIFECYCLE_STATUS),
            createdAt,
            createdBy,
            now,
            SYSTEM_ACTOR);
    List<String> refs = normalizeRefs(model.connectionRefs());
    if (refs.isEmpty()) {
      refs = List.of(model.connectionName());
    }
    return new DefinitionStream(
        envelope,
        model.versionNum(),
        model.versionName(),
        model.namespace(),
        model.streamId(),
        toDefinitionTransport(model.transport()),
        new DefinitionStream.Address(model.address().topic(), model.address().channel(), model.address().streamId()),
        model.schemaId(),
        model.schemaKey(),
        model.messageType(),
        toDefinitionSerde(model.serde()),
        toDefinitionCompatibility(model.compatibilityMode()),
        toDefinitionResolution(model.resolutionPolicy()),
        refs,
        normalizeRefs(model.dependencyRefs()),
        null,
        null);
  }

  private StreamModel fromDefinition(DefinitionStream definition) {
    String group = requireGroup();
    DefinitionEnvelope envelope = definition.envelope();
    String scoped = Strings.requireNonBlank(envelope == null ? null : envelope.name(), "stream.envelope.name");
    IdentityParsingSupport.ScopedName parsed =
        IdentityParsingSupport.parseScopedName(scoped, "stream.namespace", "stream.name");
    String namespace =
        Strings.requireNonBlank(
            Strings.firstNonBlank(definition.namespace(), parsed.namespace()),
            "stream.namespace");
    String name = parsed.name();
    String schemaName = parseSchemaNameFromKey(definition.schemaKey());
    SchemaModel schema = null;
    SchemaManager manager = schemaManagerSupplier.get();
    if (manager != null && schemaName != null) {
      schema = manager.get(group, schemaName).orElseGet(() -> manager.get(schemaName).orElse(null));
    }
    String streamId = Strings.firstNonBlank(definition.streamId(), StreamModel.buildStreamId(group, namespace, name));
    String connectionName =
        Strings.requireNonBlank(resolveConnectionName(definition.connectionRefs()), "stream.connectionName");
    return StreamModel.builder()
        .withName(name)
        .withNamespace(namespace)
        .withUuid(envelope == null ? null : envelope.uuid())
        .withEnabled(envelope == null || envelope.enabled() == null || Boolean.TRUE.equals(envelope.enabled()))
        .withLifecycleStatus(envelope == null ? null : envelope.lifecycleStatus())
        .withVersionNum(definition.versionNum())
        .withVersionName(definition.versionName())
        .withStreamId(streamId)
        .withTransport(fromDefinitionTransport(definition.transport()))
        .withAddress(new StreamModel.Address(definition.address().topic(), definition.address().channel(), definition.address().streamId()))
        .withConnectionName(connectionName)
        .withSchemaName(schemaName)
        .withSchemaId(definition.schemaId())
        .withSchemaKey(definition.schemaKey())
        .withSchema(schema)
        .withMessageType(definition.messageType())
        .withSerde(fromDefinitionSerde(definition.serde()))
        .withCompatibilityMode(fromDefinitionCompatibility(definition.compatibilityMode()))
        .withResolutionPolicy(fromDefinitionResolution(definition.resolutionPolicy()))
        .withConnectionRefs(definition.connectionRefs())
        .withDependencyRefs(definition.dependencyRefs())
        .build();
  }

  private StreamLookup parseLookup(String streamName) {
    String value = Strings.requireNonBlank(streamName, "streamName");
    if (value.startsWith("stream:")) {
      CanonicalIds.ParsedId parsed = CanonicalIds.parse(value);
      if (!parsed.isStream()) {
        throw new IllegalArgumentException("streamName must be a stream id");
      }
      String group = requireGroup();
      if (!group.equalsIgnoreCase(parsed.group())) {
        throw new IllegalArgumentException("Stream group mismatch: " + parsed.group());
      }
      return new StreamLookup(
          IdentityParsingSupport.requireNamespace(parsed.namespace(), "stream.namespace"),
          IdentityParsingSupport.requireName(parsed.name(), "stream.name"));
    }
    String[] parts = value.split("/");
    if (parts.length == 1) {
      throw new IllegalArgumentException(
          "streamName must include namespace (expected namespace/name): " + value);
    }
    if (parts.length == 2) {
      return new StreamLookup(
          IdentityParsingSupport.requireNamespace(parts[0], "stream.namespace"),
          IdentityParsingSupport.requireName(parts[1], "stream.name"));
    }
    if (parts.length == 3 && requireGroup().equalsIgnoreCase(parts[0])) {
      return new StreamLookup(
          IdentityParsingSupport.requireNamespace(parts[1], "stream.namespace"),
          IdentityParsingSupport.requireName(parts[2], "stream.name"));
    }
    throw new IllegalArgumentException("Invalid streamName format: " + value);
  }

  private DataStatusPublisher createDataStatusPublisher(StreamModel model) {
    if (!(session instanceof SessionImpl sessionImpl)) {
      return null;
    }
    if (sessionImpl.controlPlaneConnection() == null || sessionImpl.context() == null) {
      return null;
    }
    SessionContext context = sessionImpl.context();
    String group = Strings.trimToNull(context.getGroup());
    String instance = Strings.trimToNull(context.getInstance());
    if (group == null || instance == null) {
      return null;
    }
    String namespace = IdentityParsingSupport.requireNamespace(model.namespace(), "stream.namespace");
    String statusName = "stream/" + namespace + "/" + model.name();
    String producerId =
        Strings.firstNonBlank(
            Strings.trimToNull(model.streamId()),
            "stream:" + group + "/" + namespace + "/" + model.name());
    Map<String, String> tags = new LinkedHashMap<>();
    tags.put("namespace", namespace);
    tags.put("streamName", model.name());
    tags.put("transport", model.transport() == null ? "UNKNOWN" : model.transport().name());
    tags.put("serde", model.serde() == null ? "UNKNOWN" : model.serde().name());
    tags.put("topic", model.address() == null ? null : model.address().topic());
    tags.put("connectionName", model.connectionName());
    return new DataStatusPublisher(
        sessionImpl.controlPlaneConnection(),
        group,
        statusName,
        instance,
        DefinitionStatusData.ProducerType.STREAM,
        producerId,
        null,
        sessionImpl.runtimePlacementHost(),
        Strings.trimToNull(context.getAppName()),
        Strings.trimToNull(context.getInstance()),
        Strings.trimToNull(context.getGroup()),
        tags,
        sessionImpl.dataStatusCadenceSeconds(),
        sessionImpl.statusScheduler());
  }

  private void stopRuntime(String namespace, String name) {
    StreamRuntime runtime = runtimes.remove(runtimeKey(namespace, name));
    if (runtime != null) {
      runtime.close();
    }
  }

  private static String scopedName(String namespace, String name) {
    return IdentityParsingSupport.requireNamespace(namespace, "stream.namespace")
        + "/"
        + IdentityParsingSupport.requireName(name, "stream.name");
  }

  private static String runtimeKey(String namespace, String name) {
    return IdentityParsingSupport.requireNamespace(namespace, "stream.namespace").toLowerCase(Locale.ROOT)
        + "/"
        + IdentityParsingSupport.requireName(name, "stream.name").toLowerCase(Locale.ROOT);
  }

  private static String parseSchemaNameFromKey(String schemaKey) {
    String normalized = Strings.trimToNull(schemaKey);
    if (normalized == null) {
      return null;
    }
    if (!normalized.startsWith("schema:")) {
      return normalized;
    }
    String payload = normalized.substring("schema:".length());
    int slash = payload.indexOf('/');
    return slash < 0 || slash + 1 >= payload.length() ? payload : payload.substring(slash + 1);
  }

  private String resolveSchemaUuid(String group, String schemaName) {
    return requireCache()
        .get(DefinitionSchema.class, group, schemaName)
        .map(DefinitionSchema::envelope)
        .map(DefinitionEnvelope::uuid)
        .map(Strings::trimToNull)
        .orElse(null);
  }

  private static List<String> normalizeRefs(List<String> refs) {
    if (Strings.isEmpty(refs)) {
      return List.of();
    }
    List<String> normalized = new ArrayList<>();
    for (String ref : refs) {
      String value = Strings.trimToNull(ref);
      if (value != null) {
        normalized.add(value);
      }
    }
    return normalized.isEmpty() ? List.of() : List.copyOf(normalized);
  }

  private static String resolveConnectionName(List<String> refs) {
    if (Strings.isEmpty(refs)) {
      return null;
    }
    String value = Strings.trimToNull(refs.get(0));
    if (value == null) {
      return null;
    }
    if (value.startsWith("connection:")) {
      value = value.substring("connection:".length());
    }
    int slash = value.indexOf('/');
    return slash >= 0 && slash + 1 < value.length() ? value.substring(slash + 1) : value;
  }

  private static SchemaCompatibility toSchemaCompatibility(StreamModel.SchemaCompatibilityMode mode) {
    if (mode == null) {
      return SchemaCompatibility.BACKWARDS;
    }
    return switch (mode) {
      case NONE -> SchemaCompatibility.NONE;
      case FORWARD -> SchemaCompatibility.FORWARD;
      case FULL, TRANSITIVE_FULL -> SchemaCompatibility.FULL;
      case BACKWARD, TRANSITIVE_BACKWARD -> SchemaCompatibility.BACKWARDS;
    };
  }

  private static DefinitionStream.TransportType toDefinitionTransport(StreamModel.Transport value) {
    return DefinitionStream.TransportType.AMPS;
  }

  private static StreamModel.Transport fromDefinitionTransport(DefinitionStream.TransportType value) {
    return StreamModel.Transport.AMPS;
  }

  private static DefinitionStream.SerdeType toDefinitionSerde(StreamModel.Serde value) {
    if (value == null) {
      return DefinitionStream.SerdeType.JSON;
    }
    return switch (value) {
      case NVFIX -> DefinitionStream.SerdeType.NVFIX;
      case JSON -> DefinitionStream.SerdeType.JSON;
      case PROTOBUF -> DefinitionStream.SerdeType.PROTOBUF;
      case BYTES -> DefinitionStream.SerdeType.BYTES;
    };
  }

  private static StreamModel.Serde fromDefinitionSerde(DefinitionStream.SerdeType value) {
    if (value == null) {
      return StreamModel.Serde.JSON;
    }
    return switch (value) {
      case NVFIX -> StreamModel.Serde.NVFIX;
      case JSON, UNKNOWN -> StreamModel.Serde.JSON;
      case PROTOBUF -> StreamModel.Serde.PROTOBUF;
      case BYTES -> StreamModel.Serde.BYTES;
    };
  }

  private static DefinitionStream.SchemaCompatibilityMode toDefinitionCompatibility(StreamModel.SchemaCompatibilityMode value) {
    if (value == null) {
      return DefinitionStream.SchemaCompatibilityMode.BACKWARD;
    }
    return switch (value) {
      case NONE -> DefinitionStream.SchemaCompatibilityMode.NONE;
      case BACKWARD -> DefinitionStream.SchemaCompatibilityMode.BACKWARD;
      case FORWARD -> DefinitionStream.SchemaCompatibilityMode.FORWARD;
      case FULL -> DefinitionStream.SchemaCompatibilityMode.FULL;
      case TRANSITIVE_BACKWARD -> DefinitionStream.SchemaCompatibilityMode.TRANSITIVE_BACKWARD;
      case TRANSITIVE_FULL -> DefinitionStream.SchemaCompatibilityMode.TRANSITIVE_FULL;
    };
  }

  private static StreamModel.SchemaCompatibilityMode fromDefinitionCompatibility(DefinitionStream.SchemaCompatibilityMode value) {
    if (value == null) {
      return StreamModel.SchemaCompatibilityMode.BACKWARD;
    }
    return switch (value) {
      case NONE -> StreamModel.SchemaCompatibilityMode.NONE;
      case BACKWARD, UNKNOWN -> StreamModel.SchemaCompatibilityMode.BACKWARD;
      case FORWARD -> StreamModel.SchemaCompatibilityMode.FORWARD;
      case FULL -> StreamModel.SchemaCompatibilityMode.FULL;
      case TRANSITIVE_BACKWARD -> StreamModel.SchemaCompatibilityMode.TRANSITIVE_BACKWARD;
      case TRANSITIVE_FULL -> StreamModel.SchemaCompatibilityMode.TRANSITIVE_FULL;
    };
  }

  private static DefinitionStream.SchemaResolutionPolicy toDefinitionResolution(StreamModel.SchemaResolutionPolicy value) {
    if (value == null) {
      return DefinitionStream.SchemaResolutionPolicy.COMPATIBLE;
    }
    return switch (value) {
      case EXACT_ID -> DefinitionStream.SchemaResolutionPolicy.EXACT_ID;
      case LATEST -> DefinitionStream.SchemaResolutionPolicy.LATEST;
      case COMPATIBLE -> DefinitionStream.SchemaResolutionPolicy.COMPATIBLE;
    };
  }

  private static StreamModel.SchemaResolutionPolicy fromDefinitionResolution(DefinitionStream.SchemaResolutionPolicy value) {
    if (value == null) {
      return StreamModel.SchemaResolutionPolicy.COMPATIBLE;
    }
    return switch (value) {
      case EXACT_ID -> StreamModel.SchemaResolutionPolicy.EXACT_ID;
      case LATEST -> StreamModel.SchemaResolutionPolicy.LATEST;
      case COMPATIBLE, UNKNOWN -> StreamModel.SchemaResolutionPolicy.COMPATIBLE;
    };
  }

  private static Map<String, byte[]> resolvePublishHeaders(StreamModel model, StreamPublishOptions options) {
    Map<String, byte[]> headers = new LinkedHashMap<>();
    if (options != null && Strings.isNotEmpty(options.headers())) {
      options.headers().forEach(
          (key, value) -> {
            String normalizedKey = Strings.trimToNull(key);
            if (normalizedKey != null && value != null) {
              headers.put(normalizedKey, value.clone());
            }
          });
    }
    String schemaId = Strings.trimToNull(options == null ? null : options.schemaId());
    if (schemaId == null) {
      schemaId = Strings.trimToNull(model.schemaId());
    }
    if (schemaId != null) {
      headers.put(HEADER_SCHEMA_ID, schemaId.getBytes(StandardCharsets.UTF_8));
    }
    String messageType = Strings.trimToNull(options == null ? null : options.messageType());
    if (messageType == null) {
      messageType = Strings.trimToNull(model.messageType());
    }
    if (messageType != null) {
      headers.put(HEADER_MESSAGE_TYPE, messageType.getBytes(StandardCharsets.UTF_8));
    }
    String contentType = Strings.trimToNull(options == null ? null : options.contentType());
    if (contentType == null) {
      contentType = contentTypeFor(model.serde());
    }
    if (contentType != null) {
      headers.put(HEADER_CONTENT_TYPE, contentType.getBytes(StandardCharsets.UTF_8));
    }
    return headers.isEmpty() ? Map.of() : Map.copyOf(headers);
  }

  private static String encodeOptions(Map<String, byte[]> headers) {
    if (Strings.isEmpty(headers)) {
      return null;
    }
    List<String> values = new ArrayList<>();
    for (Map.Entry<String, byte[]> entry : headers.entrySet()) {
      String key = Strings.trimToNull(entry.getKey());
      byte[] value = entry.getValue();
      if (key == null || value == null) {
        continue;
      }
      values.add(key + "=" + new String(value, StandardCharsets.UTF_8));
    }
    return values.isEmpty() ? null : String.join(";", values);
  }

  private static String contentTypeFor(StreamModel.Serde serde) {
    if (serde == null) {
      return null;
    }
    return switch (serde) {
      case NVFIX -> "application/nvfix";
      case JSON -> "application/json";
      case PROTOBUF -> "application/x-protobuf";
      case BYTES -> "application/octet-stream";
    };
  }

  private static byte[] encodePayload(StreamModel model, Object value) {
    if (value instanceof byte[] bytes) {
      return bytes;
    }
    return switch (model.serde()) {
      case NVFIX -> encodeNvfix(toMap(value)).getBytes(StandardCharsets.UTF_8);
      case JSON -> {
        try {
          yield MAPPER.writeValueAsBytes(value);
        } catch (Exception ex) {
          throw new IllegalArgumentException("Failed to encode JSON payload", ex);
        }
      }
      case PROTOBUF -> throw new IllegalArgumentException("PROTOBUF publish requires byte[] payload");
      case BYTES -> throw new IllegalArgumentException("BYTES publish requires byte[] payload");
    };
  }

  private static <T> T decodePayload(StreamModel model, byte[] payload, Class<T> targetType) {
    Objects.requireNonNull(targetType, "targetType");
    byte[] bytes = payload == null ? new byte[0] : payload;
    if (targetType == byte[].class) {
      return targetType.cast(bytes);
    }
    if (targetType == String.class) {
      return targetType.cast(new String(bytes, StandardCharsets.UTF_8));
    }
    return switch (model.serde()) {
      case JSON -> {
        try {
          yield MAPPER.readValue(bytes, targetType);
        } catch (Exception ex) {
          throw new IllegalArgumentException("Failed to decode JSON payload", ex);
        }
      }
      case NVFIX -> {
        Map<String, String> decoded = decodeNvfix(bytes);
        if (targetType == Object.class || Map.class.isAssignableFrom(targetType)) {
          @SuppressWarnings("unchecked")
          T cast = (T) decoded;
          yield cast;
        }
        yield MAPPER.convertValue(decoded, targetType);
      }
      case PROTOBUF, BYTES -> throw new UnsupportedOperationException("Typed decode for " + model.serde() + " supports byte[] or String only");
    };
  }

  private static Map<String, Object> toMap(Object value) {
    if (value instanceof Map<?, ?> map) {
      Map<String, Object> out = new LinkedHashMap<>();
      for (Map.Entry<?, ?> entry : map.entrySet()) {
        String key = Strings.trimToNull(entry.getKey() == null ? null : String.valueOf(entry.getKey()));
        if (key != null && entry.getValue() != null) {
          out.put(key, entry.getValue());
        }
      }
      return out;
    }
    return MAPPER.convertValue(value, Map.class);
  }

  private static String encodeNvfix(Map<String, Object> fields) {
    if (Strings.isEmpty(fields)) {
      throw new IllegalArgumentException("NVFIX payload requires fields");
    }
    StringBuilder builder = new StringBuilder();
    for (Map.Entry<String, Object> entry : fields.entrySet()) {
      String key = Strings.trimToNull(entry.getKey());
      if (key == null || entry.getValue() == null) {
        continue;
      }
      builder.append(key).append('=').append(entry.getValue()).append('\u0001');
    }
    if (builder.isEmpty()) {
      throw new IllegalArgumentException("NVFIX payload requires at least one non-null field");
    }
    return builder.toString();
  }

  private static Map<String, String> decodeNvfix(byte[] payload) {
    Map<String, String> out = new LinkedHashMap<>();
    String text = payload == null ? "" : new String(payload, StandardCharsets.UTF_8);
    for (String token : text.split("\u0001")) {
      String normalized = Strings.trimToNull(token);
      if (normalized == null) {
        continue;
      }
      int equals = normalized.indexOf('=');
      if (equals <= 0 || equals + 1 >= normalized.length()) {
        continue;
      }
      String key = Strings.trimToNull(normalized.substring(0, equals));
      String value = Strings.trimToNull(normalized.substring(equals + 1));
      if (key != null && value != null) {
        out.put(key, value);
      }
    }
    return out;
  }

  private static Map<String, byte[]> parseHeaders(AmpsConnection.AmpsMessage message) {
    String raw = Strings.trimToNull(message == null ? null : message.options());
    if (raw == null) {
      return Map.of();
    }
    Map<String, byte[]> headers = new LinkedHashMap<>();
    for (String token : raw.split(";")) {
      String normalized = Strings.trimToNull(token);
      if (normalized == null) {
        continue;
      }
      int equals = normalized.indexOf('=');
      if (equals <= 0 || equals + 1 >= normalized.length()) {
        continue;
      }
      String key = Strings.trimToNull(normalized.substring(0, equals));
      String value = Strings.trimToNull(normalized.substring(equals + 1));
      if (key != null && value != null) {
        headers.put(key, value.getBytes(StandardCharsets.UTF_8));
      }
    }
    return headers.isEmpty() ? Map.of() : Map.copyOf(headers);
  }

  private static long messageTimestamp(AmpsConnection.AmpsMessage message) {
    Message raw = message == null ? null : message.raw();
    if (raw == null) {
      return System.currentTimeMillis();
    }
    String text = Strings.trimToNull(raw.getTimestamp());
    if (text == null) {
      return System.currentTimeMillis();
    }
    try {
      return Instant.parse(text).toEpochMilli();
    } catch (Exception ignore) {
    }
    try {
      return Long.parseLong(text);
    } catch (Exception ignore) {
    }
    return System.currentTimeMillis();
  }

  private static boolean isBoundary(AmpsConnection.AmpsMessage message) {
    Message raw = message == null ? null : message.raw();
    if (raw == null) {
      return false;
    }
    return raw.getCommand() == Message.Command.GroupBegin || raw.getCommand() == Message.Command.GroupEnd;
  }

  private record StreamLookup(String namespace, String name) {}

  @FunctionalInterface
  private interface MessageConsumer {
    void onMessage(StreamMessage message);
  }

  private final class StreamRuntime implements Stream {
    private final StreamModel model;
    private final AmpsConnection amps;
    private final Runnable onClose;
    private final DataStatusPublisher statusPublisher;
    private final AtomicLong messagesPublished = new AtomicLong(0L);
    private final AtomicLong messagesReceived = new AtomicLong(0L);
    private final CopyOnWriteArrayList<ManagedSubscription> subscriptions = new CopyOnWriteArrayList<>();
    private final AtomicBoolean closed = new AtomicBoolean(false);

    private StreamRuntime(StreamModel model, AmpsConnection amps, Runnable onClose) {
      this.model = model;
      this.amps = amps;
      this.onClose = onClose == null ? () -> {} : onClose;
      this.statusPublisher = createDataStatusPublisher(model);
      if (statusPublisher != null) {
        statusPublisher.start();
        statusPublisher.publishOnline("stream runtime opened");
      }
    }

    @Override
    public StreamPublishResult publish(byte[] payload, StreamPublishOptions options) {
      ensureOpen();
      try {
        String encodedOptions = encodeOptions(resolvePublishHeaders(model, options));
        amps.publish(model.address().topic(), payload == null ? new byte[0] : payload, encodedOptions);
        long published = messagesPublished.incrementAndGet();
        if (statusPublisher != null) {
          statusPublisher.putTag("messagesPublished", Long.toString(published));
        }
        return new StreamPublishResult(model.streamId(), true, model.address().topic());
      } catch (Exception ex) {
        if (statusPublisher != null) {
          statusPublisher.publishError(
              "publish failed: "
                  + Strings.firstNonBlank(
                      Strings.trimToNull(ex.getMessage()), ex.getClass().getSimpleName()));
        }
        throw new IllegalStateException("Failed to publish stream " + model.streamId(), ex);
      }
    }

    @Override
    public StreamPublishResult publish(Object value, StreamPublishOptions options) {
      return publish(encodePayload(model, value), options);
    }

    @Override
    public StreamSubscription subscribeBytes(StreamBytesListener listener) {
      Objects.requireNonNull(listener, "listener");
      return subscribeInternal(StreamSubscribeOptions.defaults(), listener::onMessage);
    }

    @Override
    public <T> StreamSubscription subscribe(Class<T> targetType, StreamListener<T> listener, StreamSubscribeOptions options) {
      Objects.requireNonNull(targetType, "targetType");
      Objects.requireNonNull(listener, "listener");
      StreamSubscribeOptions resolved = options == null ? StreamSubscribeOptions.defaults() : options;
      return subscribeInternal(
          resolved,
          message -> listener.onMessage(decodePayload(model, message.payload(), targetType), message.headers()));
    }

    @Override
    public void close() {
      if (!closed.compareAndSet(false, true)) {
        return;
      }
      for (ManagedSubscription subscription : List.copyOf(subscriptions)) {
        subscription.close();
      }
      subscriptions.clear();
      try {
        amps.close();
      } catch (Exception ignore) {
      }
      if (statusPublisher != null) {
        statusPublisher.putTag("messagesPublished", Long.toString(messagesPublished.get()));
        statusPublisher.putTag("messagesReceived", Long.toString(messagesReceived.get()));
        statusPublisher.close();
      }
      onClose.run();
    }

    private ManagedSubscription subscribeInternal(StreamSubscribeOptions options, MessageConsumer consumer) {
      ensureOpen();
      AmpsConnection.SubscriptionSpec spec =
          new AmpsConnection.SubscriptionSpec(
              options.sowSubscribe(),
              options.sowOnly(),
              model.address().topic(),
              options.options(),
              options.filter(),
              options.bookmarkStart(),
              options.bookmarkFile());
      AtomicBoolean subscriptionClosed = new AtomicBoolean(false);
      try {
        AmpsConnection.Subscription nativeSubscription =
            amps.subscribe(
                spec,
                (AmpsConnection.SubscriptionListener)
                    message -> {
                  if (closed.get() || subscriptionClosed.get() || isBoundary(message)) {
                    return;
                  }
                  byte[] payload = message.dataBytes();
                  Map<String, byte[]> headers = parseHeaders(message);
                  BasicStreamMessage streamMessage =
                      new BasicStreamMessage(
                          model.streamId(),
                          Strings.firstNonBlank(message.topic(), model.address().topic()),
                          messageTimestamp(message),
                          headers,
                          payload == null ? new byte[0] : payload);
                  if (statusPublisher != null) {
                    long received = messagesReceived.incrementAndGet();
                    statusPublisher.putTag("messagesReceived", Long.toString(received));
                  }
                  consumer.onMessage(streamMessage);
                });
        ManagedSubscription managed = new ManagedSubscription(subscriptionClosed, nativeSubscription, () -> subscriptions.removeIf(ManagedSubscription::isClosed));
        subscriptions.add(managed);
        return managed;
      } catch (Exception ex) {
        throw new IllegalStateException("Failed to subscribe stream " + model.streamId(), ex);
      }
    }

    private void ensureOpen() {
      if (closed.get()) {
        throw new IllegalStateException("Stream runtime is closed: " + model.streamId());
      }
    }
  }

  private static final class ManagedSubscription implements StreamSubscription {
    private final AtomicBoolean closed;
    private final AmpsConnection.Subscription delegate;
    private final Runnable onClose;

    private ManagedSubscription(AtomicBoolean closed, AmpsConnection.Subscription delegate, Runnable onClose) {
      this.closed = closed;
      this.delegate = delegate;
      this.onClose = onClose == null ? () -> {} : onClose;
    }

    @Override
    public void close() {
      if (!closed.compareAndSet(false, true)) {
        return;
      }
      try {
        if (delegate != null) {
          delegate.close();
        }
      } finally {
        onClose.run();
      }
    }

    @Override
    public boolean isClosed() {
      return closed.get();
    }
  }

  private static final class BasicStreamMessage implements StreamMessage {
    private final String streamId;
    private final String topic;
    private final long timestampEpochMillis;
    private final Map<String, byte[]> headers;
    private final byte[] payload;

    private BasicStreamMessage(
        String streamId,
        String topic,
        long timestampEpochMillis,
        Map<String, byte[]> headers,
        byte[] payload) {
      this.streamId = streamId;
      this.topic = topic;
      this.timestampEpochMillis = timestampEpochMillis;
      this.headers = headers == null ? Map.of() : Map.copyOf(headers);
      this.payload = payload == null ? new byte[0] : payload.clone();
    }

    @Override
    public String streamId() {
      return streamId;
    }

    @Override
    public String topic() {
      return topic;
    }

    @Override
    public long timestampEpochMillis() {
      return timestampEpochMillis;
    }

    @Override
    public Map<String, byte[]> headers() {
      return headers;
    }

    @Override
    public byte[] payload() {
      return payload.clone();
    }
  }
