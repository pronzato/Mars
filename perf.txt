package org.pronzato.fabric.api.control.entitlements.simple.data;

import java.util.List;
import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import java.util.Locale;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryCache;
import org.pronzato.fabric.api.control.discovery.CanonicalIds;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEntitlement;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEnvelope;
import org.pronzato.fabric.api.control.entitlements.core.EntitlementsAction;
import org.pronzato.fabric.api.control.entitlements.core.EntitlementObligations;
import org.pronzato.fabric.api.control.entitlements.core.EntitlementsScopeType;
import org.pronzato.fabric.api.control.entitlements.core.data.DataEntitlements;
import org.pronzato.fabric.api.control.entitlements.provider.DefaultEntitlementProviderRegistry;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementProviderRegistry;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementSubjectCandidates;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementProviderSelector;
import org.pronzato.fabric.api.control.entitlements.provider.EntitlementSubjectProvider;
import org.pronzato.fabric.api.control.entitlements.provider.ResolveRequest;
import org.pronzato.fabric.api.control.entitlements.provider.SubjectContext;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public final class DataEntitlementsSimple implements DataEntitlements {
  private static final Logger LOGGER = LoggerFactory.getLogger(DataEntitlementsSimple.class);

  // Canonical, deterministic action strings derived from the shared enum.
  // (Works for any enum-based EntitlementsAction.)
  private static final String READ_ACTION =
      EntitlementsAction.READ.name().toLowerCase(Locale.ROOT);
  private static final String WRITE_ACTION =
      EntitlementsAction.WRITE.name().toLowerCase(Locale.ROOT);

  private final DiscoveryCache DiscoveryCache;
  private final String sessionGroup;
  private final EntitlementProviderSelector providerSelector;

  public DataEntitlementsSimple(DiscoveryCache DiscoveryCache, String sessionGroup) {
    this(
        DiscoveryCache,
        sessionGroup,
        new DefaultEntitlementProviderRegistry(DiscoveryCache));
  }

  public DataEntitlementsSimple(
      DiscoveryCache DiscoveryCache, String sessionGroup, EntitlementSubjectProvider provider) {
    this(DiscoveryCache, sessionGroup, new DefaultEntitlementProviderRegistry(provider));
  }

  public DataEntitlementsSimple(
      DiscoveryCache DiscoveryCache, String sessionGroup, EntitlementProviderRegistry providerRegistry) {
    this(
        DiscoveryCache,
        sessionGroup,
        providerRegistry,
        new EntitlementProviderSelector(DiscoveryCache, providerRegistry));
  }

  public DataEntitlementsSimple(
      DiscoveryCache DiscoveryCache,
      String sessionGroup,
      EntitlementProviderRegistry providerRegistry,
      EntitlementProviderSelector providerSelector) {
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.sessionGroup = sessionGroup;
    Objects.requireNonNull(providerRegistry, "providerRegistry");
    this.providerSelector = Objects.requireNonNull(providerSelector, "providerSelector");
  }

  @Override
  public Decision authorizeRead(Identity identity, String datasetCanonicalId) {
    return authorize(identity, datasetCanonicalId, READ_ACTION);
  }

  @Override
  public Decision authorizeWrite(Identity identity, String datasetCanonicalId) {
    return authorize(identity, datasetCanonicalId, WRITE_ACTION);
  }

  private Decision authorize(Identity identity, String datasetCanonicalId, String requestedAction) {
    String datasetId = DataEntitlements.requireDatasetResourceId(datasetCanonicalId);
    String datasetGroup = CanonicalIds.parse(datasetId).group();
    String namespaceId = namespaceResourceId(datasetId);

    String subject = DataEntitlements.subject(identity);

    // Direct subject match only (exact). No normalization here.
    Set<String> subjectCandidates = directSubjectCandidates(subject);

    SubjectContext subjectContext;
    try {
      EntitlementProviderSelector.Selection selectedProvider =
          providerSelector.select(EntitlementsScopeType.DATASET, datasetId);
      subjectContext =
          selectedProvider.provider().resolve(
              new ResolveRequest(
                  identity.subject(),
                  identity.delegatedSubject(),
                  EntitlementsScopeType.DATASET.name(),
                  datasetId,
                  Map.of()));
    } catch (RuntimeException ex) {
      return new Decision(false, "DENIED SUBJECT_PROVIDER_UNAVAILABLE: " + safeReason(ex), null);
    }
    Set<String> membershipCandidates =
        EntitlementSubjectCandidates.membershipSubjectCandidates(subjectContext.subjectId());
    Set<String> groups = subjectContext.effectiveGroups();

    List<DefinitionEntitlement> entitlements = DiscoveryCache.list(DefinitionEntitlement.class);

    // 1) Dataset scope is authoritative if present.
    ScopeEvaluation datasetScope =
        evaluateScope(
            entitlements, datasetId, subjectCandidates, groups, requestedAction, true, datasetGroup);
    if (datasetScope.exists()) {
      if (datasetScope.deny()) {
        boolean namespacePolicyExists = scopeExists(entitlements, namespaceId, false, datasetGroup);
        logDeny(
            subject,
            membershipCandidates.size(),
            datasetId,
            true,
            namespacePolicyExists,
            "explicit_deny");
        return new Decision(false, "DENY matched rules", null);
      }
      if (datasetScope.allow()) {
        return new Decision(true, "ALLOW matched rules", null, datasetScope.obligations());
      }
      boolean namespacePolicyExists = scopeExists(entitlements, namespaceId, false, datasetGroup);
      logDeny(
          subject,
          membershipCandidates.size(),
          datasetId,
          true,
          namespacePolicyExists,
          "no_matching_binding");
      return new Decision(false, "DENIED no matching policy", null);
    }

    // 2) Namespace scope fallback only if dataset scope does not exist.
    ScopeEvaluation namespaceScope =
        evaluateScope(
            entitlements, namespaceId, subjectCandidates, groups, requestedAction, false, datasetGroup);
    if (namespaceScope.exists()) {
      if (namespaceScope.deny()) {
        logDeny(subject, membershipCandidates.size(), datasetId, false, true, "explicit_deny");
        return new Decision(false, "DENY matched rules", null);
      }
      if (namespaceScope.allow()) {
        return new Decision(true, "ALLOW matched rules", null, namespaceScope.obligations());
      }
      logDeny(subject, membershipCandidates.size(), datasetId, false, true, "no_matching_binding");
      return new Decision(false, "DENIED no matching policy", null);
    }

    // 3) No policy -> deny.
    logDeny(subject, membershipCandidates.size(), datasetId, false, false, "no_policy");
    return new Decision(false, "DENIED no matching policy", null);
  }

  private ScopeEvaluation evaluateScope(
      List<DefinitionEntitlement> entitlements,
      String scopeId,
      Set<String> subjectCandidates,
      Set<String> groups,
      String requestedAction,
      boolean datasetScope,
      String datasetGroup) {
    boolean exists = false;
    boolean deny = false;
    boolean allow = false;
    EntitlementObligations allowObligations = EntitlementObligations.empty();
    boolean allowObligationsChosen = false;

    for (DefinitionEntitlement entitlement : entitlements) {
      if (!enabled(entitlement == null || entitlement.envelope() == null
          ? null
          : entitlement.envelope().enabled())) {
        continue;
      }
      if (!isInScope(entitlement == null ? null : entitlement.envelope(), datasetGroup)) {
        continue;
      }

      String canonicalScopeId =
          datasetScope
              ? canonicalDatasetScopeId(entitlement == null ? null : entitlement.scopeId())
              : canonicalNamespaceScopeId(entitlement == null ? null : entitlement.scopeId());

      if (!scopeId.equals(canonicalScopeId)) {
        continue;
      }

      exists = true;

      RuleMatch match =
          matchRules(entitlement.policies(), subjectCandidates, groups, requestedAction);

      deny = deny || match.deny();
      allow = allow || match.allow();
      if (!allowObligationsChosen && match.allow()) {
        allowObligations = match.allowObligations();
        allowObligationsChosen = true;
      }
    }

    return new ScopeEvaluation(exists, allow, deny, allowObligations);
  }

  private boolean scopeExists(
      List<DefinitionEntitlement> entitlements,
      String scopeId,
      boolean datasetScope,
      String datasetGroup) {
    for (DefinitionEntitlement entitlement : entitlements) {
      if (!enabled(entitlement == null || entitlement.envelope() == null
          ? null
          : entitlement.envelope().enabled())) {
        continue;
      }
      if (!isInScope(entitlement == null ? null : entitlement.envelope(), datasetGroup)) {
        continue;
      }
      String canonicalScopeId =
          datasetScope
              ? canonicalDatasetScopeId(entitlement == null ? null : entitlement.scopeId())
              : canonicalNamespaceScopeId(entitlement == null ? null : entitlement.scopeId());
      if (scopeId.equals(canonicalScopeId)) {
        return true;
      }
    }
    return false;
  }

  private RuleMatch matchRules(
      List<DefinitionEntitlement.PolicyDocument> policies,
      Set<String> subjectCandidates,
      Set<String> groups,
      String requestedAction) {
    if (policies == null || policies.isEmpty()) {
      return RuleMatch.NONE;
    }

    boolean deny = false;
    boolean allow = false;
    EntitlementObligations allowObligations = EntitlementObligations.empty();
    boolean allowObligationsChosen = false;

    for (DefinitionEntitlement.PolicyDocument policy : policies) {
      if (policy == null || !enabled(policy.enabled()) || policy.bindings() == null) {
        continue;
      }
      for (DefinitionEntitlement.PolicyBinding binding : policy.bindings()) {
        if (binding == null || !enabled(binding.enabled())) {
          continue;
        }

        String bindingAction = requireSupportedAction(binding.action(), policy.policyId());

        // Action-specific evaluation (read evaluates read only, write evaluates write only)
        if (!Objects.equals(requestedAction, bindingAction)) {
          continue;
        }

        if (!matchesSubject(binding.subject(), subjectCandidates, groups)) {
          continue;
        }

        if ("DENY".equalsIgnoreCase(binding.effect())) {
          deny = true;
        } else if ("ALLOW".equalsIgnoreCase(binding.effect())) {
          allow = true;
          if (!allowObligationsChosen) {
            allowObligations = parseObligations(binding.obligations());
            allowObligationsChosen = true;
          }
        }
      }
    }

    return new RuleMatch(allow, deny, allowObligations);
  }

  private static EntitlementObligations parseObligations(DefinitionEntitlement.ObligationSet obligationSet) {
    if (obligationSet == null) {
      return EntitlementObligations.empty();
    }

    List<String> rowFilters =
        copyRowFilterExpressions(obligationSet.rowFilters());
    List<String> columnsAllow = copyStrings(obligationSet.columnsAllow());
    Map<String, String> columnsMask = copyColumnMasks(obligationSet.columnsMask());

    return new EntitlementObligations(rowFilters, columnsAllow, columnsMask);
  }

  private static List<String> copyRowFilterExpressions(List<DefinitionEntitlement.RowFilter> rowFilters) {
    if (rowFilters == null || rowFilters.isEmpty()) {
      return List.of();
    }
    List<String> copied = new java.util.ArrayList<>(rowFilters.size());
    for (DefinitionEntitlement.RowFilter rowFilter : rowFilters) {
      copied.add(rowFilter == null ? null : rowFilter.expression());
    }
    return copied;
  }

  private static List<String> copyStrings(List<String> values) {
    if (values == null || values.isEmpty()) {
      return List.of();
    }
    return new java.util.ArrayList<>(values);
  }

  private static Map<String, String> copyColumnMasks(List<DefinitionEntitlement.ColumnMask> columnMasks) {
    if (columnMasks == null || columnMasks.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, String> copied = new LinkedHashMap<>();
    for (DefinitionEntitlement.ColumnMask columnMask : columnMasks) {
      if (columnMask == null) {
        copied.put(null, null);
        continue;
      }
      String value = columnMask.replacement();
      if (value == null) {
        value = columnMask.strategy();
      }
      copied.put(columnMask.column(), value);
    }
    return copied;
  }

  private boolean isInScope(DefinitionEnvelope envelope, String datasetGroup) {
    return envelope != null && trimToNull(envelope.group()) != null;
  }

  private static String namespaceResourceId(String datasetResourceId) {
    CanonicalIds.ParsedId parsed = CanonicalIds.parse(datasetResourceId);
    // canonical namespace id
    return EntitlementsScopeType.NAMESPACE.name().toLowerCase(Locale.ROOT)
        + ":"
        + parsed.group()
        + "/"
        + parsed.namespace();
  }

  private static String canonicalDatasetScopeId(String scopeId) {
    if (scopeId == null) {
      return null;
    }
    try {
      return DataEntitlements.requireDatasetResourceId(scopeId);
    } catch (RuntimeException ex) {
      return null;
    }
  }

  private static String canonicalNamespaceScopeId(String scopeId) {
    if (scopeId == null) {
      return null;
    }
    try {
      CanonicalIds.ParsedId parsed = CanonicalIds.parse(scopeId);
      if (!parsed.isNamespace()) {
        return null;
      }
      return parsed.canonical();
    } catch (RuntimeException ex) {
      return null;
    }
  }

  private static void logDeny(
      String subject,
      int membershipCandidatesCount,
      String datasetId,
      boolean datasetPolicyExists,
      boolean namespacePolicyExists,
      String reason) {
    LOGGER.info("[DataEntitlements][DENY] subject={} membershipCandidates={} datasetId={} datasetPolicyExists={} namespacePolicyExists={} reason={}", subject, membershipCandidatesCount, datasetId, datasetPolicyExists, namespacePolicyExists, reason);
  }

  private static Set<String> directSubjectCandidates(String subject) {
    String normalized = trimToNull(subject);
    if (normalized == null) {
      return Set.of();
    }
    return Set.of(normalized);
  }

  private static String requireSupportedAction(String action, String policyId) {
    // Allow only READ/WRITE actions from the canonical EntitlementsAction enum.
    if (Objects.equals(action, READ_ACTION) || Objects.equals(action, WRITE_ACTION)) {
      return action;
    }
    String renderedAction = action == null ? "<null>" : action;
    String renderedPolicyId = policyId == null ? "<null>" : policyId;
    throw new IllegalStateException(
        "Unsupported dataset entitlement action '"
            + renderedAction
            + "' in policy "
            + renderedPolicyId
            + "; allowed actions are 'read' and 'write'");
  }

  private static boolean matchesSubject(String bindingSubject, Set<String> subjectCandidates, Set<String> groups) {
    String normalized = trimToNull(bindingSubject);
    if (normalized == null) {
      return false;
    }

    if (normalized.startsWith("group:")) {
      String normalizedGroup = normalizeGroup(normalized);
      for (String group : groups) {
        if (Objects.equals(normalizedGroup, normalizeGroup(group))) {
          return true;
        }
      }
      return false;
    }

    return subjectCandidates.contains(normalized);
  }

  private static String normalizeGroup(String value) {
    String normalized = trimToNull(value);
    if (normalized == null) {
      return null;
    }
    if (normalized.startsWith("group:") && normalized.length() > "group:".length()) {
      return normalized.substring("group:".length());
    }
    return normalized;
  }

  private static boolean enabled(Boolean value) {
    return value == null || value.booleanValue();
  }

  private static String trimToNull(String value) {
    if (value == null) {
      return null;
    }
    String normalized = value.trim();
    return normalized.isEmpty() ? null : normalized;
  }

  private static String safeReason(RuntimeException error) {
    if (error == null) {
      return "unknown";
    }
    String message = trimToNull(error.getMessage());
    return message == null ? error.getClass().getSimpleName() : message;
  }

  private record ScopeEvaluation(
      boolean exists, boolean allow, boolean deny, EntitlementObligations obligations) {}

  private record RuleMatch(boolean allow, boolean deny, EntitlementObligations allowObligations) {
    private static final RuleMatch NONE =
        new RuleMatch(false, false, EntitlementObligations.empty());
  }
}
