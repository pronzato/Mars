  private static String resolveVaultPath(String relativePath) {
    Path cwd = Path.of("").toAbsolutePath().normalize();
    List<Path> candidates =
        List.of(
            cwd.resolve("vault").resolve(relativePath),
            cwd.resolve("..").resolve("vault").resolve(relativePath),
            cwd.resolve("..").resolve("..").resolve("vault").resolve(relativePath));
    for (Path candidate : candidates) {
      if (Files.exists(candidate)) {
        return candidate.normalize().toString();
      }
    }
    return Path.of("vault").resolve(relativePath).toString();
  }

@Route("")
public class StitchStudioRootView extends Div implements BeforeEnterObserver {
  private static final String FALLBACK_ROUTE = "stitch/reference";
  private final String defaultRoute;

  public StitchStudioRootView(Stitch stitch) {
    this.defaultRoute =
        stitch
            .features()
            .status()
            .flatMap(feature -> feature.screen(StitchStatusFeature.APPS))
            .map(screen -> screen.routePath())
            .orElse(FALLBACK_ROUTE);
  }

  @Override
  public void beforeEnter(BeforeEnterEvent event) {
    event.forwardTo(defaultRoute);
  }
}
