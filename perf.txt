  public ViewManagerImpl(RowManager rowManager, DiscoveryCache DiscoveryCache, String group) {
    this(
        rowManager,
        DiscoveryCache,
        group,
        null,
        CheckpointStores.failFast(
            "ViewManager checkpoint store is not configured. Configure session with withCheckpointPath(...)"));
  }

  public ViewManagerImpl(
      RowManager rowManager, DiscoveryCache DiscoveryCache, String group, Fabric.Session session) {
    this(
        rowManager,
        DiscoveryCache,
        group,
        session,
        CheckpointStores.failFast(
            "ViewManager checkpoint store is not configured. Configure session with withCheckpointPath(...)"));
  }

  public ViewManagerImpl(
      RowManager rowManager,
      DiscoveryCache DiscoveryCache,
      String group,
      Fabric.Session session,
      CheckpointStore checkpointStore) {
    this.rowManager = Objects.requireNonNull(rowManager, "rowManager");
    this.localDataManager = rowManager instanceof RowManagerImpl local ? local : null;
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.group = Strings.requireNonBlank(group, "session group");
    this.schemaManager =
        session == null
            ? new SchemaManagerImpl(() -> this.DiscoveryCache, () -> this.group, () -> null)
            : Objects.requireNonNull(session.schemas(), "session.schemas");
    this.modelMapper = new ViewModelMapper();
    this.graphPlanner = new ViewGraphPlanner();
    this.checkpointStore = Objects.requireNonNull(checkpointStore, "checkpointStore");
    if (session instanceof SessionImpl sessionImpl) {
      this.scheduler = sessionImpl.statusScheduler();
      this.ownsScheduler = false;
    } else {
      this.scheduler =
          Executors.newSingleThreadScheduledExecutor(
              runnable -> {
                Thread thread = new Thread(runnable, "fabric-view-runtime");
                thread.setDaemon(true);
                return thread;
              });
      this.ownsScheduler = true;
    }
  }

  @Override
  public void register(ViewModel model) {
    ViewModel normalized = Objects.requireNonNull(model, "model");
    String viewName = Strings.requireNonBlank(normalized.name(), "model.name");
    String namespace = Strings.firstNonBlank(normalized.namespace(), DEFAULT_NAMESPACE);
    String schemaName = modelMapper.deterministicSchemaName(namespace, viewName);

    SchemaModel schemaToRegister =
        Objects.requireNonNull(normalized.schema(), "model.schema")
            .toBuilder()
            .withName(schemaName)
            .withType(SchemaType.TABULAR)
            .withCompatibility(
                normalized.schemaCompatibility() == null
                    ? SchemaCompatibility.BACKWARDS
                    : normalized.schemaCompatibility())
            .build();
    schemaManager.register(schemaName, SchemaType.TABULAR, schemaToRegister);

    DefinitionDataset definition =
        modelMapper.toDefinitionDataset(normalized, group, DEFAULT_ACTOR, schemaName);
    try (DiscoveryCache.ConfigCacheTx tx =
        DiscoveryCache.beginTx("view:" + group + "/" + namespace + "/" + viewName)) {
      DiscoveryCache.upsertTx(tx, definition);
      NamespaceEntitlementDefaults.ensureDefaultNamespaceAccess(
          DiscoveryCache, tx, definition, DEFAULT_ACTOR);
      DiscoveryCache.commitTx(tx);
    }

    String outputDatasetId = canonicalDatasetId(definition);
    ViewRuntimeImpl runtime = runtimes.remove(outputDatasetId);
    CloseSupport.closeQuietly(runtime);
  }

  @Override
  public Optional<ViewModel> getModel(String viewName) {
    DefinitionDataset definition = resolveViewDefinition(viewName);
    String schemaName = requireSchemaName(definition);
    SchemaModel schema =
        schemaManager
            .get(schemaName)
            .orElseThrow(
                () ->
                    new IllegalStateException(
                        "View schema not found: group="
                            + group
                            + " schemaName="
                            + schemaName
                            + " view="
                            + canonicalDatasetId(definition)));
    return Optional.of(modelMapper.fromDefinitionDataset(definition, schema));
  }

  @Override
  public List<ViewModel> listModels() {
    List<DefinitionDataset> definitions = listViewDefinitions();
    definitions.sort(
        Comparator.comparing(
            definition ->
                Strings.requireNonBlank(
                    definition.envelope() == null ? null : definition.envelope().name(),
                    "view.envelope.name"),
            String.CASE_INSENSITIVE_ORDER));
    List<ViewModel> models = new ArrayList<>(definitions.size());
    for (DefinitionDataset definition : definitions) {
      try {
        String schemaName = requireSchemaName(definition);
        Optional<SchemaModel> schema = schemaManager.get(schemaName);
        if (schema.isEmpty()) {
          LOGGER.warn("[View][list][missing-schema] group={} datasetId={} schemaName={}", group, canonicalDatasetId(definition), schemaName);
          continue;
        }
        models.add(modelMapper.fromDefinitionDataset(definition, schema.get()));
      } catch (RuntimeException ex) {
        LOGGER.warn("[View][list][skip] group={} datasetId={} reason={}", group, safeDatasetId(definition), ex.toString());
      }
    }
    return List.copyOf(models);
  }

  @Override
  public ViewRuntime open(String viewName) {
    DefinitionDataset definition = resolveViewDefinition(viewName);
    String datasetId = canonicalDatasetId(definition);
    return runtimes.computeIfAbsent(datasetId, key -> new ViewRuntimeImpl(key));
  }

  @Override
  public ViewRuntime start(String viewName) {
    ViewRuntime runtime = open(viewName);
    runtime.start();
    return runtime;
  }

  @Override
  public void stop(String viewName) {
    DefinitionDataset definition = resolveViewDefinition(viewName);
    String datasetId = canonicalDatasetId(definition);
    ViewRuntimeImpl runtime = runtimes.get(datasetId);
    if (runtime != null) {
      runtime.stop();
    }
  }

  @Override
  public ViewRefreshResult refresh(String viewName) {
    return open(viewName).refresh();
  }

  @Override
  public ViewRefreshResult refreshGraph(String rootViewName) {
    return open(rootViewName).refreshGraph();
  }

  @Override
  public ViewStatus status(String viewName) {
    return open(viewName).status();
  }

  @Override
  public void close() {
    for (ViewRuntimeImpl runtime : runtimes.values()) {
      CloseSupport.closeQuietly(runtime);
    }
    runtimes.clear();
    if (ownsScheduler) {
      try {
        scheduler.shutdownNow();
      } catch (Exception ignore) {
      }
    }
  }

  private ViewRefreshResult refreshCanonicalDatasetId(String rootDatasetId, boolean includeGraph) {
    DatasetId root = DatasetId.parse(rootDatasetId);
    DefinitionDataset rootDefinition = resolveDatasetDefinition(root);
    if (rootDefinition.view() == null) {
      throw new IllegalStateException("Dataset is not a view: " + rootDatasetId);
    }

    List<String> ordered =
        includeGraph
            ? graphPlanner.plan(rootDatasetId, buildViewDependencyGraph())
            : List.of(rootDatasetId);

    return localDataManager.withDuckdbSession(
        session -> {
          ViewRefreshResult rootResult = null;
          for (String candidateDatasetId : ordered) {
            DefinitionDataset candidate = resolveDatasetDefinition(DatasetId.parse(candidateDatasetId));
            if (candidate.view() == null) {
              continue;
            }
            ViewRefreshResult result = refreshSingleView(candidate, session);
            if (candidateDatasetId.equals(rootDatasetId)) {
              rootResult = result;
            }
          }
          if (rootResult == null) {
            throw new IllegalStateException("View refresh did not include root view: " + rootDatasetId);
          }
          return rootResult;
        });
  }

  private ViewRefreshResult refreshSingleView(DefinitionDataset definition, DuckdbSession session) {
    if (localDataManager == null) {
      throw new IllegalStateException(
          "View refresh requires DIRECT mode local RowManager implementation");
    }

    String datasetId = canonicalDatasetId(definition);
    String viewName = datasetBaseName(definition);
    DataWriteAuthorization.requireWrite(
        localDataManager.dataEntitlementsForSqlPlanning(),
        localDataManager.effectiveIdentityForSqlPlanning(),
        datasetId,
        "VIEW_REFRESH",
        null);
    DefinitionDataset.View view =
        Objects.requireNonNull(definition.view(), "definition.view");
    SchemaModel expectedSchema =
        schemaManager
            .get(requireSchemaName(definition))
            .orElseThrow(
                () ->
                    new IllegalStateException(
                        "View schema not found: group="
                            + group
                            + " datasetId="
                            + datasetId
                            + " schemaName="
                            + view.schemaName()));

    long startedEpochMs = System.currentTimeMillis();
    String startedUtc = Instant.ofEpochMilli(startedEpochMs).toString();
    DefinitionDataset running = withRunningRefresh(definition, startedUtc);
    upsertDataset(running, "view-refresh-running:" + datasetId);
    LOGGER.info("[View] refresh-start view={} mode={} dataset={}", viewName, "DIRECT", datasetId);
    try {
      List<DatasetId> dependencies = parseDependencies(view.dependencies(), datasetId);
      List<DefinitionDataset> dependencyDefs = resolveDependencyDefinitions(dependencies);
      for (int i = 0; i < dependencies.size(); i++) {
        createDependencyAlias(session, dependencies.get(i), dependencyDefs.get(i));
      }

      String querySql = Strings.requireNonBlank(view.querySql(), "view.querySql");
      validateResultSchema(session, querySql, expectedSchema, datasetId);
      materializeViewTarget(session, definition, querySql);
      long rowCount = queryRowCount(session, definition, querySql);

      long durationMs = Math.max(0L, System.currentTimeMillis() - startedEpochMs);
      String endedUtc = Instant.now().toString();
      DefinitionDataset updated =
          withTerminalRefresh(
              running,
              startedUtc,
              endedUtc,
              durationMs,
              "SUCCESS",
              null,
              rowCount,
              buildInputFingerprints(dependencies),
              "schemaHash=" + Strings.firstNonBlank(expectedSchema.getHash(), "n/a"));
      upsertDataset(updated, "view-refresh-success:" + datasetId);
      writeViewCheckpoint(datasetId, "SUCCESS", endedUtc, durationMs, rowCount, null);
      ViewStatus status = statusFromDefinition(updated, ViewRefreshState.STOPPED);
      LOGGER.info("[View] refresh-success view={} rows={} durationMs={}", viewName, rowCount, durationMs);
      return new ViewRefreshResult(viewName, datasetId, true, rowCount, durationMs, null, status);
    } catch (RuntimeException ex) {
      long durationMs = Math.max(0L, System.currentTimeMillis() - startedEpochMs);
      String endedUtc = Instant.now().toString();
      String truncatedError = truncateError(ex.getMessage());
      DefinitionDataset failed =
          withTerminalRefresh(
              running,
              startedUtc,
              endedUtc,
              durationMs,
              "FAILED",
              truncatedError,
              null,
              Map.of(),
              null);
      upsertDataset(failed, "view-refresh-failed:" + datasetId);
      writeViewCheckpoint(datasetId, "FAILED", endedUtc, durationMs, null, truncatedError);
      LOGGER.warn("[View] refresh-fail view={} error={}", viewName, truncatedError);
      throw ex;
    }
  }

  private void createDependencyAlias(
      DuckdbSession session, DatasetId dependencyId, DefinitionDataset dependencyDefinition) {
    localDataManager.prepareDatasetForQuery(dependencyId);
    String alias = quoteIdentifier(dependencyId.name());
    String fromExpression = queryFromExpression(dependencyDefinition);
    session.exec("create or replace temp view " + alias + " as select * from " + fromExpression);
  }

  private String queryFromExpression(DefinitionDataset definition) {
    if (definition.type() == DatasetType.DUCKDB
        && definition.binding() != null
        && definition.binding().duckdb() != null) {
      return Strings.requireNonBlank(
          definition.binding().duckdb().relationName(),
          "dataset.duckdb.relationName");
    }
    if (definition.type() == DatasetType.PARQUET
        && definition.binding() != null
        && definition.binding().parquet() != null) {
      DefinitionDataset.ParquetBinding parquet = definition.binding().parquet();
      String uri = Strings.requireNonBlank(parquet.uri(), "dataset.parquet.uri");
      if (isS3Uri(uri)) {
        throw new IllegalStateException(
            "View refresh in this prompt supports LOCAL parquet only: " + uri);
      }
      String glob = Strings.firstNonBlank(parquet.glob(), DEFAULT_PARQUET_GLOB);
      String base = trimTrailingSlash(Path.of(uri).normalize().toString().replace('\\', '/'));
      String parquetGlob = glob.startsWith("/") ? base + glob : base + "/" + glob;
      StringBuilder sql =
          new StringBuilder("read_parquet('")
              .append(escapeSqlLiteral(parquetGlob))
              .append("'");
      if (Boolean.TRUE.equals(parquet.hivePartitioning())) {
        sql.append(", hive_partitioning=true");
      }
      return sql.append(")").toString();
    }
    throw new IllegalStateException(
        "View dependency dataset is not queryable: " + canonicalDatasetId(definition));
  }

  private void validateResultSchema(
      DuckdbSession session,
      String querySql,
      SchemaModel expectedSchema,
      String datasetId) {
    String sql = "select * from (" + querySql + ") as __fabric_view_schema where 1=0";
    List<SchemaModel.Field> actual = readSchemaFromQuery(session, sql, datasetId);
    List<SchemaModel.Field> expected = expectedSchema.getFields();

    if (expected.size() != actual.size()) {
      throw new IllegalStateException(
          "View schema mismatch dataset="
              + datasetId
              + " reason=FIELD_COUNT expected="
              + expected.size()
              + " actual="
              + actual.size()
              + " expectedFields="
              + summarizeFields(expected)
              + " actualFields="
              + summarizeFields(actual));
    }

    for (int i = 0; i < expected.size(); i++) {
      SchemaModel.Field expectedField = expected.get(i);
      SchemaModel.Field actualField = actual.get(i);
      if (!Objects.equals(expectedField.getName(), actualField.getName())
          || expectedField.getType() != actualField.getType()
          || expectedField.isNullable() != actualField.isNullable()) {
        throw new IllegalStateException(
            "View schema mismatch dataset="
                + datasetId
                + " index="
                + i
                + " expected="
                + formatField(expectedField)
                + " actual="
                + formatField(actualField)
                + " expectedFields="
                + summarizeFields(expected)
                + " actualFields="
                + summarizeFields(actual));
      }
    }
  }

  private static List<SchemaModel.Field> readSchemaFromQuery(
      DuckdbSession session, String sql, String datasetId) {
    try (java.sql.Statement statement = session.createStatement();
        ResultSet resultSet = statement.executeQuery(sql)) {
      ResultSetMetaData metadata = resultSet.getMetaData();
      int count = metadata.getColumnCount();
      List<SchemaModel.Field> fields = new ArrayList<>(count);
      for (int index = 1; index <= count; index++) {
        String name = Strings.requireNonBlank(metadata.getColumnLabel(index), "columnLabel");
        SchemaModel.FieldType type = toSchemaType(metadata.getColumnType(index), metadata.getColumnTypeName(index));
        boolean nullable = metadata.isNullable(index) != ResultSetMetaData.columnNoNulls;
        fields.add(
            SchemaModel.Field.builder().withName(name).withType(type).withNullable(nullable).build());
      }
      return List.copyOf(fields);
    } catch (SQLException ex) {
      throw new IllegalStateException(
          "Failed to inspect view query schema dataset=" + datasetId + " error=" + ex.getMessage(), ex);
    }
  }

  private static SchemaModel.FieldType toSchemaType(int jdbcType, String typeName) {
    return switch (jdbcType) {
      case Types.CHAR,
          Types.VARCHAR,
          Types.LONGVARCHAR,
          Types.NCHAR,
          Types.NVARCHAR,
          Types.LONGNVARCHAR -> SchemaModel.FieldType.STRING;
      case Types.BIT, Types.BOOLEAN -> SchemaModel.FieldType.BOOL;
      case Types.TINYINT, Types.SMALLINT, Types.INTEGER -> SchemaModel.FieldType.INT32;
      case Types.BIGINT -> SchemaModel.FieldType.INT64;
      case Types.REAL,
          Types.FLOAT,
          Types.DOUBLE,
          Types.DECIMAL,
          Types.NUMERIC -> SchemaModel.FieldType.FLOAT64;
      case Types.DATE -> SchemaModel.FieldType.DATE;
      case Types.TIMESTAMP, Types.TIMESTAMP_WITH_TIMEZONE -> SchemaModel.FieldType.TIMESTAMP;
      case Types.BINARY,
          Types.VARBINARY,
          Types.LONGVARBINARY,
          Types.BLOB -> SchemaModel.FieldType.BINARY;
      default ->
          throw new IllegalStateException(
              "Unsupported JDBC type for view schema validation: jdbcType="
                  + jdbcType
                  + " typeName="
                  + Strings.firstNonBlank(typeName, "n/a"));
    };
  }

  private void materializeViewTarget(
      DuckdbSession session, DefinitionDataset definition, String querySql) {
    DefinitionDataset.Target target =
        Objects.requireNonNull(definition.view().target(), "definition.view.target");
    ViewTargetType targetType = parseTargetType(target.targetType());
    String tableName = Strings.requireNonBlank(target.duckdbTableName(), "view.target.duckdbTableName");

    if (targetType == ViewTargetType.DUCKDB_TABLE
        || targetType == ViewTargetType.DUCKDB_TABLE_AND_PARQUET) {
      session.exec("create or replace table " + quoteIdentifier(tableName) + " as " + querySql);
    }

    if (targetType == ViewTargetType.PARQUET_SNAPSHOT
        || targetType == ViewTargetType.DUCKDB_TABLE_AND_PARQUET) {
      Path snapshotFile = resolveSnapshotFile(target.parquetPath());
      try {
        Files.createDirectories(snapshotFile.getParent());
        Files.deleteIfExists(snapshotFile);
      } catch (Exception ex) {
        throw new IllegalStateException(
            "Failed to prepare view parquet snapshot path: " + snapshotFile, ex);
      }
      String sourceSql =
          targetType == ViewTargetType.DUCKDB_TABLE_AND_PARQUET
              ? "select * from " + quoteIdentifier(tableName)
              : "select * from (" + querySql + ") as __fabric_view_snapshot";
      session.exec(
          "copy ("
              + sourceSql
              + ") to '"
              + escapeSqlLiteral(snapshotFile.toString().replace('\\', '/'))
              + "' (format parquet)");
    }
  }

  private static Path resolveSnapshotFile(String parquetPath) {
    String normalized = Strings.requireNonBlank(parquetPath, "view.target.parquetPath");
    Path path = Path.of(normalized).toAbsolutePath().normalize();
    String name = Strings.firstNonBlank(path.getFileName() == null ? null : path.getFileName().toString(), "");
    if (name.toLowerCase(Locale.ROOT).endsWith(".parquet")) {
      return path;
    }
    return path.resolve(SNAPSHOT_FILE_NAME);
  }

  private static long queryRowCount(
      DuckdbSession session, DefinitionDataset definition, String querySql) {
    DefinitionDataset.Target target = definition.view().target();
    ViewTargetType targetType = parseTargetType(target == null ? null : target.targetType());
    String sql;
    if (targetType == ViewTargetType.DUCKDB_TABLE
        || targetType == ViewTargetType.DUCKDB_TABLE_AND_PARQUET) {
      String tableName =
          Strings.requireNonBlank(
              target == null ? null : target.duckdbTableName(), "view.target.duckdbTableName");
      sql = "select count(*) as row_count from " + quoteIdentifier(tableName);
    } else {
      sql = "select count(*) as row_count from (" + querySql + ") as __fabric_view_count";
    }

    try (java.sql.Statement statement = session.createStatement();
        ResultSet resultSet = statement.executeQuery(sql)) {
      if (!resultSet.next()) {
        return 0L;
      }
      return resultSet.getLong(1);
    } catch (SQLException ex) {
      throw new IllegalStateException("Failed to query view row count: " + ex.getMessage(), ex);
    }
  }

  private List<DatasetId> parseDependencies(List<String> dependencies, String datasetId) {
    return ViewManagerImplRefresh.parseDependencies(
        dependencies, datasetId, this::resolveDependencyDatasetId);
  }

  private List<DefinitionDataset> resolveDependencyDefinitions(List<DatasetId> dependencies) {
    if (dependencies == null || dependencies.isEmpty()) {
      return List.of();
    }
    List<DefinitionDataset> resolved = new ArrayList<>(dependencies.size());
    for (DatasetId dependency : dependencies) {
      resolved.add(resolveDatasetDefinition(dependency));
    }
    return List.copyOf(resolved);
  }

  private void upsertDataset(DefinitionDataset definition, String context) {
    try (DiscoveryCache.ConfigCacheTx tx = DiscoveryCache.beginTx(context)) {
      DiscoveryCache.upsertTx(tx, definition);
      DiscoveryCache.commitTx(tx);
    }
  }

  private DefinitionDataset withRunningRefresh(DefinitionDataset definition, String startedUtc) {
    DefinitionDataset.View existingView = Objects.requireNonNull(definition.view(), "definition.view");
    DefinitionDataset.LastRefresh lastRefresh =
        new DefinitionDataset.LastRefresh(
            startedUtc,
            null,
            null,
            "RUNNING",
            null,
            null,
            Map.of(),
            null);
    DefinitionDataset.View updatedView =
        new DefinitionDataset.View(
            existingView.querySql(),
            existingView.dependencies(),
            existingView.refresh(),
            existingView.target(),
            existingView.schemaName(),
            existingView.schemaCompatibility(),
            lastRefresh);
    return new DefinitionDataset(
        updatedEnvelope(definition.envelope(), DEFAULT_ACTOR),
        definition.type(),
        definition.namespace(),
        definition.binding(),
        updatedView);
  }

  private DefinitionDataset withTerminalRefresh(
      DefinitionDataset definition,
      String startedUtc,
      String endedUtc,
      long durationMillis,
      String status,
      String errorMessage,
      Long rowCount,
      Map<String, String> inputFingerprints,
      String outputSnapshot) {
    DefinitionDataset.View existingView = Objects.requireNonNull(definition.view(), "definition.view");
    DefinitionDataset.LastRefresh lastRefresh =
        new DefinitionDataset.LastRefresh(
            startedUtc,
            endedUtc,
            durationMillis,
            Strings.firstNonBlank(Strings.trimToNull(status), "FAILED"),
            Strings.trimToNull(errorMessage),
            rowCount,
            inputFingerprints == null ? Map.of() : inputFingerprints,
            Strings.trimToNull(outputSnapshot));
    DefinitionDataset.View updatedView =
        new DefinitionDataset.View(
            existingView.querySql(),
            existingView.dependencies(),
            existingView.refresh(),
            existingView.target(),
            existingView.schemaName(),
            existingView.schemaCompatibility(),
            lastRefresh);
    return new DefinitionDataset(
        updatedEnvelope(definition.envelope(), DEFAULT_ACTOR),
        definition.type(),
        definition.namespace(),
        definition.binding(),
        updatedView);
  }

  private DatasetId resolveDependencyDatasetId(String dependency, String ownerDatasetId) {
    try {
      return DatasetId.parse(dependency);
    } catch (RuntimeException ignored) {
    }
    try {
      DefinitionDataset resolvedView = resolveViewDefinition(dependency);
      return DatasetId.parse(canonicalDatasetId(resolvedView));
    } catch (RuntimeException ex) {
      throw new IllegalStateException(
          "Invalid view dependency value="
              + dependency
              + " viewDataset="
              + ownerDatasetId
              + " reason="
              + ex.getMessage(),
          ex);
    }
  }

  private Map<String, List<String>> buildViewDependencyGraph() {
    List<DefinitionDataset> definitions = listViewDefinitions();
    definitions.sort(Comparator.comparing(ViewManagerImpl::canonicalDatasetId, String.CASE_INSENSITIVE_ORDER));
    LinkedHashMap<String, List<String>> graph = new LinkedHashMap<>();
    for (DefinitionDataset definition : definitions) {
      String datasetId = canonicalDatasetId(definition);
      List<String> dependencies = new ArrayList<>();
      for (DatasetId dependency : parseDependencies(definition.view().dependencies(), datasetId)) {
        dependencies.add(dependency.toString());
      }
      dependencies.sort(String.CASE_INSENSITIVE_ORDER);
      graph.put(datasetId, List.copyOf(dependencies));
    }
    return Map.copyOf(graph);
  }

  }

  private DefinitionDataset resolveViewDefinition(String viewNameOrDatasetId) {
    String normalized = Strings.requireNonBlank(viewNameOrDatasetId, "viewName");
    if (normalized.regionMatches(true, 0, "dataset:", 0, "dataset:".length())) {
      DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(normalized));
      if (definition.view() == null) {
        throw new IllegalArgumentException("Dataset is not a view: " + normalized);
      }
      return definition;
    }

    List<DefinitionDataset> candidates = new ArrayList<>();
    for (DefinitionDataset definition : listViewDefinitions()) {
      String envelopeName =
          Strings.requireNonBlank(
              definition.envelope() == null ? null : definition.envelope().name(),
              "view.envelope.name");
      if (normalized.contains("/")) {
        if (normalized.equals(envelopeName)) {
          candidates.add(definition);
        }
      } else if (normalized.equals(datasetBaseName(definition))) {
        candidates.add(definition);
      }
    }

    if (candidates.isEmpty()) {
      throw new IllegalArgumentException("Unknown view: " + group + "/" + normalized);
    }
    if (candidates.size() > 1) {
      List<String> ids = new ArrayList<>();
      for (DefinitionDataset candidate : candidates) {
        ids.add(canonicalDatasetId(candidate));
      }
      ids.sort(String.CASE_INSENSITIVE_ORDER);
      throw new IllegalArgumentException(
          "Ambiguous view name '" + normalized + "'. Candidates=" + ids);
    }
    return candidates.get(0);
  }

  private List<DefinitionDataset> listViewDefinitions() {
    List<DefinitionDataset> listed = new ArrayList<>(DiscoveryCache.list(DefinitionDataset.class, group));
    List<DefinitionDataset> filtered = new ArrayList<>();
    for (DefinitionDataset definition : listed) {
      if (definition != null && definition.view() != null) {
        filtered.add(definition);
      }
    }
    return filtered;
  }

  private DefinitionDataset resolveDatasetDefinition(DatasetId datasetId) {
    for (DefinitionDataset definition : DiscoveryCache.list(DefinitionDataset.class, datasetId.group())) {
      if (definition == null) {
        continue;
      }
      DefinitionEnvelope envelope = definition.envelope();
      if (envelope == null) {
        continue;
      }
      if (!datasetId.group().equals(envelope.group())) {
        continue;
      }
      if (!datasetId.namespace().equals(definition.namespace())) {
        continue;
      }
      if (!datasetId.name().equals(datasetBaseName(definition))) {
        continue;
      }
      return definition;
    }
    throw new IllegalStateException("Dataset definition not found: " + datasetId);
  }

  private static String requireSchemaName(DefinitionDataset definition) {
    DefinitionDataset.View view = Objects.requireNonNull(definition.view(), "definition.view");
    return Strings.requireNonBlank(view.schemaName(), "view.schemaName");
  }

  private static ViewStatus statusFromDefinition(DefinitionDataset definition, ViewRefreshState fallbackState) {
    DefinitionDataset.View view = definition == null ? null : definition.view();
    DefinitionDataset.LastRefresh last = view == null ? null : view.lastRefresh();
    return new ViewStatus(
        fallbackState,
        last == null ? null : last.lastRefreshStartUtc(),
        last == null ? null : last.lastRefreshEndUtc(),
        last == null ? null : last.lastDurationMs(),
        last == null ? null : last.lastStatus(),
        last == null ? null : last.lastError(),
        last == null ? null : last.lastRowCount(),
        last == null ? Map.of() : last.lastInputFingerprints(),
        last == null ? null : last.lastOutputSnapshot());
  }

  private static ViewTargetType parseTargetType(String token) {
    String normalized = Strings.trimToNull(token);
    if (normalized == null) {
      return ViewTargetType.DUCKDB_TABLE;
    }
    try {
      return ViewTargetType.valueOf(normalized.toUpperCase(Locale.ROOT));
    } catch (IllegalArgumentException ex) {
      return ViewTargetType.DUCKDB_TABLE;
    }
  }

  private static ViewRefreshType parseRefreshType(String token) {
    String normalized = Strings.trimToNull(token);
    if (normalized == null) {
      return ViewRefreshType.MANUAL;
    }
    try {
      return ViewRefreshType.valueOf(normalized.toUpperCase(Locale.ROOT));
    } catch (IllegalArgumentException ex) {
      return ViewRefreshType.MANUAL;
    }
  }

  private static String canonicalDatasetId(DefinitionDataset definition) {
    DefinitionEnvelope envelope = Objects.requireNonNull(definition.envelope(), "definition.envelope");
    String namespace = Strings.requireNonBlank(definition.namespace(), "definition.namespace");
    String name = datasetBaseName(definition);
    return DatasetId.of(envelope.group(), namespace, name).toString();
  }

  private static String datasetBaseName(DefinitionDataset definition) {
    DefinitionEnvelope envelope = Objects.requireNonNull(definition.envelope(), "definition.envelope");
    String namespace = Strings.requireNonBlank(definition.namespace(), "definition.namespace");
    String envelopeName = Strings.requireNonBlank(envelope.name(), "definition.envelope.name");
    String prefix = namespace + "/";
    if (!envelopeName.startsWith(prefix)) {
      throw new IllegalStateException(
          "Dataset definition invalid envelope.name. Expected namespace/baseName with namespace='"
              + namespace
              + "' and envelope.name='"
              + envelopeName
              + "'");
    }
    String base = envelopeName.substring(prefix.length());
    if (base.isBlank() || base.indexOf('/') >= 0) {
      throw new IllegalStateException(
          "Dataset definition invalid envelope.name. Expected namespace/baseName with namespace='"
              + namespace
              + "' and envelope.name='"
              + envelopeName
              + "'");
    }
    return base;
  }

  private static String safeDatasetId(DefinitionDataset definition) {
    try {
      return definition == null ? "<null>" : canonicalDatasetId(definition);
    } catch (RuntimeException ex) {
      return "<invalid>";
    }
  }

  private static String summarizeFields(List<SchemaModel.Field> fields) {
    List<String> parts = new ArrayList<>();
    for (SchemaModel.Field field : fields) {
      parts.add(formatField(field));
    }
    return parts.toString();
  }

  private static String formatField(SchemaModel.Field field) {
    return field.getName() + ":" + field.getType() + ":" + (field.isNullable() ? "NULLABLE" : "NOT_NULL");
  }

  private static boolean isS3Uri(String value) {
    return value.regionMatches(true, 0, "s3://", 0, "s3://".length())
        || value.regionMatches(true, 0, "s3a://", 0, "s3a://".length());
  }

  private static String trimTrailingSlash(String value) {
    int end = value.length();
    while (end > 0 && value.charAt(end - 1) == '/') {
      end--;
    }
    return end == value.length() ? value : value.substring(0, end);
  }

  private static String quoteIdentifier(String identifier) {
    String normalized = Strings.requireNonBlank(identifier, "identifier");
    return "\"" + normalized.replace("\"", "\"\"") + "\"";
  }

  private static String escapeSqlLiteral(String value) {
    return value.replace("'", "''");
  }


  private final class ViewRuntimeImpl implements ViewRuntime {
    private final Object lifecycleLock = new Object();
    private final String datasetId;
    private volatile ViewStatus status;
    private volatile ScheduledFuture<?> periodicTask;

    private ViewRuntimeImpl(String datasetId) {
      this.datasetId = datasetId;
      DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(datasetId));
      this.status = statusFromDefinition(definition, ViewRefreshState.STOPPED);
    }

    @Override
    public ViewRuntime start() {
      synchronized (lifecycleLock) {
        DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(datasetId));
        ViewRefreshModel refresh = toRefreshModel(definition);
        if (refresh.type() != ViewRefreshType.PERIODIC) {
          return this;
        }
        if (periodicTask != null && !periodicTask.isCancelled() && !periodicTask.isDone()) {
          return this;
        }

        long intervalMillis =
            refresh.intervalMillis() == null || refresh.intervalMillis() <= 0L
                ? 1000L
                : refresh.intervalMillis();
        if (refresh.runOnStart()) {
          refreshGraph();
        }
        periodicTask =
            scheduler.scheduleAtFixedRate(
                () -> {
                  try {
                    refreshGraph();
                  } catch (RuntimeException ex) {
                    synchronized (lifecycleLock) {
                      DefinitionDataset current = resolveDatasetDefinition(DatasetId.parse(datasetId));
                      status = statusFromDefinition(current, ViewRefreshState.FAILED);
                    }
                  }
                },
                intervalMillis,
                intervalMillis,
                TimeUnit.MILLISECONDS);
        status = statusFromDefinition(definition, ViewRefreshState.RUNNING);
        return this;
      }
    }

    @Override
    public void stop() {
      synchronized (lifecycleLock) {
        ScheduledFuture<?> task = periodicTask;
        periodicTask = null;
        if (task != null) {
          task.cancel(false);
        }
        DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(datasetId));
        status = statusFromDefinition(definition, ViewRefreshState.STOPPED);
      }
    }

    @Override
    public ViewRefreshResult refresh() {
      synchronized (lifecycleLock) {
        try {
          ViewRefreshResult result = refreshCanonicalDatasetId(datasetId, false);
          DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(datasetId));
          status =
              statusFromDefinition(
                  definition,
                  periodicTask == null || periodicTask.isCancelled()
                      ? ViewRefreshState.STOPPED
                      : ViewRefreshState.RUNNING);
          return new ViewRefreshResult(
              result.viewName(),
              result.outputDatasetId(),
              result.success(),
              result.rowCount(),
              result.durationMillis(),
              result.message(),
              status);
        } catch (RuntimeException ex) {
          DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(datasetId));
          status = statusFromDefinition(definition, ViewRefreshState.FAILED);
          throw ex;
        }
      }
    }

    @Override
    public ViewRefreshResult refreshGraph() {
      synchronized (lifecycleLock) {
        try {
          ViewRefreshResult result = refreshCanonicalDatasetId(datasetId, true);
          DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(datasetId));
          status =
              statusFromDefinition(
                  definition,
                  periodicTask == null || periodicTask.isCancelled()
                      ? ViewRefreshState.STOPPED
                      : ViewRefreshState.RUNNING);
          return new ViewRefreshResult(
              result.viewName(),
              result.outputDatasetId(),
              result.success(),
              result.rowCount(),
              result.durationMillis(),
              result.message(),
              status);
        } catch (RuntimeException ex) {
          DefinitionDataset definition = resolveDatasetDefinition(DatasetId.parse(datasetId));
          status = statusFromDefinition(definition, ViewRefreshState.FAILED);
          throw ex;
        }
      }
    }

    @Override
    public ViewStatus status() {
      synchronized (lifecycleLock) {
        return status;
      }
    }

    @Override
    public void close() {
      stop();
    }

    private ViewRefreshModel toRefreshModel(DefinitionDataset definition) {
      DefinitionDataset.Refresh refresh =
          definition.view() == null ? null : definition.view().refresh();
      return ViewRefreshModel.builder()
          .withType(parseRefreshType(refresh == null ? null : refresh.type()))
          .withIntervalMillis(refresh == null ? null : refresh.intervalMillis())
          .withRunOnStart(Boolean.TRUE.equals(refresh == null ? null : refresh.runOnStart()))
          .build();
    }
  }
