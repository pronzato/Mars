public final class ConfigSchemaCodec {

  private static final Set<String> ROOT_KEYS =
      Set.of("displayName", "description", "groups", "structs", "fields", "requiredKeys");
  private static final Set<String> GROUP_KEYS = Set.of("id", "label", "description", "order");
  private static final Set<String> STRUCT_KEYS = Set.of("id", "label", "description", "fields");
  private static final Set<String> FIELD_KEYS =
      Set.of(
          "key",
          "label",
          "description",
          "groupId",
          "order",
          "type",
          "required",
          "editable",
          "enum",
          "default",
          "min",
          "max",
          "pattern",
          "fields",
          "itemType",
          "itemsEnum",
          "structRef",
          "minItems",
          "maxItems");

  private static final Comparator<Group> GROUP_ORDER =
      Comparator.comparingInt((Group group) -> group.order() == null ? Integer.MAX_VALUE : group.order())
          .thenComparing(group -> group.id() == null ? "" : group.id());

  private static final Comparator<Struct> STRUCT_ORDER =
      Comparator.comparing((Struct struct) -> struct.id() == null ? "" : struct.id())
          .thenComparing(struct -> struct.label() == null ? "" : struct.label());

  private static final Comparator<Field> FIELD_ORDER =
      Comparator.comparing(
              (Field field) -> isBlank(field.groupId()) ? null : field.groupId(),
              Comparator.nullsLast(String::compareTo))
          .thenComparingInt((Field field) -> field.order() == null ? Integer.MAX_VALUE : field.order())
          .thenComparing(field -> field.key() == null ? "" : field.key());

  private ConfigSchemaCodec() {}

  public static ConfigSchemaDoc parse(String schemaJson) {
    if (schemaJson == null || schemaJson.isBlank()) {
      throw validation("schema text is required", List.of(new ValidationIssue("schema", "schema text is required")));
    }
    try {
      return parse(DiscoveryCodec.mapper().readTree(schemaJson));
    } catch (ValidationException ex) {
      throw ex;
    } catch (Exception ex) {
      throw validation(
          "invalid JSON: " + ex.getMessage(),
          List.of(new ValidationIssue("schema", "invalid JSON: " + ex.getMessage())),
          ex);
    }
  }

  public static ConfigSchemaDoc parse(JsonNode schemaNode) {
    List<ValidationIssue> issues = new ArrayList<>();
    ConfigSchemaDoc doc = parseNode(schemaNode, issues);
    issues.addAll(validate(doc));
    if (!issues.isEmpty()) {
      throw validation(issues.get(0).message(), issues);
    }
    return doc;
  }

  public static List<ValidationIssue> validate(ConfigSchemaDoc doc) {
    List<ValidationIssue> issues = new ArrayList<>();
    if (doc == null) {
      issues.add(new ValidationIssue("schema", "schema is required"));
      return List.copyOf(issues);
    }
    if (isBlank(doc.displayName())) {
      issues.add(new ValidationIssue("schema.displayName", "displayName is required"));
    }

    List<Group> groups = list(doc.groups());
    List<Struct> structs = list(doc.structs());
    List<Field> fields = list(doc.fields());
    List<String> requiredKeys = normalizedStrings(doc.requiredKeys(), "schema.requiredKeys", issues);

    Set<String> groupIds = new HashSet<>();
    for (int i = 0; i < groups.size(); i++) {
      Group group = groups.get(i);
      String path = "schema.groups[" + i + "]";
      if (group == null) {
        issues.add(new ValidationIssue(path, "group is required"));
        continue;
      }
      if (isBlank(group.id())) {
        issues.add(new ValidationIssue(path + ".id", "id is required"));
      } else if (!groupIds.add(group.id())) {
        issues.add(new ValidationIssue(path + ".id", "duplicate group id: " + group.id()));
      }
      if (isBlank(group.label())) {
        issues.add(new ValidationIssue(path + ".label", "label is required"));
      }
      if (group.order() == null) {
        issues.add(new ValidationIssue(path + ".order", "order is required"));
      }
    }

    Map<String, Struct> structById = new LinkedHashMap<>();
    for (int i = 0; i < structs.size(); i++) {
      Struct struct = structs.get(i);
      String path = "schema.structs[" + i + "]";
      if (struct == null) {
        issues.add(new ValidationIssue(path, "struct is required"));
        continue;
      }
      if (isBlank(struct.id())) {
        issues.add(new ValidationIssue(path + ".id", "id is required"));
      } else if (structById.putIfAbsent(struct.id(), struct) != null) {
        issues.add(new ValidationIssue(path + ".id", "duplicate struct id: " + struct.id()));
      }
      if (isBlank(struct.label())) {
        issues.add(new ValidationIssue(path + ".label", "label is required"));
      }
    }

    Set<String> structIds = Set.copyOf(structById.keySet());

    Set<String> topLevelKeys = new HashSet<>();
    validateFieldScope(fields, "schema.fields", groupIds, structIds, topLevelKeys, issues);

    for (int i = 0; i < structs.size(); i++) {
      Struct struct = structs.get(i);
      if (struct == null || isBlank(struct.id())) {
        continue;
      }
      validateFieldScope(struct.fields(), "schema.structs[" + i + "].fields", groupIds, structIds, null, issues);
    }

    Set<String> seenRequired = new HashSet<>();
    for (int i = 0; i < requiredKeys.size(); i++) {
      String key = requiredKeys.get(i);
      String path = "schema.requiredKeys[" + i + "]";
      if (!seenRequired.add(key)) {
        issues.add(new ValidationIssue(path, "duplicate required key: " + key));
      }
      if (!topLevelKeys.contains(key)) {
        issues.add(new ValidationIssue(path, "required key not found in top-level fields: " + key));
      }
    }

    detectCircularStructReferences(structById, issues);
    return List.copyOf(issues);
  }

  public static String toCanonicalJson(ConfigSchemaDoc doc) {
    try {
      return DiscoveryCodec.mapper().writeValueAsString(toCanonicalJsonNode(doc));
    } catch (ValidationException ex) {
      throw ex;
    } catch (Exception ex) {
      throw new IllegalStateException("failed to serialize config schema", ex);
    }
  }

  public static JsonNode toCanonicalJsonNode(ConfigSchemaDoc doc) {
    List<ValidationIssue> issues = validate(doc);
    if (!issues.isEmpty()) {
      throw validation(issues.get(0).message(), issues);
    }
    ObjectMapper mapper = DiscoveryCodec.mapper();
    ObjectNode root = mapper.createObjectNode();
    root.put("displayName", doc.displayName());
    if (!isBlank(doc.description())) {
      root.put("description", doc.description());
    }
    ArrayNode groups = root.putArray("groups");
    sortedGroups(doc.groups()).forEach(group -> groups.add(toCanonicalGroup(group, mapper)));
    List<Struct> structs = sortedStructs(doc.structs());
    if (!structs.isEmpty()) {
      ArrayNode structNodes = root.putArray("structs");
      structs.forEach(struct -> structNodes.add(toCanonicalStruct(struct, mapper)));
    }
    ArrayNode fields = root.putArray("fields");
    sortedFields(doc.fields()).forEach(field -> fields.add(toCanonicalField(field, mapper)));
    List<String> requiredKeys = sortedDistinctStrings(doc.requiredKeys());
    if (!requiredKeys.isEmpty()) {
      ArrayNode requiredNode = root.putArray("requiredKeys");
      requiredKeys.forEach(requiredNode::add);
    }
    return root;
  }

  public static String schemaHashBase64Sha256(String schemaJson) {
    try {
      MessageDigest digest = MessageDigest.getInstance("SHA-256");
      byte[] hashed = digest.digest(schemaJson == null ? new byte[0] : schemaJson.getBytes(StandardCharsets.UTF_8));
      return Base64.getEncoder().encodeToString(hashed);
    } catch (Exception ex) {
      throw new IllegalStateException("failed to compute schema hash", ex);
    }
  }

  private static ConfigSchemaDoc parseNode(JsonNode node, List<ValidationIssue> issues) {
    if (node == null || !node.isObject()) {
      issues.add(new ValidationIssue("schema", "schema must be a JSON object"));
      return new ConfigSchemaDoc(null, null, List.of(), List.of(), List.of(), List.of());
    }

    unknownKeys(node, ROOT_KEYS, "schema", issues);
    String displayName = requiredText(node, "displayName", "schema.displayName", issues);
    String description = optionalText(node, "description", "schema.description", issues);
    List<Group> groups = parseGroups(node.get("groups"), "schema.groups", issues);
    List<Struct> structs = parseStructs(node.get("structs"), "schema.structs", issues);
    List<Field> fields = parseFields(node.get("fields"), "schema.fields", issues);
    List<String> requiredKeys = stringArray(node.get("requiredKeys"), "schema.requiredKeys", false, issues);
    return new ConfigSchemaDoc(displayName, description, groups, structs, fields, requiredKeys);
  }

  private static List<Group> parseGroups(JsonNode node, String path, List<ValidationIssue> issues) {
    if (node == null || node.isNull()) {
      issues.add(new ValidationIssue(path, "groups is required"));
      return List.of();
    }
    if (!node.isArray()) {
      issues.add(new ValidationIssue(path, "groups must be an array"));
      return List.of();
    }
    List<Group> groups = new ArrayList<>();
    for (int i = 0; i < node.size(); i++) {
      JsonNode row = node.get(i);
      String rowPath = path + "[" + i + "]";
      if (row == null || !row.isObject()) {
        issues.add(new ValidationIssue(rowPath, "group must be a JSON object"));
        continue;
      }
      unknownKeys(row, GROUP_KEYS, rowPath, issues);
      groups.add(
          new Group(
              requiredText(row, "id", rowPath + ".id", issues),
              requiredText(row, "label", rowPath + ".label", issues),
              optionalText(row, "description", rowPath + ".description", issues),
              requiredInteger(row, "order", rowPath + ".order", issues)));
    }
    return List.copyOf(groups);
  }

  private static List<Struct> parseStructs(JsonNode node, String path, List<ValidationIssue> issues) {
    if (node == null || node.isNull() || node.isMissingNode()) {
      return List.of();
    }
    if (!node.isArray()) {
      issues.add(new ValidationIssue(path, "structs must be an array"));
      return List.of();
    }
    List<Struct> structs = new ArrayList<>();
    for (int i = 0; i < node.size(); i++) {
      JsonNode row = node.get(i);
      String rowPath = path + "[" + i + "]";
      if (row == null || !row.isObject()) {
        issues.add(new ValidationIssue(rowPath, "struct must be a JSON object"));
        continue;
      }
      unknownKeys(row, STRUCT_KEYS, rowPath, issues);
      structs.add(
          new Struct(
              requiredText(row, "id", rowPath + ".id", issues),
              requiredText(row, "label", rowPath + ".label", issues),
              optionalText(row, "description", rowPath + ".description", issues),
              parseFields(row.get("fields"), rowPath + ".fields", issues)));
    }
    return List.copyOf(structs);
  }

  private static List<Field> parseFields(JsonNode node, String path, List<ValidationIssue> issues) {
    if (node == null || node.isNull()) {
      issues.add(new ValidationIssue(path, "fields is required"));
      return List.of();
    }
    if (!node.isArray()) {
      issues.add(new ValidationIssue(path, "fields must be an array"));
      return List.of();
    }
    List<Field> fields = new ArrayList<>();
    for (int i = 0; i < node.size(); i++) {
      fields.add(parseField(node.get(i), path + "[" + i + "]", issues));
    }
    return List.copyOf(fields);
  }

  private static Field parseField(JsonNode node, String path, List<ValidationIssue> issues) {
    if (node == null || !node.isObject()) {
      issues.add(new ValidationIssue(path, "field must be a JSON object"));
      return emptyField();
    }
    unknownKeys(node, FIELD_KEYS, path, issues);
    return new Field(
        requiredText(node, "key", path + ".key", issues),
        requiredText(node, "label", path + ".label", issues),
        optionalText(node, "description", path + ".description", issues),
        optionalText(node, "groupId", path + ".groupId", issues),
        requiredInteger(node, "order", path + ".order", issues),
        requiredType(node, "type", path + ".type", issues, true),
        requiredBoolean(node, "required", path + ".required", issues),
        optionalBooleanDefaultTrue(node, "editable", path + ".editable", issues),
        stringArray(node.get("enum"), path + ".enum", false, issues),
        node.has("default") ? copy(node.get("default")) : null,
        optionalNumber(node, "min", path + ".min", issues),
        optionalNumber(node, "max", path + ".max", issues),
        optionalText(node, "pattern", path + ".pattern", issues),
        nestedFields(node.get("fields"), path + ".fields", issues),
        requiredType(node, "itemType", path + ".itemType", issues, false),
        stringArray(node.get("itemsEnum"), path + ".itemsEnum", false, issues),
        optionalText(node, "structRef", path + ".structRef", issues),
        optionalInteger(node, "minItems", path + ".minItems", issues),
        optionalInteger(node, "maxItems", path + ".maxItems", issues));
  }

  private static List<Field> nestedFields(JsonNode node, String path, List<ValidationIssue> issues) {
    if (node == null || node.isNull() || node.isMissingNode()) {
      return List.of();
    }
    if (!node.isArray()) {
      issues.add(new ValidationIssue(path, "must be an array"));
      return List.of();
    }
    List<Field> fields = new ArrayList<>();
    for (int i = 0; i < node.size(); i++) {
      fields.add(parseField(node.get(i), path + "[" + i + "]", issues));
    }
    return List.copyOf(fields);
  }

  private static void validateFieldScope(
      List<Field> fields,
      String scopePath,
      Set<String> groupIds,
      Set<String> structIds,
      Set<String> keysOut,
      List<ValidationIssue> issues) {
    Set<String> keysInScope = new HashSet<>();
    for (int i = 0; i < list(fields).size(); i++) {
      Field field = list(fields).get(i);
      String path = scopePath + "[" + i + "]";
      if (field == null) {
        issues.add(new ValidationIssue(path, "field is required"));
        continue;
      }
      validateField(field, path, groupIds, structIds, issues);
      if (isBlank(field.key())) {
        continue;
      }
      if (!keysInScope.add(field.key())) {
        issues.add(new ValidationIssue(path + ".key", "duplicate key in scope: " + field.key()));
      }
      if (keysOut != null) {
        keysOut.add(field.key());
      }

      if (field.type() == FieldType.STRUCT && !list(field.fields()).isEmpty()) {
        validateFieldScope(field.fields(), path + ".fields", groupIds, structIds, null, issues);
      }
      if (field.type() == FieldType.ARRAY && field.itemType() == FieldType.STRUCT && !list(field.fields()).isEmpty()) {
        validateFieldScope(field.fields(), path + ".fields", groupIds, structIds, null, issues);
      }
    }
  }

  private static void validateField(
      Field field,
      String path,
      Set<String> groupIds,
      Set<String> structIds,
      List<ValidationIssue> issues) {
    if (isBlank(field.key())) {
      issues.add(new ValidationIssue(path + ".key", "key is required"));
    }
    if (isBlank(field.label())) {
      issues.add(new ValidationIssue(path + ".label", "label is required"));
    }
    if (field.order() == null) {
      issues.add(new ValidationIssue(path + ".order", "order is required"));
    }
    if (field.type() == null) {
      issues.add(new ValidationIssue(path + ".type", "type is required"));
      return;
    }
    if (field.required() == null) {
      issues.add(new ValidationIssue(path + ".required", "required is required"));
    }
    if (field.editable() == null) {
      issues.add(new ValidationIssue(path + ".editable", "editable is required"));
    }
    if (!isBlank(field.groupId()) && !groupIds.contains(field.groupId())) {
      issues.add(new ValidationIssue(path + ".groupId", "unknown groupId: " + field.groupId()));
    }
    if (field.min() != null && field.max() != null && field.min().compareTo(field.max()) > 0) {
      issues.add(new ValidationIssue(path, "min cannot be greater than max"));
    }
    if (field.minItems() != null && field.maxItems() != null && field.minItems() > field.maxItems()) {
      issues.add(new ValidationIssue(path, "minItems cannot be greater than maxItems"));
    }

    boolean hasStructRef = !isBlank(field.structRef());
    boolean hasInlineFields = !list(field.fields()).isEmpty();

    switch (field.type()) {
      case STRING -> {
        rejectPresent(field.min(), path + ".min", "min is only allowed for INTEGER/NUMBER", issues);
        rejectPresent(field.max(), path + ".max", "max is only allowed for INTEGER/NUMBER", issues);
        if (field.defaultValue() != null && !field.defaultValue().isTextual()) {
          issues.add(new ValidationIssue(path + ".default", "default must be a string"));
        }
        if (!field.enumValues().isEmpty()
            && field.defaultValue() != null
            && !field.enumValues().contains(field.defaultValue().asText())) {
          issues.add(new ValidationIssue(path + ".default", "default must be one of enum values"));
        }
      }
      case INTEGER -> {
        rejectPresent(field.enumValues(), path + ".enum", "enum is only allowed for STRING", issues);
        rejectPresent(field.pattern(), path + ".pattern", "pattern is only allowed for STRING", issues);
        if (field.defaultValue() != null && !field.defaultValue().isIntegralNumber()) {
          issues.add(new ValidationIssue(path + ".default", "default must be an integer"));
        }
      }
      case NUMBER -> {
        rejectPresent(field.enumValues(), path + ".enum", "enum is only allowed for STRING", issues);
        rejectPresent(field.pattern(), path + ".pattern", "pattern is only allowed for STRING", issues);
        if (field.defaultValue() != null && !field.defaultValue().isNumber()) {
          issues.add(new ValidationIssue(path + ".default", "default must be numeric"));
        }
      }
      case BOOLEAN -> {
        rejectPresent(field.enumValues(), path + ".enum", "enum is only allowed for STRING", issues);
        rejectPresent(field.pattern(), path + ".pattern", "pattern is only allowed for STRING", issues);
        rejectPresent(field.min(), path + ".min", "min is only allowed for INTEGER/NUMBER", issues);
        rejectPresent(field.max(), path + ".max", "max is only allowed for INTEGER/NUMBER", issues);
        if (field.defaultValue() != null && !field.defaultValue().isBoolean()) {
          issues.add(new ValidationIssue(path + ".default", "default must be boolean"));
        }
      }
      case STRUCT -> {
        rejectPresent(field.enumValues(), path + ".enum", "enum is only allowed for STRING", issues);
        rejectPresent(field.pattern(), path + ".pattern", "pattern is only allowed for STRING", issues);
        rejectPresent(field.min(), path + ".min", "min is only allowed for INTEGER/NUMBER", issues);
        rejectPresent(field.max(), path + ".max", "max is only allowed for INTEGER/NUMBER", issues);
        rejectPresent(field.itemType(), path + ".itemType", "itemType is only allowed for ARRAY", issues);
        rejectPresent(field.itemsEnum(), path + ".itemsEnum", "itemsEnum is only allowed for ARRAY<STRING>", issues);
        rejectPresent(field.minItems(), path + ".minItems", "minItems is only allowed for ARRAY", issues);
        rejectPresent(field.maxItems(), path + ".maxItems", "maxItems is only allowed for ARRAY", issues);
        if (hasStructRef && hasInlineFields) {
          issues.add(new ValidationIssue(path, "Cannot define both structRef and fields on the same field"));
        }
        if (!hasStructRef && !hasInlineFields) {
          issues.add(new ValidationIssue(path, "STRUCT must define either structRef or fields"));
        }
        if (hasStructRef && !structIds.contains(field.structRef().trim())) {
          issues.add(new ValidationIssue(path + ".structRef", "unknown structRef: " + field.structRef()));
        }
        if (field.defaultValue() != null && !field.defaultValue().isObject()) {
          issues.add(new ValidationIssue(path + ".default", "default must be an object"));
        }
      }
      case ARRAY -> {
        rejectPresent(field.enumValues(), path + ".enum", "enum is only allowed for STRING", issues);
        rejectPresent(field.pattern(), path + ".pattern", "pattern is only allowed for STRING", issues);
        rejectPresent(field.min(), path + ".min", "min is only allowed for INTEGER/NUMBER", issues);
        rejectPresent(field.max(), path + ".max", "max is only allowed for INTEGER/NUMBER", issues);

        FieldType itemType = field.itemType();
        if (itemType == null) {
          issues.add(new ValidationIssue(path + ".itemType", "itemType is required for ARRAY"));
        } else {
          if (itemType == FieldType.ARRAY) {
            issues.add(new ValidationIssue(path + ".itemType", "ARRAY itemType cannot be ARRAY"));
          }
          if (!list(field.itemsEnum()).isEmpty() && itemType != FieldType.STRING) {
            issues.add(new ValidationIssue(path + ".itemsEnum", "itemsEnum is only allowed when itemType=STRING"));
          }
          if (itemType == FieldType.STRUCT) {
            if (hasStructRef && hasInlineFields) {
              issues.add(new ValidationIssue(path, "Cannot define both structRef and fields on the same field"));
            }
            if (!hasStructRef && !hasInlineFields) {
              issues.add(new ValidationIssue(path, "ARRAY with itemType STRUCT must define either structRef or fields"));
            }
            if (hasStructRef && !structIds.contains(field.structRef().trim())) {
              issues.add(new ValidationIssue(path + ".structRef", "unknown structRef: " + field.structRef()));
            }
          } else {
            rejectPresent(field.structRef(), path + ".structRef", "structRef is only allowed when itemType=STRUCT", issues);
            rejectPresent(field.fields(), path + ".fields", "fields is only allowed when itemType=STRUCT", issues);
          }
        }
        validateArrayDefault(field, path + ".default", issues);
      }
    }

    if (field.type() != FieldType.STRING) {
      rejectPresent(field.pattern(), path + ".pattern", "pattern is only allowed for STRING", issues);
    }

    if (field.type() != FieldType.STRUCT
        && !(field.type() == FieldType.ARRAY && field.itemType() == FieldType.STRUCT)) {
      rejectPresent(field.fields(), path + ".fields", "fields is only allowed for STRUCT or ARRAY<STRUCT>", issues);
      rejectPresent(field.structRef(), path + ".structRef", "structRef is only allowed for STRUCT or ARRAY<STRUCT>", issues);
    }

    if (field.type() != FieldType.ARRAY) {
      rejectPresent(field.itemType(), path + ".itemType", "itemType is only allowed for ARRAY", issues);
      rejectPresent(field.itemsEnum(), path + ".itemsEnum", "itemsEnum is only allowed for ARRAY<STRING>", issues);
      rejectPresent(field.minItems(), path + ".minItems", "minItems is only allowed for ARRAY", issues);
      rejectPresent(field.maxItems(), path + ".maxItems", "maxItems is only allowed for ARRAY", issues);
    }
  }

  private static void validateArrayDefault(Field field, String path, List<ValidationIssue> issues) {
    if (field.defaultValue() == null) {
      return;
    }
    if (!field.defaultValue().isArray()) {
      issues.add(new ValidationIssue(path, "default must be an array"));
      return;
    }
    if (field.itemType() == null) {
      return;
    }
    for (int i = 0; i < field.defaultValue().size(); i++) {
      JsonNode item = field.defaultValue().get(i);
      String itemPath = path + "[" + i + "]";
      switch (field.itemType()) {
        case STRING -> {
          if (!item.isTextual()) {
            issues.add(new ValidationIssue(itemPath, "item must be a string"));
          } else if (!list(field.itemsEnum()).isEmpty() && !field.itemsEnum().contains(item.asText())) {
            issues.add(new ValidationIssue(itemPath, "item must be one of itemsEnum"));
          }
        }
        case INTEGER -> {
          if (!item.isIntegralNumber()) {
            issues.add(new ValidationIssue(itemPath, "item must be an integer"));
          }
        }
        case NUMBER -> {
          if (!item.isNumber()) {
            issues.add(new ValidationIssue(itemPath, "item must be numeric"));
          }
        }
        case BOOLEAN -> {
          if (!item.isBoolean()) {
            issues.add(new ValidationIssue(itemPath, "item must be boolean"));
          }
        }
        case STRUCT -> {
          if (!item.isObject()) {
            issues.add(new ValidationIssue(itemPath, "item must be an object"));
          }
        }
        case ARRAY -> issues.add(new ValidationIssue(itemPath, "ARRAY itemType cannot be ARRAY"));
      }
    }
  }

  private static void detectCircularStructReferences(Map<String, Struct> structById, List<ValidationIssue> issues) {
    if (structById.isEmpty()) {
      return;
    }
    Set<String> visiting = new LinkedHashSet<>();
    Set<String> visited = new HashSet<>();
    Deque<String> stack = new ArrayDeque<>();
    for (String id : structById.keySet()) {
      if (visited.contains(id)) {
        continue;
      }
      List<String> cycle = findCycle(id, structById, visiting, visited, stack);
      if (!cycle.isEmpty()) {
        issues.add(
            new ValidationIssue(
                "schema.structs",
                "Circular STRUCT reference detected: " + String.join(" -> ", cycle)));
        return;
      }
    }
  }

  private static List<String> findCycle(
      String current,
      Map<String, Struct> structById,
      Set<String> visiting,
      Set<String> visited,
      Deque<String> stack) {
    if (visiting.contains(current)) {
      List<String> path = new ArrayList<>(stack);
      int index = path.indexOf(current);
      if (index < 0) {
        return List.of(current, current);
      }
      List<String> cycle = new ArrayList<>(path.subList(index, path.size()));
      cycle.add(current);
      return List.copyOf(cycle);
    }
    if (visited.contains(current)) {
      return List.of();
    }

    visiting.add(current);
    stack.addLast(current);

    Struct struct = structById.get(current);
    if (struct != null) {
      for (String ref : referencedStructIds(struct.fields())) {
        if (!structById.containsKey(ref)) {
          continue;
        }
        List<String> cycle = findCycle(ref, structById, visiting, visited, stack);
        if (!cycle.isEmpty()) {
          return cycle;
        }
      }
    }

    stack.removeLast();
    visiting.remove(current);
    visited.add(current);
    return List.of();
  }

  private static List<String> referencedStructIds(List<Field> fields) {
    List<String> refs = new ArrayList<>();
    collectStructRefs(fields, refs);
    return List.copyOf(refs);
  }

  private static void collectStructRefs(List<Field> fields, List<String> refs) {
    for (Field field : list(fields)) {
      if (field == null || field.type() == null) {
        continue;
      }
      if (field.type() == FieldType.STRUCT) {
        if (!isBlank(field.structRef())) {
          refs.add(field.structRef().trim());
        } else {
          collectStructRefs(field.fields(), refs);
        }
        continue;
      }
      if (field.type() == FieldType.ARRAY && field.itemType() == FieldType.STRUCT) {
        if (!isBlank(field.structRef())) {
          refs.add(field.structRef().trim());
        } else {
          collectStructRefs(field.fields(), refs);
        }
      }
    }
  }

  private static ObjectNode toCanonicalGroup(Group group, ObjectMapper mapper) {
    ObjectNode node = mapper.createObjectNode();
    node.put("id", group.id());
    node.put("label", group.label());
    if (!isBlank(group.description())) {
      node.put("description", group.description());
    }
    node.put("order", group.order());
    return node;
  }

  private static ObjectNode toCanonicalStruct(Struct struct, ObjectMapper mapper) {
    ObjectNode node = mapper.createObjectNode();
    node.put("id", struct.id());
    node.put("label", struct.label());
    if (!isBlank(struct.description())) {
      node.put("description", struct.description());
    }
    ArrayNode fields = node.putArray("fields");
    sortedFields(struct.fields()).forEach(field -> fields.add(toCanonicalField(field, mapper)));
    return node;
  }

  private static ObjectNode toCanonicalField(Field field, ObjectMapper mapper) {
    ObjectNode node = mapper.createObjectNode();
    node.put("key", field.key());
    node.put("label", field.label());
    if (!isBlank(field.description())) {
      node.put("description", field.description());
    }
    if (!isBlank(field.groupId())) {
      node.put("groupId", field.groupId());
    }
    node.put("order", field.order());
    node.put("type", field.type().name());
    node.put("required", Boolean.TRUE.equals(field.required()));
    node.put("editable", !Boolean.FALSE.equals(field.editable()));
    if (!list(field.enumValues()).isEmpty()) {
      ArrayNode enumNode = node.putArray("enum");
      field.enumValues().forEach(enumNode::add);
    }
    if (field.defaultValue() != null) {
      node.set("default", copy(field.defaultValue()));
    }
    if (field.min() != null) {
      node.put("min", field.min());
    }
    if (field.max() != null) {
      node.put("max", field.max());
    }
    if (!isBlank(field.pattern())) {
      node.put("pattern", field.pattern());
    }

    if (field.type() == FieldType.STRUCT) {
      if (!isBlank(field.structRef())) {
        node.put("structRef", field.structRef().trim());
      } else {
        ArrayNode fields = node.putArray("fields");
        sortedFields(field.fields()).forEach(child -> fields.add(toCanonicalField(child, mapper)));
      }
    }

    if (field.type() == FieldType.ARRAY) {
      node.put("itemType", field.itemType().name());
      if (field.itemType() == FieldType.STRING && !list(field.itemsEnum()).isEmpty()) {
        ArrayNode itemsEnum = node.putArray("itemsEnum");
        field.itemsEnum().forEach(itemsEnum::add);
      }
      if (field.itemType() == FieldType.STRUCT) {
        if (!isBlank(field.structRef())) {
          node.put("structRef", field.structRef().trim());
        } else {
          ArrayNode fields = node.putArray("fields");
          sortedFields(field.fields()).forEach(child -> fields.add(toCanonicalField(child, mapper)));
        }
      }
      if (field.minItems() != null) {
        node.put("minItems", field.minItems());
      }
      if (field.maxItems() != null) {
        node.put("maxItems", field.maxItems());
      }
    }
    return node;
  }

  private static void unknownKeys(JsonNode node, Set<String> allowed, String path, List<ValidationIssue> issues) {
    node.fieldNames().forEachRemaining(key -> {
      if (!allowed.contains(key)) {
        issues.add(new ValidationIssue(path + "." + key, "unknown property"));
      }
    });
  }

  private static String requiredText(JsonNode node, String key, String path, List<ValidationIssue> issues) {
    JsonNode value = node.get(key);
    if (value == null || value.isNull()) {
      issues.add(new ValidationIssue(path, key + " is required"));
      return null;
    }
    if (!value.isTextual() || value.asText().trim().isEmpty()) {
      issues.add(new ValidationIssue(path, key + " must be a non-empty string"));
      return null;
    }
    return value.asText().trim();
  }

  private static String optionalText(JsonNode node, String key, String path, List<ValidationIssue> issues) {
    if (!node.has(key)) {
      return null;
    }
    JsonNode value = node.get(key);
    if (value == null || value.isNull()) {
      return null;
    }
    if (!value.isTextual()) {
      issues.add(new ValidationIssue(path, key + " must be a string"));
      return null;
    }
    String normalized = value.asText().trim();
    return normalized.isEmpty() ? null : normalized;
  }

  private static FieldType requiredType(
      JsonNode node, String key, String path, List<ValidationIssue> issues, boolean required) {
    if (!node.has(key)) {
      if (required) {
        issues.add(new ValidationIssue(path, key + " is required"));
      }
      return null;
    }
    JsonNode value = node.get(key);
    if (value == null || value.isNull()) {
      if (required) {
        issues.add(new ValidationIssue(path, key + " is required"));
      }
      return null;
    }
    if (!value.isTextual()) {
      issues.add(new ValidationIssue(path, key + " must be a supported type"));
      return null;
    }
    try {
      FieldType type = FieldType.valueOf(value.asText().trim());
      if ("itemType".equals(key) && type == FieldType.ARRAY) {
        issues.add(new ValidationIssue(path, "ARRAY itemType cannot be ARRAY"));
        return null;
      }
      return type;
    } catch (Exception ex) {
      issues.add(new ValidationIssue(path, "unsupported type: " + value.asText()));
      return null;
    }
  }

  private static List<String> stringArray(
      JsonNode node, String path, boolean required, List<ValidationIssue> issues) {
    if (node == null || node.isNull() || node.isMissingNode()) {
      if (required) {
        issues.add(new ValidationIssue(path, "value is required"));
      }
      return List.of();
    }
    if (!node.isArray()) {
      issues.add(new ValidationIssue(path, "value must be an array"));
      return List.of();
    }
    List<String> values = new ArrayList<>();
    for (int i = 0; i < node.size(); i++) {
      JsonNode item = node.get(i);
      String itemPath = path + "[" + i + "]";
      if (item == null || item.isNull() || !item.isTextual() || item.asText().trim().isEmpty()) {
        issues.add(new ValidationIssue(itemPath, "value must be a non-empty string"));
      } else {
        values.add(item.asText().trim());
      }
    }
    return List.copyOf(values);
  }

  private static List<String> normalizedStrings(List<String> values, String path, List<ValidationIssue> issues) {
    if (values == null) {
      return List.of();
    }
    List<String> normalized = new ArrayList<>();
    for (int i = 0; i < values.size(); i++) {
      String value = values.get(i);
      if (isBlank(value)) {
        issues.add(new ValidationIssue(path + "[" + i + "]", "value must be a non-empty string"));
      } else {
        normalized.add(value.trim());
      }
    }
    return List.copyOf(normalized);
  }

  private static List<String> sortedDistinctStrings(List<String> values) {
    LinkedHashSet<String> deduped = new LinkedHashSet<>();
    list(values).stream().filter(Objects::nonNull).map(String::trim).filter(v -> !v.isEmpty()).forEach(deduped::add);
    List<String> sorted = new ArrayList<>(deduped);
    sorted.sort(String::compareTo);
    return List.copyOf(sorted);
  }

  private static void rejectPresent(Object value, String path, String message, List<ValidationIssue> issues) {
    if (value == null) {
      return;
    }
    if (value instanceof Collection<?> collection && collection.isEmpty()) {
      return;
    }
    if (value instanceof String text && text.isBlank()) {
      return;
    }
    issues.add(new ValidationIssue(path, message));
  }

    return value == null || value.trim().isEmpty();
  }

  }

  public static final class ValidationException extends IllegalArgumentException {
    private final List<ValidationIssue> issues;

    private ValidationException(String message, List<ValidationIssue> issues, Throwable cause) {
      super(message, cause);
      this.issues = issues == null ? List.of() : List.copyOf(issues);
    }

    public List<ValidationIssue> issues() {
      return issues;
    }
  }
