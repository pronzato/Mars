import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.nio.file.StandardOpenOption;
import java.util.UUID;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public final class GenerateVwapConfigCopies {
    private static final Path DEFAULT_SEED_PATH = Paths.get(
            "vault", "git", "dev", "lab", "config_layer", "config__vwap__algo001.json");

    private static final Pattern PARENT_ID_PATTERN = Pattern.compile("\"parentId\"\\s*:\\s*\"([^\"]+)\"");

    private GenerateVwapConfigCopies() {
    }

    public static void main(String[] args) throws IOException {
        if (args.length != 1) {
            System.err.println("Usage: java GenerateVwapConfigCopies <N>");
            System.err.println("Example: java GenerateVwapConfigCopies 99");
            System.exit(1);
        }

        int n;
        try {
            n = Integer.parseInt(args[0]);
        } catch (NumberFormatException ex) {
            System.err.println("N must be an integer.");
            System.exit(1);
            return;
        }

        if (n < 1) {
            System.err.println("N must be >= 1.");
            System.exit(1);
        }

        if (!Files.exists(DEFAULT_SEED_PATH)) {
            System.err.println("Seed file not found: " + DEFAULT_SEED_PATH.toAbsolutePath());
            System.exit(1);
        }

        String seedJson = Files.readString(DEFAULT_SEED_PATH, StandardCharsets.UTF_8);
        String parentId = extractParentId(seedJson);

        Path configLayerDir = DEFAULT_SEED_PATH.getParent();
        int created = 0;

        for (int i = 2; i <= n + 1; i++) {
            String suffix = String.format("%03d", i);
            String algoId = "config:lab/vwap/algo" + suffix;
            String fileName = "config__vwap__algo" + suffix + ".json";
            String envelopeName = "config__vwap__algo" + suffix;
            String shortName = "L" + suffix;
            String displayName = "VWAP - Algo " + suffix;
            String uuid = UUID.randomUUID().toString();

            String json = seedJson;
            json = replaceJsonStringValue(json, "resourceId", algoId);
            json = replaceJsonStringValue(json, "parentId", parentId);
            json = replaceJsonStringValue(json, "shortName", shortName);
            json = replaceJsonStringValue(json, "displayName", displayName);
            json = replaceJsonStringValue(json, "configId", algoId);
            json = replaceJsonStringValue(json, "effectiveConfigId", algoId);
            json = replaceJsonStringValue(json, "name", envelopeName);
            json = replaceJsonStringValue(json, "uuid", uuid);

            Path out = configLayerDir.resolve(fileName);
            Files.writeString(
                    out,
                    ensureTrailingNewline(json),
                    StandardCharsets.UTF_8,
                    StandardOpenOption.CREATE,
                    StandardOpenOption.TRUNCATE_EXISTING,
                    StandardOpenOption.WRITE);
            created++;
        }

        System.out.println("Seed file: " + DEFAULT_SEED_PATH.toAbsolutePath());
        System.out.println("ParentId used for all generated files: " + parentId);
        System.out.println("Created/updated files: " + created + " (algo002..algo" + String.format("%03d", n + 1) + ")");
    }

    private static String extractParentId(String json) {
        Matcher m = PARENT_ID_PATTERN.matcher(json);
        if (!m.find()) {
            throw new IllegalStateException("Could not find parentId in seed JSON.");
        }
        return m.group(1);
    }

    private static String replaceJsonStringValue(String json, String key, String value) {
        Pattern p = Pattern.compile("(\\\"" + Pattern.quote(key) + "\\\"\\s*:\\s*\\\")[^\\\"]*(\\\")");
        Matcher m = p.matcher(json);
        if (!m.find()) {
            throw new IllegalStateException("Could not find key in seed JSON: " + key);
        }
        String replacement = m.group(1) + Matcher.quoteReplacement(value) + m.group(2);
        return m.replaceFirst(replacement);
    }

    private static String ensureTrailingNewline(String json) {
        if (json.endsWith("\n")) {
            return json;
        }
        return json + "\n";
    }
}
