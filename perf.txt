public final class AssetLifecycleGraphBuilder {
  private AssetLifecycleGraphBuilder() {}

  public static AssetLifecycleGraph build(List<Definition> definitions) {
    if (definitions == null || definitions.isEmpty()) {
      return new AssetLifecycleGraph(Map.of(), List.of(), Map.of(), Map.of());
    }

    Map<String, Semantic> configSemanticByKey = indexConfigSemantic(definitions);
    Map<String, AssetLifecycleNode> nodes = new LinkedHashMap<>();
    List<DefinitionContext> contexts = new ArrayList<>();
    NodeIndex index = new NodeIndex();

    for (Definition definition : definitions) {
      if (definition == null) {
        continue;
      }
      DefinitionContext context = toContext(definition);
      if (context == null) {
        continue;
      }
      if (nodes.putIfAbsent(context.assetId(), context.node()) == null) {
        contexts.add(context);
        index.register(context);
      }
    }

    Map<String, Integer> datasetProducerCounts = countIngestDatasetProducers(contexts, index, nodes);
    List<AssetLifecycleEdge> edges = new ArrayList<>();
    Set<String> dedupe = new LinkedHashSet<>();

    for (DefinitionContext context : contexts) {
      Definition definition = context.definition();
      if (definition instanceof DefinitionDataset dataset) {
        inferDatasetEdges(context, dataset, index, nodes, edges, dedupe);
      } else if (definition instanceof DefinitionService service) {
        inferServiceEdges(context, service, index, nodes, edges, dedupe);
      } else if (definition instanceof DefinitionStream stream) {
        inferStreamEdges(context, stream, index, nodes, edges, dedupe);
      } else if (definition instanceof DefinitionIngest ingest) {
        inferIngestEdges(context, ingest, index, nodes, datasetProducerCounts, edges, dedupe);
      } else if (definition instanceof DefinitionIngestSource ingestSource) {
        inferIngestSourceEdges(context, ingestSource, index, nodes, edges, dedupe);
      } else if (definition instanceof DefinitionSchema schema) {
        inferSchemaEdges(context, schema, index, nodes, edges, dedupe);
      } else if (definition instanceof DefinitionConfigLayer config) {
        inferConfigEdges(context, config, index, nodes, edges, dedupe);
      } else if (definition instanceof DefinitionEntitlement entitlement) {
        inferEntitlementEdges(context, entitlement, index, nodes, edges, dedupe);
      } else if (definition instanceof DefinitionConnection connection) {
        inferConnectionEdges(context, connection, index, nodes, edges, dedupe);
      }
      inferSemanticHintEdges(
          context, definition, configSemanticByKey, index, nodes, edges, dedupe);
    }

    edges.sort(
        Comparator.comparing(AssetLifecycleEdge::fromAssetId)
            .thenComparing(AssetLifecycleEdge::toAssetId)
            .thenComparing(AssetLifecycleEdge::relationshipType));
    Map<String, List<AssetLifecycleEdge>> outgoing = new LinkedHashMap<>();
    Map<String, List<AssetLifecycleEdge>> incoming = new LinkedHashMap<>();
    for (AssetLifecycleEdge edge : edges) {
      outgoing.computeIfAbsent(edge.fromAssetId(), ignored -> new ArrayList<>()).add(edge);
      incoming.computeIfAbsent(edge.toAssetId(), ignored -> new ArrayList<>()).add(edge);
    }
    return new AssetLifecycleGraph(nodes, edges, outgoing, incoming);
  }

  private static DefinitionContext toContext(Definition definition) {
    DefinitionEnvelope envelope = definition.envelope();
    if (envelope == null || !hasText(envelope.group()) || !hasText(envelope.name())) {
      return null;
    }
    DefinitionType type = envelope.type() == null ? DefinitionType.UNKNOWN : envelope.type();
    String group = envelope.group().trim();
    String name = envelope.name().trim();
    String namespace = "";
    String assetId;

    if (definition instanceof DefinitionConfigLayer) {
      return null;
    }
    if (definition instanceof DefinitionDataset dataset) {
      namespace = hasText(dataset.namespace()) ? dataset.namespace().trim() : "default";
      assetId = "dataset:" + group + "/" + namespace + "/" + name;
    } else if (definition instanceof DefinitionService service) {
      assetId = hasText(service.resourceId()) ? service.resourceId().trim() : "service:" + group + "/" + name;
    } else if (definition instanceof DefinitionStream stream) {
      namespace = hasText(stream.namespace()) ? stream.namespace().trim() : "default";
      assetId = hasText(stream.streamId()) ? stream.streamId().trim() : "stream:" + group + "/" + namespace + "/" + name;
    } else if (definition instanceof DefinitionIngest ingest) {
      String[] datasetParts = splitDatasetId(ingest.datasetId());
      if (datasetParts != null) {
        namespace = datasetParts[1];
      }
      assetId = "ingest:" + group + "/" + name;
    } else if (definition instanceof DefinitionIngestSource ingestSource) {
      namespace = hasText(ingestSource.namespace()) ? ingestSource.namespace().trim() : "default";
      assetId = "ingest-source:" + group + "/" + name;
    } else if (definition instanceof DefinitionSchema) {
      assetId = "schema:" + group + "/" + name;
    } else if (definition instanceof DefinitionConfigLayer config) {
      if (!isBaseConfig(config)) {
        return null;
      }
      assetId = hasText(config.configId()) ? config.configId().trim() : "config:" + group + "/" + name + "/base";
      String[] configParts = splitConfigId(assetId);
      if (configParts != null) {
        namespace = configParts[1];
      }
    } else if (definition instanceof DefinitionEntitlement) {
      assetId = "entitlement:" + group + "/" + name;
    } else if (definition instanceof DefinitionConnection) {
      assetId = "connection:" + group + "/" + name;
    } else {
      assetId = type.name().toLowerCase(Locale.ROOT) + ":" + group + "/" + name;
    }

    return new DefinitionContext(
        definition,
        assetId,
        new AssetLifecycleNode(assetId, type, group, name, namespace, "definition"),
        group,
        namespace);
  }

  private static Semantic semantic(
      Definition definition, Map<String, Semantic> configSemanticByKey) {
    if (definition instanceof DefinitionDataset dataset) {
      return dataset.semantic();
    }
    if (definition instanceof DefinitionService service) {
      return service.semantic();
    }
    if (definition instanceof DefinitionStream stream) {
      return stream.semantic();
    }
    if (definition instanceof DefinitionIngest ingest) {
      return ingest.semantic();
    }
    if (definition instanceof DefinitionIngestSource ingestSource) {
      return ingestSource.semantic();
    }
    if (definition instanceof DefinitionSchema schema) {
      return schema.semantic();
    }
    if (definition instanceof DefinitionConfigLayer config) {
      String group =
          trimToNull(config == null || config.envelope() == null ? null : config.envelope().group());
      String configSetId = normalizeConfigSetId(config == null ? null : config.configId());
      if (group != null && configSetId != null && configSemanticByKey != null) {
        Semantic fromMeta = configSemanticByKey.get(configSemanticKey(group, configSetId));
        if (fromMeta != null) {
          return fromMeta;
        }
      }
      return null;
    }
    return null;
  }

  private static boolean isBaseConfig(DefinitionConfigLayer config) {
    if (config == null) {
      return true;
    }
    return !hasText(config.parentId());
  }

  private static String[] splitConfigId(String configId) {
    if (!hasText(configId) || !configId.startsWith("config:")) {
      return null;
    }
    String[] parts = configId.substring("config:".length()).split("/");
    if (parts.length < 3) {
      return null;
    }
    return new String[] {parts[0], parts[1], parts[2]};
  }

  private static String[] splitDatasetId(String datasetId) {
    if (!hasText(datasetId) || !datasetId.startsWith("dataset:")) {
      return null;
    }
    String[] parts = datasetId.substring("dataset:".length()).split("/");
    if (parts.length < 3) {
      return null;
    }
    return new String[] {parts[0], parts[1], parts[2]};
  }

  private static boolean hasText(String value) {
    return value != null && !value.trim().isEmpty();
  }

  private static String normalizeToken(String value) {
    if (value == null || value.trim().isEmpty()) {
      return "unknown";
    }
    return value.trim().replace('\\', '/');
  }

  private static Map<String, Semantic> indexConfigSemantic(List<Definition> definitions) {
    if (definitions == null || definitions.isEmpty()) {
      return Map.of();
    }
    LinkedHashMap<String, Semantic> indexed = new LinkedHashMap<>();
    for (Definition definition : definitions) {
      if (!(definition instanceof DefinitionConfig configMeta)
          || configMeta.semantic() == null) {
        continue;
      }
      String group =
          trimToNull(
              configMeta.envelope() == null ? null : configMeta.envelope().group());
      if (group == null) {
        group = trimToNull(configMeta.group());
      }
      String configSetId = normalizeConfigSetId(configMeta.configId());
      if (group == null || configSetId == null) {
        continue;
      }
      indexed.put(configSemanticKey(group, configSetId), configMeta.semantic());
    }
    return indexed.isEmpty() ? Map.of() : Map.copyOf(indexed);
  }

  private static String configSemanticKey(String group, String configSetId) {
    return group + "|" + configSetId;
  }

  private static String normalizeConfigSetId(String value) {
    String normalized = trimToNull(value);
    if (normalized == null) {
      return null;
    }
    if (!normalized.startsWith("config:")) {
      return normalized;
    }
    String payload = normalized.substring("config:".length());
    String[] parts = payload.split("/");
    if (parts.length < 2) {
      return null;
    }
    return trimToNull(parts[1]);
  }

  private static String trimToNull(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }

  private static String storageAssetId(String locator) {
    return "storage:" + normalizeToken(locator);
  }

  private record DefinitionContext(
      Definition definition,
      String assetId,
      AssetLifecycleNode node,
      String group,
      String namespace) {}

  private static final class NodeIndex {
    private final Map<DefinitionType, Map<String, String>> byTypeGroupName =
        new EnumMap<>(DefinitionType.class);
    private final Map<String, String> byDatasetKey = new LinkedHashMap<>();
    private final Map<String, String> byStreamKey = new LinkedHashMap<>();
    private final Map<String, String> byConfigId = new LinkedHashMap<>();
    private final Map<String, String> byConfigFamily = new LinkedHashMap<>();
    private final Map<String, String> bySchemaId = new LinkedHashMap<>();
    private final Map<String, String> byIngestKey = new LinkedHashMap<>();
    private final Map<String, String> byIngestSourceKey = new LinkedHashMap<>();
    private final Map<String, String> byRawAssetId = new LinkedHashMap<>();

    private void register(DefinitionContext context) {
      byRawAssetId.put(context.assetId(), context.assetId());
      byTypeGroupName
          .computeIfAbsent(context.node().type(), ignored -> new LinkedHashMap<>())
          .put(groupNameKey(context.group(), context.node().name()), context.assetId());
      if (context.definition() instanceof DefinitionDataset dataset) {
        String namespace = hasText(dataset.namespace()) ? dataset.namespace().trim() : "default";
        byDatasetKey.put(groupNamespaceNameKey(context.group(), namespace, context.node().name()), context.assetId());
      } else if (context.definition() instanceof DefinitionStream stream) {
        String namespace = hasText(stream.namespace()) ? stream.namespace().trim() : "default";
        byStreamKey.put(groupNamespaceNameKey(context.group(), namespace, context.node().name()), context.assetId());
      } else if (context.definition() instanceof DefinitionConfigLayer config) {
        if (hasText(config.configId())) {
          String configId = config.configId().trim();
          byConfigId.put(configId, context.assetId());
          String[] parts = splitConfigId(configId);
          if (parts != null) {
            byConfigFamily.put(parts[0] + "|" + parts[1], context.assetId());
          }
        }
      } else if (context.definition() instanceof DefinitionSchema schema) {
        bySchemaId.put("schema:" + context.group() + "/" + context.node().name(), context.assetId());
        if (hasText(schema.hash())) {
          bySchemaId.put("schema-hash:" + schema.hash().trim(), context.assetId());
        }
      } else if (context.definition() instanceof DefinitionIngest ingest) {
        String namespace = context.namespace();
        if (!hasText(namespace)) {
          String[] datasetParts = splitDatasetId(ingest.datasetId());
          namespace = datasetParts == null ? "default" : datasetParts[1];
        }
        byIngestKey.put(
            groupNamespaceNameKey(context.group(), namespace, context.node().name()), context.assetId());
      } else if (context.definition() instanceof DefinitionIngestSource ingestSource) {
        String namespace = hasText(ingestSource.namespace()) ? ingestSource.namespace().trim() : "default";
        byIngestSourceKey.put(
            groupNamespaceNameKey(context.group(), namespace, context.node().name()), context.assetId());
      }
    }

    private String resolveSchemaReference(
        String schemaId,
        String schemaGroup,
        String schemaName,
        String fallbackGroup,
        Map<String, AssetLifecycleNode> nodes) {
      if (hasText(schemaId)) {
        return resolveReference(schemaId, DefinitionType.SCHEMA, fallbackGroup, null, nodes);
      }
      String group = hasText(schemaGroup) ? schemaGroup.trim() : fallbackGroup;
      if (hasText(group) && hasText(schemaName)) {
        String byName = resolveReference(schemaName, DefinitionType.SCHEMA, group, null, nodes);
        if (hasText(byName)) {
          return byName;
        }
        String explicit = "schema:" + group + "/" + schemaName.trim();
        return bySchemaId.getOrDefault(explicit, explicit);
      }
      return null;
    }

    private String resolveReference(
        String raw,
        DefinitionType expectedType,
        String sourceGroup,
        String sourceNamespace,
        Map<String, AssetLifecycleNode> nodes) {
      if (!hasText(raw)) {
        return null;
      }
      String normalized = raw.trim();
      if (nodes.containsKey(normalized)) {
        return normalized;
      }
      if (isCanonical(normalized)) {
        if (normalized.startsWith("config:")) {
          String[] parts = splitConfigId(normalized);
          if (parts != null) {
            String family = byConfigFamily.get(parts[0] + "|" + parts[1]);
            if (hasText(family)) {
              return family;
            }
          }
        }
        return normalized;
      }
      if (expectedType == DefinitionType.DATASET) {
        String namespace = hasText(sourceNamespace) ? sourceNamespace : "default";
        String candidate = byDatasetKey.get(groupNamespaceNameKey(sourceGroup, namespace, normalized));
        return hasText(candidate) ? candidate : "dataset:" + sourceGroup + "/" + namespace + "/" + normalized;
      }
      if (expectedType == DefinitionType.STREAM) {
        String namespace = hasText(sourceNamespace) ? sourceNamespace : "default";
        String candidate = byStreamKey.get(groupNamespaceNameKey(sourceGroup, namespace, normalized));
        return hasText(candidate) ? candidate : "stream:" + sourceGroup + "/" + namespace + "/" + normalized;
      }
      if (expectedType == DefinitionType.CONFIG_LAYER) {
        if (normalized.startsWith("config:")) {
          String[] parts = splitConfigId(normalized);
          if (parts != null) {
            String family = byConfigFamily.get(parts[0] + "|" + parts[1]);
            if (hasText(family)) {
              return family;
            }
          }
          return normalized;
        }
        String candidate = byConfigId.get("config:" + sourceGroup + "/" + normalized + "/base");
        return hasText(candidate) ? candidate : "config:" + sourceGroup + "/" + normalized + "/base";
      }
      if (expectedType == DefinitionType.CONNECTION) {
        String candidate = byTypeGroupName(DefinitionType.CONNECTION, sourceGroup, normalized);
        return hasText(candidate) ? candidate : "connection:" + sourceGroup + "/" + normalized;
      }
      if (expectedType == DefinitionType.INGEST) {
        String namespace = hasText(sourceNamespace) ? sourceNamespace : "default";
        String candidate = byIngestKey.get(groupNamespaceNameKey(sourceGroup, namespace, normalized));
        if (!hasText(candidate)) {
          candidate = byTypeGroupName(DefinitionType.INGEST, sourceGroup, normalized);
        }
        return hasText(candidate) ? candidate : "ingest:" + sourceGroup + "/" + normalized;
      }
      if (expectedType == DefinitionType.INGEST_SOURCE) {
        String namespace = hasText(sourceNamespace) ? sourceNamespace : "default";
        String candidate = byIngestSourceKey.get(groupNamespaceNameKey(sourceGroup, namespace, normalized));
        if (!hasText(candidate)) {
          candidate = byTypeGroupName(DefinitionType.INGEST_SOURCE, sourceGroup, normalized);
        }
        return hasText(candidate) ? candidate : "ingest-source:" + sourceGroup + "/" + normalized;
      }
      if (expectedType == DefinitionType.SCHEMA) {
        String byName = byTypeGroupName(DefinitionType.SCHEMA, sourceGroup, normalized);
        if (hasText(byName)) {
          return byName;
        }
        String explicit = "schema:" + sourceGroup + "/" + normalized;
        return bySchemaId.getOrDefault(explicit, explicit);
      }
      if (expectedType != null) {
        String byName = byTypeGroupName(expectedType, sourceGroup, normalized);
        if (hasText(byName)) {
          return byName;
        }
      }
      String fallback = byRawAssetId.get(normalized);
      return hasText(fallback) ? fallback : normalized;
    }

    private String byTypeGroupName(DefinitionType type, String group, String name) {
      Map<String, String> values = byTypeGroupName.get(type);
      if (values == null || !hasText(group) || !hasText(name)) {
        return null;
      }
      return values.get(groupNameKey(group, name));
    }

    private static String groupNameKey(String group, String name) {
      return normalize(group) + "|" + normalize(name);
    }

    private static String groupNamespaceNameKey(String group, String namespace, String name) {
      return normalize(group) + "|" + normalize(namespace) + "|" + normalize(name);
    }

    private static String normalize(String value) {
      return value == null ? "" : value.trim().toLowerCase(Locale.ROOT);
    }

    private static boolean isCanonical(String value) {
      if (!hasText(value)) {
        return false;
      }
      String trimmed = value.trim();
      int separator = trimmed.indexOf(':');
      if (separator <= 0 || separator >= trimmed.length() - 1 || trimmed.contains(" ")) {
        return false;
      }
      String prefix = trimmed.substring(0, separator).toLowerCase(Locale.ROOT);
      return switch (prefix) {
        case "dataset",
            "stream",
            "service",
            "endpoint",
            "materialized",
            "feed",
            "config",
            "schema",
            "connection",
            "ingest",
            "ingest-source",
            "entitlement",
            "namespace",
            "app",
            "user",
            "group",
            "role",
            "doc",
            "storage",
            "external" -> true;
        default -> false;
      };
    }
  }

  private static void inferDatasetEdges(
      DefinitionContext context,
      DefinitionDataset dataset,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    DefinitionDataset.View view = dataset.view();
    if (view != null && view.dependencies() != null) {
      for (String dependency : view.dependencies()) {
        String target = index.resolveReference(dependency, null, context.group(), context.namespace(), nodes);
        addEdge(
            context.assetId(),
            target,
            RelationshipType.DERIVED_FROM,
            OwnershipType.SHARED,
            PromotionBehavior.REQUIRED_WITH_PARENT,
            DeleteBehavior.BLOCK,
            RetentionBehavior.KEEP_HISTORY,
            true,
            true,
            "dataset.view.dependencies",
            "Dataset view dependency",
            null,
            edges,
            dedupe);
      }
    }

    DefinitionDataset.Binding binding = dataset.binding();
    if (binding == null) {
      return;
    }
    DefinitionDataset.ParquetBinding parquet = binding.parquet();
    if (parquet != null) {
      if (hasText(parquet.connectionName())) {
        String connectionAsset =
            index.resolveReference(
                parquet.connectionName(), DefinitionType.CONNECTION, context.group(), null, nodes);
        addEdge(
            context.assetId(),
            connectionAsset,
            RelationshipType.USES_CONNECTION,
            OwnershipType.SHARED,
            PromotionBehavior.REBIND_IN_TARGET,
            DeleteBehavior.DETACH,
            RetentionBehavior.KEEP_HISTORY,
            true,
            true,
            "dataset.binding.parquet.connectionName",
            "Parquet dataset uses external connection",
            null,
            edges,
            dedupe);
      }
      if (hasText(parquet.uri())) {
        addEdge(
            context.assetId(),
            storageAssetId(parquet.uri()),
            RelationshipType.STORES_IN,
            OwnershipType.EXTERNAL,
            PromotionBehavior.NEVER_PROMOTE,
            DeleteBehavior.MANUAL,
            RetentionBehavior.ARCHIVE_ONLY,
            false,
            true,
            "dataset.binding.parquet.uri",
            "Parquet dataset storage target",
            parquet.uri(),
            edges,
            dedupe);
      }
    }
    DefinitionDataset.DuckdbBinding duckdb = binding.duckdb();
    if (duckdb != null && hasText(duckdb.databasePath())) {
      addEdge(
          context.assetId(),
          storageAssetId(duckdb.databasePath()),
          RelationshipType.STORES_IN,
          OwnershipType.EXTERNAL,
          PromotionBehavior.NEVER_PROMOTE,
          DeleteBehavior.MANUAL,
          RetentionBehavior.ARCHIVE_ONLY,
          false,
          true,
          "dataset.binding.duckdb.databasePath",
          "DuckDB file path backing dataset",
          duckdb.databasePath(),
          edges,
          dedupe);
    }
    DefinitionDataset.ExternalBinding external = binding.external();
    if (external != null && hasText(external.reference())) {
      addEdge(
          context.assetId(),
          "external:" + external.reference().trim(),
          RelationshipType.CONSUMES,
          OwnershipType.EXTERNAL,
          PromotionBehavior.REBIND_IN_TARGET,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          false,
          true,
          "dataset.binding.external.reference",
          "External dataset reference",
          external.reference(),
          edges,
          dedupe);
    }
    DefinitionDataset.OdbcBinding odbc = binding.odbc();
    if (odbc != null && hasText(odbc.connectionName())) {
      String ref =
          index.resolveReference(
              odbc.connectionName(), DefinitionType.CONNECTION, context.group(), null, nodes);
      addEdge(
          context.assetId(),
          ref,
          RelationshipType.USES_CONNECTION,
          OwnershipType.SHARED,
          PromotionBehavior.REBIND_IN_TARGET,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          true,
          true,
          "dataset.binding.odbc.connectionName",
          "ODBC dataset source connection",
          null,
          edges,
          dedupe);
    }
  }

  private static void inferServiceEdges(
      DefinitionContext context,
      DefinitionService service,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    if (service.schemaRef() != null) {
      String schemaId =
          index.resolveSchemaReference(
              service.schemaRef().schemaId(),
              service.schemaRef().group(),
              service.schemaRef().name(),
              context.group(),
              nodes);
      addEdge(
          context.assetId(),
          schemaId,
          RelationshipType.USES_SCHEMA,
          OwnershipType.SHARED,
          PromotionBehavior.REQUIRED_WITH_PARENT,
          DeleteBehavior.BLOCK,
          RetentionBehavior.KEEP_HISTORY,
          true,
          true,
          "service.schemaRef",
          "Service contract schema",
          null,
          edges,
          dedupe);
    }
    if (service.connectionRefs() != null) {
      for (String connectionRef : service.connectionRefs()) {
        String connectionId =
            index.resolveReference(connectionRef, DefinitionType.CONNECTION, context.group(), null, nodes);
        addEdge(
            context.assetId(),
            connectionId,
            RelationshipType.USES_CONNECTION,
            OwnershipType.SHARED,
            PromotionBehavior.REBIND_IN_TARGET,
            DeleteBehavior.DETACH,
            RetentionBehavior.KEEP_HISTORY,
            true,
            true,
            "service.connectionRefs",
            "Service connection dependency",
            null,
            edges,
            dedupe);
      }
    }
    if (service.amps() != null && hasText(service.amps().connection())) {
      String connectionId =
          index.resolveReference(service.amps().connection(), DefinitionType.CONNECTION, context.group(), null, nodes);
      addEdge(
          context.assetId(),
          connectionId,
          RelationshipType.USES_CONNECTION,
          OwnershipType.SHARED,
          PromotionBehavior.REBIND_IN_TARGET,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          true,
          true,
          "service.amps.connection",
          "AMPS transport connection",
          null,
          edges,
          dedupe);
    }
  }

  private static void inferStreamEdges(
      DefinitionContext context,
      DefinitionStream stream,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    if (hasText(stream.schemaId()) || hasText(stream.schemaKey())) {
      String schemaId =
          index.resolveSchemaReference(stream.schemaId(), context.group(), stream.schemaKey(), context.group(), nodes);
      addEdge(
          context.assetId(),
          schemaId,
          RelationshipType.USES_SCHEMA,
          OwnershipType.SHARED,
          PromotionBehavior.REQUIRED_WITH_PARENT,
          DeleteBehavior.BLOCK,
          RetentionBehavior.KEEP_HISTORY,
          true,
          true,
          "stream.schema",
          "Stream payload schema",
          null,
          edges,
          dedupe);
    }
    if (stream.connectionRefs() != null) {
      for (String connectionRef : stream.connectionRefs()) {
        String connectionId =
            index.resolveReference(connectionRef, DefinitionType.CONNECTION, context.group(), null, nodes);
        addEdge(
            context.assetId(),
            connectionId,
            RelationshipType.USES_CONNECTION,
            OwnershipType.SHARED,
            PromotionBehavior.REBIND_IN_TARGET,
            DeleteBehavior.DETACH,
            RetentionBehavior.KEEP_HISTORY,
            true,
            true,
            "stream.connectionRefs",
            "Stream transport connection",
            null,
            edges,
            dedupe);
      }
    }
    if (stream.dependencyRefs() != null) {
      for (String ref : stream.dependencyRefs()) {
        String dependency = index.resolveReference(ref, null, context.group(), context.namespace(), nodes);
        addEdge(
            context.assetId(),
            dependency,
            RelationshipType.DEPENDS_ON,
            OwnershipType.SHARED,
            PromotionBehavior.OPTIONAL_WITH_PARENT,
            DeleteBehavior.BLOCK,
            RetentionBehavior.KEEP_HISTORY,
            false,
            true,
            "stream.dependencyRefs",
            "Stream dependency reference",
            null,
            edges,
            dedupe);
      }
    }
  }

  private static void inferIngestEdges(
      DefinitionContext context,
      DefinitionIngest ingest,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      Map<String, Integer> datasetProducerCounts,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    String ingestSourceId =
        index.resolveReference(
            context.node().name(), DefinitionType.INGEST_SOURCE, context.group(), context.namespace(), nodes);
    addEdge(
        context.assetId(),
        ingestSourceId,
        RelationshipType.OWNS,
        OwnershipType.OWNED,
        PromotionBehavior.REQUIRED_WITH_PARENT,
        DeleteBehavior.CASCADE_SOFT,
        RetentionBehavior.KEEP_HISTORY,
        true,
        false,
        "ingest.name->ingestSource",
        "Ingest source definition paired by ingest name",
        null,
        edges,
        dedupe);

    if (hasText(ingest.datasetId())) {
      String datasetId =
          index.resolveReference(
              ingest.datasetId(), DefinitionType.DATASET, context.group(), context.namespace(), nodes);
      int producers = datasetProducerCounts.getOrDefault(datasetId, 0);
      boolean ownedDataset = producers <= 1 && sameGroup(nodes, context.assetId(), datasetId);
      addEdge(
          context.assetId(),
          datasetId,
          RelationshipType.PRODUCES,
          ownedDataset ? OwnershipType.OWNED : OwnershipType.SHARED,
          ownedDataset ? PromotionBehavior.REQUIRED_WITH_PARENT : PromotionBehavior.REBIND_IN_TARGET,
          ownedDataset ? DeleteBehavior.CASCADE_SOFT : DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          true,
          !ownedDataset,
          "ingest.datasetId",
          ownedDataset
              ? "Dataset owned by single ingest pipeline"
              : "Dataset is shared and should rebind in target lane",
          null,
          edges,
          dedupe);
    }

    if (hasText(ingest.outputSchemaId())) {
      String outputSchemaId =
          index.resolveSchemaReference(
              ingest.outputSchemaId(), context.group(), ingest.outputSchemaId(), context.group(), nodes);
      boolean derivedSchema =
          hasText(outputSchemaId)
              && outputSchemaId.startsWith("schema:" + context.group() + "/ingest:");
      addEdge(
          context.assetId(),
          outputSchemaId,
          RelationshipType.USES_SCHEMA,
          derivedSchema ? OwnershipType.DERIVED : OwnershipType.SHARED,
          PromotionBehavior.REQUIRED_WITH_PARENT,
          derivedSchema ? DeleteBehavior.CASCADE_SOFT : DeleteBehavior.BLOCK,
          RetentionBehavior.KEEP_HISTORY,
          true,
          !derivedSchema,
          "ingest.outputSchemaId",
          "Ingest output schema contract",
          null,
          edges,
          dedupe);
    }
  }

  private static void inferSchemaEdges(
      DefinitionContext context,
      DefinitionSchema schema,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    if (schema.refs() != null) {
      for (DefinitionSchema.Ref ref : schema.refs()) {
        if (ref == null || !hasText(ref.refId())) {
          continue;
        }
        DefinitionType expectedType = definitionTypeFromToken(ref.refType());
        String target =
            index.resolveReference(ref.refId(), expectedType, context.group(), context.namespace(), nodes);
        RelationshipType relationshipType =
            relationshipTypeForSchemaRef(expectedType, target, nodes);
        addEdge(
            context.assetId(),
            target,
            relationshipType,
            OwnershipType.SHARED,
            PromotionBehavior.OPTIONAL_WITH_PARENT,
            expectedType == DefinitionType.SCHEMA ? DeleteBehavior.BLOCK : DeleteBehavior.DETACH,
            RetentionBehavior.KEEP_HISTORY,
            expectedType == DefinitionType.SCHEMA,
            true,
            "schema.refs",
            "Schema reference",
            null,
            edges,
            dedupe);
      }
    }
    if (hasText(schema.doc())) {
      addEdge(
          context.assetId(),
          "doc:" + normalizeToken(schema.doc()),
          RelationshipType.DOCUMENTED_BY,
          OwnershipType.EXTERNAL,
          PromotionBehavior.NEVER_PROMOTE,
          DeleteBehavior.MANUAL,
          RetentionBehavior.ARCHIVE_ONLY,
          false,
          true,
          "schema.doc",
          "Schema documentation reference",
          schema.doc(),
          edges,
          dedupe);
    }
  }

  private static void inferEntitlementEdges(
      DefinitionContext context,
      DefinitionEntitlement entitlement,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    addEntitlementGovernedAssetEdge(
        context, index, nodes, entitlement.scopeId(), definitionTypeFromToken(entitlement.scopeType()), "entitlement.scopeId", edges, dedupe);

    if (entitlement.policies() != null) {
      for (DefinitionEntitlement.PolicyDocument policy : entitlement.policies()) {
        if (policy == null) {
          continue;
        }
        addEntitlementGovernedAssetEdge(
            context,
            index,
            nodes,
            policy.resourceId(),
            definitionTypeFromToken(policy.scopeType() == null ? policy.type() : policy.scopeType()),
            "entitlement.policy.resourceId",
            edges,
            dedupe);
        if (policy.bindings() != null) {
          for (DefinitionEntitlement.PolicyBinding binding : policy.bindings()) {
            if (binding == null || !hasText(binding.subject())) {
              continue;
            }
            addEdge(
                context.assetId(),
                binding.subject().trim(),
                RelationshipType.REFERENCES,
                OwnershipType.EXTERNAL,
                PromotionBehavior.NEVER_PROMOTE,
                DeleteBehavior.DETACH,
                RetentionBehavior.KEEP_HISTORY,
                false,
                true,
                "entitlement.policy.bindings.subject",
                "Policy subject reference",
                null,
                edges,
                dedupe);
            if (binding.obligations() != null && binding.obligations().datasetScope() != null) {
              for (String datasetScope : binding.obligations().datasetScope()) {
                addEntitlementGovernedAssetEdge(
                    context,
                    index,
                    nodes,
                    datasetScope,
                    DefinitionType.DATASET,
                    "entitlement.policy.bindings.obligations.datasetScope",
                    edges,
                    dedupe);
              }
            }
          }
        }
      }
    }
    if (entitlement.relationships() != null) {
      for (DefinitionEntitlement.RelationshipTuple relationship : entitlement.relationships()) {
        if (relationship == null) {
          continue;
        }
        addEntitlementGovernedAssetEdge(
            context,
            index,
            nodes,
            relationship.resourceId(),
            definitionTypeFromToken(relationship.resourceType()),
            "entitlement.relationships.resourceId",
            edges,
            dedupe);
        addEntitlementGovernedAssetEdge(
            context,
            index,
            nodes,
            relationship.resourceSubject(),
            null,
            "entitlement.relationships.resourceSubject",
            edges,
            dedupe);
      }
    }
  }

  private static void inferConnectionEdges(
      DefinitionContext context,
      DefinitionConnection connection,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    if (connection.client() != null && hasText(connection.client().physicalRef())) {
      String physicalRef = connection.client().physicalRef().trim();
      String resolvedPhysical =
          index.resolveReference(physicalRef, DefinitionType.CONNECTION, context.group(), null, nodes);
      if (!hasText(resolvedPhysical) || !nodes.containsKey(resolvedPhysical)) {
        String[] slashParts = physicalRef.split("/");
        if (slashParts.length == 2) {
          resolvedPhysical =
              index.resolveReference(
                  slashParts[1], DefinitionType.CONNECTION, slashParts[0], null, nodes);
        }
      }
      addEdge(
          context.assetId(),
          resolvedPhysical,
          RelationshipType.USES_CONNECTION,
          OwnershipType.SHARED,
          PromotionBehavior.REBIND_IN_TARGET,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          true,
          true,
          "connection.client.physicalRef",
          "Client connection depends on physical connection",
          null,
          edges,
          dedupe);
    }
    if (connection.bitbucketRepo() != null && hasText(connection.bitbucketRepo().authConnectionName())) {
      String authConnection =
          index.resolveReference(
              connection.bitbucketRepo().authConnectionName(),
              DefinitionType.CONNECTION,
              context.group(),
              null,
              nodes);
      addEdge(
          context.assetId(),
          authConnection,
          RelationshipType.USES_CONNECTION,
          OwnershipType.SHARED,
          PromotionBehavior.REBIND_IN_TARGET,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          true,
          true,
          "connection.bitbucketRepo.authConnectionName",
          "Bitbucket auth connection",
          null,
          edges,
          dedupe);
    }
    if (connection.odbc() != null && hasText(connection.odbc().connectionName())) {
      String ref =
          index.resolveReference(connection.odbc().connectionName(), DefinitionType.CONNECTION, context.group(), null, nodes);
      addEdge(
          context.assetId(),
          ref,
          RelationshipType.USES_CONNECTION,
          OwnershipType.SHARED,
          PromotionBehavior.REBIND_IN_TARGET,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          false,
          true,
          "connection.odbc.connectionName",
          "ODBC connection chain reference",
          null,
          edges,
          dedupe);
    }
    if (connection.s3() != null && hasText(connection.s3().bucket())) {
      String uri =
          "s3://"
              + connection.s3().bucket().trim()
              + (hasText(connection.s3().basePrefix()) ? "/" + connection.s3().basePrefix().trim() : "");
      addEdge(
          context.assetId(),
          storageAssetId(uri),
          RelationshipType.STORES_IN,
          OwnershipType.EXTERNAL,
          PromotionBehavior.NEVER_PROMOTE,
          DeleteBehavior.MANUAL,
          RetentionBehavior.ARCHIVE_ONLY,
          false,
          true,
          "connection.s3.bucket",
          "S3 storage locator",
          uri,
          edges,
          dedupe);
    }
    if (connection.physical() != null && hasText(connection.physical().uriOrEndpoint())) {
      addEdge(
          context.assetId(),
          "endpoint:" + normalizeToken(connection.physical().uriOrEndpoint()),
          RelationshipType.DEPENDS_ON,
          OwnershipType.EXTERNAL,
          PromotionBehavior.NEVER_PROMOTE,
          DeleteBehavior.MANUAL,
          RetentionBehavior.ARCHIVE_ONLY,
          false,
          true,
          "connection.physical.uriOrEndpoint",
          "Physical endpoint locator",
          connection.physical().uriOrEndpoint(),
          edges,
          dedupe);
    }
  }

  private static void inferSemanticHintEdges(
      DefinitionContext context,
      Definition definition,
      Map<String, Semantic> configSemanticByKey,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    Semantic metadata = semantic(definition, configSemanticByKey);
    if (metadata == null || metadata.relationshipHints() == null || metadata.relationshipHints().isEmpty()) {
      return;
    }
    for (Semantic.RelationshipHint hint : metadata.relationshipHints()) {
      if (hint == null || !hasText(hint.targetAssetId())) {
        continue;
      }
      String targetToken = hint.targetAssetId().trim();
      if (targetToken.toLowerCase(Locale.ROOT).startsWith("todo:")) {
        continue;
      }
      String target = index.resolveReference(targetToken, null, context.group(), context.namespace(), nodes);
      addEdge(
          context.assetId(),
          target,
          mapHintType(hint.relationshipType()),
          OwnershipType.SHARED,
          PromotionBehavior.OPTIONAL_WITH_PARENT,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          false,
          true,
          "semantic.relationshipHints",
          hint.description(),
          hint.joinExpression(),
          edges,
          dedupe);
    }
  }

  private static Map<String, Integer> countIngestDatasetProducers(
      List<DefinitionContext> contexts, NodeIndex index, Map<String, AssetLifecycleNode> nodes) {
    Map<String, Integer> counts = new LinkedHashMap<>();
    for (DefinitionContext context : contexts) {
      if (!(context.definition() instanceof DefinitionIngest ingest) || !hasText(ingest.datasetId())) {
        continue;
      }
      String datasetId =
          index.resolveReference(ingest.datasetId(), DefinitionType.DATASET, context.group(), context.namespace(), nodes);
      counts.put(datasetId, counts.getOrDefault(datasetId, 0) + 1);
    }
    return counts;
  }

  private static boolean sameGroup(
      Map<String, AssetLifecycleNode> nodes, String leftAssetId, String rightAssetId) {
    AssetLifecycleNode left = nodes.get(leftAssetId);
    AssetLifecycleNode right = nodes.get(rightAssetId);
    return left != null && right != null && left.group().equalsIgnoreCase(right.group());
  }

  private static RelationshipType mapHintType(Semantic.RelationshipType type) {
    if (type == null) {
      return RelationshipType.DEPENDS_ON;
    }
    return switch (type) {
      case DEPENDS_ON -> RelationshipType.DEPENDS_ON;
      case DERIVED_FROM -> RelationshipType.DERIVED_FROM;
      case JOINS_TO -> RelationshipType.JOINS_TO;
      case CONSUMES -> RelationshipType.CONSUMES;
      case PRODUCES -> RelationshipType.PRODUCES;
      case DOCUMENTED_BY -> RelationshipType.DOCUMENTED_BY;
      case GOVERNED_BY -> RelationshipType.GOVERNED_BY;
    };
  }

  private static DefinitionType definitionTypeFromToken(String token) {
    if (!hasText(token)) {
      return null;
    }
    String normalized = token.trim().toLowerCase(Locale.ROOT);
    return switch (normalized) {
      case "dataset", "metaset", "materialized" -> DefinitionType.DATASET;
      case "service" -> DefinitionType.SERVICE;
      case "stream", "feed" -> DefinitionType.STREAM;
      case "schema" -> DefinitionType.SCHEMA;
      case "config" -> DefinitionType.CONFIG_LAYER;
      case "connection" -> DefinitionType.CONNECTION;
      case "entitlement" -> DefinitionType.ENTITLEMENT;
      case "ingest" -> DefinitionType.INGEST;
      case "ingest_source", "ingest-source" -> DefinitionType.INGEST_SOURCE;
      default -> null;
    };
  }

  private static void addEntitlementGovernedAssetEdge(
      DefinitionContext context,
      NodeIndex index,
      Map<String, AssetLifecycleNode> nodes,
      String rawTarget,
      DefinitionType expectedType,
      String source,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    if (!hasText(rawTarget)) {
      return;
    }
    String target = index.resolveReference(rawTarget, expectedType, context.group(), context.namespace(), nodes);
    if (!hasText(target)) {
      return;
    }
    boolean governableTarget = isGovernableTarget(target, expectedType, nodes);
    if (governableTarget) {
      addEdge(
          target,
          context.assetId(),
          RelationshipType.GOVERNED_BY,
          OwnershipType.SHARED,
          PromotionBehavior.OPTIONAL_WITH_PARENT,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          false,
          true,
          source,
          "Asset governed by entitlement policy",
          null,
          edges,
          dedupe);
      addEdge(
          context.assetId(),
          target,
          RelationshipType.DEPENDS_ON,
          OwnershipType.SHARED,
          PromotionBehavior.OPTIONAL_WITH_PARENT,
          DeleteBehavior.DETACH,
          RetentionBehavior.KEEP_HISTORY,
          false,
          true,
          source,
          "Entitlement scope depends on governed asset",
          null,
          edges,
          dedupe);
      return;
    }
    addEdge(
        context.assetId(),
        target,
        RelationshipType.REFERENCES,
        OwnershipType.EXTERNAL,
        PromotionBehavior.OPTIONAL_WITH_PARENT,
        DeleteBehavior.DETACH,
        RetentionBehavior.KEEP_HISTORY,
        false,
        true,
        source,
        "Entitlement references non-governable target",
        null,
        edges,
        dedupe);
  }

  private static RelationshipType relationshipTypeForSchemaRef(
      DefinitionType expectedType, String targetAssetId, Map<String, AssetLifecycleNode> nodes) {
    if (expectedType == DefinitionType.SCHEMA) {
      return RelationshipType.DEPENDS_ON;
    }
    if (expectedType == DefinitionType.DATASET) {
      return RelationshipType.DERIVED_FROM;
    }
    if (expectedType == DefinitionType.SERVICE
        || expectedType == DefinitionType.STREAM
        || expectedType == DefinitionType.CONFIG_LAYER
        || expectedType == DefinitionType.INGEST
        || expectedType == DefinitionType.INGEST_SOURCE) {
      return RelationshipType.DEPENDS_ON;
    }
    if (isInternalAssetId(targetAssetId, nodes)) {
      return RelationshipType.DEPENDS_ON;
    }
    return RelationshipType.REFERENCES;
  }

  private static boolean isGovernableTarget(
      String assetId, DefinitionType expectedType, Map<String, AssetLifecycleNode> nodes) {
    if (expectedType == DefinitionType.DATASET
        || expectedType == DefinitionType.SERVICE
        || expectedType == DefinitionType.STREAM
        || expectedType == DefinitionType.CONFIG_LAYER
        || expectedType == DefinitionType.INGEST) {
      return true;
    }
    if (!hasText(assetId)) {
      return false;
    }
    AssetLifecycleNode node = nodes.get(assetId);
    if (node != null) {
      return node.type() == DefinitionType.DATASET
          || node.type() == DefinitionType.SERVICE
          || node.type() == DefinitionType.STREAM
          || node.type() == DefinitionType.CONFIG_LAYER
          || node.type() == DefinitionType.INGEST;
    }
    return assetId.startsWith("dataset:")
        || assetId.startsWith("service:")
        || assetId.startsWith("stream:")
        || assetId.startsWith("config:")
        || assetId.startsWith("ingest:");
  }

  private static boolean isInternalAssetId(
      String assetId, Map<String, AssetLifecycleNode> nodes) {
    if (!hasText(assetId)) {
      return false;
    }
    if (nodes.containsKey(assetId)) {
      return true;
    }
    return assetId.startsWith("dataset:")
        || assetId.startsWith("service:")
        || assetId.startsWith("stream:")
        || assetId.startsWith("schema:")
        || assetId.startsWith("config:")
        || assetId.startsWith("connection:")
        || assetId.startsWith("ingest:")
        || assetId.startsWith("ingest-source:")
        || assetId.startsWith("entitlement:")
        || assetId.startsWith("app:")
        || assetId.startsWith("platform:")
        || assetId.startsWith("environment:");
  }

  private static void addEdge(
      String fromAssetId,
      String toAssetId,
      RelationshipType relationshipType,
      OwnershipType ownershipType,
      PromotionBehavior promotionBehavior,
      DeleteBehavior deleteBehavior,
      RetentionBehavior retentionBehavior,
      boolean required,
      boolean shared,
      String source,
      String notes,
      String targetLocator,
      List<AssetLifecycleEdge> edges,
      Set<String> dedupe) {
    if (!hasText(fromAssetId) || !hasText(toAssetId)) {
      return;
    }
    AssetLifecycleEdge edge =
        new AssetLifecycleEdge(
            fromAssetId.trim(),
            toAssetId.trim(),
            relationshipType,
            ownershipType,
            promotionBehavior,
            deleteBehavior,
            retentionBehavior,
            required,
            shared,
            source,
            notes,
            targetLocator);
    String key =
        edge.fromAssetId()
            + "|"
            + edge.toAssetId()
            + "|"
            + edge.relationshipType()
            + "|"
            + edge.promotionBehavior()
            + "|"
            + edge.deleteBehavior();
    if (dedupe.add(key)) {
      edges.add(edge);
    }
  }
