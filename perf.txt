final class DiscoveryCacheType {

  private DiscoveryCacheType() {}

  static DefinitionType toType(
      Class<?> clazz,
      Map<Class<? extends Definition>, DefinitionType> classToType,
      Set<String> unmappedDefinitionClasses,
      Logger logger) {
    if (clazz == null || !Definition.class.isAssignableFrom(clazz)) {
      return DefinitionType.UNKNOWN;
    }
    @SuppressWarnings("unchecked")
    Class<? extends Definition> definitionClass = (Class<? extends Definition>) clazz;
    DefinitionType type = classToType.get(definitionClass);
    if (type != null) {
      return type;
    }
    String className = definitionClass.getName();
    if (unmappedDefinitionClasses.add(className)) {
      logger.warn(
          "[DiscoveryCache][Compatibility] unmapped definition class {}; treating as UNKNOWN",
          className);
    }
    return DefinitionType.UNKNOWN;
  }

  static boolean isCacheableType(
      DefinitionType type, Map<DefinitionType, Class<? extends Definition>> cacheableDefinitions) {
    return type != null && type != DefinitionType.UNKNOWN && cacheableDefinitions.containsKey(type);
  }

  static DefinitionType definitionType(Definition definition) {
    DefinitionEnvelope envelope = definitionEnvelope(definition);
    return envelope == null ? null : envelope.type();
  }

  static String definitionGroup(Definition definition) {
    DefinitionEnvelope envelope = definitionEnvelope(definition);
    return envelope == null ? null : envelope.group();
  }

  static String definitionName(Definition definition) {
    DefinitionEnvelope envelope = definitionEnvelope(definition);
    return envelope == null ? null : envelope.name();
  }

  static DefinitionEnvelope definitionEnvelope(Definition definition) {
    return definition == null ? null : definition.envelope();
  }

  static String normalize(String value) {
    return Strings.trimToNull(value);
  }

  static String safe(Object value) {
    return value == null ? "" : String.valueOf(value);
  }

  static String schemaName(DefinitionSchema schema) {
    DefinitionEnvelope envelope = schema == null ? null : schema.envelope();
    return Strings.firstNonBlank(envelope == null ? null : envelope.name(), "");
  }

  static String schemaTypeName(DefinitionSchema schema) {
    return schema == null || schema.schemaType() == null ? "" : schema.schemaType().name();
  }

  static Map<DefinitionType, Class<? extends Definition>> cacheableDefinitions() {
    EnumMap<DefinitionType, Class<? extends Definition>> map = new EnumMap<>(DefinitionType.class);
    map.put(DefinitionType.APP, DefinitionApp.class);
    map.put(DefinitionType.CONNECTION, DefinitionConnection.class);
    map.put(DefinitionType.DATASET, DefinitionDataset.class);
    map.put(DefinitionType.STREAM, DefinitionStream.class);
    map.put(DefinitionType.INGEST, DefinitionIngest.class);
    map.put(DefinitionType.INGEST_SOURCE, DefinitionIngestSource.class);
    map.put(DefinitionType.ENVIRONMENT, DefinitionEnvironment.class);
    map.put(DefinitionType.PLATFORM, DefinitionPlatform.class);
    map.put(DefinitionType.SCHEMA, DefinitionSchema.class);
    map.put(DefinitionType.SERVICE, DefinitionService.class);
    map.put(DefinitionType.ENTITLEMENT, DefinitionEntitlement.class);
    map.put(DefinitionType.DASHBOARD, DefinitionDashboard.class);
    map.put(DefinitionType.CONFIG_LAYER, DefinitionConfigLayer.class);
    map.put(DefinitionType.CONFIG, DefinitionConfig.class);
    map.put(DefinitionType.CONFIG_CHANGE, DefinitionConfigChange.class);
    map.put(DefinitionType.STATUS_APP, DefinitionStatusApp.class);
    // No concrete class is modeled yet, but the bootstrap record is still cacheable.
    map.put(DefinitionType.SECRETS_KDF, Definition.class);
    return Map.copyOf(map);
  }

  static Map<Class<? extends Definition>, DefinitionType> classToType(
      Map<DefinitionType, Class<? extends Definition>> cacheableDefinitions) {
    ConcurrentHashMap<Class<? extends Definition>, DefinitionType> reverse = new ConcurrentHashMap<>();
    for (Map.Entry<DefinitionType, Class<? extends Definition>> entry : cacheableDefinitions.entrySet()) {
      Class<? extends Definition> definitionClass = entry.getValue();
      if (definitionClass == Definition.class) {
        continue;
      }
      reverse.putIfAbsent(definitionClass, entry.getKey());
    }
    return Map.copyOf(reverse);
  }
}

public final class DiscoveryCodec {
  private static final Logger LOGGER = LoggerFactory.getLogger(DiscoveryCodec.class);
  private static final long UNKNOWN_TYPE_LOG_WINDOW_MS = 30_000L;
  private static final ConcurrentHashMap<String, Long> UNKNOWN_TYPE_LAST_LOG =
      new ConcurrentHashMap<>();
  private static final ObjectMapper MAPPER =
      new ObjectMapper()
          .setSerializationInclusion(JsonInclude.Include.NON_NULL)
          .disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES)
          .enable(DeserializationFeature.READ_UNKNOWN_ENUM_VALUES_USING_DEFAULT_VALUE)
          .disable(SerializationFeature.FAIL_ON_EMPTY_BEANS)
          .findAndRegisterModules();

  private DiscoveryCodec() {}

  public static ObjectMapper mapper() {
    return MAPPER;
  }

  public static Definition decode(String json) {
    String normalized = Strings.trimToNull(json);
    if (normalized == null) {
      throw new IllegalArgumentException("definition data is required");
    }
    try {
      return decode(MAPPER.readTree(normalized));
    } catch (IOException ex) {
      throw new IllegalArgumentException("definition data must be valid JSON", ex);
    }
  }

  public static Definition decode(byte[] data) {
    if (data == null || data.length == 0) {
      throw new IllegalArgumentException("definition data is required");
    }
    try {
      return decode(MAPPER.readTree(data));
    } catch (IOException ex) {
      throw new IllegalArgumentException("definition data must be valid JSON", ex);
    }
  }

  public static Definition decode(JsonNode node) {
    if (node == null || node.isNull() || node.isMissingNode() || !node.isObject()) {
      return null;
    }
    ObjectNode root = (ObjectNode) node;
    JsonNode envelopeNode = root.get("envelope");
    if (envelopeNode == null || envelopeNode.isNull() || !envelopeNode.isObject()) {
      return null;
    }
    ObjectNode envelopeRoot = (ObjectNode) envelopeNode;
    String typeText = asText(envelopeRoot.get("type"));
    String group = asText(envelopeRoot.get("group"));
    String name = asText(envelopeRoot.get("name"));
    String uuid = asText(envelopeRoot.get("uuid"));
    DefinitionType type = parseType(typeText);
    if (type == DefinitionType.UNKNOWN || !isSupportedDecodeType(type)) {
      logUnknownTypeWithDedupe(group, typeText, name, uuid);
      return null;
    }
    DefinitionEnvelope envelope = decodeEnvelope(envelopeRoot, type);
    if (envelope == null) {
      return null;
    }
    try {
      Definition decoded =
          switch (type) {
            case APP -> MAPPER.treeToValue(root, DefinitionApp.class);
            case ENVIRONMENT -> MAPPER.treeToValue(root, DefinitionEnvironment.class);
            case CONNECTION -> MAPPER.treeToValue(root, DefinitionConnection.class);
            case DATASET -> MAPPER.treeToValue(root, DefinitionDataset.class);
            case STREAM -> MAPPER.treeToValue(root, DefinitionStream.class);
            case INGEST -> MAPPER.treeToValue(root, DefinitionIngest.class);
            case INGEST_SOURCE -> MAPPER.treeToValue(root, DefinitionIngestSource.class);
            case PLATFORM -> MAPPER.treeToValue(root, DefinitionPlatform.class);
            case SCHEMA -> MAPPER.treeToValue(root, DefinitionSchema.class);
            case SERVICE -> MAPPER.treeToValue(root, DefinitionService.class);
            case ENTITLEMENT -> MAPPER.treeToValue(root, DefinitionEntitlement.class);
            case DASHBOARD -> MAPPER.treeToValue(root, DefinitionDashboard.class);
            case CONFIG -> MAPPER.treeToValue(root, DefinitionConfig.class);
            case CONFIG_LAYER -> MAPPER.treeToValue(root, DefinitionConfigLayer.class);
            case CONFIG_CHANGE -> MAPPER.treeToValue(root, DefinitionConfigChange.class);
            case STATUS_APP -> MAPPER.treeToValue(root, DefinitionStatusApp.class);
            case STATUS_SERVICE -> MAPPER.treeToValue(root, DefinitionStatusService.class);
            case STATUS_DATA -> MAPPER.treeToValue(root, DefinitionStatusData.class);
            case HEARTBEAT_SERVICE -> MAPPER.treeToValue(root, DefinitionHbService.class);
            default -> throw new IllegalArgumentException("Unsupported definition type=" + type);
          };
      if (decoded.envelope() == null) {
        throw new DefinitionDecodeException(
            "Decoded definition missing envelope for type=" + type, envelope, root.deepCopy(), null);
      }
      return decoded;
    } catch (DefinitionDecodeException ex) {
      throw ex;
    } catch (Exception ex) {
      throw new DefinitionDecodeException(
          "Unable to decode definition for type=" + type, envelope, root.deepCopy(), ex);
    }
  }

  private static DefinitionEnvelope decodeEnvelope(ObjectNode envelopeNode, DefinitionType type) {
    String group = asText(envelopeNode.get("group"));
    String name = asText(envelopeNode.get("name"));
    if (isBlank(group) || isBlank(name)) {
      return null;
    }
    DefinitionEnvelope envelope =
        new DefinitionEnvelope(
            group,
            type,
            name,
            asText(envelopeNode.get("uuid")),
            asBoolean(envelopeNode.get("enabled")),
            asText(envelopeNode.get("lifecycleStatus")),
            asText(envelopeNode.get("createdTimeUtc")),
            asText(envelopeNode.get("createdBy")),
            asText(envelopeNode.get("updatedTimeUtc")),
            asText(envelopeNode.get("updatedBy")));
    return envelope;
  }

  private static DefinitionType parseType(String value) {
    if (isBlank(value)) {
      return DefinitionType.UNKNOWN;
    }
    try {
      return DefinitionType.valueOf(value);
    } catch (IllegalArgumentException ex) {
      return DefinitionType.UNKNOWN;
    }
  }

  private static String asText(JsonNode node) {
    if (node == null || node.isNull() || !node.isTextual()) {
      return null;
    }
    return node.textValue();
  }

  private static Boolean asBoolean(JsonNode node) {
    return node == null || node.isNull() || !node.isBoolean() ? null : node.booleanValue();
  }

  private static boolean isBlank(String value) {
    return value == null || value.isBlank();
  }

  private static void logUnknownTypeWithDedupe(
      String group, String typeText, String name, String uuid) {
    String key = safe(group) + "|" + safe(typeText) + "|" + safe(name) + "|" + safe(uuid);
    long now = System.currentTimeMillis();
    Long last = UNKNOWN_TYPE_LAST_LOG.get(key);
    if (last != null && now - last <= UNKNOWN_TYPE_LOG_WINDOW_MS) {
      return;
    }
    UNKNOWN_TYPE_LAST_LOG.put(key, now);
    LOGGER.debug("[Discovery][UnknownType][Ignored] group={} typeText={} name={} uuid={}", group, typeText, name, uuid);
  }

  private static String safe(String value) {
    return value == null ? "" : value;
  }

  private static boolean isSupportedDecodeType(DefinitionType type) {
    if (type == null) {
      return false;
    }
    return switch (type) {
      case APP,
          ENVIRONMENT,
          CONNECTION,
          DATASET,
          STREAM,
          INGEST,
          INGEST_SOURCE,
          PLATFORM,
          SCHEMA,
          SERVICE,
          ENTITLEMENT,
          DASHBOARD,
          CONFIG,
          CONFIG_LAYER,
          CONFIG_CHANGE,
          STATUS_APP,
          STATUS_SERVICE,
          STATUS_DATA,
          HEARTBEAT_SERVICE -> true;
      default -> false;
    };
  }

  public static final class DefinitionDecodeException extends RuntimeException {
    private final DefinitionEnvelope envelope;
    private final JsonNode payload;

    public DefinitionDecodeException(
        String message, DefinitionEnvelope envelope, JsonNode payload, Throwable cause) {
      super(message, cause);
      this.envelope = envelope;
      this.payload = payload;
    }

    public DefinitionEnvelope getEnvelope() {
      return envelope;
    }

    public JsonNode getPayload() {
      return payload;
    }
  }
}
