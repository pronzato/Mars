public final class DuckdbCatalogImpl implements DuckdbCatalog {
  private static final Logger LOGGER = LoggerFactory.getLogger(DuckdbCatalogImpl.class);
  private static final String PLATFORM_GROUP = "platform";
  private static final String DEFAULT_NAMESPACE = "default";
  private static final String DEFAULT_PARQUET_GLOB = "**/*.parquet";
  private static final Comparator<DatasetId> DATASET_ID_ORDER =
      Comparator.comparing(DatasetId::toString, String.CASE_INSENSITIVE_ORDER)
          .thenComparing(DatasetId::toString);
  private static final Comparator<MetadataTable> METADATA_TABLE_ORDER =
      Comparator.comparing(MetadataTable::namespace, String.CASE_INSENSITIVE_ORDER)
          .thenComparing(MetadataTable::namespace)
          .thenComparing(MetadataTable::name, String.CASE_INSENSITIVE_ORDER)
          .thenComparing(MetadataTable::name)
          .thenComparing(MetadataTable::tableType, String.CASE_INSENSITIVE_ORDER)
          .thenComparing(MetadataTable::tableType);

  private final Object lock = new Object();
  private final DiscoveryCache DiscoveryCache;
  private final RowManagerImpl rowManager;
  private final String sessionGroup;
  private final String defaultNamespace;
  private final ViewGraphPlanner viewGraphPlanner = new ViewGraphPlanner();
  private final IngestModelMapper ingestModelMapper = new IngestModelMapper();
  private final LinkedHashSet<DatasetId> explicitInstalls = new LinkedHashSet<>();
  private final DuckdbDatasetProviderRegistry datasetProviderRegistry;
  private volatile CatalogSnapshot snapshot;

  public DuckdbCatalogImpl(DiscoveryCache DiscoveryCache, RowManagerImpl rowManager, String sessionGroup) {
    this(DiscoveryCache, rowManager, sessionGroup, DEFAULT_NAMESPACE);
  }

  public DuckdbCatalogImpl(
      DiscoveryCache DiscoveryCache, RowManagerImpl rowManager, String sessionGroup, String defaultNamespace) {
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.rowManager = Objects.requireNonNull(rowManager, "rowManager");
    this.sessionGroup = Strings.requireNonBlank(sessionGroup, "sessionGroup");
    this.defaultNamespace = Strings.requireNonBlank(defaultNamespace, "defaultNamespace");
    this.datasetProviderRegistry = DuckdbDatasetProviderRegistry.defaults();
    this.snapshot = CatalogSnapshot.empty(this.sessionGroup);
  }

  @Override
  public void installAll() {
    CatalogSnapshot updated;
    synchronized (lock) {
      updated = rebuildSnapshotLocked();
      snapshot = updated;
    }
    rowManager.withDuckdbSession((Consumer<DuckdbSession>) session -> installSnapshotIntoSession(session, updated));
  }

  @Override
  public void install(String datasetId) {
    DatasetId parsed = DatasetId.parse(Strings.requireNonBlank(datasetId, "datasetId"));
    DefinitionDataset definition = fetchDefinition(parsed);
    synchronized (lock) {
      explicitInstalls.add(parsed);
      if (isActive(definition)) {
        snapshot = upsertLocked(snapshot, toEntry(definition), true);
      }
    }
    rowManager.withDuckdbSession(
        (Consumer<DuckdbSession>) session -> installSnapshotIntoSession(session, snapshot));
  }

  @Override
  public void onDatasetUpsert(DefinitionDataset definition) {
    if (definition == null) {
      return;
    }
    DatasetId datasetId = tryDatasetId(definition);
    if (datasetId == null) {
      return;
    }
    boolean explicit = false;
    synchronized (lock) {
      if (!sessionGroup.equals(datasetId.group()) && !explicitInstalls.contains(datasetId)) {
        return;
      }
      explicit = explicitInstalls.contains(datasetId);
      if (!isActive(definition)) {
        snapshot = removeLocked(snapshot, datasetId, false);
        return;
      }
      snapshot = upsertLocked(snapshot, toEntry(definition), explicit);
    }
    rowManager.withDuckdbSession((Consumer<DuckdbSession>) session -> installDataset(session, toEntry(definition)));
  }

  @Override
  public void onDatasetDelete(String group, String namespace, String name) {
    String resolvedGroup = Strings.requireNonBlank(group, "group");
    String resolvedNamespace = Strings.requireNonBlank(namespace, "namespace");
    String resolvedName = Strings.requireNonBlank(name, "name");
    DatasetId datasetId = DatasetId.of(resolvedGroup, resolvedNamespace, resolvedName);
    if (!sessionGroup.equals(datasetId.group())) {
      synchronized (lock) {
        if (!explicitInstalls.remove(datasetId)) {
          return;
        }
      }
    }

    CatalogEntry removed;
    synchronized (lock) {
      removed = snapshot.entriesById().get(datasetId);
      snapshot = removeLocked(snapshot, datasetId, true);
    }
    String viewName =
        removed == null
            ? duckdbViewName(resolvedGroup, resolvedNamespace, resolvedName)
            : removed.physicalViewName();
    rowManager.withDuckdbSession(
        (Consumer<DuckdbSession>) session -> dropView(session, datasetId, viewName));
  }

  @Override
  public QueryPlan planSql(String sql, Integer limit) {
    String rawSql = Strings.requireNonBlank(sql, "sql");
    CatalogSnapshot current = snapshot;
    DatasetNameResolver resolver =
        new DatasetNameResolver(sessionGroup, defaultNamespace, current);
    List<DatasetId> directDatasets = resolver.resolveAllTableTokensInSql(rawSql);
    String rewrittenSql = rewriteSqlForQuery(rawSql, resolver, current);
    List<DatasetId> expanded = expandDependencies(directDatasets, current);
    return new QueryPlan(applyLimit(rewrittenSql, limit), expanded);
  }

  @Override
  public QueryPlan planSqlPublic(String sql, Integer limit) {
    String rawSql = Strings.requireNonBlank(sql, "sql");
    CatalogSnapshot current = snapshot;
    DatasetNameResolver resolver =
        new DatasetNameResolver(sessionGroup, defaultNamespace, current);
    List<DatasetId> directDatasets = resolver.resolveAllTableTokensInSql(rawSql);
    String rewrittenSql = rewriteSql(rawSql, resolver, current);
    List<DatasetId> expanded = expandDependencies(directDatasets, current);
    return new QueryPlan(applyLimit(rewrittenSql, limit), expanded);
  }

  @Override
  public void prepareSession(DuckdbSession session) {
    Objects.requireNonNull(session, "session");
    installSnapshotIntoSession(session, snapshot);
  }

  @Override
  public void close() {
    synchronized (lock) {
      explicitInstalls.clear();
      snapshot = CatalogSnapshot.empty(sessionGroup);
    }
  }

  public List<MetadataTable> metadataTables() {
    CatalogSnapshot current = snapshot;
    if (current.orderedEntries().isEmpty()) {
      return List.of();
    }
    List<MetadataTable> tables = new ArrayList<>();
    for (CatalogEntry entry : current.orderedEntries()) {
      if (entry == null) {
        continue;
      }
      for (String tableType : metadataTableTypes(entry)) {
        tables.add(new MetadataTable(entry.datasetId(), entry.namespace(), entry.name(), tableType));
      }
    }
    tables.sort(METADATA_TABLE_ORDER);
    return List.copyOf(tables);
  }

  public Optional<SchemaModel> metadataSchemaModel(DatasetId datasetId) {
    Objects.requireNonNull(datasetId, "datasetId");
    CatalogEntry entry = snapshot.entriesById().get(datasetId);
    if (entry == null) {
      return Optional.empty();
    }
    SchemaModel schemaModel = resolveSchemaForMasking(entry);
    if (schemaModel == null) {
      return Optional.empty();
    }
    return Optional.of(schemaModel);
  }

  public record MetadataTable(
      DatasetId datasetId,
      String namespace,
      String name,
      String tableType) {}

  private CatalogSnapshot rebuildSnapshotLocked() {
    LinkedHashMap<DatasetId, CatalogEntry> byId = new LinkedHashMap<>();
    for (DefinitionDataset definition : DiscoveryCache.list(DefinitionDataset.class, sessionGroup)) {
      if (!isActive(definition)) {
        continue;
      }
      CatalogEntry entry = toEntry(definition);
      byId.put(entry.datasetId(), entry);
    }
    for (DatasetId datasetId : explicitInstalls) {
      DefinitionDataset definition = fetchDefinition(datasetId);
      if (!isActive(definition)) {
        continue;
      }
      CatalogEntry entry = toEntry(definition);
      byId.put(entry.datasetId(), entry);
    }
    return buildSnapshotById(byId.values());
  }

  private CatalogSnapshot upsertLocked(
      CatalogSnapshot current, CatalogEntry entry, boolean includeExplicitGroup) {
    LinkedHashMap<DatasetId, CatalogEntry> byId = new LinkedHashMap<>(current.entriesById());
    if (!sessionGroup.equals(entry.datasetId().group()) && !includeExplicitGroup) {
      return current;
    }
    byId.put(entry.datasetId(), entry);
    CatalogSnapshot updated = buildSnapshotById(byId.values());
    LOGGER.debug("[Catalog] update datasetId={} type={} viewName={}", entry.datasetId(), typeToken(entry.definition()), entry.physicalViewName());
    return updated;
  }

  private CatalogSnapshot removeLocked(CatalogSnapshot current, DatasetId datasetId, boolean logDrop) {
    LinkedHashMap<DatasetId, CatalogEntry> byId = new LinkedHashMap<>(current.entriesById());
    CatalogEntry removed = byId.remove(datasetId);
    CatalogSnapshot updated = buildSnapshotById(byId.values());
    if (removed != null && logDrop) {
      LOGGER.debug("[Catalog] drop datasetId={} viewName={}", removed.datasetId(), removed.physicalViewName());
    }
    return updated;
  }

  private CatalogSnapshot buildSnapshotById(Collection<CatalogEntry> entries) {
    List<CatalogEntry> ordered = new ArrayList<>(entries);
    ordered.sort(
        Comparator.comparing((CatalogEntry entry) -> entry.datasetId().toString(), String.CASE_INSENSITIVE_ORDER)
            .thenComparing(entry -> entry.datasetId().toString()));
    Map<DatasetId, CatalogEntry> byId = new LinkedHashMap<>();
    for (CatalogEntry entry : ordered) {
      byId.put(entry.datasetId(), entry);
    }
    Map<String, DatasetId> preliminaryQualified = new HashMap<>();
    Map<String, List<DatasetId>> preliminaryUnqualified = new HashMap<>();
    for (CatalogEntry entry : ordered) {
      String namespaceQualified = qualifiedNameKey(entry.namespace(), entry.name());
      putQualifiedPreferSessionGroup(preliminaryQualified, namespaceQualified, entry.datasetId());
      preliminaryQualified.put(
          qualifiedNameKey(entry.datasetId().group(), entry.namespace(), entry.name()),
          entry.datasetId());
      if (sessionGroup.equals(entry.datasetId().group())) {
        String unqualifiedKey = entry.name().toLowerCase(Locale.ROOT);
        preliminaryUnqualified
            .computeIfAbsent(unqualifiedKey, ignored -> new ArrayList<>())
            .add(entry.datasetId());
      }
    }
    Map<String, List<DatasetId>> preliminaryUnqualifiedImmutable = new HashMap<>();
    for (Map.Entry<String, List<DatasetId>> entry : preliminaryUnqualified.entrySet()) {
      List<DatasetId> values = new ArrayList<>(entry.getValue());
      values.sort(DATASET_ID_ORDER);
      preliminaryUnqualifiedImmutable.put(entry.getKey(), List.copyOf(values));
    }
    CatalogSnapshot preliminary =
        new CatalogSnapshot(
            sessionGroup,
            List.copyOf(ordered),
            Map.copyOf(byId),
            Map.copyOf(preliminaryQualified),
            Map.copyOf(preliminaryUnqualifiedImmutable));

    List<CatalogEntry> withDependencies = new ArrayList<>(ordered.size());
    DatasetNameResolver resolver = new DatasetNameResolver(sessionGroup, defaultNamespace, preliminary);
    for (CatalogEntry entry : ordered) {
      if (!entry.logicalView()) {
        withDependencies.add(entry);
        continue;
      }
      List<DatasetId> dependencies = resolveViewDependencies(entry, resolver);
      withDependencies.add(
          new CatalogEntry(
              entry.datasetId(),
              entry.definition(),
              entry.namespace(),
              entry.name(),
              entry.physicalViewName(),
              dependencies,
              true));
    }

    Map<DatasetId, CatalogEntry> finalById = new LinkedHashMap<>();
    Map<String, DatasetId> qualified = new HashMap<>();
    Map<String, List<DatasetId>> unqualified = new HashMap<>();
    for (CatalogEntry entry : withDependencies) {
      finalById.put(entry.datasetId(), entry);
      String namespaceQualified = qualifiedNameKey(entry.namespace(), entry.name());
      putQualifiedPreferSessionGroup(qualified, namespaceQualified, entry.datasetId());
      qualified.put(
          qualifiedNameKey(entry.datasetId().group(), entry.namespace(), entry.name()),
          entry.datasetId());
      if (sessionGroup.equals(entry.datasetId().group())) {
        String unqualifiedKey = entry.name().toLowerCase(Locale.ROOT);
        unqualified.computeIfAbsent(unqualifiedKey, ignored -> new ArrayList<>()).add(entry.datasetId());
      }
    }
    Map<String, List<DatasetId>> immutableUnqualified = new HashMap<>();
    for (Map.Entry<String, List<DatasetId>> entry : unqualified.entrySet()) {
      List<DatasetId> values = new ArrayList<>(entry.getValue());
      values.sort(DATASET_ID_ORDER);
      immutableUnqualified.put(entry.getKey(), List.copyOf(values));
    }
    return new CatalogSnapshot(
        sessionGroup,
        List.copyOf(withDependencies),
        Map.copyOf(finalById),
        Map.copyOf(qualified),
        Map.copyOf(immutableUnqualified));
  }

  private List<DatasetId> resolveViewDependencies(CatalogEntry entry, DatasetNameResolver resolver) {
    DefinitionDataset.View view = entry.definition().view();
    if (view == null) {
      return List.of();
    }
    LinkedHashSet<DatasetId> dependencies = new LinkedHashSet<>();
    List<String> configured = view.dependencies();
    if (configured != null && !configured.isEmpty()) {
      for (String dependency : configured) {
        DatasetId parsed = tryParseDependencyToken(dependency, resolver);
        if (parsed != null && !parsed.equals(entry.datasetId())) {
          dependencies.add(parsed);
        }
      }
    } else {
      String querySql = Strings.trimToNull(view.querySql());
      if (querySql != null) {
        try {
          for (DatasetId dependency : resolver.resolveAllTableTokensInSql(querySql)) {
            if (!dependency.equals(entry.datasetId())) {
              dependencies.add(dependency);
            }
          }
        } catch (IllegalArgumentException ignore) {
          // Keep catalog rebuild deterministic even when configured view dependencies are temporarily missing.
        }
      }
    }
    return List.copyOf(dependencies);
  }

  private DatasetId tryParseDependencyToken(String dependency, DatasetNameResolver resolver) {
    String token = Strings.trimToNull(dependency);
    if (token == null) {
      return null;
    }
    try {
      if (token.regionMatches(true, 0, "dataset:", 0, "dataset:".length())) {
        return DatasetId.parse(token);
      }
      if (token.indexOf('/') >= 0) {
        String[] parts = token.split("/", 2);
        if (parts.length != 2) {
          return null;
        }
        return DatasetId.of(sessionGroup, parts[0], parts[1]);
      }
      return resolver.resolveTableToken(token);
    } catch (IllegalArgumentException ex) {
      return null;
    }
  }

  private void installSnapshotIntoSession(DuckdbSession session, CatalogSnapshot current) {
    for (CatalogEntry entry : current.orderedEntries()) {
      if (!entry.logicalView()) {
        installDataset(session, entry);
      }
    }
    installLogicalViews(session, current);
  }

  private void installLogicalViews(DuckdbSession session, CatalogSnapshot current) {
    Map<String, List<String>> graph = new LinkedHashMap<>();
    for (CatalogEntry entry : current.orderedEntries()) {
      if (!entry.logicalView()) {
        continue;
      }
      List<String> dependencies = new ArrayList<>();
      for (DatasetId dependency : entry.dependencies()) {
        CatalogEntry dependencyEntry = current.entriesById().get(dependency);
        if (dependencyEntry != null && dependencyEntry.logicalView()) {
          dependencies.add(dependency.toString());
        }
      }
      dependencies.sort(String.CASE_INSENSITIVE_ORDER);
      graph.put(entry.datasetId().toString(), List.copyOf(dependencies));
    }

    LinkedHashSet<String> ordered = new LinkedHashSet<>();
    List<String> roots = new ArrayList<>(graph.keySet());
    roots.sort(String.CASE_INSENSITIVE_ORDER);
    for (String root : roots) {
      ordered.addAll(viewGraphPlanner.plan(root, graph));
    }

    DatasetNameResolver resolver = new DatasetNameResolver(sessionGroup, defaultNamespace, current);
    for (String datasetId : ordered) {
      CatalogEntry entry = current.entriesById().get(DatasetId.parse(datasetId));
      if (entry == null) {
        continue;
      }
      String querySql = Strings.requireNonBlank(entry.definition().view().querySql(), "view.querySql");
      String rewritten;
      try {
        rewritten = rewriteSql(querySql, resolver, current);
      } catch (IllegalArgumentException ex) {
        dropView(session, entry.datasetId(), entry.physicalViewName());
        LOGGER.debug("[Catalog] install-skipped datasetId={} type={} viewName={} reason=unresolved-view-dependency message={}", entry.datasetId(), typeToken(entry.definition()), entry.physicalViewName(), ex.getMessage());
        continue;
      }
      String sql =
          "create or replace view "
              + quoteIdentifier(entry.physicalViewName())
              + " as "
              + stripTrailingSemicolon(rewritten);
      exec(session, sql);
      LOGGER.debug("[Catalog] install datasetId={} type={} viewName={}", entry.datasetId(), typeToken(entry.definition()), entry.physicalViewName());
    }
  }

  private void installDataset(DuckdbSession session, CatalogEntry entry) {
    try (DuckdbPreparedDataset prepared = prepareDatasetForCatalog(session, entry)) {
      if (!prepared.installed()) {
        return;
      }
      LOGGER.debug("[Catalog] install datasetId={} type={} viewName={}", entry.datasetId(), typeToken(entry.definition()), entry.physicalViewName());
    } catch (IllegalStateException ex) {
      if (isMissingParquetFilesError(ex)) {
        LOGGER.debug("[Catalog] install-skipped datasetId={} type={} viewName={} reason=no-matching-parquet-files", entry.datasetId(), typeToken(entry.definition()), entry.physicalViewName());
        return;
      }
      if (isS3CredentialUnavailableError(ex)) {
        LOGGER.warn("[Catalog] install-skipped datasetId={} type={} viewName={} reason=s3-credentials-unavailable message={}", entry.datasetId(), typeToken(entry.definition()), entry.physicalViewName(), Exceptions.rootMessageNullable(ex));
        return;
      }
      if (isMissingDuckdbRelationError(ex) && isDuckdbActive(entry.definition().binding() == null ? null : entry.definition().binding().duckdb())) {
        LOGGER.debug("[Catalog] install-skipped datasetId={} type={} viewName={} reason=missing-duckdb-active-relation", entry.datasetId(), typeToken(entry.definition()), entry.physicalViewName());
        return;
      }
      throw ex;
    } catch (Exception ex) {
      throw new IllegalStateException(
          "Failed to prepare dataset install for " + entry.datasetId() + ": " + ex.getMessage(),
          ex);
    }
  }

  private DuckdbPreparedDataset prepareDatasetForCatalog(
      DuckdbSession session, CatalogEntry entry) {
    DefinitionDataset definition = entry.definition();
    DefinitionDataset.View view = definition.view();
    String viewName = quoteIdentifier(entry.physicalViewName());

    if (view != null) {
      DefinitionDataset.Target target = view.target();
      String targetParquetPath = target == null ? null : Strings.trimToNull(target.parquetPath());
      if (targetParquetPath != null) {
        String parquetExpression = parquetFromPath(targetParquetPath, null, false);
        exec(
            session,
            "create or replace view " + viewName + " as select * from " + parquetExpression);
        return DuckdbPreparedDataset.installed(entry.datasetId(), entry.physicalViewName(), null);
      }
      String targetTable = target == null ? null : Strings.trimToNull(target.duckdbTableName());
      if (targetTable != null) {
        exec(
            session,
            "create or replace view " + viewName + " as select * from " + targetTable);
        return DuckdbPreparedDataset.installed(entry.datasetId(), entry.physicalViewName(), null);
      }
      if (entry.logicalView()) {
        return DuckdbPreparedDataset.noOp(entry.datasetId(), entry.physicalViewName());
      }
    }
    DuckdbDatasetPrepareRequest request =
        new DuckdbDatasetPrepareRequest(
            entry.datasetId(),
            definition,
            entry.physicalViewName(),
            session,
            DiscoveryCache,
            rowManager,
            DuckdbDatasetPrepareMode.CATALOG);
    return datasetProviderRegistry.prepare(request);
  }

  private static String rewriteSql(String sql, DatasetNameResolver resolver, CatalogSnapshot snapshot) {
    List<DatasetToken> tokens = resolver.scanTableTokensInSql(sql);
    if (tokens.isEmpty()) {
      return stripTrailingSemicolon(sql);
    }
    StringBuilder rewritten = new StringBuilder(sql.length() + (tokens.size() * 16));
    int cursor = 0;
    for (DatasetToken token : tokens) {
      DatasetId datasetId = resolver.resolveTableToken(token.normalizedToken());
      CatalogEntry entry = snapshot.entriesById().get(datasetId);
      if (entry == null) {
        throw new IllegalArgumentException("Unknown table token '" + token.token() + "'");
      }
      rewritten.append(sql, cursor, token.start());
      rewritten.append(withAlias(quoteIdentifier(entry.physicalViewName()), token));
      cursor = token.end();
    }
    rewritten.append(sql, cursor, sql.length());
    return stripTrailingSemicolon(rewritten.toString());
  }

  private String rewriteSqlForQuery(String sql, DatasetNameResolver resolver, CatalogSnapshot snapshot) {
    List<DatasetToken> tokens = resolver.scanTableTokensInSql(sql, true);
    if (tokens.isEmpty()) {
      return stripTrailingSemicolon(sql);
    }
    DataEntitlements entitlements = rowManager.dataEntitlementsForSqlPlanning();
    DataEntitlements.Identity identity = rowManager.effectiveIdentityForSqlPlanning();
    Map<DatasetId, DataRowFilterPlanner.RowFilterPlan> rowPlansByDataset = new HashMap<>();
    Map<DatasetId, DataColumnMaskPlanner.ColumnMaskPlan> maskPlansByDataset = new HashMap<>();
    StringBuilder rewritten = new StringBuilder(sql.length() + (tokens.size() * 32));
    int cursor = 0;
    for (DatasetToken token : tokens) {
      DatasetId datasetId = resolver.resolveTableToken(token.normalizedToken());
      CatalogEntry entry = snapshot.entriesById().get(datasetId);
      if (entry == null) {
        throw new IllegalArgumentException("Unknown table token '" + token.token() + "'");
      }
      DataRowFilterPlanner.RowFilterPlan rowPlan =
          rowPlansByDataset.computeIfAbsent(
              datasetId,
              ignored -> DataRowFilterPlanner.planRead(entitlements, identity, datasetId.toString()));
      if (!rowPlan.allowed()) {
        throw new IllegalStateException(firstDenyReason(rowPlan.reasons(), "row filter", datasetId));
      }

      DataColumnMaskPlanner.ColumnMaskPlan maskPlan =
          maskPlansByDataset.computeIfAbsent(
              datasetId,
              ignored ->
                  DataColumnMaskPlanner.planRead(
                      entitlements,
                      identity,
                      datasetId.toString(),
                      resolveSchemaForMasking(entry)));
      if (!maskPlan.allowed()) {
        throw new IllegalStateException(firstDenyReason(maskPlan.reasons(), "column mask", datasetId));
      }

      String replacement = quoteIdentifier(entry.physicalViewName());
      String predicate = Strings.trimToNull(rowPlan.predicateSql());
      if (predicate != null) {
        replacement = "(select * from " + replacement + " where " + predicate + ")";
      }
      String projection = Strings.trimToNull(maskPlan.projectionSql());
      if (projection != null) {
        replacement = "(select " + projection + " from " + replacement + ")";
      }
      rewritten.append(sql, cursor, token.start());
      rewritten.append(withAlias(replacement, token));
      cursor = token.end();
    }
    rewritten.append(sql, cursor, sql.length());
    return stripTrailingSemicolon(rewritten.toString());
  }

  private static String withAlias(String replacement, DatasetToken token) {
    String alias = Strings.trimToNull(token.alias());
    if (alias == null) {
      return replacement;
    }
    if (token.aliasUsesAsKeyword()) {
      return replacement + " as " + alias;
    }
    return replacement + " " + alias;
  }

  private static String firstDenyReason(
      List<String> reasons, String policyType, DatasetId datasetId) {
    if (reasons == null || reasons.isEmpty()) {
      return "DENIED: no " + policyType + " decision for dataset " + datasetId;
    }
    String reason = Strings.trimToNull(reasons.get(0));
    if (reason == null) {
      return "DENIED: no " + policyType + " decision for dataset " + datasetId;
    }
    return reason;
  }

  private SchemaModel resolveSchemaForMasking(CatalogEntry entry) {
    String schemaName = resolveSchemaName(entry);
    if (schemaName == null) {
      return null;
    }
    Optional<DefinitionSchema> definition = resolveSchemaDefinition(entry.datasetId().group(), schemaName);
    return definition.map(SchemaModelMapper::fromDefinition).orElse(null);
  }

  private Optional<DefinitionSchema> resolveSchemaDefinition(String datasetGroup, String schemaName) {
    List<String> groups = new ArrayList<>();
    addDistinct(groups, datasetGroup);
    addDistinct(groups, sessionGroup);
    addDistinct(groups, PLATFORM_GROUP);
    for (String group : groups) {
      Optional<DefinitionSchema> schema = DiscoveryCache.get(DefinitionSchema.class, group, schemaName);
      if (schema.isPresent() && isTabularSchema(schema.get())) {
        return schema;
      }
    }
    return Optional.empty();
  }

  private String resolveSchemaName(CatalogEntry entry) {
    DefinitionDataset definition = entry == null ? null : entry.definition();
    DefinitionDataset.View view = definition == null ? null : definition.view();
    String viewSchemaName = view == null ? null : Strings.trimToNull(view.schemaName());
    if (viewSchemaName != null) {
      return viewSchemaName;
    }
    String ingestSchemaName = resolveSchemaNameFromIngest(entry.datasetId());
    if (ingestSchemaName != null) {
      return ingestSchemaName;
    }
    String deterministic =
        ingestModelMapper.deterministicSchemaName(
            entry.datasetId().group(), entry.datasetId().namespace(), entry.datasetId().name());
    return resolveSchemaDefinition(entry.datasetId().group(), deterministic).isPresent()
        ? deterministic
        : null;
  }

  private String resolveSchemaNameFromIngest(DatasetId datasetId) {
    List<String> schemaNames = new ArrayList<>();
    List<DefinitionIngest> ingests = DiscoveryCache.list(DefinitionIngest.class, datasetId.group());
    for (DefinitionIngest ingest : ingests) {
      if (ingest == null) {
        continue;
      }
      String ingestDatasetId = Strings.trimToNull(ingest.datasetId());
      String outputSchemaId = Strings.trimToNull(ingest.outputSchemaId());
      if (ingestDatasetId == null || outputSchemaId == null) {
        continue;
      }
      try {
        if (DatasetId.parse(ingestDatasetId).equals(datasetId)) {
          schemaNames.add(outputSchemaId);
        }
      } catch (RuntimeException ignore) {
        // skip invalid ingest references for deterministic planning
      }
    }
    if (schemaNames.isEmpty()) {
      return null;
    }
    schemaNames.sort(String.CASE_INSENSITIVE_ORDER.thenComparing(Comparator.naturalOrder()));
    return schemaNames.get(0);
  }

  private static boolean isTabularSchema(DefinitionSchema schema) {
    DefinitionSchema.SchemaType schemaType = schema == null ? null : schema.schemaType();
    return schemaType == null || schemaType == DefinitionSchema.SchemaType.TABULAR;
  }

  private static void addDistinct(List<String> values, String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized != null && !values.contains(normalized)) {
      values.add(normalized);
    }
  }

  private List<DatasetId> expandDependencies(List<DatasetId> directDatasets, CatalogSnapshot current) {
    LinkedHashSet<DatasetId> visited = new LinkedHashSet<>();
    List<DatasetId> roots = new ArrayList<>(directDatasets);
    roots.sort(DATASET_ID_ORDER);
    for (DatasetId root : roots) {
      collectDependencies(root, current, visited, new LinkedHashSet<>());
    }
    return List.copyOf(visited);
  }

  private static void collectDependencies(
      DatasetId datasetId,
      CatalogSnapshot current,
      LinkedHashSet<DatasetId> visited,
      LinkedHashSet<DatasetId> visiting) {
    if (visited.contains(datasetId)) {
      return;
    }
    if (!visiting.add(datasetId)) {
      return;
    }
    CatalogEntry entry = current.entriesById().get(datasetId);
    if (entry != null) {
      List<DatasetId> dependencies = new ArrayList<>(entry.dependencies());
      dependencies.sort(DATASET_ID_ORDER);
      for (DatasetId dependency : dependencies) {
        collectDependencies(dependency, current, visited, visiting);
      }
    }
    visiting.remove(datasetId);
    visited.add(datasetId);
  }

  private CatalogEntry toEntry(DefinitionDataset definition) {
    DatasetId datasetId = requireDatasetId(definition);
    String namespace = datasetId.namespace();
    String name = datasetId.name();
    String viewName = duckdbViewName(datasetId.group(), namespace, name);
    boolean logicalView = isLogicalView(definition);
    return new CatalogEntry(
        datasetId,
        definition,
        namespace,
        name,
        viewName,
        List.of(),
        logicalView);
  }

  private static boolean isLogicalView(DefinitionDataset definition) {
    if (definition == null || definition.view() == null) {
      return false;
    }
    DefinitionDataset.Target target = definition.view().target();
    if (target == null) {
      return true;
    }
    return Strings.trimToNull(target.duckdbTableName()) == null
        && Strings.trimToNull(target.parquetPath()) == null;
  }

  private DefinitionDataset fetchDefinition(DatasetId datasetId) {
    String name = datasetId.namespace() + "/" + datasetId.name();
    Optional<DefinitionDataset> definition =
        DiscoveryCache.get(DefinitionDataset.class, datasetId.group(), name);
    return definition.orElseThrow(
        () -> new IllegalArgumentException("Dataset definition not found for install: " + datasetId));
  }

  private static DatasetId tryDatasetId(DefinitionDataset definition) {
    try {
      return requireDatasetId(definition);
    } catch (RuntimeException ex) {
      return null;
    }
  }

  private static DatasetId requireDatasetId(DefinitionDataset definition) {
    Objects.requireNonNull(definition, "definition");
    DefinitionEnvelope envelope = Objects.requireNonNull(definition.envelope(), "definition.envelope");
    String group = Strings.requireNonBlank(envelope.group(), "definition.envelope.group");
    String namespace = Strings.requireNonBlank(definition.namespace(), "definition.namespace");
    String baseName = datasetBaseName(definition);
    return DatasetId.of(group, namespace, baseName);
  }

  private static String datasetBaseName(DefinitionDataset definition) {
    DefinitionEnvelope envelope = Objects.requireNonNull(definition.envelope(), "definition.envelope");
    String namespace = Strings.requireNonBlank(definition.namespace(), "definition.namespace");
    String envelopeName = Strings.requireNonBlank(envelope.name(), "definition.envelope.name");
    String prefix = namespace + "/";
    if (!envelopeName.startsWith(prefix)) {
      throw new IllegalStateException(
          "Dataset definition invalid envelope.name. Expected namespace/baseName with namespace='"
              + namespace
              + "' and envelope.name='"
              + envelopeName
              + "'");
    }
    String base = envelopeName.substring(prefix.length());
    if (base.isBlank() || base.indexOf('/') >= 0) {
      throw new IllegalStateException(
          "Dataset definition invalid envelope.name. Expected namespace/baseName with namespace='"
              + namespace
              + "' and envelope.name='"
              + envelopeName
              + "'");
    }
    return base;
  }

  private static boolean isActive(DefinitionDataset definition) {
    if (definition == null || definition.envelope() == null) {
      return false;
    }
    DefinitionEnvelope envelope = definition.envelope();
    if (Boolean.FALSE.equals(envelope.enabled())) {
      return false;
    }
    String status = Strings.trimToNull(envelope.lifecycleStatus());
    return status == null || "ACTIVE".equalsIgnoreCase(status);
  }

  private static String applyLimit(String sql, Integer limit) {
    if (limit == null) {
      return sql;
    }
    if (limit <= 0) {
      throw new IllegalArgumentException("limit must be > 0");
    }
    return "select * from (" + sql + ") as __fabric_sql limit " + limit;
  }

  private static void exec(DuckdbSession session, String sql) {
    if (sql == null || sql.isBlank()) {
      return;
    }
    String[] statements = sql.split(";\\s*\\n|;\\s*$|;\\s*");
    for (String statementSql : statements) {
      String trimmed = Strings.trimToNull(statementSql);
      if (trimmed == null) {
        continue;
      }
      try (Statement statement = session.createStatement()) {
        statement.execute(trimmed);
      } catch (SQLException ex) {
        throw new IllegalStateException("DuckDB SQL failed: " + ex.getMessage(), ex);
      }
    }
  }

  private static void dropView(DuckdbSession session, DatasetId datasetId, String viewName) {
    String sql = "drop view if exists " + quoteIdentifier(viewName);
    exec(session, sql);
    LOGGER.debug("[Catalog] drop datasetId={} viewName={}", datasetId, viewName);
  }

  private void putQualifiedPreferSessionGroup(
      Map<String, DatasetId> qualified, String key, DatasetId candidate) {
    DatasetId existing = qualified.get(key);
    if (existing == null) {
      qualified.put(key, candidate);
      return;
    }
    boolean existingSession = sessionGroup.equals(existing.group());
    boolean candidateSession = sessionGroup.equals(candidate.group());
    if (!existingSession && candidateSession) {
      qualified.put(key, candidate);
    }
  }

  private static List<String> metadataTableTypes(CatalogEntry entry) {
    if (entry == null) {
      return List.of("TABLE");
    }
    if (entry.logicalView()) {
      return List.of("VIEW");
    }
    DefinitionDataset definition = entry.definition();
    DefinitionDataset.Binding binding = definition == null ? null : definition.binding();
    DefinitionDataset.DuckdbBinding duckdb = binding == null ? null : binding.duckdb();
    if (isDuckdbActive(duckdb)) {
      return List.of("TABLE", "STREAM");
    }
    return List.of("TABLE");
  }
