import java.nio.file.Path;

import java.security.PrivilegedActionException;
import java.security.PrivilegedExceptionAction;
import java.util.Base64;
import java.util.HexFormat;
import java.util.LinkedHashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.UUID;
import javax.security.auth.Subject;
import javax.security.auth.callback.Callback;
import javax.security.auth.callback.CallbackHandler;
import javax.security.auth.callback.UnsupportedCallbackException;
import javax.security.auth.kerberos.KerberosPrincipal;
import javax.security.auth.login.AppConfigurationEntry;
import javax.security.auth.login.Configuration;
import javax.security.auth.login.LoginContext;
import javax.security.auth.login.LoginException;
import org.ietf.jgss.GSSContext;
import org.ietf.jgss.GSSCredential;
import org.ietf.jgss.GSSException;
import org.ietf.jgss.GSSManager;
import org.ietf.jgss.GSSName;
import org.ietf.jgss.Oid;

import org.pronzato.fabric.api.util.Strings;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public final class KerberosClient implements AutoCloseable {
  private static final Logger LOGGER = LoggerFactory.getLogger(KerberosClient.class);
  private static final String LOGIN_CONTEXT_PREFIX = "FabricKerberosClient";
  private static final String JVM_KRB5_CONF = "java.security.krb5.conf";
  private static final String JVM_KRB5_CONFIG = "sun.security.krb5.config";
  private static final String JVM_USE_SUBJECT_CREDS_ONLY = "javax.security.auth.useSubjectCredsOnly";
  private static final String JVM_KRB5_DEBUG = "sun.security.krb5.debug";
  private static final Object JVM_KRB5_LOCK = new Object();
  private static final Oid SPNEGO_OID = oid("1.3.6.1.5.5.2");
  private static Path installedJvmKrb5Path;
  private static String installedJvmKrb5Fingerprint;

  private final KerberosConfig config;
  private final String loginContextName =
      LOGIN_CONTEXT_PREFIX + "-" + UUID.randomUUID().toString().replace("-", "");

  private Subject subject;
  private LoginContext loginContext;
  private Path resolvedKeytabPath;
  private Path resolvedKrb5Path;
  private Path materializedKeytabPath;
  private Path materializedKrb5Path;
  private boolean started;

  public KerberosClient(KerberosConfig config) {
    this.config = Objects.requireNonNull(config, "config");
    this.config.validate();
  }

  public KerberosConfig getConfig() {
    return config;
  }

  public Path getResolvedKeytabPath() {
    return resolvedKeytabPath;
  }

  public Path getResolvedKrb5Path() {
    return resolvedKrb5Path;
  }

  public static void ensureJvmKrb5Configuration(
      Path resolvedKrb5Path, boolean debug, String realmHint) {
    try {
      installJvmKrb5ForStart(resolvedKrb5Path, debug, realmHint);
    } catch (Exception ex) {
      throw new IllegalStateException(
          "Failed to configure JVM Kerberos bootstrap for krb5.conf=" + resolvedKrb5Path, ex);
    }
  }

  public synchronized KerberosClient start() {
    if (started) {
      return this;
    }
    if (!config.isEnabled()) {
      started = true;
      return this;
    }
    try {
      resolvedKeytabPath = resolveKeytabPath(config);
      resolvedKrb5Path = resolveKrb5Path(config.getKrb5ConfigSource());
      installJvmKrb5ForStart(resolvedKrb5Path, config.isDebug(), config.getRealm());
      login();
      started = true;
      return this;
    } catch (Exception ex) {
      close();
      throw new IllegalStateException("Failed to start Kerberos context '" + config.getName() + "'", ex);
    }
  }

  public synchronized Subject getSubject() {
    ensureActive();
    if (subject == null) {
      throw new IllegalStateException("Kerberos Subject is not initialized");
    }
    return subject;
  }

  @SuppressWarnings("removal")
  public <T> T doAs(PrivilegedExceptionAction<T> action) throws Exception {
    Objects.requireNonNull(action, "action");
    try {
      return Subject.doAs(getSubject(), action);
    } catch (PrivilegedActionException ex) {
      throw ex.getException();
    }
  }

  @SuppressWarnings("removal")
  public void doAs(Runnable action) {
    Objects.requireNonNull(action, "action");
    Subject.doAs(
        getSubject(),
        (PrivilegedAction<Void>)
            () -> {
              action.run();
              return null;
            });
  }

  public String buildNegotiateHeader(String spn) {
    return buildNegotiateHeader(spn, getSubject());
  }

  public String buildNegotiateHeaderForSpn(String spn) {
    return buildNegotiateHeader(spn);
  }

  public String buildNegotiateHeaderForSpn(String spn, Subject subject) {
    return buildNegotiateHeader(spn, subject);
  }

  public String buildNegotiateHeaderForHostHttp(String hostFqdn) {
    return buildNegotiateHeader(KerberosSpn.httpSpnForHost(hostFqdn));
  }

  @SuppressWarnings("removal")
  public String buildNegotiateHeader(String spn, Subject subject) {
    ensureActive();
    String target = require(spn, "spn");
    try {
      byte[] token = mintSpnegoTokenAsSubject(target, Objects.requireNonNull(subject, "subject"));
      return "Negotiate " + Base64.getEncoder().encodeToString(token);
    } catch (Exception ex) {
      throw new IllegalStateException(
          "Failed to build Negotiate header for SPN '" + target + "'", ex);
    }
  }

  @SuppressWarnings("removal")
  public byte[] mintSpnegoTokenAsSubject(String spn, Subject subject) throws Exception {
    String target = require(spn, "spn");
    Subject subjectForMint = Objects.requireNonNull(subject, "subject");
    try {
      return Subject.doAs(
          subjectForMint,
          (PrivilegedExceptionAction<byte[]>)
              () -> createInitToken(target, config.isAllowSpnegoFallbackToKrb5()));
    } catch (PrivilegedActionException ex) {
      throw ex.getException();
    }
  }

  public synchronized void reloadKrb5Config(KerberosKrb5Source source) {
    ensureActive();
    Path oldResolved = resolvedKrb5Path;
    Path oldMaterialized = materializedKrb5Path;
    LoginContext oldLogin = loginContext;
    Subject oldSubject = subject;
    try {
      resolvedKrb5Path = resolveKrb5Path(Objects.requireNonNull(source, "source"));
      installJvmKrb5ForReload(resolvedKrb5Path, config.isDebug(), config.getRealm());
      login();
      if (oldLogin != null) {
        oldLogin.logout();
      }
      if (oldMaterialized != null && !oldMaterialized.equals(materializedKrb5Path)) {
        Io.deleteIfExists(oldMaterialized);
      }
    } catch (Exception ex) {
      resolvedKrb5Path = oldResolved;
      subject = oldSubject;
      loginContext = oldLogin;
      restoreJvmKrb5AfterFailedReload(oldResolved, config.isDebug(), config.getRealm());
      throw new IllegalStateException("Failed to reload krb5 source", ex);
    }
  }

  public KerberosResult acceptSpnego(byte[] token) {
    return acceptSpnego(token, null);
  }

  public KerberosResult acceptSpnego(byte[] token, String acceptorPrincipal) {
    Objects.requireNonNull(token, "token");
    if (token.length == 0) {
      throw new IllegalArgumentException("token is required");
    }
    if (acceptorPrincipal != null) {
      require(acceptorPrincipal, "acceptorPrincipal");
    }
    try {
      return doAs(() -> acceptToken(token));
    } catch (Exception ex) {
      throw new IllegalStateException("Failed to accept SPNEGO token", ex);
    }
  }

  public KerberosResult acceptSpnegoToken(byte[] token) {
    return acceptSpnego(token);
  }

  public KerberosResult acceptSpnegoToken(byte[] token, String acceptorPrincipal) {
    return acceptSpnego(token, acceptorPrincipal);
  }

  @SuppressWarnings("removal")
  public KerberosResult acceptSpnegoTokenAsSubject(
      byte[] token, String acceptorPrincipal, Subject acceptorSubject) {
    Objects.requireNonNull(token, "token");
    if (token.length == 0) {
      throw new IllegalArgumentException("token is required");
    }
    require(acceptorPrincipal, "acceptorPrincipal");
    Subject subjectForAccept = Objects.requireNonNull(acceptorSubject, "acceptorSubject");
    try {
      return Subject.doAs(
          subjectForAccept,
          (PrivilegedExceptionAction<KerberosResult>)
              () -> {
                GSSManager manager = GSSManager.getInstance();
                GSSContext context = manager.createContext((GSSCredential) null);
                try {
                  byte[] response = context.acceptSecContext(token, 0, token.length);
                  if (!context.isEstablished()) {
                    throw new IllegalStateException("SPNEGO accept did not establish context");
                  }
                  String sourcePrincipal =
                      context.getSrcName() == null ? null : context.getSrcName().toString();
                  Subject delegated = delegatedSubject(context);
                  return new KerberosResult(sourcePrincipal, true, response, delegated);
                } finally {
                  context.dispose();
                }
              });
    } catch (PrivilegedActionException ex) {
      throw new IllegalStateException("Failed to accept SPNEGO token", ex.getException());
    }
  }

  public KerberosResult acceptSpnegoTokenAsSubject(
      byte[] token,
      String acceptorPrincipal,
      Subject acceptorSubject,
      String keytabPath,
      String krb5Path) {
    return acceptSpnegoTokenAsSubject(token, acceptorPrincipal, acceptorSubject);
  }

  @Override
  public synchronized void close() {
    started = false;
    if (loginContext != null) {
      try {
        loginContext.logout();
      } catch (LoginException ignore) {
      }
      loginContext = null;
    }
    subject = null;
    resolvedKeytabPath = null;
    resolvedKrb5Path = null;
    Io.deleteIfExists(materializedKeytabPath);
    Io.deleteIfExists(materializedKrb5Path);
    materializedKeytabPath = null;
    materializedKrb5Path = null;
  }

  private KerberosResult acceptToken(byte[] token) throws GSSException {
    GSSManager manager = GSSManager.getInstance();
    GSSContext ctx = manager.createContext((GSSCredential) null);
    try {
      byte[] response = ctx.acceptSecContext(token, 0, token.length);
      String principal = ctx.getSrcName() == null ? null : ctx.getSrcName().toString();
      Subject delegated = delegatedSubject(ctx);
      return new KerberosResult(principal, ctx.isEstablished(), response, delegated);
    } finally {
      ctx.dispose();
    }
  }

  private void ensureActive() {
    if (!started) {
      throw new IllegalStateException("Kerberos context is not started");
    }
    if (!config.isEnabled()) {
      throw new IllegalStateException("Kerberos context is disabled");
    }
  }

  private void login() throws LoginException {
    LoginContext ctx =
        new LoginContext(
            loginContextName,
            null,
            NoPromptCallbackHandler.INSTANCE,
            new LoginConfig(
                config.getEffectivePrincipal(),
                resolvedKeytabPath,
                resolvedKrb5Path,
                config.isDebug()));
    ctx.login();
    loginContext = ctx;
    subject = Objects.requireNonNull(ctx.getSubject(), "subject");
  }

  private Path resolveKeytabPath(KerberosConfig cfg) throws Exception {
    if (cfg.getKeytabPath() != null) {
      if (!Files.isRegularFile(cfg.getKeytabPath())) {
        throw new IllegalArgumentException("Kerberos keytab path does not exist: " + cfg.getKeytabPath());
      }
      return cfg.getKeytabPath();
    }
    byte[] bytes = cfg.getKeytabBytes();
    if (bytes == null || bytes.length == 0) {
      throw new IllegalArgumentException("Kerberos keytab bytes are required");
    }
    Path temp = Io.writeTempFile("fabric-kerberos-keytab-", ".keytab", bytes);
    materializedKeytabPath = temp;
    return temp;
  }

  private Path resolveKrb5Path(KerberosKrb5Source source) throws Exception {
    KerberosKrb5Source value = source == null ? KerberosKrb5Source.DEFAULT : source;
    if (value instanceof KerberosKrb5Source.FilePath filePath) {
      Path path = filePath.getPath().toAbsolutePath().normalize();
      if (!Files.isRegularFile(path)) {
        throw new IllegalArgumentException("krb5 path does not exist: " + path);
      }
      materializedKrb5Path = null;
      return path;
    }
    if (value instanceof KerberosKrb5Source.Bytes bytes) {
      Path temp = Io.writeTempFile("fabric-krb5-", ".conf", bytes.getData());
      materializedKrb5Path = temp;
      return temp;
    }
    String resourcePath;
    if (value instanceof KerberosKrb5Source.Classpath classpath) {
      resourcePath = classpath.getResourcePath();
    } else {
      resourcePath = KerberosKrb5Source.DEFAULT_RESOURCE;
    }
    Path temp = materializeClasspathKrb5(resourcePath);
    materializedKrb5Path = temp;
    return temp;
  }

  private static void installJvmKrb5ForStart(Path resolvedKrb5Path, boolean debug, String realmHint)
      throws Exception {
    Path normalized = requireExistingPath(resolvedKrb5Path, "resolvedKrb5Path");
    String fingerprint = sha256Hex(Files.readAllBytes(normalized));
    synchronized (JVM_KRB5_LOCK) {
      if (installedJvmKrb5Path != null
          && !Objects.equals(installedJvmKrb5Fingerprint, fingerprint)) {
        throw new IllegalStateException(
            "Kerberos runtime requested krb5.conf "
                + normalized
                + " but JVM is already configured with "
                + installedJvmKrb5Path
                + ". The JDK uses a process-global krb5.conf; reuse the same krb5.conf for all Kerberos contexts or restart the process.");
      }
      applyJvmKrb5Properties(normalized, debug);
      installedJvmKrb5Path = normalized;
      installedJvmKrb5Fingerprint = fingerprint;
      Krb5Diagnostic diagnostic = parseKrb5Diagnostic(normalized);
      String realm = Strings.trimToNull(realmHint);
      if (realm == null) {
        realm = diagnostic.realm();
      }
      LOGGER.debug("[Kerberos] resolvedKrb5Path={} realm={} kdcHostPort={}", normalized, realm == null ? "<unknown>" : realm, diagnostic.kdcHostPort() == null ? "<unknown>" : diagnostic.kdcHostPort());
    }
  }

  private static void installJvmKrb5ForReload(Path resolvedKrb5Path, boolean debug, String realmHint)
      throws Exception {
    Path normalized = requireExistingPath(resolvedKrb5Path, "resolvedKrb5Path");
    String fingerprint = sha256Hex(Files.readAllBytes(normalized));
    synchronized (JVM_KRB5_LOCK) {
      applyJvmKrb5Properties(normalized, debug);
      installedJvmKrb5Path = normalized;
      installedJvmKrb5Fingerprint = fingerprint;
      Krb5Diagnostic diagnostic = parseKrb5Diagnostic(normalized);
      String realm = Strings.trimToNull(realmHint);
      if (realm == null) {
        realm = diagnostic.realm();
      }
      LOGGER.debug("[Kerberos] krb5 reload applied resolvedKrb5Path={} realm={} kdcHostPort={}", normalized, realm == null ? "<unknown>" : realm, diagnostic.kdcHostPort() == null ? "<unknown>" : diagnostic.kdcHostPort());
    }
  }

  private static void restoreJvmKrb5AfterFailedReload(
      Path resolvedKrb5Path, boolean debug, String realmHint) {
    if (resolvedKrb5Path == null) {
      return;
    }
    try {
      installJvmKrb5ForReload(resolvedKrb5Path, debug, realmHint);
    } catch (Exception ex) {
      LOGGER.warn("[Kerberos] failed to restore JVM krb5 after reload failure", ex);
    }
  }

  private static void applyJvmKrb5Properties(Path resolvedKrb5Path, boolean debug) {
    String absolute = resolvedKrb5Path.toAbsolutePath().normalize().toString();
    System.setProperty(JVM_KRB5_CONF, absolute);
    System.setProperty(JVM_KRB5_CONFIG, absolute);
    System.setProperty(JVM_USE_SUBJECT_CREDS_ONLY, "false");
    String configuredDebug = System.getProperty(JVM_KRB5_DEBUG);
    boolean externallyConfigured = configuredDebug != null;
    boolean externallyEnabled = Boolean.parseBoolean(configuredDebug);
    // JDK krb5 debug is intentionally opt-in and should only be enabled for troubleshooting.
    if (debug || externallyEnabled) {
      System.setProperty(JVM_KRB5_DEBUG, "true");
    } else if (!externallyConfigured) {
      System.clearProperty(JVM_KRB5_DEBUG);
    }
  }

  private static Path materializeClasspathKrb5(String resourcePath) throws Exception {
    byte[] data = Io.readClasspathBytes(resourcePath, KerberosConfig.class);
    return Io.writeTempFile("fabric-kerberos-krb5-", ".conf", data);
  }

  private static String sha256Hex(byte[] data) throws Exception {
    MessageDigest digest = MessageDigest.getInstance("SHA-256");
    return HexFormat.of().formatHex(digest.digest(data == null ? new byte[0] : data));
  }

  private static Path requireExistingPath(Path value, String label) {
    if (value == null) {
      throw new IllegalArgumentException(label + " is required");
    }
    Path normalized = value.toAbsolutePath().normalize();
    if (!Files.isRegularFile(normalized)) {
      throw new IllegalArgumentException(label + " does not exist: " + normalized);
    }
    return normalized;
  }

  private static Krb5Diagnostic parseKrb5Diagnostic(Path path) {
    String realm = null;
    String kdc = null;
    try {
      List<String> lines = Files.readAllLines(path);
      for (String line : lines) {
        String trimmed = stripComment(line);
        if (trimmed.isEmpty()) {
          continue;
        }
        String lower = trimmed.toLowerCase(java.util.Locale.ROOT);
        if (realm == null && lower.startsWith("default_realm")) {
          realm = parseAssignmentValue(trimmed);
          continue;
        }
        if (kdc == null && lower.startsWith("kdc")) {
          kdc = parseAssignmentValue(trimmed);
        }
        if (realm != null && kdc != null) {
          break;
        }
      }
    } catch (Exception ignore) {
    }
    return new Krb5Diagnostic(Strings.trimToNull(realm), Strings.trimToNull(kdc));
  }

  private static String stripComment(String line) {
    if (line == null) {
      return "";
    }
    String value = line;
    int hash = value.indexOf('#');
    if (hash >= 0) {
      value = value.substring(0, hash);
    }
    int semicolon = value.indexOf(';');
    if (semicolon >= 0) {
      value = value.substring(0, semicolon);
    }
    return value.trim();
  }

  private static String parseAssignmentValue(String line) {
    if (line == null) {
      return null;
    }
    int eq = line.indexOf('=');
    if (eq < 0 || eq + 1 >= line.length()) {
      return null;
    }
    return Strings.trimToNull(line.substring(eq + 1));
  }

  private record Krb5Diagnostic(String realm, String kdcHostPort) {}

  private static byte[] createInitToken(String spn, boolean allowFallback) throws GSSException {
    if (allowFallback) {
      // SPNEGO-only by design to match standalone runtime behavior.
      return createInitToken(spn);
    }
    return createInitToken(spn);
  }

  private static byte[] createInitToken(String spn) throws GSSException {
    GSSManager manager = GSSManager.getInstance();
    String normalized = require(spn, "spn");
    GSSName server = manager.createName(normalized, GSSName.NT_USER_NAME);
    GSSContext ctx = manager.createContext(server, SPNEGO_OID, null, GSSContext.DEFAULT_LIFETIME);
    try {
      ctx.requestMutualAuth(false);
      ctx.requestCredDeleg(false);
      ctx.requestConf(false);
      ctx.requestInteg(true);
      byte[] token = ctx.initSecContext(new byte[0], 0, 0);
      if (token == null || token.length == 0) {
        throw new IllegalStateException("SPNEGO init returned empty token");
      }
      return token;
    } finally {
      ctx.dispose();
    }
  }

  private static Subject delegatedSubject(GSSContext ctx) throws GSSException {
    GSSCredential credential = ctx.getDelegCred();
    if (credential == null) {
      return null;
    }
    Subject delegated = new Subject();
    delegated.getPrivateCredentials().add(credential);
    GSSName name = credential.getName();
    if (name != null) {
      delegated.getPrincipals().add(new KerberosPrincipal(name.toString()));
    }
    return delegated;
  }

  private static String require(String value, String label) {
    String trimmed = Strings.trimToNull(value);
    if (trimmed == null) {
      throw new IllegalArgumentException(label + " is required");
    }
    return trimmed;
  }

  private static Oid oid(String value) {
    try {
      return new Oid(value);
    } catch (GSSException ex) {
      throw new ExceptionInInitializerError(ex);
    }
  }

  private static final class NoPromptCallbackHandler implements CallbackHandler {
    private static final NoPromptCallbackHandler INSTANCE = new NoPromptCallbackHandler();

    @Override
    public void handle(Callback[] callbacks) throws UnsupportedCallbackException {
      if (callbacks != null && callbacks.length > 0) {
        throw new UnsupportedCallbackException(
            callbacks[0], "Kerberos interactive prompts are disabled");
      }
    }
  }

  private static final class LoginConfig extends Configuration {
    private final AppConfigurationEntry[] entries;

    private LoginConfig(String principal, Path keytabPath, Path krb5Path, boolean debug) {
      this.entries =
          new AppConfigurationEntry[] {
            new AppConfigurationEntry(
                "com.sun.security.auth.module.Krb5LoginModule",
                AppConfigurationEntry.LoginModuleControlFlag.REQUIRED,
                options(principal, keytabPath, krb5Path, debug))
          };
    }

    @Override
    public AppConfigurationEntry[] getAppConfigurationEntry(String name) {
      return entries;
    }

    private static Map<String, String> options(
        String principal, Path keytabPath, Path krb5Path, boolean debug) {
      Map<String, String> options = new LinkedHashMap<>();
      options.put("refreshKrb5Config", "true");
      options.put("isInitiator", "true");
      options.put("doNotPrompt", "true");
      options.put("renewTGT", "false");
      options.put("useKeyTab", "true");
      options.put("keyTab", keytabPath.toAbsolutePath().toString());
      options.put("storeKey", "true");
      options.put("useTicketCache", "false");
      options.put("principal", principal);
      options.put("debug", Boolean.toString(debug));
      if (krb5Path != null) {
        String path = krb5Path.toAbsolutePath().toString();
        options.put("krb5Config", path);
        options.put("krb5Conf", path);
      }
      return options;
    }
  }
}
