package org.pronzato.fabric.api.control.entitlements.provider;

import java.lang.reflect.Method;
import java.util.ArrayList;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.Set;
import org.pronzato.fabric.api.control.discovery.CanonicalIds;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryCache;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionDataset;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEntitlement;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEnvelope;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionEnvironment;
import org.pronzato.fabric.api.control.discovery.definitions.DefinitionService;
import org.pronzato.fabric.api.control.discovery.definitions.common.EntitlementsProvider;
import org.pronzato.fabric.api.control.entitlements.core.EntitlementsScopeType;
import org.pronzato.fabric.api.control.entitlements.core.data.DataEntitlements;
import org.pronzato.fabric.api.control.entitlements.core.services.ServiceEntitlements;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public final class EntitlementProviderSelector {
  private static final Logger LOGGER = LoggerFactory.getLogger(EntitlementProviderSelector.class);
  private static final String PLATFORM_GROUP = "platform";
  private static final String ENVIRONMENT_NAME = "environment";
  private static final String KEY_SUBJECT_PROVIDER = "entitlements.subjectProvider";
  private static final String KEY_PROVIDER = "entitlements.provider";

  private final DiscoveryCache DiscoveryCache;
  private final EntitlementProviderRegistry registry;
  private final String configuredEnvironmentDefaultProviderId;

  public EntitlementProviderSelector(DiscoveryCache DiscoveryCache, EntitlementProviderRegistry registry) {
    this(DiscoveryCache, registry, null);
  }

  public EntitlementProviderSelector(
      DiscoveryCache DiscoveryCache,
      EntitlementProviderRegistry registry,
      String configuredEnvironmentDefaultProviderId) {
    this.DiscoveryCache = Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    this.registry = Objects.requireNonNull(registry, "registry");
    this.configuredEnvironmentDefaultProviderId = trimToNull(configuredEnvironmentDefaultProviderId);
  }

  public Selection select(EntitlementsScopeType scopeType, String resourceId) {
    EntitlementsScopeType normalizedScopeType = Objects.requireNonNull(scopeType, "scopeType");
    String normalizedResourceId = canonicalResourceId(normalizedScopeType, resourceId);

    ProviderChoice resourceProvider = resourceProviderChoice(normalizedScopeType, normalizedResourceId);
    ProviderChoice namespaceProvider = namespaceProviderChoice(normalizedScopeType, normalizedResourceId);
    ProviderChoice environmentProvider = environmentProviderChoice();
    ProviderChoice configured = firstNonNull(resourceProvider, namespaceProvider, environmentProvider);

    String defaultProvider = defaultProviderForTrace();
    Selection selected;
    if (configured != null) {
      selected = explicitSelection(configured, normalizedScopeType, normalizedResourceId);
    } else {
      EntitlementSubjectProvider provider = registry.defaultProvider();
      if (provider == null) {
        throw new IllegalStateException("No default entitlement subject provider is registered");
      }
      selected =
          new Selection(defaultProviderId(provider), provider, SelectionSource.REGISTRY_DEFAULT, false);
    }

    if (LOGGER.isDebugEnabled()) {
      LOGGER.debug(
          "[Entitlements][ProviderTrace] resource={} resourceProvider={} namespaceProvider={} environmentProvider={} defaultProvider={} selectedProvider={}",
          normalizedResourceId,
          providerId(resourceProvider),
          providerId(namespaceProvider),
          providerId(environmentProvider),
          defaultProvider,
          selected.providerId());
      LOGGER.debug(
          "[Entitlements][ProviderSelected] resource={} provider={} source={}",
          normalizedResourceId,
          selectedProviderForLog(selected, defaultProvider),
          sourceLabel(selected.source()));
    }

    return selected;
  }

  private Selection explicitSelection(
      ProviderChoice configured, EntitlementsScopeType scopeType, String resourceId) {
    EntitlementSubjectProvider provider = registry.provider(configured.providerId());
    if (provider == null) {
      throw new IllegalStateException(
          "Unknown entitlement subject provider '"
              + configured.providerId()
              + "' configured at "
              + configured.source()
              + " scopeType="
              + scopeType.name()
              + " resourceId="
              + resourceId);
    }
    return new Selection(configured.providerId(), provider, configured.source(), true);
  }

  private ProviderChoice resourceProviderChoice(EntitlementsScopeType scopeType, String resourceId) {
    String fromDefinition = resourceDefinitionProviderId(scopeType, resourceId);
    if (fromDefinition != null) {
      return new ProviderChoice(fromDefinition, SelectionSource.RESOURCE);
    }

    String fromEntitlements = entitlementsScopeProviderId(scopeType, resourceId);
    if (fromEntitlements != null) {
      return new ProviderChoice(fromEntitlements, SelectionSource.RESOURCE);
    }
    return null;
  }

  private ProviderChoice namespaceProviderChoice(EntitlementsScopeType scopeType, String resourceId) {
    String namespaceId = namespaceResourceId(scopeType, resourceId);
    if (namespaceId == null) {
      return null;
    }
    String fromEntitlements = entitlementsScopeProviderId(EntitlementsScopeType.NAMESPACE, namespaceId);
    if (fromEntitlements == null) {
      return null;
    }
    return new ProviderChoice(fromEntitlements, SelectionSource.NAMESPACE);
  }

  private ProviderChoice environmentProviderChoice() {
    String configured = trimToNull(configuredEnvironmentDefaultProviderId);
    if (configured != null) {
      return new ProviderChoice(configured, SelectionSource.ENVIRONMENT);
    }

    DefinitionEnvironment environment = findEnvironmentDefinition();
    if (environment == null) {
      return null;
    }
    String value = providerIdFromEnvironment(environment.entitlements());
    if (value == null) {
      return null;
    }
    return new ProviderChoice(value, SelectionSource.ENVIRONMENT);
  }

  private String resourceDefinitionProviderId(EntitlementsScopeType scopeType, String resourceId) {
    CanonicalIds.ParsedId parsed = CanonicalIds.parse(resourceId);
    if (scopeType == EntitlementsScopeType.SERVICE && parsed.isService()) {
      DefinitionService definition = findServiceDefinition(parsed.group(), parsed.name());
      return providerIdFromService(definition);
    }
    if (scopeType == EntitlementsScopeType.DATASET && parsed.isDataset()) {
      DefinitionDataset definition = findDatasetDefinition(parsed.group(), parsed.namespace(), parsed.name());
      return providerIdFromDataset(definition);
    }
    return null;
  }

  private String providerIdFromService(DefinitionService definition) {
    if (definition == null) {
      return null;
    }
    String typed = providerIdFromTypedConfig(definition.entitlements());
    if (typed != null) {
      return typed;
    }
    return providerIdFromObject(definition);
  }

  private String providerIdFromDataset(DefinitionDataset definition) {
    if (definition == null) {
      return null;
    }
    String typed = providerIdFromTypedConfig(definition.entitlements());
    if (typed != null) {
      return typed;
    }
    return providerIdFromObject(definition);
  }

  private String providerIdFromEnvironment(DefinitionEnvironment.Entitlements entitlements) {
    if (entitlements == null) {
      return null;
    }
    String typed = trimToNull(entitlements.defaultProviderId());
    if (typed != null) {
      return typed;
    }
    return providerIdFromObject(entitlements);
  }

  private DefinitionService findServiceDefinition(String group, String name) {
    List<DefinitionService> services = DiscoveryCache.list(DefinitionService.class, group);
    if (services == null || services.isEmpty()) {
      return null;
    }
    for (DefinitionService service : services) {
      DefinitionEnvelope envelope = service == null ? null : service.envelope();
      if (envelope != null && Objects.equals(name, envelope.name())) {
        return service;
      }
    }
    return null;
  }

  private DefinitionDataset findDatasetDefinition(String group, String namespace, String name) {
    List<DefinitionDataset> datasets = DiscoveryCache.list(DefinitionDataset.class, group);
    if (datasets == null || datasets.isEmpty()) {
      return null;
    }
    String canonicalName = trimToNull(namespace) == null ? null : namespace + "/" + name;
    for (DefinitionDataset dataset : datasets) {
      DefinitionEnvelope envelope = dataset == null ? null : dataset.envelope();
      if (envelope == null) {
        continue;
      }
      if (Objects.equals(name, envelope.name()) || Objects.equals(canonicalName, envelope.name())) {
        return dataset;
      }
    }
    return null;
  }

  private DefinitionEnvironment findEnvironmentDefinition() {
    List<DefinitionEnvironment> environments = DiscoveryCache.list(DefinitionEnvironment.class, PLATFORM_GROUP);
    if (environments == null || environments.isEmpty()) {
      return null;
    }
    for (DefinitionEnvironment environment : environments) {
      DefinitionEnvelope envelope = environment == null ? null : environment.envelope();
      if (envelope != null && Objects.equals(ENVIRONMENT_NAME, envelope.name())) {
        return environment;
      }
    }
    return null;
  }

  private String entitlementsScopeProviderId(EntitlementsScopeType scopeType, String scopeId) {
    List<DefinitionEntitlement> definitions = DiscoveryCache.list(DefinitionEntitlement.class);
    if (definitions == null || definitions.isEmpty()) {
      return null;
    }

    LinkedHashSet<String> matches = new LinkedHashSet<>();
    for (DefinitionEntitlement definition : definitions) {
      if (definition == null || !enabled(definition.envelope() == null ? null : definition.envelope().enabled())) {
        continue;
      }
      String canonicalScopeId = canonicalScopeId(scopeType, definition.scopeId());
      if (!scopeId.equals(canonicalScopeId)) {
        continue;
      }
      String configured = providerIdFromEntitlement(definition);
      if (configured != null) {
        matches.add(configured);
      }
    }

    if (matches.isEmpty()) {
      return null;
    }
    if (matches.size() == 1) {
      return matches.iterator().next();
    }
    List<String> values = new ArrayList<>(matches);
    values.sort(String.CASE_INSENSITIVE_ORDER);
    throw new IllegalStateException(
        "Conflicting entitlement subject provider configuration for scopeType="
            + scopeType.name()
            + " scopeId="
            + scopeId
            + " providers="
            + values);
  }

  private String providerIdFromEntitlement(DefinitionEntitlement definition) {
    if (definition == null) {
      return null;
    }
    String fromLabels = providerIdFromMap(definition.labels());
    if (fromLabels != null) {
      return fromLabels;
    }
    if (definition.policies() == null || definition.policies().isEmpty()) {
      return null;
    }
    LinkedHashSet<String> matches = new LinkedHashSet<>();
    for (DefinitionEntitlement.PolicyDocument policy : definition.policies()) {
      if (policy == null) {
        continue;
      }
      String fromMetadata = providerIdFromMap(policy.metadata());
      if (fromMetadata != null) {
        matches.add(fromMetadata);
      }
    }
    if (matches.isEmpty()) {
      return null;
    }
    if (matches.size() == 1) {
      return matches.iterator().next();
    }
    List<String> values = new ArrayList<>(matches);
    values.sort(String.CASE_INSENSITIVE_ORDER);
    throw new IllegalStateException(
        "Conflicting policy metadata entitlement subject provider configuration providers="
            + values);
  }

  private static String providerIdFromTypedConfig(EntitlementsProvider config) {
    if (config == null) {
      return null;
    }
    return trimToNull(config.providerId());
  }

  private String providerIdFromObject(Object object) {
    if (object == null) {
      return null;
    }

    String fromAccessor = providerIdFromAccessor(object, "subjectProvider");
    if (fromAccessor != null) {
      return fromAccessor;
    }

    fromAccessor = providerIdFromAccessor(object, "provider");
    if (fromAccessor != null) {
      return fromAccessor;
    }

    Object labels = valueFromAccessor(object, "labels");
    if (labels instanceof Map<?, ?> map) {
      return providerIdFromMap(castStringMap(map));
    }

    Object metadata = valueFromAccessor(object, "metadata");
    if (metadata instanceof Map<?, ?> map) {
      return providerIdFromMap(castStringMap(map));
    }
    return null;
  }

  private static String providerIdFromAccessor(Object object, String accessor) {
    Object value = valueFromAccessor(object, accessor);
    if (!(value instanceof String stringValue)) {
      return null;
    }
    return trimToNull(stringValue);
  }

  private static Object valueFromAccessor(Object object, String accessor) {
    if (object == null || accessor == null || accessor.isBlank()) {
      return null;
    }
    try {
      Method method = object.getClass().getMethod(accessor);
      return method.invoke(object);
    } catch (Exception ignored) {
      return null;
    }
  }

  private static Map<String, String> castStringMap(Map<?, ?> value) {
    if (value == null || value.isEmpty()) {
      return Map.of();
    }
    java.util.LinkedHashMap<String, String> casted = new java.util.LinkedHashMap<>();
    for (Map.Entry<?, ?> entry : value.entrySet()) {
      Object key = entry.getKey();
      Object raw = entry.getValue();
      if (!(key instanceof String keyString) || !(raw instanceof String valueString)) {
        continue;
      }
      casted.put(keyString, valueString);
    }
    return casted.isEmpty() ? Map.of() : Map.copyOf(casted);
  }

  private static String providerIdFromMap(Map<String, String> values) {
    if (values == null || values.isEmpty()) {
      return null;
    }
    String fromSubjectProvider = trimToNull(values.get(KEY_SUBJECT_PROVIDER));
    if (fromSubjectProvider != null) {
      return fromSubjectProvider;
    }
    return trimToNull(values.get(KEY_PROVIDER));
  }

  private static String canonicalResourceId(EntitlementsScopeType scopeType, String resourceId) {
    if (scopeType == EntitlementsScopeType.DATASET) {
      return DataEntitlements.requireDatasetResourceId(resourceId);
    }
    if (scopeType == EntitlementsScopeType.SERVICE) {
      return ServiceEntitlements.requireAssetResourceId(resourceId);
    }
    if (scopeType == EntitlementsScopeType.NAMESPACE) {
      CanonicalIds.ParsedId parsed = CanonicalIds.parse(resourceId);
      if (!parsed.isNamespace()) {
        throw new IllegalArgumentException("resourceId must be canonical namespace:<group>/<namespace>");
      }
      return parsed.canonical();
    }
    return requireTrimmed(resourceId, "resourceId");
  }

  private static String canonicalScopeId(EntitlementsScopeType scopeType, String scopeId) {
    if (scopeId == null) {
      return null;
    }
    try {
      if (scopeType == EntitlementsScopeType.DATASET) {
        return DataEntitlements.requireDatasetResourceId(scopeId);
      }
      if (scopeType == EntitlementsScopeType.SERVICE) {
        return ServiceEntitlements.requireAssetResourceId(scopeId);
      }
      if (scopeType == EntitlementsScopeType.NAMESPACE) {
        CanonicalIds.ParsedId parsed = CanonicalIds.parse(scopeId);
        return parsed.isNamespace() ? parsed.canonical() : null;
      }
      return trimToNull(scopeId);
    } catch (RuntimeException ex) {
      return null;
    }
  }

  private static String namespaceResourceId(EntitlementsScopeType scopeType, String resourceId) {
    if (scopeType == EntitlementsScopeType.NAMESPACE) {
      return canonicalScopeId(EntitlementsScopeType.NAMESPACE, resourceId);
    }
    CanonicalIds.ParsedId parsed = CanonicalIds.parse(resourceId);
    if (scopeType == EntitlementsScopeType.DATASET && parsed.isDataset()) {
      return "namespace:" + parsed.group() + "/" + parsed.namespace();
    }
    return null;
  }

  private static boolean enabled(Boolean value) {
    return value == null || value.booleanValue();
  }

  private static String requireTrimmed(String value, String name) {
    String normalized = trimToNull(value);
    if (normalized == null) {
      throw new IllegalArgumentException(name + " is required");
    }
    return normalized;
  }

  private static <T> T firstNonNull(T first, T second, T third) {
    if (first != null) {
      return first;
    }
    if (second != null) {
      return second;
    }
    return third;
  }

  private static String providerId(ProviderChoice choice) {
    return choice == null ? null : choice.providerId();
  }

  private String defaultProviderForTrace() {
    if (registry instanceof DefaultEntitlementProviderRegistry defaultRegistry) {
      return defaultRegistry.defaultProviderIdHint();
    }
    return "default";
  }

  private static String sourceLabel(SelectionSource source) {
    if (source == SelectionSource.REGISTRY_DEFAULT) {
      return "default";
    }
    return source.name().toLowerCase(java.util.Locale.ROOT);
  }

  private static String selectedProviderForLog(Selection selected, String defaultProviderId) {
    if (selected == null) {
      return null;
    }
    if (selected.source() == SelectionSource.REGISTRY_DEFAULT
        && Objects.equals("default", selected.providerId())) {
      return defaultProviderId;
    }
    return selected.providerId();
  }

  private String defaultProviderId(EntitlementSubjectProvider provider) {
    EntitlementSubjectProvider platformProvider =
        registry.provider(DefaultEntitlementProviderRegistry.PLATFORM_MODEL_PROVIDER_ID);
    if (platformProvider != null && platformProvider == provider) {
      return DefaultEntitlementProviderRegistry.PLATFORM_MODEL_PROVIDER_ID;
    }
    return "default";
  }

  private static String trimToNull(String value) {
    if (value == null) {
      return null;
    }
    String normalized = value.trim();
    return normalized.isEmpty() ? null : normalized;
  }

  public record Selection(
      String providerId,
      EntitlementSubjectProvider provider,
      SelectionSource source,
      boolean explicit) {
    public Selection {
      providerId = trimToNull(providerId);
      provider = Objects.requireNonNull(provider, "provider");
      source = Objects.requireNonNull(source, "source");
    }
  }

  public enum SelectionSource {
    RESOURCE,
    NAMESPACE,
    ENVIRONMENT,
    REGISTRY_DEFAULT
  }

  private record ProviderChoice(String providerId, SelectionSource source) {
    private ProviderChoice {
      providerId = requireTrimmed(providerId, "providerId");
      source = Objects.requireNonNull(source, "source");
    }
  }
}
