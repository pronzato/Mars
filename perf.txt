public final class FlightDatasetStreamProducer {
  private static final Logger LOGGER = LoggerFactory.getLogger(FlightDatasetStreamProducer.class);
  private static final long IPC_READ_ALLOCATOR_BYTES = 64L * 1024L * 1024L;
  private static final String OPERATION_DATASET_STREAM_SUBSCRIBE = "flight.dataset.stream.subscribe";
  private static final long STREAM_QUEUE_POLL_MILLIS = 100L;
  private static volatile DiagnosticsHook diagnosticsHook = DiagnosticsHook.NO_OP;

  private final TabularManager tabularManager;
  private final ServiceEntitlements serviceEntitlements;
  private final String serviceResourceId;
  private final DiscoveryCache discoveryCache;
  private final CheckpointStore checkpointStore;
  private final DuckdbNotifier changeNotifier;
  private final RootAllocator ipcReadAllocator = new RootAllocator(IPC_READ_ALLOCATOR_BYTES);
  private final DatasetStreamHubRegistry hubRegistry;
  private final ArrowBatchPublisher batchPublisher;
  private final Set<DatasetId> allowedDatasetIds;

  public FlightDatasetStreamProducer(
      TabularManager tabularManager,
      ServiceEntitlements serviceEntitlements,
      String serviceResourceId,
      DiscoveryCache discoveryCache,
      CheckpointStore checkpointStore,
      DuckdbNotifier changeNotifier) {
    this(
        tabularManager,
        serviceEntitlements,
        serviceResourceId,
        discoveryCache,
        checkpointStore,
        changeNotifier,
        Set.of());
  }

  public FlightDatasetStreamProducer(
      TabularManager tabularManager,
      ServiceEntitlements serviceEntitlements,
      String serviceResourceId,
      DiscoveryCache discoveryCache,
      CheckpointStore checkpointStore,
      DuckdbNotifier changeNotifier,
      Set<DatasetId> allowedDatasetIds) {
    this(
        tabularManager,
        serviceEntitlements,
        serviceResourceId,
        discoveryCache,
        checkpointStore,
        changeNotifier,
        allowedDatasetIds,
        new DatasetStreamHubRegistry());
  }

  FlightDatasetStreamProducer(
      TabularManager tabularManager,
      ServiceEntitlements serviceEntitlements,
      String serviceResourceId,
      DiscoveryCache discoveryCache,
      CheckpointStore checkpointStore,
      DuckdbNotifier changeNotifier,
      Set<DatasetId> allowedDatasetIds,
      DatasetStreamHubRegistry hubRegistry) {
    this.tabularManager = Objects.requireNonNull(tabularManager, "tabularManager");
    this.serviceEntitlements = Objects.requireNonNull(serviceEntitlements, "serviceEntitlements");
    this.serviceResourceId = Objects.requireNonNull(serviceResourceId, "serviceResourceId");
    this.discoveryCache = discoveryCache;
    this.checkpointStore = Objects.requireNonNull(checkpointStore, "checkpointStore");
    this.changeNotifier = Objects.requireNonNull(changeNotifier, "changeNotifier");
    this.allowedDatasetIds =
        allowedDatasetIds == null || allowedDatasetIds.isEmpty()
            ? Set.of()
            : Set.copyOf(allowedDatasetIds);
    this.hubRegistry = Objects.requireNonNull(hubRegistry, "hubRegistry");
    this.batchPublisher = new ArrowBatchPublisher(this.tabularManager);
  }

  public void start(
      RequestIdentity identity,
      Ticket ticket,
      FlightProducer.ServerStreamListener listener,
      Runnable onCancel) {
    start(identity, null, ticket, listener, onCancel);
  }

  public void start(
      RequestIdentity identity,
      FabricRequestHeaders requestHeaders,
      Ticket ticket,
      FlightProducer.ServerStreamListener listener,
      Runnable onCancel) {
    FlightDatasetStreamTicket streamTicket;
    try {
      streamTicket = FlightDatasetStreamTicketCodec.decode(ticket);
    } catch (Exception ex) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription("Invalid dataset stream ticket: " + ex.getMessage())
          .withCause(ex)
          .toRuntimeException();
    }
    runDatasetStream(identity, requestHeaders, streamTicket, listener, onCancel);
  }

  public void close() {
    hubRegistry.closeAll();
    try {
      ipcReadAllocator.close();
    } catch (Exception ignore) {
    }
  }

  public static void setDiagnosticsHookForTests(DiagnosticsHook hook) {
    diagnosticsHook = hook == null ? DiagnosticsHook.NO_OP : hook;
  }

  private void runDatasetStream(
      RequestIdentity identity,
      FabricRequestHeaders requestHeaders,
      FlightDatasetStreamTicket streamTicket,
      FlightProducer.ServerStreamListener listener,
      Runnable onCancel) {
    DatasetId datasetId = streamTicket.datasetId();
    authorizeInvoke(identity.effectiveUserSubject(), OPERATION_DATASET_STREAM_SUBSCRIBE, datasetId.toString());
    DatasetStreamSpec streamSpec = resolveStreamSpec(datasetId, streamTicket.accessMode());
    installDatasetIfPresent(datasetId);
    RowManagerImpl rowManager = localRowManager();
    if (!streamSpec.isPublic()) {
      requireDatasetReadAccess(rowManager, identity, datasetId);
    }
    String filterSql = validateStreamFilterSql(streamTicket.filterSql());
    CheckpointKey checkpointKey = streamCheckpointKey(streamTicket, identity, datasetId, streamSpec.accessMode());
    long startSeq = resolveStartSequence(rowManager, streamTicket, checkpointKey, datasetId);
    StreamSubscriptionContext subscriptionContext =
        new StreamSubscriptionContext(
            datasetId,
            streamSpec.accessMode(),
            streamSpec.isPublic() ? null : identity.effectiveUserSubject(),
            identity.callerAppSubject(),
            requestHeaders == null ? null : requestHeaders.requestMetadata(),
            streamTicket.subscriptionId(),
            streamTicket.start(),
            startSeq,
            streamTicket.batchMaxRows(),
            streamTicket.flushMillis(),
            filterSql,
            streamSpec.isPublic() ? null : identity);

    ByteBuffer schemaIpc = batchPublisher.querySchemaIpc(subscriptionContext);
    Schema schema = inspectIpc(schemaIpc).schema();
    DatasetStreamHub hub =
        hubRegistry.getOrCreate(
            datasetId,
            () ->
                new DatasetStreamHub(
                    datasetId,
                    streamSpec.accessMode(),
                    streamSpec.primaryKeyColumns(),
                    rowManager,
                    changeNotifier,
                    batchPublisher,
                    hubRegistry));
    long lastSeq = startSeq;

    try (DatasetStreamHub.SubscriberHandle handle = hub.addSubscriber(subscriptionContext);
        VectorSchemaRoot outputRoot = VectorSchemaRoot.create(schema, ipcReadAllocator)) {
      VectorLoader loader = new VectorLoader(outputRoot);
      listener.start(outputRoot);
      while (!listener.isCancelled() && !Thread.currentThread().isInterrupted()) {
        diagnosticsHook.onAwait();
        DatasetStreamHub.Delivery delivery = handle.poll(STREAM_QUEUE_POLL_MILLIS);
        if (delivery == null) {
          continue;
        }
        if (delivery.failure() != null) {
          throw mapStreamFailure(
              delivery.failure(), identity.effectiveUserSubject(), datasetId, streamSpec.isPublic());
        }
        lastSeq = Math.max(lastSeq, delivery.lastSeq());
        saveCheckpoint(checkpointKey, lastSeq);
        if (delivery.terminal()) {
          break;
        }
        if (delivery.ipc() != null && delivery.ipc().hasRemaining()) {
          diagnosticsHook.onReadChanges();
          streamBatchToListener(delivery.ipc(), loader, listener);
        }
      }
      saveCheckpoint(checkpointKey, lastSeq);
      if (!listener.isCancelled()) {
        listener.completed();
      }
      if (listener.isCancelled() && onCancel != null) {
        onCancel.run();
      }
    } catch (InterruptedException ex) {
      Thread.currentThread().interrupt();
    } catch (FlightRuntimeException ex) {
      throw ex;
    } catch (RuntimeException ex) {
      throw mapStreamFailure(ex, identity.effectiveUserSubject(), datasetId, streamSpec.isPublic());
    } catch (Exception ex) {
      throw CallStatus.INTERNAL
          .withDescription("Dataset stream failed: " + ex.getMessage())
          .withCause(ex)
          .toRuntimeException();
    } finally {
      saveCheckpoint(checkpointKey, lastSeq);
    }
  }

  private DatasetStreamSpec resolveStreamSpec(DatasetId datasetId, String requestedAccessMode) {
    Optional<DefinitionDataset> definition =
        discoveryCache == null
            ? Optional.empty()
            : discoveryCache.get(
                DefinitionDataset.class,
                datasetId.group(),
                datasetId.namespace() + "/" + datasetId.name());
    if (definition.isEmpty()) {
      throw CallStatus.NOT_FOUND
          .withDescription("DATASET_NOT_FOUND dataset=" + datasetId)
          .toRuntimeException();
    }
    DefinitionDataset dataset = definition.get();
    String datasetAccessMode = dataset.effectiveAccessMode();
    String requested = normalizeAccessMode(requestedAccessMode);
    String accessMode = requested == null ? datasetAccessMode : requested;
    if (requested != null && !requested.equals(datasetAccessMode)) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "STREAM_ACCESS_MODE_MISMATCH dataset="
                  + datasetId
                  + " requested="
                  + requested
                  + " effective="
                  + datasetAccessMode)
          .toRuntimeException();
    }
    DefinitionDataset.DuckdbBinding duckdb =
        dataset.binding() == null ? null : dataset.binding().duckdb();
    String targetType = duckdb == null ? null : Strings.trimToNull(duckdb.targetType());
    if (duckdb == null || !"DUCKDB_ACTIVE".equalsIgnoreCase(targetType)) {
      String type = dataset.type() == null ? "UNKNOWN" : dataset.type().name();
      if ("PUBLIC".equals(accessMode)) {
        throw CallStatus.INVALID_ARGUMENT
            .withDescription(
                "PUBLIC_DATASET_UNSUPPORTED dataset=" + datasetId + " type=" + type)
            .toRuntimeException();
      }
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "DATASET_NOT_STREAMABLE dataset=" + datasetId + " type=" + type)
          .toRuntimeException();
    }
    List<String> primaryKey = DuckdbActiveChangeLog.normalizedPrimaryKeyColumns(duckdb.primaryKey());
    if (primaryKey.isEmpty()) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "DATASET_NOT_STREAMABLE dataset=" + datasetId + " type=DUCKDB_ACTIVE")
          .toRuntimeException();
    }
    return new DatasetStreamSpec(accessMode, primaryKey);
  }

  private static String normalizeAccessMode(String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      return null;
    }
    if ("PUBLIC".equalsIgnoreCase(normalized)) {
      return "PUBLIC";
    }
    if ("PRIVATE".equalsIgnoreCase(normalized)) {
      return "PRIVATE";
    }
    throw CallStatus.INVALID_ARGUMENT
        .withDescription("accessMode must be PUBLIC or PRIVATE")
        .toRuntimeException();
  }

  private void authorizeInvoke(String callerSubject, String operationName, String requestPayload) {
    ServiceEntitlements.Decision decision =
        serviceEntitlements.authorize(
            new ServiceEntitlements.Identity(callerSubject, null),
            new ServiceEntitlements.Request(
                ServiceEntitlements.Transport.ARROW_FLIGHT_SQL,
                ServiceEntitlements.Action.INVOKE,
                serviceResourceId,
                operationName,
                null,
                requestPayload,
                Map.of(),
                Map.of()));
    if (decision == null || !decision.allowed()) {
      String reason = decision == null ? "DENIED" : Strings.firstNonBlank(decision.reason(), "DENIED");
      throw CallStatus.UNAUTHORIZED
          .withDescription(
              reason
                  + " [serviceId="
                  + serviceResourceId
                  + ", operation="
                  + operationName
                  + ", subject="
                  + safeSubject(callerSubject)
                  + "]")
          .toRuntimeException();
    }
  }

  private RowManagerImpl localRowManager() {
    if (tabularManager instanceof TabularManagerImpl localTabularManager) {
      return localTabularManager.rowManager();
    }
    throw CallStatus.UNIMPLEMENTED
        .withDescription("Dataset streaming requires DIRECT mode tabular manager")
        .toRuntimeException();
  }

  private void installDatasetIfPresent(DatasetId datasetId) {
    if (datasetId != null && !isAllowedDataset(datasetId)) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription(
              "Dataset is not allowed for this service datasetId="
                  + datasetId
                  + " serviceId="
                  + serviceResourceId)
          .toRuntimeException();
    }
    if (!(tabularManager instanceof TabularManagerImpl localTabularManager)) {
      return;
    }
    try {
      localTabularManager.installDataset(datasetId.toString());
    } catch (RuntimeException ex) {
      throw CallStatus.INVALID_ARGUMENT
          .withDescription("Dataset definition not available for query datasetId=" + datasetId)
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private long resolveStartSequence(
      RowManagerImpl rowManager,
      FlightDatasetStreamTicket ticket,
      CheckpointKey checkpointKey,
      DatasetId datasetId) {
    if (ticket.start() == FlightDatasetStreamTicket.Start.BEGINNING) {
      return 0L;
    }
    if (ticket.start() == FlightDatasetStreamTicket.Start.FROM_SEQUENCE) {
      return Math.max(0L, ticket.fromSequence());
    }
    if (ticket.start() == FlightDatasetStreamTicket.Start.RESUME) {
      Long checkpointSeq = loadCheckpointSeq(checkpointKey);
      return checkpointSeq == null ? 0L : checkpointSeq;
    }
    return Math.max(0L, rowManager.duckDbActiveChangeMaxSeq(datasetId));
  }

  private boolean isAllowedDataset(DatasetId datasetId) {
    return datasetId == null || allowedDatasetIds.isEmpty() || allowedDatasetIds.contains(datasetId);
  }

  private static void requireDatasetReadAccess(
      RowManagerImpl rowManager, RequestIdentity identity, DatasetId datasetId) {
    DataEntitlements.Decision decision =
        rowManager
            .dataEntitlementsForSqlPlanning()
            .authorizeRead(
                new DataEntitlements.Identity(identity.effectiveUserSubject(), null),
                datasetId.toString());
    if (decision == null || !decision.allowed()) {
      String reason = decision == null ? "DENIED" : Strings.firstNonBlank(decision.reason(), "DENIED");
      throw CallStatus.UNAUTHORIZED
          .withDescription(
              "READ_DENIED dataset="
                  + datasetId
                  + " subject="
                  + safeSubject(identity.effectiveUserSubject())
                  + " reason="
                  + reason)
          .toRuntimeException();
    }
  }

  private CheckpointKey streamCheckpointKey(
      FlightDatasetStreamTicket ticket, RequestIdentity identity, DatasetId datasetId, String accessMode) {
    String scope = Strings.firstNonBlank(Strings.trimToNull(ticket.subscriptionId()), "default");
    if ("PUBLIC".equals(accessMode)) {
      return new CheckpointKey(datasetId.group(), CheckpointType.STREAM, datasetId.toString(), scope);
    }
    String caller = CheckpointKey.normalizeToken(identity.effectiveUserSubject());
    return new CheckpointKey(
        datasetId.group(), CheckpointType.STREAM, datasetId.toString(), scope + "_" + caller);
  }

  private static String validateStreamFilterSql(String filterSql) {
    try {
      return StreamFilterSqlValidator.validate(filterSql);
    } catch (IllegalArgumentException ex) {
      throw CallStatus.INVALID_ARGUMENT.withDescription(ex.getMessage()).withCause(ex).toRuntimeException();
    }
  }

  private Long loadCheckpointSeq(CheckpointKey key) {
    if (key == null) {
      return null;
    }
    Optional<CheckpointRecord> record = checkpointStore.load(key);
    if (record.isEmpty() || record.get().payload() == null) {
      return null;
    }
    String raw = record.get().payload().get("lastSeq");
    if (Strings.trimToNull(raw) == null) {
      return null;
    }
    try {
      long parsed = Long.parseLong(raw);
      return parsed < 0L ? null : parsed;
    } catch (NumberFormatException ex) {
      return null;
    }
  }

  private void saveCheckpoint(CheckpointKey key, long lastSeq) {
    if (key == null) {
      return;
    }
    try {
      checkpointStore.save(
          CheckpointRecord.of(
              key,
              Map.of(
                  "lastSeq", Long.toString(Math.max(0L, lastSeq)),
                  "updatedUtc", Instant.now().toString())));
    } catch (RuntimeException ex) {
      LOGGER.warn("[FlightStream][checkpoint-save-failed] key={} lastSeq={} message={}", key, Math.max(0L, lastSeq), ex.getMessage());
    }
  }

  private void streamBatchToListener(
      ByteBuffer ipc, VectorLoader loader, FlightProducer.ServerStreamListener listener) {
    IpcSummary summary = inspectIpc(ipc);
    if (summary.rowCount() <= 0L) {
      return;
    }
    try (ArrowStreamReader reader =
        new ArrowStreamReader(Bytes.asInputStream(ipc), ipcReadAllocator)) {
      VectorSchemaRoot root = reader.getVectorSchemaRoot();
      while (reader.loadNextBatch()) {
        if (root.getRowCount() <= 0) {
          continue;
        }
        try (ArrowRecordBatch batch = new VectorUnloader(root).getRecordBatch()) {
          loader.load(batch);
          listener.putNext();
        }
      }
    } catch (Exception ex) {
      throw CallStatus.INTERNAL
          .withDescription("Failed to stream dataset batch")
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private IpcSummary inspectIpc(ByteBuffer ipc) {
    if (ipc == null || !ipc.hasRemaining()) {
      throw CallStatus.INTERNAL.withDescription("Query returned empty Arrow IPC payload").toRuntimeException();
    }
    try (ArrowStreamReader reader =
        new ArrowStreamReader(Bytes.asInputStream(ipc), ipcReadAllocator)) {
      VectorSchemaRoot root = reader.getVectorSchemaRoot();
      Schema schema = root.getSchema();
      long rowCount = 0L;
      while (reader.loadNextBatch()) {
        rowCount += root.getRowCount();
      }
      return new IpcSummary(schema, rowCount);
    } catch (Exception ex) {
      throw CallStatus.INTERNAL
          .withDescription("Failed to inspect Arrow IPC payload")
          .withCause(ex)
          .toRuntimeException();
    }
  }

  private RuntimeException mapStreamFailure(
      RuntimeException ex, String callerSubject, DatasetId datasetId, boolean publicStream) {
    if (ex instanceof FlightRuntimeException flightRuntimeException) {
      return flightRuntimeException;
    }
    String message = Strings.trimToNull(ex.getMessage());
    String details =
        "serviceId="
            + serviceResourceId
            + ", datasetId="
            + datasetId
            + ", subject="
            + safeSubject(callerSubject);
    if (message == null) {
      return CallStatus.INTERNAL
          .withDescription("Dataset stream failed [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    }
    String lower = message.toLowerCase();
    if (!publicStream && lower.contains("access denied for dataset")) {
      return CallStatus.UNAUTHORIZED
          .withDescription(
              "READ_DENIED dataset="
                  + datasetId
                  + " subject="
                  + safeSubject(callerSubject)
                  + " reason=DENIED")
          .withCause(ex)
          .toRuntimeException();
    }
    if (lower.contains("duckdb sql failed")
        || lower.contains("unknown table token")
        || lower.contains("ambiguous table token")
        || lower.contains("dataset not found")) {
      return CallStatus.INVALID_ARGUMENT
          .withDescription(message + " [" + details + "]")
          .withCause(ex)
          .toRuntimeException();
    }
    return CallStatus.INTERNAL
        .withDescription(message + " [" + details + "]")
        .withCause(ex)
        .toRuntimeException();
  }

  private static String safeSubject(String subject) {
    String normalized = Strings.trimToNull(subject);
    return normalized == null ? "<unknown>" : normalized;
  }

  private record DatasetStreamSpec(String accessMode, List<String> primaryKeyColumns) {
    private boolean isPublic() {
      return "PUBLIC".equals(accessMode);
    }
  }

  private record IpcSummary(Schema schema, long rowCount) {}

  @FunctionalInterface
  public interface DiagnosticsHook {
    DiagnosticsHook NO_OP = () -> {};

    void onReadChanges();

    default void onAwait() {}
  }
}
