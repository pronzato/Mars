final class RemoteFlightClientSupport {
  private static final Logger LOGGER = LoggerFactory.getLogger(RemoteFlightClientSupport.class);
  private static final long MAX_ALLOCATOR_BYTES = 64L * 1024L * 1024L;
  private static final String KERBEROS_AUTH_FAILURE = "Kerberos authentication failed";
  private static final String UNAUTHENTICATED = "UNAUTHENTICATED";

  private RemoteFlightClientSupport() {}

  static String endpointSecurityContext(ServiceEndpoint endpoint) {
    Objects.requireNonNull(endpoint, "endpoint");
    return "serviceId="
        + endpoint.serviceResourceId()
        + ", endpoint="
        + endpoint.connectHost()
        + ":"
        + endpoint.port()
        + ", tlsEnabled="
        + endpoint.tlsEnabled()
        + ", kerberosEnabled="
        + endpoint.kerberosEnabled();
  }

  static ServiceEndpoint resolveArrowService(
      DiscoveryCache DiscoveryCache,
      String group,
      String serviceName,
      DefinitionService.ServiceType expectedType,
      ServiceInstanceResolver endpointResolver) {
    return resolveArrowService(
        DiscoveryCache, group, serviceName, expectedType, endpointResolver, null, null);
  }

  static ServiceEndpoint resolveArrowService(
      DiscoveryCache DiscoveryCache,
      String group,
      String serviceName,
      DefinitionService.ServiceType expectedType,
      ServiceInstanceResolver endpointResolver,
      ServiceRoutingContext routingContext,
      ServiceRoutingState routingState) {
    Objects.requireNonNull(DiscoveryCache, "DiscoveryCache");
    String resolvedGroup = Strings.requireNonBlank(group, "service group");
    String resolvedServiceName = Strings.requireNonBlank(serviceName, "service name");

    Optional<DefinitionService> definition =
        DiscoveryCache.get(DefinitionService.class, resolvedGroup, resolvedServiceName);
    if (definition.isEmpty()) {
      throw new IllegalArgumentException("Unknown service: " + resolvedServiceName);
    }

    DefinitionService service = definition.get();
    DefinitionService.ServiceType actualType = service.serviceType();
    if (actualType != null && actualType != expectedType) {
      throw new IllegalArgumentException(
          "Service is not "
              + expectedType
              + ": "
              + resolvedServiceName
              + " type="
              + actualType);
    }
    DefinitionService.Arrow arrow = service.arrow();
    if (arrow == null) {
      throw new IllegalArgumentException(
          "Arrow service config is required for service: " + resolvedServiceName);
    }
    String serviceResourceId =
        Strings.requireNonBlank(service.resourceId(), "service.resourceId");
    ServiceRoutingMode routingMode =
        ServiceRoutingMode.requireSupported(
            service.arrow() == null || service.arrow().routing() == null
                ? null
                : service.arrow().routing().mode(),
            serviceResourceId,
            "arrow.routing.mode",
            "client");
    boolean tlsEnabled = arrow.tls() != null && Boolean.TRUE.equals(arrow.tls().enabled());
    boolean kerberosEnabled = arrow.kerberos() != null && arrow.kerberos().enabled();
    String kerberosSpn = arrow.kerberos() == null ? null : Strings.trimToNull(arrow.kerberos().spn());

    if (endpointResolver == null) {
      throw new IllegalStateException(
          "Arrow service endpoint resolver is not initialized for service: " + serviceResourceId);
    }
    ServiceHbCache.Target target =
        routingMode == ServiceRoutingMode.ROUND_ROBIN
            ? endpointResolver.chooseRoundRobin(
                resolvedGroup, resolvedServiceName, serviceResourceId)
            : endpointResolver.choose(
                resolvedGroup,
                resolvedServiceName,
                serviceResourceId,
                routingMode,
                routingContext == null
                    ? ServiceRoutingContext.minimal(serviceResourceId, transportName(expectedType))
                    : new ServiceRoutingContext(
                        serviceResourceId,
                        Strings.firstNonBlank(routingContext.transport(), transportName(expectedType)),
                        routingContext.effectiveSubject(),
                        routingContext.callerAppSubject(),
                        routingContext.routingKey(),
                        routingContext.requestId(),
                        routingContext.operation()),
                routingState);
    if (target == null) {
      throw new IllegalStateException("No available instances for service: " + serviceResourceId);
    }
    if (LOGGER.isDebugEnabled()) {
      LOGGER.debug(
          "[RemoteFlightClient][routing] mode={} serviceId={} group={} service={} endpoint={} instance={} subject={} callerApp={} routingKey={} transport={}",
          routingMode,
          serviceResourceId,
          resolvedGroup,
          resolvedServiceName,
          target.host + ":" + target.port,
          target.instanceId,
          routingContext == null ? null : routingContext.effectiveSubject(),
          routingContext == null ? null : routingContext.callerAppSubject(),
          routingContext == null ? null : routingContext.routingKey(),
          transportName(expectedType));
    }
    String connectHost = ServiceAdvertise.requireRoutableHost(target.host, "grpc target host");
    int port = target.port;
    boolean resolvedTls = tlsEnabled || target.tls;
    return new ServiceEndpoint(
        resolvedGroup,
        resolvedServiceName,
        serviceResourceId,
        service,
        connectHost,
        port,
        resolvedTls,
        kerberosEnabled,
        kerberosSpn);
  }

  static ClientResources openClient(
      ServiceEndpoint endpoint,
      ServiceRunOptions runOptions,
      String sessionKerberosPrincipal,
      java.nio.file.Path sessionKerberosKeytab) {
    Objects.requireNonNull(endpoint, "endpoint");
    ServiceRunOptions options = runOptions == null ? ServiceRunOptions.defaults() : runOptions;

    RootAllocator allocator = new RootAllocator(MAX_ALLOCATOR_BYTES);
    ManagedChannel channel = null;
    FlightClient client = null;
    String derivedSpn = null;
    String diagnostics = null;
    try {
      if (endpoint.kerberosEnabled()) {
        derivedSpn = deriveClientSpn(endpoint, sessionKerberosPrincipal);
      }
      diagnostics = clientDiagnostics(endpoint, options, sessionKerberosPrincipal, sessionKerberosKeytab, derivedSpn);
      channel =
          FlightClientChannelSupport.createChannel(
              endpoint.connectHost(),
              endpoint.port(),
              endpoint.tlsEnabled(),
              endpoint.kerberosEnabled(),
              endpoint.kerberosSpn(),
              options,
              sessionKerberosPrincipal,
              sessionKerberosKeytab);
      client = FlightGrpcUtils.createFlightClient(allocator, channel, List.of());
      return new ClientResources(allocator, channel, client);
    } catch (Exception ex) {
      CloseSupport.closeQuietly(client);
      shutdownQuietly(channel);
      CloseSupport.closeQuietly(allocator);
      throw new IllegalStateException(
          "Failed to open Arrow Flight client for service "
              + endpoint.serviceName()
              + " at "
              + endpoint.connectHost()
              + ":"
              + endpoint.port()
              + " derivedSpn="
              + derivedSpn
              + " diagnostics="
              + diagnostics,
          ex);
    }
  }

  static CallOption[] buildCallOptions(
      CallerAppIdentity callerApp,
      String env,
      String region,
      String site,
      ClientResources client,
      String datasetId) {
    return buildCallOptions(callerApp, env, region, site, client, datasetId, true);
  }

  static CallOption[] buildCallOptions(
      CallerAppIdentity callerApp,
      String env,
      String region,
      String site,
      ClientResources client,
      String datasetId,
      boolean includeDelegatedUser) {
    Objects.requireNonNull(callerApp, "callerApp");
    Objects.requireNonNull(client, "client");

    FlightCallHeaders headers = new FlightCallHeaders();
    UserPrincipal principal = UserContext.get().orElse(null);
    String oboUser =
        includeDelegatedUser && principal != null ? Strings.trimToNull(principal.username()) : null;
    FabricRequestHeaders requestHeaders =
        FabricRequestHeaders.outbound(
            callerApp,
            oboUser,
            new FabricRequestHeaders.RequestMetadata(
                FabricRequestHeaders.generateRequestId(),
                UUID.randomUUID().toString(),
                env,
                region,
                site));
    requestHeaders.applyToFlightHeaders(headers);
    if (Strings.isNotBlank(datasetId)) {
      headers.insert(FlightSqlRequestHeadersMiddleware.HEADER_DATASET_ID, datasetId);
    }
    return new CallOption[] {new HeaderCallOption(headers)};
  }

  static String kerberosAuthFailureDetails(ServiceEndpoint endpoint, Throwable error) {
    return kerberosAuthFailureDetails(endpoint, error, null);
  }

  static String kerberosAuthFailureDetails(
      ServiceEndpoint endpoint, Throwable error, String sessionKerberosPrincipal) {
    if (!isKerberosAuthFailure(error)) {
      return null;
    }
    Objects.requireNonNull(endpoint, "endpoint");
    String derivedSpn = null;
    try {
      derivedSpn = deriveClientSpn(endpoint, sessionKerberosPrincipal);
    } catch (RuntimeException ignore) {
      derivedSpn = Strings.trimToNull(endpoint.kerberosSpn());
    }
    return "kerberosAuthContext{serviceId="
        + endpoint.serviceResourceId()
        + ",connectHost="
        + endpoint.connectHost()
        + ",port="
        + endpoint.port()
        + ",derivedSpn="
        + derivedSpn
        + "}";
  }

  static boolean isKerberosAuthFailure(Throwable error) {
    Throwable cursor = error;
    while (cursor != null) {
      if (cursor instanceof FlightRuntimeException flightEx
          && flightEx.status() != null
          && flightEx.status().code() == FlightStatusCode.UNAUTHENTICATED
          && containsIgnoreCase(flightEx.getMessage(), KERBEROS_AUTH_FAILURE)) {
        return true;
      }
      String message = cursor.getMessage();
      if (containsIgnoreCase(message, UNAUTHENTICATED)
          && containsIgnoreCase(message, KERBEROS_AUTH_FAILURE)) {
        return true;
      }
      cursor = cursor.getCause();
    }
    return false;
  }

  static String clientDiagnostics(
      ServiceEndpoint endpoint,
      ServiceRunOptions runOptions,
      String sessionKerberosPrincipal,
      java.nio.file.Path sessionKerberosKeytab,
      String derivedSpn) {
    Objects.requireNonNull(endpoint, "endpoint");
    ServiceRunOptions options = runOptions == null ? ServiceRunOptions.defaults() : runOptions;
    java.nio.file.Path trustCert =
        options.tlsTrustCert() == null ? options.clientTrustCert() : options.tlsTrustCert();
    return "client{"
        + "serviceGroup="
        + endpoint.group()
        + ",serviceName="
        + endpoint.serviceName()
        + ",tlsEnabled="
        + endpoint.tlsEnabled()
        + ",trustCert="
        + safePath(trustCert)
        + ",trustCertExists="
        + pathExists(trustCert)
        + ",kerberosEnabled="
        + endpoint.kerberosEnabled()
        + ",kerberosSpn="
        + Strings.trimToNull(endpoint.kerberosSpn())
        + ",derivedSpn="
        + Strings.trimToNull(derivedSpn)
        + ",sessionPrincipal="
        + Strings.trimToNull(sessionKerberosPrincipal)
        + ",keytab="
        + safePath(sessionKerberosKeytab)
        + ",keytabExists="
        + pathExists(sessionKerberosKeytab)
        + "}";
  }

  static String summarizeThrowable(Throwable error) {
    if (error == null) {
      return "<none>";
    }
    StringBuilder summary = new StringBuilder();
    Throwable cursor = error;
    int depth = 0;
    while (cursor != null && depth < 8) {
      if (depth > 0) {
        summary.append(" <- ");
      }
      summary
          .append(cursor.getClass().getSimpleName())
          .append(":")
          .append(Strings.firstNonBlank(Strings.trimToNull(cursor.getMessage()), "<empty>"));
      cursor = cursor.getCause();
      depth++;
    }
    if (cursor != null) {
      summary.append(" <- ...");
    }
    return summary.toString();
  }

  private static String safePath(java.nio.file.Path path) {
    return path == null ? null : path.toAbsolutePath().normalize().toString();
  }

  private static boolean pathExists(java.nio.file.Path path) {
    return path != null && java.nio.file.Files.exists(path);
  }

  private static boolean containsIgnoreCase(String value, String needle) {
    if (value == null || needle == null) {
      return false;
    }
    return value.toLowerCase(java.util.Locale.ROOT).contains(needle.toLowerCase(java.util.Locale.ROOT));
  }

  private static String deriveClientSpn(ServiceEndpoint endpoint, String sessionKerberosPrincipal) {
    Objects.requireNonNull(endpoint, "endpoint");
    String baseSpn = KerberosSpn.resolveSpn(endpoint.kerberosSpn(), endpoint.connectHost());
    String realm = KerberosSpn.extractRealm(sessionKerberosPrincipal);
    if (realm == null) {
      realm = KerberosConfig.DEFAULT_REALM;
    }
    return KerberosSpn.normalizeSpnWithRealm(baseSpn, realm);
  }

  private static String transportName(DefinitionService.ServiceType expectedType) {
    if (expectedType == DefinitionService.ServiceType.ARROW_FLIGHT_SQL) {
      return "ARROW_FLIGHT_SQL";
    }
    if (expectedType == DefinitionService.ServiceType.ARROW_FLIGHT_INGEST) {
      return "ARROW_FLIGHT";
    }
    return "ARROW";
  }


  private static void shutdownQuietly(ManagedChannel channel) {
    if (channel == null) {
      return;
    }
    try {
      channel.shutdownNow();
      channel.awaitTermination(250, TimeUnit.MILLISECONDS);
    } catch (Exception ignore) {
    }
  }

  record ServiceEndpoint(
      String group,
      String serviceName,
      String serviceResourceId,
      DefinitionService definition,
      String connectHost,
      int port,
      boolean tlsEnabled,
      boolean kerberosEnabled,
      String kerberosSpn) {}

  static final class ClientResources implements AutoCloseable {
    private final BufferAllocator allocator;
    private final ManagedChannel channel;
    private final FlightClient flightClient;

    ClientResources(BufferAllocator allocator, ManagedChannel channel, FlightClient flightClient) {
      this.allocator = Objects.requireNonNull(allocator, "allocator");
      this.channel = Objects.requireNonNull(channel, "channel");
      this.flightClient = Objects.requireNonNull(flightClient, "flightClient");
    }

    BufferAllocator allocator() {
      return allocator;
    }

    FlightClient flightClient() {
      return flightClient;
    }

    @Override
    public void close() {
      try {
        flightClient.close();
      } catch (InterruptedException ex) {
        Thread.currentThread().interrupt();
      } catch (Exception ignore) {
      }
      shutdownQuietly(channel);
      CloseSupport.closeQuietly(allocator);
    }
  }
}
