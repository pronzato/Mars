package org.pronzato.fabric.api.control.status.app;

import java.net.InetAddress;
import java.time.Clock;
import java.time.Duration;
import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Objects;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.ScheduledFuture;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicBoolean;
import java.util.concurrent.atomic.AtomicLong;
import org.pronzato.fabric.api.control.ControlConnection;
import org.pronzato.fabric.api.control.discovery.cache.DiscoveryTopics;
import org.pronzato.fabric.api.control.discovery.codec.DiscoveryCodec;

public final class AppStatusPublisher implements AutoCloseable {
  private static final Logger LOG = LoggerFactory.getLogger(AppStatusPublisher.class);
  private static final int MAX_ATTEMPTS = 3;
  private static final String DEFAULT_VERSION = "DEV";
  private static final int DEFAULT_HEARTBEAT_SECONDS = 30;

  private final ControlConnection control;
  private final SessionContext context;
  private final DefinitionApp appDefinition;
  private final TickListener tickListener;
  private final long heartbeatIntervalMs;
  private final int heartbeatIntervalSeconds;
  private final int heartbeatTimeoutSeconds;
  private final String intervalSource;
  private final String configuredUserTag;
  private final Boolean otelEnabled;
  private final Clock clock;
  private final AppMetricsCollector metricsCollector;
  private final ObjectMapper mapper = DiscoveryCodec.mapper();
  private final ScheduledExecutorService executor;
  private final boolean ownsExecutor;
  private final AtomicBoolean running = new AtomicBoolean(false);
  private final AtomicLong publishCount = new AtomicLong(0L);

  private volatile DefinitionStatusApp current;
  private volatile String lastPayload;
  private volatile long lastPublishedEpochMs;
  private volatile ScheduledFuture<?> heartbeatTask;

  public AppStatusPublisher(
      ControlConnection control, SessionContext context, DefinitionApp appDefinition) {
    this(control, context, appDefinition, null, null, null);
  }

  public AppStatusPublisher(
      ControlConnection control,
      SessionContext context,
      DefinitionApp appDefinition,
      TickListener tickListener) {
    this(control, context, appDefinition, null, null, tickListener);
  }

  public AppStatusPublisher(
      ControlConnection control,
      SessionContext context,
      DefinitionApp appDefinition,
      ScheduledExecutorService scheduler,
      TickListener tickListener) {
    this(control, context, appDefinition, null, scheduler, tickListener);
  }

  public AppStatusPublisher(
      ControlConnection control,
      SessionContext context,
      DefinitionApp appDefinition,
      String configuredUserTag,
      ScheduledExecutorService scheduler,
      TickListener tickListener) {
    this(control, context, appDefinition, configuredUserTag, null, scheduler, tickListener);
  }

  public AppStatusPublisher(
      ControlConnection control,
      SessionContext context,
      DefinitionApp appDefinition,
      String configuredUserTag,
      Boolean otelEnabled,
      ScheduledExecutorService scheduler,
      TickListener tickListener) {
    this.control = Objects.requireNonNull(control, "control");
    this.context = Objects.requireNonNull(context, "context");
    this.appDefinition = appDefinition;
    this.tickListener = tickListener;
    this.configuredUserTag = Strings.trimToNull(configuredUserTag);
    this.otelEnabled = otelEnabled;
    this.clock = Clock.systemUTC();
    this.metricsCollector = new AppMetricsCollector();
    this.heartbeatIntervalSeconds = resolveHeartbeatIntervalSeconds(appDefinition);
    this.heartbeatIntervalMs = Duration.ofSeconds(heartbeatIntervalSeconds).toMillis();
    this.heartbeatTimeoutSeconds = (int) Math.max(heartbeatIntervalSeconds * 3L, 60L);
    this.intervalSource = resolveIntervalSource(appDefinition);
    if (scheduler == null) {
      this.executor =
          Executors.newSingleThreadScheduledExecutor(
              runnable -> {
                Thread thread = new Thread(runnable, "fabric-app-heartbeat");
                thread.setDaemon(true);
                return thread;
              });
      this.ownsExecutor = true;
    } else {
      this.executor = scheduler;
      this.ownsExecutor = false;
    }
  }

  public void start() {
    if (!running.compareAndSet(false, true)) {
      return;
    }
    DefinitionStatusApp started = buildInitialStatus();
    publish(started);
    publish(transitionToRunning(started, clock.millis()));
    heartbeatTask =
        executor.scheduleAtFixedRate(
        this::publishHeartbeatSafe, heartbeatIntervalMs, heartbeatIntervalMs, TimeUnit.MILLISECONDS);
  }

  public boolean started() {
    return running.get();
  }

  public int publishIntervalSeconds() {
    return heartbeatIntervalSeconds;
  }

  public String intervalSource() {
    return intervalSource;
  }

  public String lastPublishedPayload() {
    return lastPayload;
  }

  public long lastPublishedEpochMs() {
    return lastPublishedEpochMs;
  }

  public long publishCount() {
    return publishCount.get();
  }

  @Override
  public void close() {
    if (!running.getAndSet(false)) {
      return;
    }
    ScheduledFuture<?> task = heartbeatTask;
    heartbeatTask = null;
    if (task != null) {
      task.cancel(true);
    }
    publishStoppedSafe();
    if (ownsExecutor) {
      executor.shutdownNow();
    }
  }

  public void stop() {
    close();
  }

  private void publish(DefinitionStatusApp status) {
    Objects.requireNonNull(status, "status");
    DefinitionStatusApp normalized = normalizeStatus(status);
    String group = Strings.trimToNull(normalized.group());
    if (group == null) {
      throw new IllegalStateException("STATUS_GROUP_MISSING: group is required");
    }
    current = normalized;
    int attempt = 0;
    while (attempt < MAX_ATTEMPTS) {
      try {
        publishOnce(normalized);
        return;
      } catch (Exception ex) {
        attempt++;
        if (attempt >= MAX_ATTEMPTS) {
          LOG.warn(
              "[AppStatusPublisher] failed to publish status for {}/{}/{} after {} attempts: {}",
              normalized.group(),
              normalized.name(),
              normalized.instance(),
              attempt,
              ex.toString());
        } else {
          try {
            Thread.sleep(50L * attempt);
          } catch (InterruptedException ie) {
            Thread.currentThread().interrupt();
            return;
          }
        }
      }
    }
  }

  private void publishHeartbeatSafe() {
    if (!running.get()) {
      return;
    }
    try {
      DefinitionStatusApp baseline = current;
      if (baseline == null) {
        return;
      }
      long now = clock.millis();
      DefinitionStatusApp heartbeat = withLastHeartbeatEpochMs(baseline, now);
      publish(heartbeat);
    } catch (Exception ex) {
      LOG.warn("[AppStatusPublisher] heartbeat publish failed", ex);
    }
  }

  private void publishOnce(DefinitionStatusApp status) throws Exception {
    if (status == null) {
      return;
    }
    String payload = mapper.writeValueAsString(status);
    control.amps().publish(DiscoveryTopics.STATUS, payload);

    long publishedAt =
        status.lastHeartbeatEpochMs() == null
            ? clock.millis()
            : Math.max(0L, status.lastHeartbeatEpochMs());
    this.lastPayload = payload;
    this.lastPublishedEpochMs = publishedAt;
    this.publishCount.incrementAndGet();
    if (tickListener != null) {
      try {
        tickListener.onPublished(publishedAt, status, payload);
      } catch (Exception ignore) {
      }
    }
  }

  private DefinitionStatusApp buildInitialStatus() {
    long now = clock.millis();
    String group = requireField(context.getGroup(), "group");
    String name = requireField(context.getAppName(), "name");
    String instance = requireField(context.getInstance(), "instance");
    String env = Strings.trimToNull(context.getEnv());
    String appType = Strings.trimToNull(appDefinition == null ? null : appDefinition.appType());

    Map<String, String> tags = buildTags(group, name, instance, appType, context, configuredUserTag);
    String buildVersion = Strings.trimToNull(tags.get("buildVersion"));
    String configVersion = Strings.trimToNull(tags.get("configVersion"));
    String codeVersion = Strings.trimToNull(tags.get("codeVersion"));
    String codeCommitId = Strings.trimToNull(tags.get("codeCommitId"));
    String resolvedVersion = firstNonBlank(buildVersion, codeVersion, configVersion);
    AppRuntimeType runtimeType = mapRuntimeType(appType);

    DefinitionEnvelope envelope =
        new DefinitionEnvelope(
            group, DefinitionType.STATUS_APP, name, null, null, null, null, null, null, null);
    return new DefinitionStatusApp(
        envelope,
        group,
        DefinitionType.STATUS_APP,
        name,
        instance,
        Strings.trimToNull(context.getSessionId()),
        buildSessionKey(group, name, instance, context.getEnv(), context.getRegion(), context.getSite()),
        runtimeType,
        resolveHostname(),
        resolvedVersion,
        env,
        appType,
        DefinitionStatusApp.State.STARTED,
        now,
        now,
        now,
        now,
        null,
        1L,
        buildVersion,
        configVersion,
        codeVersion,
        codeCommitId,
        null,
        null,
        heartbeatIntervalSeconds,
        heartbeatTimeoutSeconds,
        otelEnabled,
        tags,
        null,
        null,
        null,
        null,
        null,
        null,
        null);
  }

  private DefinitionStatusApp transitionToRunning(DefinitionStatusApp baseline, long now) {
    if (baseline == null) {
      return null;
    }
    long normalizedNow = Math.max(0L, now);
    Long startedAt = baseline.startedAtEpochMs() == null ? normalizedNow : baseline.startedAtEpochMs();
    Long lastStart = baseline.lastStartEpochMs() == null ? normalizedNow : baseline.lastStartEpochMs();
    return new DefinitionStatusApp(
        baseline.envelope(),
        baseline.group(),
        baseline.type(),
        baseline.name(),
        baseline.instance(),
        baseline.sessionId(),
        baseline.sessionKey(),
        baseline.runtimeType(),
        baseline.host(),
        baseline.version(),
        baseline.env(),
        baseline.appType(),
        DefinitionStatusApp.State.RUNNING,
        startedAt,
        normalizedNow,
        normalizedNow,
        lastStart,
        baseline.lastStopEpochMs(),
        baseline.runCount(),
        baseline.buildVersion(),
        baseline.configVersion(),
        baseline.codeVersion(),
        baseline.codeCommitId(),
        baseline.lastExitCode(),
        baseline.lastErrorMessage(),
        baseline.heartbeatIntervalSeconds(),
        baseline.heartbeatTimeoutSeconds(),
        baseline.otel(),
        baseline.tags(),
        baseline.message(),
        baseline.deployEnabled(),
        baseline.deployStatus(),
        baseline.deployAtEpochMs(),
        baseline.deployMessage(),
        baseline.deploySuccess(),
        baseline.metrics());
  }

  private DefinitionStatusApp transitionToStopped(DefinitionStatusApp baseline, long now) {
    if (baseline == null) {
      return null;
    }
    long normalizedNow = Math.max(0L, now);
    return new DefinitionStatusApp(
        baseline.envelope(),
        baseline.group(),
        baseline.type(),
        baseline.name(),
        baseline.instance(),
        baseline.sessionId(),
        baseline.sessionKey(),
        baseline.runtimeType(),
        baseline.host(),
        baseline.version(),
        baseline.env(),
        baseline.appType(),
        DefinitionStatusApp.State.STOPPED,
        baseline.startedAtEpochMs(),
        normalizedNow,
        normalizedNow,
        baseline.lastStartEpochMs(),
        normalizedNow,
        baseline.runCount(),
        baseline.buildVersion(),
        baseline.configVersion(),
        baseline.codeVersion(),
        baseline.codeCommitId(),
        baseline.lastExitCode(),
        baseline.lastErrorMessage(),
        baseline.heartbeatIntervalSeconds(),
        baseline.heartbeatTimeoutSeconds(),
        baseline.otel(),
        baseline.tags(),
        "graceful shutdown",
        baseline.deployEnabled(),
        baseline.deployStatus(),
        baseline.deployAtEpochMs(),
        baseline.deployMessage(),
        baseline.deploySuccess(),
        baseline.metrics());
  }

  private void publishStoppedSafe() {
    try {
      DefinitionStatusApp baseline = current;
      if (baseline == null) {
        return;
      }
      publish(transitionToStopped(baseline, clock.millis()));
    } catch (Exception ex) {
      LOG.warn("[AppStatusPublisher] stop publish failed", ex);
    }
  }

  private DefinitionStatusApp normalizeStatus(DefinitionStatusApp status) {
    if (status == null) {
      return null;
    }
    long now = clock.millis();
    String group = Strings.trimToNull(status.group());
    String name = Strings.trimToNull(status.name());
    String instance = Strings.trimToNull(status.instance());
    if (group == null || name == null || instance == null) {
      return null;
    }
    AppRuntimeType runtimeType = status.runtimeType() == null ? AppRuntimeType.UNKNOWN : status.runtimeType();
    String host = Strings.firstNonBlank(status.host(), resolveHostname());
    String version = resolveVersion(status.version());
    Long startedAtEpochMs =
        status.startedAtEpochMs() == null || status.startedAtEpochMs() <= 0L
            ? now
            : status.startedAtEpochMs();
    Long lastHeartbeatEpochMs =
        status.lastHeartbeatEpochMs() == null || status.lastHeartbeatEpochMs() <= 0L
            ? now
            : status.lastHeartbeatEpochMs();
    Integer intervalSeconds =
        status.heartbeatIntervalSeconds() == null
            ? heartbeatIntervalSeconds
            : status.heartbeatIntervalSeconds();
    Integer timeoutSeconds =
        status.heartbeatTimeoutSeconds() == null
            ? heartbeatTimeoutSeconds
            : status.heartbeatTimeoutSeconds();
    Map<String, String> tags = status.tags() == null || status.tags().isEmpty() ? Map.of() : Map.copyOf(status.tags());
    AppMetrics metrics = status.metrics();
    if (metricsCollector != null) {
      try {
        AppMetrics collected = metricsCollector.collect();
        if (collected != null) {
          metrics = collected;
        }
      } catch (Exception ignore) {
      }
    }
    DefinitionEnvelope envelope = status.envelope();
    if (envelope == null) {
      envelope =
          new DefinitionEnvelope(
              group, DefinitionType.STATUS_APP, name, null, null, null, null, null, null, null);
    }
    DefinitionStatusApp.State state =
        status.state() == null ? DefinitionStatusApp.State.UNKNOWN : status.state();
    return new DefinitionStatusApp(
        envelope,
        group,
        DefinitionType.STATUS_APP,
        name,
        instance,
        Strings.trimToNull(status.sessionId()),
        Strings.trimToNull(status.sessionKey()),
        runtimeType,
        host,
        version,
        Strings.trimToNull(status.env()),
        Strings.trimToNull(status.appType()),
        state,
        startedAtEpochMs,
        lastHeartbeatEpochMs,
        status.lastStateChangeEpochMs(),
        status.lastStartEpochMs(),
        status.lastStopEpochMs(),
        status.runCount(),
        Strings.trimToNull(status.buildVersion()),
        Strings.trimToNull(status.configVersion()),
        Strings.trimToNull(status.codeVersion()),
        Strings.trimToNull(status.codeCommitId()),
        status.lastExitCode(),
        Strings.trimToNull(status.lastErrorMessage()),
        intervalSeconds,
        timeoutSeconds,
        status.otel(),
        tags,
        Strings.trimToNull(status.message()),
        status.deployEnabled(),
        Strings.trimToNull(status.deployStatus()),
        status.deployAtEpochMs(),
        Strings.trimToNull(status.deployMessage()),
        status.deploySuccess(),
        metrics);
  }

  private static DefinitionStatusApp withLastHeartbeatEpochMs(DefinitionStatusApp status, long lastHeartbeatEpochMs) {
    if (status == null) {

    long heartbeat = Math.max(0L, lastHeartbeatEpochMs);
    return new DefinitionStatusApp(
        status.envelope(),
        status.group(),
        status.type(),
        status.name(),
        status.instance(),
        status.sessionId(),
        status.sessionKey(),
        status.runtimeType(),
        status.host(),
        status.version(),
        status.env(),
        status.appType(),
        status.state(),
        status.startedAtEpochMs(),
        heartbeat,
        status.lastStateChangeEpochMs(),
        status.lastStartEpochMs(),
        status.lastStopEpochMs(),
        status.runCount(),
        status.buildVersion(),
        status.configVersion(),
        status.codeVersion(),
        status.codeCommitId(),
        status.lastExitCode(),
        status.lastErrorMessage(),
        status.heartbeatIntervalSeconds(),
        status.heartbeatTimeoutSeconds(),
        status.otel(),

        status.deployEnabled(),
        status.deployStatus(),
        status.deployAtEpochMs(),
        status.deployMessage(),
        status.deploySuccess(),
        status.metrics());
  }

  private static int resolveHeartbeatIntervalSeconds(DefinitionApp appDefinition) {
    Integer configured = null;
    if (appDefinition != null
        && appDefinition.runtimeControls() != null
        && appDefinition.runtimeControls().status() != null) {
      configured = appDefinition.runtimeControls().status().publishIntervalSeconds();
    }
    if (configured != null && configured > 0) {
      return configured;
    }
    return DEFAULT_HEARTBEAT_SECONDS;
  }

  private static String resolveIntervalSource(DefinitionApp appDefinition) {
    Integer configured = null;
    if (appDefinition != null
        && appDefinition.runtimeControls() != null
        && appDefinition.runtimeControls().status() != null) {
      configured = appDefinition.runtimeControls().status().publishIntervalSeconds();
    }
    if (configured != null && configured > 0) {
      return "runtimeControls.status.publishIntervalSeconds";
    }
    return "default";
  }

  private String resolveVersion(String provided) {
    String trimmed = Strings.trimToNull(provided);
    if (trimmed != null) {
      return trimmed;
    }
    String manifestVersion = manifestVersion();
    if (manifestVersion != null) {
      return manifestVersion;
    }
    return DEFAULT_VERSION;
  }

  private static String manifestVersion() {
    Package pkg = AppStatusPublisher.class.getPackage();
    if (pkg == null) {
      return null;
    }
    String impl = Strings.trimToNull(pkg.getImplementationVersion());
    if (impl != null) {
      return impl;
    }
    return Strings.trimToNull(pkg.getSpecificationVersion());
  }

  private static String resolveHostname() {
    try {
      return InetAddress.getLocalHost().getHostName();
    } catch (Exception ex) {
      return "localhost";
    }
  }

  private static Map<String, String> buildTags(
      String group,
      String app,
      String instance,
      String appType,
      SessionContext context,
      String configuredUserTag) {
    Map<String, String> tags = new LinkedHashMap<>();
    putIfPresent(tags, "group", group);
    putIfPresent(tags, "app", app);
    putIfPresent(tags, "instance", instance);
    putIfPresent(tags, "env", context == null ? null : context.getEnv());
    putIfPresent(tags, "region", context == null ? null : context.getRegion());
    putIfPresent(tags, "site", context == null ? null : context.getSite());
    putIfPresent(tags, "appType", appType);
    putIfPresent(tags, "pid", Long.toString(ProcessHandle.current().pid()));
    putIfPresent(tags, "user", firstNonBlank(configuredUserTag, resolveRuntimeUser()));
    return tags.isEmpty() ? Map.of() : Map.copyOf(tags);
  }

  private static String resolveRuntimeUser() {
    try {
      return ProcessHandle.current().info().user().orElse(null);
    } catch (Exception ex) {
      return null;
    }
  }

  private static void putIfPresent(Map<String, String> tags, String key, String value) {
    if (tags == null || key == null) {
      return;
    }
    String normalized = Strings.trimToNull(value);
    if (normalized != null) {
      tags.put(key, normalized);
    }
  }

  private static String buildSessionKey(
      String group, String appName, String instance, String environment, String region, String site) {
    String resolvedGroup = Strings.trimToNull(group);
    String resolvedApp = Strings.trimToNull(appName);
    if (resolvedGroup == null || resolvedApp == null) {
      return null;
    }
    String resolvedInstance = Strings.trimToNull(instance);
    if (resolvedInstance == null) {
      resolvedInstance = "unknown";
    }
    StringBuilder builder =
        new StringBuilder("app:")
            .append(resolvedGroup)
            .append('/')
            .append(resolvedApp)
            .append('|')
            .append(resolvedInstance);
    appendPlacement(builder, "env", environment);
    appendPlacement(builder, "region", region);
    appendPlacement(builder, "site", site);
    return builder.toString();
  }

  private static void appendPlacement(StringBuilder builder, String label, String value) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null || "UNKNOWN".equalsIgnoreCase(normalized)) {
      return;
    }
    builder.append('|').append(label).append('=').append(normalized);
  }

  private static AppRuntimeType mapRuntimeType(String appType) {
    String normalized = Strings.trimToNull(appType);
    if (normalized == null) {
      return AppRuntimeType.UNKNOWN;
    }
    String upper = normalized.toUpperCase();
    if (upper.contains("JAVA")) {
      return AppRuntimeType.JAVA;
    }
    if (upper.contains("PYTHON")) {
      return AppRuntimeType.PYTHON;
    }
    if (upper.contains("SCRIPT") || upper.contains("SHELL")) {
      return AppRuntimeType.SCRIPT;
    }
    return AppRuntimeType.UNKNOWN;
  }

  private static String requireField(String value, String fieldName) {
    String normalized = Strings.trimToNull(value);
    if (normalized == null) {
      throw new IllegalStateException("status publish " + fieldName + " is required");
    }
    return normalized;
  }

  private static String firstNonBlank(String... values) {
    if (values == null) {
      return null;
    }
    for (String value : values) {
      String normalized = Strings.trimToNull(value);
      if (normalized != null) {
        return normalized;
      }
    }
    return null;
  }

  @FunctionalInterface
  public interface TickListener {
    void onPublished(long publishEpochMs, DefinitionStatusApp status, String payload);
  }
}
