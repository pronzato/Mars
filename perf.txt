  private static ConfigPublisher createAuditPublisher(
      ControlConnection controlConnection, DefinitionEnvironment environmentDefinition) {
    ConfigUsageEventsSettings settings = ConfigUsageEventsSettings.fromEnvironment(environmentDefinition);
    if (!settings.enabled() || settings.transport() == ConfigUsageEventsTransport.NOOP) {
      return ConfigPublisherNoop.INSTANCE;
    }
    if (settings.transport() == ConfigUsageEventsTransport.S3) {
      LOGGER.warn("[ConfigUsageAudit] transport=S3 is not implemented; using NOOP");
      return ConfigPublisherNoop.INSTANCE;
    }
    if (settings.transport() != ConfigUsageEventsTransport.AMPS) {
      LOGGER.warn("[ConfigUsageAudit] unsupported transport={} ; using NOOP", settings.transport());
      return ConfigPublisherNoop.INSTANCE;
    }
    String connection = Strings.firstNonBlank(settings.connection(), DISCOVERY_CONNECTION_NAME);
    if (!DISCOVERY_CONNECTION_NAME.equalsIgnoreCase(connection)) {
      LOGGER.warn("[ConfigUsageAudit] unsupported AMPS connection={} expected={} ; using NOOP", connection, DISCOVERY_CONNECTION_NAME);
      return ConfigPublisherNoop.INSTANCE;
    }
    if (controlConnection == null) {
      LOGGER.warn("[ConfigUsageAudit] AMPS requested but ControlConnection is null; using NOOP");
      return ConfigPublisherNoop.INSTANCE;
    }
    return new ConfigPublisherAmp(
        controlConnection,
        new ConfigPublisherAmp.Settings(
            connection,
            settings.topic(),
            settings.maxRetries(),
            settings.initialBackoffMillis(),
            settings.maxBackoffMillis(),
            settings.backoffMultiplier()));
  }

  private enum ConfigUsageEventsTransport {
    NOOP,
    AMPS,
    S3;

    static ConfigUsageEventsTransport fromWire(String raw) {
      String normalized = Strings.trimToNull(raw);
      if (normalized == null) {
        return NOOP;
      }
      return switch (normalized.toUpperCase(Locale.ROOT)) {
        case "NOOP", "NONE" -> NOOP;
        case "AMPS" -> AMPS;
        case "S3" -> S3;
        default -> NOOP;
      };
    }
  }

  private record ConfigUsageEventsSettings(
      boolean enabled,
      ConfigUsageEventsTransport transport,
      String connection,
      String topic,
      int maxRetries,
      long initialBackoffMillis,
      long maxBackoffMillis,
      double backoffMultiplier) {

    static ConfigUsageEventsSettings fromEnvironment(DefinitionEnvironment environmentDefinition) {
      DefinitionEnvironment.Configs configs =
          environmentDefinition == null ? null : environmentDefinition.configs();
      return fromModel(configs == null ? null : configs.events());
    }

    static ConfigUsageEventsSettings fromModel(DefinitionEnvironment.Events events) {
      if (events == null || !Boolean.TRUE.equals(events.enabled())) {
        return disabled();
      }
      ConfigUsageEventsTransport transport = ConfigUsageEventsTransport.fromWire(events.transport());
      DefinitionEnvironment.EventsAmps amps = events.amps();
      String connection =
          Strings.firstNonBlank(
              amps == null ? null : Strings.trimToNull(amps.connection()),
              DISCOVERY_CONNECTION_NAME);
      String topic =
          Strings.firstNonBlank(
              amps == null ? null : Strings.trimToNull(amps.topic()),
              DiscoveryTopics.EVENTS_CONFIG);
      int maxRetries = amps == null || amps.maxRetries() == null ? 3 : Math.max(0, amps.maxRetries());
      long initialBackoffMillis =
          amps == null || amps.initialBackoffMillis() == null
              ? 100L
              : Math.max(1L, amps.initialBackoffMillis());
      long maxBackoffMillis =
          amps == null || amps.maxBackoffMillis() == null
              ? 2_000L
              : Math.max(initialBackoffMillis, amps.maxBackoffMillis());
      double backoffMultiplier =
          amps == null || amps.backoffMultiplier() == null
              ? 2.0d
              : amps.backoffMultiplier() <= 1.0d ? 2.0d : amps.backoffMultiplier();
      return new ConfigUsageEventsSettings(
          true,
          transport,
          connection,
          topic,
          maxRetries,
          initialBackoffMillis,
          maxBackoffMillis,
          backoffMultiplier);
    }

    static ConfigUsageEventsSettings disabled() {
      return new ConfigUsageEventsSettings(
          false,
          ConfigUsageEventsTransport.NOOP,
          DISCOVERY_CONNECTION_NAME,
          DiscoveryTopics.EVENTS_CONFIG,
          0,
          100L,
          2_000L,
          2.0d);
    }
  }

  <T> T decode(
      Resolution id,
      LayerEntry layer,
      JsonNode resolved,
      Instant resolvedAtUtc,
      Instant effectiveAtUtc,
      ConfigDecoder<T> decoder) {
    JsonNode payload = resolved == null ? NullNode.getInstance() : resolved;
    payload = groupedPresentation(id, layer, payload, resolvedAtUtc, effectiveAtUtc);
    try {
      return decoder.decode(mapper, payload);
    } catch (Exception ex) {
      throw new IllegalArgumentException(
          "Failed to decode payload for config "
              + id.group()
              + "/"
              + id.configId()
              + "/"
              + id.layerId(),
          ex);
    }
  }

  private JsonNode groupedPresentation(
      Resolution id,
      LayerEntry layer,
      JsonNode resolvedPayload,
      Instant resolvedAtUtc,
      Instant effectiveAtUtc) {
    DefinitionConfigLayer definition = layer == null ? null : layer.definition();
    DefinitionConfigLayer payload = definition;
    String schemaKey = required(payload == null ? null : payload.schemaKey(), "schemaKey");
    ConfigSchemaDoc schemaDoc = resolveSchemaDoc(id.group(), schemaKey);
    ConfigPresentation.FabricMetadata fabric =
        ConfigPresentation.fabricMetadata(
            id.group(),
            id.configId(),
            id.layerId(),
            schemaKey,
            resolvedAtUtc,
            effectiveAtUtc,
            payload == null ? null : payload.effective(),
            Strings.trimToNull(definition == null ? null : definition.configId()),
            Strings.trimToNull(definition == null ? null : definition.effectiveConfigId()));
    return ConfigPresentation.grouped(
        schemaDoc, resolvedPayload, fabric, ConfigPresentation.Options.defaults());
  }

  ConfigCodegen generateForLayer(
      Resolution resolvedId,
      LayerEntry layer,
      ConfigManager.Language language,
      String basePackageOrModule) {
    AuditContext auditContext = captureAuditContext();
    DefinitionConfigLayer payload = layer == null ? null : layer.definition();
    String schemaName = required(payload == null ? null : payload.schemaKey(), "schemaName");
    ConfigManager.Language requiredLanguage = Objects.requireNonNull(language, "language");
    ConfigCodegen generated =
        generateCodeZip(
        resolvedId.group(), schemaName, requiredLanguage, Strings.trimToNull(basePackageOrModule));
    ConfigResolved resolved =
        resolver != null
            ? resolver.resolve(
                resolvedId.group(), resolvedId.configId(), resolvedId.layerId(), Instant.now())
            : fallbackResolved(
                resolvedId,
                layer,
                deepCopy(resolveDeclaredAt(layer.definition(), Instant.now())),
                Instant.now());
    publishAudit(
        new ConfigEventGenerated(
            envelope(
                ConfigEventType.CONFIG_GENERATED,
                resolvedId.group(),
                resolvedId.configId(),
                resolvedId.layerId(),
                schemaName,
                auditContext,
                Instant.now()),
            requiredLanguage.name().toLowerCase(),
            Strings.trimToNull(basePackageOrModule),
            generated.filename(),
            generated.stats(),
            resolved == null ? null : resolved.lineage()));
    return generated;
  }

  @Override
  public void close() {
    if (!closed.compareAndSet(false, true)) {
      return;
    }
    for (SubscriptionImpl subscription : subscriptions) {
      try {
        subscription.close();
      } catch (Exception ignore) {
      }
    }
    callbackExecutor.shutdownNow();
    scheduleExecutor.shutdownNow();
    CloseSupport.closeQuietly(auditPublisher);
  }

  UpdateSnapshot toUpdateSnapshotAt(
      DefinitionConfigLayer definition, ParsedLayerIdentity identity, Instant asOfUtc) {
    DefinitionEnvelope envelope = definition == null ? null : definition.envelope();
    DefinitionConfigLayer payload = definition;
    List<DefinitionConfigLayer.Effective> copiedEffective = copyEffective(payload == null ? null : payload.effective());
    Instant resolvedAt = asOfUtc == null ? Instant.now() : asOfUtc;
    JsonNode changedPayload =
        definition == null ? null : deepCopy(resolveDeclaredAt(definition, resolvedAt));
    JsonNode declared;
    ConfigLineage lineage;
    if (resolver != null) {
      ConfigResolved resolvedConfig =
          resolver.resolve(identity.group(), identity.configId(), identity.layerId(), resolvedAt);
      declared = deepCopy(resolvedPayloadAsJsonNode(resolvedConfig.payload()));
      lineage = resolvedConfig == null ? null : resolvedConfig.lineage();
    } else {
      declared = deepCopy(resolveDeclaredAt(definition, resolvedAt));
      lineage = fallbackLineage(definition, identity.layerId(), declared);
    }
    return new UpdateSnapshot(
        new ConfigUpdate(
            identity.group(),
            identity.configId(),
            identity.layerId(),
            Strings.trimToNull(payload == null ? null : payload.schemaKey()),
            Strings.trimToNull(definition == null ? null : definition.configId()),
            Strings.trimToNull(definition == null ? null : definition.effectiveConfigId()),
            Strings.trimToNull(envelope == null ? null : envelope.uuid()),
            Strings.trimToNull(envelope == null ? null : envelope.updatedTimeUtc()),
            declared,
            copiedEffective),
        lineage,
        changedPayload);
  }

  ConfigResolved fallbackResolved(
      Resolution resolvedId, LayerEntry layer, JsonNode declared, Instant resolvedAt) {
    if (resolvedId == null) {
      return null;
    }
    DefinitionConfigLayer definition = layer == null ? null : layer.definition();
    return new ConfigResolved(
        resolvedId.configId(),
        resolvedId.layerId(),
        deepCopy(declared),
        Strings.trimToNull(
            definition == null || definition.envelope() == null
                ? null
                : definition.envelope().updatedTimeUtc()),
        Map.of(),
        fallbackLineage(definition, resolvedId.layerId(), declared));
  }

  ConfigLineage fallbackLineage(DefinitionConfigLayer definition, String layerId, JsonNode declared) {
    String configVersionId = Strings.trimToNull(definition == null ? null : definition.configId());
    String payloadHash = ConfigManagerImplAudit.hashJson(HASH_MAPPER, declared);
    return new ConfigLineage(
        configVersionId,
        1,
        true,
        configVersionId == null ? List.of() : List.of(configVersionId),
        layerId == null ? List.of() : List.of(layerId),
        payloadHash,
        payloadHash,
        HASH_VERSION);
  }

  ConfigLineage resolveLineage(String group, String configId, String layerId, Instant at) {
    if (resolver == null
        || Strings.isBlank(group)
        || Strings.isBlank(configId)
        || Strings.isBlank(layerId)
        || at == null) {
      return null;
    }
    try {
      ConfigResolved resolved = resolver.resolve(group, configId, layerId, at);
      return resolved == null ? null : resolved.lineage();
    } catch (Exception ex) {
      LOGGER.debug("[ConfigUsageAudit] resolve-lineage ignored group={} configId={} layerId={} message={}", group, configId, layerId, ex.toString());
      return null;
    }
  }

  AuditContext captureAuditContext() {
    CallerAppContext.ContextValue caller = CallerAppContext.get().orElse(null);
    String subject = Strings.trimToNull(UserContext.getUsernameOrDefault(null));
    String delegatedUser =
        ConfigManagerImplAudit.resolveDelegatedUser(UserContext.getAttributesOrEmpty());
    String requestId = caller == null ? nextAuditRequestId() : caller.requestId();
    return new AuditContext(
        requestId,
        caller == null ? null : caller.traceId(),
        caller == null ? null : caller.callerAppSubject(),
        subject,

  LayerEntry requireLayer(String group, String configId, String layerId) {
    String requiredGroup = required(group, "group");
    String requiredconfigId = required(configId, "configId");
    String requiredLayerId = required(layerId, "layerId");
    List<LayerEntry> layers = findConfigLayers(requiredGroup, requiredconfigId);
    for (LayerEntry layer : layers) {
      if (requiredLayerId.equals(layer.identity().layerId())) {
        return layer;
      }
    }
    throw new IllegalArgumentException(
        "layer not found for config " + requiredconfigId + ": " + requiredLayerId);
  }

  List<LayerEntry> findConfigLayers(String group, String configId) {
    List<LayerEntry> results = findConfigLayersOrEmpty(group, configId);
    if (results.isEmpty()) {
      throw new IllegalArgumentException("config not found: " + configId);
    }
    return results;
  }

  List<LayerEntry> findConfigLayersOrEmpty(String group, String configId) {
    List<DefinitionConfigLayer> all = configView.listConfigLayers(group);
    List<LayerEntry> results = new ArrayList<>();
    for (DefinitionConfigLayer definition : all) {
      ParsedLayerIdentity identity = parseLayerIdentity(definition, group);
      if (identity == null) {
        LOGGER.debug("[ConfigManager][Ignored] reason=identity-parse definition={}", definition);
        continue;
      }
      if (!group.equals(identity.group()) || !configId.equals(identity.configId())) {
        continue;
      }
      results.add(new LayerEntry(definition, identity));
    }
    return List.copyOf(results);
  }

  Map<String, List<LayerEntry>> groupByConfig(String group) {
    List<DefinitionConfigLayer> all = configView.listConfigLayers(group);
    Map<String, List<LayerEntry>> grouped = new LinkedHashMap<>();
    for (DefinitionConfigLayer definition : all) {
      ParsedLayerIdentity identity = parseLayerIdentity(definition, group);
      if (identity == null || !group.equals(identity.group())) {
        continue;
      }
      grouped.computeIfAbsent(identity.configId(), ignored -> new ArrayList<>())
          .add(new LayerEntry(definition, identity));
    }
    return grouped;
  }

  static LayerEntry representativeLayer(List<LayerEntry> layers) {
    if (layers == null || layers.isEmpty()) {
      return null;
    }
    for (LayerEntry layer : layers) {
      if (LAYER_BASE.equals(layer.identity().layerId())) {
        return layer;
      }
    }
    return layers.get(0);
  }

  static List<LayerEntry> sortLayers(List<LayerEntry> layers) {
    List<LayerEntry> sorted = new ArrayList<>(layers == null ? List.of() : layers);
    sorted.sort(
        Comparator.comparing((LayerEntry entry) -> !LAYER_BASE.equals(entry.identity().layerId()))
            .thenComparing(entry -> entry.identity().layerId(), Comparator.nullsLast(String::compareTo)));
    return List.copyOf(sorted);
  }

  JsonNode resolveLiveDeclared(DefinitionConfigLayer definition) {
    DefinitionConfigLayer payload = definition;
    JsonNode declared = payload == null ? null : payload.declared();
    if (declared == null || !declared.isObject()) {
      throw new IllegalStateException("declared payload must be a JSON object");
    }
    return declared.deepCopy();
  }

  JsonNode resolveDeclaredAt(DefinitionConfigLayer definition, Instant at) {
    JsonNode liveDeclared = resolveLiveDeclared(definition);
    DefinitionConfigLayer payload = definition;
    List<DefinitionConfigLayer.Effective> effectiveEntries = payload == null ? null : payload.effective();
    if (effectiveEntries == null || effectiveEntries.isEmpty()) {
      return liveDeclared;
    }
    List<TimedEffective> sorted = new ArrayList<>(effectiveEntries.size());
    for (DefinitionConfigLayer.Effective entry : effectiveEntries) {
      if (entry == null || Strings.trimToNull(entry.effectiveTimeUtc()) == null) {
        continue;
      }
      JsonNode declared = entry.declared();
      if (declared == null || !declared.isObject()) {
        throw new IllegalStateException("effective declared payload must be a JSON object");
      }
      Instant expiresAt = null;
      if (Strings.trimToNull(entry.expiresAtUtc()) != null) {
        try {
          expiresAt = Instant.parse(entry.expiresAtUtc());
        } catch (Exception ex) {
          throw new IllegalStateException("invalid expiresAtUtc: " + entry.expiresAtUtc(), ex);
        }
      }
      Instant effectiveAt;
      try {
        effectiveAt = Instant.parse(entry.effectiveTimeUtc());
      } catch (Exception ex) {
        throw new IllegalStateException("invalid effectiveTimeUtc: " + entry.effectiveTimeUtc(), ex);
      }
      if (expiresAt != null && !expiresAt.isAfter(effectiveAt)) {
        continue;
      }
      sorted.add(new TimedEffective(effectiveAt, expiresAt, entry));
    }
    sorted.sort(Comparator.comparing(TimedEffective::effectiveTime));
    DefinitionConfigLayer.Effective winner = null;
    for (TimedEffective timed : sorted) {
      if (timed.effectiveTime().isAfter(at)) {
        break;
      }
      if (timed.expiresAt() != null && !at.isBefore(timed.expiresAt())) {
        continue;
      }
      winner = timed.entry();
    }
    if (winner == null) {
      return liveDeclared;
    }
    return winner.declared().deepCopy();
  }

  List<DefinitionConfigLayer.Effective> copyEffective(List<DefinitionConfigLayer.Effective> values) {
    if (values == null || values.isEmpty()) {
      return List.of();
    }
    List<DefinitionConfigLayer.Effective> copied = new ArrayList<>(values.size());
    for (DefinitionConfigLayer.Effective value : values) {
      if (value == null) {
        continue;
      }
      copied.add(
          new DefinitionConfigLayer.Effective(
              value.effectiveTimeUtc(), value.expiresAtUtc(), deepCopy(value.declared())));
    }
    return List.copyOf(copied);
  }

  static List<DefinitionConfigLayer.Effective> upsertScheduledEffective(
      List<DefinitionConfigLayer.Effective> baseline,
      Instant effectiveAt,
      Instant expiresAt,
      JsonNode declaredPayload) {
    String effectiveUtc = effectiveAt.toString();
    String expiresUtc = expiresAt == null ? null : expiresAt.toString();
    List<DefinitionConfigLayer.Effective> rewritten = new ArrayList<>();
    if (baseline != null) {
      for (DefinitionConfigLayer.Effective entry : baseline) {
        if (entry == null) {
          continue;
        }
        if (!effectiveUtc.equals(entry.effectiveTimeUtc())) {
          rewritten.add(entry);
        }
      }
    }
    rewritten.add(new DefinitionConfigLayer.Effective(effectiveUtc, expiresUtc, declaredPayload.deepCopy()));
    rewritten.sort(Comparator.comparing(entry -> Instant.parse(entry.effectiveTimeUtc())));
    return List.copyOf(rewritten);
  }

  static DefinitionConfigLayer.History findHistorySnapshot(
      List<DefinitionConfigLayer.History> history, String configId) {
    if (history == null || history.isEmpty() || configId == null) {
      return null;
    }
    for (DefinitionConfigLayer.History snapshot : history) {
      if (snapshot == null) {
        continue;
      }
      if (configId.equals(Strings.trimToNull(snapshot.configId()))) {
        return snapshot;
      }
    }
    return null;
  }

  static List<String> topLevelKeys(JsonNode declared) {
    if (declared == null || !declared.isObject()) {
      return List.of();
    }
    List<String> keys = new ArrayList<>();
    declared.fieldNames().forEachRemaining(keys::add);
    keys.sort(String::compareTo);
    return List.copyOf(keys);
  }

  ParsedLayerIdentity parseLayerIdentity(DefinitionConfigLayer definition, String fallbackGroup) {
    if (definition == null) {
      return null;
    }
    DefinitionEnvelope envelope = definition.envelope();
    String group = Strings.trimToNull(envelope == null ? null : envelope.group());
    if (group == null) {
      group = Strings.trimToNull(fallbackGroup);
    }

    String resourceId = Strings.trimToNull(definition.resourceId());
    if (resourceId != null) {
      try {
        CanonicalIds.ParsedId parsed = CanonicalIds.parse(resourceId);
        if ("config".equals(parsed.type())) {
          String parsedGroup = Strings.trimToNull(parsed.group());
          if (parsedGroup != null) {
            group = parsedGroup;
          }
          String configId = Strings.trimToNull(parsed.namespace());
          String layerId = Strings.trimToNull(parsed.name());
          if (group != null && configId != null && layerId != null) {
            return new ParsedLayerIdentity(group, configId, layerId, resourceId);
          }
        }
      } catch (RuntimeException ex) {
        LOGGER.debug("[ConfigManager][Ignored] reason=invalid-resource-id value={}", resourceId);
      }
    }
    return null;
  }

  static final class SubscriptionImpl implements ConfigSubscription {
    final ConfigManagerImpl manager;
    final String subscriptionId;
    final String group;
    final String configId;
    final String layerId;
    final ConfigUpdateListener listener;
    final AuditContext auditContext;
    final AtomicLong callbackCount = new AtomicLong(0L);
    final AtomicBoolean closed = new AtomicBoolean(false);
    final ConcurrentHashMap<String, String> lastSeenByLayer = new ConcurrentHashMap<>();
    final ConcurrentHashMap<String, CopyOnWriteArrayList<ScheduledFuture<?>>> scheduledByLayer =
        new ConcurrentHashMap<>();
    volatile AutoCloseable cacheHandle;

    SubscriptionImpl(
        ConfigManagerImpl manager,
        String subscriptionId,
        String group,
        String configId,
        String layerId,
        ConfigUpdateListener listener,
        AuditContext auditContext) {
      this.manager = Objects.requireNonNull(manager, "manager");
      this.subscriptionId = Strings.requireNonBlank(subscriptionId, "subscriptionId");
      this.group = group;
      this.configId = configId;
      this.layerId = layerId;
      this.listener = listener;
      this.auditContext = auditContext;
    }

    @Override
    public void close() {
      ConfigManagerImplSubscription.closeSubscription(this);
    }

    void addScheduled(String layer, ScheduledFuture<?> future) {
      if (future == null) {
        return;
      }
      String key = required(layer, "layerId");
      if (closed.get()) {
        future.cancel(false);
        return;
      }
      scheduledByLayer.computeIfAbsent(key, ignored -> new CopyOnWriteArrayList<>()).add(future);
    }

    void cancelScheduled(String layer) {
      if (layer == null) {
        return;
      }
      CopyOnWriteArrayList<ScheduledFuture<?>> futures = scheduledByLayer.remove(layer);
      if (futures == null) {
        return;
      }
      for (ScheduledFuture<?> future : futures) {
        if (future != null) {
          future.cancel(false);
        }
      }
    }

    void pruneScheduled(String layer) {
      if (layer == null) {
        return;
      }
      CopyOnWriteArrayList<ScheduledFuture<?>> futures = scheduledByLayer.get(layer);
      if (futures == null) {
        return;
      }
      futures.removeIf(future -> future == null || future.isDone() || future.isCancelled());
      if (futures.isEmpty()) {
        scheduledByLayer.remove(layer, futures);
      }
    }
  }
