/**
 * Configuration for the OIDC SSO implementation.
 *
 * <p>Values can be loaded via system properties or environment variables. Required fields
 * must be provided explicitly; no real secrets are hard-coded.
 */
@JsonDeserialize(builder = FabricOidcConfig.Builder.class)
public final class FabricOidcConfig {

  public static final String DEFAULT_SCOPE = "openid profile email";
  public static final String DEFAULT_USERNAME_CLAIM = "preferred_username";
  public static final String DEFAULT_LOGIN_PATH = "/login";
  public static final String DEFAULT_CALLBACK_PATH = "/oidc/callback";
  public static final int DEFAULT_CLOCK_SKEW_SECONDS = 60;

  private final String issuerBaseUrl;
  private final String authorizationEndpoint;
  private final String tokenEndpoint;
  private final String jwksEndpoint;
  private final String clientId;
  private final String clientSecret;
  private final String redirectUri;
  private final String scope;
  private final String usernameClaim;
  private final List<String> groupClaims;
  private final List<String> allowAnonPaths;
  private final boolean httpsRequiredForSameSiteNone;
  private final String loginPath;
  private final String callbackPath;
  private final int clockSkewSeconds;

  private FabricOidcConfig(Builder builder) {
    this.issuerBaseUrl = builder.issuerBaseUrl;
    this.authorizationEndpoint = builder.authorizationEndpoint;
    this.tokenEndpoint = builder.tokenEndpoint;
    this.jwksEndpoint = builder.jwksEndpoint;
    this.clientId = builder.clientId;
    this.clientSecret = builder.clientSecret;
    this.redirectUri = builder.redirectUri;
    this.scope = builder.scope;
    this.usernameClaim = builder.usernameClaim;
    this.groupClaims = builder.groupClaims;
    this.allowAnonPaths = builder.allowAnonPaths;
    this.httpsRequiredForSameSiteNone = builder.httpsRequiredForSameSiteNone;
    this.loginPath = builder.loginPath;
    this.callbackPath = builder.callbackPath;
    this.clockSkewSeconds = builder.clockSkewSeconds;
  }

  public static Builder builder() {
    return new Builder();
  }

  public static FabricOidcConfig fromSystem() {
    return Builder.fromSystem().build();
  }

  public String authorizationUrl() {
    return resolveEndpoint(issuerBaseUrl, authorizationEndpoint);
  }

  public String tokenUrl() {
    return resolveEndpoint(issuerBaseUrl, tokenEndpoint);
  }

  public String discoveryUrl() {
    String base = issuerBaseUrl == null ? null : issuerBaseUrl.trim();
    if (Strings.isBlank(base)) {
      return null;
    }
    String normalized = trimTrailingSlash(base);
    return normalized + "/.well-known/openid-configuration";
  }

  public static String resolveEndpoint(String base, String endpoint) {
    if (Strings.isBlank(endpoint)) {
      return null;
    }
    String trimmed = endpoint.trim();
    if (trimmed.startsWith("http://") || trimmed.startsWith("https://")) {
      return trimmed;
    }
    if (Strings.isBlank(base)) {
      return trimmed;
    }
    return join(base.trim(), trimmed);
  }

  static String join(String base, String endpoint) {
    String normalizedBase = trimTrailingSlash(base == null ? "" : base.trim());
    String normalizedEndpoint = endpoint == null ? "" : endpoint.trim();
    if (normalizedEndpoint.startsWith("/")) {
      return normalizedBase + normalizedEndpoint;
    }
    return normalizedBase + "/" + normalizedEndpoint;
  }

  static String trimTrailingSlash(String value) {
    if (Strings.isBlank(value)) {
      return value;
    }
    String trimmed = value.trim();
    while (trimmed.endsWith("/") && trimmed.length() > 1) {
      trimmed = trimmed.substring(0, trimmed.length() - 1);
    }
    return trimmed;
  }

  public static final class Builder {

    private String issuerBaseUrl;
    private String authorizationEndpoint;
    private String tokenEndpoint;
    private String jwksEndpoint;
    private String clientId;
    private String clientSecret;
    private String redirectUri;
    private String externalBaseUrl;
    private String redirectPath = DEFAULT_CALLBACK_PATH;
    private String scope = DEFAULT_SCOPE;
    private String usernameClaim = DEFAULT_USERNAME_CLAIM;
    private List<String> groupClaims = List.of();
    private List<String> allowAnonPaths =
        List.of(DEFAULT_LOGIN_PATH, DEFAULT_CALLBACK_PATH, "/health", "/favicon.ico");
    private boolean httpsRequiredForSameSiteNone = true;
    private String loginPath = DEFAULT_LOGIN_PATH;
    private String callbackPath = DEFAULT_CALLBACK_PATH;
    private int clockSkewSeconds = DEFAULT_CLOCK_SKEW_SECONDS;

    private Builder() {
    }

    public static Builder fromSystem() {
      Builder builder = new Builder();
      String issuer = firstNonBlank(
          System.getProperty("fabric.oidc.issuerBaseUrl"),
          System.getProperty("fabric.oidc.issuer"),
          System.getenv("FABRIC_OIDC_ISSUER_BASE_URL"),
          System.getenv("FABRIC_OIDC_ISSUER"));
      if (Strings.isNotBlank(issuer)) {
        builder.withIssuerBaseUrl(issuer);
      }
      String authorizationEndpoint = firstNonBlank(
          System.getProperty("fabric.oidc.authorizationEndpoint"),
          System.getProperty("fabric.oidc.authEndpoint"),
          System.getenv("FABRIC_OIDC_AUTHORIZATION_ENDPOINT"),
          System.getenv("FABRIC_OIDC_AUTH_ENDPOINT"));
      if (Strings.isNotBlank(authorizationEndpoint)) {
        builder.withAuthorizationEndpoint(authorizationEndpoint);
      }
      String tokenEndpoint = firstNonBlank(
          System.getProperty("fabric.oidc.tokenEndpoint"),
          System.getenv("FABRIC_OIDC_TOKEN_ENDPOINT"));
      if (Strings.isNotBlank(tokenEndpoint)) {
        builder.withTokenEndpoint(tokenEndpoint);
      }
      String jwksEndpoint = firstNonBlank(
          System.getProperty("fabric.oidc.jwksEndpoint"),
          System.getenv("FABRIC_OIDC_JWKS_ENDPOINT"));
      if (Strings.isNotBlank(jwksEndpoint)) {
        builder.withJwksEndpoint(jwksEndpoint);
      }
      String clientId = firstNonBlank(
          System.getProperty("fabric.oidc.clientId"),
          System.getenv("FABRIC_OIDC_CLIENT_ID"));
      if (Strings.isNotBlank(clientId)) {
        builder.withClientId(clientId);
      }
      String clientSecret = firstNonBlank(
          System.getProperty("fabric.oidc.clientSecret"),
          System.getenv("FABRIC_OIDC_CLIENT_SECRET"));
      if (Strings.isNotBlank(clientSecret)) {
        builder.withClientSecret(clientSecret);
      }
      String redirectUri = firstNonBlank(
          System.getProperty("fabric.oidc.redirectUri"),
          System.getenv("FABRIC_OIDC_REDIRECT_URI"));
      if (Strings.isNotBlank(redirectUri)) {
        builder.withRedirectUri(redirectUri);
      }
      String externalBaseUrl = firstNonBlank(
          System.getProperty("fabric.oidc.externalBaseUrl"),
          System.getenv("FABRIC_OIDC_EXTERNAL_BASE_URL"));
      if (Strings.isNotBlank(externalBaseUrl)) {
        builder.withExternalBaseUrl(externalBaseUrl);
      }
      String redirectPath = firstNonBlank(
          System.getProperty("fabric.oidc.redirectPath"),
          System.getenv("FABRIC_OIDC_REDIRECT_PATH"));
      if (Strings.isNotBlank(redirectPath)) {
        builder.withRedirectPath(redirectPath);
      }
      String scope = firstNonBlank(
          System.getProperty("fabric.oidc.scope"),
          System.getenv("FABRIC_OIDC_SCOPE"));
      if (Strings.isNotBlank(scope)) {
        builder.withScope(scope);
      }
      String usernameClaim = firstNonBlank(
          System.getProperty("fabric.oidc.usernameClaim"),
          System.getenv("FABRIC_OIDC_USERNAME_CLAIM"));
      if (Strings.isNotBlank(usernameClaim)) {
        builder.withUsernameClaim(usernameClaim);
      }
      String groupClaims = firstNonBlank(
          System.getProperty("fabric.oidc.groupClaims"),
          System.getenv("FABRIC_OIDC_GROUP_CLAIMS"));
      if (Strings.isNotBlank(groupClaims)) {
        builder.withGroupClaims(parseList(groupClaims));
      }
      String allowAnonPaths = firstNonBlank(
          System.getProperty("fabric.oidc.allowAnonPaths"),
          System.getenv("FABRIC_OIDC_ALLOW_ANON_PATHS"));
      if (Strings.isNotBlank(allowAnonPaths)) {
        builder.withAllowAnonPaths(parseList(allowAnonPaths));
      }
      String loginPath = firstNonBlank(
          System.getProperty("fabric.oidc.loginPath"),
          System.getenv("FABRIC_OIDC_LOGIN_PATH"));
      if (Strings.isNotBlank(loginPath)) {
        builder.withLoginPath(loginPath);
      }
      String callbackPath = firstNonBlank(
          System.getProperty("fabric.oidc.callbackPath"),
          System.getenv("FABRIC_OIDC_CALLBACK_PATH"));
      if (Strings.isNotBlank(callbackPath)) {
        builder.withCallbackPath(callbackPath);
      }
      String httpsRequired = firstNonBlank(
          System.getProperty("fabric.oidc.httpsRequiredForSameSiteNone"),
          System.getenv("FABRIC_OIDC_HTTPS_REQUIRED_FOR_SAMESITE_NONE"));
      if (Strings.isNotBlank(httpsRequired)) {
        builder.withHttpsRequiredForSameSiteNone(parseBoolean(httpsRequired, true));
      }
      String clockSkew = firstNonBlank(
          System.getProperty("fabric.oidc.clockSkewSeconds"),
          System.getenv("FABRIC_OIDC_CLOCK_SKEW_SECONDS"));
      if (Strings.isNotBlank(clockSkew)) {
        builder.withClockSkewSeconds(parseInt(clockSkew, DEFAULT_CLOCK_SKEW_SECONDS));
      }
      return builder;
    }

    @JsonProperty("issuerBaseUrl")
    public Builder withIssuerBaseUrl(String issuerBaseUrl) {
      this.issuerBaseUrl = normalizeBaseUrl(issuerBaseUrl);
      return this;
    }

    @JsonProperty("authorizationEndpoint")
    public Builder withAuthorizationEndpoint(String authorizationEndpoint) {
      this.authorizationEndpoint = trimToNull(authorizationEndpoint);
      return this;
    }

    @JsonProperty("tokenEndpoint")
    public Builder withTokenEndpoint(String tokenEndpoint) {
      this.tokenEndpoint = trimToNull(tokenEndpoint);
      return this;
    }

    @JsonProperty("jwksEndpoint")
    public Builder withJwksEndpoint(String jwksEndpoint) {
      this.jwksEndpoint = trimToNull(jwksEndpoint);
      return this;
    }

    @JsonProperty("clientId")
    public Builder withClientId(String clientId) {
      this.clientId = trimToNull(clientId);
      return this;

      this.clientSecret = trimToNull(clientSecret);

      return this;

      this.usernameClaim = trimToNull(usernameClaim);

      this.groupClaims = normalizeList(groupClaims);

    public Builder withHttpsRequiredForSameSiteNone(boolean httpsRequiredForSameSiteNone) {
      this.httpsRequiredForSameSiteNone = httpsRequiredForSameSiteNone;
      return this;
    }

    @JsonProperty("loginPath")
    public Builder withLoginPath(String loginPath) {
      this.loginPath = normalizePath(loginPath, DEFAULT_LOGIN_PATH);
      return this;
    }

    @JsonProperty("callbackPath")
    public Builder withCallbackPath(String callbackPath) {
      this.callbackPath = normalizePath(callbackPath, DEFAULT_CALLBACK_PATH);
      return this;
    }

    @JsonProperty("clockSkewSeconds")
    public Builder withClockSkewSeconds(int clockSkewSeconds) {
      this.clockSkewSeconds = Math.max(0, clockSkewSeconds);
      return this;
    }

    public FabricOidcConfig build() {
      if (Strings.isBlank(redirectUri)) {
        if (Strings.isNotBlank(externalBaseUrl)) {
          String path = Strings.isBlank(redirectPath) ? DEFAULT_CALLBACK_PATH : redirectPath.trim();
          redirectUri = join(externalBaseUrl, path);
        }
      }
      if (Strings.isNotBlank(loginPath)) {
        loginPath = normalizePath(loginPath, DEFAULT_LOGIN_PATH);
      }
      if (Strings.isNotBlank(callbackPath)) {
        callbackPath = normalizePath(callbackPath, DEFAULT_CALLBACK_PATH);
      }
      allowAnonPaths = normalizeAllowlist(allowAnonPaths, loginPath, callbackPath);
      scope = Strings.isBlank(scope) ? DEFAULT_SCOPE : scope;
      usernameClaim = Strings.isBlank(usernameClaim) ? DEFAULT_USERNAME_CLAIM : usernameClaim;
      validateRequired();
      URI.create(redirectUri);
      return new FabricOidcConfig(this);
    }

    private void validateRequired() {
      List<String> missing = new ArrayList<>();
      if (Strings.isBlank(issuerBaseUrl)) {
        missing.add("issuerBaseUrl");
      }
      if (Strings.isBlank(authorizationEndpoint)) {
        missing.add("authorizationEndpoint");
      }
      if (Strings.isBlank(tokenEndpoint)) {
        missing.add("tokenEndpoint");
      }
      if (Strings.isBlank(clientId)) {
        missing.add("clientId");
      }
      if (Strings.isBlank(clientSecret)) {
        missing.add("clientSecret");
      }
      if (Strings.isBlank(redirectUri)) {
        missing.add("redirectUri");
      }
      if (!missing.isEmpty()) {
        throw new IllegalStateException(
            "OIDC config missing required values: " + String.join(", ", missing));
      }
    }
  }

  private static String normalizeBaseUrl(String base) {
    if (Strings.isBlank(base)) {
      return null;
    }
    return trimTrailingSlash(base.trim());
  }

  private static String normalizePath(String value, String fallback) {
    String trimmed = trimToNull(value);
    if (trimmed == null) {
      return fallback;
    }
    if (!trimmed.startsWith("/")) {
      return "/" + trimmed;
    }
    return trimmed;
  }

  private static List<String> normalizeAllowlist(
      List<String> allowlist, String loginPath, String callbackPath) {
    Set<String> merged = new LinkedHashSet<>();
    if (allowlist != null) {
      merged.addAll(allowlist);
    }
    if (Strings.isNotBlank(loginPath)) {
      merged.add(loginPath);
    }
    if (Strings.isNotBlank(callbackPath)) {
      merged.add(callbackPath);
    }
    if (merged.isEmpty()) {
      merged.add(DEFAULT_LOGIN_PATH);
      merged.add(DEFAULT_CALLBACK_PATH);
      merged.add("/health");
      merged.add("/favicon.ico");
    }
    List<String> normalized = new ArrayList<>();
    for (String entry : merged) {
      String trimmed = trimToNull(entry);
      if (trimmed != null) {
        normalized.add(trimmed.startsWith("/") ? trimmed : "/" + trimmed);
      }
    }
    return List.copyOf(normalized);
  }

  private static List<String> normalizeList(List<String> values) {
    if (values == null || values.isEmpty()) {
      return List.of();
    }
    List<String> normalized = new ArrayList<>();
    for (String value : values) {
      String trimmed = trimToNull(value);
      if (trimmed != null) {
        normalized.add(trimmed);
      }
    }
    return List.copyOf(normalized);
  }

  private static List<String> parseList(String value) {
    if (Strings.isBlank(value)) {
      return List.of();
    }
    String[] parts = value.split("[,;]");
    List<String> normalized = new ArrayList<>();
    for (String part : parts) {
      String trimmed = trimToNull(part);
      if (trimmed != null) {
        normalized.add(trimmed);
      }
    }
    return List.copyOf(normalized);
  }

  private static String trimToNull(String value) {
    if (value == null) {
      return null;
    }
    String trimmed = value.trim();
    return trimmed.isEmpty() ? null : trimmed;
  }

  private static String firstNonBlank(String... values) {
    if (values == null) {
      return null;
    }
    for (String value : values) {
      if (Strings.isNotBlank(value)) {
        return value.trim();
      }
    }
    return null;
  }

  private static boolean parseBoolean(String raw, boolean fallback) {
    if (Strings.isBlank(raw)) {
      return fallback;
    }
    String trimmed = raw.trim();
    if ("true".equalsIgnoreCase(trimmed) || "1".equals(trimmed) || "yes".equalsIgnoreCase(trimmed)) {
      return true;
    }
    if ("false".equalsIgnoreCase(trimmed) || "0".equals(trimmed) || "no".equalsIgnoreCase(trimmed)) {
      return false;
    }
    return fallback;
  }

  private static int parseInt(String raw, int fallback) {
    if (Strings.isBlank(raw)) {
      return fallback;
    }
    try {
      return Integer.parseInt(raw.trim());
    } catch (NumberFormatException ex) {
      return fallback;
    }
  }
